487bool DIOPING::Do(XDWORD nretries, XDWORD timebetweenchecks, bool exitfirstgoodreply)
488{
489 if(!urltarget) return false;
490 if(!xtimer) return false;
491
492 if(xmutexping)
493 {
494 xmutexping->Lock();
495 }
496
497 DIOSTREAMICMPCONFIG diostreamICMPconfig;
498 XSTRING targetIP;
499
500 DeleteAllReplys();
501
502 urltarget->ResolveURL(targetIP);
503
504 diostreamICMPconfig.GetRemoteURL()->Set(urltarget->Get());
505
506 diostreamICMPconfig.SetMode(DIOSTREAMMODE_CLIENT);
507
508 DIOSTREAMICMP* diostreamICMP = (DIOSTREAMICMP*)GEN_DIOFACTORY.CreateStreamIO(&diostreamICMPconfig);
509 if(!diostreamICMP)
510 {
511 if(xmutexping)
512 {
513 xmutexping->UnLock();
514 }
515
516 return false;
517 }
518
519 if(!diostreamICMP->Open())
520 {
521 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
522
523 if(xmutexping)
524 {
525 xmutexping->UnLock();
526 }
527
528 return false;
529 }
530
531 if(!diostreamICMP->WaitToConnected(5))
532 {
533 diostreamICMP->Close();
534 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
535
536 if(xmutexping)
537 {
538 xmutexping->UnLock();
539 }
540
541 return false;
542 }
543
544 DIOPING_ECHOREQUEST echorequest;
545 int ID = 0xCAFE;
546 int nseq = 2;
547 int nloop = nretries;
548
549 while(nloop)
550 {
551 HASHCRC32 crc32;
552 XDWORD crc32result;
553 XSTRING address;
554 XBUFFER xbuffer;
555 bool status = false;
556
557 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
558 if(!xrand)
559 {
560 break;
561 }
562
563 memset((XBYTE*)&echorequest, 0, sizeof(DIOPING_ECHOREQUEST));
564
565 // Fill in echo request
566 echorequest.icmphdr.type = DIOPING_ICMP_ECHOREQ;
567 echorequest.icmphdr.code = 0;
568 echorequest.icmphdr.ID = ID;
569 echorequest.icmphdr.seq = nseq++;
570
571 //Fill in some data to send
572 for(int nret=0; nret<DIOPING_REQ_TICKETSIZE; nret++)
573 {
574 echorequest.cdata[nret] = xrand->Max(255);
575 }
576
577 GEN_XFACTORY.DeleteRand(xrand);
578
579 crc32.ResetResult();
580 crc32.Do((XBYTE*)echorequest.cdata, DIOPING_REQ_TICKETSIZE);
581 crc32result = crc32.GetResultCRC32();
582
583 echorequest.applicationID = applicationID;
584 echorequest.time = (XDWORD)xtimer->GetMicroSecondsTickCounter(); // Save tick count when sent
585 echorequest.icmphdr.checksum = CalculeCheckSum((XWORD *)&echorequest, sizeof(DIOPING_ECHOREQUEST)); // Put data in packet and compute checksum
586
587 if(echorequest.icmphdr.checksum)
588 {
589 diostreamICMP->GetInXBuffer()->Delete();
590 diostreamICMP->GetOutXBuffer()->Delete();
591 diostreamICMP->DeleteAllDatagrams();
592
593 status = diostreamICMP->WriteDatagram(targetIP, (XBYTE*)&echorequest, sizeof(DIOPING_ECHOREQUEST));
594 if(status)
595 {
596 status = diostreamICMP->WaitToFlushOutXBuffer(3);
597 }
598 }
599
600 if(!status)
601 {
602 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not write packet! %s"), targetIP.Get());
603 break;
604 }
605
606 while(1)
607 {
608 status = diostreamICMP->WaitToFilledReadingBuffer(sizeof(DIOPING_ECHOREPLY), 1);
609 if(!status)
610 {
611 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not ready packet! %s"), targetIP.Get());
612 if(exitfirstgoodreply) nloop = 0; else nloop--;
613 break;
614 }
615
616 status = diostreamICMP->ReadDatagram(address, xbuffer);
617 if(!status)
618 {
619 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not read packet! %s"), targetIP.Get());
620 if(exitfirstgoodreply) nloop = 0; else nloop--;
621 break;
622 }
623
624 if(xbuffer.GetSize() < sizeof(DIOPING_ECHOREPLY))
625 {
626 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] invalid size packet! %s, "), targetIP.Get());
627 continue;
628 }
629
630 DIOPING_ECHOREPLY echoreply;
631
632 memcpy((XBYTE*)&echoreply.iphdr, xbuffer.Get(), sizeof(DIOPING_ECHOREPLY));
633
634 if(echoreply.echorequest.icmphdr.type == DIOPING_ICMP_ECHOREPLY)
635 {
636 crc32.ResetResult();
637 crc32.Do((XBYTE*)echoreply.echorequest.cdata, DIOPING_REQ_TICKETSIZE);
638
639 if(crc32result == crc32.GetResultCRC32())
640 {
641 XWORD checksum = CalculeCheckSum((XWORD *)&echoreply.echorequest, sizeof(DIOPING_ECHOREQUEST));
642 if(!checksum)
643 {
644 DIOIP ip;
645 XSTRING fromIP;
646
647 // Calculate elapsed time
648
649 ip.Set(echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b1,
650 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b2,
651 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b3,
652 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b4);
653
654 ip.GetXString(fromIP);
655
656 if(!fromIP.Compare(targetIP))
657 {
658 if(echoreply.echorequest.applicationID == applicationID)
659 {
660 XDWORD elapsed;
661
662 elapsed = ((XDWORD)xtimer->GetMicroSecondsTickCounter() - echoreply.echorequest.time);
663
664 AddReply(nloop+1, fromIP.Get(), DIOPING_REQ_TICKETSIZE, (elapsed/1000), echoreply.iphdr.TTL);
665
666 DIOPING_XEVENT xevent(this, DIOPING_XEVENT_TYPE_DOPING);
667 xevent.SetPingReply((DIOPINGREPLY*)replys.Get(replys.GetSize()-1));
668 PostEvent(&xevent);
669
670 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Ping] received! %s %s (elapsed time %d)"), fromIP.Get(), targetIP.Get(), elapsed);
671
672 if(exitfirstgoodreply)
673 {
674 nloop = 0;
675 break;
676 }
677
678 GEN_XSLEEP.MilliSeconds(timebetweenchecks);
679
680 nloop--;
681 break;
682 }
683 else
684 {
685 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] application ID not equal! %s %s"), fromIP.Get(), targetIP.Get());
686 }
687 }
688 else
689 {
690 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not IP equal! %s %s"), fromIP.Get(), targetIP.Get());
691 }
692
693 }
694 else
695 {
696 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] error Checksum! %s "), targetIP.Get());
697 }
698 }
699 else
700 {
701 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] error CRC! %s "), targetIP.Get());
702 }
703 }
704 else
705 {
706 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] Is not reply packet! %s "), targetIP.Get());
707 }
708 }
709 }
710
711 diostreamICMP->Close();
712 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
713
714 bool status = WasConnected();
715 if(!status)
716 {
717 //XTRACE_PRINTCOLOR((status?XTRACE_COLOR_BLUE:XTRACE_COLOR_RED), __L("[Ping] to [%s]: %s"), targetIP.Get(), (status?__L("Ok."): __L("ERROR!")));
718 }
719
720 if(xmutexping)
721 {
722 xmutexping->UnLock();
723 }
724
725 return status;
726}

890XWORD DIOPING::CalculeCheckSum(XWORD* addr, int len)
891{
892 int nleft = len;
893 XWORD* w = addr;
894 XWORD answer;
895 int sum = 0;
896
897 // Our algorithm is simple, using a 32 bit accumulator (sum),
898 // we add sequential 16 bit words to it, and at the end, fold
899 // back all the carry bits from the top 16 bits into the lower
900 // 16 bits.
901
902 while(nleft>1)
903 {
904 sum += *w++;
905 nleft -= 2;
906 }
907
908 // mop up an odd byte, if necessary
909
910 if( nleft == 1 )
911 {
912 XWORD u = 0;
913
914 *(XBYTE*)(&u) = *(XBYTE*)w ;
915 sum += u;
916 }
917
918
919 // add back carry outs from top 16 bits to low 16 bits
920
921 sum = (sum >> 16) + (sum & 0xffff); // add hi 16 to low 16
922 sum += (sum >> 16); // add carry
923 answer = ~sum; // truncate to 16 bits
924
925 return answer;
926}
