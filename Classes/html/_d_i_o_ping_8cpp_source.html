494bool DIOPING::Do(XDWORD nretries, XDWORD timebetweenchecks, bool exitfirstgoodreply)
495{
496 if(!urltarget) return false;
497 if(!xtimer) return false;
498
499 if(xmutexping)
500 {
501 xmutexping->Lock();
502 }
503
504 DIOSTREAMICMPCONFIG diostreamICMPconfig;
505 XSTRING targetIP;
506
507 DeleteAllReplys();
508
509 urltarget->ResolveURL(targetIP);
510
511 diostreamICMPconfig.GetRemoteURL()->Set(urltarget->Get());
512
513 diostreamICMPconfig.SetMode(DIOSTREAMMODE_CLIENT);
514
515 DIOSTREAMICMP* diostreamICMP = (DIOSTREAMICMP*)GEN_DIOFACTORY.CreateStreamIO(&diostreamICMPconfig);
516 if(!diostreamICMP)
517 {
518 if(xmutexping)
519 {
520 xmutexping->UnLock();
521 }
522
523 return false;
524 }
525
526 if(!diostreamICMP->Open())
527 {
528 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
529
530 if(xmutexping)
531 {
532 xmutexping->UnLock();
533 }
534
535 return false;
536 }
537
538 if(!diostreamICMP->WaitToConnected(5))
539 {
540 diostreamICMP->Close();
541 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
542
543 if(xmutexping)
544 {
545 xmutexping->UnLock();
546 }
547
548 return false;
549 }
550
551 DIOPING_ECHOREQUEST echorequest;
552 int ID = 0xCAFE;
553 int nseq = 2;
554 int nloop = nretries;
555
556 while(nloop)
557 {
558 HASHCRC32 crc32;
559 XDWORD crc32result;
560 XSTRING address;
561 XBUFFER xbuffer;
562 bool status = false;
563
564 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
565 if(!xrand)
566 {
567 break;
568 }
569
570 memset((XBYTE*)&echorequest, 0, sizeof(DIOPING_ECHOREQUEST));
571
572 // Fill in echo request
573 echorequest.icmphdr.type = DIOPING_ICMP_ECHOREQ;
574 echorequest.icmphdr.code = 0;
575 echorequest.icmphdr.ID = ID;
576 echorequest.icmphdr.seq = nseq++;
577
578 //Fill in some data to send
579 for(int nret=0; nret<DIOPING_REQ_TICKETSIZE; nret++)
580 {
581 echorequest.cdata[nret] = xrand->Max(255);
582 }
583
584 GEN_XFACTORY.DeleteRand(xrand);
585
586 crc32.ResetResult();
587 crc32.Do((XBYTE*)echorequest.cdata, DIOPING_REQ_TICKETSIZE);
588 crc32result = crc32.GetResultCRC32();
589
590 echorequest.applicationID = applicationID;
591 echorequest.time = (XDWORD)xtimer->GetMicroSecondsTickCounter(); // Save tick count when sent
592 echorequest.icmphdr.checksum = CalculeCheckSum((XWORD *)&echorequest, sizeof(DIOPING_ECHOREQUEST)); // Put data in packet and compute checksum
593
594 if(echorequest.icmphdr.checksum)
595 {
596 diostreamICMP->GetInXBuffer()->Delete();
597 diostreamICMP->GetOutXBuffer()->Delete();
598 diostreamICMP->DeleteAllDatagrams();
599
600 status = diostreamICMP->WriteDatagram(targetIP, (XBYTE*)&echorequest, sizeof(DIOPING_ECHOREQUEST));
601 if(status)
602 {
603 status = diostreamICMP->WaitToFlushOutXBuffer(3);
604 }
605 }
606
607 if(!status)
608 {
609 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not write packet! %s"), targetIP.Get());
610 break;
611 }
612
613 while(1)
614 {
615 status = diostreamICMP->WaitToFilledReadingBuffer(sizeof(DIOPING_ECHOREPLY), 1);
616 if(!status)
617 {
618 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not ready packet! %s"), targetIP.Get());
619 if(exitfirstgoodreply) nloop = 0; else nloop--;
620 break;
621 }
622
623 status = diostreamICMP->ReadDatagram(address, xbuffer);
624 if(!status)
625 {
626 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not read packet! %s"), targetIP.Get());
627 if(exitfirstgoodreply) nloop = 0; else nloop--;
628 break;
629 }
630
631 if(xbuffer.GetSize() < sizeof(DIOPING_ECHOREPLY))
632 {
633 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] invalid size packet! %s, "), targetIP.Get());
634 continue;
635 }
636
637 DIOPING_ECHOREPLY echoreply;
638
639 memcpy((XBYTE*)&echoreply.iphdr, xbuffer.Get(), sizeof(DIOPING_ECHOREPLY));
640
641 if(echoreply.echorequest.icmphdr.type == DIOPING_ICMP_ECHOREPLY)
642 {
643 crc32.ResetResult();
644 crc32.Do((XBYTE*)echoreply.echorequest.cdata, DIOPING_REQ_TICKETSIZE);
645
646 if(crc32result == crc32.GetResultCRC32())
647 {
648 XWORD checksum = CalculeCheckSum((XWORD *)&echoreply.echorequest, sizeof(DIOPING_ECHOREQUEST));
649 if(!checksum)
650 {
651 DIOIP ip;
652 XSTRING fromIP;
653
654 // Calculate elapsed time
655
656 ip.Set(echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b1,
657 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b2,
658 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b3,
659 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b4);
660
661 ip.GetXString(fromIP);
662
663 if(!fromIP.Compare(targetIP))
664 {
665 if(echoreply.echorequest.applicationID == applicationID)
666 {
667 XDWORD elapsed;
668
669 elapsed = ((XDWORD)xtimer->GetMicroSecondsTickCounter() - echoreply.echorequest.time);
670
671 AddReply(nloop+1, fromIP.Get(), DIOPING_REQ_TICKETSIZE, (elapsed/1000), echoreply.iphdr.TTL);
672
673 DIOPING_XEVENT xevent(this, DIOPING_XEVENT_TYPE_DOPING);
674 xevent.SetPingReply((DIOPINGREPLY*)replys.Get(replys.GetSize()-1));
675 PostEvent(&xevent);
676
677 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Ping] received! %s %s (elapsed time %d)"), fromIP.Get(), targetIP.Get(), elapsed);
678
679 if(exitfirstgoodreply)
680 {
681 nloop = 0;
682 break;
683 }
684
685 GEN_XSLEEP.MilliSeconds(timebetweenchecks);
686
687 nloop--;
688 break;
689 }
690 else
691 {
692 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] application ID not equal! %s %s"), fromIP.Get(), targetIP.Get());
693 }
694 }
695 else
696 {
697 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] not IP equal! %s %s"), fromIP.Get(), targetIP.Get());
698 }
699
700 }
701 else
702 {
703 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] error Checksum! %s "), targetIP.Get());
704 }
705 }
706 else
707 {
708 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] error CRC! %s "), targetIP.Get());
709 }
710 }
711 else
712 {
713 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Ping] Is not reply packet! %s "), targetIP.Get());
714 }
715 }
716 }
717
718 diostreamICMP->Close();
719 GEN_DIOFACTORY.DeleteStreamIO(diostreamICMP);
720
721 bool status = WasConnected();
722 if(!status)
723 {
724 //XTRACE_PRINTCOLOR((status?XTRACE_COLOR_BLUE:XTRACE_COLOR_RED), __L("[Ping] to [%s]: %s"), targetIP.Get(), (status?__L("[Ok.]"):__L("[Error!]")));
725 }
726
727 if(xmutexping)
728 {
729 xmutexping->UnLock();
730 }
731
732 return status;
733}

897XWORD DIOPING::CalculeCheckSum(XWORD* addr, int len)
898{
899 int nleft = len;
900 XWORD* w = addr;
901 XWORD answer;
902 int sum = 0;
903
904 // Our algorithm is simple, using a 32 bit accumulator (sum),
905 // we add sequential 16 bit words to it, and at the end, fold
906 // back all the carry bits from the top 16 bits into the lower
907 // 16 bits.
908
909 while(nleft>1)
910 {
911 sum += *w++;
912 nleft -= 2;
913 }
914
915 // mop up an odd byte, if necessary
916
917 if( nleft == 1 )
918 {
919 XWORD u = 0;
920
921 *(XBYTE*)(&u) = *(XBYTE*)w ;
922 sum += u;
923 }
924
925
926 // add back carry outs from top 16 bits to low 16 bits
927
928 sum = (sum >> 16) + (sum & 0xffff); // add hi 16 to low 16
929 sum += (sum >> 16); // add carry
930 answer = ~sum; // truncate to 16 bits
931
932 return answer;
933}
