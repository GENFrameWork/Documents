148bool GRPLINUXSCREENFRAMEBUFFER::Create(bool show)
149{
150 Clean();
151
152 handlefb = open(GRPLINUXSCREENFRAMEBUFFER_DEVICE, O_RDWR);
153 if(handlefb!=-1)
154 {
155 int consoletty = open("/dev/tty1", O_RDWR);
156 if(ioctl(consoletty, KDSETMODE, KD_GRAPHICS) != -1)
157 {
158 if(!ScreenResolution(width, height))
159 {
160 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("ERROR! Change screen mode: error[%d]"), errno);
161 }
162
163 CreateBuffers();
164
165 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("X: %d, Y: %d mode: %d"), width, height, GetMode());
166
167 int zero = 0;
168 if(ioctl(handlefb, FBIO_WAITFORVSYNC, &zero) == -1)
169 {
170 XSTRING errorstr;
171
172 errorstr = strerror(errno);
173 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("Frame Buffer VSYNC: %s"), errorstr.Get());

175
176 return GRPSCREEN::Create(show);
177 }
178 }
179
180 XTRACE_PRINTCOLOR(4, __L("ERROR! open device screen: %d"), errno);

212bool GRPLINUXSCREENFRAMEBUFFER::Update(GRPCANVAS* canvas)
213{
214 if(handlefb == -1) return false;
215 if(!buffersize) return false;
216
217
218 if(ioctl(handlefb, FBIO_WAITFORVSYNC, 0))
219 {
220 //printf("VSync failed.\n");
221 }
222
223 XBYTE* fbp = (XBYTE*)mmap(0, buffersize, PROT_READ | PROT_WRITE, MAP_SHARED, handlefb, 0);
224 if(fbp == (XBYTE*)-1) return false;
225
226 if(IsEqualSizeTo(canvas) == ISEQUAL)
227 {
228 memcpy(fbp, (XBYTE*)canvas->GetBuffer(), buffersize);
229 }
230 else
231 {
232 //canvas->CopyBufferRenderToScreen(this);
233 //memcpy(fbp,(XBYTE*)buffer, buffersize);
234
235 canvas->CopyBufferRenderToBufferScreen((XBYTE*)(fbp), width, height, isbufferinverse);
236 }
237
238 munmap((XBYTE*)fbp, buffersize);
239
240 return true;
241}
