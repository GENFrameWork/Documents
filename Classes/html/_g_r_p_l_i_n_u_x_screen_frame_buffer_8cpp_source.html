154bool GRPLINUXSCREENFRAMEBUFFER::Create(bool show)
155{
156 Clean();
157
158 handlefb = open(GRPLINUXSCREENFRAMEBUFFER_DEVICE, O_RDWR);
159 if(handlefb!=-1)
160 {
161 int consoletty = open("/dev/tty1", O_RDWR);
162 if(ioctl(consoletty, KDSETMODE, KD_GRAPHICS) != -1)
163 {
164 if(!ScreenResolution(width, height))
165 {
166 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[LINUX FrameBuffer] Change screen mode: error [%d]"), errno);
167 }
168
169 int zero = 0;
170 if(ioctl(handlefb, FBIO_WAITFORVSYNC, &zero) == -1)
171 {
172 XSTRING errorstr;
173
174 errorstr = strerror(errno);
175
176 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[LINUX FrameBuffer] Frame Buffer VSYNC: %s"), errorstr.Get());
177 }
178
179 return GRPSCREEN::Create(show);
180 }
181 }
182
183 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[LINUX FrameBuffer] Open device screen: error [%d]"), errno);
184
185 return false;
186}

200bool GRPLINUXSCREENFRAMEBUFFER::Update(GRPCANVAS* canvas)
201{
202 if(handlefb == -1)
203 {
204 return false;
205 }
206
207 if(ioctl(handlefb, FBIO_WAITFORVSYNC, 0))
208 {
209 //printf("VSync failed.\n");
210 }
211
212 XBYTE* fbp = (XBYTE*)mmap(0, canvas->Buffer_GetSize(), PROT_READ | PROT_WRITE, MAP_SHARED, handlefb, 0);
213 if(fbp == (XBYTE*)-1)
214 {
215 return false;
216 }
217
218 if(rotation != GRPSCREENROTATION_NONE)
219 {
220 Rotate(fbp, canvas);
221 }
222 else
223 {
224 memcpy(fbp, (XBYTE*)canvas->Buffer_Get(), canvas->Buffer_GetSize());
225 }
226
227 munmap((XBYTE*)fbp, canvas->Buffer_GetSize());
228
229 return true;
230}

356{
357 if(handlefb == -1) return false;
358
359 struct fb_fix_screeninfo fixed_info;
360 struct fb_var_screeninfo variable_info;
361 int screen_width;
362 int screen_height;
363
364 if(ioctl(handlefb, FBIOGET_VSCREENINFO, &variable_info) == -1)
365 {
366 return false;
367 }
368
369 screen_width = variable_info.xres;
370 screen_height = variable_info.yres;
371
372 variable_info.xres = width;
373 variable_info.yres = height;
374
375 variable_info.xres_virtual = variable_info.xres;
376 variable_info.yres_virtual = variable_info.yres;
377
378 if(ioctl(handlefb, FBIOPUT_VSCREENINFO, &variable_info) == -1)
379 {
380 ioctl(handlefb, FBIOGET_VSCREENINFO, &variable_info);
381
382 SetWidth(screen_width);
383 SetHeight(screen_height);
384
385 return false;
386 }
387
388 if(ioctl(handlefb, FBIOGET_FSCREENINFO, &fixed_info) == -1)
389 {
390 return false;

399 XTRACE_PRINTCOLOR(4, __L("ERROR! 4"));
