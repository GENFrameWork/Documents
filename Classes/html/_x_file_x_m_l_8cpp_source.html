1220{
1221 DeleteAllElements();
1222
1223 XSTRING string;
1224 XSTRING* stringml;
1225 bool quote = false;
1226 bool cdata = false;
1227 int c = 0;
1228
1229 string.Empty();
1230
1231 while(c<GetNLines())
1232 {
1233 stringml = GetLine(c);
1234
1235 for(int d=0;d<(int)stringml->GetSize();d++)
1236 {
1237 XCHAR letter[2] = { 0, 0 };
1238 letter[0] = (*stringml)[d];
1239
1240 if((*stringml)[d]==__C('\"')) quote=!quote;
1241
1242 if((*stringml)[d]==__C('<')&&(!quote))
1243 {
1244 if(string.Find(__L("[CDATA["),false)!=XSTRING_NOTFOUND) cdata=true;
1245
1246 if(!string.IsEmpty() && string.HaveCharacters() && (!cdata))
1247 {
1248 DecodeLine(string,false);
1249 string.Empty();
1250 }
1251
1252 quote = false;
1253 string += letter;
1254 }
1255 else
1256 {
1257 if(((*stringml)[d]==__C('>'))&&(!quote))
1258 {
1259 if(string.Find(__L("]]"),false)!=XSTRING_NOTFOUND) cdata=false;
1260
1261 string += letter;
1262 if(!string.IsEmpty() && string.HaveCharacters() && (!cdata))
1263 {
1264 DecodeLine(string,true);
1265 string.Empty();
1266 }
1267 }
1268 else string += letter;
1269 }
1270 }

1272 c++;

1611int XFILEXML::DecodeLine(XSTRING& string,bool iselement)

1619

1621
1622 int iini = str.FindCharacter(__C('<'));
1623 int iend = str.FindCharacter(__C('>'),0,true);
1624

1626 if(iend!=XSTRING_NOTFOUND) iend--;

1641 elementtype = XFILEXMLELEMENTTYPE_CDATA;
