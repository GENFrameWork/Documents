1213{
1214 DeleteAllElements();
1215
1216 XSTRING string;
1217 XSTRING* stringml;
1218 bool quote = false;
1219 bool cdata = false;
1220 int c = 0;
1221
1222 string.Empty();
1223
1224 while(c<GetNLines())
1225 {
1226 stringml = GetLine(c);
1227
1228 for(int d=0;d<(int)stringml->GetSize();d++)
1229 {
1230 XCHAR letter[2] = { 0, 0 };
1231 letter[0] = (*stringml)[d];
1232
1233 if((*stringml)[d]==__C('\"')) quote=!quote;
1234
1235 if((*stringml)[d]==__C('<')&&(!quote))
1236 {
1237 if(string.Find(__L("[CDATA["),false)!=XSTRING_NOTFOUND) cdata=true;
1238
1239 if(!string.IsEmpty() && string.HaveCharacters() && (!cdata))
1240 {
1241 DecodeLine(string,false);
1242 string.Empty();
1243 }
1244
1245 quote = false;
1246 string += letter;
1247 }
1248 else
1249 {
1250 if(((*stringml)[d]==__C('>'))&&(!quote))
1251 {
1252 if(string.Find(__L("]]"),false)!=XSTRING_NOTFOUND) cdata=false;
1253
1254 string += letter;
1255 if(!string.IsEmpty() && string.HaveCharacters() && (!cdata))
1256 {
1257 DecodeLine(string,true);
1258 string.Empty();
1259 }
1260 }
1261 else string += letter;

1604int XFILEXML::DecodeLine(XSTRING& string,bool iselement)
1605{

1607 XFILEXMLELEMENTTYPELINE typeline = XFILEXMLELEMENTTYPELINE_UNKNOWN;

1612
1613 str.DeleteNoCharacters(XSTRINGCONTEXT_ALLSTRING);
1614
1615 int iini = str.FindCharacter(__C('<'));
1616 int iend = str.FindCharacter(__C('>'),0,true);
1617
1618 if(iini!=XSTRING_NOTFOUND) iini++;
1619 if(iend!=XSTRING_NOTFOUND) iend--;
1620
1621 switch(str[iini])

1627 typeline = XFILEXMLELEMENTTYPELINE_END;

1631 elementtype = XFILEXMLELEMENTTYPE_COMMENT;

1634 elementtype = XFILEXMLELEMENTTYPE_CDATA;

1637 if((str[iini+1]==__C('-')) && (str[iini+2]==__C('-')))
1638 is_multicomment=true;

1640
1641 break;

1651 case XFILEXMLELEMENTTYPE_UNKNOWN : break;
