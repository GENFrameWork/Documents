613bool XFILECFG::Load(XPATH& xpath)
614{
615 bool status = false;
616
617 if(values.IsEmpty()) return status;
618
619 if(fileini->IsOpen())
620 {
621 EndFile();
622 }
623
624 if(IniFile(xpath))
625 {
626 AjustRemarks();
627
628 for(int c=0;c<(int)values.GetSize();c++)
629 {
630 XFILECFGVALUE* cfgvalue = (XFILECFGVALUE*)values.Get(c);
631 if(cfgvalue)
632 {
633 XSTRING strvalue;
634
635 if(fileini->ReadValue(cfgvalue->GetGroup()->Get(), cfgvalue->GetID()->Get() , strvalue))
636 {
637 switch(cfgvalue->GetType())
638 {
639 case XFILECFG_VALUETYPE_UNKNOWN : break;
640
641 case XFILECFG_VALUETYPE_INT : { int* value = (int*)cfgvalue->GetValue();
642 if(!value) return false;
643
644 (*value) = strvalue.ConvertToInt();
645 }
646 break;
647
648 case XFILECFG_VALUETYPE_MASK : { XWORD* value = (XWORD*)cfgvalue->GetValue();
649 if(!value) return false;
650
651 strvalue.UnFormat(__L("%04X"), value);
652 }
653 break;
654
655 case XFILECFG_VALUETYPE_FLOAT : { float* value = (float*)cfgvalue->GetValue();
656 if(!value) return false;
657
658 (*value) = strvalue.ConvertToFloat();
659 }
660 break;
661
662 case XFILECFG_VALUETYPE_STRING : { XSTRING* value = (XSTRING*)cfgvalue->GetValue();
663 if(!value) return false;
664
665 (*value) = strvalue.Get();
666
667 value->DeleteCharacter(__C('\t'));
668 value->DeleteCharacter(__C('\n'));
669 value->DeleteCharacter(__C('\r'));
670
671 value->DeleteCharacter(__C(' '), XSTRINGCONTEXT_TO_END);
672 }
673 break;
674
675 case XFILECFG_VALUETYPE_BOOLEAN : { bool* value = (bool*)cfgvalue->GetValue();
676 if(!value) return false;
677
678 (*value) = strvalue.ConvertToBoolean();
679 }
680 break;
681 }
682 }
683 }
684 }
685
686 status = EndFile();
687 }
688
689 return status;
690}
