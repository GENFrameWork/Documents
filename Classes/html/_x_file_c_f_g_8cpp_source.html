606bool XFILECFG::Load(XPATH& xpath)
607{
608 bool status = false;
609
610 if(values.IsEmpty()) return status;
611
612 if(fileini->IsOpen())
613 {
614 EndFile();
615 }
616
617 if(IniFile(xpath))
618 {
619 AjustRemarks();
620
621 for(int c=0;c<(int)values.GetSize();c++)
622 {
623 XFILECFGVALUE* cfgvalue = (XFILECFGVALUE*)values.Get(c);
624 if(cfgvalue)
625 {
626 XSTRING strvalue;
627
628 if(fileini->ReadValue(cfgvalue->GetGroup()->Get(), cfgvalue->GetID()->Get() , strvalue))
629 {
630 switch(cfgvalue->GetType())
631 {
632 case XFILECFG_VALUETYPE_UNKNOWN : break;
633
634 case XFILECFG_VALUETYPE_INT : { int* value = (int*)cfgvalue->GetValue();
635 if(!value) return false;
636
637 (*value) = strvalue.ConvertToInt();
638 }
639 break;
640
641 case XFILECFG_VALUETYPE_MASK : { XWORD* value = (XWORD*)cfgvalue->GetValue();
642 if(!value) return false;
643
644 strvalue.UnFormat(__L("%04X"), value);
645 }
646 break;
647
648 case XFILECFG_VALUETYPE_FLOAT : { float* value = (float*)cfgvalue->GetValue();
649 if(!value) return false;
650
651 (*value) = strvalue.ConvertToFloat();
652 }
653 break;
654
655 case XFILECFG_VALUETYPE_STRING : { XSTRING* value = (XSTRING*)cfgvalue->GetValue();
656 if(!value) return false;
657
658 (*value) = strvalue.Get();
659
660 value->DeleteCharacter(__C('\t'));
661 value->DeleteCharacter(__C('\n'));
662 value->DeleteCharacter(__C('\r'));
663
664 value->DeleteCharacter(__C(' '), XSTRINGCONTEXT_TO_END);
665 }
666 break;
667
668 case XFILECFG_VALUETYPE_BOOLEAN : { bool* value = (bool*)cfgvalue->GetValue();
669 if(!value) return false;
670
671 (*value) = strvalue.ConvertToBoolean();
672 }
673 break;
674 }
675 }
676 }
677 }
678
679 status = EndFile();
680 }
681
682 return status;
683}
