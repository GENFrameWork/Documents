103bool DIOSPIOLEDDISPLAYSSD1331::IniDevice()
104{
105 if(!DIODEVICESPI::IniDevice()) return false;
106
107 diostreamcfg->SetMode(DIOSTREAMMODE_MASTER);
108 diostreamcfg->SetSPIMode(DIOSTREAMSPI_MODE_3);
109 diostreamcfg->SetNBitsWord(8);
110 diostreamcfg->SetSpeed(16*1000*1000);
111 diostreamcfg->SetDelay(0);
112 diostreamcfg->SetIsOnlyWrite(true);
113
114
115 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) != DIOGPIO_ID_NOTDEFINED)
116 {
117 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , DIOGPIO_MODE_OUTPUT);
118 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , true);
119 }
120
121 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) != DIOGPIO_ID_NOTDEFINED)
122 {
123 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , DIOGPIO_MODE_OUTPUT);
124 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , true);
125 }
126
127 if(diostream->Open())
128 {
129 if(diostream->WaitToConnected(timeout))
130 {
131 width = 96;
132 height = 64;
133
134 sizebuffer = ((width*height)*(sizeof(XWORD)));
135
136 if(!Initializate()) return false;
137
138 return DIODEVICE::Ini();
139 }
140 }
141
142 return false;
143}
