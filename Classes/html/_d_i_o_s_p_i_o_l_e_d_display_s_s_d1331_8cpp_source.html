118 diostreamcfg->SetDelay(0);
119 diostreamcfg->SetIsOnlyWrite(true);

121
122 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) != DIOGPIO_ID_NOTDEFINED)

130 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , DIOGPIO_MODE_OUTPUT);
131 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , true);
132 }
133
134 if(diostream->Open())
135 {
136 if(diostream->WaitToConnected(timeout))
137 {
138 width = 96;
139 height = 64;
140
141 sizebuffer = ((width*height)*(sizeof(XWORD)));
142
143 if(!Initializate()) return false;
144
