484bool XDIR::Copy(XPATH& xpathsource,XPATH& xpathtarget)
485{
486 #define COPYFILE_MAXBUFFER 2048
487
488 bool status = true;
489
490 XFILE* GEN_XFACTORY_CREATE(fsource, Create_File())
491 XFILE* GEN_XFACTORY_CREATE(ftarget, Create_File());
492
493 if(fsource && ftarget)
494 {
495 if(fsource->Open(xpathsource))
496 {
497 if(ftarget->Create(xpathtarget))
498 {
499 XBYTE* buffer = new XBYTE[COPYFILE_MAXBUFFER];
500 if(buffer)
501 {
502 XDWORD rsize = 0;
503
504 do { rsize = COPYFILE_MAXBUFFER;
505
506 memset(buffer,0,rsize);
507
508 fsource->Read(buffer,&rsize);
509
510 if(rsize) ftarget->Write(buffer,rsize);
511
512 } while(rsize == COPYFILE_MAXBUFFER);
513
514 delete[] buffer;
515 }
516
517 ftarget->Close();
518
519 } else status = false;
520
521 fsource->Close();
522
523 } else status = false;
524
525 } else status = false;
526
527 GEN_XFACTORY.Delete_File(fsource);
528 GEN_XFACTORY.Delete_File(ftarget);
529
530 return status;
531}
