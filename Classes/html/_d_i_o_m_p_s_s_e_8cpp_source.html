156bool DIOMPSSE::Open(int index)
157{
158 if(!dioenumUSBdevices) return false;
159
160 EnumDevices();
161
162 XVECTOR<DIOSTREAMDEVICE*>* devices = dioenumUSBdevices->GetDevices();
163 if(!devices) return false;
164 if(devices->IsEmpty()) return false;
165
166 USBdevice = (DIOSTREAMDEVICEUSB*)devices->Get(index);
167 if(!USBdevice) return false;
168
169 switch(USBdevice->GetDescriptor()->GetBCDDevice())
170 {
171 case 0x200 : if(USBdevice->GetDescriptor()->GetSerialNumber() == 0) chiptype = DIOMPSSE_CHIP_TYPE_BM;
172 break;
173
174 case 0x400 : chiptype = DIOMPSSE_CHIP_TYPE_BM; break;
175 case 0x500 : chiptype = DIOMPSSE_CHIP_TYPE_2232C; break;
176 case 0x600 : chiptype = DIOMPSSE_CHIP_TYPE_R; break;
177 case 0x700 : chiptype = DIOMPSSE_CHIP_TYPE_2232H; break;
178 case 0x800 : chiptype = DIOMPSSE_CHIP_TYPE_4232H; break;
179 case 0x900 : chiptype = DIOMPSSE_CHIP_TYPE_232H; break;
180
181 default : chiptype = DIOMPSSE_CHIP_TYPE_UNKNOWN; break;
182 }
183
184 diostreamUSBcfg->GetResource()->Set((*USBdevice->GetResource()));
185
186 if(diostreambase)
187 {
188 if(diostreambase->IsConnected()) Close();
189 GEN_DIOFACTORY.DeleteStreamIO(diostreambase);
190
191 diostreambase = NULL;
192 }
193
194 diostreambase = GEN_DIOFACTORY.CreateStreamIO(diostreamUSBcfg);
195 if(!diostreambase) return false;
196
197 if(!diostreambase->Open()) return false;
198
199 if(!Reset()) return false;
200
201 isopen = true;
202
203 return isopen;
204}

376bool DIOMPSSE::EnumDevices()
377{
378 if(!dioenumUSBdevices) return false;
379
380 if(!dioenumUSBdevices->Search()) return false;
381
382 XVECTOR<DIOSTREAMDEVICE*>* devices = dioenumUSBdevices->GetDevices();
383 if(!devices) return false;
384 if(devices->IsEmpty()) return false;
385
386 DIOSTREAMDEVICEUSB* device;
387 int index = 0;
388
389 do{ device = (DIOSTREAMDEVICEUSB*)devices->Get(index);
390 if(device)
391 {
392 if((device->GetVendorID() == 0x403) && ((device->GetProductID() == 0x6001) ||
393 (device->GetProductID() == 0x6010) ||
394 (device->GetProductID() == 0x6011) ||
395 (device->GetProductID() == 0x6014)))
396 {
397 index++;
398 }
399 else
400 {
401 devices->Delete(device);
402 delete device;
403 }
404 }
405
406 } while(device);
407
408 return true;
409}

424bool DIOMPSSE::SendCommand(XBYTE command, XBUFFER& buffer)
425{
426 XBUFFER datasend;
427 XBYTE lowsize;
428 XBYTE highsize;
429 bool status;
430
431 if(buffer.GetSize() > 65535) return false;
432
433 lowsize = (buffer.GetSize() & 0x00FF);
434 highsize = ((buffer.GetSize() & 0xFF00) >> 0x08);
435
436 datasend.Add((XBYTE)command);
437 datasend.Add((XBYTE)lowsize);
438 datasend.Add((XBYTE)highsize);
439
440 datasend.Add(buffer);
441
442 status = diostreambase->Write(datasend);
443 if(status) status = diostreambase->WaitToFlushOutXBuffer(5);
444
445 return status;
446}
