194bool CIPHERKEYPRIVATERSA::Check(CIPHERKEYPUBLICRSA& publickey)
195{
196 XMPINTEGER PQ;
197 XMPINTEGER DE;
198 XMPINTEGER P1;
199 XMPINTEGER Q1;
200 XMPINTEGER H;
201 XMPINTEGER I;
202 XMPINTEGER G;
203 XMPINTEGER G2;
204 XMPINTEGER L1;
205 XMPINTEGER L2;
206 XMPINTEGER publicmodulus;
207 XMPINTEGER publicexponent;
208
209 if(!publickey.Check()) return false;
210
211 if((!prime1factor.GetLimbs()) || (!prime2factor.GetLimbs()) || (!exponent.GetLimbs())) return false;
212
213 publickey.Get(publicmodulus, publicexponent);
214
215 PQ.Ini();
216 DE.Ini();
217 P1.Ini();
218 Q1.Ini();
219 H.Ini();
220 I.Ini();
221 G.Ini();
222 G2.Ini();
223 L1.Ini();
224 L2.Ini();
225
226 if(!PQ.Multiplication(&prime1factor, &prime2factor)) return false;
227 if(!DE.Multiplication(&exponent, &publicexponent)) return false;
228 if(!P1.SubtractionSigned(&prime1factor, 1)) return false;
229 if(!Q1.SubtractionSigned(&prime2factor, 1)) return false;
230 if(!H.Multiplication(&P1, &Q1)) return false;
231 if(!G.GreatestCommonDivisor(&G, &publicexponent, &H)) return false;
232 if(!G2.GreatestCommonDivisor(&G2, &P1, &Q1)) return false;
233 if(!L1.Division(&L1, &L2, &H, &G2 )) return false;
234 if(!I.Module(&I, &DE, &L1)) return false;
235
236 // Check for a valid PKCS1v2 private key
237 if((PQ.CompareSignedValues(publicmodulus) != 0) ||
238 (L2.CompareSignedValues(0) != 0) ||
239 (I.CompareSignedValues(1) != 0) ||
240 (G.CompareSignedValues(1) != 0)) return false;
241
242 PQ.End();
243 DE.End();
244 P1.End();
245 Q1.End();
246 H.End();
247 I.End();
248 G.End();
249 G2.End();
250 L1.End();
251 L2.End();
252
253 return true;
254}
