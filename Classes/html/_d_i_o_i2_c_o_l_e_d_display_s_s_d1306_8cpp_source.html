132bool DIOI2COLEDDISPLAYSSD1306::IniDevice()
133{
134 if(!DIODEVICEI2C::IniDevice()) return false;
135
136 if(diostream->Open())
137 {
138 if(diostream->WaitToConnected(timeout))
139 {
140 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFF); // 0xAE
141 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_CLOCK_DIV_RATIO_SET); // 0xD5
142 SendCommand(0xF0); // Suggest ratio
143 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_MULTIPLEX_RADIO_SET); // 0xA8
144 SendCommand((64 - 1)); // Display_Height - 1
145
146 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFFSET_SET); // 0xD3 + No offset
147 SendCommand(00); // No offset
148 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_CHARGE_PUMP_SET); // 0x8D
149 SendCommand(0x14); // Enable charge pump
150 SendCommand((DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_START_LINE_SET | 0x00)); // 0x40 | Start line
151 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_MEMORY_ADDRESSING_SET); // 0x20
152 SendCommand(0x00); // Horizontal Addressing - Operate like KS0108
153 SendCommand((DIOI2COLEDDISPLAYSSD1306_COMMAND_SEGMENT_REMAP_SET | 0x01)); // 0xA0 - Left towards Right
154 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_COM_OUTPUT_SCAN_DEC); // 0xC8 - Up towards Down
155 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_COM_PINS_SET); // 0xDA
156 //SendCommand(0x12); // Sequential COM pin configuration
157
158 if((width == 128) && (height == 64)) SendCommand(0x12); // Sequential COM pin configuration
159 if((width == 128) && (height == 32)) SendCommand(0x02); // Sequential COM pin configuration
160 if((width == 96) && (height == 16)) SendCommand(0x02); // Sequential COM pin configuration
161
162 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_CONSTRAST_SET); // 0x81
163 SendCommand(0xFF); // 0xFF
164 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_PRECHARGE_PERIOD_SET); // 0xD9
165 SendCommand(0xF1); // 0xF1
166
167 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_VCOMH_DESELECT_LEVEL_SET); // 0xDB
168 SendCommand(0x20); // 0x20
169
170 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_ALL_ON_RESUME); // 0xA4
171 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_NORMAL); // 0xA6
172 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_SCROLL_DEACTIVATE); // 0x2E

201bool DIOI2COLEDDISPLAYSSD1306::Clear(XBYTE color)
202{
203 XBYTE nblocks;
204
205 nblocks = (width >> 4) * (height >> 3);
206
207 // Set columns
208 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_COLUMN_ADDRESS_SET); // 0x21
209 SendCommand(0x00); // Start
210 SendCommand(width - 1); // End
211
212 //Set rows
213 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_PAGE_ADDRESS_SET); // 0x22
214 SendCommand(0x00); // Start
215 SendCommand(height - 1); // End
216
217 XBYTE buffer[16];
218
219 memset(buffer, color, 16);
220
221 //Send buffer
222 for(XBYTE c=0; c<nblocks; c++)
223 {
224 SendData(buffer, 16);
225 }
226
227 return true;
228}

282bool DIOI2COLEDDISPLAYSSD1306::PutPixel(XWORD x, XWORD y, bool color)
283{
284 if(!buffer) return false;
285
286 // Goto to point
287 SetXY((XBYTE)x, (XBYTE)y);
288
289 XBYTE data = 0;
290
291 // Read data
292 data = buffer[DIOI2COLEDDISPLAYSSD1306_POINTER(x, y)];
293
294 //Set pixel
295 if(color)
296 DIOI2COLEDDISPLAYSSD1306_BITSET(data, y % 8);
297 else DIOI2COLEDDISPLAYSSD1306_BITCLEAR(data, y % 8);
298
299 //Sent data
300 buffer[DIOI2COLEDDISPLAYSSD1306_POINTER(x, y)] = data;
301
302 return true;
303}
