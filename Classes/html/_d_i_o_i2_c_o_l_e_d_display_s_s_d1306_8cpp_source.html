140{
141 if(!DIODEVICEI2C::IniDevice()) return false;
142
143 if(diostream->Open())
144 {
145 if(diostream->WaitToConnected(timeout))

147 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFF); // 0xAE

151 SendCommand((64 - 1)); // Display_Height - 1
152

154 SendCommand(00); // No offset
155 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_CHARGE_PUMP_SET); // 0x8D
156 SendCommand(0x14); // Enable charge pump
157 SendCommand((DIOI2COLEDDISPLAYSSD1306_COMMAND_DISPLAY_START_LINE_SET | 0x00)); // 0x40 | Start line
158 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_MEMORY_ADDRESSING_SET); // 0x20
159 SendCommand(0x00); // Horizontal Addressing - Operate like KS0108

161 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_COM_OUTPUT_SCAN_DEC); // 0xC8 - Up towards Down

208bool DIOI2COLEDDISPLAYSSD1306::Clear(XBYTE color)
209{
210 XBYTE nblocks;
211
212 nblocks = (width >> 4) * (height >> 3);
213
214 // Set columns
215 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_COLUMN_ADDRESS_SET); // 0x21
216 SendCommand(0x00); // Start
217 SendCommand(width - 1); // End
218
219 //Set rows
220 SendCommand(DIOI2COLEDDISPLAYSSD1306_COMMAND_PAGE_ADDRESS_SET); // 0x22
221 SendCommand(0x00); // Start
222 SendCommand(height - 1); // End
223
224 XBYTE buffer[16];
225
226 memset(buffer, color, 16);
227
228 //Send buffer
229 for(XBYTE c=0; c<nblocks; c++)
230 {
231 SendData(buffer, 16);
232 }
233
234 return true;
235}

289bool DIOI2COLEDDISPLAYSSD1306::PutPixel(XWORD x, XWORD y, bool color)
290{
291 if(!buffer) return false;
292
293 // Goto to point
294 SetXY((XBYTE)x, (XBYTE)y);
295
296 XBYTE data = 0;
297
298 // Read data
299 data = buffer[DIOI2COLEDDISPLAYSSD1306_POINTER(x, y)];
300
301 //Set pixel
302 if(color)
303 DIOI2COLEDDISPLAYSSD1306_BITSET(data, y % 8);
304 else DIOI2COLEDDISPLAYSSD1306_BITCLEAR(data, y % 8);
305
306 //Sent data
307 buffer[DIOI2COLEDDISPLAYSSD1306_POINTER(x, y)] = data;
308
309 return true;
310}

506bool DIOI2COLEDDISPLAYSSD1306::Send(XBYTE control, XBYTE* data, int sizedata)
507{

511
