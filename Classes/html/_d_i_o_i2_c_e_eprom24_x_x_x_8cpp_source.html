152bool DIOI2CEEPROM24XXX::Read(XDWORD address, XBYTE* buffer, XDWORD size)
153{
154 if(!diostream) return false;
155 if(!IsInitialized()) return false;
156
157 if(diostreamcfg->GetRemoteAddress() == DIOSTREAMI2CCONFIG_ADDRESSINVALID) return false;
158
159 if((address+size)>maxsize) return false;
160
161 bool status = true;
162
163 for(XDWORD c=0; c<size; c++)
164 {
165 xtimerout->Reset();
166
167 while(1)
168 {
169 status = ReadBYTE(address+c, buffer[c]);
170 if(!status)
171 {
172 if(xtimerout->GetMeasureSeconds()>(XDWORD)timeout) break;
173
174 } else break;
175 }
176 }
177
178 return status;

345bool DIOI2CEEPROM24XXX::ReadBYTE(XDWORD address, XBYTE& data)
346{
347 if(!diostream) return false;
348 if(!IsInitialized()) return false;
349
350 XBYTE remotebasebackup = (XBYTE)diostreamcfg->GetRemoteAddress();
351 XBYTE remotebase = remotebasebackup;
352 XBUFFER xbuffer;
353 bool status;
354
355 switch(type)
356 {
357 case DIOI2CEEPROM24XXXTYPE_UNKNOWN :
358 default : return false;
359
360 case DIOI2CEEPROM24XXXTYPE_08 :
361 case DIOI2CEEPROM24XXXTYPE_16 : { if((address)>=maxsize) return false;
362
363 XBYTE page = (XBYTE)(address / maxsizepage);
364 remotebase+=(XBYTE)page;
365
366 XBYTE offset = (address % maxsizepage);

368 xbuffer.Add((XBYTE)offset);
369
370 //XTRACE_PRINTCOLOR(1, __L("%d, %d"), page, offset);

372 break;

374 case DIOI2CEEPROM24XXXTYPE_65 :

376 xbuffer.Add((XWORD)address);

381 if(diostream->Write(xbuffer)) status = diostream->WaitToFlushOutXBuffer(timeout);

386 //XTRACE_PRINTCOLOR(1, __L("%02X, %02X"), xbuffer.GetByte(0), data);
