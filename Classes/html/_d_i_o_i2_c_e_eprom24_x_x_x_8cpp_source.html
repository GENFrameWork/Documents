160{
161 if(!diostream) return false;
162 if(!IsInitialized()) return false;
163
164 if(diostreamcfg->GetRemoteAddress() == DIOSTREAMI2CCONFIG_ADDRESSINVALID) return false;
165
166 if((address+size)>maxsize) return false;
167
168 bool status = true;
169

172 xtimerout->Reset();

352bool DIOI2CEEPROM24XXX::ReadBYTE(XDWORD address, XBYTE& data)
353{
354 if(!diostream) return false;
355 if(!IsInitialized()) return false;
356
357 XBYTE remotebasebackup = (XBYTE)diostreamcfg->GetRemoteAddress();
358 XBYTE remotebase = remotebasebackup;
359 XBUFFER xbuffer;
360 bool status;
361
362 switch(type)
363 {
364 case DIOI2CEEPROM24XXXTYPE_UNKNOWN :
365 default : return false;
366
367 case DIOI2CEEPROM24XXXTYPE_08 :
368 case DIOI2CEEPROM24XXXTYPE_16 : { if((address)>=maxsize) return false;
369
370 XBYTE page = (XBYTE)(address / maxsizepage);
371 remotebase+=(XBYTE)page;
372
373 XBYTE offset = (address % maxsizepage);
374
375 xbuffer.Add((XBYTE)offset);
376 }
377 break;
378
379 case DIOI2CEEPROM24XXXTYPE_65 :
380 case DIOI2CEEPROM24XXXTYPE_515 : if((address)>=maxsize) return false;
381 xbuffer.Add((XWORD)address);
382 break;
383 }
384
385 diostreamcfg->SetRemoteAddress(remotebase);
386 if(diostream->Write(xbuffer)) status = diostream->WaitToFlushOutXBuffer(timeout);
387
388 status = diostream->WaitToFilledReadingBuffer(1, timeout);
389 if(status) status = diostream->Read(&data, 1)?true:false;
390
391 diostreamcfg->SetRemoteAddress(remotebasebackup);
