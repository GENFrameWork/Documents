104bool GRPWINDOWSCONTEXT::Create(void* handle)
105{
106 // deal with weird windows context creation
107
108
109 #ifdef GRP_OPENGL_ACTIVE
110
111 GRPWINDOWSSCREEN* windowsscreen = (GRPWINDOWSSCREEN*)this->screen;
112 if(!windowsscreen) return false;
113
114 // Choose the corect pixel format
115 windowsscreen->ChooseValidPixelFormat();
116 // make dummy opengl context
117
118 hdlopenglrc = wglCreateContext(windowsscreen->GetHDC());
119 wglMakeCurrent(windowscreen->GetHDC(), hdlopenglrc);
120
121 glewExperimental = true;
122 if(glewInit() != GLEW_OK) return false;
123
124 // Set the pixel format usiong wgl extensions
125 int nPixelFormat;
126 BOOL bValidPixFormat;
127 UINT nMaxFormats = 1;
128 UINT nNumFormats;
129 float pfAttribFList[] = { 0, 0 };
130 int piAttribIList[] = { WGL_DRAW_TO_WINDOW_ARB, GL_TRUE,
131 WGL_SUPPORT_OPENGL_ARB, GL_TRUE,
132 WGL_ACCELERATION_ARB, WGL_FULL_ACCELERATION_ARB,
133 WGL_COLOR_BITS_ARB, 32,
134 WGL_ALPHA_BITS_ARB, 8,
135 WGL_DEPTH_BITS_ARB, 16,
136 WGL_STENCIL_BITS_ARB, 8,
137 WGL_DOUBLE_BUFFER_ARB, GL_TRUE,
138 WGL_PIXEL_TYPE_ARB, WGL_TYPE_RGBA_ARB,
139 WGL_SAMPLE_BUFFERS_ARB, (multisampling != 0) ? 1 : 0, //Number of mutisample buffers (must be 1 at time of writing)
140 WGL_SAMPLES_ARB, (int)multisampling, //Number of samples
141 0, 0 };
142
143 bValidPixFormat = wglChoosePixelFormatARB(windowsscreen->GetHDC(), piAttribIList, pfAttribFList, nMaxFormats, &nPixelFormat, &nNumFormats );
144 if(!bValidPixFormat)
145 {
146 XTRACE_PRINTCOLOR(4,__L("Invalid Pixel Format! Try back..."));
147
148 const int attribList[] ={ WGL_DRAW_TO_WINDOW_ARB , GL_TRUE ,
149 WGL_SUPPORT_OPENGL_ARB , GL_TRUE ,
150 WGL_DOUBLE_BUFFER_ARB , GL_TRUE ,
151 WGL_PIXEL_TYPE_ARB , WGL_TYPE_RGBA_ARB ,
152 WGL_COLOR_BITS_ARB , 32 ,
153 WGL_DEPTH_BITS_ARB , 24 ,
154 WGL_STENCIL_BITS_ARB , 8 ,
155 0, //End
156 };
157
158 bValidPixFormat = wglChoosePixelFormatARB(windowsscreen->GetHDC(), attribList, pfAttribFList, nMaxFormats, &nPixelFormat, &nNumFormats );
159 if(!bValidPixFormat)
160 {
161 XTRACE_PRINTCOLOR(4,__L("\n\n\n\nInvalid Pixel Format!"));
162 return false;
163 }
164 }
165
166 windowsscreen->SetPixelFormat(nPixelFormat);
167
168 // delete old window
169 windowsscreen->Delete();
170
171 // make real window
172 windowsscreen->Create(handle);
173
174 // delete old context
175 Destroy();
176
177 // make real context
178 hdlopenglrc = wglCreateContext(windowsscreen->GetHDC());
179 wglMakeCurrent(windowsscreen->GetHDC(), hdlopenglrc);
180
181 openglctrl.Ini();
182
183 return true;
184
185 #else
186
187 return false;
188
189 #endif
190}
