111void ANDROIDNATIVEACTIVITY_EVENTLOOP::Run(ANDROIDNATIVEACTIVITY_HANDLER* activityhandler, ANDROIDNATIVEINPUT_HANDLER* inputhandler)
112{
113 int32_t result;
114 int32_t events;
115 android_poll_source* source;
116
117 //app_dummy();
118
119 this->activityhandler = activityhandler;
120 this->inputhandler = inputhandler;
121
122 while (true)
123 {
124 // Event processing loop.
125 while((result = ALooper_pollOnce(enabled?0:-1, NULL, &events, (void**)&source)) >= 0)
126 {
127 // An event has to be processed.
128 if(source != NULL) source->process(application, source);
129
130 // Application is getting destroyed.
131 if(application->destroyRequested) return;
132 }
133
134 if((enabled) && (!quit))
135 {
136 if(activityhandler->OnStep() != STATUS_OK)
137 {
138 quit = true;
139 ANativeActivity_finish(application->activity);
140 }
141 }
142 }
143}

202int32_t ANDROIDNATIVEACTIVITY_EVENTLOOP::ProcessInputEvent(AInputEvent* event, android_app* application)
203{
204 int32_t eventtype = AInputEvent_getType(event);
205
206 this->application = application;
207
208 switch(eventtype)
209 {
210 case AINPUT_EVENT_TYPE_MOTION : switch (AInputEvent_getSource(event))
211 {
212 case AINPUT_SOURCE_TOUCHSCREEN : return inputhandler->OnTouchEvent(event); break;
213 case AINPUT_SOURCE_TRACKBALL : return inputhandler->OnTrackballEvent(event); break;
214 }
215 break;
216
217 case AINPUT_EVENT_TYPE_KEY : return inputhandler->OnKeyboardEvent(event); break;
218 }
219
220 return 0;
221}

234void ANDROIDNATIVEACTIVITY_EVENTLOOP::ProcessApplicationEvent(int32_t command, android_app* application)
235{
236 this->application = application;
237
238 switch(command)
239 {
240 case APP_CMD_CONFIG_CHANGED : activityhandler->OnConfigurationChanged();
241 break;
242
243 case APP_CMD_INIT_WINDOW : activityhandler->OnCreateWindow();
244 break;
245
246 case APP_CMD_DESTROY : activityhandler->OnDestroy();
247 break;
248
249 case APP_CMD_GAINED_FOCUS : Activate();
250 activityhandler->OnGainFocus();
251 break;
252
253 case APP_CMD_LOST_FOCUS : activityhandler->OnLostFocus();
254 Deactivate();
255 break;
256
257 case APP_CMD_LOW_MEMORY : activityhandler->OnLowMemory();
258 break;
259
260 case APP_CMD_PAUSE : activityhandler->OnPause();
261 Deactivate();
262 break;
263
264 case APP_CMD_RESUME : Activate();
265 activityhandler->OnResume();
266 break;
267
268 case APP_CMD_SAVE_STATE : activityhandler->OnSaveState(&application->savedState, &application->savedStateSize);
269 break;
270
271 case APP_CMD_START : activityhandler->OnStart();
272 break;
273
274 case APP_CMD_STOP : activityhandler->OnStop();
275 break;
276
277 case APP_CMD_TERM_WINDOW : activityhandler->OnDestroyWindow();
278 Deactivate();
279 break;
280
281 default : break;
282 }
283}
