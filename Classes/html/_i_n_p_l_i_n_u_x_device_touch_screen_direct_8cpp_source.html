137bool INPLINUXDEVICETOUCHSCREENDIRECT::Update()
138{
139 if((!created)||(!enabled)) return false;
140
141 struct input_event event;
142
143 int eventsize = sizeof(struct input_event);
144 int bytesread = 0;
145
146 //--------------- clean up old states
147
148 INPBUTTON* button = GetButton(INPBUTTON_ID_TOUCHSCREEN);
149 if(button)
150 {
151 button->SetPressed(false);
152
153 touchscr_x = -1;
154 touchscr_y = -1;
155
156 UpdateTouchScreenData();
157 }
158
159 //--------------- process
160
161 if(touchscr.GetSize() == 0)
162 {
163 OpenAllDevicesID();
164 }
165
166 for(XDWORD e=0; e<touchscr.GetSize(); e++)
167 {
168 while(true)
169 {
170 bytesread=0;
171
172 memset(&event, 0, sizeof(event));
173
174 errno = 0;
175 bytesread = read(touchscr.Get(e)->GetFileDescriptor(), &event, sizeof(event)/*eventsize*/);
176
177 // if device is diconnected we receive a ENODEV message (happens for example every time a touchscreen is poweredoff)
178
179 if(bytesread == -1 && errno == ENODEV)
180 {
181 // we lost the handle to the event file, need to reclaim it
182 touchscr.DeleteContents();
183 OpenAllDevicesID();
184 }
185
186 if(bytesread==-1 && errno!=EAGAIN) break;
187
188 if((bytesread < sizeof(struct input_event)) || (bytesread == -1)) break;
189
190 if(bytesread == sizeof(event)/*eventsize*/)
191 {
192 switch(event.type)
193 {
194 case EV_ABS : switch(event.code)
195 {
196 case ABS_X : { touchscr_x = event.value;
197 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Touch] x: %d"), touchscr_x);
198
199 }
200 break;
201
202 case ABS_Y : { touchscr_y = grpscreen->GetHeight() - event.value;
203 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Touch] y: %d"), touchscr_y);
204
205 INPBUTTON* button = GetButton(INPBUTTON_ID_TOUCHSCREEN);
206 if(button) button->SetPressed(true);
207 }
208 break;
209 }
210 break;
211
212 default : continue;
213 }
214
215 } else break;
216 }
217 }
218
219 UpdateTouchScreenData();
220
221 return true;
222}

275bool INPLINUXDEVICETOUCHSCREENDIRECT::OpenAllDevicesID()
276{
277 INPLINUXFACTORY* inpdevices = (INPLINUXFACTORY*)(&INPFACTORY::GetInstance());
278 XVECTOR<INPLINUXDEVICEID*> devicesID;
279
280 if(inpdevices) inpdevices->GetDeviceHandlers(INPDEVICE_TYPE_TOUCHSCREEN, devicesID);
281
282 for(XDWORD c=0; c<devicesID.GetSize(); c++)
283 {
284 INPLINUXDEVICEID* deviceID = devicesID.Get(c);
285 if(deviceID)
286 {
287 XSTRING event;
288 int filedescriptor = INPLINUXDEVICEID_INVALID;
289
290 event.Format(__L("/dev/input/event%d"), deviceID->GetEventIndex());
291
292 XBUFFER charstr;
293
294 event.ConvertToASCII(charstr);
295 filedescriptor = open(charstr.GetPtrChar(), O_RDONLY | O_NONBLOCK);
296
297 if(filedescriptor != -1)
298 {
299 char mouseOEMname[_MAXSTR];
300 XSTRING namehardware;
301
302 // we get the keyboard name
303 memset(mouseOEMname, 0, _MAXSTR);
304
305 ioctl(filedescriptor, EVIOCGNAME(_MAXSTR), mouseOEMname);
306 namehardware.Set(mouseOEMname);
307
308 deviceID->SetFileDescriptor(filedescriptor);
309 //deviceID->GetName()->Set(namehardware);
310
311 touchscr.Add(deviceID);
312 }
313 }
314 }
315
316 devicesID.DeleteAll();
317
318 return touchscr.GetSize()?true:false;
319}
