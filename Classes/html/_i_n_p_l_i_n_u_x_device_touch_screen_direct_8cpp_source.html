144bool INPLINUXDEVICETOUCHSCREENDIRECT::Update()
145{
146 if((!created)||(!enabled)) return false;
147
148 struct input_event event;
149
150 int eventsize = sizeof(struct input_event);
151 int bytesread = 0;
152
153 //--------------- clean up old states
154
155 INPBUTTON* button = GetButton(INPBUTTON_ID_TOUCHSCREEN);
156 if(button)
157 {
158 button->SetPressed(false);
159
160 touchscr_x = -1;
161 touchscr_y = -1;
162
163 UpdateTouchScreenData();
164 }
165
166 //--------------- process
167
168 if(touchscr.GetSize() == 0)
169 {
170 OpenAllDevicesID();
171 }
172
173 for(XDWORD e=0; e<touchscr.GetSize(); e++)
174 {
175 while(true)
176 {
177 bytesread=0;
178
179 memset(&event, 0, sizeof(event));
180
181 errno = 0;
182 bytesread = read(touchscr.Get(e)->GetFileDescriptor(), &event, sizeof(event)/*eventsize*/);
183
184 // if device is diconnected we receive a ENODEV message (happens for example every time a touchscreen is poweredoff)
185
186 if(bytesread == -1 && errno == ENODEV)
187 {
188 // we lost the handle to the event file, need to reclaim it
189 touchscr.DeleteContents();
190 OpenAllDevicesID();
191 }
192
193 if(bytesread==-1 && errno!=EAGAIN) break;
194
195 if((bytesread < sizeof(struct input_event)) || (bytesread == -1)) break;
196
197 if(bytesread == sizeof(event)/*eventsize*/)
198 {
199 switch(event.type)
200 {
201 case EV_ABS : switch(event.code)
202 {
203 case ABS_X : { touchscr_x = event.value;
204 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Touch] x: %d"), touchscr_x);
205
206 }
207 break;
208
209 case ABS_Y : { touchscr_y = grpscreen->GetHeight() - event.value;
210 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Touch] y: %d"), touchscr_y);
211
212 INPBUTTON* button = GetButton(INPBUTTON_ID_TOUCHSCREEN);
213 if(button) button->SetPressed(true);
214 }
215 break;
216 }
217 break;
218
219 default : continue;
220 }
221
222 } else break;
223 }
224 }
225
226 UpdateTouchScreenData();
227
228 return true;
229}

282bool INPLINUXDEVICETOUCHSCREENDIRECT::OpenAllDevicesID()
283{
284 INPLINUXFACTORY* inpdevices = (INPLINUXFACTORY*)(&INPFACTORY::GetInstance());
285 XVECTOR<INPLINUXDEVICEID*> devicesID;
286
287 if(inpdevices) inpdevices->GetDeviceHandlers(INPDEVICE_TYPE_TOUCHSCREEN, devicesID);
288
289 for(XDWORD c=0; c<devicesID.GetSize(); c++)
290 {
291 INPLINUXDEVICEID* deviceID = devicesID.Get(c);
292 if(deviceID)
293 {
294 XSTRING event;
295 int filedescriptor = INPLINUXDEVICEID_INVALID;
296
297 event.Format(__L("/dev/input/event%d"), deviceID->GetEventIndex());
298
299 XBUFFER charstr;
300
301 event.ConvertToASCII(charstr);
302 filedescriptor = open(charstr.GetPtrChar(), O_RDONLY | O_NONBLOCK);
303
304 if(filedescriptor != -1)
305 {
306 char mouseOEMname[_MAXSTR];
307 XSTRING namehardware;
308
309 // we get the keyboard name
310 memset(mouseOEMname, 0, _MAXSTR);
311
312 ioctl(filedescriptor, EVIOCGNAME(_MAXSTR), mouseOEMname);
313 namehardware.Set(mouseOEMname);
314
315 deviceID->SetFileDescriptor(filedescriptor);
316 //deviceID->GetName()->Set(namehardware);
317
318 touchscr.Add(deviceID);
319 }
320 }
321 }
322
323 devicesID.DeleteAll();
324
325 return touchscr.GetSize()?true:false;
326}

364bool INPLINUXDEVICETOUCHSCREENDIRECT::UpdateTouchScreenData()

374

376 {
377 cursor->Set(touchscr_x, touchscr_y);
