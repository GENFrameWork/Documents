120bool GRPVIDEOFILEAVI::Open(XCHAR* path)
121{
122 if(!fileRIFF->Open(path)) return false;
123
124 if(!fileRIFF->ReadAllLists()) return false;
125
126 XDWORD sizedata = 0;
127
128 memset((XBYTE*)&mainheader , 0, sizeof(GRPVIDEOFILEAVI_MAINHEADER));
129 memset((XBYTE*)&streamheader , 0, sizeof(GRPVIDEOFILEAVI_STREAMHEADER));
130 memset((XBYTE*)&bitmapheader , 0, sizeof(GRPVIDEOFILEAVI_BITMAPHEADER));
131
132 XFILERIFF_LIST* chunk_mainheader = fileRIFF->GetChunk(__L("avih"), __L("hdrl"));
133 if(!chunk_mainheader) return false;
134
135 fileRIFF->GetChunkData(chunk_mainheader, (XBYTE*)&mainheader, sizedata);
136
137 XFILERIFF_LIST* strl_list = fileRIFF->GetList(__L("strl"), __L("hdrl"));
138 if(!strl_list) return false;
139
140 XFILERIFF_LIST* strh_chunk = fileRIFF->GetChunk(__L("strh"), __L("strl"));
141 if(!strh_chunk) return false;
142
143 fileRIFF->GetChunkData(strh_chunk, (XBYTE*)&streamheader, sizedata);
144
145 XFILERIFF_LIST* strf_chunk = fileRIFF->GetChunk(__L("strf"), __L("strl"));
146 if(!strf_chunk) return false;
147
148 fileRIFF->GetChunkData(strf_chunk, (XBYTE*)&bitmapheader, sizedata);
149
150 propertys.width = mainheader.width;
151 propertys.height = mainheader.height;
152 propertys.nframes = mainheader.totalframes;
153 propertys.framerate = streamheader.rate;
154
155 frames_node = fileRIFF->GetListNode(__L("movi"));
156 if(!frames_node) return false;
157
158 XFILERIFF_LIST* index_list = fileRIFF->GetChunk(__L("idx1"), __L("AVI "));
159 if(!index_list) return false;
160
161 fileRIFF->GetFileBase()->SetPosition(index_list->GetPositionFileData());
162
163 for(XDWORD index=0; index < propertys.nframes; index++)
164 {
165 GRPVIDEOFILEAVI_INDEXENTRY* indexentry = new GRPVIDEOFILEAVI_INDEXENTRY;
166 if(!indexentry) return false;
167
168 if(!fileRIFF->GetFileBase()->Read((XBYTE*)indexentry, sizeof(GRPVIDEOFILEAVI_INDEXENTRY))) return false;
169
170 indexentrys.Add(indexentry);
171 }
172
173 iscreate = false;
174
175 return true;
176}

354XBYTE* GRPVIDEOFILEAVI::GetDataFrame(XDWORD index_frame, XDWORD& sizeframe)
355{
356 if(!frames_node) return NULL;
357 if(index_frame >= propertys.nframes) return NULL;
358
359 XFILERIFF_LIST_NODE* frame_node = frames_node->GetChild(index_frame);
360 if(!frame_node) return NULL;
361
362 XFILERIFF_LIST* frame_list = frame_node->GetData();
363 if(!frame_list) return NULL;
364
365 XDWORD datasize = frame_list->GetSize();
366
367 GRPVIDEOFILEAVI_INDEXENTRY* indexentry = indexentrys.Get(index_frame);
368 if(!indexentry) return NULL;
369
370 if(datasize != indexentry->chunklength) return NULL;
371
372 XBYTE* data = new XBYTE[datasize];
373 if(!data) return NULL;
374
375 if(!fileRIFF->GetFileBase()->SetPosition(frame_list->GetPositionFileData()))
376 {
377 delete [] data;
378 return NULL;
379 }
380
381 if(!fileRIFF->GetFileBase()->Read(data, datasize))
382 {
383 delete [] data;
384 return NULL;
385 }
386
387 sizeframe = datasize;
388
389 return data;
390}

405bool GRPVIDEOFILEAVI::AddFrame(XBYTE* dataframe, XDWORD dataframesize)
406{
407 if(!dataframe) return false;
408 if(!dataframesize) return false;
409
410 if(!movi_node) return false;
411
412
413 XDWORD framesize = dataframesize;
414
415 //framesize += ((framesize % 2)?1:0);
416
417 //--------------------------------------------------------------------
418 // AVI->movi->00dc
419
420 XFILERIFF_LIST_NODE* frame_node = fileRIFF->CreateChunkNode(GRPVIDEOFILEAVI_TYPECHUNKFRAME, framesize);
421 if(!frame_node) return false;
422
423 fileRIFF->UpdateFilePosition(frame_node);
424 if(!fileRIFF->WriteListToFile(frame_node->GetData(), dataframe, framesize)) return false;
425
426 mainheader.totalframes++;
427 movi_node->AddChild(frame_node);
428
429
430 GRPVIDEOFILEAVI_INDEXENTRY* indexentry = new GRPVIDEOFILEAVI_INDEXENTRY();
431 if(indexentry)
432 {
433 indexentry->chunkID = fileRIFF->GetTypeFromString(GRPVIDEOFILEAVI_TYPECHUNKFRAME);
434 indexentry->flags = 16;
435 indexentry->chunklength = framesize;
436 indexentry->chunkoffset = offsetforentrys;
437
438 indexentrys.Add(indexentry);
439
440 //offsetforentrys += (framesize + sizeof(GRPVIDEOFILEAVI_INDEXENTRY));
441
442 offsetforentrys += (framesize + 8);
443 }
444
445 return true;
446}
