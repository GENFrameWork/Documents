325void DIOSTREAMCIPHER::ThreadRunFunction(void* param)
326{
327 DIOSTREAMCIPHER* diostreamcipher = (DIOSTREAMCIPHER*)param;
328 if(!diostreamcipher) return;
329
330 DIOSTREAM* diostream = diostreamcipher->diostream;
331 if(!diostream) return;
332
333 CIPHER* cipher = diostreamcipher->cipher;
334 if(!cipher) return;
335
336 int br = diostream->GetInXBuffer()->GetSize();
337 if(br>DIOSTREAMCIPHER_HEAD_SIZE)
338 {
339 if(diostreamcipher->mutexread) diostreamcipher->mutexread->UnLock();
340
341 XDWORD magicnumber;
342 XBYTE ciphertype;
343 XBYTE cipherpadding;
344 XDWORD sizebuffer;
345 XDWORD CRC32[2];
346 XBUFFER input;
347
348 input.Resize(DIOSTREAMCIPHER_HEAD_SIZE);
349
350 diostream->GetInXBuffer()->Get(input.Get(), DIOSTREAMCIPHER_HEAD_SIZE, 0);
351
352 diostreamcipher->hashcrc32->ResetResult();
353 CRC32[0] = diostreamcipher->hashcrc32->Do(input.Get(), DIOSTREAMCIPHER_HEAD_SIZE - sizeof(XDWORD));
354
355 input.Extract(magicnumber , 0);
356 input.Extract(ciphertype , 0);
357 input.Extract(cipherpadding , 0);
358 input.Extract(sizebuffer , 0);
359 input.Extract(CRC32[1] , 0);
360
361 if((magicnumber == DIOSTREAMCIPHER_HEAD_MAGICNUMBER) && (CRC32[0] == CRC32[1]))
362 {
363 if(diostream->GetInXBuffer()->GetSize() >= sizebuffer + DIOSTREAMCIPHER_HEAD_SIZE)
364 {
365 input.Empty();
366
367 input.Resize(DIOSTREAMCIPHER_HEAD_SIZE + sizebuffer);
368
369 diostream->GetInXBuffer()->Extract(input.Get(), 0, sizebuffer + DIOSTREAMCIPHER_HEAD_SIZE);
370 input.Extract(NULL, 0, DIOSTREAMCIPHER_HEAD_SIZE);
371
372 cipher->Uncipher(input);
373
374 diostreamcipher->GetInXBuffer()->Add(cipher->GetResult()->Get(), cipher->GetResult()->GetSize() - cipherpadding);
375 }
376 }
377
378 if(diostreamcipher->mutexread) diostreamcipher->mutexread->UnLock();
379 }
380
381
382 int bw = diostreamcipher->GetOutXBuffer()->GetSize();
383 if(bw)
384 {
385 if(diostreamcipher->mutexwrite) diostreamcipher->mutexwrite->Lock();
386
387 XBUFFER output;
388
389 cipher->Cipher(diostreamcipher->GetOutXBuffer()->Get(), diostreamcipher->GetOutXBuffer()->GetSize());
390
391 output.Add((XDWORD)DIOSTREAMCIPHER_HEAD_MAGICNUMBER);
392 output.Add((XBYTE)cipher->GetType());
393 output.Add((XBYTE)(cipher->GetResult()->GetSize() - bw));
394 output.Add((XDWORD)cipher->GetResult()->GetSize());
395
396 diostreamcipher->hashcrc32->ResetResult();
397 output.Add((XDWORD)diostreamcipher->hashcrc32->Do(output));
398 output.Add((XBYTE*)cipher->GetResult()->Get(), cipher->GetResult()->GetSize());
399
400 if(diostream->Write(output) == output.GetSize())
401 {
402 //if(diostream->WaitToFlushOutXBuffer())
403 diostreamcipher->GetOutXBuffer()->Extract(NULL, 0 , bw);
404 }
405
406 if(diostreamcipher->mutexwrite) diostreamcipher->mutexwrite->UnLock();
407 }
408}
