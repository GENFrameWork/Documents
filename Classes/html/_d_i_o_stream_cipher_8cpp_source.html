314void DIOSTREAMCIPHER::ThreadRunFunction(void* param)
315{
316 DIOSTREAMCIPHER* diostreamcipher = (DIOSTREAMCIPHER*)param;
317 if(!diostreamcipher) return;
318
319 DIOSTREAM* diostream = diostreamcipher->diostream;
320 if(!diostream) return;
321
322 CIPHER* cipher = diostreamcipher->cipher;
323 if(!cipher) return;
324
325 int br = diostream->GetInXBuffer()->GetSize();
326 if(br>DIOSTREAMCIPHER_HEAD_SIZE)
327 {
328 if(diostreamcipher->mutexread) diostreamcipher->mutexread->UnLock();
329
330 XDWORD magicnumber;
331 XBYTE ciphertype;
332 XBYTE cipherpadding;
333 XDWORD sizebuffer;
334 XDWORD CRC32[2];
335 XBUFFER input;
336
337 input.Resize(DIOSTREAMCIPHER_HEAD_SIZE);
338
339 diostream->GetInXBuffer()->Get(input.Get(), DIOSTREAMCIPHER_HEAD_SIZE, 0);
340
341 diostreamcipher->hashcrc32->ResetResult();
342 CRC32[0] = diostreamcipher->hashcrc32->Do(input.Get(), DIOSTREAMCIPHER_HEAD_SIZE - sizeof(XDWORD));
343
344 input.Extract(magicnumber , 0);
345 input.Extract(ciphertype , 0);
346 input.Extract(cipherpadding , 0);
347 input.Extract(sizebuffer , 0);
348 input.Extract(CRC32[1] , 0);
349
350 if((magicnumber == DIOSTREAMCIPHER_HEAD_MAGICNUMBER) && (CRC32[0] == CRC32[1]))
351 {
352 if(diostream->GetInXBuffer()->GetSize() >= sizebuffer + DIOSTREAMCIPHER_HEAD_SIZE)
353 {
354 input.Empty();
355
356 input.Resize(DIOSTREAMCIPHER_HEAD_SIZE + sizebuffer);
357
358 diostream->GetInXBuffer()->Extract(input.Get(), 0, sizebuffer + DIOSTREAMCIPHER_HEAD_SIZE);
359 input.Extract(NULL, 0, DIOSTREAMCIPHER_HEAD_SIZE);
360
361 cipher->Uncipher(input);
362
363 diostreamcipher->GetInXBuffer()->Add(cipher->GetResult()->Get(), cipher->GetResult()->GetSize() - cipherpadding);
364 }
365 }
366
367 if(diostreamcipher->mutexread) diostreamcipher->mutexread->UnLock();
368 }
369
370
371 int bw = diostreamcipher->GetOutXBuffer()->GetSize();
372 if(bw)
373 {
374 if(diostreamcipher->mutexwrite) diostreamcipher->mutexwrite->Lock();
375
376 XBUFFER output;
377
378 cipher->Cipher(diostreamcipher->GetOutXBuffer()->Get(), diostreamcipher->GetOutXBuffer()->GetSize());
379
380 output.Add((XDWORD)DIOSTREAMCIPHER_HEAD_MAGICNUMBER);
381 output.Add((XBYTE)cipher->GetType());
382 output.Add((XBYTE)(cipher->GetResult()->GetSize() - bw));
383 output.Add((XDWORD)cipher->GetResult()->GetSize());
384
385 diostreamcipher->hashcrc32->ResetResult();
386 output.Add((XDWORD)diostreamcipher->hashcrc32->Do(output));
387 output.Add((XBYTE*)cipher->GetResult()->Get(), cipher->GetResult()->GetSize());
388
389 if(diostream->Write(output) == output.GetSize())
390 {
391 //if(diostream->WaitToFlushOutXBuffer())
392 diostreamcipher->GetOutXBuffer()->Extract(NULL, 0 , bw);
393 }
394
395 if(diostreamcipher->mutexwrite) diostreamcipher->mutexwrite->UnLock();
396 }
397}
