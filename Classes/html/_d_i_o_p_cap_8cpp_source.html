688bool DIOPCAPFRAME::GetHeader(DIOPCAPUDPHEADER& header)
689{
690 DIOPCAPIPHEADER ipheader;
691 if(!GetHeader(ipheader))
692 {
693 return false;
694 }
695
696 int iplen = (ipheader.ver_ihl & 0xf) * 4;
697
698 UDP_header = (DIOPCAPUDPHEADER*)((XBYTE*)data->Get() + sizeof(DIOPCAPETHERNETHEADER) + iplen);
699 if(!UDP_header)
700 {
701 return false;
702 }
703
704 memcpy((XBYTE*)&header,(XBYTE*)(UDP_header),sizeof(DIOPCAPUDPHEADER));
705
706 if(hardwareuselittleendian)
707 {
708 SWAPWORD(header.sourceport); // Source port
709 SWAPWORD(header.targetport); // Target port
710 SWAPWORD(header.datagramlen); // Datagram length
711 SWAPWORD(header.checksum); // Checksum
712 }
