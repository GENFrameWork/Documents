697bool DIOPCAPFRAME::GetHeader(DIOPCAPUDPHEADER& header)
698{
699 DIOPCAPIPHEADER ipheader;
700 if(!GetHeader(ipheader))
701 {
702 return false;
703 }
704
705 int iplen = (ipheader.ver_ihl & 0xf) * 4;
706
707 UDP_header = (DIOPCAPUDPHEADER*)((XBYTE*)data->Get() + sizeof(DIOPCAPETHERNETHEADER) + iplen);
708 if(!UDP_header)
709 {
710 return false;
711 }
712
713 memcpy((XBYTE*)&header,(XBYTE*)(UDP_header),sizeof(DIOPCAPUDPHEADER));
714
715 if(hardwareuselittleendian)
716 {
717 SWAPWORD(header.sourceport); // Source port
718 SWAPWORD(header.targetport); // Target port
719 SWAPWORD(header.datagramlen); // Datagram length
720 SWAPWORD(header.checksum); // Checksum
721 }
