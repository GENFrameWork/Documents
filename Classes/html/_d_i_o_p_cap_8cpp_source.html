690bool DIOPCAPFRAME::GetHeader(DIOPCAPUDPHEADER& header)
691{
692 DIOPCAPIPHEADER ipheader;
693 if(!GetHeader(ipheader))
694 {
695 return false;
696 }
697
698 int iplen = (ipheader.ver_ihl & 0xf) * 4;
699
700 UDP_header = (DIOPCAPUDPHEADER*)((XBYTE*)data->Get() + sizeof(DIOPCAPETHERNETHEADER) + iplen);
701 if(!UDP_header)
702 {
703 return false;
704 }
705
706 memcpy((XBYTE*)&header,(XBYTE*)(UDP_header),sizeof(DIOPCAPUDPHEADER));
707
708 if(hardwareuselittleendian)
709 {
710 SWAPWORD(header.sourceport); // Source port
711 SWAPWORD(header.targetport); // Target port
712 SWAPWORD(header.datagramlen); // Datagram length
713 SWAPWORD(header.checksum); // Checksum
714 }
715
716 return true;

1163bool DIOPCAP::Frames_Add(XBYTE* data,XDWORD size, bool isloopback)
1164{
1165 if(!data) return false;
1166 if(!size) return false;
1167
1168 bool status = false;
1169
1170 if(xmutexframes) xmutexframes->Lock();
1171
1172 XDWORD allheaderssize = 0;
1173
1174 DIOPCAPFRAME* frame = new DIOPCAPFRAME(GEN_XSYSTEM.HardwareUseLittleEndian(), isloopback);
1175 if(frame)
1176 {
1177 if(frame->SetData(data,size))
1178 {
1179 XBYTE* framedata = frame->GetData()->Get();
1180 if(framedata)
1181 {
1182 if(!isloopback)
1183 {
1184 DIOPCAPETHERNETHEADER* ethernet_header = (DIOPCAPETHERNETHEADER*)framedata;
1185 if(ethernet_header)
1186 {
1187 if(ethernet_header->type != DIOPCAPETHERNETTYPE_IP)
1188 {
1189 frame->GetSourceMAC()->Set(ethernet_header->MACsource);
1190 frame->GetTargetMAC()->Set(ethernet_header->MACtarget);
1191
1192 frame->Set((DIOPCAPETHERNETHEADER*)framedata);
1193
1194 allheaderssize = sizeof(DIOPCAPETHERNETHEADER);
1195 framedata += sizeof(DIOPCAPETHERNETHEADER);
1196 }
1197 }
1198 }
1199 else
1200 {
1201 allheaderssize = sizeof(XDWORD);
1202 framedata += sizeof(XDWORD);
1203 }
1204
1205 DIOPCAPIPHEADER* IP_header = (DIOPCAPIPHEADER*)framedata;
1206 if(IP_header)
1207 {
1208 frame->GetSourceIP()->Set(IP_header->sourceaddr);
1209 frame->GetTargetIP()->Set(IP_header->targetaddr);
1210
1211 frame->Set((DIOPCAPIPHEADER*)framedata);
1212
1213 allheaderssize += sizeof(DIOPCAPIPHEADER);
1214 framedata += sizeof(DIOPCAPIPHEADER);
1215
1216 if(framedata)
1217 {
1218 switch(IP_header->protocol)
1219 {
1220 case DIOPCAPIPPROTOCOLTYPE_ICMP : frame->SetProtocolType(DIOPCAPPROTOCOL_TYPE_ICMP);
1221 break;
1222
1223 case DIOPCAPIPPROTOCOLTYPE_UDP : { DIOPCAPUDPHEADER* UDPheader = (DIOPCAPUDPHEADER*)framedata;
1224 if(UDPheader)
1225 {
1226 frame->SetProtocolType(DIOPCAPPROTOCOL_TYPE_UDP);
1227
1228 frame->SetSourcePort(SwapWORD(UDPheader->sourceport));
1229 frame->SetTargetPort(SwapWORD(UDPheader->targetport));
1230
1231 frame->Set((DIOPCAPUDPHEADER*)framedata);
1232
1233 allheaderssize += sizeof(DIOPCAPUDPHEADER);
1234 framedata += sizeof(DIOPCAPUDPHEADER);
1235 }
1236 }
1237 break;
1238
1239 case DIOPCAPIPPROTOCOLTYPE_TCP : { DIOPCAPTCPHEADER* TCPheader = (DIOPCAPTCPHEADER*)framedata;
1240 if(TCPheader)
1241 {
1242 XWORD TCPheadersize = (SwapWORD(TCPheader->hlenflags) >> 12) *4;
1243
1244 frame->SetProtocolType(DIOPCAPPROTOCOL_TYPE_TCP);
1245
1246 frame->SetSourcePort(SwapWORD(TCPheader->sourceport));
1247 frame->SetTargetPort(SwapWORD(TCPheader->targetport));
1248
1249 frame->Set((DIOPCAPTCPHEADER*)framedata);
1250
1251 allheaderssize += TCPheadersize;
1252 framedata += TCPheadersize;
1253 }
1254 }
1255 break;
1256 }
1257 }
1258 }
1259
1260
1261 if(frame->GetData()->GetSize() >= allheaderssize)
1262 {
1263 frame->SetAllHeadersSize(allheaderssize);
1264
1265 XDWORD sizepayload = (XDWORD)(frame->GetData()->GetSize() - allheaderssize);
1266 if(sizepayload)
1267 {
1268 frame->Set(framedata, sizepayload);
1269 }
1270
1271 frames.Add(frame);
1272 status = true;
1273 }
1274 else
1275 {
1276 delete frame;
1277 frame = NULL;
1278 }
1279 }
1280 }
1281 }
1282
1283 if(xmutexframes) xmutexframes->UnLock();
1284
1285 return status;
1286}
