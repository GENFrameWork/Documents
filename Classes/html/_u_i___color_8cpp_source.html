309bool UI_COLOR::SetFromString(XCHAR* string)
310{
311 SetInvalid();
312
313 if(!string)
314 {
315 return false;
316 }
317
318 XSTRING colorstr;
319 int ncommas = 0;
320 int _alpha = 0;
321 bool status = false;
322
323 colorstr = string;
324 if(colorstr.IsEmpty())
325 {
326 return false;
327 }
328
329 for(XDWORD c=0; c<colorstr.GetSize(); c++)
330 {
331 XCHAR character = colorstr.Get()[c];
332 if(character == __C(','))
333 {
334 ncommas++;
335 }
336 }
337
338 switch(ncommas)
339 {
340 case 0 : // only name
341 { XSTRING* resolved_colorstr = GEN_UI_COLORS.Get(colorstr);
342 if(resolved_colorstr)
343 {
344 status = SetFromString(resolved_colorstr->Get());
345 }
346 }
347 break;
348
349 case 1 : // name + int alpha
350 { XSTRING _colorstr;
351 XSTRING* resolved_colorstr = NULL;
352
353 _colorstr.AdjustSize(_MAXSTR);
354 colorstr.UnFormat(__L("%s,%d"), _colorstr.Get(), &_alpha);
355 _colorstr.AdjustSize();
356
357 resolved_colorstr = GEN_UI_COLORS.Get(_colorstr);
358 if(resolved_colorstr)
359 {
360 status = SetFromString(resolved_colorstr->Get());
361 }
362
363 alpha = GetAlphaForPercent(_alpha);
364 }
365 break;
366
367 case 2 : // int colors
368 status = colorstr.UnFormat(__L("%d,%d,%d"), &red, &green, &blue);
369 alpha = 255;
370 break;
371
372 case 3 : // int colors + int alpha
373 status = colorstr.UnFormat(__L("%d,%d,%d,%d"), &red, &green, &blue, &_alpha);
374 alpha = GetAlphaForPercent(_alpha);
375 break;
376
377 default : break;
378 }
379
380 if(!status)
381 {
382 SetInvalid();
383 }
384
385 valid = true;
386
387 return status;
388}
