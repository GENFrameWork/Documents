160bool XFILECOBOL_PIC::ResolveMask()
161{
162 if(mask.IsEmpty())
163 {
164 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC Missing"));
165 return false;
166 }
167
168 hassign = false;
169 ndecimals = 0;
170 sizedata = 0;
171
172 XDWORD index = 0;
173
174 while(index < mask.GetSize())
175 {
176 switch(mask.Get()[index])
177 {
178 case __C('X') : type = XFILECOBOL_PIC_TYPE_BINARY;
179 maskdecondensed += __C('X');
180 sizedata++;
181 index++;
182 break;
183
184 case __C('A') : type = XFILECOBOL_PIC_TYPE_ALPHA;
185 maskdecondensed += __C('A');
186 sizedata++;
187 index++;
188 break;
189
190 case __C('9') : if(type != XFILECOBOL_PIC_TYPE_DECIMAL) type = XFILECOBOL_PIC_TYPE_NUMERIC;
191 if(type == XFILECOBOL_PIC_TYPE_DECIMAL) ndecimals++;
192 maskdecondensed += __C('9');
193 sizedata++;
194 index++;
195 break;
196
197 case __C(',') : type = XFILECOBOL_PIC_TYPE_DECIMAL;
198 maskdecondensed += __C(',');
199 sizedata++;
200 index++;
201 break;
202
203 case __C('S') : hassign = true;
204 maskdecondensed += __C('S');
205 index++;
206 break;
207
208 case __C('(') : { XSTRING modificator;
209 int sizemodificator;
210
211 int end = mask.Find(__L(")"), true, index+1);
212 if(end == XSTRING_NOTFOUND)
213 {
214 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Missing ')'"), mask.Get());
215 return false;
216 }
217
218 mask.Copy(index, end+1, modificator);
219 modificator.UnFormat(__L("(%d)"), &sizemodificator);
220
221 if(!sizemodificator)
222 {
223 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Modificador is zero '(0)'"), mask.Get());
224 return false;
225 }
226
227 for(int c=0; c<sizemodificator-1; c++)
228 {
229 maskdecondensed += mask.Get()[index-1];
230 sizedata++;
231 if(type == XFILECOBOL_PIC_TYPE_DECIMAL) ndecimals++;
232 }
233
234 index = end;
235 }
236 break;
237
238 case __C(')') : index++;
239 break;
240
241 case __C('V') : type = XFILECOBOL_PIC_TYPE_DECIMAL;
242 maskdecondensed += __C('V');
243 index++;
244 break;
245
246 default : XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Unknown Specifier '%c'"), mask.Get(), mask.Get()[index]);
247 return false;
248 break;
249 }
250 }
251
252 switch(computationaltype)
253 {
254 case 1 : // Float point 4 Bytes
255 break;
256
257 case 2 : // Float Point 8 Bytes
258 break;
259
260 case 3 : // BCD + Sing
261 sizedata = ((sizedata + 1) / 2) + (((sizedata + 1) % 2)?1:0);
262 break;
263
264 case 4 : break;
265
266 case 5 : // Conversion Packet number standard binary data
267 if((sizedata >= 1) && (sizedata <= 2))
268 {
269 sizedata = 1;
270 }
271 else
272 {
273 if((sizedata >=3) && (sizedata <= 4))
274 {
275 sizedata = 2;
276 }
277 else
278 {
279 if((sizedata >= 5) && (sizedata <= 9))
280 {
281 sizedata = 4;
282 }
283 else
284 {
285 if((sizedata >= 10) && (sizedata <= 18))
286 {
287 sizedata = 8;
288 }
289 else
290 {
291 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Size of modificator to long %s > 18 "), mask.Get(), sizedata);
292 return false;
293 }
294 }
295 }
296 }
297 break;
298 }
299
300 return true;
301}

1118 if(!picture) return false;
1119 if(!picture->GetMask()) return false;
1120 if(!bufferdata.GetSize()) return false;
1121
1122 XSTRING mask;
1123
1124 mask = picture->GetMaskDecondensed()->Get();
1125 if(!mask.GetSize()) return false;
1126
1127 int indexdata = 0;
1128
1129 switch(picture->GetComputationalType())
1130 {
1131 case 0 : for(XDWORD c=0; c<mask.GetSize(); c++)
1132 {
1133 switch(mask.Get()[c])
1134 {
1135 case __C('S') : { XBYTE lastdata = bufferdata.GetLastByte();
1136 if(lastdata & XFILECOBOL_MASK_SIGN)

1143 }
