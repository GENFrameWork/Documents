204bool DIOLIGHTSENSORLDRANALOG::ReadDirect(XDWORD& level)
205{
206 if(!diogpio) return false;
207 if(!xtimer) return false;
208 if(!IsInitialized()) return false;
209
210 level = 0;
211
212 xtimer->Reset();
213
214 // Discharge capacitor
215 diogpio->SetMode(pindata, false);
216 diogpio->Set(pindata , false);
217 GEN_XSLEEP.MilliSeconds(100);
218
219 diogpio->SetMode(pindata, true);
220 while(!diogpio->Get(pindata))
221 {
222 level++;
223 if(!(level%100))
224 {
225 if(xtimer->GetMeasureSeconds()>=1) return false;
226 }
227 }
228
229 if(!level) return false;
230
231 return true;
232}

266 if(!sensor) return;
