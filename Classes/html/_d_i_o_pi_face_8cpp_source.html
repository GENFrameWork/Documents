107bool DIOPIFACE::Configure()
108{
109 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IOCON , 8); // Enable hardware addressing
110 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_GPIOA , 0x00); // Turn on port A
111 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IODIRA , 0); // Set port A as an output
112 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IODIRB , 0xFF); // Set port B as an input
113 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_GPPUB , 0xFF); // Turn on port B pullups
114
115 return true;
116}

130bool DIOPIFACE::ReadInputs(XBYTE& value)
131{
132 XBYTE _value[3] = { 0xFF, 0xFF, 0xFF };
133
134 if(!diostream) return false;
135
136 diostream->GetInXBuffer()->Delete();
137
138 if(!Write_Register(DIOSPIGPIOMCP23S17_READ_CMD, DIOPIFACE_INPUTPORT, 0xFF)) return false;
139
140 if(!diostream->WaitToFilledReadingBuffer(3, timeout)) return false;
141
142 bool status = diostream->Read(_value, 3)?true:false;
143
144 value = _value[2] ^ 0xFF;
145
146 return status;
147}
