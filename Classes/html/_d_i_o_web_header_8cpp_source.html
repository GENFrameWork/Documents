113bool DIOWEBHEADER::Read(DIOSTREAMTCPIP* diostream, int timeout)
114{
115 if(!diostream) return false;
116
117 XSTRING line;
118 bool status;
119
120 DeleteAllLines();
121
122 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer())
123 if(!xtimerout) return false;
124
125 do{ status = diostream->ReadStr(line);
126 if(status)
127 {
128 if(line.IsEmpty())
129 {
130 break;
131 }
132 else
133 {
134 //XTRACE_PRINTCOLOR(3, __L("WEB CLIENT HEADER LINE: %s"), line.Get());
135 AddLine(line);
136 }
137 }
138
139 if(timeout)
140 {
141 if(xtimerout->GetMeasureSeconds()>(XDWORD)timeout) break;
142
143 GEN_XSLEEP.MilliSeconds(1);
144 }
145
146 } while((diostream->GetStatus() == DIOSTREAMSTATUS_CONNECTED) || diostream->GetInXBuffer()->GetSize());
147
148 GEN_XFACTORY.DeleteTimer(xtimerout);
149
150
151 if(lines.IsEmpty()) return false;
152
153 return true;
154}

366bool DIOWEBHEADER::GetFormat(bool isserver, int type, void* data)
367{
368 if(lines.IsEmpty()) return false;
369
370 if(!data) return false;
371
372 XSTRING line;
373 int version = 0;
374 int subversion = 0;
375
376 line =((XSTRING*)lines.Get(0))->Get();
377
378 if(isserver)
379 {
380 int code;
381 XSTRING codestring(DIOWEBHEADER_MAXLINE);
382
383 line.UnFormat(__L("HTTP/%d.%d %d %s"),&version,&subversion,&code,codestring.Get());
384 codestring.AdjustSize();
385
386 switch(type)
387 {
388 case 0: (*(int*)data) = version; break;
389 case 1: (*(int*)data) = subversion; break;
390 case 2: (*(int*)data) = code; break;
391 case 3: (*(XSTRING*)data) = codestring; break;
392 }
393 }
394 else
395 {
396 XSTRING method(DIOWEBHEADER_MAXLINE);
397 XSTRING resource(DIOWEBHEADER_MAXLINE);
398
399 if(line.GetSize())
400 {
401 int foundindex = line.Find(__L("HTTP/"), true);
402
403 if(foundindex != XSTRING_NOTFOUND) line.UnFormat(__L("%s %s HTTP/%d.%d"), method.Get(), resource.Get(), &version, &subversion);
404 }
405
406 method.AdjustSize();
407 resource.AdjustSize();
408
409 switch(type)
410 {
411 case 0: if(data) (*(XSTRING*)data) = method; break;
412 case 1: if(data) (*(XSTRING*)data) = resource; break;
413 case 2: if(data) (*(int*)data) = version; break;
414 case 3: if(data) (*(int*)data) = subversion; break;
415 }
416 }
417
418 return true;
419}
