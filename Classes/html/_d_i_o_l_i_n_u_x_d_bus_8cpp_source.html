577{
578 DBusMessageIter iter_variant;
579 bool status = false;
580
581 if(in) return false;

583
584 if(dbus_message_iter_get_arg_type(&iter) != DBUS_TYPE_VARIANT) return false;
585
586 dbus_message_iter_recurse(&iter, &iter_variant);
587
588 switch(dbus_message_iter_get_arg_type(&iter_variant))
589 {
590 case DBUS_TYPE_UINT32 : { XDWORD value = 0;
591 dbus_message_iter_get_basic(&iter_variant, &value);
592 result = value;
593 }
594 break;
595
596 case DBUS_TYPE_STRING :
597 case DBUS_TYPE_OBJECT_PATH : { char* value_str_ptr = NULL;
598 dbus_message_iter_get_basic(&iter_variant, &value_str_ptr);
599 if(value_str_ptr)
600 {
601 result = value_str_ptr;
602 status = true;
603 }
604 }
605 break;
606
607
608 }

1032void DIOLINUXDBUS::Thread_Signals(void* param)
1033{
1034 DIOLINUXDBUS* dbus = (DIOLINUXDBUS*)param;
1035 if(!dbus) return;
1036
1037 if(!dbus->connection) return;
1038 if(dbus->signals.IsEmpty()) return;
1039
1040 DBusMessage* handler_message = NULL;
1041
1042 dbus_connection_read_write(dbus->connection, 0);
1043 handler_message = dbus_connection_pop_message(dbus->connection);
1044
1045 if(!handler_message)
1046 {
1047 GEN_XSLEEP.Seconds(1);
1048 return;
1049 }
1050
1051 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DBUS] Message ..."));
1052
1053 DIOLINUXDBUS_SIGNAL* signal = NULL;
1054 bool found_signal = false;
1055
1056 for(XDWORD c=0; c<dbus->signals.GetSize(); c++)
1057 {
1058 signal = dbus->signals.Get(c);
1059 if(signal)
1060 {
1061 // ---------------------------------------------------------------------------------------
1062 // check if the message is a signal from the correct interface and with the correct name
1063
1064 XBUFFER typechar;
1065 XBUFFER namechar;
1066
1067 (*signal->GetType()).ConvertToASCII(typechar);
1068 (*signal->GetName()).ConvertToASCII(namechar);
1069
1070 if(dbus_message_is_signal(handler_message, typechar.GetPtrChar(), namechar.GetPtrChar()))
1071 {
1072 found_signal = true;
1073 }
1074
1075 if(found_signal)
1076 {
1077 break;
1078 }
1079 else
1080 {
1081 signal = NULL;
1082 }
1083 }
1084 }
1085
1086
1087 if(found_signal && signal)
1088 {
1089 DIOLINUXDBUS_MESSAGE message;
1090
1091 message.SetHandler(handler_message);
1092
1093 DIOLINUXDBUS_XEVENT xevent(dbus, DIOLINUXDBUS_XEVENT_TYPE_SIGNAL);
1094
1095 xevent.SetMessage(&message);
1096 xevent.SetSignal(signal);
1097
1098 dbus->PostEvent(&xevent);
1099
1100 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DBUS] Signal produced [%s] [%s] ..."), signal->GetType()->Get(), signal->GetName()->Get());
1101 }
1102 else
1103 {
1104 dbus_message_unref(handler_message);
1105 }
1106}
