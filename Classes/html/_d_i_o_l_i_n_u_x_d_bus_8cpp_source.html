587
588 if(in) return false;

592
593 dbus_message_iter_recurse(&iter, &iter_variant);
594
595 switch(dbus_message_iter_get_arg_type(&iter_variant))

598 dbus_message_iter_get_basic(&iter_variant, &value);

606 if(value_str_ptr)

608 result = value_str_ptr;
609 status = true;

1039void DIOLINUXDBUS::Thread_Signals(void* param)
1040{
1041 DIOLINUXDBUS* dbus = (DIOLINUXDBUS*)param;
1042 if(!dbus) return;
1043
1044 if(!dbus->connection) return;
1045 if(dbus->signals.IsEmpty()) return;
1046
1047 DBusMessage* handler_message = NULL;
1048
1049 dbus_connection_read_write(dbus->connection, 0);
1050 handler_message = dbus_connection_pop_message(dbus->connection);
1051
1052 if(!handler_message)
1053 {
1054 GEN_XSLEEP.Seconds(1);
1055 return;
1056 }
1057
1058 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DBUS] Message ..."));
1059
1060 DIOLINUXDBUS_SIGNAL* signal = NULL;
1061 bool found_signal = false;
1062
1063 for(XDWORD c=0; c<dbus->signals.GetSize(); c++)
1064 {
1065 signal = dbus->signals.Get(c);
1066 if(signal)
1067 {
1068 // ---------------------------------------------------------------------------------------
1069 // check if the message is a signal from the correct interface and with the correct name
1070
1071 XBUFFER typechar;
1072 XBUFFER namechar;
1073
1074 (*signal->GetType()).ConvertToASCII(typechar);
1075 (*signal->GetName()).ConvertToASCII(namechar);
1076
1077 if(dbus_message_is_signal(handler_message, typechar.GetPtrChar(), namechar.GetPtrChar()))
1078 {
1079 found_signal = true;
1080 }
1081
1082 if(found_signal)
1083 {
1084 break;
1085 }
1086 else
1087 {
1088 signal = NULL;
1089 }
1090 }
1091 }
1092
1093
1094 if(found_signal && signal)
1095 {
1096 DIOLINUXDBUS_MESSAGE message;
1097
1098 message.SetHandler(handler_message);
1099
1100 DIOLINUXDBUS_XEVENT xevent(dbus, DIOLINUXDBUS_XEVENT_TYPE_SIGNAL);
1101
1102 xevent.SetMessage(&message);
1103 xevent.SetSignal(signal);
1104
1105 dbus->PostEvent(&xevent);
1106
1107 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DBUS] Signal produced [%s] [%s] ..."), signal->GetType()->Get(), signal->GetName()->Get());
1108 }
1109 else
1110 {
1111 dbus_message_unref(handler_message);
1112 }
1113}
