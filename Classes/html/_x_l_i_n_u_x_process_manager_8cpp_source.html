487{
488 DIR* dir;
489 struct dirent* ent;
490 char* endptr;
491 char buf[512];
492
493 if(!(dir = opendir("/proc"))) return -1;
494
495 while((ent = readdir(dir)) != NULL)
496 {
497 // if endptr is not a null character, the directory is not entirely numeric, so ignore it
498 long lpid = strtol(ent->d_name, &endptr, 10);
499 if(*endptr != '\0')
500 {
501 continue;
502 }
503
504 // try to open the cmdline file
505 snprintf(buf, sizeof(buf), "/proc/%ld/status", lpid);
506 FILE* fp = fopen(buf, "r");
507 if(fp)
508 {
509 if(fgets(buf, sizeof(buf), fp) != NULL)
510 {
511 // check the first token in the file, the program name
512 char* first = strtok(buf, " ");
513
514 XPATH nameall;
515
516 nameall = first;
517 nameall.SetOnlyNamefile();
518 nameall.DeleteCharacter(__C(' ') , XSTRINGCONTEXT_FROM_FIRST);
519 nameall.DeleteCharacter(__C('\t'), XSTRINGCONTEXT_FROM_FIRST);
520 nameall.DeleteCharacter(__C('\n'), XSTRINGCONTEXT_TO_END);
521 nameall.DeleteCharacter(__C('\r'), XSTRINGCONTEXT_TO_END);
522
523 if(!nameall.Compare(applicationname, false))
524 {
525 if(ID) (*ID) = (XDWORD)lpid;
526
527 fclose(fp);
528 closedir(dir);
529
530 return true;
