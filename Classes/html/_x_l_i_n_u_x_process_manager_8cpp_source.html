484bool XLINUXPROCESSMANAGER::Application_IsRunning(XCHAR* applicationname, XDWORD* ID)
485{
486 DIR* dir;
487 struct dirent* ent;
488 char* endptr;
489 char buf[512];
490
491 if(!(dir = opendir("/proc"))) return -1;
492
493 while((ent = readdir(dir)) != NULL)
494 {
495 // if endptr is not a null character, the directory is not entirely numeric, so ignore it
496 long lpid = strtol(ent->d_name, &endptr, 10);
497 if(*endptr != '\0')
498 {
499 continue;
500 }
501
502 // try to open the cmdline file
503 snprintf(buf, sizeof(buf), "/proc/%ld/status", lpid);
504 FILE* fp = fopen(buf, "r");
505 if(fp)
506 {
507 if(fgets(buf, sizeof(buf), fp) != NULL)
508 {
509 // check the first token in the file, the program name
510 char* first = strtok(buf, " ");
511
512 XPATH nameall;
513
514 nameall = first;
515 nameall.SetOnlyNamefile();
516 nameall.DeleteCharacter(__C(' ') , XSTRINGCONTEXT_FROM_FIRST);
517 nameall.DeleteCharacter(__C('\t'), XSTRINGCONTEXT_FROM_FIRST);
518 nameall.DeleteCharacter(__C('\n'), XSTRINGCONTEXT_TO_END);
519 nameall.DeleteCharacter(__C('\r'), XSTRINGCONTEXT_TO_END);
520
521 if(!nameall.Compare(applicationname, false))
522 {
523 if(ID) (*ID) = (XDWORD)lpid;
524
525 fclose(fp);
526 closedir(dir);
527
528 return true;
529
530 }
531 }
532
533 fclose(fp);
534 }
535 }
536
537 closedir(dir);

539 return false;
