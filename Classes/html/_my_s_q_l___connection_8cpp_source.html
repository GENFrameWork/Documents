107bool MYSQL_CONNECTION::Connect()
108{
109 connection = mysql_init(NULL);
110 if(!connection)
111 {
112 XSTRING error;
113 error.Set(mysql_error(connection));
114
115 database->Error(error.Get());
116
117 return false;

120 DB_SQL_STRING databaseurl;
121 DB_SQL_STRING databaseuser;
122 DB_SQL_STRING databasepassword;
123 DB_SQL_STRING databasename;
124 DB_SQL_STRING databaseport;
125 DB_SQL_STRING databasetimeout;
126 DB_SQL_STRING reconnect;
127
128 if(!FindOption(__L("URL"), &databaseurl))
129 {
130 database->Error(__L("DATABASE URL argument not supplied"));
131 return false;
132 }
133
134 if(!FindOption(__L("USER"), &databaseuser))

136 database->Error(__L("DATABASE User Name argument not supplied"));

142 database->Error(__L("DATABASE User Name argument not supplied"));
143 return false;

145
146 FindOption(__L("DATABASE") , &databasename);
147 FindOption(__L("PORT") , &databaseport);
148
149 XBUFFER url;
150 XBUFFER user;
151 XBUFFER pswd;
152 XBUFFER name;
153
154 databaseurl.ConvertToASCII(url);
155 databaseuser.ConvertToASCII(user);
156 databasepassword.ConvertToASCII(pswd);
157 databasename.ConvertToASCII(name);
158
159 unsigned int port = databaseport.ConvertToDWord();
160
161 bool success=true;
162
163 if (FindOption(__L("RECONNECT"), &reconnect))
164 {
165 int reconnect_opt = reconnect.ConvertToInt();
166 mysql_options(connection, MYSQL_OPT_RECONNECT, &reconnect_opt);
167 }
168 else
169 {
170 int reconnect_opt = 1;
171 mysql_options(connection, MYSQL_OPT_RECONNECT, &reconnect_opt);
172 }
173
174 if (FindOption(__L("TIMEOUT"), &databasetimeout))
175 {
176 int timeoutseconds=databasetimeout.ConvertToInt();
177 mysql_options(connection,MYSQL_OPT_CONNECT_TIMEOUT ,&timeoutseconds);
178 }
179
180 if(mysql_real_connect(connection, url.GetPtrChar(), user.GetPtrChar(), pswd.GetPtrChar(), name.GetPtrChar(), port, NULL, 0) == NULL)
181 {
182 DB_SQL_STRING error;
183
184 error.Set(mysql_error(connection));
185 database->Error(error.Get());
186 mysql_close(connection);
187
188 success = false;
189 }
190
191 return success;
192}
