161void Call_System_GetType(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
162{
163 if(!library) return;
164 if(!script) return;
165 if(!params) return;
166 if(!returnvalue) return;
167
168 returnvalue->Set();
169
170 switch(GEN_XSYSTEM.GetPlatform())
171 {
172 case XSYSTEM_PLATFORM_UNKNOWN :
173 default : (*returnvalue) = ""; break;
174 case XSYSTEM_PLATFORM_WINDOWS : (*returnvalue) = "Windows"; break;
175 case XSYSTEM_PLATFORM_LINUX : (*returnvalue) = "Linux"; break;
176 case XSYSTEM_PLATFORM_LINUX_EMBEDDED : (*returnvalue) = "LinuxEmbedded"; break;
177 case XSYSTEM_PLATFORM_ANDROID : (*returnvalue) = "Android"; break;
178 case XSYSTEM_PLATFORM_STM32 : (*returnvalue) = "STM32"; break;
179 };
180}

270void Call_System_GetEnviromentVar(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
271{
272 if(!library) return;
273 if(!script) return;
274 if(!params) return;
275 if(!returnvalue) return;
276
277 returnvalue->Set();
278
279 if(!params->GetSize())
280 {
281 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
282 return;
283 }
284
285 XSTRING* namevar = (XSTRING*)params->Get(0)->GetData();
286 if(!namevar) return;
287
288 XCHAR* variable = GEN_XSYSTEM.GetEnviromentVariable(namevar->Get());
289 if(variable)
290 {
291 (*returnvalue) = variable;
292 }
293}
