214bool XFSMACHINESTATE::DeleteTransition(int outputID)
215{
216 int c;
217
218 for(c=0;c<maxtransitions;c++)
219 {
220 if(outputstates[c]==outputID) break;
221 }
222
223 if(c>=maxtransitions) return false;

424bool XFSMACHINE::AddState(int state, int event, int tostate,...)
425{
426 XFSMACHINESTATE* _state;
427 int narg = 0;
428 va_list arg;
429
430 va_start(arg, tostate);
431 while(va_arg(arg, int) != XFSMACHINESTATE_EVENTDEFEND)
432 {
433 narg++;
434 }
435 va_end(arg);
436
437 if((narg)&&(narg%2)) return false;
438
439 narg+=2;
440
441 _state = new XFSMACHINESTATE(state,(narg/2));
442 if(!_state) return false;
443
444 _state->AddTransition(event,tostate);
445
446 va_start(arg,tostate);
447
448 int _event;
449 int _tostate;
450
451 do { _event = va_arg(arg, int);
452 if(_event != XFSMACHINESTATE_EVENTDEFEND)
453 {
454 _tostate = va_arg(arg, int);
455 _state->AddTransition(_event,_tostate);
456 }
457
458 } while(_event != XFSMACHINESTATE_EVENTDEFEND);
459
460 va_end(arg);
461
462 if(!AddState(_state))
463 {
464 delete _state;
465 return false;
466 }
467
468 return true;
