207bool XFSMACHINESTATE::DeleteTransition(int outputID)
208{
209 int c;
210
211 for(c=0;c<maxtransitions;c++)
212 {
213 if(outputstates[c]==outputID) break;
214 }
215
216 if(c>=maxtransitions) return false;

417bool XFSMACHINE::AddState(int state, int event, int tostate,...)
418{
419 XFSMACHINESTATE* _state;
420 int narg = 0;
421 va_list arg;
422
423 va_start(arg, tostate);
424 while(va_arg(arg, int) != XFSMACHINESTATE_EVENTDEFEND)
425 {
426 narg++;
427 }
428 va_end(arg);
429
430 if((narg)&&(narg%2)) return false;
431
432 narg+=2;
433
434 _state = new XFSMACHINESTATE(state,(narg/2));
435 if(!_state) return false;
436
437 _state->AddTransition(event,tostate);
438
439 va_start(arg,tostate);
440
441 int _event;
442 int _tostate;
443
444 do { _event = va_arg(arg, int);
445 if(_event != XFSMACHINESTATE_EVENTDEFEND)
446 {
447 _tostate = va_arg(arg, int);
448 _state->AddTransition(_event,_tostate);
449 }

451 } while(_event != XFSMACHINESTATE_EVENTDEFEND);
452
453 va_end(arg);

455 if(!AddState(_state))
456 {
457 delete _state;
