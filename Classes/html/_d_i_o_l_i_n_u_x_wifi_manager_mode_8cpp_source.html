557bool DIOLINUXWIFIMANAGERMODE::WPA_GetListRemoteAP(XVECTOR<XSTRING*>& nameAPs)
558{
559 XSTRING prefixcommand;
560 XSTRING command;
561 XSTRING output;
562 bool status;
563
564
565 nameAPs.DeleteContents();
566 nameAPs.DeleteAll();
567
568 prefixcommand.Format(DIOLINUXWIFIMANAGERMODE_PREFIXCOMMAND, netinterface.Get());
569
570 command.Format(__L("%s scan") , prefixcommand.Get());
571 status = WPA_CLI_Command(command, output);
572 if(!status) return false;
573
574 GEN_XSLEEP.Seconds(3);
575
576 command.Format(__L("%s scan_results") , prefixcommand.Get());
577 status = WPA_CLI_Command(command, output);
578
579 int indexfr = output.Find(__L("\n"), true);
580 if(indexfr == XSTRING_NOTFOUND) return false;
581
582 indexfr++;
583
584 int index = 0;
585 int indexstart = 0;
586
587 do{
588 index = output.Find(__L("\n"), true, indexfr);
589 if(index == XSTRING_NOTFOUND ) break;
590
591 for(int c=index; c>0 ; c--)
592 {
593 if(output.Get()[c] == __C(']'))
594 {
595 indexstart = c;
596 break;
597 }
598 }
599
600 indexfr = index+1;
601
602 if(indexstart)
603 {
604 XSTRING* newstr = new XSTRING();
605 if(newstr)
606 {
607 output.Copy(indexstart+1, index, (*newstr));
608 newstr->DeleteCharacter(__C('\t'), XSTRINGCONTEXT_FROM_FIRST);
609
610 nameAPs.Add(newstr);
611 }
612 }
613
614 } while(index != XSTRING_NOTFOUND);
615
616 return true;
617}
