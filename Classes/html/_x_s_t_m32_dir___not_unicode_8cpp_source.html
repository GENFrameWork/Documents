328{
329 if(!xpath) return false;

351 searchelement->GetPathSearch()->Set(pathsearch);

357
358 pathsearch.ConvertToASCII(charpathsearch);
359 _patternsearch.ConvertToASCII(charpatternsearch);

361 fresult = f_findfirst(&searchinfo->dir, &searchinfo->fileinfo, (const TCHAR*)charpathsearch.GetPtrChar(), (const TCHAR*)charpatternsearch.GetPtrChar());
362 if(fresult != FR_OK || !searchinfo->fileinfo.fname[0])
363 {
364 searchelement->DeleteFindFileData();

366 }
