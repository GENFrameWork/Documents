521SNDITEM* SNDFACTORY::CreateItem(XCHAR* path)
522{
523 if(!soundactive)
524 {
525 return NULL;

527
528 if(!path)
529 {
530 return NULL;
531 }
532

536 return NULL;

540
541 XPATH xpath;

599SNDITEM* SNDFACTORY::CreateItem(XDWORD frecuency, XDWORD duration)
600{
601 if(!soundactive)
602 {
603 return NULL;
604 }
605
606 if((!frecuency) || (!duration))
607 {
608 return NULL;
609 }
610
611 SNDITEM* item = new SNDITEM();
612 if(!item)
613 {
614 return NULL;
615 }
616
617 item->SetType(SNDITEM_TYPE_NOTE);
618
619 item->GetID()->Format(__L("frec: %d (Hz) - time: %d (msec)"), frecuency, duration);
620
621 SNDNOTE* note = new SNDNOTE();
622 if(!note)
623 {
624 delete item;
625 return NULL;
626 }
627
628 note->SetFrequency(frecuency);
629 note->SetDuration(duration);
630
631 item->SetStatus(SNDITEM_STATUS_NONE);
632 item->SetSoundNote(note);
633
634 sounditems.Add(item);
635
636 return item;

664bool SNDFACTORY::Sound_IsAnyActive()
665{
666 if(!soundactive)
667 {
668 return false;
669 }
670
671 XDWORD counter = 0;
672
673 for(XDWORD c=0; c<sounditems.GetSize(); c++)
674 {
675 SNDITEM* item = sounditems.Get(c);
676 if(item)
677 {
678 if((item->GetStatus() != SNDITEM_STATUS_END) &&
679 (item->GetStatus() != SNDITEM_STATUS_NONE))
680 {
681 counter++;
682 }
683 }
684 }
685
686 return counter?true:false;
687}
