302bool DIODNSPROTOCOL_MITM_SERVER::Update()
303{
304 XBUFFER receivedbuffer[2];
305 DIODNSPROTOCOL_HEADER* header = NULL;
306 DIODNSPROTOCOL_QUESTION* question = NULL;
307 XWORD askcount = 0;
308 XBYTE* reader = NULL;
309 int stop = 0;
310 bool status = false;
311
312 if(!diostreamudp)
313 {
314 return false;
315 }
316
317 if(diostreamudp->GetInXBuffer()->GetSize() < sizeof(DIODNSPROTOCOL_HEADER))
318 {
319 return false;
320 }
321
322 receivedbuffer[0].Empty();
323 receivedbuffer[0].Resize(65535);
324
325 XDWORD sizeread = diostreamudp->Read(receivedbuffer[0]);
326 if(!sizeread)
327 {
328 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DSN MitM Server] Error to Read header"));
329 return false;
330 }
331
332 header = (DIODNSPROTOCOL_HEADER*)receivedbuffer[0].Get();
333 askcount = SwapWORD(header->q_count);
334
335 reader = &receivedbuffer[0].Get()[sizeof(DIODNSPROTOCOL_HEADER)];
336
337 XSTRING nameURLstr;
338
339 for(XWORD c=0; c<askcount; c++)
340 {
341 XBYTE* nameURL = NULL;
342
343 nameURL = DIODNSPROTOCOLCLIENT::GetBufferName(reader, receivedbuffer[0].Get(), &stop);
344 nameURLstr.Set((char*)nameURL);
345
346 DIODNSPROTOCOL_MITM_SERVER_XEVENT xevent(this, DIODNSPROTOCOL_MITM_SERVER_XEVENT_TYPE_ASKDNS);
347 xevent.GetAskedURL()->Set(nameURLstr);
348 xevent.GetAskedBuffer()->CopyFrom(receivedbuffer[0]);
349
350 PostEvent(&xevent);
351
352 reader += stop;
353 question = (DIODNSPROTOCOL_QUESTION*)reader;
354 }
355
356 for(XDWORD c=0; c<dnsservers.GetSize(); c++)
357 {
358 DIODNSRESOLVER_DNSSERVER* DNSserver = dnsservers.Get(c);
359 if(DNSserver)
360 {
361 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DSN MitM Server] %d server..."), c);
362
363 status = Detour(DNSserver, receivedbuffer[0], receivedbuffer[1]);
364 if(status)
365 {
366 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[DSN MitM Server] Buffer receiver"));
367 XTRACE_PRINTDATABLOCKCOLOR(XTRACE_COLOR_BLUE, receivedbuffer[1]);
368
369 break;
370 }
371 else
372 {
373 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DSN MitM Server] Error to Detour"));
374 }
375 }
376 }
377
378 status = diostreamudp->Write(receivedbuffer[1]);
379 if(status)
380 {
381 //status = diostreamudp->WaitToFlushOutXBuffer(3);
382 if(status)
383 {
384 DIODNSPROTOCOL_MITM_SERVER_XEVENT xevent(this, DIODNSPROTOCOL_MITM_SERVER_XEVENT_TYPE_ANSWERDNS);
385 xevent.GetAskedURL()->Set(nameURLstr);
386 xevent.GetAnsweredBuffer()->CopyFrom(receivedbuffer[1]);
387
388 PostEvent(&xevent);
389
390 status = true;
391 }
392 else
393 {
394 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DSN MitM Server] Error to Write answer buffer"));
395 }
396 }
397 else
398 {
399 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DSN MitM Server] Error to Write answer"));
400 }

608bool DIODNSPROTOCOL_MITM_SERVER::Detour(DIODNSRESOLVER_DNSSERVER* DNSserver, XBUFFER& sendbuffer, XBUFFER& receivedbuffer)
609{
610 if(!DNSserver)
611 {
612 return false;
613 }
614
615 DIOSTREAMUDPCONFIG dioudpcfg;
616 DIOSTREAMUDP* dioudp = NULL;
617 XSTRING remoteIP;
618 int serverport = 0;
619 bool status = false;
620
621 DNSserver->GetIP()->GetXString(remoteIP);
622
623 dioudpcfg.SetMode(DIOSTREAMMODE_CLIENT);
624 dioudpcfg.SetIsUsedDatagrams(false);
625 dioudpcfg.GetRemoteURL()->Set(remoteIP);
626
627 serverport = DNSserver->GetPort();
628
629 if(!serverport)
630 {
631 dioudpcfg.SetRemotePort(DIODNSPROTOCOL_DEFAULTPORT);
632 }
633 else
634 {
635 dioudpcfg.SetRemotePort(serverport);
636 }
637
638
639 dioudp = (DIOSTREAMUDP*)GEN_DIOFACTORY.CreateStreamIO(&dioudpcfg);
640 if(dioudp)
641 {
642 if(dioudp->Open())
643 {
644 status = dioudp->Write(sendbuffer);
645 if(status)
646 {
647 status = diostreamudp->WaitToFlushOutXBuffer(3);
648 }
649
650 if(status)
651 {
652 receivedbuffer.Empty();
653 receivedbuffer.Resize(65535);
654
655 status = diostreamudp->WaitToFilledReadingBuffer(sizeof(DIODNSPROTOCOL_HEADER), 2);
656 if(status)
657 {
658 status = diostreamudp->Read(receivedbuffer);
659 }
660 }
661
662 dioudp->Close();
663 }
664
665 GEN_DIOFACTORY.DeleteStreamIO(dioudp);
666 }
667
668 return status;
669}
