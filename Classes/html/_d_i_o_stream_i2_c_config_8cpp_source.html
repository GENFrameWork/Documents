150bool DIOSTREAMI2CCONFIG::SetLocalDeviceNameByPort()
151{
152 int revision = 0;
153
154 switch(GEN_XSYSTEM.GetTypeHardware(&revision))
155 {
156 case XSYSTEM_HARDWARETYPE_PC : localdevicename.Format(__L("/dev/i2c-%d"), port);
157 break;
158
159 case XSYSTEM_HARDWARETYPE_RASPBERRYPI : if(revision == 1)
160 localdevicename = __L("/dev/i2c-0");
161 else localdevicename = __L("/dev/i2c-1");
162 break;
163
164 default : break;
165 }
166
167 return true;
168}
