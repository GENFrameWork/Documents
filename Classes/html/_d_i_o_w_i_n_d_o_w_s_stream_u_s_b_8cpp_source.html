246void DIOWINDOWSSTREAMUSB::ThreadConnection(void* data)
247{
248 DIOWINDOWSSTREAMUSB* diostream = (DIOWINDOWSSTREAMUSB*)data;
249 if(!diostream) return;
250
251 if(diostream->GetEvent()==DIOWINDOWSUSBFSMEVENT_NONE) // No hay nuevos Eventos
252 {
253 switch(diostream->GetCurrentState())
254 {
255 case DIOWINDOWSUSBFSMSTATE_NONE : break;
256
257 case DIOWINDOWSUSBFSMSTATE_CONNECTED : break;
258
259 case DIOWINDOWSUSBFSMSTATE_WAITINGTOREAD : { if(!diostream->IsBlockRead())
260 {
261 XBYTE buffer[DIOSTREAM_MAXBUFFER];
262
263 XDWORD br = diostream->ReadBuffer(buffer, DIOSTREAM_MAXBUFFER);
264 if(br)
265 {
266 diostream->inbuffer->Add(buffer, br);
267 }
268 }
269
270 if(::GetLastError() == 5)
271 {
272 diostream->SetEvent(DIOWINDOWSUSBFSMEVENT_DISCONNECTING);
273 break;
274 }
275
276 if(!diostream->IsBlockWrite())
277 {
278 int esize = diostream->outbuffer->GetSize();
279 if(esize)
280 {
281 diostream->outbuffer->SetBlocked(true);
282 XDWORD bw = diostream->WriteBuffer(diostream->outbuffer->Get(), esize);
283 diostream->outbuffer->SetBlocked(false);
284 if(bw) diostream->outbuffer->Extract(NULL, 0, bw);
285 }
286 }
287 }
288 break;
289
290 case DIOWINDOWSUSBFSMSTATE_SENDINGDATA : break;
291
292 case DIOWINDOWSUSBFSMSTATE_DISCONNECTING : break;
293
294 }
295 }
296 else
297 {
298 if(diostream->GetEvent()<DIOWINDOWSUSB_LASTEVENT)
299 {
300 diostream->CheckTransition();
301
302 switch(diostream->GetCurrentState())
303 {
304 case DIOWINDOWSUSBFSMSTATE_NONE : break;
305
306 case DIOWINDOWSUSBFSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
307 xevent.SetDIOStream(diostream);
308
309 diostream->PostEvent(&xevent);
310
311 diostream->SetEvent(DIOWINDOWSUSBFSMEVENT_WAITINGTOREAD);
312 }
313 break;
314
315 case DIOWINDOWSUSBFSMSTATE_WAITINGTOREAD : break;
316
317 case DIOWINDOWSUSBFSMSTATE_SENDINGDATA : break;
318
319 case DIOWINDOWSUSBFSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
320 xevent.SetDIOStream(diostream);
321 diostream->PostEvent(&xevent);
322
323 diostream->threadconnection->Run(false);
324 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
325 }
326 break;
327 }
328 }
329 }
330}

373XDWORD DIOWINDOWSSTREAMUSB::WriteBuffer(XBYTE* buffer, XDWORD size)
374{
375 if(GetStatus()==DIOSTREAMSTATUS_DISCONNECTED) return 0;
376
377 if(size == 0) return 0;
378
379 DWORD bw = 0;
380 OVERLAPPED oviwrite = { 0 };
381 bool status;
382
383 oviwrite.hEvent = CreateEvent(0,TRUE,FALSE,0);
384 if(!oviwrite.hEvent) return bw;
385
386 if(!WriteFile(handle,buffer,size,&bw,&oviwrite))
387 {
388 if(GetLastError()!=ERROR_IO_PENDING)
389 {
390 status = false;
391 }
392 else
393 {
394 switch(WaitForSingleObject(oviwrite.hEvent,writetimeout))
395 {
396 case WAIT_OBJECT_0 : if(!GetOverlappedResult(handle,&oviwrite,&bw,TRUE))
397 status = false;
398 else status = true;
399 break;
400
401 case WAIT_TIMEOUT : CancelIo(handle);
402 status=false;
403 break;
404
405 default : status=false;
406 break;
407
408 }
409 }
410
411 } else status = true;
412
413 CloseHandle(oviwrite.hEvent);
414
415 if(!status) return 0;
416
417 return (XDWORD)bw;
418}
