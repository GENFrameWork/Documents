321bool APPFLOWEXTENDED_APPLICATIONSTATUS::Update()
322{
323
324 if(checkresourceshardware && cfg)
325 {
326 static bool iniappcheck = false;
327
328 if(!iniappcheck)
329 {
330 checkresourceshardware->Ini(cfg);
331 iniappcheck = true;
332 }
333
334 bool ischecktotalCPUusage = cfg->CheckResourcesHardware_GetTotalCPUUsageCheckCadence()?true:false;
335 bool ischeckappCPUusage = cfg->CheckResourcesHardware_GetAppCPUUsageCheckCadence()?true:false;
336
337 if(ischecktotalCPUusage || ischeckappCPUusage)
338 {
339 averange.Empty();
340
341 if(ischecktotalCPUusage)
342 {
343 averange.AddFormat(__L("avg. %d%% (max. %d%%)") , checkresourceshardware->GetCPUTotalCPUUsageAverange()
344 , checkresourceshardware->GetCPUTotalCPUUsageMax());
345 }
346
347 if(ischeckappCPUusage)
348 {
349 if(ischecktotalCPUusage)
350 {
351 averange.AddFormat(__L(", "));
352 }
353
354 if(ischeckappCPUusage)
355 {
356 averange.AddFormat(__L("app %s avg. %d%% (max. %d%%)") , cfg->CheckResourcesHardware_GetAppCPUUsageProcessName()->Get()
357 , checkresourceshardware->GetCPUAppCPUUsageAverange()
358 , checkresourceshardware->GetCPUAppCPUUsageMax());
359 }
360 }
361 }
362 }
363
364 GEN_XSYSTEM.GetMemoryInfo(memory_total, memory_free);
365 memory_freepercent = GEN_XSYSTEM.GetFreeMemoryPercent();
366
367 XDATETIME* datetime = GEN_XFACTORY.CreateDateTime();
368 if(datetime)
369 {
370 datetime->Read();
371 datetime->GetDateTimeToString(XDATETIME_FORMAT_STANDARD | XDATETIME_FORMAT_TEXTMONTH | XDATETIME_FORMAT_ADDDAYOFWEEK, currentdate);
372
373 GEN_XFACTORY.DeleteDateTime(datetime);
374 }
375
376 if(xtimeroperatingtime)
377 {
378 xtimeroperatingtime->GetMeasureString(operatingtime, true);
379 }
380
381 return true;
382}
