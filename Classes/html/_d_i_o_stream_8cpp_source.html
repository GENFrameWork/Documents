466bool DIOSTREAM::ReadStr(XSTRING& string)
467{
468 bool foundreturn = false;
469 int stringsize = 0;
470
471 inbuffer->SetBlocked(true);
472
473 for(int c=0;c<(int)inbuffer->GetSize();c++)
474 {
475 if((inbuffer->GetByte(c) =='\n') || (inbuffer->GetByte(c) == '\r'))
476 {
477 stringsize++;
478
479 if(c+1<(int)inbuffer->GetSize())
480 {
481 if((inbuffer->GetByte(c+1) == '\n') || (inbuffer->GetByte(c+1) =='\r')) stringsize++;
482 }
483
484 foundreturn = true;
485
486 break;
487
488 } else stringsize++;
489 }
490
491 inbuffer->SetBlocked(false);
492
493 if(!foundreturn) return false;
494
495 XBYTE* buffer = new XBYTE[stringsize+1];
496 if(!buffer) return false;
497
498 memset(buffer, 0, stringsize+1);
499
500 int br = Read((XBYTE*)buffer, stringsize);
501
502 if(br != stringsize) return false;
503
504 int c=(int)strlen((const char*)buffer);
505 while(c>=0)
506 {
507 if((buffer[c]=='\n')||(buffer[c]=='\r')) buffer[c]=0;
508 c--;
509 }
510
511 string.Empty();
512 string = (char*)buffer;
513
514 delete [] buffer;
515
516 return true;
517}
