490bool DIOSTREAM::ReadStr(XSTRING& string)
491{
492 bool foundreturn = false;
493 int stringsize = 0;
494
495 inbuffer->SetBlocked(true);
496
497 for(int c=0;c<(int)inbuffer->GetSize();c++)
498 {
499 if((inbuffer->GetByte(c) =='\n') || (inbuffer->GetByte(c) == '\r'))
500 {
501 stringsize++;
502
503 if(c+1<(int)inbuffer->GetSize())
504 {
505 if((inbuffer->GetByte(c+1) == '\n') || (inbuffer->GetByte(c+1) =='\r')) stringsize++;
506 }
507
508 foundreturn = true;
509
510 break;
511
512 } else stringsize++;
513 }
514
515 inbuffer->SetBlocked(false);
516
517 if(!foundreturn) return false;
518
519 XBYTE* buffer = new XBYTE[stringsize+1];
520 if(!buffer) return false;
521
522 memset(buffer, 0, stringsize+1);
523
524 int br = Read((XBYTE*)buffer, stringsize);
525
526 if(br != stringsize) return false;
527
528 int c=(int)strlen((const char*)buffer);
529 while(c>=0)
530 {
531 if((buffer[c]=='\n')||(buffer[c]=='\r')) buffer[c]=0;
532 c--;
533 }
534
535 string.Empty();
536 string = (char*)buffer;
537
538 delete [] buffer;
539
540 return true;
541}
