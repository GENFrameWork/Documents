532bool DIOPCAP_FILTERS::Entrys_Check(DIOPCAPFRAME* frame)
533{
534 bool valid = false;
535
536 if(!frame)
537 {
538 return false;
539 }
540
541 if(!entrys.GetSize())
542 {
543 return true;
544 }
545
546 for(XDWORD c=0; c<entrys.GetSize(); c++)
547 {
548 DIOPCAP_FILTER_ENTRY* entry = entrys.Get(c);
549 if(entry)
550 {
551 XDWORD protocoltype = entry->GetProtocolType();
552
553 if((protocoltype & DIOPCAPPROTOCOL_TYPE_ICMP) == frame->GetProtocolType())
554 {
555 valid = true;
556 }
557
558 if((protocoltype & DIOPCAPPROTOCOL_TYPE_UDP) == frame->GetProtocolType())
559 {
560 valid = true;
561 }
562
563 if((protocoltype & DIOPCAPPROTOCOL_TYPE_TCP) == frame->GetProtocolType())
564 {
565 valid = true;
566 }
567
568 if(valid)
569 {
570 if(!entry->Ports_GetAll()->GetSize())
571 {
572 return valid;
573 }
574
575 for(XDWORD d=0; d<entry->Ports_GetAll()->GetSize(); d++)
576 {
577 XDWORD port_target = entry->Ports_GetAll()->Get(d);
578 XDWORD port_source = (port_target & ~DIOPCAP_FILTERS_PORT_TARGET);
579
580 if(port_source == frame->GetSourcePort())
581 {
582 return valid;
583 }
584
585 if(port_target == frame->GetTargetPort())
586 {
587 return valid;
588 }
589 }
590 }
591 }
592 }
593
594 return false;
595}
