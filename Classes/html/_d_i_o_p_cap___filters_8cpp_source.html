539bool DIOPCAP_FILTERS::Entrys_Check(DIOPCAPFRAME* frame)
540{
541 bool valid = false;
542
543 if(!frame)
544 {
545 return false;
546 }
547
548 if(!entrys.GetSize())
549 {
550 return true;
551 }
552
553 for(XDWORD c=0; c<entrys.GetSize(); c++)
554 {
555 DIOPCAP_FILTER_ENTRY* entry = entrys.Get(c);
556 if(entry)
557 {
558 XDWORD protocoltype = entry->GetProtocolType();
559
560 if((protocoltype & DIOPCAPPROTOCOL_TYPE_ICMP) == frame->GetProtocolType())
561 {
562 valid = true;
563 }
564
565 if((protocoltype & DIOPCAPPROTOCOL_TYPE_UDP) == frame->GetProtocolType())
566 {
567 valid = true;
568 }
569
570 if((protocoltype & DIOPCAPPROTOCOL_TYPE_TCP) == frame->GetProtocolType())
571 {
572 valid = true;
573 }
574
575 if(valid)
576 {
577 if(!entry->Ports_GetAll()->GetSize())
578 {
579 return valid;
580 }
581
582 for(XDWORD d=0; d<entry->Ports_GetAll()->GetSize(); d++)
583 {
584 XDWORD port_target = entry->Ports_GetAll()->Get(d);
585 XDWORD port_source = (port_target & ~DIOPCAP_FILTERS_PORT_TARGET);
586
587 if(port_source == frame->GetSourcePort())
588 {
589 return valid;
590 }
591
592 if(port_target == frame->GetTargetPort())
593 {
594 return valid;
595 }
596 }
597 }
598 }
599 }
600
601 return false;
602}
