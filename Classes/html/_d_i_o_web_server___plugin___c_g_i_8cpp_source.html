132bool DIOWEBSERVER_PLUGIN_CGI::PageExtension(XPATH& pathfile, DIOWEBSERVER_REQUEST* request, DIOWEBSERVER_QUERYSTRINGS* querystrings, DIOWEBHEADER_RESULT& headerresult, XSTRING& result)
133{
134 if(!request) return false;
135 if(!querystrings) return false;
136
137 XSTRING allparam;
138
139 headerresult = DIOWEBHEADER_RESULT_OK;
140
141 querystrings->GetAllParam(allparam);
142
143 XPATH pathexec;
144
145 pathexec = pathCGIinterpreter.Get();
146 pathexec.Slash_Add();
147 pathexec += nameexec;
148
149 XSTRING methodstring;
150 int returnerror = 0;
151 XSTRING in;
152 int sizein = 0;
153 XSTRING sizeinstr;
154 bool status;
155
156 if(request->GetData())
157 {
158 sizein = request->GetData()->GetSize();
159 if(sizein)
160 {
161
162 in.Add((*request->GetData()));
163 sizeinstr.Format(__L("%d"), sizein);
164 }
165 }
166
167 switch(request->GetMethod())
168 {
169 case DIOWEBHEADER_METHOD_GET : methodstring = __L("\"GET\""); break;
170 case DIOWEBHEADER_METHOD_POST : methodstring = __L("\"POST\""); break;
171 case DIOWEBHEADER_METHOD_PUT : methodstring = __L("\"PUT\""); break;
172 case DIOWEBHEADER_METHOD_DELETE : methodstring = __L("\"DELETE\""); break;
173 default : break;
174 }
175
176
177 GEN_XSYSTEM.SetEnviromentVariable(__L("GATEWAY_INTERFACE") , __L("CGI/1.1"));
178 GEN_XSYSTEM.SetEnviromentVariable(__L("REQUEST_METHOD") , __L("POST"));
179 GEN_XSYSTEM.SetEnviromentVariable(__L("SCRIPT_FILENAME") , pathfile.Get());
180 GEN_XSYSTEM.SetEnviromentVariable(__L("CONTENT_LENGTH") , sizeinstr.Get());
181 GEN_XSYSTEM.SetEnviromentVariable(__L("REDIRECT_STATUS") , __L("true"));
182 GEN_XSYSTEM.SetEnviromentVariable(__L("CONTENT_TYPE") , __L("application/x-www-form-urlencoded"));
183
184 if(request->GetMethod() == DIOWEBHEADER_METHOD_GET) GEN_XSYSTEM.SetEnviromentVariable(__L("QUERY_STRING"), allparam.GetSize()?allparam.Get():__L("\"\""));
185
186 status = GEN_XPROCESSMANAGER.Application_Execute(pathexec.Get(), NULL, &in, &result, &returnerror);
187
188 GEN_XSYSTEM.DelEnviromentVariable(__L("GATEWAY_INTERFACE"));
189 GEN_XSYSTEM.DelEnviromentVariable(__L("REQUEST_METHOD"));
190 GEN_XSYSTEM.DelEnviromentVariable(__L("SCRIPT_FILENAME"));
191 GEN_XSYSTEM.DelEnviromentVariable(__L("CONTENT_LENGTH"));
192 GEN_XSYSTEM.DelEnviromentVariable(__L("REDIRECT_STATUS"));
193 GEN_XSYSTEM.DelEnviromentVariable(__L("CONTENT_TYPE"));
194 GEN_XSYSTEM.DelEnviromentVariable(__L("QUERY_STRING"));
195
196 return status;
197}
