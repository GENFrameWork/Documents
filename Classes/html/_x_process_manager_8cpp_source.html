627bool XPROCESSMANAGER::Application_Terminate(XCHAR* processname, XDWORD exitcode)
628{
629 XVECTOR<XPROCESS*> apprunning;
630 bool status = false;
631
632 if(!GEN_XPROCESSMANAGER.Application_GetRunningList(apprunning)) return false;
633
634 for(XDWORD c=0; c<apprunning.GetSize(); c++)
635 {
636 XPROCESS* xprocess = apprunning.Get(c);
637 if(xprocess)
638 {
639 if(!xprocess->GetName()->Compare(processname, true))
640 {
641 status = Application_Terminate((*xprocess), exitcode);
642 break;
643 }
644 }
645 }
646
647 apprunning.DeleteContents();
648 apprunning.DeleteAll();
649
650 return status;
651}
