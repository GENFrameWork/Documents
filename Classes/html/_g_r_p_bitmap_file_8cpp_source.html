416GRPBITMAPFILE_TYPE GRPBITMAPFILE::GetTypeFromExtenxion(XCHAR* xpath)
417{
418 XPATH _xpath;
419 XSTRING ext;
420 GRPBITMAPFILE_TYPE type = GRPBITMAPFILE_TYPE_UNKNOWN;
421
422 _xpath = xpath;
423
424 _xpath.GetExt(ext);
425
426 if(!ext.Compare(__L(".jpg"), true)) type = GRPBITMAPFILE_TYPE_JPG;
427 if(!ext.Compare(__L(".jpeg"), true)) type = GRPBITMAPFILE_TYPE_JPG;
428 if(!ext.Compare(__L(".png"), true)) type = GRPBITMAPFILE_TYPE_PNG;
429 if(!ext.Compare(__L(".bmp"), true)) type = GRPBITMAPFILE_TYPE_BMP;
430 if(!ext.Compare(__L(".tga"), true)) type = GRPBITMAPFILE_TYPE_TGA;
431
432 return type;
433}

448GRPBITMAP* GRPBITMAPFILE::Load(XCHAR* xpath, GRPPROPERTYMODE forcemode)
449{
450 if(!xpath) return NULL;
451
452 XPATH _xpath;
453 GRPBITMAPFILE_TYPE type = GetTypeFromExtenxion(xpath);
454 GRPBITMAPFILEBASE* grpfilebase = NULL;
455 GRPBITMAP* bitmap = NULL;
456
457 _xpath = xpath;
458
459 switch(type)
460 {
461 case GRPBITMAPFILE_TYPE_UNKNOWN :
462 default : break;
463
464 #ifdef GRP_BITMAP_FILE_BMP_ACTIVE
465 case GRPBITMAPFILE_TYPE_BMP : grpfilebase = new GRPBITMAPFILEBMP();
466 break;
467 #endif
468
469 #ifdef GRP_BITMAP_FILE_JPG_ACTIVE
470 case GRPBITMAPFILE_TYPE_JPG : grpfilebase = new GRPBITMAPFILEJPG();
471 break;
472 #endif
473
474 #ifdef GRP_BITMAP_FILE_PNG_ACTIVE
475 case GRPBITMAPFILE_TYPE_PNG : grpfilebase = new GRPBITMAPFILEPNG();
476 break;
477 #endif
478
479 #ifdef GRP_BITMAP_FILE_JPG_ACTIVE
480 case GRPBITMAPFILE_TYPE_TGA : grpfilebase = new GRPBITMAPFILETGA();
481 break;
482 #endif
483 }
484
485 if(grpfilebase)
486 {
487 bitmap = grpfilebase->CreateBitmapFromFile(_xpath, forcemode);
488
489 delete grpfilebase;

645GRPBITMAPSEQUENCE* GRPBITMAPFILE::LoadSequence(XCHAR* namemask, int inisequence, int endsequence)
646{
647 if(xpath.IsEmpty()) return NULL;
648
649 GRPBITMAPSEQUENCE* bitmapsequence = new GRPBITMAPSEQUENCE();
650 if(!bitmapsequence) return NULL;
651
652 XSTRING namefile;
653
654 //xpath.GetNamefileExt(namefile);
655 //if(!namefile.IsEmpty()) return NULL;
656
657 // The sequence is in individual files in a directory. Sequence numbers must be continuos
658 int nfiles = inisequence;
659
660 while((nfiles < endsequence)|| (endsequence < 0))
661 {
662 XPATH xpathtemp;
663 bool existfile = false;
664
665 namefile.Format(namemask, nfiles);
666 xpathtemp.Create(2, xpath.Get(), namefile.Get());
667
668 XFILE* GEN_XFACTORY_CREATE(xfile, Create_File())
669 if(xfile)
670 {
671 existfile = xfile->Exist(xpathtemp);
672 GEN_XFACTORY.Delete_File(xfile);
673 }
674
675 if(!existfile) break;
676
677 GRPBITMAP* bitmap = Load(xpathtemp);
678 if(!bitmap) break;
679
680 bitmapsequence->AddFrame(bitmap);
681
682 nfiles++;
683 }
684
685 return bitmapsequence;
686}
