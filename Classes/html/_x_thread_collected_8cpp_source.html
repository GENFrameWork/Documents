873void XTHREADSCOLLECTED::ThreadRunFunction(void* param)
874{
875 XTHREADSCOLLECTED* xthreadscollected = (XTHREADSCOLLECTED*)param;
876 if(!xthreadscollected) return;
877
878 if(!xthreadscollected->xthreadsvectormutex) return;
879
880 xthreadscollected->xthreadsvectormutex->Lock();
881
882 for(XDWORD c=0; c<xthreadscollected->xthreadsvector.GetSize(); c++)
883 {
884 XTHREADCOLLECTED* xthreadcollected = xthreadscollected->xthreadsvector.Get(c);
885 if(xthreadcollected)
886 {
887 switch(xthreadcollected->GetStatus())
888 {
889 case XTHREADSTATUS_RUN : xthreadcollected->SetIsInFunction(true);
890 xthreadcollected->GetFunction()(xthreadcollected->GetParam());
891 xthreadcollected->SetIsInFunction(false);
892
893 xthreadscollected->xthread->Wait();
894 break;
895
896 case XTHREADSTATUS_EXIT : xthreadcollected->SetStatus(XTHREADSTATUS_END);
897 break;
898 }
899 }
900 }
901
902 xthreadscollected->xthreadsvectormutex->UnLock();
903}
