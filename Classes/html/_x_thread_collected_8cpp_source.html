880void XTHREADSCOLLECTED::ThreadRunFunction(void* param)
881{
882 XTHREADSCOLLECTED* xthreadscollected = (XTHREADSCOLLECTED*)param;
883 if(!xthreadscollected) return;
884
885 if(!xthreadscollected->xthreadsvectormutex) return;
886
887 xthreadscollected->xthreadsvectormutex->Lock();
888
889 for(XDWORD c=0; c<xthreadscollected->xthreadsvector.GetSize(); c++)
890 {
891 XTHREADCOLLECTED* xthreadcollected = xthreadscollected->xthreadsvector.Get(c);
892 if(xthreadcollected)
893 {
894 switch(xthreadcollected->GetStatus())
895 {
896 case XTHREADSTATUS_RUN : xthreadcollected->SetIsInFunction(true);
897 xthreadcollected->GetFunction()(xthreadcollected->GetParam());
898 xthreadcollected->SetIsInFunction(false);
899
900 xthreadscollected->xthread->Wait();
901 break;
902
903 case XTHREADSTATUS_EXIT : xthreadcollected->SetStatus(XTHREADSTATUS_END);
904 break;
905 }
906 }
907 }
908
909 xthreadscollected->xthreadsvectormutex->UnLock();
910}
