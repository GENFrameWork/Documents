670bool XSYSTEM::HostFile(bool add, XCHAR* domain, XCHAR* IP)
671{
672 XPATH xpath;
673 XFILETXT xfileTXT;
674 bool status = false;
675
676 #ifdef WINDOWS
677 xpath = __L("C:\\Windows\\System32\\drivers\\etc\\hosts");
678 #endif
679
680 #if defined(LINUX) || defined(ANDROID)
681 xpath = __L("/etc/hosts");
682 #endif
683
684 if(xfileTXT.Open(xpath, false))
685 {
686 xfileTXT.ReadAllFile();
687
688 if(add)
689 {
690 int nline = IsLineInHostFile(xfileTXT, domain, IP);
691 if(nline == XSTRING_NOTFOUND)
692 {
693 XSTRING line;
694
695 line.Format(__L("%s %s"), IP, domain);
696
697 xfileTXT.AddLine(line);
698 xfileTXT.WriteAllFile();
699 }
700 }
701 else
702 {
703 int nline = IsLineInHostFile(xfileTXT, domain, IP);
704 if(nline != XSTRING_NOTFOUND)
705 {
706 xfileTXT.DeleteLine(nline);
707 status = xfileTXT.WriteAllFile();
708 }
709 }
710
711 xfileTXT.Close();
712 }
713
714 return status;
715}

750int XSYSTEM::IsLineInHostFile(XFILETXT& xfileTXT, XCHAR* domain, XCHAR* IP)
751{
752 bool found = false;
753 int c = 0;
754
755 if(!xfileTXT.GetNLines()) return XSTRING_NOTFOUND;
756
757 do{ XSTRING* line = xfileTXT.GetLine(c);
758 if(line)
759 {
760 int index = line->Find(domain, true);
761 if(index != XSTRING_NOTFOUND)
762 {
763 found = true;
764
765 for(XDWORD d=0; d<line->GetSize(); d++)
766 {
767 XCHAR character = line->Get()[d];
768
769 if((character != __C('\t') && (character != __C(' '))))
770 {
771 if(character == __C('#'))
772 {
773 found = false;
774 }
775
776 break;
777 }
778 }
779
780 if(found) break;
781 }
782 }
783
784 c++;
785
786 } while(c<xfileTXT.GetNLines());
787
788 if(!found) return XSTRING_NOTFOUND;
789
790 return c;
791}
