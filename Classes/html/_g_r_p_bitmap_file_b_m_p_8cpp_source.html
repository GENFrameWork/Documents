184bool GRPBITMAPFILEBMPHEADER::Load(XFILE* file)
185{
186 if(!file->Read((XBYTE*)&headerinfo.type , sizeof(XWORD))) return false;
187 if(!file->Read((XBYTE*)&headerinfo.size , sizeof(XDWORD))) return false;
188 if(!file->Read((XBYTE*)&headerinfo.reserved1 , sizeof(XWORD))) return false;
189 if(!file->Read((XBYTE*)&headerinfo.reserved2 , sizeof(XWORD))) return false;
190 if(!file->Read((XBYTE*)&headerinfo.offsetdata , sizeof(XDWORD))) return false;
191
192 if(!file->Read((XBYTE*)&headerinfo.headersize , sizeof(XDWORD))) return false;
193 if(!file->Read((XBYTE*)&headerinfo.width , sizeof(XDWORD))) return false;
194 if(!file->Read((XBYTE*)&headerinfo.height , sizeof(XDWORD))) return false;
195 if(!file->Read((XBYTE*)&headerinfo.planes , sizeof(XWORD))) return false;
196 if(!file->Read((XBYTE*)&headerinfo.bitsperpixel , sizeof(XWORD))) return false;
197 if(!file->Read((XBYTE*)&headerinfo.compression , sizeof(XDWORD))) return false;
198 if(!file->Read((XBYTE*)&headerinfo.sizebmp , sizeof(XDWORD))) return false;
199 if(!file->Read((XBYTE*)&headerinfo.xpixelspermeter , sizeof(XDWORD))) return false;
200 if(!file->Read((XBYTE*)&headerinfo.ypixelspermeter , sizeof(XDWORD))) return false;
201 if(!file->Read((XBYTE*)&headerinfo.colorused , sizeof(XDWORD))) return false;
202 if(!file->Read((XBYTE*)&headerinfo.colorimportant , sizeof(XDWORD))) return false;
203
204
205 // Check: "BM" signature in the file for WINDOWS NO OS/2 valid format.
206 if(headerinfo.type!=0x4D42) return false;
207
208 switch(headerinfo.bitsperpixel)
209 {
210 case 8 : mode = GRPPROPERTYMODE_08_INDEX;
211 break;
212
213 case 16 : { XDWORD maskcolor[4];
214 if(!file->Read((XBYTE*)maskcolor, 16)) return false;
215
216 if((maskcolor[0] == 0x00007c00) && (maskcolor[1] == 0x000003E0) && (maskcolor[2] == 0x0000001F) && (maskcolor[3] == 0x00008000)) mode = GRPPROPERTYMODE_16_RGBA_5551;
217 if((maskcolor[0] == 0x0000F800) && (maskcolor[1] == 0x000007E0) && (maskcolor[2] == 0x0000001F) && (maskcolor[3] == 0x00000000)) mode = GRPPROPERTYMODE_16_RGB_565;
218 }
219 break;
220
221
222 case 24 : mode = GRPPROPERTYMODE_24_RGB_888;
223 break;
224
225 case 32 : mode = GRPPROPERTYMODE_32_RGBA_8888;
226 break;
227 }
228
229
230 DeleteAllPalette();
231
232 int ncpalette = headerinfo.colorused;
233
234 if(ncpalette)
235 {
236 for(int c=0;c<ncpalette;c++)
237 {
238 XBYTE red;
239 XBYTE green;
240 XBYTE blue;
241 XBYTE alpha;
242
243 if(!file->Read(&blue ,1)) return false;
244 if(!file->Read(&green ,1)) return false;
245 if(!file->Read(&red ,1)) return false;
246 if(!file->Read(&alpha ,1)) return false;
247
248 GRPBITMAPFILEBMPCOLORPALETE* colorpalette = new GRPBITMAPFILEBMPCOLORPALETE(red, green, blue, alpha);
249 if(colorpalette) palette.Add((XBYTE)c, colorpalette);
250 }
251 }
252
253 return true;
254}

539GRPBITMAP* GRPBITMAPFILEBMP::CreateBitmapFromFile(XPATH& xpath, GRPPROPERTYMODE forcemode)
540{
541 if(!Open(xpath)) return NULL;
542
543 GRPBITMAPFILEBMPHEADER header;
544 GRPBITMAP* bitmap = NULL;
545//XBYTE* buffer = NULL;
546 bool status = true;
547
548 if(!ReadHeaderFile(&header))
549 {
550 Close();
551 return NULL;
552 }
553
554 int width = header.GetWidth();
555 int height = header.GetHeight();
556 int bytesppixel = (header.GetBitsPerPixel()/8);
557
558 bitmap = GRPFACTORY::GetInstance().CreateBitmap(width, height, header.GetMode());
559 if(bitmap)
560 {
561 int sizebmp = (((header.GetWidth()*bytesppixel)+3)&(~3))*header.GetHeight();
562 XBYTE* databuffer = NULL;
563 int index = 0;
564
565 databuffer = new XBYTE[sizebmp];
566 if(databuffer)
567 {
568 file->SetPosition(header.GetOffsetData());
569
570 if(!file->Read(databuffer, sizebmp)) return NULL;
571
572 for(int y=height-1; y>-1; y--)
573 {
574 for(int x=0; x<width; x++)
575 {
576 GRP2DCOLOR_RGBA8 color;
577
578 switch(header.GetMode())
579 {
580 case GRPPROPERTYMODE_08_INDEX : header.GetColorPalette(databuffer[index], color);
581 color.a = 255;
582 bitmap->PutPixel(x, y, (GRP2DCOLOR*)&color);
583 break;
584
585 case GRPPROPERTYMODE_16_RGBA_5551 : { XWORD rgb = databuffer[index+1];
586 rgb <<= 8;
587 rgb |= databuffer[index+0];
588
589 color.r = (rgb >> 7) & 0xF8;
590 color.g = (rgb >> 2) & 0xF8;
591 color.b = (rgb << 3) & 0xF8;
592 color.a = (rgb & 0x8000)?255:0;
593
594 bitmap->PutBlendPixel(x, y, (GRP2DCOLOR*)&color, (int)color.a);
595 }
596 break;
597
598 case GRPPROPERTYMODE_16_RGB_565 : { XWORD rgb = databuffer[index+1];
599 rgb <<= 8;
600 rgb |= databuffer[index+0];
601
602 color.r = (rgb >> 8) & 0xF8;
603 color.g = (rgb >> 3) & 0xFC;
604 color.b = (rgb << 3) & 0xF8;
605 color.a = 255;
606
607 bitmap->PutPixel(x, y, (GRP2DCOLOR*)&color);
608 }
609 break;
610
611 case GRPPROPERTYMODE_24_RGB_888 : color.r = databuffer[index+2];
612 color.g = databuffer[index+1];
613 color.b = databuffer[index+0];
614 color.a = 255;
615
616 bitmap->PutPixel(x, y, (GRP2DCOLOR*)&color);
617 break;
618
619 case GRPPROPERTYMODE_32_RGBA_8888 : color.r = databuffer[index+2];
620 color.g = databuffer[index+1];
621 color.b = databuffer[index+0];
622 color.a = databuffer[index+3];
623
624 bitmap->PutBlendPixel(x, y, (GRP2DCOLOR*)&color, (int)color.a);
625 break;
626 default: break;
627 }
628
629 index += bytesppixel;
630 }
631 }
632
633 } else status = false;
634
635 delete [] databuffer;
636
637 } else status = false;
638
639
640 if(!status)
641 {
642 if(bitmap)
643 {
644 GRPFACTORY::GetInstance().DeleteBitmap(bitmap);
645 bitmap = NULL;
646 }
647 }
648
649 Close();
650
651 return ForceModeOfBitmap(forcemode, bitmap);
652}
