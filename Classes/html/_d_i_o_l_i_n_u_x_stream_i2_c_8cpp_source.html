130bool DIOLINUXSTREAMI2C::Open()
131{
132 if(!config) return false;
133
134 if(!threadconnection) return false;
135
136 XBUFFER charstr;
137
138 (*config->GetLocalDeviceName()).ConvertToASCII(charstr);
139 handle = open(charstr.GetPtrChar(), O_RDWR);
140 if(handle<0) return false;
141
142 switch(config->GetAddressSize())
143 {
144 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_DEFAULT :
145 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_7BITS : break;
146 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_10BITS : if(ioctl(handle, DIOLINUXSTREAMI2C_TENBIT, 1) < 0) return false;
147 break;
148 }
149
150 if(ioctl(handle, DIOLINUXSTREAMI2C_SLAVE , config->GetRemoteAddress()) < 0) return false;
151
152 SetEvent(DIOSTREAMI2C_FSMEVENT_CONNECTED);
153
154 status = DIOSTREAMSTATUS_CONNECTED;
155
156 ResetXBuffers();
157 ResetConnectionStatistics();
158
159 return threadconnection->Ini();
160}

295void DIOLINUXSTREAMI2C::ThreadConnection(void* data)
296{
297 DIOLINUXSTREAMI2C* diostream = (DIOLINUXSTREAMI2C*)data;
298 if(!diostream) return;
299
300 if(diostream->GetEvent()==DIOSTREAMI2C_FSMEVENT_NONE) // No hay nuevos Eventos
301 {
302 switch(diostream->GetCurrentState())
303 {
304 case DIOSTREAMI2C_FSMSTATE_NONE : break;
305
306 case DIOSTREAMI2C_FSMSTATE_CONNECTED : break;
307
308 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : {
309 if(!diostream->IsBlockWrite())
310 {
311 XDWORD size = diostream->outbuffer->GetSize();
312 if(size)
313 {
314 diostream->outbuffer->SetBlocked(true);
315 XDWORD bw = diostream->WriteDirect(diostream->outbuffer->Get(), size);
316 diostream->outbuffer->SetBlocked(false);
317
318 if(bw)
319 {
320 //XTRACE_PRINTDATABLOCKCOLOR(XTRACE_COLORRED, diostream->outbuffer->Get(), size);
321 diostream->outbuffer->Extract(NULL, 0 , bw);
322 }
323 }
324 }
325
326
327 if(!diostream->IsBlockRead())
328 {
329 if(diostream->sizeread)
330 {
331 XBYTE* _data = new XBYTE[diostream->sizeread];
332 if(_data)
333 {
334 if(diostream->ReadDirect(_data, diostream->sizeread))
335 {
336 //XTRACE_PRINTDATABLOCKCOLOR(XTRACE_COLORGREEN, _data, diostream->sizeread);
337 diostream->inbuffer->Add(_data, diostream->sizeread);
338 diostream->sizeread = 0;
339 }
340
341 delete [] _data;
342 }
343 }
344 }
345 }
346 break;
347
348
349 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
350
351 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : break;
352
353 }
354 }
355 else
356 {
357 if(diostream->GetEvent()<DIOSTREAMI2C_LASTEVENT)
358 {
359 diostream->CheckTransition();
360
361 switch(diostream->GetCurrentState())
362 {
363 case DIOSTREAMI2C_FSMSTATE_NONE : break;
364
365 case DIOSTREAMI2C_FSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
366 xevent.SetDIOStream(diostream);
367 diostream->PostEvent(&xevent);
368
369 diostream->SetEvent(DIOSTREAMI2C_FSMEVENT_WAITINGTOREAD);
370 }
371 break;
372
373 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : break;
374
375 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
376
377 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
378 xevent.SetDIOStream(diostream);
379 diostream->PostEvent(&xevent);
380
381 diostream->threadconnection->Run(false);
382 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
383 }
384 break;
385 }
386 }
387 }
388}
