154bool XWINDOWSDIR::Make(XCHAR* path, bool recursive)
155{
156 if(!path)
157 {
158 return false;
159 }
160
161 if(!path[0])
162 {
163 return false;
164 }
165
166 if(Exist(path))
167 {
168 return true;
169 }
170
171 XPATH xpath;
172
173 xpath = path;
174
175 xpath.Slash_Add();
176 xpath.Slash_Normalize(true);
177
178 if(recursive)
179 {
180 XPATH xpathsequence;
181 XSTRING pathpart;
182 int index = 0;
183
184 xpathsequence.Empty();
185
186 do{
187 if(xpath.GetPathInSequence(index,pathpart))
188 {
189 xpathsequence += pathpart;
190 xpathsequence += __L("\\");
191
192 if(pathpart.Character_GetLast()!=__C(':'))
193 {
194 if(!Exist(xpathsequence.Get()))
195 {
196 //XTRACE_PRINTCOLOR(2, __L("---> Make: %s"), xpathsequence.Get());
197
198 int result = CreateDirectory(xpathsequence.Get(), NULL);
199 if(!result)
200 {
201 return false;
202 }
203 }
204 }
205
206 } else break;
207
208 index++;
209
210 } while(1);
211 }
212 else
213 {
214 int result = CreateDirectory(xpath.Get(), NULL);
215 if(!result)
216 {
217 return false;
218 }
219 }
220
221 return true;
222}

271
272 if(all)
273 {
274 xpathname = path;

338bool XWINDOWSDIR::FirstSearch(XCHAR* xpath,XCHAR* patternsearch,XDIRELEMENT* searchelement)
339{
340 if(!xpath)
341 {
342 return false;
343 }
344
345 if(!patternsearch)
346 {
347 return false;
348 }
349
350 XBYTE* findfiledata = new XBYTE[sizeof(WIN32_FIND_DATA)];
351 if(!findfiledata)
352 {
353 return false;
354 }
355
356 memset(findfiledata,0,sizeof(WIN32_FIND_DATA));
357
358 searchelement->SetFindFileData(findfiledata);
359

362 XPATH pathsearch;
363

365 pathsearch.Slash_Add();
366 pathsearch += patternsearch;
367
368 WIN32_FIND_DATA* search = (WIN32_FIND_DATA*)searchelement->GetFindFileData();
369
370 searchelement->SetHandle((void*)FindFirstFile(pathsearch.Get() , search));
371 if((HANDLE)(searchelement->GetHandle()) == INVALID_HANDLE_VALUE)
372 {
373 searchelement->SetHandle((void*)-1);
374 searchelement->DeleteFindFileData();
375
376 return false;
377 }
378
379 XSTRING searchfilename(search->cFileName);
380
381 if((!searchfilename.Compare(__L(".")))||
382 (!searchfilename.Compare(__L(".."))))
383 {
384 return NextSearch(searchelement);
385 }
386
387 ReconvertSearchData(searchelement);
388
389 return true;

476bool XWINDOWSDIR::SetFileDate(XDIRELEMENT* searchelement)
477{
478 WIN32_FIND_DATA* search = (WIN32_FIND_DATA*)searchelement->GetFindFileData();
479 XDATETIME* datetimefile = NULL;
480 SYSTEMTIME st;
481

483 {
484 switch(c)
485 {

487 FileTimeToSystemTime(&search->ftCreationTime, &st);
488 break;
489
490 case 1 : datetimefile = searchelement->GetDateTimeFile_Modificated();
491 FileTimeToSystemTime(&search->ftLastWriteTime, &st);
492 break;
493

495 FileTimeToSystemTime(&search->ftLastAccessTime, &st);
496 break;
497 }

501 datetimefile->SetYear(st.wYear);
