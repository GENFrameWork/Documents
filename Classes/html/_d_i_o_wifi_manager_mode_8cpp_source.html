190bool DIOWIFIMANAGERMODE::CheckNetInterface(DIOIP* IP)
191{
192 bool status = false;
193
194 DIOSTREAMENUMDEVICES* enumdevicesIPlocal = NULL;
195
196 enumdevicesIPlocal = GEN_DIOFACTORY.CreateStreamEnumDevices(DIOSTREAMENUMTYPE_IP_LOCAL);
197 if(enumdevicesIPlocal)
198 {
199 enumdevicesIPlocal->Search();
200 while(enumdevicesIPlocal->IsSearching())
201 {
202 XSLEEP::GetInstance().MilliSeconds(10);
203 }
204
205 for(int c=0; c<(int)enumdevicesIPlocal->GetDevices()->GetSize(); c++)
206 {
207 DIOSTREAMDEVICEIP* deviceIP = (DIOSTREAMDEVICEIP*)enumdevicesIPlocal->GetDevices()->Get(c);
208 if(deviceIP)
209 {
210 bool isdeviceselect = false;
211
212 if(!netinterface.IsEmpty())
213 {
214 if(!deviceIP->GetName()->Compare(netinterface, true)) isdeviceselect = true;
215 }
216 else
217 {
218 if(deviceIP->GetIPType() == DIOSTREAMIPDEVICE_TYPE_WIFI) isdeviceselect = true;
219 }
220
221
222 if(isdeviceselect)
223 {
224 status = true;
225
226 if(deviceIP->GetIP()->IsEmpty()) status = false;
227 if(!deviceIP->GetIP()->IsLocal()) status = false;
228
229 if(IP) IP->Set(deviceIP->GetIP()->Get());
230
231 break;
232 }
233 }
234 }
235
236 GEN_DIOFACTORY.DeleteStreamEnumDevices(enumdevicesIPlocal);
237 }
238
239 return status;
240}
