197bool DIOWIFIMANAGERMODE::CheckNetInterface(DIOIP* IP)
198{
199 bool status = false;
200
201 DIOSTREAMENUMDEVICES* enumdevicesIPlocal = NULL;
202
203 enumdevicesIPlocal = GEN_DIOFACTORY.CreateStreamEnumDevices(DIOSTREAMENUMTYPE_IP_LOCAL);
204 if(enumdevicesIPlocal)
205 {
206 enumdevicesIPlocal->Search();
207 while(enumdevicesIPlocal->IsSearching())
208 {
209 XSLEEP::GetInstance().MilliSeconds(10);
210 }
211
212 for(int c=0; c<(int)enumdevicesIPlocal->GetDevices()->GetSize(); c++)
213 {
214 DIOSTREAMDEVICEIP* deviceIP = (DIOSTREAMDEVICEIP*)enumdevicesIPlocal->GetDevices()->Get(c);
215 if(deviceIP)
216 {
217 bool isdeviceselect = false;
218
219 if(!netinterface.IsEmpty())
220 {
221 if(!deviceIP->GetName()->Compare(netinterface, true)) isdeviceselect = true;
222 }
223 else
224 {
225 if(deviceIP->GetIPType() == DIOSTREAMIPDEVICE_TYPE_WIFI) isdeviceselect = true;
226 }
227
228
229 if(isdeviceselect)
230 {
231 status = true;
232
233 if(deviceIP->GetIP()->IsEmpty()) status = false;
234 if(!deviceIP->GetIP()->IsLocal()) status = false;
235
236 if(IP) IP->Set(deviceIP->GetIP()->Get());
237
238 break;
239 }
240 }
241 }
242
243 GEN_DIOFACTORY.DeleteStreamEnumDevices(enumdevicesIPlocal);
244 }
245
246 return status;
247}
