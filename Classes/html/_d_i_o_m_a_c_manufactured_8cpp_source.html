289bool DIOMACMANUFACTURED::File_Convert(XBUFFER& xbuffer, XPATH& xpath)
290{
291 if(xpath.IsEmpty()) return false;
292
293 XFILETXT* xfiletxtoui = new XFILETXT();
294 if(!xfiletxtoui) return false;
295
296 XMAP<XDWORD, XSTRING*> idmap;
297 XMAP<XDWORD, int> idmapindex;
298 XDWORD c;
299
300 idmap.DeleteAll();
301 idmapindex.DeleteAll();
302
303 idmap.SetAddInLimit(10000);
304 idmapindex.SetAddInLimit(10000);
305
306 xfiletxtoui->AddBufferLines(XFILETXTFORMATCHAR_ASCII, xbuffer);
307
308 int nlines = xfiletxtoui->GetNLines();
309 for(int c=0;c<nlines;c++)
310 {
311 XSTRING* line = xfiletxtoui->GetLine(c);
312 if(line)
313 {
314 if(!line->IsEmpty())
315 {
316 int start = line->Find(__L("(base 16)"),true);
317 if(start!=XSTRING_NOTFOUND)
318 {
319 XDWORD ID = 0;
320
321 //XTRACE_PRINTCOLOR(2,__L("Line %d / %d"), c, nlines);
322
323 start+=11;
324 line->UnFormat(__L("%X (base 16)\t\t"),&ID);
325
326 if(idmap.Find(ID)==-1)
327 {
328 XSTRING* name = new XSTRING();
329 if(name)
330 {
331 line->Copy(start,(*name));
332 idmap.Add(ID,name);
333
334 //XTRACE_PRINTCOLOR(2,__L("[%08X] [%s]"), ID, name->Get());
335 }
336 }
337 }
338 }
339 }
340 }
341
342 delete xfiletxtoui;
343
344 XDWORD nrecords = idmap.GetSize();
345 HASHMD5 hashmd5;
346 XDWORD initablepos = 0;
347 XDWORD filepos = 0;
348
349 XFILEXDB* xfilexdb = new XFILEXDB( &hashmd5, xpath);
350 if(xfilexdb)
351 {
352 xfilexdb->Set(XFILE_XDB_ID, XFILEXDB_TYPE, DIOMACMANUFACTURED_FILE_VERSION, DIOMACMANUFACTURED_FILE_IDSTRING);
353
354 if(xfilexdb->Create(xpath))
355 {
356 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&nrecords, sizeof(XDWORD));
357 xfilexdb->GetPrimaryFile()->GetPosition(initablepos);
358
359 XDWORD ID = 0;
360
361 for(c=0;c<nrecords;c++)
362 {
363 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&ID, sizeof(XDWORD));
364 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&ID, sizeof(XDWORD));
365 }
366
367 for(c=0;c<nrecords;c++)
368 {
369 size_t _name = (size_t)idmap.GetElement(c);
370 XSTRING* name = (XSTRING*)_name;
371 if(name)
372 {
373 ID = (XDWORD)idmap.GetKey(c);
374
375 xfilexdb->GetPrimaryFile()->GetPosition(filepos);
376
377 XDWORD size = (name->GetSize()+1)*sizeof(XWORD);
378
379 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&size , sizeof(XDWORD));
380
381 XBUFFER xbufferexchange;
382
383 (*name)->ConvertToBufferExchange(xbufferexchange, true);
384 xfilexdb->GetPrimaryFile()->Write(xbufferexchange->Get(), size);
385
386 idmapindex.Add(ID,filepos);
387
388 delete name;
389 }
390 }
391
392 xfilexdb->GetPrimaryFile()->SetPosition(initablepos-sizeof(XDWORD));
393 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&nrecords,sizeof(XDWORD));
394
395 for(c=0;c<nrecords;c++)
396 {
397 ID = (XDWORD)idmapindex.GetKey(c);
398 filepos = (XDWORD)idmapindex.GetElement(c);
399
400 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&ID , sizeof(XDWORD));
401 xfilexdb->GetPrimaryFile()->Write((XBYTE*)&filepos, sizeof(XDWORD));
402 }
403
404 xfilexdb->CloseFile();
405 }
406 }
407
408 idmap.DeleteAll();
409 idmapindex.DeleteAll();
410
411 return true;
412}
