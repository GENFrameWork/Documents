100void ANDROIDJNI::InitializeJNI(android_app* application, jint javaversion)
101{
102 currentjavaVM = application->activity->vm;
103 currentjavaversion = javaversion;
104
105 JNIEnv* env = ANDROIDJNI::GetJNIEnv();
106 if(!env)
107 {
108 return;
109 }
110
111 jobject mainactivity = application->activity->clazz;
112
113 jclass MainClass = env->GetObjectClass(mainactivity);
114 jclass classClass = env->FindClass("java/lang/Class");
115 jclass classLoaderClass = env->FindClass("java/lang/ClassLoader");
116 jmethodID getClassLoaderMethod = env->GetMethodID(classClass, "getClassLoader", "()Ljava/lang/ClassLoader;");
117 jobject classLoader = env->CallObjectMethod(MainClass, getClassLoaderMethod);
118
119 ClassLoader = env->NewGlobalRef(classLoader);
120
121 FindClassMethod = env->GetMethodID(classLoaderClass, "findClass", "(Ljava/lang/String;)Ljava/lang/Class;");
122}

134JNIEnv* ANDROIDJNI::GetJNIEnv()
135{
136 static XDWORD threadslot = 0;
137 if(!threadslot)
138 {
139 pthread_key_create((pthread_key_t*)&threadslot, &JavaEnvDestructor);
140 }
141
142 JNIEnv* env = NULL;
143
144 if(!currentjavaVM)
145 {
146 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[ANDROID] No Current Java VM"));
147 }
148
149 jint getenvresult = currentjavaVM->GetEnv((void**)&env, currentjavaversion);
150 if(getenvresult == JNI_EDETACHED)
151 {
152 jint attachresult = currentjavaVM->AttachCurrentThread(&env, NULL);
153
154 if(attachresult == JNI_ERR)
155 {
156 return NULL;
157 }
158 }
159 else
160 {
161 if(getenvresult != JNI_OK)
162 {
163 return NULL;
164 }
165 }
166
167 return env;
168}
