127bool APPFLOWGRAPHICS::Ini(XVECTOR<XSTRING*>* execparams)
128{
129 RegisterEvent(GRPXEVENT_TYPE_SCREEN_CREATING);
130 RegisterEvent(GRPXEVENT_TYPE_SCREEN_CANVASCREATING);
131 RegisterEvent(GRPXEVENT_TYPE_SCREEN_CREATED);
132 RegisterEvent(GRPXEVENT_TYPE_SCREEN_DELETING);
133 RegisterEvent(GRPXEVENT_TYPE_SCREEN_DELETED);
134
135 if(!APPFLOWCONSOLE::Ini(execparams))
136 {
137 return false;
138 }
139
140 if(GetExitType() != APPFLOWBASE_EXITTYPE_UNKNOWN)
141 {
142 return false;
143 }
144
145 if(initoptions & APPFLOWGRAPHICS_INIOPTION_CREATEMAINSCREEN)
146 {
147 if(!CreateMainScreenProcess((initoptions & APPFLOWGRAPHICS_INIOPTION_SHOWMAINSCREEN)?true:false)) return false;
148 }
149
150 return true;
151}

228bool APPFLOWGRAPHICS::CreateMainScreenProcess(bool show)
229{
230 if(mainscreen) return false;
231
232 mainscreen = GEN_GRPFACTORY.CreateScreen();
233 if(!mainscreen)
234 {
235 // XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[APP Graphics] Could not initialize the main screen ..."));
236 return false;
237 }
238
239 #ifdef GRP_OPENGL_ACTIVE
240 maincontext = GRPFACTORY::GetInstance().CreateContext();
241 if(!maincontext)
242 {
243 // XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[APP Graphics] Could not initialize the Graphics Context ..."));
244 return false;
245 }
246 #endif
247
248 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATING , this);
249 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_CANVASCREATING , this);
250 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
251 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETING , this);
252 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETED , this);
253
254 #ifndef ANDROID
255
256 GRPXEVENT grpeventini(this, GRPXEVENT_TYPE_SCREEN_CREATING);
257 grpeventini.SetScreen(mainscreen);
258 PostEvent(&grpeventini);
259
260 bool status = !grpeventini.GetError();
261
262 if(status)
263 {
264 status = mainscreen->Create(show);
265 if(status)
266 {
267 GRPXEVENT grpeventini(this, GRPXEVENT_TYPE_SCREEN_CANVASCREATING);
268 grpeventini.SetScreen(mainscreen);
269 PostEvent(&grpeventini);
270
271 GRPVIEWPORT* viewport = mainscreen->GetViewport(0);
272 if(!viewport)
273 {
274 if(!mainscreen->CreateViewport(GRPVIEWPORT_ID_MAIN , 0.0f, 0.0f, (float)mainscreen->GetWidth() , (float)mainscreen->GetHeight(), 0, 0, mainscreen->GetWidth(), mainscreen->GetHeight()))
275 {
276 return false;
277 }
278 }
279 }
280 }
281
282 if(status)
283 {
284 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[App Graphics] Screen created"));
285
286 GRPXEVENT grpeventend(this, GRPXEVENT_TYPE_SCREEN_CREATED);
287 grpeventend.SetScreen(mainscreen);
288 grpeventend.SetError(status);
289 PostEvent(&grpeventend);
290 }
291 else
292 {
293 //XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[App Graphics] Error to creating screen..."));
294 }
295
296 if(!status) return false;
297
298 #ifdef INP_ACTIVE
299 if(initoptions & APPFLOWGRAPHICS_INIOPTION_INPUT)
300 {
301 keyboard = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_KEYBOARD);
302 if(!keyboard)
303 {
304 keyboard = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_KEYBOARD, mainscreen);
305 if(keyboard) GEN_INPMANAGER.AddDevice(keyboard);
306 }
307
308 mouse = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_MOUSE);
309 if(!mouse)
310 {
311 mouse = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_MOUSE, mainscreen);
312 if(mouse) GEN_INPMANAGER.AddDevice(mouse);
313 }
314
315 touchscreen = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_TOUCHSCREEN);
316 if(!touchscreen)
317 {
318 touchscreen = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_TOUCHSCREEN, mainscreen);
319 if(touchscreen) GEN_INPMANAGER.AddDevice(touchscreen);
320 }
321
322 }
323 #endif
324
325 #endif
326
327 return true;
328}

340bool APPFLOWGRAPHICS::DeleteMainScreenProcess()
341{
342 if(!mainscreen)
343 {
344 return false;
345 }
346
347 GRPXEVENT grpeventini(this, GRPXEVENT_TYPE_SCREEN_DELETING);
348 grpeventini.SetScreen(mainscreen);
349 PostEvent(&grpeventini);
350
351 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATING , this);
352 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CANVASCREATING , this);
353 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
354
355 if(mainscreen)
356 {
357 mainscreen->Delete();
358 GRPFACTORY::GetInstance().DeleteScreen(mainscreen);
359 mainscreen = NULL;
360 }
361
362 GRPXEVENT grpeventend(this, GRPXEVENT_TYPE_SCREEN_DELETED);
363 grpeventini.SetScreen(mainscreen);
364 PostEvent(&grpeventend);
365
366 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETED , this);
367 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
368
369 return true;
370}
