115 if(file->Read(header, GRPBITMAPFILEPNG_HEADERSIZE))
116 {

118 {
119 png = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
120 if(png)
121 {
122 info = png_create_info_struct(png);
123 if(info)
124 {
125 if(!setjmp(png_jmpbuf(png)))

151 png_destroy_read_struct(&png, &info, NULL);
152

154
155 switch(colortype)
156 {
157 case PNG_COLOR_TYPE_RGB : mode = GRPPROPERTYMODE_24_RGB_888;
158 case PNG_COLOR_TYPE_RGB_ALPHA : mode = GRPPROPERTYMODE_32_RGBA_8888;
159 }
160
161 bitmap = GRPFACTORY::GetInstance().CreateBitmap(width, height, mode);
162 if(bitmap)
163 {
164 int index = 0;
165
166 for(int y=0; y<(int)height; y++)
167 {
168 index = 0;
169
170 for(int x=0; x<(int)width; x++)
171 {
172 GRP2DCOLOR_RGBA8 color;
173
174 color.r = (XBYTE)(*(XBYTE*)(databuffer[y] + index + 0));
175 color.g = (XBYTE)(*(XBYTE*)(databuffer[y] + index + 1));
176 color.b = (XBYTE)(*(XBYTE*)(databuffer[y] + index + 2));
177
178 if(colortype == PNG_COLOR_TYPE_RGB)
179 {
180 color.a = 255;
181 }
182 else
183 {
184 color.a = (XBYTE)(*(XBYTE*)(databuffer[y] + index + 3));
185 }
186
187 bitmap->PutPixel(x, y, (GRP2DCOLOR*)&color);
188
189 index += bytesppixel;
190 }
191 }
192
193 }
194
195 for(int y=0; y<(int)height; y++)
196 {
197 delete [] databuffer[y];
198 }
199
200 delete [] databuffer;
201 }
202 }
203 }
204 }
205 }
206 }

208 Close();

210 return ForceModeOfBitmap(forcemode, bitmap);

359bool GRPBITMAPFILEPNG::CreateFileFromBitmap(XPATH& xpath, GRPBITMAP* bitmap, int quality)
360{
361 if(!bitmap) return false;

363 if(!Create(xpath)) return false;
364
365 png_structp png;
366 png_infop info;
367 png_bytep* databuffer = NULL;
368 png_uint_32 width = bitmap->GetWidth();
369 png_uint_32 height = bitmap->GetHeight();
370 png_byte bitdepth = 8;
371 int bytesppixel = 4;
372 int rowbytes = 0;
373 bool status = true;
374
375 png = png_create_write_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
376 if(png)
377 {
378 info = png_create_info_struct(png);
379 if(info)
380 {
381 if(!setjmp(png_jmpbuf(png)))
382 {
383 png_set_write_fn(png , (void *)this, GRPBITMAPFILEPNG::WriteFileData, NULL);
384
385 if(!setjmp(png_jmpbuf(png)))
386 {
387 rowbytes = width * 4;
388
389 databuffer = (png_bytep*)new XBYTE[sizeof(png_bytep) * height];
390 for(int y=0; y<(int)height; y++)
391 {
392 databuffer[y] = (png_byte*)new XBYTE[rowbytes];
393 }
394
395 png_set_IHDR(png, info, width ,
396 height ,
397 bitdepth ,
398 PNG_COLOR_TYPE_RGB_ALPHA ,
399 PNG_INTERLACE_NONE ,
400 PNG_COMPRESSION_TYPE_BASE ,
401 PNG_FILTER_TYPE_BASE);
402
403 png_write_info(png, info);
404
405 if(!setjmp(png_jmpbuf(png)))
406 {
407 int index = 0;
408
409 for(int y=0; y<(int)height; y++)
410 {
411 index = 0;
412

416 if(color)

418 XBYTE* ptr = (XBYTE*)(databuffer[y] + index);
419

426 index += bytesppixel;

428 }

430 png_write_image(png, databuffer);
431
432 if(!setjmp(png_jmpbuf(png)))

436 }
