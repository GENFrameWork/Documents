339bool DIOSNMP_BASE::Open(XCHAR* IPtarget, int port, bool checkavailableNMS, XCHAR* localIP)
340{
341 if(!IPtarget) return false;
342 if(!IPtarget[0]) return false;
343
344 diostreamudpcfg = new DIOSTREAMUDPCONFIG();
345 if(!diostreamudpcfg) return false;
346
347 diostreamudpcfg->SetMode(DIOSTREAMMODE_CLIENT);
348
349 if(localIP) diostreamudpcfg->GetLocalIP()->Set(localIP);
350
351 diostreamudpcfg->GetRemoteURL()->Set(IPtarget);
352 diostreamudpcfg->SetRemotePort(port);
353
354 bool status = false;
355
356 diostreamudp = (DIOSTREAMUDP*)GEN_DIOFACTORY.CreateStreamIO(diostreamudpcfg);
357 if(diostreamudp)
358 {
359 if(diostreamudp->Open())
360 {
361 if(checkavailableNMS)
362 {
363 //--------------------------------------------
364 // Write something to received a ICMP result.
365 // and check if something is the other side.
366
367 status = diostreamudp->WaitToConnected(3);
368 if(status)
369 {
370 XBYTE data[] = { 0xAA,0x55 };
371 diostreamudp ->Write(data,sizeof(data));
372
373 status = diostreamudp->WaitToDisconnected(3);
374 }
375
376 } else status = true;
377 }
378 }
379
380 if(!status) Close();
381
382 return status;
383}

539 if(OCTETstring)
540 {
541 xber.SetOCTETSTRING(OCTETstring);

643bool DIOSNMP::Get(DIOSNMP_OPERATION operation, XCHAR* community, XCHAR* OIDstr, DIOSNMP_XBER& xberout)
644{
645 #define MAX_READBUFFER 2048
646
647 if(!diostreamudpcfg) return false;
648 if(!diostreamudp) return false;
649
650 XBUFFER xbuffer;
651 XBUFFER xbufferdata;
652 XDWORD number = 0;
653 DIOSNMP_XBER xber;
654 DIOSNMP_XBER value;
655 DIOSNMP_XBER bindings;
656 DIOSNMP_XBER request;
657 DIOSNMP_XBER data_next_request;
658 DIOSNMP_XBER header;
659
660 // ----------------------------------------------------------------
661 // Get-Next
662
663 xber.SetOID(OIDstr); value.Sequence_AddTo(xber); // OID
664 xber.SetNULL(); value.Sequence_AddTo(xber); // NULL
665
666 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
667 if(xrand)
668 {
669 number = xrand->Max(65000);
670 GEN_XFACTORY.DeleteRand(xrand);
671 }
672
673 xber.SetINTEGER(number); bindings.Sequence_AddTo(xber); // Request-ID
674 xber.SetINTEGER(0); bindings.Sequence_AddTo(xber); // Error-status
675 xber.SetINTEGER(0); bindings.Sequence_AddTo(xber); // Error-index
676
677 request.Sequence_AddTo(value);
678 bindings.Sequence_AddTo(request);
679
680 data_next_request.Sequence_AddCommandTo(DIOSNMP_OPERATION_GETNEXTREQUEST, bindings);
681
682 xber.SetINTEGER(version); header.Sequence_AddTo(xber); // Version
683 xber.SetOCTETSTRING(community); header.Sequence_AddTo(xber); // Community
684
685 header.Sequence_AddTo(data_next_request);
686
687 header.GetDump(xbuffer);
688
689 XDWORD bw = diostreamudp->Write(xbuffer.Get(), xbuffer.GetSize());
690 if(bw!= xbuffer.GetSize()) return false;
691
692 xbuffer.Empty();
693 xbuffer.Resize(MAX_READBUFFER);
694
695 XTIMER* GEN_XFACTORY_CREATE(timeout, CreateTimer())
696 if(!timeout) return false;
697
698 do{ XDWORD sizeread = diostreamudp->GetInXBuffer()->GetSize();
699 if(sizeread)
700 {
701 XDWORD br = diostreamudp->Read(xbuffer.Get(), sizeread);
702 if(br)
703 {
704 timeout->Reset();
705 xbufferdata.Add(xbuffer.Get(), br);
706 }
707 }
708
709 if(timeout->GetMeasureSeconds() > 2) break;
710
711 } while(1);
712
713 GEN_XFACTORY.DeleteTimer(timeout);
714
715 if(xbufferdata.GetSize())
716 {
717 xberout.SetTagType(xbufferdata.Get()[0]);
718 xberout.SetSize(xbufferdata.Get()[1]);
719 xberout.GetData()->Add(&xbufferdata.Get()[2], xbufferdata.GetSize());
720 }
721
722 return xbufferdata.GetSize()?true:false;
723}
