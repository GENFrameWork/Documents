116bool APPBASE::Ini(XVECTOR<XSTRING*>* execparams)
117{
118 this->execparams = execparams;
119
120 GEN_XFACTORY_CREATE(xtimerglobal, CreateTimer())
121 if(!xtimerglobal) return false;
122
123 xtimerglobal->Reset();
124
125 firstupdate = false;
126
127 XDWORD code = GEN_XSYSTEM.GetLanguageSO();
128
129 if(code != XLANGUAGE_ISO_639_3_CODE_INVALID) XTRANSLATION_GEN::GetInstance().Sentences_AddToTranslation(code);
130
131 if(!AppProc_Ini()) return false;
132
133 return true;
134}

169bool APPBASE::End()
170{
171 bool status;
172
173 status = AppProc_LastUpdate();
174 if(status) status = AppProc_End();
175
176 GEN_XPUBLISHER.DeRegisterAllEvents();
177
178 if(xtimerglobal)
179 {
180 GEN_XFACTORY.DeleteTimer(xtimerglobal);
181 xtimerglobal = NULL;
182 }
183
184 applicationname.Empty();
185 applicationexecutable.Empty();
186 applicationID.Empty();
187
188 return status;

445void APPBASE::SetSystemChangeStatus(XSYSTEM_CHANGESTATUSTYPE systemchangestatustype)
446{
447 if(this->systemchangestatustype == systemchangestatustype)
448 {
449 return;
450 }
451
452 this->systemchangestatustype = systemchangestatustype;
453
454
455 XSTRING statusstr;
456
457 switch(systemchangestatustype)
458 {
459 case XSYSTEM_CHANGESTATUSTYPE_NONE :
460 default : statusstr = __L("normal"); break;
461 case XSYSTEM_CHANGESTATUSTYPE_REBOOT : statusstr = __L("reboot"); break;
462 case XSYSTEM_CHANGESTATUSTYPE_CONSOLE_CONNECT : statusstr = __L("console connect"); break;
463 case XSYSTEM_CHANGESTATUSTYPE_CONSOLE_DISCONNECT : statusstr = __L("console disconnect"); break;
464 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGON : statusstr = __L("session logon"); break;
465 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGOFF : statusstr = __L("session logooff"); break;
466 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOCK : statusstr = __L("session lock"); break;
467 case XSYSTEM_CHANGESTATUSTYPE_SESSION_UNLOCK : statusstr = __L("session unlock"); break;
468 case XSYSTEM_CHANGESTATUSTYPE_POWEROFF : statusstr = __L("power off"); break;
469 case XSYSTEM_CHANGESTATUSTYPE_SUSPEND : statusstr = __L("suspend"); break;
470 }
471
472 //XTRACE_PRINTCOLOR(XTRACE_COLOR_GREEN, __L("[System] Mode change to [%s]"), statusstr.Get());
473
474 APP_XEVENT xevent(this, APP_XEVENT_TYPE_CHANGESTATUSTYPE);
475 xevent.SetChangeStatusType(systemchangestatustype);
476 PostEvent(&xevent);
477}
