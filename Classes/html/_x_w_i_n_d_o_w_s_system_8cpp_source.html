139double XWINDOWSSYSTEM_CPUUSAGE::GetTotalCpuUsage()
140{
141 #ifndef BUILDER
142 FILETIME idletime;
143 FILETIME kerneltime;
144 FILETIME usertime;
145
146 if(GetSystemTimes(&idletime, &kerneltime, &usertime) == 0)
147 {
148 return -1;
149 }
150
151 ULONGLONG sysidlediff = SubtractTimes(idletime , lastsysidletime);
152 ULONGLONG syskerneldiff = SubtractTimes(kerneltime , lastsyskerneltime);
153 ULONGLONG sysuserdiff = SubtractTimes(usertime , lastsysusertime);
154
155 ULONGLONG systotal = syskerneldiff + sysuserdiff;
156 ULONGLONG totalsystime = systotal + sysidlediff;
157
158 lastsysidletime = idletime;
159 lastsyskerneltime = kerneltime;
160 lastsysusertime = usertime;
161
162 if(totalsystime == 0)
163 {
164 return -1;
165 }
166
167 return (systotal * 100.0) / totalsystime;
168
169 #else
170 return 0.0f;
171 #endif
172}

186double XWINDOWSSYSTEM_CPUUSAGE::GetProcessCpuUsage(DWORD processID)
187{
188 #ifndef BUILDER
189 HANDLE hprocess = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, processID);
190 if(hprocess == NULL)
191 {
192 return -1;
193 }
194
195 FILETIME proccreationtime;
196 FILETIME procexittime;
197 FILETIME prockerneltime;
198 FILETIME procusertime;
199
200 if(GetProcessTimes(hprocess, &proccreationtime, &procexittime, &prockerneltime, &procusertime) == 0)
201 {
202 CloseHandle(hprocess);
203 return -1;
204 }
205
206 ULONGLONG prockerneldiff = SubtractTimes(prockerneltime, lastprockerneltime);
207 ULONGLONG procuserdiff = SubtractTimes(procusertime, lastprocusertime);
208
209 lastprockerneltime = prockerneltime;
210 lastprocusertime = procusertime;
211
212 CloseHandle(hprocess);
213
214 if(!lastsystemtime)
215 {
216 lastsystemtime = GetTickCount64();
217 lastprocesstime = prockerneldiff + procuserdiff;
218
219 return -1;
220 }
221
222 ULONGLONG currenttime = GetTickCount64();
223 ULONGLONG timediff = currenttime - lastsystemtime;
224 ULONGLONG processtime = prockerneldiff + procuserdiff;
225 ULONGLONG processtimediff = processtime - lastprocesstime;
226
227 lastsystemtime = currenttime;
228 lastprocesstime = processtime;
229
230 return (processtimediff * 100.0) / (timediff * GetNumberOfProcessors());
231 #else
232 return 0.0f;
233 #endif
234}

635 if(!cus) return 0;
636

638
639 newperftime100nsec = perfdata->PerfTime100nSec;
640
641 if(cus->firsttime)
642 {
643 cus->firsttime = false;
644 cus->oldvalue = newvalue;
645 cus->oldperftime100nsec = newperftime100nsec;
646
647 return 0;
648 }
649
650 LONGLONG lnValueDelta = newvalue - cus->oldvalue;
651 double DeltaPerfTime100nSec = (double)newperftime100nsec.QuadPart - (double)cus->oldperftime100nsec.QuadPart;
652
653 cus->oldvalue = newvalue;
654 cus->oldperftime100nsec = newperftime100nsec;
655
656 double a = (double)lnValueDelta / DeltaPerfTime100nSec;

658 double f = (1.0 - a) * 100.0;

679int XWINDOWSSYSTEM::GetCPUUsageForProcessName(XCHAR* processname)
680{
681 int cpuusage = 0;
682
683 #ifndef BUILDER
684 HANDLE hprocesssnap;
685 PROCESSENTRY32 pe32;
686
687 // Take a snapshot of all processes in the system.
688 hprocesssnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
689 if(hprocesssnap == INVALID_HANDLE_VALUE ) return false;
690
691 // Set the size of the structure before using it.
692 memset(&pe32, 0, sizeof(PROCESSENTRY32));
693 pe32.dwSize = sizeof(PROCESSENTRY32);
694
695 // Retrieve information about the first process, and exit if unsuccessful
696 if(!Process32First(hprocesssnap, &pe32))
697 {
698 CloseHandle(hprocesssnap); // clean the snapshot object
699 return false;
700 }
701
702 // Now walk the snapshot of processes
703 do{ XSTRING nametask(pe32.szExeFile);
704
705 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("Process %s : [%s]"), processname, nametask.Get());
706
707 if(!nametask.Compare(processname, true))
708 {
709 cpuusage = GetCPUUsageForProcessID((XDWORD)pe32.th32ProcessID);
710 }
711
712 } while(Process32Next(hprocesssnap, &pe32));
713
714 CloseHandle(hprocesssnap);
715
716 #endif
717
718 return cpuusage;

980bool XWINDOWSSYSTEM::ShutDown(XSYSTEM_CHANGESTATUSTYPE type)
981{
982 if(type == XSYSTEM_CHANGESTATUSTYPE_NONE) return false;
983
984 HANDLE hToken;
985 TOKEN_PRIVILEGES tkp;
986
987 // Get a token for this process.
988 if(!OpenProcessToken(GetCurrentProcess(), TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY, &hToken)) return false;
989
990 // Get the LUID for the shutdown privilege.
991 LookupPrivilegeValue(NULL, SE_SHUTDOWN_NAME, &tkp.Privileges[0].Luid);
992
993 tkp.PrivilegeCount = 1; // one privilege to set
994 tkp.Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;
995
996 // Get the shutdown privilege for this process.
997 AdjustTokenPrivileges(hToken, FALSE, &tkp, 0, (PTOKEN_PRIVILEGES)NULL, 0);
998 if(GetLastError() != ERROR_SUCCESS) return false;
999
1000 switch(type)
1001 {
1002 case XSYSTEM_CHANGESTATUSTYPE_REBOOT : if(!ExitWindowsEx(EWX_REBOOT | EWX_FORCE, SHTDN_REASON_MAJOR_OPERATINGSYSTEM | SHTDN_REASON_MINOR_UPGRADE | SHTDN_REASON_FLAG_PLANNED)) return false;
1003 break;
1004
1005 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGOFF : if(!ExitWindowsEx(EWX_LOGOFF | EWX_FORCE, SHTDN_REASON_MAJOR_OPERATINGSYSTEM | SHTDN_REASON_MINOR_UPGRADE | SHTDN_REASON_FLAG_PLANNED)) return false;
1006 break;
1007
1008 case XSYSTEM_CHANGESTATUSTYPE_POWEROFF : if(!ExitWindowsEx(EWX_POWEROFF | EWX_FORCE, SHTDN_REASON_MAJOR_OPERATINGSYSTEM | SHTDN_REASON_MINOR_UPGRADE | SHTDN_REASON_FLAG_PLANNED)) return false;
1009 break;
1010
1011 case XSYSTEM_CHANGESTATUSTYPE_SUSPEND :
1012 #ifndef BUILDER
1013 if(!SetSuspendState(TRUE, FALSE, FALSE)) return false;
1014 #endif
1015 break;
1016
1017 default : return false;
1018
1019
1020 }
1021
1022 return true;
1023}
