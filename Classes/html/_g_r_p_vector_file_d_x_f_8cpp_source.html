115GRPVECTORFILERESULT GRPVECTORFILEDXF::DetectType()
116{
117 GRPVECTORFILERESULT result;
118
119 if(pathfile.IsEmpty())
120 {
121 type = GRPVECTORFILETYPE_UNKNOWN;
122
123 return GRPVECTORFILERESULT_ERRORNOTFILE;
124 }
125
126 XSTRING extension;
127 pathfile.GetExt(extension);
128
129 if(extension.IsEmpty())
130 {
131 type = GRPVECTORFILETYPE_UNKNOWN;
132
133 return GRPVECTORFILERESULT_ERRORNOTFILE;
134 }
135
136 if(!extension.Compare(GRPVECTORFILEDXF_EXTENSION, true))
137 {
138 type = GRPVECTORFILETYPE_DXF;
139 }
140
141 if(DetectFileFormatText(pathfile))
142 {
143 XFILETXT* file = new XFILETXT();
144 if(file)
145 {
146 if(file->Open(pathfile))
147 {
148 int nlines = 100;
149
150 file->ReadNLines(nlines);
151
152 nlines = file->GetNLines();
153
154 if(ParserHaveAnySection(file, nlines))
155 {
156 result = GRPVECTORFILERESULT_OK;
157 }
158 else
159 {
160 result = GRPVECTORFILERESULT_ERRORINVALIDFORMAT;
161 }
162
163 file->Close();
164 }
165 else
166 {
167 result = GRPVECTORFILERESULT_ERRORNOTFILE;
168 }
169
170 delete file;
171 }
172 else
173 {
174 result = GRPVECTORFILERESULT_ERRORNOTMEMORY;
175 }
176 }
177 else
178 {
179 result = GRPVECTORFILERESULT_ERRORINVALIDFORMAT;
180 }
181
182 return result;
183}

195GRPVECTORFILERESULT GRPVECTORFILEDXF::Load()
196{
197 GRPVECTORFILEDXFRESULT result = GRPVECTORFILERESULT_ERRORNOTMEMORY;
198
199 XFILETXT* file = new XFILETXT();
200 if(!file)
201 {
202 return GRPVECTORFILERESULT_ERRORNOTMEMORY;
203 }
204
205 if(file->Open(pathfile))
206 {
207 XTIMER* xtimer = GEN_XFACTORY.CreateTimer();
208 if(xtimer)
209 {
210 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("Reading TXT file ..."));
211
212 xtimer->Reset();
213
214 file->ReadAllFile();
215
216 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("Read TXT file: %ld Milliseconds."), xtimer->GetMeasureMilliSeconds());
217
218 GEN_XFACTORY.DeleteTimer(xtimer);
219 }
220
221 result = ParserTextFile (file);
222
223 file->Close();
224 }
225 else
226 {
227 result = GRPVECTORFILERESULT_ERRORNOTFILE;
228 }
229
230 delete file;
231
232 return result.TakeAsBase();

334GRPVECTORFILEDXFRESULT GRPVECTORFILEDXF::ParserTextSections(XFILETXT* fileTXT, XVECTOR<GRPVECTORFILEDXFTEXTSECTION*>& sections)
335{
336 GRPVECTORFILEDXFTEXTSECTION* section = NULL;
337 GRPVECTORFILEDXFRESULT result = GRPVECTORFILERESULT_ERRORUNKNOWN;
338 int c = 0;
339
340 do{ if(!section)
341 {
342 section = new GRPVECTORFILEDXFTEXTSECTION ();
343 }
344
345 XSTRING* line = fileTXT->GetLine(c);
346 ParserTextFilePrepareLine(line);
347
348 if(line && !line->Compare(__L("0"),true))
349 {
350 line = fileTXT->GetLine(c+1);
351 ParserTextFilePrepareLine(line);
352
353 if(line && !line->Compare(__L("SECTION"), false))
354 {
355 line = fileTXT->GetLine(c+2);
356 ParserTextFilePrepareLine(line);
357 if(line && !line->Compare(__L("2"),true))
358 {
359 c += 3;
360
361 line = fileTXT->GetLine(c);
362 ParserTextFilePrepareLine(line);
363
364 section->iniline = c+1;
365 section->name = line->Get();
366
367 c++;
368 }
369 }
370
371 if(line && !line->Compare(__L("ENDSEC"), false))
372 {
373 section->endline = c-1;
374 section->type = GRPVECTORFILEDXFTEXTSECTION::GetTypeSection(section->name);
375
376 #ifdef XTRACE_ACTIVE
377 if(section->type == GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_UNKNOWN)
378 {
379 XSTRING message;
380
381 message.Format(__L("section %s Unknown"), section->name.Get());
382
383 GRPVECTORFILE_XEVENT vfEvent(this, GRPVECTORFILE_XEVENTTYPE_PARTUNKNOWN);
384
385 vfEvent.SetType(GRPVECTORFILETYPE_DXF);
386 vfEvent.GetPath()->Set(fileTXT->GetPrimaryFile()->GetPathNameFile());
387 vfEvent.GetMsg()->Set(message);
388
389 PostEvent(&vfEvent);
390 }
391 #endif
392
393 GRPVECTORFILEDXFTEXTSECTION* sectionType = GRPVECTORFILEDXFTEXTSECTION::CreateInstance(section->type);
394 if(sectionType)
395 {
396 section->CopyTo(sectionType);
397 sectionType->SetGrpVectorFile(this);
398 sections.Add(sectionType);
399
400 delete section;
401 }
402 else
403 {
404 sections.Add(section);
405 }
406
407 section = NULL;
408 }
409
410 if(line && !line->Compare(__L("EOF"), false))
411 {
412 c++;
413 break;
414 }
415 }
416
417 c++;
418
419 } while(c < fileTXT->GetNLines());
420
421 if(section && (section->type == GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_UNKNOWN))
422 {
423 delete section;
424 }
425
426 if(!sections.IsEmpty())
427 {
428 result = GRPVECTORFILERESULT_OK;
429 }
430 else
431 {
432 result = GRPVECTORFILEDXFRESULT_NOTSECTIONS;
433 }
434
435 return result;
436}

503GRPVECTORFILEDXFRESULT GRPVECTORFILEDXF::ParserTextFile(XFILETXT* fileTXT)
504{
505 GRPVECTORFILEDXFRESULT result = GRPVECTORFILERESULT_ERRORUNKNOWN;
506
507 sections.DeleteContents();
508 sections.DeleteAll();
509
510 result = ParserTextSections (fileTXT, sections);
511 if(result == (GRPVECTORFILEDXFRESULT)GRPVECTORFILERESULT_OK)
512 {
513 for(XDWORD c=0; c<sections.GetSize(); c++)
514 {
515 GRPVECTORFILEDXFTEXTSECTION* section = sections.Get(c);
516 if(section)
517 {
518 result = section->ParserTextSection(fileTXT);
519 if(result != (GRPVECTORFILEDXFRESULT)GRPVECTORFILERESULT_OK)
520 {
521 break;
522 }
523 }
524 }
525
526 if(!sections.GetSize())
527 {
528 result = GRPVECTORFILEDXFRESULT_NOTSECTIONS;
529 }
530 }
531
532 GRPVECTORFILEDXFTEXTSECTIONHEADER* sectionheader = (GRPVECTORFILEDXFTEXTSECTIONHEADER*)GetSection(GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_HEADER);
533 if(sectionheader)
534 {
535 config.SetHeader(sectionheader);
536 }
537
538 #ifdef XTRACE_ACTIVE
539 ShowTraceAllSections();
540 #endif
541
542 return result;
543}

594bool GRPVECTORFILEDXF::ShowTraceAllSections()
595{
596 for(XDWORD c=0; c<sections.GetSize(); c++)
597 {
598 GRPVECTORFILEDXFTEXTSECTION* section = sections.Get(c);
599 if(section)
600 {
601 XSTRING line;
602
603 line.Format(__L("[Vector file DXF load] Section [%d, %d]: %s"), section->iniline, section->endline, section->name.Get());
604
605 switch(section->type)
606 {
607 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_UNKNOWN :
608 default : break;
609
610 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_HEADER : { GRPVECTORFILEDXFTEXTSECTIONHEADER* sectionheader = (GRPVECTORFILEDXFTEXTSECTIONHEADER*)GetSection (section->type);
611 if(sectionheader)
612 {
613 line.AddFormat(__L(" -> NVariables (%d)"), sectionheader->GetVariables()->GetSize());
614 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, line.Get());
615 }
616 }
617 break;
618
619 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_CLASSES :
620 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_TABLES : break;
621
622 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_BLOCKS : { GRPVECTORFILEDXFTEXTSECTIONBLOCKS* sectionblocks = (GRPVECTORFILEDXFTEXTSECTIONBLOCKS*)GetSection(section->type);
623 if(sectionblocks)
624 {
625 line.AddFormat(__L(" -> (%d)"), sectionblocks->GetBlocks()->GetSize());
626 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, line.Get());
627
628 for(XDWORD d=0; d<sectionblocks->GetBlocks()->GetSize(); d++)
629 {
630 GRPVECTORFILEDXFTEXTBLOCK* block = sectionblocks->GetBlocks()->Get(d);
631 if(block)
632 {
633 if(!block->IsEndBlock())

635 GRPVECTORFILEDXFTEXTSECTIONENTITIES* sectionentities = block->GetSectionEntities();
636 if(sectionentities)

638 line.Format(__L("[Vector file DXF load] Block (%2d) [%s] ENTITIES "), d+1, block->GetName()->Get());
639 GenerateSectionsEntities(sectionentities,line);
640 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, line.Get());
641 }
642 }
643 }
644 }
645 }
646 }
647 break;
648
649 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_ENTITIES : { GRPVECTORFILEDXFTEXTSECTIONENTITIES* sectionentities = (GRPVECTORFILEDXFTEXTSECTIONENTITIES*)GetSection (section->type);
650 if(sectionentities)
651 {
652 GenerateSectionsEntities(sectionentities,line);
653 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, line.Get());
654 }
655 }
656 break;
657 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_OBJECTS :
658 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_ACDSDATA :
659 case GRPVECTORFILEDXFTEXTSECTION_TYPESECTION_THUMBNAILIMAGE : break;

661 }
662 }
663
664 return true;
