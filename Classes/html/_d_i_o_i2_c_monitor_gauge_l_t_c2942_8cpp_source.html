114float DIOI2CMONITORGAUGELTC2942::GetVoltage (void)
115{
116 if(!diostream) return false;
117
118 float voltage_return = 0;
119 XBYTE buffer[2] = { 0, 0 };
120
121 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_VOL_H, buffer, 2)) return 0;
122
123 XDWORD read_value = buffer[1] | (buffer[0] << 8);
124
125 if(read_value == DIOI2CMONITORGAUGELTC2942_VOLTAGE_READ_ERROR)
126 {
127 return 0;
128 }
129
130 voltage_return = (float)read_value;
131
132 return ((voltage_return* 6000) / 65535)/1000;
133}

147bool DIOI2CMONITORGAUGELTC2942::SetAC(XWORD ac)
148{
149 XBYTE ctrl;
150
151 // Backup control register
152 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL,ctrl)) return false;
153
154 // Shutdown analog section before programming new AC value
155 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl | DIOI2CMONITORGAUGELTC2942_CTL_SHUTDOWN)) return false;
156
157 // Program new AC value
158 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_AC_H, ac >> 8)!=0) return false;
159
160 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_AC_L, ac & 0xFF)!=0) return false;
161
162 // Restore control register
163 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl)!=0) return false;
164
165 return true;
166}

259bool DIOI2CMONITORGAUGELTC2942::Initialize()
260{
261 XBYTE ctrl;
262
263 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl)) return false;
264
265 // Enable auto measurement of battery voltage and temperature
266 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_ADC_MSK; // clear ADC mode bits
267 ctrl |= DIOI2CMONITORGAUGELTC2942_ADC_AUTO; // configure mode
268
269 // Enable analog section of the chip
270 ctrl &= ~DIOI2CMONITORGAUGELTC2942_CTL_SHUTDOWN;
271
272 // Set prescaler M value
273 //
274 // 0.085mAh • 50momh • M
275 // qLSB = ----------------------- =
276 // RSENSE • 128
277
278 // 0.085mAh • 50momh • 16
279 // qLSB = ----------------------- = 0,0133mAh -> MaxValue: 870,38mAh (not valid)
280 // 40momh • 128
281
282 // 0.085mAh • 50momh • 32
283 // qLSB = ----------------------- = 0,0265625mAh -> MaxValue: 1.740,7mAh valid
284 // 40momh • 128
285
286 // M=32 for 1050mA battery AC max value 39.530
287
288 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_PSCM_MSK; // clear prescaler M bits
289 ctrl |= DIOI2CMONITORGAUGELTC2942_PSCM_32; // configure new value
290
291 // Disable AL/CC pin
292 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_ALCC_MSK; // clear AL/CC bits
293 ctrl |= DIOI2CMONITORGAUGELTC2942_ALCC_DISABLED; // configure new value
294
295 // Write control register
296 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl))
297 {
298 return false;
299 }
300
301 // Checks if there was power off and sets de acumulator in middle value
302 XWORD currentAC = GetAC();
303 if(currentAC >= 0x7FF0) SetAC(DIOI2CMONITORGAUGELTC2942_AC_MAXVALUE/2);
304
305 return true;
306}

340bool DIOI2CMONITORGAUGELTC2942::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
341{
342 if(!diostream) return false;
343
344 bool status;
345
346 status = diostream->Write(&offset, 1)?true:false;
347 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
348
349 if(!status) return false;
350
351 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
352 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
353
354 return status;
355}
