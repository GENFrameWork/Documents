811GRPVECTORFILERESULT GRPVECTORFILEDXFTEXTSECTIONENTITIES::ParserTextSection(XFILETXT* fileTXT)
812{
813 XVECTOR<GRPVECTORFILEDXFTEXTPART*> parts;
814 GRPVECTORFILEDXFTEXTPART* part = NULL;
815 GRPVECTORFILEDXFXDATACTRL* xdatactrl = NULL;
816 int indexline = iniline;
817 XSTRING* line;
818
819 enumentities.DeleteKeyContents();
820 enumentities.DeleteAll();
821
822 part = new GRPVECTORFILEDXFTEXTPART();
823
824 do{ line = fileTXT->GetLine(indexline);
825 if(line)
826 {
827 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
828
829 if(line && !line->Compare(__L("0"),true))
830 {
831 indexline++;
832 line = fileTXT->GetLine(indexline);
833 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
834
835 if(!line->IsNumber())
836 {
837 if(IsKnownEntity(*line))
838 {
839 if(part && (part->iniline != -1))
840 {
841 part->endline = indexline-2;
842
843 parts.Add(part);
844 part = NULL;
845
846 part = new GRPVECTORFILEDXFTEXTPART ();
847 }
848
849 if(part && (part->iniline == -1))
850 {
851 part->name = line->Get();
852 part->iniline = indexline + 1;
853 }
854 }
855 else
856 {
857 XSTRING message;
858
859 message.Format(__L("entity %s Unknown"), line->Get());
860
861 GRPVECTORFILE_XEVENT vfevent(GetGrpVectorFile(), GRPVECTORFILE_XEVENTTYPE_PARTUNKNOWN);
862
863 vfevent.SetType(GRPVECTORFILETYPE_DXF);
864 vfevent.GetPath()->Set(fileTXT->GetPrimaryFile()->GetPathNameFile());
865 vfevent.GetMsg()->Set(message);
866
867 PostEvent(&vfevent, GetGrpVectorFile());
868 }
869 }
870 }
871 }
872
873 indexline++;
874
875 } while(indexline < endline);
876
877 if(part && (part->iniline != -1))
878 {
879 part->endline = indexline-1;
880
881 parts.Add(part);
882 part = NULL;
883 }
884
885 if(part && part->iniline == -1)
886 {
887 delete part;
888 part = NULL;
889 }
890
891 indexline = 0;
892
893 for(XDWORD c=0; c<parts.GetSize(); c++)
894 {
895 part = parts.Get(c);
896 if(part)
897 {
898 GRPVECTORFILEDXFENTITY* entity = new GRPVECTORFILEDXFENTITY();
899 if(entity)
900 {
901 entity->GetName()->Set(part->name);
902
903 indexline = part->iniline;
904
905 do{ line = fileTXT->GetLine(indexline);
906 if(line)
907 {
908 indexline++;
909 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
910
911 int type = line->ConvertToInt();
912
913
914 GRPVECTORFILEDXFVALUE* value = new GRPVECTORFILEDXFVALUE();
915 if(value)
916 {
917 GRPVECTORFILEDXFTEXTSECTIONENTITYDEFTYPE* defType = IsKnownTypeValue(part->name, type);
918 if(!defType)
919 {
920 #ifdef TEST_ONLY_DEFINE_IN_ENTITITES
921
922 XSTRING message;
923
