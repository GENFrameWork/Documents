187bool APPFLOWINTERNETSERVICES::Ini(APPFLOWCFG* cfg, XDWORD timeoutgetpublicip)
188{
189 bool status = false;
190
191 this->cfg = cfg;
192
193 xscheduler = new XSCHEDULER();
194 if(!xscheduler) return false;
195
196 XSCHEDULERTASK* xtask;
197
198 //-------------------------------------------------------------------
199
200 if(cfg->InternetServices_GetCheckInternetStatusCadence())
201 {
202 xtask = new XSCHEDULERTASK( xscheduler);
203 if(!xtask) return false;
204
205 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, cfg->InternetServices_GetCheckInternetStatusCadence());
206 xtask->SetID(APPFLOWINTERNETSERVICES_TASKID_CHECKCONNECTIONINTERNET);
207 xtask->SetIsStartImmediatelyCycles(true);
208 xtask->SetIsActive(true);
209
210 xscheduler->Task_Add(xtask);
211
212 checkinternetconnection = new DIOCHECKINTERNETCONNECTION(5);
213 status =(checkinternetconnection)?true:false;
214 }
215
216 //-------------------------------------------------------------------
217
218 if(cfg->InternetServices_GetCheckIPsChangeCadence())
219 {
220 xtask = new XSCHEDULERTASK(xscheduler);
221 if(!xtask) return false;
222
223 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, 5); // Get IP first time
224 xtask->SetID(APPFLOWINTERNETSERVICES_TASKID_GETIPS);
225 xtask->SetIsStartImmediatelyCycles(false);
226 xtask->SetIsActive(true);
227
228 xscheduler->Task_Add(xtask);
229
230 #ifdef APPFLOW_CFG_DYNDNSMANAGER_ACTIVE
231 if(dyndnsmanager)
232 {
233 for(XDWORD c=0; c<cfg->DNSManager_GetURLs()->GetSize(); c++)
234 {
235 dyndnsmanager->AddDNS((*cfg->DNSManager_GetURL(c)));
236 }
237 }
238 else return true;
239 #endif
240 }
241
242 //-------------------------------------------------------------------
243
244 if(cfg->InternetServices_GetUpdateTimeByNTPCadence())
245 {
246 for(XDWORD c=0; c<cfg->InternetServices_GetUpdateTimeNTPServers()->GetSize(); c++)
247 {
248 XSTRING* server = new XSTRING();
249 if(server)
250 {
251 server->Set(cfg->InternetServices_GetUpdateTimeNTPServer(c)->Get());
252 NTPservers.Add(server);
253
254 server = NULL;
255 }
256 }
257
258 xtask = new XSCHEDULERTASK( xscheduler);
259 if(!xtask) return false;
260
261 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, cfg->InternetServices_GetUpdateTimeByNTPCadence());
262 xtask->SetID(APPFLOWINTERNETSERVICES_TASKID_CHECKNTPDATETIME);
263 xtask->SetIsStartImmediatelyCycles(true);
264 xtask->SetIsActive(true);
265
266 xscheduler->Task_Add(xtask);
267 }
268
269 //-------------------------------------------------------------------
270
271 SubscribeEvent(XEVENT_TYPE_SCHEDULER, xscheduler);
272 if(!xscheduler->Ini()) return false;
273
274 //-------------------------------------------------------------------
275
276 if(timeoutgetpublicip)
277 {
278 bool haveinternet = true;
279
280 if(checkinternetconnection)
281 {
282 haveinternet = checkinternetconnection->Check();
283 }
284
285 if(haveinternet)
286 {
287 if(cfg->InternetServices_GetCheckIPsChangeCadence() > 0)
288 {
289 XTIMER* GEN_XFACTORY_CREATE(timeout, CreateTimer())
290 if(timeout)
291 {
292 do{ if(!publicIP.IsEmpty()) break;
293
294 GEN_XSLEEP.MilliSeconds(50);
295
296 } while(timeout->GetMeasureSeconds() <= timeoutgetpublicip);
297 }
298
299 GEN_XFACTORY.DeleteTimer(timeout);
300 }
301 }
302 }
303
304 return true;
305}
