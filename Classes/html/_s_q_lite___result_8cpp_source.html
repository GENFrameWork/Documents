208bool SQLITE_RESULT::ProcessRow()
209{
210 row.Clear();
211 header.DeleteContents();
212 header.DeleteAll();
213
214 int ncol = (int)this->GetNumColumns();
215
216 for(int e=0; e<ncol; e++)
217 {
218 DB_SQL_VARIANT* variant=new DB_SQL_VARIANT();
219 if(!variant) return false;
220
221 DB_SQL_STRING columnvalue;
222 char* text=(char*)sqlite3_column_text(query->ppstmt, (int)e);
223 int len = sqlite3_column_bytes(query->ppstmt, (int)e);
224 columnvalue.Set(text);
225
226 DB_SQL_STRING* columnname=new DB_SQL_STRING();
227 if(!columnname)
228 {
229 delete(variant);
230 return false;
231 }
232
233 columnname->Set(sqlite3_column_name(query->ppstmt, (int)e));
234
235 int sqtype=sqlite3_column_type(query->ppstmt, (int)e);
236 switch(sqtype)
237 {
238 case SQLITE_INTEGER : { int i = columnvalue.ConvertToInt();
239 (*variant) = i;
240 }
241 break;
242
243 case SQLITE_FLOAT : { float i = columnvalue.ConvertToFloat();
244 (*variant) = i;
245 }
246 break;
247
248 case SQLITE_BLOB : { // unsupported
249 // XTRACE_PRINTCOLOR(0,__L("Unsupported BLOB type"));
250 }
251 break;
252
253 case SQLITE_NULL : break;
254
255 case SQLITE_TEXT : { XSTRING received;
256
257 received.ConvertFromUTF8((XBYTE*)text,len);
258 (*variant) = received;
259 }
260 break;
261 }
262
263 header.Add(columnname);
264
265 row.Add(variant);
266 }
267
268 return true;
269}
