222bool XTRANSLATION::Translate_Add(XDWORD ID, XCHAR* sentence, XDWORD fixed)
223{
224 if(!ID) return false;
225 if(!sentence) return false;
226
227 XDWORD sizesentence = XSTRING::GetSize(sentence);
228 if(fixed)
229 {
230 if(sizesentence > fixed) sizesentence = fixed;
231 }
232
233 XCHAR* newsentence = new XCHAR[sizesentence+1];
234 if(newsentence)
235 {
236 memset(newsentence, 0 , (sizesentence+1) * sizeof(XCHAR));
237 memcpy(newsentence, sentence , sizesentence * sizeof(XCHAR));
238
239 sentences[ID] = newsentence;
240 }
241
242 return true;
243}

330bool XTRANSLATION::LoadLanguageAvailable()
331{
332 #ifdef XFILE_ACTIVE
333
334 XFILEJSON xfilejson;
335
336 if(!xfilejson.Open(xpath)) return false;
337 if(!xfilejson.ReadAllFile()) return false;
338 if(!xfilejson.DecodeAllLines()) return false;
339
340 XFILEJSONARRAY* languages = (XFILEJSONARRAY*)xfilejson.GetObj(__L("languages"));
341 if(languages)
342 {
343 for(XDWORD c=0; c<languages->GetValues()->GetSize(); c++)
344 {
345 XFILEJSONVALUE* value = languages->GetValues()->Get(c);
346 if(value)
347 {
348 if(value->GetType() == XFILEJSONVALUETYPE_STRING)
349 {
350 XSTRING strvalue = (XSTRING)(value->GetValueString());
351
352 if(!strvalue.IsEmpty())
353 {
354 XDWORD code = ISO_639_3.Code_GetByCodeAlpha3(strvalue.Get());
355 if(code != XLANGUAGE_ISO_639_3_CODE_INVALID)
356 {
357 languageavailables.Add(code);
358 }
359 else
360 {
361 XDWORD code = ISO_639_3.Code_GetByEnglishName(strvalue.Get());
362 if(code != XLANGUAGE_ISO_639_3_CODE_INVALID)
363 {
364 languageavailables.Add(code);
365 }
366 else
367 {
368 XDWORD code = ISO_639_3.Code_GetByAlias(strvalue.Get());
369 if(code != XLANGUAGE_ISO_639_3_CODE_INVALID)
370 {
371 languageavailables.Add(code);
372 }
373 }
374 }
375 }

377 }
378 }
379 }

382 xfilejson.Close();

465bool XTRANSLATION::Translate_Load()
466{
467 #ifdef XFILE_ACTIVE
468
469 XFILEJSON xfilejson;
470
471 if(!xfilejson.Open(xpath)) return false;
472 if(!xfilejson.ReadAllFile()) return false;
473 if(!xfilejson.DecodeAllLines()) return false;
474
475 int languageindex = GetIndexLanguageByCode(code);
476 if(languageindex == -1) return false;
477
478 XFILEJSONARRAY* translations = (XFILEJSONARRAY*)xfilejson.GetObj(__L("translations"));
479 if(translations)
480 {
481 for(XDWORD c=0; c<translations->GetValues()->GetSize(); c++)
482 {
483 XFILEJSONVALUE* translation = translations->GetValues()->Get(c);
484 if(translation)
485 {
486 XSTRING IDstr = translation->GetName()->Get();
487 if(!IDstr.IsEmpty())
488 {
489 XDWORD ID = IDstr.ConvertToDWord();
490 if(ID < XTRANSLATION_MAXSENTENCES)
491 {
492 XFILEJSONARRAY* translationarray = (XFILEJSONARRAY*)translation->GetValuePointer();
493 if(translationarray)
494 {
495 XDWORD sizesentencefixed = (XDWORD)translationarray->GetValues()->Get(0)->GetValueInteger();
496 XSTRING sentence = (XSTRING)translationarray->GetValues()->Get(languageindex+1)->GetValueString();
497
498 Translate_Add(ID, sentence.Get(), sizesentencefixed);
499 }
500 }
501 }
502 }
503 }
504 }
505
506 xfilejson.Close();
507
508 return true;
509
510 #else
511
512 return false;
513
514 #endif
515}
