423{
424 Clean();
425
426 type = DIONODEITEMHANDLER_TYPE_GPIODIGITAL;
427 name = __L("GPIO Digital");
428
429 for(XDWORD c=0; c<entrysGPIO->GetSize(); c++)
430 {
431 DIONODEITEMHANDLER_ENTRYGPIO* entryGPIO = new DIONODEITEMHANDLER_ENTRYGPIO();
432 if(entryGPIO)
433 {
434 entryGPIO->CopyFrom(entrysGPIO->Get(c));
435 this->entrysGPIO.Add(entryGPIO);
436
437 if(entryGPIO->GetPin() != DIONODEITEMHANDLER_INVALIDPARAM)
438 {
439 if(!GEN_DIOGPIO.GPIOEntry_SetIDByPin(entryGPIO->GetEntryID(), entryGPIO->GetGPIO()))
440 {
441 GEN_DIOGPIO.GPIOEntry_CreateByPin(entryGPIO->GetEntryID(), entryGPIO->GetPin());
442 }
443 }
444
445 if(entryGPIO->GetGPIO() != DIONODEITEMHANDLER_INVALIDPARAM)
446 {
447 if(!GEN_DIOGPIO.GPIOEntry_SetIDByGPIO(entryGPIO->GetEntryID(), entryGPIO->GetGPIO()))

628{

630 {

634 nodeitem->SetCategory(DIONODEITEM_CATEGORY_GPIO_DIGITAL);
635
636 for(XDWORD c=0; c<entrysGPIO.GetSize(); c++)
637 {
638 DIONODEITEMHANDLER_ENTRYGPIO* entryGPIO = entrysGPIO.Get(c);
639 if(entryGPIO)

642 if(value)

648
649 nodeitem->GetValues()->Add(value);

651 }
652 }
