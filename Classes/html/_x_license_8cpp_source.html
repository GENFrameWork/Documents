283bool XLICENSE::GenerateMachineID(XLICENSEID& licenseID)
284{
285 DIOSTREAMIPLOCALENUMDEVICES* enumdevices = (DIOSTREAMIPLOCALENUMDEVICES*)GEN_DIOFACTORY.CreateStreamEnumDevices(DIOSTREAMENUMTYPE_IP_LOCAL);
286 if(enumdevices)
287 {
288 enumdevices->Search();
289
290 XQWORD basicID = 0;
291 XSTRING basicstring;
292 XBUFFER basicxbuffer;
293
294 for(int c=0;c<(int)enumdevices->GetDevices()->GetSize();c++)
295 {
296 DIOSTREAMDEVICEIP* device = (DIOSTREAMDEVICEIP*)enumdevices->GetDevices()->Get(c);
297 if(device)
298 {
299 DIOMAC* mac = device->GetMAC();
300 if(mac)
301 {
302 XQWORD macID = 0;
303
304 for(int d=0; d<DIOMAC_MAXSIZE; d++)
305 {
306 macID |= mac->Get()[d];
307 macID<<=8;
308 }
309
310 basicID ^= macID;
311 }
312 }
313 }
314
315 GEN_DIOFACTORY.DeleteStreamEnumDevices(enumdevices);
316
317 basicstring.Format(__L("%02X%02X%llX%llX%02X%02X"), XLICENSE_VERSION, XLICENSE_SUBVERSION, basicID, ~basicID, XLICENSE_VERSION, XLICENSE_SUBVERSION);
318
319 HASHSHA1* sha1 = new HASHSHA1();
320 if(sha1)
321 {
322 basicstring.ConvertHexStringToBuffer(basicxbuffer);
323 sha1->Do(basicxbuffer);
324
325 XDWORD index = 0;
326 XDWORD partID = 0;
327 int sizeofpart = sizeof(XDWORD)/sizeof(XBYTE);
328
329 for(int c=0; c<XLICENSE_MAXIDPARTS; c++)
330 {
331 partID = 0;
332 for(int d=0; d<sizeofpart; d++)
333 {
334 partID |= (XBYTE)sha1->GetResult()->GetByte(index++);
335 if(d!=(sizeofpart-1)) partID<<=8;
336 if(index>=sha1->GetResult()->GetSize()) index = 0;
337 }
338
339 licenseID.SetPart(c, partID);
340 }
341
342 delete sha1;
343
344 return true;
345 }
346 }
347
348 return false;
349}

365bool XLICENSE::Generate(XLICENSEID& licenseID, XSTRING* applicationID, XBUFFER* license)
366{
367 CIPHERAES cipherAES;
368 CIPHERKEYSYMMETRICAL key;
369 XBUFFER xbufferkey;
370 XBUFFER xbuffercipher;
371 XBUFFER* _license = license;
372 XSTRING* _applicationID = applicationID;
373
374 if(!_license) _license = this->license;
375 if(!_license) return false;
376
377 if(!_applicationID) _applicationID = &this->applicationID;
378 if(!_applicationID) return false;
379
380 _license->Delete();
381 licenseID.GetXBuffer(xbuffercipher);
382
383 GetBufferKeyFromMachineID((*_applicationID), xbufferkey);
384
385 key.Set(xbufferkey);
386
387 cipherAES.SetKey(&key);
388 cipherAES.SetChainingMode(CIPHERCHAININGMODE_CBC);
389 cipherAES.SetPaddingType(XBUFFER_PADDINGTYPE_NONE);
390
391 if(cipherAES.Cipher(xbuffercipher))
392 {
393 _license->Delete();
394 _license->Add(cipherAES.GetResult());
395
396 return true;
397 }
398
399 return false;
400}

415bool XLICENSE::GetBufferKeyFromMachineID(XSTRING& applicationID, XBUFFER& xbufferkey)
416{
417 HASHSHA1 sha1;
418
419 xbufferkey.Delete();
420
421
422 XBUFFER charstr;
423
424 applicationID.ConvertToASCII(charstr);
425 xbufferkey.Add((XBYTE*)charstr.Get(), applicationID.GetSize());
426
427 sha1.Do(xbufferkey);
428
429 xbufferkey.Delete();
430 xbufferkey.Add(sha1.GetResult());
431
432 xbufferkey.Resize(16);
433
434 return xbufferkey.GetSize()?true:false;
435}

451bool XLICENSE::CipherExpirationDate(bool cipher, XSTRING& applicationID, XSTRING& expirationdate)
452{
453 XBUFFER xbufferkey;
454 CIPHERAES cipherAES;
455 CIPHERKEYSYMMETRICAL key;
456 bool status = false;
457
458 if(!GetBufferKeyFromMachineID(applicationID, xbufferkey)) return false;
459
460 key.Set(xbufferkey);
461
462 cipherAES.SetKey(&key);
463 cipherAES.SetChainingMode(CIPHERCHAININGMODE_CBC);
464 cipherAES.SetPaddingType(XBUFFER_PADDINGTYPE_SPACES);
465
466 if(cipher)
467 {
468 XBUFFER charstr;
469
470 expirationdate.ConvertToASCII(charstr);
471
472 status = cipherAES.Cipher((XBYTE*)charstr.Get(), expirationdate.GetSize());
473 if(status)
474 {
475 cipherAES.GetResult()->ConvertToBase64(expirationdate);
476 return true;
477 }
478 }
479 else
480 {
481 XBUFFER xbuffer;
482
483 xbuffer.ConvertFromBase64(expirationdate);
484
485 status = cipherAES.Uncipher(xbuffer);
486 if(status)
487 {
488 expirationdate.Empty();
489
490 for(XDWORD c=0; c<cipherAES.GetResult()->GetSize(); c++)
491 {
492 expirationdate.Add((XCHAR)cipherAES.GetResult()->GetByte(c));
493 }
494
495 expirationdate.DeleteCharacter(0x20, XSTRINGCONTEXT_TO_END);
496
497 return true;
498 }
499 }
500
501 return status;
502}

668bool XLICENSE::LoadFromBuffer(XBUFFER& xbuffer, XSTRING& applicationID, XBUFFER* license, XSTRING* expirationdate)
669{
670 if(!license) return false;
671
672 XFILEINI fileini;
673 XLICENSEID licenseID;
674 XSTRING licenseIDstring;
675 XSTRING licenseIDstring2;
676 XSTRING licensestring;
677 XSTRING licenseexprirationstring;
678 XBUFFER* _license = license;
679
680 if(!_license) _license = this->license;
681 if(!_license) return false;
682
683 fileini.AddBufferLines(XFILETXTFORMATCHAR_ASCII, xbuffer);
684
685 fileini.ConvertFromLines();
686
687 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_ID , licenseIDstring);
688 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE , licensestring);
689 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_EXPIRATION , licenseexprirationstring);
690
691 GenerateMachineID(licenseID);
692 licenseID.GetXString(licenseIDstring2);
693
694 if(licenseIDstring.Compare(licenseIDstring2)) return false;
695
696 CipherExpirationDate(false, applicationID, licenseexprirationstring);
697
698 if(expirationdate) (*expirationdate) = licenseexprirationstring;
699
700 licensestring.ConvertHexStringToBuffer((*_license));
701
702 return true;
703}

720bool XLICENSE::CheckMasterCreation(XPATH& xpath, XLICENSEID& licenseID, XSTRING& applicationID, int expirationseconds)
721{
722 XFILEINI fileini;
723 XSTRING licensestring;
724
725 if(!fileini.Open(xpath)) return false;
726 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE, licensestring);
727 fileini.Close();
728
729 if(!applicationID.Compare(licensestring))
730 {
731 XFILEINI fileini;
732 XSTRING licenseIDstring;
733 XSTRING licensestring;
734 XSTRING licenseexpiration;
735 XBUFFER licencia;
736 XDATETIME* expirationdatetime;
737
738 licenseID.GetXString(licenseIDstring);
739 if(!Generate(licenseID, &applicationID, &licencia)) return false;
740
741 licensestring.ConvertHexStringFromBuffer(licencia);
742
743 GEN_XFACTORY_CREATE(expirationdatetime, CreateDateTime())
744 if(!expirationdatetime) return false;
745
746 expirationdatetime->SetToZero();
747 if(expirationseconds)
748 {
749 expirationdatetime->Read();
750 expirationdatetime->AddSeconds((int)expirationseconds);
751 }
752
753 expirationdatetime->GetDateTimeToString(XDATETIME_FORMAT_STANDARD, licenseexpiration);
754 GEN_XFACTORY.DeleteDateTime(expirationdatetime);
755
756 if(!CipherExpirationDate(true, applicationID, licenseexpiration)) return false;
757
758 if(!fileini.Create(xpath)) return false;
759
760 fileini.CreateSection(XLICENSE_FILESECTION);
761 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_ID , licenseIDstring);
762 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE , licensestring);
763 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_EXPIRATION, licenseexpiration);
764
765 fileini.Close();
766
767 return true;
768 }
769
770 return false;
771}
