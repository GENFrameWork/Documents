141int SCRIPT_LNG_LUA::Run(int* returnval)
142{
143 errorcode = SCRIPT_ERRORCODE_NONE;
144 iscancelexec = false;
145
146 int status = 0;
147
148 if(HaveMainFunction())
149 {
150 //Exec to ajust the Stack in Lua with Main Function. lua_getglobal don´t work well without this.
151 lua_pcall(state, 0, LUA_MULTRET, 0);
152 }
153
154 lua_getglobal(state, SCRIPT_LNG_LUA_MAINFUNCTIONNAME);
155
156 status = lua_isfunction(state, lua_gettop(state));
157 if(status)
158 {
159 status = lua_pcall(state, 0, 0, 0);
160 }
161 else
162 {
163 lua_setglobal(state, SCRIPT_LNG_LUA_MAINFUNCTIONNAME);
164 status = lua_pcall(state, 0, LUA_MULTRET, 0);
165 }
166
167
168 if(status != LUA_OK)
169 {
170 switch(status)
171 {
172 case LUA_YIELD : errorcode = SCRIPT_LNG_LUA_ERRORCODE_YIELD; break;
173 case LUA_ERRRUN : errorcode = SCRIPT_LNG_LUA_ERRORCODE_RUN; break;
174 case LUA_ERRSYNTAX : errorcode = SCRIPT_LNG_LUA_ERRORCODE_SYNTAX; break;
175 case LUA_ERRMEM : errorcode = SCRIPT_LNG_LUA_ERRORCODE_MEM; break;
176 case LUA_ERRGCMM : errorcode = SCRIPT_LNG_LUA_ERRORCODE_GCMM; break;
177 case LUA_ERRERR : errorcode = SCRIPT_LNG_LUA_ERRORCODE_ERRERR; break;
178 }
179
180 HaveError(errorcode);
181
182 return errorcode;
183 }
184
185 return errorcode;
186}

228bool SCRIPT_LNG_LUA::HaveError(int errorcode)
229{
230 if(errorcode != SCRIPT_ERRORCODE_NONE)
231 {
232 static XCHAR* errorstr[]= { __L("None") ,
233 __L("Internal error") ,
234 __L("Syntax error") ,
235 __L("Insufficient parameters") ,
236 __L("Thread not yield") ,
237 __L("runtime error") ,
238 __L("memory allocation error") ,
239 __L("error while running a __gc metamethod") ,
240 __L("error while running the error handler function") ,
241 };
242 XSTRING currenttoken;
243 XPATH namefile;
244 XSTRING errorstring;
245 int nline = 0;
246
247 currenttoken = lua_tostring(state, -1);
248
249 namefile.AdjustSize(_MAXSTR);
250 errorstring.AdjustSize(_MAXSTR);
251 currenttoken.UnFormat(__L("[string \"%s\"]:%d:%s"), namefile.Get(), &nline, errorstring.Get());
252 namefile.AdjustSize();
253 errorstring.AdjustSize();
254
255 namescript.Set(namefile);
256 namefile.SetOnlyNamefileExt();
257
258 SCRIPT_XEVENT xevent(this, SCRIPT_XEVENT_TYPE_ERROR);
259
260 xevent.SetError(errorcode);
261 xevent.GetErrorText()->Set(errorstr[errorcode]);
262 xevent.GetCurrentToken()->Set(currenttoken);
263 xevent.SetNLine(nline);
264
265 XTRACE_PRINTCOLOR(4, __L("Script [%s] ERROR %d: %s, line %d \"%s\"") , namefile.Get(), errorcode, errorstr[errorcode], nline, errorstring.Get());
266
267 PostEvent(&xevent);
268
269 this->errorcode = errorcode;
270 iscancelexec = true;
271 }
272
273 return (errorcode == SCRIPT_ERRORCODE_NONE)?false:true;
274}

286bool SCRIPT_LNG_LUA::HaveMainFunction()
287{
288 XSTRING mainfunctionname;
289 XSTRING script;
290 int start = 0;
291 int end = 0;
292 XSTRING line;
293
294 script = GetScript()->Get();
295 if(script.IsEmpty()) return false;
296
297 mainfunctionname = SCRIPT_LNG_LUA_MAINFUNCTIONNAME;
298
299 int index = GetScript()->Find(mainfunctionname, false);
300 if(index == XSTRING_NOTFOUND) return false;
301
302 for(start=index; start>0; start--)
303 {
304 if((script.Get()[start] == __C('\r')) || (script.Get()[start] == __C('\n')))
305 {
306 break;
307 }
308 }
309
310 for(end=index; end<(int)script.GetSize() ; end++)
311 {
312 if(script.Get()[end] == __C(')'))
313 {
314 break;
315 }
316 }
317
318 script.Copy(start, end+1, line);
319
320 line.DeleteCharacter(__C('\n'));
321 line.DeleteCharacter(__C('\r'));
322 line.DeleteCharacter(__C(' '));
323
324 XSTRING searchmask;
325
326 searchmask.Format(__L("function%s()"), mainfunctionname.Get());
327
328 if(!line.Compare(searchmask)) return true;
329
330 return false;
331}

359int LUA_LibraryCallBack(lua_State* state)
360{
361 lua_Debug ar;
362 XSTRING namefunction;
363 SCRIPT_LNG_LUA* script = *static_cast<SCRIPT_LNG_LUA**>(lua_getextraspace(state));
364
365 if(!script)
366 {
367 return 0;
368 }
369
370 lua_getstack (state, 0, &ar);
371 if(lua_getinfo(state, "Snlu", &ar)) namefunction = ar.name;
372
373 if(namefunction.IsEmpty())
374 {
375 return 0;
376 }
377
378 SCRIPT_LIB_FUNCTION* libfunction = script->GetLibraryFunction(namefunction.Get());
379 if(!libfunction)
380 {
381 return 0;
382 }
383
384 if(!libfunction->GetFunctionLibrary())
385 {
386 return 0;
387 }
388
389 XVECTOR<XVARIANT*> params;
390 XVARIANT returnvalue;
391
392 int nargs = lua_gettop(state);
393
394 for(int c=1; c <= nargs; ++c)
395 {
396 XVARIANT* variant = new XVARIANT();
397 if(variant)
398 {
399 switch(lua_type(state, c))
400 {
401 case LUA_TNONE : break;
402 case LUA_TNIL : /*(*variant) = NULL;*/ break;
403 case LUA_TBOOLEAN : (*variant) = (lua_toboolean(state, c)?true:false); break;
404 case LUA_TLIGHTUSERDATA : break;
405 case LUA_TNUMBER : /*
406 {
407 float number = (float)lua_tonumber(state, c);
408 double intpart = 0.0f;
409 double fracpart = 0.0f;
410
411 fracpart = modf(number, &intpart);
412
413 if(!fracpart)
414 (*variant) = (int)(number);
415 else (*variant) = (float)(number);
416 }
417 */
418
419 (*variant) = (double)lua_tonumber(state, c);
420 break;
421
422
423 case LUA_TSTRING : (*variant) = (char*)(lua_tostring(state, c)); break;
424 case LUA_TTABLE : break;
425 case LUA_TFUNCTION : // XVARIANT don´t suppor pointers yet
426 //(*variant) = (void*)(lua_tocfunction(state,c));
427 break;
428 case LUA_TUSERDATA : break;
429 case LUA_TTHREAD : break;
430 case LUA_NUMTAGS : break;
431 }
432
433 params.Add(variant);
434
435 }
436 }
437
438 libfunction->GetFunctionLibrary()(libfunction->GetLibrary(), script, &params, &returnvalue);
439
440 params.DeleteContents();
441 params.DeleteAll();
442
443 // Push the return value on top of the stack. NOTE: We haven't popped the
444 // input arguments to our function. To be honest, I haven't checked if we
445 // must, but at least in stack machines like the JVM, the stack will be
446 // cleaned between each function call.
447
448 int nreturnvalues = 0;
449
450 switch(returnvalue.GetType())
451 {
452 case XVARIANT_TYPE_NULL : lua_pushnil (state); nreturnvalues++; break;
453 case XVARIANT_TYPE_BOOLEAN : lua_pushboolean (state, (int)(returnvalue)); nreturnvalues++; break;
454 case XVARIANT_TYPE_INTEGER : lua_pushnumber(state, (int)(returnvalue)); nreturnvalues++; break;
455 case XVARIANT_TYPE_CHAR : lua_pushnumber(state, (int)(returnvalue)); nreturnvalues++; break;
456 case XVARIANT_TYPE_XCHAR : break;
457 case XVARIANT_TYPE_FLOAT : lua_pushnumber(state, (float)(returnvalue)); nreturnvalues++; break;
458
459 case XVARIANT_TYPE_STRING : { XSTRING stringreturnvalue;
460
461 stringreturnvalue = (const XSTRING&)returnvalue;
462
463 XBUFFER charstr;
464
465 stringreturnvalue.ConvertToASCII(charstr);
466 lua_pushstring(state, charstr.GetPtrChar());
467
468 nreturnvalues++;
469 }
470 break;
471
472 case XVARIANT_TYPE_DATE : break;
473 case XVARIANT_TYPE_TIME : break;
474 case XVARIANT_TYPE_DATETIME : break;
475 }
476
477
478 // Let Lua know how many return values we've passed
479 return nreturnvalues;
480}
