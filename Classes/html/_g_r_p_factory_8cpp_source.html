245GRPCANVAS* GRPFACTORY::CreateCanvas(GRPPROPERTIES* properties)
246{
247 GRPCANVAS* canvas = NULL;
248 GRPPROPERTYMODE mode = GRPPROPERTYMODE_XX_UNKNOWN;
249
250 if(!properties) return NULL;
251
252 mode = properties->GetMode();
253 if(mode == GRPPROPERTYMODE_XX_UNKNOWN) return NULL;
254
255 if(!properties->GetWidth() || !properties->GetHeight()) return NULL;
256
257 switch(mode)
258 {
259 case GRPPROPERTYMODE_08_INDEX : break;
260 case GRPPROPERTYMODE_16_RGBA_4444 : break;
261 case GRPPROPERTYMODE_16_RGBA_5551 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb555, agg::rgba8>(); break;
262 case GRPPROPERTYMODE_16_RGB_565 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb565, agg::rgba8>(); break;
263 case GRPPROPERTYMODE_16_SRGB_565 : break;
264 case GRPPROPERTYMODE_24_RGB_888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb24 , agg::rgba8>(); break;
265 case GRPPROPERTYMODE_24_BGR_888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_bgr24 , agg::rgba8>(); break;
266 case GRPPROPERTYMODE_32_RGBA_8888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgba32, agg::rgba8>(); break;
267 case GRPPROPERTYMODE_32_BGRA_8888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_bgra32, agg::rgba8>(); break;
268 default : break;
269 }
270
271 if(canvas) canvas->CopyPropertysFrom(properties);
272
273 return canvas;
274}
