252GRPCANVAS* GRPFACTORY::CreateCanvas(GRPPROPERTIES* properties)
253{
254 GRPCANVAS* canvas = NULL;
255 GRPPROPERTYMODE mode = GRPPROPERTYMODE_XX_UNKNOWN;
256
257 if(!properties)
258 {
259 return NULL;
260 }
261
262 mode = properties->GetMode();
263 if(mode == GRPPROPERTYMODE_XX_UNKNOWN)
264 {
265 return NULL;
266 }
267
268 if(!properties->GetWidth() || !properties->GetHeight())
269 {
270 return NULL;
271 }
272
273 switch(mode)
274 {
275 case GRPPROPERTYMODE_08_INDEX : break;
276 case GRPPROPERTYMODE_16_RGBA_4444 : break;
277 case GRPPROPERTYMODE_16_RGBA_5551 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb555, agg::rgba8>(); break;
278 case GRPPROPERTYMODE_16_RGB_565 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb565, agg::rgba8>(); break;
279 case GRPPROPERTYMODE_16_SRGB_565 : break;
280 case GRPPROPERTYMODE_24_RGB_888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgb24 , agg::rgba8>(); break;
281 case GRPPROPERTYMODE_24_BGR_888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_bgr24 , agg::rgba8>(); break;
282 case GRPPROPERTYMODE_32_RGBA_8888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_rgba32, agg::rgba8>(); break;
283 case GRPPROPERTYMODE_32_BGRA_8888 : canvas = (GRPCANVAS*)new GRPCANVASAGG<agg::pixfmt_bgra32, agg::rgba8>(); break;
284 default : break;
285 }
286
287 if(canvas)
288 {
289 canvas->CopyPropertysFrom(properties);
290 }
291
292 return canvas;
293}
