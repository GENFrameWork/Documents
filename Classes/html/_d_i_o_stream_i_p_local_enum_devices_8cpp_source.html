167DIOSTREAMDEVICE* DIOSTREAMIPLOCALENUMDEVICES::GetFirstActiveDevice()
168{
169 Search();
170
171 for(int c=0;c<(int)GetDevices()->GetSize();c++)
172 {
173 DIOSTREAMDEVICEIP* device = (DIOSTREAMDEVICEIP*)GetDevices()->Get(c);
174 if(device)
175 {
176 if((!device->GetIP()->IsEmpty()) &&
177 (!device->GetIP()->IsAPIPA()) &&
178 ((device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_ETHERNET) ||
179 (device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_WIFI) ||
180 (device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_PPP)))
181 {
182 return device;
183 }
184 }
185 }
186
187 return NULL;
188}
