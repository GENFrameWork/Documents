115{
116 DelAllDevices();
117
118 int sock;
119 struct ifconf ifconf;
120 struct ifreq ifreq[DIOLINUXSTREAMIPLOCALENUMDEVICES_MAXIFS];
121 int ninterfaces;
122 int c = 0;
123
124 sock = socket(AF_INET, SOCK_STREAM, 0);
125 if (sock < 0) return false;
126
127 ifconf.ifc_buf = (char *) ifreq;
128 ifconf.ifc_len = sizeof ifreq;
129
130 if(ioctl(sock, SIOCGIFCONF, &ifconf) == -1)
131 {
132 close(sock);
133 return false;
134 }
135
136 ninterfaces = ifconf.ifc_len / sizeof(ifreq[0]);
137
138 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("N NET Interfaces: %d"), ninterfaces);
139
140 for(c=0; c<ninterfaces; c++)
141 {
142 char ip[INET_ADDRSTRLEN];
143 char mask[INET_ADDRSTRLEN];
144 struct sockaddr_in* address = (struct sockaddr_in *) &ifreq[c].ifr_addr;
145 struct sockaddr_in* addressmask = (struct sockaddr_in *) &ifreq[c].ifr_netmask;
146
147 memset(ip , 0, sizeof(ip));
148 memset(mask, 0, sizeof(mask));
149
150 if(!inet_ntop(AF_INET, &address->sin_addr, ip, sizeof(ip)))
151 {
152 close(sock);
153 return false;
154 }
155
156 if(!inet_ntop(AF_INET, &addressmask->sin_addr, mask, sizeof(mask)))
157 {
158 close(sock);
159 return false;
160 }
161
162 if(ioctl(sock, SIOCGIFHWADDR, &ifreq[c])==-1)
163 {
164 close(sock);
165 return false;
166 }
167
168 DIOSTREAMDEVICEIP* device = new DIOSTREAMDEVICEIP();
169 if(device)
170 {
171 device->SetIsActive(true);
172
173 device->SetIndex(c);
174
175 device->GetName()->Set(ifreq[c].ifr_name);
176 device->GetDescription()->Set(__L(""));
177 device->GetMAC()->Set((XBYTE*)ifreq[c].ifr_hwaddr.sa_data);
178
179 device->SetIPType(DIOSTREAMIPDEVICE_TYPE_UNKNOWN);
180
181 if(device->GetName()->Find(__L("eth") , true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_ETHERNET);
182 if(device->GetName()->Find(__L("enx") , true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_ETHERNET); // Debian Stretch form
183 if(device->GetName()->Find(__L("ens") , true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_ETHERNET); // Cent OS 7
184 if(device->GetName()->Find(__L("ppp") , true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_PPP); // Modem / ppp
185 if(device->GetName()->Find(__L("wlan"), true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_WIFI);
186 if(device->GetName()->Find(__L("wwan"), true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_WWAN);
187 if(device->GetName()->Find(__L("lo") , true) != XSTRING_NOTFOUND) device->SetIPType(DIOSTREAMIPDEVICE_TYPE_LOOPBACK);
188
189 XSTRING IP;
190 IP.Set(ip);
191 device->GetIP()->Set(IP);
192 device->GetIP()->GetMask()->Set(mask);
193
194 if(device->GetIP()->IsEmpty())
195 {
196 device->SetIsActive(false);
197 }
198
199 devices.Add(device);
200
201 //device->DebugPrintInfo();
202 }
203 }
204
205 close(sock);
206
207 return true;
