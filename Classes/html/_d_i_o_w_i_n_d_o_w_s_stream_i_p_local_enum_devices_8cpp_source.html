181 }
182
183 for(aa = adapter_addresses; aa != NULL; aa = aa->Next)
184 {
185 DIOSTREAMDEVICEIP* device = new DIOSTREAMDEVICEIP();

187 {
188 device->SetIsActive((aa->OperStatus == IfOperStatusUp?true:false));
189
190 char buffer[BUFSIZ];
191
192 device->SetIndex(index++);
193 device->SetType(DIOSTREAMDEVICE_TYPE_IP);

195 switch(aa->IfType)

197 case IF_TYPE_SOFTWARE_LOOPBACK : device->SetIPType(DIOSTREAMIPDEVICE_TYPE_LOOPBACK);
198 break;
199
200 case IF_TYPE_ETHERNET_CSMACD : device->SetIPType(DIOSTREAMIPDEVICE_TYPE_ETHERNET);
201 break;
202
203 case IF_TYPE_IEEE80211 : device->SetIPType(DIOSTREAMIPDEVICE_TYPE_WIFI);
204 break;
205
206 default : device->SetIPType(DIOSTREAMIPDEVICE_TYPE_UNKNOWN);
207 break;
208 }

210 memset(buffer, 0, BUFSIZ);
211 WideCharToMultiByte(CP_ACP, 0, aa->FriendlyName, (int)wcslen(aa->FriendlyName), buffer, BUFSIZ, NULL, NULL);

213
214 memset(buffer, 0, BUFSIZ);
215 WideCharToMultiByte(CP_ACP, 0, aa->Description, (int)wcslen(aa->Description), buffer, BUFSIZ, NULL, NULL);
216 device->GetDescription()->Set(buffer);
217
218 device->GetMAC()->Set((XBYTE*)aa->PhysicalAddress);
219
