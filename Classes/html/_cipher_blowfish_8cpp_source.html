383bool CIPHERBLOWFISH::Cipher(XBYTE* input,XDWORD size)
384{
385 if(!size) return false;
386
387 XBUFFER inputpadding;
388 CIPHERBLOWFISH_CONTEXT ctx;
389 XBYTE iv[16];
390
391 memset(iv, 0, 16);
392 if(inivector && inivector->GetSize()>=16) memcpy(iv, inivector->Get(), 16);
393
394 inputpadding.Add(input, size);
395 inputpadding.Padding_Add(paddingadjustsize, paddingtype);
396
397 result->Delete();
398 result->Add(input, size);
399 result->Padding_Add(paddingadjustsize, paddingtype);
400
401 CIPHERKEYSYMMETRICAL* key = (CIPHERKEYSYMMETRICAL*)GetKey();
402 if(!key) return false;
403 if(!key->Get()->Get()) return false;
404
405 BlowfishSetKey(&ctx, (XBYTE *)key->Get()->Get(), key->Get()->GetSize()*8);
406
407 XBYTE stream_block[16];
408 XBYTE nonce_counter[16];
409 bool status = false;
410 int offset = 0;
411
412 memset(stream_block ,0 , 16);
413 memset(nonce_counter ,0 , 16);
414
415 memcpy(nonce_counter,iv,16);
416
417 switch(this->GetChainingMode())
418 {
419 case CIPHERCHAININGMODE_UNKNOWN :
420 case CIPHERCHAININGMODE_ECB : status =false;
421 break;
422
423 case CIPHERCHAININGMODE_CBC : status = BlowfishCipher_CBC(&ctx, CIPHERBLOWFISH_ENCRYPT , result->GetSize(), iv, inputpadding.Get(), result->Get());
424 break;
425
426 case CIPHERCHAININGMODE_CFB : status = BlowfishCipher_CFB64(&ctx, CIPHERBLOWFISH_ENCRYPT , result->GetSize(), &offset, iv, inputpadding.Get(), result->Get());
427 break;
428
429 case CIPHERCHAININGMODE_CTR : status = BlowfishCipher_CTR(&ctx, result->GetSize(), &offset, nonce_counter, stream_block, inputpadding.Get(), result->Get());
430 break;
431
432 }
433
434 return status;

517 XWORD d;

552
553 Xl = *xl;
554 Xr = *xr;
555
556 for(i=0; i<CIPHERBLOWFISH_ROUNDS; ++i)

634bool CIPHERBLOWFISH::BlowfishSetKey(CIPHERBLOWFISH_CONTEXT* ctx, XBYTE* key, XDWORD keysize)
635{
636 XDWORD i;
637 XDWORD j;
638 XDWORD k;
639 XDWORD data;
640 XDWORD datal;
641 XDWORD datar;
642
643 if(keysize < CIPHERBLOWFISH_MIN_KEY || keysize > CIPHERBLOWFISH_MAX_KEY || ( keysize % 8 )) return false;
644
645 keysize >>= 3;
646
647 for( i = 0; i < 4; i++ )
648 {
649 for( j = 0; j < 256; j++ )
650 {
651 ctx->S[i][j] = S[i][j];
652 }
653 }
654
655 j = 0;
656 for(i=0; i<CIPHERBLOWFISH_ROUNDS+2; ++i)
657 {
658 data = 0x00000000;
659 for( k = 0; k < 4; ++k )
660 {
661 data = (data << 8) | key[j++];
662 if(j >= keysize) j = 0;
663 }
664 ctx->P[i] = P[i] ^ data;
665 }
666
667 datal = 0x00000000;
668 datar = 0x00000000;
669
670 for(i=0; i<CIPHERBLOWFISH_ROUNDS+2; i+=2)
671 {
672 BlowfishCipher(ctx, &datal, &datar);
673 ctx->P[i] = datal;
674 ctx->P[i + 1] = datar;
675 }
676
677 for( i = 0; i < 4; i++ )
678 {
679 for( j = 0; j < 256; j += 2 )
680 {
681 BlowfishCipher(ctx, &datal, &datar);
682 ctx->S[i][j] = datal;
683 ctx->S[i][j + 1] = datar;
684 }
685 }
686
687 return true;
688}
