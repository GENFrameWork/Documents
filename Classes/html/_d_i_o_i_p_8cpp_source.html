200bool DIOIPADDRESS::IsLocal()

202 if(address[0]==10) return true;

313bool DIOIPADDRESS::Set(XSTRING& addressstring)
314{
315 if(addressstring.IsEmpty()) return false;
316
317 int iip[DIOIP_MAXIPADDRESS] = { 0, 0, 0, 0 };
318 XBYTE bip[DIOIP_MAXIPADDRESS] = { 0, 0, 0, 0 };
319
320 if(addressstring.GetNCharacters(__C('.')) != 3) return false;
321
322 for(XDWORD c=0; c<addressstring.GetSize(); c++)
323 {
324 if((addressstring.Get()[c]<__C('0')) && (addressstring.Get()[c]>__C('9')) && (addressstring.Get()[c]!= __C('.'))) return false;
325 }
326
327 addressstring.UnFormat(__L("%d.%d.%d.%d"), &iip[0], &iip[1], &iip[2], &iip[3]);
328
329 for(int c=0; c<DIOIP_MAXIPADDRESS; c++)
330 {
331 bip[c] = (XBYTE)iip[c];
332 }
333
334 return Set(bip);
335}
