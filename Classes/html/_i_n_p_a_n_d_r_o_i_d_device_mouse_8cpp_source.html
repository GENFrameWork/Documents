168bool INPANDROIDDEVICEMOUSE::AddPosition(int index, float x, float y, bool ispressed)
169{
170 if(!grpscreen) return false;
171
172 INPCURSOR* cursor = GetCursor(index);
173 if(!cursor) return false;
174
175 cursor->SetIsChanged(false);
176
177 if((cursor->GetX() != x) || (cursor->GetY() != y))
178 {
179 GRPANDROIDSCREEN* _grpscreen = (GRPANDROIDSCREEN*)grpscreen;
180
181 // to place the origin coordinates top left instead of bottom left
182 y = (_grpscreen->GetHeight()) - y;
183
184 cursor->Set(x,y);
185 cursor->SetIsChanged(true);
186 }
187
188 cursor->AddPointToMotion(ispressed);
189
190 return true;
191}

239{
