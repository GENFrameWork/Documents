161bool INPANDROIDDEVICEMOUSE::AddPosition(int index, float x, float y, bool ispressed)
162{
163 if(!grpscreen) return false;
164
165 INPCURSOR* cursor = GetCursor(index);
166 if(!cursor) return false;
167
168 cursor->SetIsChanged(false);
169
170 if((cursor->GetX() != x) || (cursor->GetY() != y))
171 {
172 GRPANDROIDSCREEN* _grpscreen = (GRPANDROIDSCREEN*)grpscreen;
173
174 // to place the origin coordinates top left instead of bottom left
175 y = (_grpscreen->GetHeight()) - y;
176
177 cursor->Set(x,y);
178 cursor->SetIsChanged(true);
179 }
180
181 cursor->AddPointToMotion(ispressed);
182
183 return true;
184}
