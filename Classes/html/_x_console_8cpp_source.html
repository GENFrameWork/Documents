348bool XCONSOLE::PrintDataBlock(XBYTE* data, XDWORD size, XDWORD marginsize, XDWORD sizeline, bool showoffset, bool showtext)
349{
350 XSTRING margin;
351 XSTRING strdata;
352 XDWORD _size = 0;
353 XDWORD _sizeline = sizeline;
354 int index = 0;
355
356 for(XDWORD c=0;c<marginsize;c++)
357 {
358 margin += __L(" ");
359 }
360
361 while(_size<size)
362 {
363 XSTRING string;
364
365 string = margin;
366 if(_sizeline > (size - _size)) _sizeline = (size - _size);
367
368 if(showoffset)
369 {
370 strdata.Format(__L("%04X"),index);
371 string += strdata;
372 string += __L(" ");
373 }
374
375 for(XDWORD c=0; c<_sizeline; c++)
376 {
377 strdata.Format(__L("%02X "),data[index]);
378 string += strdata;
379 _size++;
380 index++;
381 }
382
383 if(_sizeline != sizeline)
384 {
385 for(XDWORD c=0; c<(sizeline-_sizeline); c++)
386 {
387 strdata.Format(__L(" "));
388 string += strdata;
389 }
390 }
391
392 if(showtext)
393 {
394 index -= _sizeline;
395 string += __L(" ");
396
397 for(XDWORD c=0; c<_sizeline; c++)
398 {
399 XCHAR character = (XCHAR)data[index];
400
401 if((character<0x20) || (character>0x80) || (character==__C('%')) || (character==__C('\\'))) character = __C('.');
402
403 strdata.Format(__L("%c"),character);
404 string += strdata;
405 index++;
406 }
407 }
408
409 Print(string.Get());
410 Print(__L("\n"));
411
412 if(_size > XCONSOLE_MAXSIZEDATABLOCK) break;
413 }
414
415 return true;

512bool XCONSOLE::WaitKey(XCHAR* text, XDWORD margin, bool prelude, XDWORD timeout)
513{
