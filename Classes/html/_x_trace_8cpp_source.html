1781XDWORD XTRACE::GetTraceFromXBuffer(XBUFFER& xbufferpacket, XDWORD& publicIP, XDWORD& localIP, XBYTE& level, XDWORD& sequence, XDATETIME* xtime, XBUFFER& data)
1782{
1783 if(!xtime) return 1;
1784 if(!xbufferpacket.GetSize()) return 1;
1785
1786 XDWORD IDpacket = XTRACE_IDPACKET;
1787
1788 XDWORD sizepacketread = 0;
1789 XDWORD dword;
1790 XDWORD sizeinbuffer = xbufferpacket.GetSize();
1791 XBUFFER xbuffer(false);
1792 XDWORD sizeheader = (sizeof(XDWORD) + // ID.
1793 sizeof(XDWORD) + // SizePacket.
1794 sizeof(XDWORD) + // public IP address origin.
1795 sizeof(XDWORD) + // local IP address origin.
1796 sizeof(XBYTE) + // Level.
1797 sizeof(XDWORD) + // Sequence.
1798 (sizeof(XDWORD)*7) + // Date Time.
1799 sizeof(XWORD)); // CRC16
1800
1801 if(!sizeinbuffer) return 1;
1802 if(sizeinbuffer < sizeheader) return 1;
1803
1804
1805 while(sizeinbuffer-4)
1806 {
1807 xbufferpacket.Get(IDpacket, 0);
1808 if(IDpacket == XTRACE_IDPACKET) break;
1809
1810 xbufferpacket.Extract(NULL, 0, 1);
1811 sizeinbuffer--;
1812 }
1813
1814 if(IDpacket != XTRACE_IDPACKET) return 1;
1815
1816 if(!sizeinbuffer) return 1;
1817 if(sizeinbuffer < sizeheader) return 1;
1818
1819 if(!xbufferpacket.Get(sizepacketread, sizeof(IDpacket)))
1820 {
1821 return 1;
1822 }
1823
1824 if(!sizepacketread)
1825 {
1826 return 2;
1827 }
1828
1829 if(sizepacketread > sizeinbuffer)
1830 {
1831 return 1;
1832 }
1833
1834
1835 XDWORD sizeassing = sizepacketread;
1836 XDWORD sizeassingmin = 5*1024;
1837
1838 if(sizeassing < sizeassingmin) sizeassing = sizeassingmin;
1839 xbuffer.Resize(sizeassing);
1840
1841 xbufferpacket.Extract(xbuffer.Get(), 0, sizepacketread);
1842
1843 XWORD crc16;
1844 XTRACE_CRC16(xbuffer.Get(), (xbuffer.GetSize()-sizeof(XWORD)), crc16)
1845
1846 XWORD crc16read;
1847
1848 xbuffer.Get(crc16read, xbuffer.GetSize()-sizeof(XWORD));
1849 if(crc16 != crc16read)
1850 {
1851 return 2;
1852 }
1853
1854 xbuffer.SetPosition(0);
1855
1856 xbuffer.Get(IDpacket);
1857 xbuffer.Get(sizepacketread);
1858 xbuffer.Get(publicIP);
1859 xbuffer.Get(localIP);
1860 xbuffer.Get(level);
1861 xbuffer.Get(sequence);
1862
1863 xbuffer.Get(dword); xtime->SetYear(dword);
1864 xbuffer.Get(dword); xtime->SetMonth(dword);
1865 xbuffer.Get(dword); xtime->SetDay(dword);
1866 xbuffer.Get(dword); xtime->SetHours(dword);
1867 xbuffer.Get(dword); xtime->SetMinutes(dword);
1868 xbuffer.Get(dword); xtime->SetSeconds(dword);
1869 xbuffer.Get(dword); xtime->SetMilliSeconds(dword);
1870
1871 XDWORD sizestr = sizepacketread - sizeheader; // CRC16
1872 data.Empty();
1873 data.Add(&xbuffer.Get()[xbuffer.GetPosition()], sizestr);
1874
1875
1876 return 0;
1877}
