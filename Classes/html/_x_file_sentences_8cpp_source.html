105bool XFILESENTENCES::ConvertFileFromDBF(XPATH& xpath)
106{
107 XFILEDBF* filedbf;
108 XDWORD c;
109
110 CloseFile();
111
112 filedbf = new XFILEDBF();
113 if(!filedbf) return false;
114
115 if(!filedbf->Open(xpath))
116 {
117 delete filedbf;
118 return false;
119 }
120
121 XFILEDBF_HEADER* header = filedbf->GetHeader();
122
123 if((!header)||(!header->Load(filedbf->GetPrimaryFile())))
124 {
125 delete filedbf;
126 return false;
127 }
128
129 XDWORD nrecords = header->GetNRecords();
130
131 if(!nrecords)
132 {
133 delete filedbf;
134 return false;
135 }
136
137 XQWORD initablepos;
138
139 Create(this->xpath);
140
141 GetPrimaryFile()->Write((XBYTE*)&nrecords,sizeof(XDWORD));
142
143 GetPrimaryFile()->GetPosition(initablepos);
144
145 for(c=0;c<nrecords;c++)
146 {
147 XDWORD idx=0;
148 for(int d=0;d<2;d++)
149 {
150 GetPrimaryFile()->Write((XBYTE*)&idx,sizeof(XDWORD));
151 }
152 }
153
154 XDWORD index;
155 XQWORD fpos = 0;
156
157 for(c=0;c<nrecords;c++)
158 {
159 XFILEDBF_RECORD* record = filedbf->ReadRecord(c);
160 XSTRING* fsentence = new XSTRING(_MAXSTR);
161
162 if(fsentence)
163 {
164 fsentence->FillChar();
165 memcpy(fsentence->Get(),(char*)record->GetData(0),record->GetSizeField(0));
166 index = fsentence->ConvertToInt();
167
168 fsentence->FillChar();
169 memcpy(fsentence->Get(),(char*)record->GetData(1),record->GetSizeField(1));
170
171 fsentence->DeleteCharacter(__C(' '),XSTRINGCONTEXT_TO_END);
172
173 GetPrimaryFile()->GetPosition(fpos);
174
175 indexmap.Add(index, (XDWORD)fpos);
176
177 XDWORD size = (XDWORD)fsentence->GetSize();
178 GetPrimaryFile()->Write((XBYTE*)&size,sizeof(XDWORD));
179
180 XBUFFER xbufferexchange;
181 (*fsentence).ConvertToBufferExchange(xbufferexchange);
182 GetPrimaryFile()->Write(xbufferexchange.Get(), xbufferexchange.GetSize());
183
184
185 delete fsentence;
186 }
187
188 delete record;
189 }
190
191 GetPrimaryFile()->SetPosition(initablepos);
192
193 XDWORD ifilepos = 0;
194 XDWORD filepos = 0;
195
196 for(XDWORD c=0;c<nrecords;c++)
197 {
198 ifilepos = indexmap.GetKey(c);
199 filepos = indexmap.GetElement(c);
200
201 GetPrimaryFile()->Write((XBYTE*)&ifilepos ,sizeof(XDWORD));
202 GetPrimaryFile()->Write((XBYTE*)&filepos ,sizeof(XDWORD));
203 }
204
205 filedbf->Close();
206 delete filedbf;
207
208 CloseFile();
209
210 return OpenFile();
