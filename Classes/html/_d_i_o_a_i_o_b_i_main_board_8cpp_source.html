100bool DIOAIOBIMAINBOARD::Ini()
101{
102 for(int c=0; c<3; c++)
103 {
104 pcf8574[c] = new DIOI2CGPIOPCF8574();
105 if(!pcf8574[c]) return false;
106 }
107
108 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_IN]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_IN , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
109 {
110 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO IN not init!"));
111 return false;
112 }
113
114 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_OUT]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_OUT , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
115 {
116 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO OUT not init!"));
117 return false;
118 }
119
120 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_RELAYS]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_RELAYS , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
121 {
122 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO Relays not init!"));
123 return false;
124 }
125
126 for(int c=1; c<9; c++)
127 {
128 SetDigitalOutput(c, false);
129 }
130
131 eeprom = new DIOI2CEEPROM24XXX();
132 if(!eeprom) return false;
133
134 eeprom->SetType(DIOI2CEEPROM24XXXTYPE_16);
135
136 if(!eeprom->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME, DIOAIOBIMAINBOARD_I2CEEPROM, DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
137 {
138 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C Serial EEPROM not init!"));
139 return false;
140 }
141
142
143 pca9685 = new DIOI2CPWMCONTROLERPCA9685();
144 if(!pca9685) return false;
145
146 if(!pca9685->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME, DIOAIOBIMAINBOARD_I2CPWM, DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
147 {
148 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C PCA9685 (PWM) not init!"));
149 return false;
150 }
151
152
153 gpio = GEN_DIOFACTORY.CreateGPIO();
154 if(!gpio)
155 {
156 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! GPIO not created!"));
157 return false;
158 }
159
160 if(!gpio->Ini())
161 {
162 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! GPIO not init!"));
163 return false;
164 }
165
166 isini = true;
167
168 return isini;
169}

651bool DIOAIOBIMAINBOARD::ResetMicroControler()
652{
653 if(!SetModeHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, false)) return false; // Reset Micro
654 if(!SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, true)) return false;
655
656 bool pinstatus = false;
657
658 if(SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, pinstatus))
659 {
660 GEN_XSLEEP.MilliSeconds(10);
661
662 pinstatus = !pinstatus;
663
664 return SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, pinstatus);
665 }
666
667 return false;
668}

680bool DIOAIOBIMAINBOARD::End()
681{
682 for(int c=0; c<3; c++)
683 {
684 if(pcf8574[c])
685 {
686 pcf8574[c]->End();
687
688 delete pcf8574[c];
689 pcf8574[c] = NULL;
690 }
691 }
692
693 if(eeprom)
694 {
695 eeprom->End();
696 delete eeprom;
697 eeprom = NULL;
698 }
699
700
701 if(pca9685)
702 {
703 pca9685->GetDIOStream()->WaitToFlushXBuffers(3);
704 pca9685->End();
705 delete pca9685;
706 pca9685 = NULL;
707 }
708
709 if(gpio)
710 {
711 SetLedA(false);
712 SetLedB(false);
713
714 gpio->End();
715 GEN_DIOFACTORY.DeleteGPIO(gpio);
716 gpio = NULL;
717 }
718
719 return true;
720}
