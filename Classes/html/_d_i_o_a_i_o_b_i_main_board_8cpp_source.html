93bool DIOAIOBIMAINBOARD::Ini()
94{
95 for(int c=0; c<3; c++)
96 {
97 pcf8574[c] = new DIOI2CGPIOPCF8574();
98 if(!pcf8574[c]) return false;
99 }
100
101 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_IN]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_IN , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
102 {
103 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO IN not init!"));
104 return false;
105 }
106
107 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_OUT]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_OUT , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
108 {
109 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO OUT not init!"));
110 return false;
111 }
112
113 if(!pcf8574[DIOAIOBIMAINBOARD_I2CDIGIGPIO_RELAYS]->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME , DIOAIOBIMAINBOARD_I2CDIGIGPIO_ID_RELAYS , DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
114 {
115 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C GPIO Relays not init!"));
116 return false;
117 }
118
119 for(int c=1; c<9; c++)
120 {
121 SetDigitalOutput(c, false);
122 }
123
124 eeprom = new DIOI2CEEPROM24XXX();
125 if(!eeprom) return false;
126
127 eeprom->SetType(DIOI2CEEPROM24XXXTYPE_16);
128
129 if(!eeprom->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME, DIOAIOBIMAINBOARD_I2CEEPROM, DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
130 {
131 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C Serial EEPROM not init!"));
132 return false;
133 }
134
135
136 pca9685 = new DIOI2CPWMCONTROLERPCA9685();
137 if(!pca9685) return false;
138
139 if(!pca9685->Ini(DIOAIOBIMAINBOARD_I2CDEVICENAME, DIOAIOBIMAINBOARD_I2CPWM, DIOAIOBIMAINBOARD_DEFAULTTIMEOUT))
140 {
141 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! I2C PCA9685 (PWM) not init!"));
142 return false;
143 }
144
145
146 gpio = GEN_DIOFACTORY.CreateGPIO();
147 if(!gpio)
148 {
149 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! GPIO not created!"));
150 return false;
151 }
152
153 if(!gpio->Ini())
154 {
155 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOAIOBIMAINBOARD_LOGSECTIONID, false, __L("AIOBI Main Board: ERROR! GPIO not init!"));
156 return false;
157 }
158
159 isini = true;
160
161 return isini;
162}

644bool DIOAIOBIMAINBOARD::ResetMicroControler()
645{
646 if(!SetModeHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, false)) return false; // Reset Micro
647 if(!SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, true)) return false;
648
649 bool pinstatus = false;
650
651 if(SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, pinstatus))
652 {
653 GEN_XSLEEP.MilliSeconds(10);
654
655 pinstatus = !pinstatus;
656
657 return SetStatusHeaderPin(DIOAIOBIMAINBOARD_HEADER_SLOT, 9, pinstatus);
658 }
659
660 return false;
661}

673bool DIOAIOBIMAINBOARD::End()
674{
675 for(int c=0; c<3; c++)
676 {
677 if(pcf8574[c])
678 {
679 pcf8574[c]->End();
680
681 delete pcf8574[c];
682 pcf8574[c] = NULL;
683 }
684 }
685
686 if(eeprom)
687 {
688 eeprom->End();
689 delete eeprom;
690 eeprom = NULL;
691 }
692
693
694 if(pca9685)
695 {
696 pca9685->GetDIOStream()->WaitToFlushXBuffers(3);
697 pca9685->End();
698 delete pca9685;
699 pca9685 = NULL;
700 }
701
702 if(gpio)
703 {
704 SetLedA(false);
705 SetLedB(false);
706
707 gpio->End();
708 GEN_DIOFACTORY.DeleteGPIO(gpio);
709 gpio = NULL;
710 }
711
712 return true;
713}
