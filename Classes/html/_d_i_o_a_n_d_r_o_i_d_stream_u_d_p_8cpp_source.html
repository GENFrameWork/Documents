251int DIOANDROIDSTREAMUDP::IsReadyConnect(int socket)
252{
253 struct timeval tv;
254 int rc;
255 fd_set fdr;
256 fd_set fdw;
257 fd_set fds;
258
259 if(socket==-1) return -1;
260
261 FD_ZERO(&fdr);
262 FD_ZERO(&fdw);
263 FD_ZERO(&fds);
264
265 FD_SET(socket,&fdr);
266 FD_SET(socket,&fdw);
267 FD_SET(socket,&fds);
268
269 tv.tv_sec = 0;
270 tv.tv_usec = 100;
271
272 rc = select(socket+1, &fdr, &fdw, &fds, &tv);
273 if(rc==-1) return -1;
274
275 int status1 = FD_ISSET(socket,&fdr) ? 1 : 0;
276 int status2 = FD_ISSET(socket,&fdw) ? 1 : 0;
277 int status3 = FD_ISSET(socket,&fds) ? 1 : 0;
278
279 if(config->IsServer())
280 {
281 if(status1 || status2 ) return 1;
282 if(status3) return -1;
283 }
284 else
285 {
286 if((!status1) && status2) return 1;
287 if(status3) return -1;
288 }
289
290 return 0;
291}
