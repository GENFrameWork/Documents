343void Call_TerminateApplicationWithWindow(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
344{
345 if(!library) return;
346 if(!script) return;
347 if(!params) return;
348 if(!returnvalue) return;
349
350 returnvalue->Set();
351
352 if(params->GetSize()<2)
353 {
354 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
355 return;
356 }
357
358 XVECTOR<XPROCESS*> applist;
359 XSTRING appname = (*params->Get(0));
360 XSTRING windowstitle = (*params->Get(1));
361 int windowsposx = 0;
362 bool status = false;
363
364 if(GEN_XPROCESSMANAGER.Application_GetRunningList(applist, true))
365 {
366 for(XDWORD c=0; c<applist.GetSize(); c++)
367 {
368 if(applist.Get(c)->GetName()->Find(appname, true)!= XSTRING_NOTFOUND)
369 {
370 if(applist.Get(c)->GetWindowTitle()->Find(windowstitle, false) != XSTRING_NOTFOUND)
371 {
372 status = GEN_XPROCESSMANAGER.Application_Terminate((*applist.Get(c)));
373 break;
374 }
375 }
376 }
377 }
378
379 applist.DeleteContents();
380 applist.DeleteAll();
381
382 (*returnvalue) = status;
383}
