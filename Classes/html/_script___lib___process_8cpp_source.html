352void Call_TerminateApplicationWithWindow(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
353{
354 if(!library) return;
355 if(!script) return;
356 if(!params) return;
357 if(!returnvalue) return;
358
359 returnvalue->Set();
360
361 if(params->GetSize()<2)
362 {
363 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
364 return;
365 }
366
367 XVECTOR<XPROCESS*> applist;
368 XSTRING appname = (*params->Get(0));
369 XSTRING windowstitle = (*params->Get(1));
370 int windowsposx = 0;
371 bool status = false;
372
373 if(GEN_XPROCESSMANAGER.Application_GetRunningList(applist, true))
374 {
375 for(XDWORD c=0; c<applist.GetSize(); c++)
376 {
377 if(applist.Get(c)->GetName()->Find(appname, true)!= XSTRING_NOTFOUND)
378 {
379 if(applist.Get(c)->GetWindowTitle()->Find(windowstitle, false) != XSTRING_NOTFOUND)
380 {
381 status = GEN_XPROCESSMANAGER.Application_Terminate((*applist.Get(c)));
382 break;
383 }
384 }
385 }
386 }
387
388 applist.DeleteContents();
389 applist.DeleteAll();
390
391 (*returnvalue) = status;
392}
