617{
618 if(!part)
619 {
620 return false;
621 }
622
623 if(!namevar)
624 {
625 return false;
626 }
627
628 if(part->name.Compare(namevar, false))
629 {
630 return false;
631 }
632
633 int type;
634 XSTRING extname;
635 XSTRING* namevarext = NULL;
636 XVARIANT* variant = NULL;
637
638 va_list arg;
639
640 va_start(arg, namevar);
641
642 type = (int)va_arg(arg, int);
643 extname = (XCHAR*)va_arg(arg, XCHAR*);
644 namevarext = (XSTRING*)va_arg(arg, XSTRING*);
645 variant = (XVARIANT*)va_arg(arg, XVARIANT*);
646
647 va_end(arg);
648
649 int indexline = 0;
650
651 for(int c=0; c<GRPVECTORFILEDXFTEXTSECTIONHEADER_MAXNVAR; c++)
652 {
653 XSTRING* line = file->GetLine(part->iniline + indexline);
654 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
655
656 indexline++;
657
658 if(type == line->ConvertToInt())
659 {
660 line = file->GetLine(part->iniline + indexline);
661 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
662
663 indexline++;
664
665 namevarext->Format(__L("%s_%d"), part->name.Get(), type);
666
667 if(!extname.IsEmpty())
668 {
669 namevarext->AddFormat(__L("_%s"), extname.Get());
670 }
671
672 GetVariableFromLine(namevar, type, line, (*variant));
673
674 break;
675 }
676 }
677
678 return true;

693GRPVECTORFILERESULT GRPVECTORFILEDXFTEXTSECTIONHEADER::ParserTextSection(XFILETXT* fileTXT)
694{
695 XVECTOR<GRPVECTORFILEDXFTEXTPART*> parts;
696 GRPVECTORFILEDXFTEXTPART* part = NULL;
697 int c = iniline;
698
699 do{ if(!part)
700 {
701 part = new GRPVECTORFILEDXFTEXTPART ();
702 }
703
704 XSTRING* line = fileTXT->GetLine(c);
705 if(!line)
706 {
707 c++;
708 continue;
709 }
710
711 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
712
713 if(line && !line->Compare(__L("9"),true))
714 {
715 c++;
716 line = fileTXT->GetLine(c);
717 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
718
719 if(line->Get()[0]== __C('$'))
720 {
721 part->iniline = c+1;
722 part->name = line->Get();
723
724 if(parts.GetSize())
725 {
726 GRPVECTORFILEDXFTEXTPART* partbefore = parts.Get(parts.GetSize()-1);
727 partbefore->endline = c-2;
728 }
729
730 parts.Add(part);
731 part = NULL;
732
733 c++;
734 }
735
736 } else c++;
737
738 } while(c < endline);
739
740 if(parts.GetSize())
741 {
742 GRPVECTORFILEDXFTEXTPART* partbefore = parts.Get(parts.GetSize()-1);
743 partbefore->endline = c-1;
744 }
745
746 if(part && part->iniline == -1)
747 {
748 delete part;
749 part = NULL;
750 }
751
752 for(c=0; c<parts.GetSize(); c++)
753 {
754 part = parts.Get(c);
755 if(part)
756 {
757 if(!IsKnownVariable(part->name))
758 {
759 XSTRING message;
760
761 message.Format(__L("Header Variable Unknown [%s]"), part->name.Get());
762
763 GRPVECTORFILE_XEVENT vfevent(GetGrpVectorFile(), GRPVECTORFILE_XEVENTTYPE_PARTUNKNOWN);
764
765 vfevent.SetType(GRPVECTORFILETYPE_DXF);
766 vfevent.GetPath()->Set(__L(""));
767 vfevent.GetMsg()->Set(message);
768
769 PostEvent(&vfevent, GetGrpVectorFile());
770 }
771 else
772 {
773 XSTRING namevar[GRPVECTORFILEDXFTEXTSECTIONHEADER_MAXNVAR];
774 XVARIANT variant[GRPVECTORFILEDXFTEXTSECTIONHEADER_MAXNVAR];
775
776 for(int d=0; d<GRPVECTORFILEDXFTEXTSECTIONHEADER_MAXNDEFVAR; d++)
777 {
778 for(int e=0; e<GRPVECTORFILEDXFTEXTSECTIONHEADER::defvariable[d].nvalues; e++)
779 {
780 if(GRPVECTORFILEDXFTEXTSECTIONHEADER::defvariable[d].value[e].valuetype)
781 {
782 ParserVariable (fileTXT, part , GRPVECTORFILEDXFTEXTSECTIONHEADER::defvariable[d].name
783 , GRPVECTORFILEDXFTEXTSECTIONHEADER::defvariable[d].value[e].valuetype
784 , GRPVECTORFILEDXFTEXTSECTIONHEADER::defvariable[d].value[e].valuename
785 , &namevar[e]
786 , &variant[e]);
787 }
788 }
789 }
790
791 for(int d=0; d<GRPVECTORFILEDXFTEXTSECTIONHEADER_MAXNVAR; d++)
792 {
793 if((!namevar[d].IsEmpty()) && (variant->GetType() != XVARIANT_TYPE_NULL))
794 {
795 AddVariable(namevar[d].Get(), &variant[d]);
796 }
797 }
798 }
799 }
800 }
801
802 parts.DeleteContents();
803 parts.DeleteAll();
804
805 //config.SetHeader(&header);
806
807 #ifdef XTRACE_ACTIVE
808 //ShowTraceAllVariables();
809 #endif
810
811 return GRPVECTORFILERESULT_OK;
