511void CIPHERKEYSFILEPEM::HandleEvent_XBER(XBER_XEVENT* event)
512{
513 if(!decodekey)
514 {
515 return;
516 }
517
518 switch(event->GetEventType())
519 {
520 case XBERXEVENT_TYPE_DECODE_START : XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Cipher Keys File PEM] Start [%d]"), decodekey->GetType());
521 break;
522
523 case XBERXEVENT_TYPE_DECODE_DATA : { XSTRING levelsstr;
524
525 levelsstr = event->GetLevelsString()->Get();
526
527 if(decodekey->GetType() == CIPHERKEYTYPE_CERTIFICATE)
528 {
529 CIPHERKEYCERTIFICATE* certificate = (CIPHERKEYCERTIFICATE*)decodekey;
530
531 // Version
532 if(!levelsstr.Compare(__L("1.1.1.1")))
533 {
534 certificate->SetVersion((XWORD)event->GetData()->Get()[0]);
535 }
536
537 // Serial
538 if(!levelsstr.Compare(__L("1.1.2")))
539 {
540 certificate->GetSerial()->CopyFrom((*event->GetData()));
541 }
542
543 // Algorithm type
544 if(!levelsstr.Compare(__L("1.1.3.1")))
545 {
546 certificate->SetAlgorithmType(event->GetProperty()->OID);
547 }
548
549 // Date Not Before
550 if(!levelsstr.Compare(__L("1.1.5.1")))
551 {
552 XVARIANT* value;
553 XSTRING valuestr;
554
555 value = (XVARIANT*)event->GetValue();
556 valuestr = (XSTRING)(*value);
557
558 certificate->ConvertDateTime(valuestr.Get(), certificate->GetDateNotBefore());
559 }
560
561 // Date Not After
562 if(!levelsstr.Compare(__L("1.1.5.2")))
563 {
564 XVARIANT* value;
565 XSTRING valuestr;
566
567 value = (XVARIANT*)event->GetValue();
568 valuestr = (XSTRING)(*value);
569
570 certificate->ConvertDateTime(valuestr.Get(), certificate->GetDateNotAfter());
571 }
572
573 if(event->GetBeforeEvent()->GetTagType() == XBER_TAGTYPE_OBJECT_IDENTIFIER)
574 {
575 XSTRING _OID;
576 XVARIANT* value;
577
578 _OID = event->GetBeforeEvent()->GetProperty()->OID;
579
580 value = (XVARIANT*)event->GetValue();
581
582 if(!_OID.Compare(__L("2.5.4.6"), false))
583 {
584 certificate->GetCountryName()->Set((XSTRING)(*value));
585 }
586
587 if(!_OID.Compare(__L("2.5.4.10"), false))
588 {
589 certificate->GetOrganizationName()->Set((XSTRING)(*value));
590 }
591
592 if(!_OID.Compare(__L("2.5.4.11"), false))
593 {
594 certificate->GetOrganizationalUnitName()->Set((XSTRING)(*value));
595 }
596
597 if(!_OID.Compare(__L("2.5.4.3"), false))
598 {
599 certificate->GetCommonName()->Set((XSTRING)(*value));
600 }
601 }
602 }
603
604 levelsstr.AddFormat(__L(" %s"), event->GetLine()->Get());
605
606 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, levelsstr.Get(), NULL);
607 }
608 break;
609
610 case XBERXEVENT_TYPE_DECODE_END : XTRACE_PRINTCOLOR((event->GetStatus()?XTRACE_COLOR_BLUE:XTRACE_COLOR_RED), __L("[Cipher Keys File PEM] End"));
611 break;
612
613 }
614}
