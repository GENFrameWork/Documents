264bool DIOSCRAPERWEB::Do(XCHAR* namewebservice, int timeoutforurl, XSTRING* localIP)
265{
266 if(!xml) return false;
267
268 XFILEXMLELEMENT* root = xml->GetRoot();
269 if(!root) return false;
270
271 XSTRING namews;
272
273 XFILEXMLELEMENT* nodewebservice = NULL;
274
275 for(int c=0; c<root->GetNElements(); c++)
276 {
277 nodewebservice = root->GetElement(c);
278 if(nodewebservice)
279 {
280 namews = nodewebservice->GetValueAttribute(__L("name"));
281 if(!namews.Compare(namewebservice)) break;
282
283 nodewebservice = NULL;
284 }
285 }
286
287 if(nodewebservice)
288 {
289 DeleteAllValues();
290
291 for(int c=0; c<nodewebservice->GetNElements(); c++)
292 {
293 bool priorityvaluesobtain = true;
294
295 XFILEXMLELEMENT* nodeweb = nodewebservice->GetElement(c);
296 if(nodeweb)
297 {
298 XBUFFER webpage;
299 DIOURL url;
300
301 if(!ChangeURL(nodeweb->GetValueAttribute(__L("url")), url))
302 {
303 return false;
304 }
305
306 if(webclient->Get(url, webpage, DIOSCRAPERWEB_DEFAULTUSERAGENT, timeoutforurl, localIP))
307 {
308 if(webpage.GetSize())
309 {
310 XSTRING format;
311
312 format = nodeweb->GetValueAttribute(__L("format"));
313
314 if(!format.Compare(__L("JSON"),true))
315 {
316 XFILEJSON* json = new XFILEJSON();
317 if(json)
318 {
319 json->DeleteAllLines();
320 json->AddBufferLines(XFILETXTFORMATCHAR_ASCII, webpage);
321 json->DecodeAllLines();
322
323 for(int d=0; d<nodeweb->GetNElements(); d++)
324 {
325 XFILEXMLELEMENT* nodevalue = nodeweb->GetElement(d);
326 if(nodevalue)
327 {
328 XSTRING name;
329 XSTRING prioritystring;
330 int priority;
331 XSTRING namevalue;
332
333 name = nodevalue->GetValueAttribute(__L("name"));
334
335 prioritystring = nodevalue->GetValueAttribute(__L("priority"));
336 priority = prioritystring.ConvertToInt();
337
338 namevalue = nodevalue->GetValueAttribute(__L("namevalue"));
339
340 XFILEJSONVALUE* jsonvalue = json->GetValue(namevalue,NULL);
341 if(jsonvalue)
342 {
343 switch(jsonvalue->GetType())
344 {
345 case XFILEJSONVALUETYPE_INTEGER : { int pvalue = (int)jsonvalue->GetValueInteger();
346 XSTRING string;
347
348 string.Format(__L("%d"),pvalue);
349
350 AddValue(name, string);
351 }
352 break;
353
354 case XFILEJSONVALUETYPE_DOUBLEINTEGER : { long pvalue = (long)jsonvalue->GetValueDoubleInteger();
355 XSTRING string;
356
357 string.Format(__L("%ld"),pvalue);
358
359 AddValue(name, string);
360 }
361 break;
362
363 case XFILEJSONVALUETYPE_FLOAT : { float pvalue = (float)jsonvalue->GetValueFloating();
364 XSTRING string;
365
366 string.Format(__L("%f"),pvalue);
367
368 AddValue(name, string);
369 }
370 break;
371
372 case XFILEJSONVALUETYPE_DOUBLEFLOAT : { double pvalue = (float)jsonvalue->GetValueDoubleFloat();
373 XSTRING string;
374
375 string.Format(__L("%f"),pvalue);
376
377 AddValue(name, string);
378 }
379 break;
380
381 case XFILEJSONVALUETYPE_STRING : { XSTRING pvalue = (XSTRING)jsonvalue->GetValueString();
382 AddValue(name, pvalue);
383 }
384 break;
385
386 case XFILEJSONVALUETYPE_OBJECT : break;
387 case XFILEJSONVALUETYPE_ARRAY : break;
388 case XFILEJSONVALUETYPE_BOOLEAN : break;
389 case XFILEJSONVALUETYPE_NULL : break;
390
391 }
392 }
393 else
394 {
395 if(priority) priorityvaluesobtain = false;
396 }
397 }
398 }
399
400 delete json;
401 }
402 }
403
404 if(!format.Compare(__L("HTML"),true))
405 {
406 for(int d=0; d<nodeweb->GetNElements(); d++)
407 {
408 XFILEXMLELEMENT* nodevalue = nodeweb->GetElement(d);
409 if(nodevalue)
410 {
411 XSTRING searchini;
412 XSTRING searchend;
413 XSTRING namevalue;
414 XSTRING prioritystring;
415 int priority;
416
417 XSTRING value;
418
419 namevalue = nodevalue->GetValueAttribute(__L("name"));
420 namevalue.ConvertHexFormatChars();
421
422 prioritystring = nodevalue->GetValueAttribute(__L("priority"));
423 priority = prioritystring.ConvertToInt();
424
425 searchini = nodevalue->GetValueAttribute(__L("start"));
426 searchini.ConvertHexFormatChars();
427
428 searchend = nodevalue->GetValueAttribute(__L("end"));;
429 searchend.ConvertHexFormatChars();
430
431 if(ExtractString(searchini, searchend, &webpage, false, value))
432 {
433 if(!GetValue(namevalue))
434 {
435 AddValue(namevalue, value);
436 }
437 }
438 else
439 {
440 if(priority) priorityvaluesobtain = false;
441 }
442 }
443 }
444 }
445
446 if(values.GetSize() && priorityvaluesobtain)
447 {
448 return true;
449 }
450 }
451 }
452 }
453 }
454 }
455
456 return false;
457}

777bool DIOSCRAPERWEB::ExtractString(XSTRING& searchini, XSTRING& searchend, XBUFFER* xbuffer, bool usebufferpos, XSTRING& result)
778{
779 result.Empty();
780
781 if(!xbuffer) return false;
782
783 if(searchini.IsEmpty()) return false;
784 if(searchend.IsEmpty()) return false;
785
786 XSTRING html;
787 int indexini;
788 int indexend;
789
790 //html = (char*)xbuffer->Get();
791 html = *xbuffer; // #Imanol, changed to avoid a buffer overrun
792
793 indexini = html.Find(searchini,false,usebufferpos?xbuffer->GetPosition():0);
794 if(indexini==XSTRING_NOTFOUND) return false;
795
796 indexend = html.Find(searchend,false,indexini + searchini.GetSize());
797 if(indexend==XSTRING_NOTFOUND) return false;
798
799 if((int)(indexini + searchini.GetSize())>=indexend) return false;
800
801 html.Copy(indexini + searchini.GetSize(),indexend,result);
802
803 result.DeleteCharacter(__C(' '),XSTRINGCONTEXT_FROM_FIRST);
804 result.DeleteCharacter(__C(' '),XSTRINGCONTEXT_TO_END);
805
806 if(result.IsEmpty()) return false;
807
808 return true;
809}
