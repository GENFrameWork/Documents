417bool DIOMODBUS_CLIENT::MakeCommand(DIOMODBUS_CLIENT_FUNCTION function, XBUFFER& ask, XBUFFER& answer, int timeout)
418{
419 if(!diostream) return false;
420 if(diostream->GetStatus() != DIOSTREAMSTATUS_CONNECTED) return false;
421
422 XBUFFER datawrite;
423
424 answer.Delete();
425
426 bool status = false;
427
428 switch(mode)
429 {
430 case DIOMODBUS_CLIENTMODE_UNKNOWN : break;
431
432 case DIOMODBUS_CLIENTMODE_RTU : { datawrite.Add((XBYTE) unit);
433 datawrite.Add((XBYTE) function);
434 datawrite.Add(ask);
435
436 XWORD crc16 = CRC16(datawrite);
437
438 datawrite.Add((XWORD)crc16);
439
440 diostream->Write(datawrite);
441
442 if(!diostream->WaitToFlushOutXBuffer(timeout)) return false;
443 if(!diostream->WaitToFilledReadingBuffer(3, timeout)) return false;
444
445 if(diostream->GetInXBuffer()->GetByte(0) != unit) return false;
446 if(diostream->GetInXBuffer()->GetByte(1) != function) return false;
447
448 XDWORD sizeanswer = 0;
449
450 switch(function)
451 {
452 case DIOMODBUS_CLIENT_FUNCTION_READ_COILS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
453 case DIOMODBUS_CLIENT_FUNCTION_READ_DISCRETE_INPUTS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
454 case DIOMODBUS_CLIENT_FUNCTION_READ_HOLDING_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
455 case DIOMODBUS_CLIENT_FUNCTION_READ_INPUT_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
456 case DIOMODBUS_CLIENT_FUNCTION_WRITE_SINGLE_COIL : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
457 case DIOMODBUS_CLIENT_FUNCTION_WRITE_SINGLE_REGISTER : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
458 case DIOMODBUS_CLIENT_FUNCTION_WRITE_MULTIPLE_COILS : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
459 case DIOMODBUS_CLIENT_FUNCTION_WRITE_MULTIPLE_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
460 case DIOMODBUS_CLIENT_FUNCTION_READ_FILE_RECORD : break;
461 case DIOMODBUS_CLIENT_FUNCTION_WRITE_FILE_RECORD : break;
462 case DIOMODBUS_CLIENT_FUNCTION_MASK_WRITE_REGISTER : break;
463 case DIOMODBUS_CLIENT_FUNCTION_READWRITE_MULTIPLE_REGISTERS : break;
464 case DIOMODBUS_CLIENT_FUNCTION_READ_FIFO_QUEUE : break;
465 case DIOMODBUS_CLIENT_FUNCTION_ENCAPSULATED_INTERFACE_TRANSPORT : break;
466
467 }
468
469 if(!sizeanswer) break;
470
471 if(!diostream->WaitToFilledReadingBuffer(sizeanswer, timeout)) return false;
472
473 answer.Resize(sizeanswer);
474
475 if(!diostream->Read(answer)) return false;
476
477 XWORD CRCread = 0;
478 XWORD CRCcheck;
479
480 answer.Extract(CRCread, sizeanswer-2);
481
482 CRCcheck = CRC16(answer);
483
484 if(CRCcheck != CRCread) return false;
485
486 status = true;
487 }
488 break;
489
490 case DIOMODBUS_CLIENTMODE_ASCII : break;
491
492 case DIOMODBUS_CLIENTMODE_TCPIP : break;
493 }
494
495 return status;
496}

510XWORD DIOMODBUS_CLIENT::CRC16(XBUFFER& data)
511{
512 XWORD poly16 = 0xA001;
513 XWORD LSB;
514 XWORD crc = 0xFFFF;
515
516 for(int d=0; d<(int)data.GetSize(); d++)
517 {
518 crc = ((crc^data.Get()[d]) | 0xFF00) & (crc | 0x00FF);
519
520 for(int c=0; c<8; c++)
521 {
522 LSB=(crc & 0x0001);
523 crc=crc/2;
524
525 if(LSB) crc = crc ^ poly16;
526 }
527 }
528
529 SWAPWORD(crc);
530
531 return crc;
532}
