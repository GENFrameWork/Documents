410bool DIOMODBUS_CLIENT::MakeCommand(DIOMODBUS_CLIENT_FUNCTION function, XBUFFER& ask, XBUFFER& answer, int timeout)
411{
412 if(!diostream) return false;
413 if(diostream->GetConnectStatus() != DIOSTREAMSTATUS_CONNECTED) return false;
414
415 XBUFFER datawrite;
416
417 answer.Delete();
418
419 bool status = false;
420
421 switch(mode)
422 {
423 case DIOMODBUS_CLIENTMODE_UNKNOWN : break;
424
425 case DIOMODBUS_CLIENTMODE_RTU : { datawrite.Add((XBYTE) unit);
426 datawrite.Add((XBYTE) function);
427 datawrite.Add(ask);
428
429 XWORD crc16 = CRC16(datawrite);
430
431 datawrite.Add((XWORD)crc16);
432
433 diostream->Write(datawrite);
434
435 if(!diostream->WaitToFlushOutXBuffer(timeout)) return false;
436 if(!diostream->WaitToFilledReadingBuffer(3, timeout)) return false;
437
438 if(diostream->GetInXBuffer()->GetByte(0) != unit) return false;
439 if(diostream->GetInXBuffer()->GetByte(1) != function) return false;
440
441 XDWORD sizeanswer = 0;
442
443 switch(function)
444 {
445 case DIOMODBUS_CLIENT_FUNCTION_READ_COILS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
446 case DIOMODBUS_CLIENT_FUNCTION_READ_DISCRETE_INPUTS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
447 case DIOMODBUS_CLIENT_FUNCTION_READ_HOLDING_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
448 case DIOMODBUS_CLIENT_FUNCTION_READ_INPUT_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERVAR; break;
449 case DIOMODBUS_CLIENT_FUNCTION_WRITE_SINGLE_COIL : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
450 case DIOMODBUS_CLIENT_FUNCTION_WRITE_SINGLE_REGISTER : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
451 case DIOMODBUS_CLIENT_FUNCTION_WRITE_MULTIPLE_COILS : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
452 case DIOMODBUS_CLIENT_FUNCTION_WRITE_MULTIPLE_REGISTERS : DIOMODBUS_CLIENTMODE_SIZEANSWERFIX; break;
453 case DIOMODBUS_CLIENT_FUNCTION_READ_FILE_RECORD : break;
454 case DIOMODBUS_CLIENT_FUNCTION_WRITE_FILE_RECORD : break;
455 case DIOMODBUS_CLIENT_FUNCTION_MASK_WRITE_REGISTER : break;
456 case DIOMODBUS_CLIENT_FUNCTION_READWRITE_MULTIPLE_REGISTERS : break;
457 case DIOMODBUS_CLIENT_FUNCTION_READ_FIFO_QUEUE : break;
458 case DIOMODBUS_CLIENT_FUNCTION_ENCAPSULATED_INTERFACE_TRANSPORT : break;
459
460 }
461
462 if(!sizeanswer) break;
463
464 if(!diostream->WaitToFilledReadingBuffer(sizeanswer, timeout)) return false;
465
466 answer.Resize(sizeanswer);
467
468 if(!diostream->Read(answer)) return false;
469
470 XWORD CRCread = 0;
471 XWORD CRCcheck;
472
473 answer.Extract(CRCread, sizeanswer-2);
474
475 CRCcheck = CRC16(answer);
476
477 if(CRCcheck != CRCread) return false;
478
479 status = true;
480 }
481 break;
482
483 case DIOMODBUS_CLIENTMODE_ASCII : break;
484
485 case DIOMODBUS_CLIENTMODE_TCPIP : break;
486 }
487
488 return status;
489}

503XWORD DIOMODBUS_CLIENT::CRC16(XBUFFER& data)
504{
505 XWORD poly16 = 0xA001;
506 XWORD LSB;
507 XWORD crc = 0xFFFF;
508
509 for(int d=0; d<(int)data.GetSize(); d++)
510 {
511 crc = ((crc^data.Get()[d]) | 0xFF00) & (crc | 0x00FF);
512
513 for(int c=0; c<8; c++)
514 {
515 LSB=(crc & 0x0001);
516 crc=crc/2;
517
518 if(LSB) crc = crc ^ poly16;
519 }
520 }
521
522 SWAPWORD(crc);
523
524 return crc;
525}
