316bool DIODNSRESOLVER::ResolveURL(XCHAR* URL, DIOIP& IPresolved, int querytype, XDWORD timeout)
317{
318 if(xmutexresolved)
319 {
320 //xmutexresolved->Lock();
321 }
322
323 if(Host_FindIP(URL, IPresolved))
324 {
325 if(xmutexresolved)
326 {
327 //xmutexresolved->UnLock();
328 }
329
330 return true;
331 }
332
333 if(!listhostresolved.GetSize())
334 {
335 GEN_DIODNSRESOLVER.Host_Add(__L("localhost"), __L("127.0.0.1"));
336 }
337
338 DIOURL url;
339
340 url = URL;
341
342 if(!url.IsAURL())
343 {
344 IPresolved.Set(URL);
345 }
346 else
347 {
348 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer());
349 if(xtimerout)
350 {
351 bool status;
352
353 if(!listDNSservers.GetSize())
354 {
355 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("8.8.8.8"));
356 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("8.8.4.4"));
357 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("64.6.64.6"));
358 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("64.6.65.6"));
359 }
360
361 DIODNSPROTOCOLCLIENT* dnsprotocolclient = new DIODNSPROTOCOLCLIENT();
362 if(dnsprotocolclient)
363 {
364 for(XDWORD c=0; c<listDNSservers.GetSize(); c++)
365 {
366 if(xtimerout->GetMeasureSeconds() > (timeout * listDNSservers.GetSize()))
367 {
368 status = false;
369 break;
370 }
371
372 DIODNSRESOLVER_DNSSERVER* serverDNS = listDNSservers.Get(c);
373 if(serverDNS)
374 {
375 XSTRING serverIP;
376
377 serverDNS->GetIP()->GetXString(serverIP);
378
379 dnsprotocolclient->SetServer(serverIP.Get(), serverDNS->GetPort());
380
381 status = dnsprotocolclient->ResolveURL(URL, IPresolved, querytype, timeout);
382 if(status)
383 {
384 break;
385 }
386 }
387 }
388
389 delete dnsprotocolclient;
390 }
391
392 GEN_XFACTORY.DeleteTimer(xtimerout);
393 }
394 }
395
396 if(xmutexresolved)
397 {
398 //xmutexresolved->UnLock();
399 }
400
401 return true;
402}

456bool DIODNSRESOLVER::DNSServer_AddDNSServer(XCHAR* server)
457{
458 if(!server) return false;
459
460 DIODNSRESOLVER_DNSSERVER* DNSserver = new DIODNSRESOLVER_DNSSERVER();
461 if(!DNSserver) return false;
462
463 XSTRING string;
464 XSTRING IPstring;
465 int port = 0;
466
467 string = server;
468 IPstring.AdjustSize(_MAXSTR);
469 string.UnFormat(__L("%s:%d"), IPstring.Get(), &port);
470 IPstring.AdjustSize();
471
472 if(IPstring.IsEmpty())
473 {
474 delete DNSserver;
475
476 return false;
477 }
478
479 DNSserver->GetIP()->Set(IPstring);
480 DNSserver->SetPort(port);
481
482 if(DNSServer_IsIPonList((*DNSserver->GetIP())))
483 {
484 delete DNSserver;
485
486 return false;
487 }
488
489 return listDNSservers.Add(DNSserver);
490}
