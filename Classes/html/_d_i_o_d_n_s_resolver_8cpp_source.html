309bool DIODNSRESOLVER::ResolveURL(XCHAR* URL, DIOIP& IPresolved, int querytype, XDWORD timeout)
310{
311 if(xmutexresolved)
312 {
313 //xmutexresolved->Lock();
314 }
315
316 if(Host_FindIP(URL, IPresolved))
317 {
318 if(xmutexresolved)
319 {
320 //xmutexresolved->UnLock();
321 }
322
323 return true;
324 }
325
326 if(!listhostresolved.GetSize())
327 {
328 GEN_DIODNSRESOLVER.Host_Add(__L("localhost"), __L("127.0.0.1"));
329 }
330
331 DIOURL url;
332
333 url = URL;
334
335 if(!url.IsAURL())
336 {
337 IPresolved.Set(URL);
338 }
339 else
340 {
341 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer());
342 if(xtimerout)
343 {
344 bool status;
345
346 if(!listDNSservers.GetSize())
347 {
348 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("8.8.8.8"));
349 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("8.8.4.4"));
350 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("64.6.64.6"));
351 GEN_DIODNSRESOLVER.DNSServer_AddDNSServer(__L("64.6.65.6"));
352 }
353
354 DIODNSPROTOCOLCLIENT* dnsprotocolclient = new DIODNSPROTOCOLCLIENT();
355 if(dnsprotocolclient)
356 {
357 for(XDWORD c=0; c<listDNSservers.GetSize(); c++)
358 {
359 if(xtimerout->GetMeasureSeconds() > (timeout * listDNSservers.GetSize()))
360 {
361 status = false;
362 break;
363 }
364
365 DIODNSRESOLVER_DNSSERVER* serverDNS = listDNSservers.Get(c);
366 if(serverDNS)
367 {
368 XSTRING serverIP;
369
370 serverDNS->GetIP()->GetXString(serverIP);
371
372 dnsprotocolclient->SetServer(serverIP.Get(), serverDNS->GetPort());
373
374 status = dnsprotocolclient->ResolveURL(URL, IPresolved, querytype, timeout);
375 if(status)
376 {
377 break;
378 }
379 }
380 }
381
382 delete dnsprotocolclient;
383 }
384
385 GEN_XFACTORY.DeleteTimer(xtimerout);
386 }
387 }
388
389 if(xmutexresolved)
390 {
391 //xmutexresolved->UnLock();
392 }
393
394 return true;
395}

449bool DIODNSRESOLVER::DNSServer_AddDNSServer(XCHAR* server)
450{
451 if(!server) return false;
452
453 DIODNSRESOLVER_DNSSERVER* DNSserver = new DIODNSRESOLVER_DNSSERVER();
454 if(!DNSserver) return false;
455
456 XSTRING string;
457 XSTRING IPstring;
458 int port = 0;
459
460 string = server;
461 IPstring.AdjustSize(_MAXSTR);
462 string.UnFormat(__L("%s:%d"), IPstring.Get(), &port);
463 IPstring.AdjustSize();
464
465 if(IPstring.IsEmpty())
466 {
467 delete DNSserver;
468
469 return false;
470 }
471
472 DNSserver->GetIP()->Set(IPstring);
473 DNSserver->SetPort(port);
474
475 if(DNSServer_IsIPonList((*DNSserver->GetIP())))
476 {
477 delete DNSserver;
478
479 return false;
480 }
481
482 return listDNSservers.Add(DNSserver);
483}
