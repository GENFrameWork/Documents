149bool DIOSTREAMSPICONFIG::SetLocalDeviceNameByPort()
150{
151 int revision = 0;
152
153 switch(GEN_XSYSTEM.GetTypeHardware(&revision))
154 {
155 case XSYSTEM_HARDWARETYPE_PC :
156 #ifdef LINUX
157 localdevicename.Format(__L("/dev/spi%d"), port-1);
158 #endif
159
160 #ifdef WINDOWS
161 localdevicename.Format(__L("spi%d"), port-1);
162 #endif
163 break;
164
165 case XSYSTEM_HARDWARETYPE_RASPBERRYPI : localdevicename.Format(__L("/dev/spidev%d.%d"), port-1, chipselect);
166 break;
167
168 default : break;
169 }
170
171 return true;
172}
