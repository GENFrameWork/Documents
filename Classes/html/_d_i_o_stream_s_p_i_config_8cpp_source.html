142bool DIOSTREAMSPICONFIG::SetLocalDeviceNameByPort()
143{
144 int revision = 0;
145
146 switch(GEN_XSYSTEM.GetTypeHardware(&revision))
147 {
148 case XSYSTEM_HARDWARETYPE_PC :
149 #ifdef LINUX
150 localdevicename.Format(__L("/dev/spi%d"), port-1);
151 #endif
152
153 #ifdef WINDOWS
154 localdevicename.Format(__L("spi%d"), port-1);
155 #endif
156 break;
157
158 case XSYSTEM_HARDWARETYPE_RASPBERRYPI : localdevicename.Format(__L("/dev/spidev%d.%d"), port-1, chipselect);
159 break;
160
161 default : break;
162 }
163
164 return true;
165}
