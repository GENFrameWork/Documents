288void HASHCOMP128V1::Make(XBYTE* input)
289{
290 XBYTE x[32];
291 XBYTE bit[128];
292 int i;
293 int j;
294 int k;
295 int l;
296 int m;
297 int n;
298 int y;
299 int z;
300 int next_bit;
301
302 // Load RAND into last 16 bytes of input
303 for(i=16; i<32; i++)
304 {
305 x[i] = input[i];
306 }
307
308 // Loop eight times
309 for(i=1;i<9;i++)
310 {
311 // Load ki into first 16 bytes of input
312 for(j=0;j<16;j++)
313 {
314 x[j] = input[j];
315 }
316
317 // Perform substitutions
318 for(j=0;j<5;j++)
319 {
320 for(k=0; k<(1<<j);k++)
321 {
322 for(l=0; l<(1<<(4-j));l++)
323 {
324 m = l + k*(1<<(5-j));
325 n = m + (1<<(4-j));
326 y = (x[m]+2*x[n]) % (1<<(9-j));
327 z = (2*x[m]+x[n]) % (1<<(9-j));
328 x[m] = table[j][y];
329 x[n] = table[j][z];
330 }
331 }
332 }
333
334 /* ( Form bits from bytes ) */
335 for(j=0;j<32;j++)
336 {
337 for(k=0;k<4;k++)
338 {
339 bit[4*j+k] = (x[j]>>(3-k)) & 1;
340 }
341 }
342
343 /* ( Permutation but not on the last loop ) */
344 if(i<8)
345 {
346 for(j=0;j<16;j++)
347 {
348 x[j+16] = 0;
349 for (k=0;k<8;k++)
350 {
351 next_bit = ((8*j + k)*17) % 128;
352 x[j+16] |= bit[next_bit] << (7-k);
353 }
354 }
355 }
356 }
357
358 /*
359 * ( At this stage the vector x[] consists of 32 nibbles.
360 * The first 8 of these are taken as the output SRES. )
361 */
362
363 /* The remainder of the code is not given explicitly in the
364 * standard, but was derived by reverse-engineering.
365 */
366
367
368 for(i=0;i<4;i++)
369 {
370 digest[i] = (x[2*i]<<4) | x[2*i+1];
371 }
372
373 for(i=0;i<6;i++)
374 {
375 digest[4+i] = (x[2*i+18]<<6) | (x[2*i+18+1]<<2) | (x[2*i+18+2]>>2);
376 }
377
378 digest[4+6] = (x[2*6+18]<<6) | (x[2*6+18+1]<<2);
379 digest[4+7] = 0;
380}
