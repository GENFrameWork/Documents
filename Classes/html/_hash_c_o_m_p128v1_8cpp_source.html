295void HASHCOMP128V1::Make(XBYTE* input)
296{
297 XBYTE x[32];
298 XBYTE bit[128];
299 int i;
300 int j;
301 int k;
302 int l;
303 int m;
304 int n;
305 int y;
306 int z;
307 int next_bit;
308
309 // Load RAND into last 16 bytes of input
310 for(i=16; i<32; i++)
311 {
312 x[i] = input[i];
313 }
314
315 // Loop eight times
316 for(i=1;i<9;i++)
317 {
318 // Load ki into first 16 bytes of input
319 for(j=0;j<16;j++)
320 {
321 x[j] = input[j];
322 }
323
324 // Perform substitutions
325 for(j=0;j<5;j++)
326 {
327 for(k=0; k<(1<<j);k++)
328 {
329 for(l=0; l<(1<<(4-j));l++)
330 {
331 m = l + k*(1<<(5-j));
332 n = m + (1<<(4-j));
333 y = (x[m]+2*x[n]) % (1<<(9-j));
334 z = (2*x[m]+x[n]) % (1<<(9-j));
335 x[m] = table[j][y];
336 x[n] = table[j][z];
337 }
338 }
339 }
340
341 /* ( Form bits from bytes ) */
342 for(j=0;j<32;j++)
343 {
344 for(k=0;k<4;k++)
345 {
346 bit[4*j+k] = (x[j]>>(3-k)) & 1;
347 }
348 }
349
350 /* ( Permutation but not on the last loop ) */
351 if(i<8)
352 {
353 for(j=0;j<16;j++)
354 {
355 x[j+16] = 0;
356 for (k=0;k<8;k++)
357 {
358 next_bit = ((8*j + k)*17) % 128;
359 x[j+16] |= bit[next_bit] << (7-k);
360 }
361 }
362 }
363 }
364
365 /*
366 * ( At this stage the vector x[] consists of 32 nibbles.
367 * The first 8 of these are taken as the output SRES. )
368 */
369
370 /* The remainder of the code is not given explicitly in the
371 * standard, but was derived by reverse-engineering.
372 */
373
374
375 for(i=0;i<4;i++)
376 {
377 digest[i] = (x[2*i]<<4) | x[2*i+1];
378 }
379
380 for(i=0;i<6;i++)
381 {
382 digest[4+i] = (x[2*i+18]<<6) | (x[2*i+18+1]<<2) | (x[2*i+18+2]>>2);
383 }
384
385 digest[4+6] = (x[2*6+18]<<6) | (x[2*6+18+1]<<2);
386 digest[4+7] = 0;
387}
