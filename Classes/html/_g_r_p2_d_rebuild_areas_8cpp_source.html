373bool GRP2DREBUILDAREAS::CreateRebuildArea(double x, double y, double width, double height, void* extra_data)
374{
375 double _x = (x - excessedge);
376 double _y = (y - excessedge);
377 double _width = (width + (excessedge*2));
378 double _height = (height + (excessedge*2));
379
380 if(_x < 0) _x = 0;
381 if(_y < 0) _y = 0;
382
383 GRPBITMAP* bitmap = GetBitmap(_x, _y, _width, _height);
384 if(!bitmap) return false;
385
386 GRP2DREBUILDAREA* newarea = new GRP2DREBUILDAREA();
387 if(!newarea)
388 {
389 delete bitmap;
390 return false;
391 }
392
393 newarea->SetXPos(_x);
394 newarea->SetYPos(_y);
395 newarea->SetBitmap(bitmap);
396 newarea->SetExtraData(extra_data);
397
398 return areas.Add(newarea);
399}
