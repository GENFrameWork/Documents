380bool GRP2DREBUILDAREAS::CreateRebuildArea(double x, double y, double width, double height, void* extra_data)
381{
382 double _x = (x - excessedge);
383 double _y = (y - excessedge);
384 double _width = (width + (excessedge*2));
385 double _height = (height + (excessedge*2));
386
387 if(_x < 0) _x = 0;
388 if(_y < 0) _y = 0;
389
390 GRPBITMAP* bitmap = GetBitmap(_x, _y, _width, _height);
391 if(!bitmap) return false;
392
393 GRP2DREBUILDAREA* newarea = new GRP2DREBUILDAREA();
394 if(!newarea)
395 {
396 delete bitmap;
397 return false;
398 }
399
400 newarea->SetXPos(_x);
401 newarea->SetYPos(_y);
402 newarea->SetBitmap(bitmap);
403 newarea->SetExtraData(extra_data);
404
405 return areas.Add(newarea);
406}
