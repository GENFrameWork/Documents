171GRPVECTORFILE* GRPVECTORFILE::CreateInstance(XPATH& pathfile)
172{
173 if(!GRPVECTORFILE::DetectFile(pathfile))
174 {
175 return NULL;
176 }
177
178 GRPVECTORFILE* vectorFile = NULL;
179 GRPVECTORFILETYPE list[] = { GRPVECTORFILETYPE_DXF
180 };
181
182 for(int c=0; c < (sizeof(list)/sizeof(GRPVECTORFILETYPE)); c++)
183 {
184 vectorFile = GRPVECTORFILE::CreateInstance(list[c]);
185 if(vectorFile)
186 {
187 vectorFile->GetPathFile()->Set(pathfile);
188
189 if(vectorFile->DetectType() == GRPVECTORFILERESULT_OK)
190 {
191 return vectorFile;
192
193 }
194 else
195 {
196 delete vectorFile;
197 }
198 }
199 }
200

281{

342bool GRPVECTORFILE::DetectFileFormatText(XPATH& pathfile)
343{
344 XFILETXTFORMATCHAR formatchar = XFILETXTFORMATCHAR_UNKNOWN;
345 bool isText = false;
346
347 XFILETXT* fileTXT = new XFILETXT();
348 if(fileTXT)
349 {
350 if(fileTXT->Open(pathfile))
351 {
352 formatchar = fileTXT->GetFormatCharFromFile();
353
354 fileTXT->Close();
355 }
356
357 delete fileTXT;
358 }
359
360 if(formatchar != XFILETXTFORMATCHAR_ASCII) return true;
361
362 XFILE* file = GEN_XFACTORY.Create_File();
363 if(file)
364 {
365 if(file->Open(pathfile))
366 {
367 #define MAX_BUFFER_TEST 1024*5
368
369 XBUFFER buffer;
370 XDWORD size = MAX_BUFFER_TEST;
371 XDWORD controlchars = 0;
372
373 buffer.Resize(MAX_BUFFER_TEST);
374 file->Read(buffer.Get(), &size);
375
376 if(size)
377 {
378 for(XDWORD c=0; c<size; c++)
379 {
380 if(((buffer.Get()[c] >= 0x00) && (buffer.Get()[c] < 0x0A)) ||
381 ((buffer.Get()[c] > 0x0D) && (buffer.Get()[c] < 0x20)))
382 {
383 controlchars++;
384 }
385 }
386
387 int percent = (controlchars*100/size);
388 if(percent < 10) isText = true;
389 }
390
391 file->Close();
392 }
393
394 delete file;
395 }
396
397 return isText;
398}
