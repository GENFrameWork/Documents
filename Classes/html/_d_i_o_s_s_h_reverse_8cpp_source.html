230bool DIOSSHREVERSE::DownloadCFG(XCHAR* URL, XSTRING& publicIP, XSTRING& localIP)
231{
232 if(!URL) return false;
233
234 XBUFFER xbuffer;
235 DIOURL _URL;
236 bool status = false;
237 bool result = false;
238
239 DIOWEBCLIENT* webclient = new DIOWEBCLIENT();
240 if(!webclient) return false;
241
242 _URL.Set(URL);
243 _URL.Slash_Add();
244 _URL.Add(DIOSSHREVERSE_CFGREMOTENAMEFILE);
245
246 status = webclient->Get(_URL, xbuffer, NULL, 3);
247 if(status)
248 {
249 XFILEINI* fileini = new XFILEINI();
250 if(fileini)
251 {
252 status = fileini->AddBufferLines(XFILETXTFORMATCHAR_ASCII, xbuffer);
253 if(status)
254 {
255 status = fileini->ConvertFromLines();
256 if(status)
257 {
258 int c = 1;
259 while(1)
260 {
261 XSTRING section;
262 XSTRING value;
263
264 section.Format(__L("ENTRY%02d"), c);
265
266 if(!fileini->SelectSection(section)) break;
267
268 status = fileini->ReadValue(__L("isactive"), value);
269 if(status)
270 {
271 if(value.ConvertToBoolean()
272 {
273 DIOURL* urlorigin = new DIOURL();
274 if(urlorigin)
275 {
276 XSTRING _publicIP;
277 XSTRING _localremoteIP;
278
279 status = fileini->ReadValue(__L("urlorigin"), (*urlorigin));
280 if(status)
281 {
282 bool validaccess = false;
283
284 urlorigin->ResolveURL(_publicIP);
285
286 if(!publicIP.Compare(_publicIP))
287 {
288 validaccess = true;
289
290 status = fileini->ReadValue(__L("localremoteip"), _localremoteIP);
291 if(status)
292 {
293 if(_localremoteIP.GetSize())
294 {
295 if(_localremoteIP.Compare(localIP)) validaccess = false;
296 }
297 }
298 }
299
300 if(validaccess)
301 {
302 localIP = __L("localhost");
303
304 status = fileini->ReadValue(__L("urltarget"), value);
305 if(status) GetURLTarget()->Set(value);
306
307 status = fileini->ReadValue(__L("localip"), value);
308 if(status) GetLocalIP()->Set(value);
309
310 status = fileini->ReadValue(__L("login"), value);
311 if(status) GetLogin()->Set(value);
312
313 status = fileini->ReadValue(__L("password"), value);
314 if(status) GetPassword()->Set(value);
315
316 status = fileini->ReadValue(__L("port"), value);
317 if(status) SetPort(value.ConvertToInt());
318
319 result = true;
320 }
321 }
322
323 delete urlorigin;
324 }
325 }
326 }
327
328 c++;
329 }
330 }
331 }
332
333 delete fileini;
334 }
335 }
336
337 delete webclient;
338
339 return result;
340}
