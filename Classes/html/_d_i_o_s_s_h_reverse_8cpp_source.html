223bool DIOSSHREVERSE::DownloadCFG(XCHAR* URL, XSTRING& publicIP, XSTRING& localIP)
224{
225 if(!URL) return false;
226
227 XBUFFER xbuffer;
228 DIOURL _URL;
229 bool status = false;
230 bool result = false;
231
232 DIOWEBCLIENT* webclient = new DIOWEBCLIENT();
233 if(!webclient) return false;
234
235 _URL.Set(URL);
236 _URL.Slash_Add();
237 _URL.Add(DIOSSHREVERSE_CFGREMOTENAMEFILE);
238
239 status = webclient->Get(_URL, xbuffer, NULL, 3);
240 if(status)
241 {
242 XFILEINI* fileini = new XFILEINI();
243 if(fileini)
244 {
245 status = fileini->AddBufferLines(XFILETXTFORMATCHAR_ASCII, xbuffer);
246 if(status)
247 {
248 status = fileini->ConvertFromLines();
249 if(status)
250 {
251 int c = 1;
252 while(1)
253 {
254 XSTRING section;
255 XSTRING value;
256
257 section.Format(__L("ENTRY%02d"), c);
258
259 if(!fileini->SelectSection(section)) break;
260
261 status = fileini->ReadValue(__L("isactive"), value);
262 if(status)
263 {
264 if(!value.Compare(__L("yes"), true))
265 {
266 DIOURL* urlorigin = new DIOURL();
267 if(urlorigin)
268 {
269 XSTRING _publicIP;
270 XSTRING _localremoteIP;
271
272 status = fileini->ReadValue(__L("urlorigin"), (*urlorigin));
273 if(status)
274 {
275 bool validaccess = false;
276
277 urlorigin->ResolveURL(_publicIP);
278
279 if(!publicIP.Compare(_publicIP))
280 {
281 validaccess = true;
282
283 status = fileini->ReadValue(__L("localremoteip"), _localremoteIP);
284 if(status)
285 {
286 if(_localremoteIP.GetSize())
287 {
288 if(_localremoteIP.Compare(localIP)) validaccess = false;
289 }
290 }
291 }
292
293 if(validaccess)
294 {
295 localIP = __L("localhost");
296
297 status = fileini->ReadValue(__L("urltarget"), value);
298 if(status) GetURLTarget()->Set(value);
299
300 status = fileini->ReadValue(__L("localip"), value);
301 if(status) GetLocalIP()->Set(value);
302
303 status = fileini->ReadValue(__L("login"), value);
304 if(status) GetLogin()->Set(value);
305
306 status = fileini->ReadValue(__L("password"), value);
307 if(status) GetPassword()->Set(value);
308
309 status = fileini->ReadValue(__L("port"), value);
310 if(status) SetPort(value.ConvertToInt());
311
312 result = true;
313 }
314 }
315
316 delete urlorigin;
317 }
318 }
319 }
320
321 c++;
322 }
323 }
324 }
325
326 delete fileini;
327 }
328 }
329
330 delete webclient;
331
332 return result;
333}
