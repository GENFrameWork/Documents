111bool SNDOPENALFACTORY::Ini()
112{
113 device = alcOpenDevice(NULL);
114 if(!device)
115 {
116 return false;
117 }
118
119 ALCenum error;
120
121 context = alcCreateContext(device, NULL);
122 error = alcGetError(device);
123
124 if(!context || (error != ALC_NO_ERROR))
125 {
126 alcCloseDevice(device);
127 return false;
128 }
129
130 alcMakeContextCurrent(context);
131 error = alcGetError(device);
132 if(!context || (error != ALC_NO_ERROR))
133 {
134 alcCloseDevice(device);
135 return false;
136 }
137
138 maxchannels = 16;
139 mastervolume = 1.0f;
140

142 if(!playmutex)

150 return false;
151 }
152
153 if(!playthread->Ini())
154 {
155 return false;
156 }
157

173bool SNDOPENALFACTORY::Update()
174{
175 if(!soundactive)
176 {
177 return false;
178 }
179
180 if(playmutex)
181 {
182 playmutex->Lock();
183 }
184
185 for(XDWORD c=0; c<soundplayitems.GetSize(); c++)
186 {
187 SNDOPENALPLAYITEM* playitem = soundplayitems.Get(c);
188 if(playitem)
189 {
190 SNDITEM* item = playitem->GetItem();
191 SNDOPENALSOURCE* source = playitem->GetSource();
192 if(item && source)
193 {
194 SNDNOTE* soundnote = NULL;
195 SNDFILE* soundfile = NULL;
196
197 switch(item->GetType())
198 {
199 case SNDITEM_TYPE_UNKNOWN : break;
200
201 case SNDITEM_TYPE_NOTE : soundnote = playitem->GetItem()->GetSoundNote();
202 break;
203
204 case SNDITEM_TYPE_FILE : soundfile = playitem->GetItem()->GetSoundFile();
205 break;
206 }
207
208 if(!soundnote && !soundfile)
209 {
210 continue;
211 }
212
213 if(playitem->GetEvent() == SNDOPENALPLAYITEM_XFSMEVENT_NONE) // Not new event
214 {
215 switch(playitem->GetCurrentState())
216 {
217 case SNDOPENALPLAYITEM_XFSMSTATE_NONE : break;
218 case SNDOPENALPLAYITEM_XFSMSTATE_INI : break;
219
220 case SNDOPENALPLAYITEM_XFSMSTATE_PLAY : { XQWORD millisecond = item->GetTimerPlay()->GetMeasureMilliSeconds();
221 bool outoftime = false;
222
223 item->SetPlayingTime((XDWORD)item->GetPlayingTime() + (XDWORD)millisecond);
224 item->SetCurrentPlayingTime((XDWORD)item->GetCurrentPlayingTime() + (XDWORD)millisecond);
225
226 item->GetTimerPlay()->Reset();
227
228 if(item->GetNTimesToPlay() == SNDFACTORY_INLOOP)
229 {
230 if(!source->IsPLaying())
231 {
232 outoftime = true;
233 }
234 }
235 else
236 {
237 if(item->GetCurrentPlayingTime() >= (item->GetDuration()))
238 {
239 outoftime = true;
240 }
241 }
242
243 if(outoftime)
244 {
245 item->SetCurrentPlayingTime(0);
246
247 if(!item->GetCounterPlay())
248 {
249 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_STOP);
250 item->AddOneNTimesPlayed();
251 }
252 else
253 {
254 item->SetCounterPlay(item->GetCounterPlay()-1);
255
256 item->GetTimerPlay()->Reset();
257
258 source->ResetPlay();
259 source->Play();
260 }
261 }
262 }
263 break;
264
265 case SNDOPENALPLAYITEM_XFSMSTATE_PAUSE : break;
266
267 case SNDOPENALPLAYITEM_XFSMSTATE_STOP : break;
268
269 case SNDOPENALPLAYITEM_XFSMSTATE_END : break;
270 }
271 }
272 else // New event
273 {
274 if(playitem->GetEvent() < SNDOPENALPLAYITEM_LASTEVENT)
275 {
276 playitem->CheckTransition();
277
278 switch(playitem->GetCurrentState())
279 {
280 case SNDOPENALPLAYITEM_XFSMSTATE_NONE : break;
281
282 case SNDOPENALPLAYITEM_XFSMSTATE_INI : { switch(item->GetType())
283 {
284 case SNDITEM_TYPE_UNKNOWN : break;
285
286 case SNDITEM_TYPE_NOTE : GenerateBuffer(playitem, soundnote);
287 break;
288
289 case SNDITEM_TYPE_FILE : GenerateBuffer(playitem, soundfile);
290 break;
291 }
292
293 if(item->GetNTimesToPlay())
294 {
295 item->SetCounterPlay(item->GetNTimesToPlay()-1);
296 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_PLAY);
297 }
298
299 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_INI);
300 xevent.SetItem(item);
301 PostEvent(&xevent);
302 }
303 break;
304
305 case SNDOPENALPLAYITEM_XFSMSTATE_PLAY : { item->GetTimerPlay()->Reset();
306 source->Play();
307 item->SetStatus(SNDITEM_STATUS_PLAY);
308
309 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_PLAY);
310 xevent.SetItem(item);
311 PostEvent(&xevent);
312 }
313 break;
314
315 case SNDOPENALPLAYITEM_XFSMSTATE_PAUSE : { source->Pause();
316 item->SetStatus(SNDITEM_STATUS_PAUSE);
317
318 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_PAUSE);
319 xevent.SetItem(item);
320 PostEvent(&xevent);
321 }
322 break;
323
324 case SNDOPENALPLAYITEM_XFSMSTATE_STOP : { source->Stop();
325
326 item->SetStatus(SNDITEM_STATUS_STOP);
327
328 if(item->GetNTimesToPlay() == SNDFACTORY_INLOOP)
329 {
330 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_END);
331 }
332 else
333 {
334 if(!item->GetCounterPlay())
335 {
336 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_END);
337 }
338 }
339
340 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_STOP);
341 xevent.SetItem(item);
342 PostEvent(&xevent);
343
344 }
345 break;
346
347 case SNDOPENALPLAYITEM_XFSMSTATE_END : { if(playmutex)
348 {
349 playmutex->Lock();
350 }
351
352 soundplayitems.Delete(playitem);
353 delete playitem;
354
355 if(playmutex)
356 {
357 playmutex->UnLock();
358 }
359
360 item->SetStatus(SNDITEM_STATUS_END);
361
362 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_END);
363 xevent.SetItem(item);
364 PostEvent(&xevent);
365 }
366 break;
367 }
368 }
369 }
370 }
371 }
372 }
373
374 if(playmutex)
375 {
376 playmutex->UnLock();
377 }
378
379 return true;
380}

449bool SNDOPENALFACTORY::Sound_Play(SNDITEM* item, SNDPLAYCFG* playCFG, int ntimestoplay)
450{
451 if(!soundactive)
452 {
453 return false;
454 }
455
456 if(!SNDFACTORY::Sound_Play(item, playCFG, ntimestoplay))
457 {
458 return false;
459 }
460
461 if(soundplayitems.GetSize() >= maxchannels)
462 {
463 return false;
464 }
465
466 SNDOPENALSOURCE* source = new SNDOPENALSOURCE();
467 if(!source)
468 {
469 return false;
470 }
471
472 SNDOPENALPLAYITEM* soundplayitem = new SNDOPENALPLAYITEM();
473 if(!soundplayitem)
474 {
475 delete source;
476 return false;
477 }
478
479 soundplayitem->SetItem(item);
480 soundplayitem->SetSource(source);
481
482 if(!soundplayitems.Add(soundplayitem))
483 {
484 return false;
485 }
486
487 if(item->GetPlayCFG()->GetVolume() != SNDFACTORY_UNDEFINED)
488 {
489 source->SetVolume(item->GetPlayCFG()->GetVolume());
490 }
491
492 if(item->GetPlayCFG()->GetPitch() != SNDFACTORY_UNDEFINED)
493 {
494 source->SetPitch(item->GetPlayCFG()->GetPitch());
495 }
496
497 if(ntimestoplay == SNDFACTORY_INLOOP)
498 {
499 source->SetInLoop(true);
500 }
501 else
502 {
503 item->SetPlayingTime(0);
504 source->SetInLoop(false);
505 }
506
507 soundplayitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_INI);
508
509 return true;
510}

659bool SNDOPENALFACTORY::Sound_WaitToEnd(SNDITEM* item, int maxtimeout, SNDFACTORY_WAITFUNCTION waitfunction)
660{
661 if(!soundactive)
662 {
663 return false;
664 }
665
666 if(!item)
667 {
668 return false;
669 }
670
671 XTIMER* timeout = NULL;
672 bool status = true;
673
674 timeout = GEN_XFACTORY.CreateTimer();
675 if(!timeout)
676 {
677 return false;
678 }
679
680 timeout->Reset();
681
682 while(item->GetStatus() != SNDITEM_STATUS_END)
683 {
684 if(maxtimeout)

686 if(timeout->GetMeasureMilliSeconds() >= maxtimeout)

694 {
695 if(!waitfunction(item))
696 {
697 status = false;
698 break;
699 }
700 }
701 else
702 {
703 GEN_XSLEEP.MilliSeconds(100);
704 }
705 }
706
707 GEN_XFACTORY.DeleteTimer(timeout);
708
709 if(waitfunction)
710 {
711 if(!waitfunction(item))
712 {
713 status = false;
714 }
715 }
716
