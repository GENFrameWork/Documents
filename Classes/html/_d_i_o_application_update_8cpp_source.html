564bool DIOAPPLICATIONUPDATE::GetVersionAvailable(DIOURL& url, int port, DIOAPPLICATIONUPDATE_VERSIONDATA& updateversiondata, int timeout, DIOAPPLICATIONUPDATE_ERROR& error)
565{
566 error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
567
568 if(!webclient) return false;
569
570 bool status = false;
571
572 DIOURL urlfile;
573 XPATH xpath;
574
575 urlfile = url;
576 urlfile.Slash_Add();
577 urlfile += DIOAPPLICATIONUPDATE_INIFILE;
578
579 urlfile.Slash_Normalize();
580
581 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_ROOT, xpath);
582 xpath.Slash_Add();
583 xpath += DIOAPPLICATIONUPDATE_INIFILE;
584
585 webclient->Set_Port(port);
586
587 status = webclient->Get(urlfile, xpath ,NULL, timeout);
588 if(!status)
589 {
590 if(stopprocess)
591 {
592 error = DIOAPPLICATIONUPDATE_ERROR_BREAKFORUSER;
593 }
594 else
595 {
596 error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
597 }
598 }
599 else
600 {
601 error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
602
603 XFILEINI* ini = new XFILEINI();
604 if(ini)
605 {
606 if(ini->Open(xpath))
607 {
608 XSTRING key;
609 XSTRING value;
610 int index;
611
612 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_VERSION , value)) updateversiondata.SetVersion(value.ConvertToInt());
613 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_SUBVERSION , value)) updateversiondata.SetSubVersion(value.ConvertToInt());
614 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_ERRORCONTROL , value)) updateversiondata.SetSubVersionError(value.ConvertToInt());
615 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_SYSTEMMUSTBEINIT , value)) updateversiondata.SetSystemMustBeInit(value.ConvertToBoolean());
616
617 //----------------------------------------------------------------------------------------
618
619 FilesToUpdate_DeleteList();
620
621 index = 0;
622
623 do{ key.Format(__L("%s%d"), DIOAPPLICATIONUPDATE_FILESSECTION_FILES, index);
624
625 ini->ReadValue(DIOAPPLICATIONUPDATE_FILESSECTION, key, value);
626 if(!value.IsEmpty())
627 {
628 XPATH xpathfile(_MAXSTR);
629 int sizefile = 0;
630 XDWORD CRC32 = 0;
631
632 int comma=value.FindCharacter(__C(','));
633 XSTRING name;
634 value.Copy(0,comma,name);
635 value.DeleteCharacters(0,comma+1);
636 value.UnFormat(__L("%d,%08X"), &sizefile, &CRC32);
637 xpathfile.Set(name.Get());
638
639 if(xpathfile.GetSize() && sizefile && CRC32)
640 {
641 DIOAPPLICATIONUPDATE_FILE* updatefile = new DIOAPPLICATIONUPDATE_FILE(xpathfile, CRC32, sizefile);
642 if(updatefile) filestoupdate.Add(updatefile);
643 }
644 }
645
646 index++;
647
648 } while(index<DIOAPPLICATIONUPDATE_MAXINDEXFILES);
649
650 //----------------------------------------------------------------------------------------
651
652 ini->Close();
653 error = DIOAPPLICATIONUPDATE_NOT_ERROR;
654 }
655
656 delete ini;
657 }
658 }
659
660 //----------------------------------------------------------------------------------------
661 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE);
662
663 xevent.SetVersionData(&updateversiondata);
664 xevent.SetFilesToUpdate(&filestoupdate);
665 xevent.SetError(error);
666
667 PostEvent(&xevent);
668 //----------------------------------------------------------------------------------------
669
670 XFILE* GEN_XFACTORY_CREATE(file, Create_File())
671 if(file)
672 {
673 file->Erase(xpath);
674 GEN_XFACTORY.Delete_File(file);
675 }
676
677 return status;
678}

697bool DIOAPPLICATIONUPDATE::Download(DIOURL& url, int port, XCHAR* addtargetpath, DIOAPPLICATIONUPDATE_VERSIONDATA& updateversiondata, int timeout, DIOAPPLICATIONUPDATE_ERROR& error)
698{
699 error = DIOAPPLICATIONUPDATE_NOT_ERROR;
700
701 if(!xdir) return false;
702
703 XPATH namefile;
704 XPATH extenxion;
705 DIOURL urlfile;
706 int c;
707 bool status = false;
708
709 if(!filestoupdate.GetSize())
710 {
711 error = DIOAPPLICATIONUPDATE_ERROR_NOTFILES;
712 return false;
713 }
714
715 for(c=0;c<(int)filestoupdate.GetSize();c++)
716 {
717 XPATH xpathfile;
718
719 progressfiletoupdate = c;
720
721 DIOAPPLICATIONUPDATE_FILE* updatefile = (DIOAPPLICATIONUPDATE_FILE*)filestoupdate.Get(c);
722 if(!updatefile) continue;
723
724 urlfile = url;
725 urlfile.Slash_Add();
726 urlfile += updatefile->GetXPathFile()->Get();
727 urlfile.Slash_Normalize();
728
729 xpathfile.Empty();
730
731 if(addtargetpath)
732 {
733 if(addtargetpath[0])
734 {
735 xpathfile = addtargetpath;
736 xpathfile.Slash_Add();
737 }
738 }
739
740 xpathfile += updatefile->GetXPathFile()->Get();
741 xpathfile.Slash_Normalize();
742
743 if(!xpathfile.IsEmpty())
744 {
745 XPATH xpath;
746
747 xpathfile.GetDriveAndPath(xpath);
748
749 if(!xpath.IsEmpty())
750 {
751 if(xdir) xdir->Make(xpath);
752 }
753 }
754
755 xpathfile.GetNamefileExt(namefile);
756 xpathfile.GetExt(extenxion);
757
758 HASHCRC32 hashCRC32;
759
760 hashCRC32.ResetResult();
761 hashCRC32.Do(xpathfile);
762
763 if(updatefile->GetCRC32() != hashCRC32.GetResultCRC32())
764 {
765 webclient->Set_Port(port);
766
767 status = webclient->Get(urlfile, xpathfile, NULL, timeout);
768 if(!status)
769 {
770 if(stopprocess)
771 error = DIOAPPLICATIONUPDATE_ERROR_BREAKFORUSER;
772 else error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
773
774 break;
775 }
776
777 hashCRC32.ResetResult();
778 hashCRC32.Do(xpathfile);
779
780 if(updatefile->GetCRC32() != hashCRC32.GetResultCRC32())
781 {
782 error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
783 status = false;
784 break;
785
786 } else status = true;
787
788 } else status = true;
789
790
791 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE);
792
793 xevent.SetVersionData(&updateversiondata);
794 xevent.SetFilesToUpdate(&filestoupdate);
795 xevent.SetIndexOfOperation(c);
796 xevent.SetError(error);
797
798 PostEvent(&xevent);
799
800
801 //-------------------------------------------------------------
802 // Descompress File
803 if(!extenxion.Compare(__L(".zip"), true))
804 {
805 XFILEUNZIP* unzip = new XFILEUNZIP();
806 if(unzip)
807 {
808 if(unzip->Open(xpathfile))
809 {
810 XPATH xpath;
811 int sourceindex = 0;
812
813 xpath = xpathfile;
814 xpath.SetOnlyDriveAndPath();
815
816 status = true;
817
818 do{ XFILECMPZIP* xfilecmp = unzip->GotoFile(sourceindex);
819 if(!xfilecmp) break;
820
821 XPATH xpathcomplete;
822
823 xpathcomplete = xpath;
824 xpathcomplete += xfilecmp->GetName();
825
826 if(xfilecmp->IsDirectory())
827 {
828 xdir->Make(xpathcomplete);
829 status = true;
830 }
831 else
832 {
833 hashCRC32.ResetResult();
834 hashCRC32.Do(xpathcomplete);
835
836 if(xfilecmp->GetCRC32() != hashCRC32.GetResultCRC32())
837 {
838 status = unzip->DecompressFile(sourceindex, xpath);
839
840 } else status = true;
841
842 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE);
843
844 xevent.SetVersionData(&updateversiondata);
845 xevent.SetFilesToUpdate(&filestoupdate);
846 xevent.SetIndexOfOperation(c);
847 xevent.GetActualUnzipFileName()->Set(xfilecmp->GetName());
848 xevent.SetError(status?DIOAPPLICATIONUPDATE_NOT_ERROR:DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION);
849
850 PostEvent(&xevent);
851 }
852
853 sourceindex++;
854
855 delete xfilecmp;
856
857 } while(status);
858
859 unzip->Close();
860
861 XFILE* GEN_XFACTORY_CREATE(file, Create_File())
862 if(file)
863 {
864 file->Erase(xpathfile);
865 GEN_XFACTORY.Delete_File(file);
866 }
867 }
868
869 delete unzip;
870 }
871 }
872
873 //-------------------------------------------------------------
874 }
875
876 if(!status) error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
877
878 return status;
879}
