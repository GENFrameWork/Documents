557bool DIOAPPLICATIONUPDATE::GetVersionAvailable(DIOURL& url, int port, DIOAPPLICATIONUPDATE_VERSIONDATA& updateversiondata, int timeout, DIOAPPLICATIONUPDATE_ERROR& error)
558{
559 error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
560
561 if(!webclient) return false;
562
563 bool status = false;
564
565 DIOURL urlfile;
566 XPATH xpath;
567
568 urlfile = url;
569 urlfile.Slash_Add();
570 urlfile += DIOAPPLICATIONUPDATE_INIFILE;
571
572 urlfile.Slash_Normalize();
573
574 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_ROOT, xpath);
575 xpath.Slash_Add();
576 xpath += DIOAPPLICATIONUPDATE_INIFILE;
577
578 webclient->Set_Port(port);
579
580 status = webclient->Get(urlfile, xpath ,NULL, timeout);
581 if(!status)
582 {
583 if(stopprocess)
584 {
585 error = DIOAPPLICATIONUPDATE_ERROR_BREAKFORUSER;
586 }
587 else
588 {
589 error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
590 }
591 }
592 else
593 {
594 error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
595
596 XFILEINI* ini = new XFILEINI();
597 if(ini)
598 {
599 if(ini->Open(xpath))
600 {
601 XSTRING key;
602 XSTRING value;
603 int index;
604
605 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_VERSION , value)) updateversiondata.SetVersion(value.ConvertToInt());
606 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_SUBVERSION , value)) updateversiondata.SetSubVersion(value.ConvertToInt());
607 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_ERRORCONTROL , value)) updateversiondata.SetSubVersionError(value.ConvertToInt());
608 if(ini->ReadValue(DIOAPPLICATIONUPDATE_GENERALSECTION , DIOAPPLICATIONUPDATE_GENERALSECTION_SYSTEMMUSTBEINIT , value)) updateversiondata.SetSystemMustBeInit(value.ConvertToBoolean());
609
610 //----------------------------------------------------------------------------------------
611
612 FilesToUpdate_DeleteList();
613
614 index = 0;
615
616 do{ key.Format(__L("%s%d"), DIOAPPLICATIONUPDATE_FILESSECTION_FILES, index);
617
618 ini->ReadValue(DIOAPPLICATIONUPDATE_FILESSECTION, key, value);
619 if(!value.IsEmpty())
620 {
621 XPATH xpathfile(_MAXSTR);
622 int sizefile = 0;
623 XDWORD CRC32 = 0;
624
625 int comma=value.FindCharacter(__C(','));
626 XSTRING name;
627 value.Copy(0,comma,name);
628 value.DeleteCharacters(0,comma+1);
629 value.UnFormat(__L("%d,%08X"), &sizefile, &CRC32);
630 xpathfile.Set(name.Get());
631
632 if(xpathfile.GetSize() && sizefile && CRC32)
633 {
634 DIOAPPLICATIONUPDATE_FILE* updatefile = new DIOAPPLICATIONUPDATE_FILE(xpathfile, CRC32, sizefile);
635 if(updatefile) filestoupdate.Add(updatefile);
636 }
637 }
638
639 index++;
640
641 } while(index<DIOAPPLICATIONUPDATE_MAXINDEXFILES);
642
643 //----------------------------------------------------------------------------------------
644
645 ini->Close();
646 error = DIOAPPLICATIONUPDATE_NOT_ERROR;
647 }
648
649 delete ini;
650 }
651 }
652
653 //----------------------------------------------------------------------------------------
654 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE);
655
656 xevent.SetVersionData(&updateversiondata);
657 xevent.SetFilesToUpdate(&filestoupdate);
658 xevent.SetError(error);
659
660 PostEvent(&xevent);
661 //----------------------------------------------------------------------------------------
662
663 XFILE* GEN_XFACTORY_CREATE(file, Create_File())
664 if(file)
665 {
666 file->Erase(xpath);
667 GEN_XFACTORY.Delete_File(file);
668 }
669
670 return status;
671}

690bool DIOAPPLICATIONUPDATE::Download(DIOURL& url, int port, XCHAR* addtargetpath, DIOAPPLICATIONUPDATE_VERSIONDATA& updateversiondata, int timeout, DIOAPPLICATIONUPDATE_ERROR& error)
691{
692 error = DIOAPPLICATIONUPDATE_NOT_ERROR;
693
694 if(!xdir) return false;
695
696 XPATH namefile;
697 XPATH extenxion;
698 DIOURL urlfile;
699 int c;
700 bool status = false;
701
702 if(!filestoupdate.GetSize())
703 {
704 error = DIOAPPLICATIONUPDATE_ERROR_NOTFILES;
705 return false;
706 }
707
708 for(c=0;c<(int)filestoupdate.GetSize();c++)
709 {
710 XPATH xpathfile;
711
712 progressfiletoupdate = c;
713
714 DIOAPPLICATIONUPDATE_FILE* updatefile = (DIOAPPLICATIONUPDATE_FILE*)filestoupdate.Get(c);
715 if(!updatefile) continue;
716
717 urlfile = url;
718 urlfile.Slash_Add();
719 urlfile += updatefile->GetXPathFile()->Get();
720 urlfile.Slash_Normalize();
721
722 xpathfile.Empty();
723
724 if(addtargetpath)
725 {
726 if(addtargetpath[0])
727 {
728 xpathfile = addtargetpath;
729 xpathfile.Slash_Add();
730 }
731 }
732
733 xpathfile += updatefile->GetXPathFile()->Get();
734 xpathfile.Slash_Normalize();
735
736 if(!xpathfile.IsEmpty())
737 {
738 XPATH xpath;
739
740 xpathfile.GetDriveAndPath(xpath);
741
742 if(!xpath.IsEmpty())
743 {
744 if(xdir) xdir->Make(xpath);
745 }
746 }
747
748 xpathfile.GetNamefileExt(namefile);
749 xpathfile.GetExt(extenxion);
750
751 HASHCRC32 hashCRC32;
752
753 hashCRC32.ResetResult();
754 hashCRC32.Do(xpathfile);
755
756 if(updatefile->GetCRC32() != hashCRC32.GetResultCRC32())
757 {
758 webclient->Set_Port(port);
759
760 status = webclient->Get(urlfile, xpathfile, NULL, timeout);
761 if(!status)
762 {
763 if(stopprocess)
764 error = DIOAPPLICATIONUPDATE_ERROR_BREAKFORUSER;
765 else error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
766
767 break;
768 }
769
770 hashCRC32.ResetResult();
771 hashCRC32.Do(xpathfile);
772
773 if(updatefile->GetCRC32() != hashCRC32.GetResultCRC32())
774 {
775 error = DIOAPPLICATIONUPDATE_ERROR_DOWNLOADFILE;
776 status = false;
777 break;
778
779 } else status = true;
780
781 } else status = true;
782
783
784 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE);
785
786 xevent.SetVersionData(&updateversiondata);
787 xevent.SetFilesToUpdate(&filestoupdate);
788 xevent.SetIndexOfOperation(c);
789 xevent.SetError(error);
790
791 PostEvent(&xevent);
792
793
794 //-------------------------------------------------------------
795 // Descompress File
796 if(!extenxion.Compare(__L(".zip"), true))
797 {
798 XFILEUNZIP* unzip = new XFILEUNZIP();
799 if(unzip)
800 {
801 if(unzip->Open(xpathfile))
802 {
803 XPATH xpath;
804 int sourceindex = 0;
805
806 xpath = xpathfile;
807 xpath.SetOnlyDriveAndPath();
808
809 status = true;
810
811 do{ XFILECMPZIP* xfilecmp = unzip->GotoFile(sourceindex);
812 if(!xfilecmp) break;
813
814 XPATH xpathcomplete;
815
816 xpathcomplete = xpath;
817 xpathcomplete += xfilecmp->GetName();
818
819 if(xfilecmp->IsDirectory())
820 {
821 xdir->Make(xpathcomplete);
822 status = true;
823 }
824 else
825 {
826 hashCRC32.ResetResult();
827 hashCRC32.Do(xpathcomplete);
828
829 if(xfilecmp->GetCRC32() != hashCRC32.GetResultCRC32())
830 {
831 status = unzip->DecompressFile(sourceindex, xpath);
832
833 } else status = true;
834
835 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE);
836
837 xevent.SetVersionData(&updateversiondata);
838 xevent.SetFilesToUpdate(&filestoupdate);
839 xevent.SetIndexOfOperation(c);
840 xevent.GetActualUnzipFileName()->Set(xfilecmp->GetName());
841 xevent.SetError(status?DIOAPPLICATIONUPDATE_NOT_ERROR:DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION);
842
843 PostEvent(&xevent);
844 }
845
846 sourceindex++;
847
848 delete xfilecmp;
849
850 } while(status);
851
852 unzip->Close();
853
854 XFILE* GEN_XFACTORY_CREATE(file, Create_File())
855 if(file)
856 {
857 file->Erase(xpathfile);
858 GEN_XFACTORY.Delete_File(file);
859 }
860 }
861
862 delete unzip;
863 }
864 }
865
866 //-------------------------------------------------------------
867 }
868
869 if(!status) error = DIOAPPLICATIONUPDATE_ERROR_INVALIDOPERATION;
870
871 return status;
872}
