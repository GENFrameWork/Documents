299bool DIOSTREAMUARTCONFIG::GetToString(XSTRING* string)
300{
301 if(!string) return false;
302
303 XSTRING xstringtmp;
304
305 string->Empty();
306
307 if(!GetLocalDeviceName()->IsEmpty())
308 {
309 string->Add(GetLocalDeviceName()->Get());
310 string->Add(__L(","));
311 }
312 else
313 {
314 if(!GetPort()) return false;
315 string->Format(__L("%d,"), GetPort());
316 }
317
318 if(!GetBaudRate()) return false;
319 xstringtmp.Format(__L("%d,"), GetBaudRate());
320 string->Add(xstringtmp);
321
322 switch(GetDataBits())
323 {
324 case DIOSTREAMUARTDATABIT_5 : string->Add(__L("5,")); break;
325 case DIOSTREAMUARTDATABIT_6 : string->Add(__L("6,")); break;
326 case DIOSTREAMUARTDATABIT_7 : string->Add(__L("7,")); break;
327 case DIOSTREAMUARTDATABIT_8 : string->Add(__L("8,")); break;
328 default : return false;
329 }
330
331 switch(GetParity())
332 {
333 case DIOSTREAMUARTPARITY_NONE : string->Add(__L("N,")); break;
334 case DIOSTREAMUARTPARITY_ODD : string->Add(__L("O,")); break;
335 case DIOSTREAMUARTPARITY_EVEN : string->Add(__L("E,")); break;
336 case DIOSTREAMUARTPARITY_MARK : string->Add(__L("M,")); break;
337 case DIOSTREAMUARTPARITY_SPACE : string->Add(__L("S,")); break;
338 default : return false;
339 }
340
341 switch(GetFlowControl())
342 {
343 case DIOSTREAMUARTFLOWCONTROL_NONE : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_NONE); break;
344 case DIOSTREAMUARTFLOWCONTROL_HARD : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_SOFT); break;
345 case DIOSTREAMUARTFLOWCONTROL_SOFT : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_HARD); break;
346 default : return false;
347 }
348
349 return true;
350}

381bool DIOSTREAMUARTCONFIG::SetFromString(XCHAR* string)
382{
383 if(!string) return false;
384
385 XSTRING _string = string;
386 XSTRING devicename;
387 int baudrate;
388 int databits;
389 XCHAR parity;
390 XSTRING stopbits;
391 XSTRING flowcontrol;
392
393 devicename.AdjustSize(_MAXSTR);
394 stopbits.AdjustSize(_MAXSTR);
395 flowcontrol.AdjustSize(_MAXSTR);
396
397 _string.UnFormat(__L("%s,%d,%d,%c,%s,%s"), devicename.Get(), &baudrate, &databits, &parity, stopbits.Get(), flowcontrol.Get());
398
399 devicename.AdjustSize();
400 stopbits.AdjustSize();
401 flowcontrol.AdjustSize();
402
403
404 if(devicename.HaveOnlyNumbers())
405 {
406 SetPort(devicename.ConvertToInt());
407 if(!GetPort()) return false;
408 }
409 else GetLocalDeviceName()->Set(devicename);
410
411 if(!baudrate) return false;
412 SetBaudRate(baudrate);
413
414 switch(databits)
415 {
416 case 5 : SetDataBits(DIOSTREAMUARTDATABIT_5); break;
417 case 6 : SetDataBits(DIOSTREAMUARTDATABIT_6); break;
418 case 7 : SetDataBits(DIOSTREAMUARTDATABIT_7); break;
419 case 8 : SetDataBits(DIOSTREAMUARTDATABIT_8); break;
420 default : return false;
421 }
422
423 switch(parity)
424 {
425 case 'N' : SetParity(DIOSTREAMUARTPARITY_NONE); break;
426 case 'O' : SetParity(DIOSTREAMUARTPARITY_ODD); break;
427 case 'E' : SetParity(DIOSTREAMUARTPARITY_EVEN); break;
428 case 'M' : SetParity(DIOSTREAMUARTPARITY_MARK); break;
429 case 'S' : SetParity(DIOSTREAMUARTPARITY_SPACE); break;
430 default : return false;
431 }
432
433 if(!stopbits.Compare(__L("1")))
434 {
435 SetStopBits(DIOSTREAMUARTSTOPBITS_ONE);
436 }
437 else
438 {
439 if(!stopbits.Compare(__L("1.5")))
440 {
441 SetStopBits(DIOSTREAMUARTSTOPBITS_ONEANDAHALF);
442 }
443 else
444 {
445 if(!stopbits.Compare(__L("2")))
446 {
447 SetStopBits(DIOSTREAMUARTSTOPBITS_TWO);
448
449 } else return false;
450 }
451 }
452
453 if(!flowcontrol.Compare(DIOSTREAMUART_FLOWCONTROL_STR_NONE, true))
454 {
455 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_NONE);
456 }
457 else
458 {
459 if(!stopbits.Compare(DIOSTREAMUART_FLOWCONTROL_STR_SOFT) || !stopbits.Compare(__L("SOFT")))
460 {
461 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_SOFT);
462 }
463 else
464 {
465 if(!stopbits.Compare(DIOSTREAMUART_FLOWCONTROL_STR_HARD) || !stopbits.Compare(__L("HARD")))
466 {
467 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_HARD);
468
469 } else return false;
470 }
471 }
472
473 return true;
474}
