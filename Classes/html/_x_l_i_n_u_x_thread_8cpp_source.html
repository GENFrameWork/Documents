318void* XLINUXTHREAD::Callback(void* thread)
319{
320 XLINUXTHREAD* th = NULL;
321 XTHREADFUNCTION func = NULL;
322 void* param;
323
324 th = (XLINUXTHREAD*)thread;
325 if(!th) return NULL;
326
327 func = (XTHREADFUNCTION)th->GetFunction();
328 if(!func) return NULL;
329
330 param = th->GetParam();
331
332 th->gotofunction = true;
333
334 if((th->priority != XTHREADPRIORITY_UNKNOWN) &&
335 (th->priority != XTHREADPRIORITY_NORMAL))
336 {
337 struct sched_param param; // scheduling priority
338
339 int policy = SCHED_FIFO; // scheduling policy
340
341 switch(th->priority)
342 {
343 #ifdef SCHED_IDLE
344 case XTHREADPRIORITY_LOW : policy = SCHED_IDLE;
345 param.sched_priority = sched_get_priority_min(policy);
346 break;
347 #endif
348
349 case XTHREADPRIORITY_HIGH : policy = SCHED_FIFO;
350 param.sched_priority = sched_get_priority_max(policy)/2;
351 break;
352
353 case XTHREADPRIORITY_REALTIME : policy = SCHED_FIFO;
354 param.sched_priority = sched_get_priority_max(policy);
355 break;
356
357 default : break;
358 }
359
360 pthread_setschedparam(th->thhandle, policy, &param);
361 }
362
363 while(1)
364 {
365 if(!th) break;
366
367 if(th->statusfunc == XTHREADSTATUS_EXIT) break;
368 if(th->statusfunc == XTHREADSTATUS_END) break;
369
370 if(th->statusfunc == XTHREADSTATUS_RUN)
371 {
372 if(func)
373 {
374 th->isinfunction = true;
375 func(param);
376 th->isinfunction = false;
377
378 } else break;
379 }
380
381 if(!th) break;
382
383 if(th->waityield) usleep(th->waityield*1000);
384 }
385
386 if(th) th->statusfunc = XTHREADSTATUS_END;
387
388 pthread_exit(0);
389
390 return NULL;
391}
