752bool XFILECSV::CreateRecordOfLine(XSTRING& line, XFILECSV_RECORD* record)
753{
754 if(line.IsEmpty()) return false;
755
756 int start = 0;
757 int end = 0;
758
759 do{ end = line.Find(__L(";"),true,start);
760 if(end == XSTRING_NOTFOUND)
761 {
762 end = line.Find(__L(","),true,start);
763 if(end != XSTRING_NOTFOUND) SetSeparator(__C(','));
764
765 } else SetSeparator(__C(';'));
766
767 XSTRING element;
768
769 if(end != XSTRING_NOTFOUND)
770 {
771 line.Copy(start,end,element);
772
773 record->AddElement(element);
774
775 start = end+1;
776 }
777 else
778 {
779 line.Copy(start,element);
780 record->AddElement(element);
781
782 break;
783 }
784
785 } while(1);
786
787 return true;
788}

803bool XFILECSV::CreateLineOfRecord(XFILECSV_RECORD* record,XSTRING& line)
804{
805 line.Empty();
806
807 XCHAR separatorstring[2] = { 0, 0 };
808 separatorstring[0]= separator;
809
810 for(int c=0;c<record->GetNElements();c++)
811 {
812 XSTRING element;
