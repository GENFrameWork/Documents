745bool XFILECSV::CreateRecordOfLine(XSTRING& line, XFILECSV_RECORD* record)
746{
747 if(line.IsEmpty()) return false;
748
749 int start = 0;
750 int end = 0;
751
752 do{ end = line.Find(__L(";"),true,start);
753 if(end == XSTRING_NOTFOUND)
754 {
755 end = line.Find(__L(","),true,start);
756 if(end != XSTRING_NOTFOUND) SetSeparator(__C(','));
757
758 } else SetSeparator(__C(';'));
759
760 XSTRING element;
761
762 if(end != XSTRING_NOTFOUND)
763 {
764 line.Copy(start,end,element);
765
766 record->AddElement(element);
767
768 start = end+1;
769 }
770 else
771 {
772 line.Copy(start,element);
773 record->AddElement(element);
774
775 break;
776 }
777
778 } while(1);
779
780 return true;
781}

796bool XFILECSV::CreateLineOfRecord(XFILECSV_RECORD* record,XSTRING& line)
797{
798 line.Empty();
799

801 separatorstring[0]= separator;
802
803 for(int c=0;c<record->GetNElements();c++)
804 {
