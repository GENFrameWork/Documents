236int DIOANDROIDSTREAMBLUETOOTH::IsReadyConnect(int socket)
237{
238 struct timeval tv;
239 int rc;
240 fd_set fdr;
241 fd_set fdw;
242 fd_set fds;
243
244 if(socket==-1) return -1;
245
246 FD_ZERO(&fdr);
247 FD_ZERO(&fdw);
248 FD_ZERO(&fds);
249
250 FD_SET((unsigned int)socket,&fdr);
251 FD_SET((unsigned int)socket,&fdw);
252 FD_SET((unsigned int)socket,&fds);
253
254 tv.tv_sec = 0;
255 tv.tv_usec = 100;
256
257 rc = select(socket+1, &fdr, &fdw, &fds, &tv);
258 if(rc==-1) return -1;
259 if(rc== 2) return -1;
260
261 int status1 = FD_ISSET(socket,&fdr) ? 1 : 0;
262 int status2 = FD_ISSET(socket,&fdw) ? 1 : 0;
263 int status3 = FD_ISSET(socket,&fds) ? 1 : 0;
264
265 if(status3) return -1;
266 if(((status1) || (status2)) && (rc==1)) return 1;
267
268 return 0;
269}

286sdp_session_t* DIOANDROIDSTREAMBLUETOOTH::SDP_RegisterService(char* service_name,char* service_dsc,char* service_prov,int rfcomm_channel)
287{
288 uint32_t service_uuid_int[] = { 0x0111, 10, 1, 1 };
289 //uint8_t rfcomm_channel = 1;
290
291 uuid_t root_uuid;
292 uuid_t l2cap_uuid;
293 uuid_t rfcomm_uuid;
294 uuid_t svc_uuid;
295
296 sdp_list_t* l2cap_list = 0;
297 sdp_list_t* rfcomm_list = 0;
298 sdp_list_t* root_list = 0;
299 sdp_list_t* proto_list = 0;
300 sdp_list_t* access_proto_list = 0;
301 sdp_data_t* channel = 0;
302//sdp_data_t* psm = 0;
303
304 sdp_record_t record;
305
306 bdaddr_t bdaddr_any = {0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
307 bdaddr_t bdaddr_local = {0x00, 0x00, 0x00, 0xff, 0xff, 0xff };
308
309
310 sdp_session_t* session = sdp_connect(&bdaddr_any, &bdaddr_local, SDP_RETRY_IF_BUSY);
311
312 memset((void *)&record, 0, sizeof(sdp_record_t));
313 record.handle = 0xffffffff;
314
315 sdp_uuid16_create(&root_uuid, PUBLIC_BROWSE_GROUP);
316 root_list = sdp_list_append(0, &root_uuid);
317 sdp_set_browse_groups(&record, root_list );
318
319 // set the general service ID
320 sdp_uuid128_create( &svc_uuid, &service_uuid_int );
321 sdp_set_service_id(&record, svc_uuid );
322
323 // set l2cap information
324 sdp_uuid16_create(&l2cap_uuid, L2CAP_UUID);
325 l2cap_list = sdp_list_append( 0, &l2cap_uuid );
326 proto_list = sdp_list_append( 0, l2cap_list );
327
328 // set rfcomm information
329 sdp_uuid16_create(&rfcomm_uuid, RFCOMM_UUID);
330 channel = sdp_data_alloc(SDP_UINT8, &rfcomm_channel);
331 rfcomm_list = sdp_list_append( 0, &rfcomm_uuid );
332 sdp_list_append( rfcomm_list, channel );
333 sdp_list_append( proto_list, rfcomm_list );
334
335 // attach protocol information to service record
336 access_proto_list = sdp_list_append( 0, proto_list );
337 sdp_set_access_protos( &record, access_proto_list );
338
339 // set the name, provider, and description
340 sdp_set_info_attr(&record, service_name, service_prov, service_dsc);
341
342 // connect to the local SDP server, register the service record, and
343 // disconnect
344 sdp_record_register(session, &record, 0);
345
346 // cleanup
347 sdp_data_free(channel);
348 sdp_list_free(l2cap_list , 0);
349 sdp_list_free(rfcomm_list , 0);
350 sdp_list_free(root_list , 0);
351 sdp_list_free(access_proto_list , 0);
352
353 return session;
354}

478void DIOANDROIDSTREAMBLUETOOTH::ThreadRunFunction(void* param)
479{
480 DIOANDROIDSTREAMBLUETOOTH* diostream = (DIOANDROIDSTREAMBLUETOOTH*)param;
481 if(!diostream) return;
482
483 if(diostream->GetEvent()==DIOANDROIDBTFSMEVENT_NONE) // No hay nuevos Eventos
484 {
485 switch(diostream->GetCurrentState())
486 {
487 case DIOANDROIDBTFSMSTATE_NONE : break;
488
489 case DIOANDROIDBTFSMSTATE_GETTINGCONNECTION : switch(diostream->IsReadyConnect(diostream->handlesocket))
490 {
491 case -1: { int sockerr = 0;
492 socklen_t sockerrlen = sizeof(sockerr);
493
494 if(!getsockopt(diostream->handlesocket, SOL_SOCKET, SO_ERROR,(void *)&sockerr, &sockerrlen))
495 {
496 if(sockerr==ECONNREFUSED) diostream->SetIsRefusedConnection(true);
497 }
498
499 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
500 }
501 break;
502
503 case 1: diostream->SetEvent(DIOANDROIDBTFSMEVENT_CONNECTED);
504 diostream->status = DIOSTREAMSTATUS_CONNECTED;
505 break;
506
507 default: diostream->status = DIOSTREAMSTATUS_GETTINGCONNECTION;
508 break;
509 }
510 break;
511
512 case DIOANDROIDBTFSMSTATE_CONNECTED : break;
513
514 case DIOANDROIDBTFSMSTATE_WAITINGTOREAD : { struct timeval waitd;
515 fd_set read_flags;
516 fd_set write_flags;
517
518 waitd.tv_sec = 0;
519 waitd.tv_usec = 100;
520
521 FD_ZERO(&read_flags);
522 FD_ZERO(&write_flags);
523
524 if(diostream->handlesocket<0)
525 {
526 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
527 break;
528 }
529
530 if(!diostream->IsBlockRead()) FD_SET(diostream->handlesocket, &read_flags);
531 if(!diostream->IsBlockWrite())
532 {
533 if(diostream->outbuffer->GetSize()) FD_SET(diostream->handlesocket, &write_flags);
534 }
535
536 int error = select(diostream->handlesocket+1, &read_flags,&write_flags,(fd_set*)0,&waitd);
537 if(error == -1)
538 {
539 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
540 break;
541 }
542
543 if(FD_ISSET(diostream->handlesocket, &read_flags)) //Socket ready for reading
544 {
545 memset(diostream->buffer,0,DIOSTREAM_MAXBUFFER);
546
547 int size = recv(diostream->handlesocket,(char*)diostream->buffer,DIOSTREAM_MAXBUFFER,0);
548 if(size<=0)
549 {
550 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
551 break;
552 }
553 else diostream->inbuffer->Add(diostream->buffer,size);
554
555 FD_CLR(diostream->handlesocket, &read_flags);
556 }
557
558 if(FD_ISSET(diostream->handlesocket, &write_flags)) //Socket ready for writing
559 {
560 int esize = diostream->outbuffer->GetSize();
561 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
562
563 if(esize)
564 {
565 memset(diostream->buffer,0,DIOSTREAM_MAXBUFFER);
566
567 diostream->outbuffer->Get(diostream->buffer,esize,0);
568
569 int size = send(diostream->handlesocket,(char*)diostream->buffer,esize,0);
570 if(size<=0)
571 {
572 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
573 break;
574
575 } else diostream->outbuffer->Extract(diostream->buffer,0,size);
576 }
577
578 FD_CLR(diostream->handlesocket, &write_flags);
579 }
580 }
581 break;
582
583 case DIOANDROIDBTFSMSTATE_SENDINGDATA : break;
584
585 case DIOANDROIDBTFSMSTATE_DISCONNECTING : break;
586 }
587 }
588 else
589 {
590 if(diostream->GetEvent()<DIOANDROIDBT_LASTEVENT)
591 {
592 diostream->CheckTransition();
593
594 switch(diostream->GetCurrentState())
595 {
596 case DIOANDROIDBTFSMSTATE_NONE : break;
597
598 case DIOANDROIDBTFSMSTATE_GETTINGCONNECTION : { struct sockaddr_rc loc_addr = { 0 };
599 struct sockaddr_rc rem_addr = { 0 };
600
601 if(diostream->config->IsServer())
602 {
603 socklen_t opt = sizeof(rem_addr);
604
605 diostream->handleserver = socket(AF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM);
606 if(diostream->handleserver < 0)
607 {
608 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
609 break;
610 }
611
612 loc_addr.rc_family = AF_BLUETOOTH;
613 //str2ba(device->GetMAC()->GetOEM(), &loc_addr.rc_bdaddr);
614 loc_addr.rc_channel = (uint8_t) 1;
615
616 int err;
617 int port = 1;
618 for(port = 1; port <= 31; port++ )
619 {
620 loc_addr.rc_channel = port;
621 err = bind(diostream->handleserver, (struct sockaddr *)&loc_addr, sizeof(loc_addr));
622 if( ! err || errno == EINVAL )
623 {
624 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
625 break;
626 }
627 }
628
629 if(port == 31)
630 {
631 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
632 break;
633 }
634
635 listen(diostream->handleserver, 1);
636
637 diostream->sdpserversession =(void*)diostream->SDP_RegisterService((char*)diostream->config->GetServerPropertys(0),
638 (char*)diostream->config->GetServerPropertys(1),
639 (char*)diostream->config->GetServerPropertys(2),port);
640 if(!diostream->sdpserversession)
641 {
642 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
643 break;
644 }
645
646
647 diostream->handlesocket = accept(diostream->handleserver, (struct sockaddr *)&rem_addr, &opt);
648
649 fcntl(diostream->handlesocket, F_SETFL, fcntl(diostream->handlesocket, F_GETFL,0) | O_NONBLOCK);
650 }
651 else
652 {
653 XSTRING localMACstring;
654 XSTRING remoteMACstring;
655
656 diostream->config->GetLocalMAC()->GetXString(localMACstring);
657 diostream->config->GetRemoteMAC()->GetXString(remoteMACstring);
658
659 diostream->handlesocket = socket(AF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM); //socket(PF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM);
660 if(diostream->handlesocket < 0)
661 {
662 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
663 break;
664 }
665
666 if(diostream->config->GetPIN()->GetSize())
667 {
668 int socket_opt = 0;
669 socklen_t size_opt = sizeof(socket_opt);
670
671 if(getsockopt(diostream->handlesocket, SOL_RFCOMM, RFCOMM_LM, &socket_opt, &size_opt) < 0)
672 {
673 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
674 break;
675 }
676
677 socket_opt |= (RFCOMM_LM_AUTH | RFCOMM_LM_ENCRYPT);
678
679 if(socket_opt != 0)
680 {
681 if(setsockopt(diostream->handlesocket, SOL_RFCOMM, RFCOMM_LM, &socket_opt, sizeof(socket_opt)) < 0)
682 {
683 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
684 break;
685 }
686 }
687
688 diostream->ManagementOfPIN(true, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
689
690 } else diostream->ManagementOfPIN(false, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
691
692
693 loc_addr.rc_family = AF_BLUETOOTH;
694 loc_addr.rc_channel = (uint8_t)0;
695
696 { XBUFFER charstr;
697
698 localMACstring.ConvertToASCII(charstr);
699 str2ba(charOEM.GetPtrChar(), &loc_addr.rc_bdaddr);
700 }
701
702 int error = bind(diostream->handlesocket, (struct sockaddr *)&loc_addr, sizeof(loc_addr));
703 if(error < 0)
704 {
705 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
706 break;
707 }
708
709 rem_addr.rc_family = AF_BLUETOOTH;
710 rem_addr.rc_channel = (uint8_t) diostream->config->GetRemoteChannel();
711
712 { XBUFFER charstr;
713
714 remoteMACstring.ConvertToASCII(charstr);
715 str2ba(charstr.GetPtrChar(), &rem_addr.rc_bdaddr);
716 }
717
718 diostream->SetIsRefusedConnection(false);
719
720 fcntl(diostream->handlesocket, F_SETFL, fcntl(diostream->handlesocket, F_GETFL,0) | O_NONBLOCK);
721
722 connect(diostream->handlesocket, (struct sockaddr *)&rem_addr, sizeof(rem_addr));
723
724 }
725 }
726 break;
727
728 case DIOANDROIDBTFSMSTATE_CONNECTED : diostream->SetEvent(DIOANDROIDBTFSMEVENT_WAITINGTOREAD);
729 break;
730
731 case DIOANDROIDBTFSMSTATE_WAITINGTOREAD : break;
732
733 case DIOANDROIDBTFSMSTATE_SENDINGDATA : break;
734
735 case DIOANDROIDBTFSMSTATE_DISCONNECTING : if(diostream->config->GetPIN()->GetSize())
736 {
737 XSTRING localMACstring;
738 XSTRING remoteMACstring;
739
740 diostream->config->GetLocalMAC()->GetXString(localMACstring);
741 diostream->config->GetRemoteMAC()->GetXString(remoteMACstring);
742
743 diostream->ManagementOfPIN(false, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
744 }
745
746 diostream->threadconnection->Run(false);
747 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
748 break;
749 }
750 }
751 }
