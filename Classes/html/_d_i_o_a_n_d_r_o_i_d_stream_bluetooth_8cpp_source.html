243int DIOANDROIDSTREAMBLUETOOTH::IsReadyConnect(int socket)
244{
245 struct timeval tv;
246 int rc;
247 fd_set fdr;
248 fd_set fdw;
249 fd_set fds;
250
251 if(socket==-1) return -1;
252
253 FD_ZERO(&fdr);
254 FD_ZERO(&fdw);
255 FD_ZERO(&fds);
256
257 FD_SET((unsigned int)socket,&fdr);
258 FD_SET((unsigned int)socket,&fdw);
259 FD_SET((unsigned int)socket,&fds);
260
261 tv.tv_sec = 0;
262 tv.tv_usec = 100;
263
264 rc = select(socket+1, &fdr, &fdw, &fds, &tv);
265 if(rc==-1) return -1;
266 if(rc== 2) return -1;
267
268 int status1 = FD_ISSET(socket,&fdr) ? 1 : 0;
269 int status2 = FD_ISSET(socket,&fdw) ? 1 : 0;
270 int status3 = FD_ISSET(socket,&fds) ? 1 : 0;
271
272 if(status3) return -1;
273 if(((status1) || (status2)) && (rc==1)) return 1;
274
275 return 0;
276}

293sdp_session_t* DIOANDROIDSTREAMBLUETOOTH::SDP_RegisterService(char* service_name,char* service_dsc,char* service_prov,int rfcomm_channel)
294{
295 uint32_t service_uuid_int[] = { 0x0111, 10, 1, 1 };
296 //uint8_t rfcomm_channel = 1;
297
298 uuid_t root_uuid;
299 uuid_t l2cap_uuid;
300 uuid_t rfcomm_uuid;
301 uuid_t svc_uuid;
302
303 sdp_list_t* l2cap_list = 0;
304 sdp_list_t* rfcomm_list = 0;
305 sdp_list_t* root_list = 0;
306 sdp_list_t* proto_list = 0;
307 sdp_list_t* access_proto_list = 0;
308 sdp_data_t* channel = 0;
309//sdp_data_t* psm = 0;
310
311 sdp_record_t record;
312
313 bdaddr_t bdaddr_any = {0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
314 bdaddr_t bdaddr_local = {0x00, 0x00, 0x00, 0xff, 0xff, 0xff };
315
316
317 sdp_session_t* session = sdp_connect(&bdaddr_any, &bdaddr_local, SDP_RETRY_IF_BUSY);
318
319 memset((void *)&record, 0, sizeof(sdp_record_t));
320 record.handle = 0xffffffff;
321
322 sdp_uuid16_create(&root_uuid, PUBLIC_BROWSE_GROUP);
323 root_list = sdp_list_append(0, &root_uuid);
324 sdp_set_browse_groups(&record, root_list );
325
326 // set the general service ID
327 sdp_uuid128_create( &svc_uuid, &service_uuid_int );
328 sdp_set_service_id(&record, svc_uuid );
329
330 // set l2cap information
331 sdp_uuid16_create(&l2cap_uuid, L2CAP_UUID);
332 l2cap_list = sdp_list_append( 0, &l2cap_uuid );
333 proto_list = sdp_list_append( 0, l2cap_list );
334
335 // set rfcomm information
336 sdp_uuid16_create(&rfcomm_uuid, RFCOMM_UUID);
337 channel = sdp_data_alloc(SDP_UINT8, &rfcomm_channel);
338 rfcomm_list = sdp_list_append( 0, &rfcomm_uuid );
339 sdp_list_append( rfcomm_list, channel );
340 sdp_list_append( proto_list, rfcomm_list );
341
342 // attach protocol information to service record
343 access_proto_list = sdp_list_append( 0, proto_list );
344 sdp_set_access_protos( &record, access_proto_list );
345
346 // set the name, provider, and description
347 sdp_set_info_attr(&record, service_name, service_prov, service_dsc);
348
349 // connect to the local SDP server, register the service record, and
350 // disconnect
351 sdp_record_register(session, &record, 0);
352
353 // cleanup
354 sdp_data_free(channel);
355 sdp_list_free(l2cap_list , 0);
356 sdp_list_free(rfcomm_list , 0);
357 sdp_list_free(root_list , 0);
358 sdp_list_free(access_proto_list , 0);
359
360 return session;
361}

433 }
434 }
435 }
436
437 if(active)
438 {
439 if(indexfound != XSTRING_NOTFOUND)
440 {
441 XSTRING* line = filetxt->GetLine(indexfound);
442 if(line) line->Format(__L("%s %s"), remote.Get(), PIN.Get());
443 }
444 else
445 {
446 XSTRING line;
447 line.Format(__L("%s %s"), remote.Get(), PIN.Get());
448 filetxt->AddLine(line);
449 }
450
451 status = filetxt->WriteAllFile();
452 }
453 else
454 {
455 filetxt->DeleteLine(indexfound);
456 }
457
458 filetxt->Close();
459
460 if(!filetxt->GetNLines())
461 {
462 filetxt->GetPrimaryFile()->Erase(xpathfile);
463 }
464 }
465 }
466
467 delete filetxt;
468 }
469
470 GEN_XFACTORY.Delete_Dir(xdir);
471
472 return status;

485void DIOANDROIDSTREAMBLUETOOTH::ThreadRunFunction(void* param)
486{
487 DIOANDROIDSTREAMBLUETOOTH* diostream = (DIOANDROIDSTREAMBLUETOOTH*)param;
488 if(!diostream) return;
489
490 if(diostream->GetEvent()==DIOANDROIDBTFSMEVENT_NONE) // No hay nuevos Eventos
491 {
492 switch(diostream->GetCurrentState())
493 {
494 case DIOANDROIDBTFSMSTATE_NONE : break;
495
496 case DIOANDROIDBTFSMSTATE_GETTINGCONNECTION : switch(diostream->IsReadyConnect(diostream->handlesocket))
497 {
498 case -1: { int sockerr = 0;
499 socklen_t sockerrlen = sizeof(sockerr);
500
501 if(!getsockopt(diostream->handlesocket, SOL_SOCKET, SO_ERROR,(void *)&sockerr, &sockerrlen))
502 {
503 if(sockerr==ECONNREFUSED) diostream->SetIsRefusedConnection(true);
504 }
505
506 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
507 }
508 break;
509
510 case 1: diostream->SetEvent(DIOANDROIDBTFSMEVENT_CONNECTED);
511 diostream->status = DIOSTREAMSTATUS_CONNECTED;
512 break;
513
514 default: diostream->status = DIOSTREAMSTATUS_GETTINGCONNECTION;
515 break;
516 }
517 break;
518
519 case DIOANDROIDBTFSMSTATE_CONNECTED : break;
520
521 case DIOANDROIDBTFSMSTATE_WAITINGTOREAD : { struct timeval waitd;
522 fd_set read_flags;
523 fd_set write_flags;
524
525 waitd.tv_sec = 0;
526 waitd.tv_usec = 100;
527
528 FD_ZERO(&read_flags);
529 FD_ZERO(&write_flags);
530
531 if(diostream->handlesocket<0)
532 {
533 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
534 break;
535 }
536
537 if(!diostream->IsBlockRead()) FD_SET(diostream->handlesocket, &read_flags);
538 if(!diostream->IsBlockWrite())
539 {
540 if(diostream->outbuffer->GetSize()) FD_SET(diostream->handlesocket, &write_flags);
541 }
542
543 int error = select(diostream->handlesocket+1, &read_flags,&write_flags,(fd_set*)0,&waitd);
544 if(error == -1)
545 {
546 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
547 break;
548 }
549
550 if(FD_ISSET(diostream->handlesocket, &read_flags)) //Socket ready for reading
551 {
552 memset(diostream->buffer,0,DIOSTREAM_MAXBUFFER);
553
554 int size = recv(diostream->handlesocket,(char*)diostream->buffer,DIOSTREAM_MAXBUFFER,0);
555 if(size<=0)
556 {
557 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
558 break;
559 }
560 else diostream->inbuffer->Add(diostream->buffer,size);
561
562 FD_CLR(diostream->handlesocket, &read_flags);
563 }
564
565 if(FD_ISSET(diostream->handlesocket, &write_flags)) //Socket ready for writing
566 {
567 int esize = diostream->outbuffer->GetSize();
568 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
569
570 if(esize)
571 {
572 memset(diostream->buffer,0,DIOSTREAM_MAXBUFFER);
573
574 diostream->outbuffer->Get(diostream->buffer,esize,0);
575
576 int size = send(diostream->handlesocket,(char*)diostream->buffer,esize,0);
577 if(size<=0)
578 {
579 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
580 break;
581
582 } else diostream->outbuffer->Extract(diostream->buffer,0,size);
583 }
584
585 FD_CLR(diostream->handlesocket, &write_flags);
586 }
587 }
588 break;
589
590 case DIOANDROIDBTFSMSTATE_SENDINGDATA : break;
591
592 case DIOANDROIDBTFSMSTATE_DISCONNECTING : break;
593 }
594 }
595 else
596 {
597 if(diostream->GetEvent()<DIOANDROIDBT_LASTEVENT)
598 {
599 diostream->CheckTransition();
600
601 switch(diostream->GetCurrentState())
602 {
603 case DIOANDROIDBTFSMSTATE_NONE : break;
604
605 case DIOANDROIDBTFSMSTATE_GETTINGCONNECTION : { struct sockaddr_rc loc_addr = { 0 };
606 struct sockaddr_rc rem_addr = { 0 };
607
608 if(diostream->config->IsServer())
609 {
610 socklen_t opt = sizeof(rem_addr);
611
612 diostream->handleserver = socket(AF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM);
613 if(diostream->handleserver < 0)
614 {
615 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
616 break;
617 }
618
619 loc_addr.rc_family = AF_BLUETOOTH;
620 //str2ba(device->GetMAC()->GetOEM(), &loc_addr.rc_bdaddr);
621 loc_addr.rc_channel = (uint8_t) 1;
622
623 int err;
624 int port = 1;
625 for(port = 1; port <= 31; port++ )
626 {
627 loc_addr.rc_channel = port;
628 err = bind(diostream->handleserver, (struct sockaddr *)&loc_addr, sizeof(loc_addr));
629 if( ! err || errno == EINVAL )
630 {
631 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
632 break;
633 }
634 }
635
636 if(port == 31)
637 {
638 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
639 break;
640 }
641
642 listen(diostream->handleserver, 1);
643
644 diostream->sdpserversession =(void*)diostream->SDP_RegisterService((char*)diostream->config->GetServerPropertys(0),
645 (char*)diostream->config->GetServerPropertys(1),
646 (char*)diostream->config->GetServerPropertys(2),port);
647 if(!diostream->sdpserversession)
648 {
649 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
650 break;
651 }
652
653
654 diostream->handlesocket = accept(diostream->handleserver, (struct sockaddr *)&rem_addr, &opt);
655
656 fcntl(diostream->handlesocket, F_SETFL, fcntl(diostream->handlesocket, F_GETFL,0) | O_NONBLOCK);
657 }
658 else
659 {
660 XSTRING localMACstring;
661 XSTRING remoteMACstring;
662
663 diostream->config->GetLocalMAC()->GetXString(localMACstring);
664 diostream->config->GetRemoteMAC()->GetXString(remoteMACstring);
665
666 diostream->handlesocket = socket(AF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM); //socket(PF_BLUETOOTH, SOCK_STREAM, BTPROTO_RFCOMM);
667 if(diostream->handlesocket < 0)
668 {
669 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
670 break;
671 }
672
673 if(diostream->config->GetPIN()->GetSize())
674 {
675 int socket_opt = 0;
676 socklen_t size_opt = sizeof(socket_opt);
677
678 if(getsockopt(diostream->handlesocket, SOL_RFCOMM, RFCOMM_LM, &socket_opt, &size_opt) < 0)
679 {
680 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
681 break;
682 }
683
684 socket_opt |= (RFCOMM_LM_AUTH | RFCOMM_LM_ENCRYPT);
685
686 if(socket_opt != 0)
687 {
688 if(setsockopt(diostream->handlesocket, SOL_RFCOMM, RFCOMM_LM, &socket_opt, sizeof(socket_opt)) < 0)
689 {
690 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
691 break;
692 }
693 }
694
695 diostream->ManagementOfPIN(true, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
696
697 } else diostream->ManagementOfPIN(false, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
698
699
700 loc_addr.rc_family = AF_BLUETOOTH;
701 loc_addr.rc_channel = (uint8_t)0;
702
703 { XBUFFER charstr;
704
705 localMACstring.ConvertToASCII(charstr);
706 str2ba(charOEM.GetPtrChar(), &loc_addr.rc_bdaddr);
707 }
708
709 int error = bind(diostream->handlesocket, (struct sockaddr *)&loc_addr, sizeof(loc_addr));
710 if(error < 0)
711 {
712 diostream->SetEvent(DIOANDROIDBTFSMEVENT_DISCONNECTING);
713 break;
714 }
715
716 rem_addr.rc_family = AF_BLUETOOTH;
717 rem_addr.rc_channel = (uint8_t) diostream->config->GetRemoteChannel();
718
719 { XBUFFER charstr;
720
721 remoteMACstring.ConvertToASCII(charstr);
722 str2ba(charstr.GetPtrChar(), &rem_addr.rc_bdaddr);
723 }
724
725 diostream->SetIsRefusedConnection(false);
726
727 fcntl(diostream->handlesocket, F_SETFL, fcntl(diostream->handlesocket, F_GETFL,0) | O_NONBLOCK);
728
729 connect(diostream->handlesocket, (struct sockaddr *)&rem_addr, sizeof(rem_addr));
730
731 }
732 }
733 break;
734
735 case DIOANDROIDBTFSMSTATE_CONNECTED : diostream->SetEvent(DIOANDROIDBTFSMEVENT_WAITINGTOREAD);
736 break;
737
738 case DIOANDROIDBTFSMSTATE_WAITINGTOREAD : break;
739
740 case DIOANDROIDBTFSMSTATE_SENDINGDATA : break;
741
742 case DIOANDROIDBTFSMSTATE_DISCONNECTING : if(diostream->config->GetPIN()->GetSize())
743 {
744 XSTRING localMACstring;
745 XSTRING remoteMACstring;
746
747 diostream->config->GetLocalMAC()->GetXString(localMACstring);
748 diostream->config->GetRemoteMAC()->GetXString(remoteMACstring);
749
750 diostream->ManagementOfPIN(false, localMACstring, remoteMACstring, (*diostream->config->GetPIN()));
751 }
752
753 diostream->threadconnection->Run(false);
754 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
755 break;
756 }
757 }
758 }
