189 {
190 //

192 // i declare CBUFFER as static.
193 // The allocated is changed only when RegQueryValueEx return ERROR_MORE_DATA
194 //
195 static CBUFFER buffer(XWINDOWSSYSTEM_TOTALBYTES);

197 LONG result;
198
199 XSTRING keyname;
200 keyname.Format(__L("%d"), objectindex);
201
202 buffer.Reset();
203 while((result = RegQueryValueEx( HKEY_PERFORMANCE_DATA , keyname.Get()
204 , NULL
205 , NULL

212 buffer.Realloc(buffersize);

219 T GetCounterValue (PPERF_OBJECT_TYPE perfobj, DWORD counterindex, LPCTSTR instancename)
220 {
221 PPERF_COUNTER_DEFINITION perfcntr = NULL;
222 PPERF_INSTANCE_DEFINITION perfinst = NULL;
223 PPERF_COUNTER_BLOCK counterblock = NULL;
224
225 // Get the first counter.
226
227 perfcntr = FirstCounter( perfobj );
228
229 for(DWORD j=0; j<perfobj->NumCounters; j++ )
230 {
231 if (perfcntr->CounterNameTitleIndex == counterindex) break;
232
233 // Get the next counter.
234 perfcntr = NextCounter( perfcntr );
235 }
236
237 if(perfobj->NumInstances == PERF_NO_INSTANCES)
238 {
239 counterblock = (PPERF_COUNTER_BLOCK) ((LPBYTE) perfobj + perfobj->DefinitionLength);
240 }
241 else
242 {
243 perfinst = FirstInstance( perfobj );
244
245 // Look for instance instancename
246 //_bstr_t bstrInstance;
247 //_bstr_t bstrInputInstance = instancename;
248
249 XSTRING instancenamestr;
250 XSTRING instancenamestr2;
251
252 instancenamestr = instancename;
253
254 for(int k=0; k<perfobj->NumInstances; k++)
255 {
256 instancenamestr2 = (XCHAR*)((PBYTE)perfinst + perfinst->NameOffset);
257
258 if(!instancenamestr.Compare(instancenamestr2))
259 {
260 counterblock = (PPERF_COUNTER_BLOCK) ((LPBYTE) perfinst + perfinst->ByteLength);
261 break;
262 }
263
264 // Get the next instance.
265
266 perfinst = NextInstance( perfinst );
267 }
268 }
269
270 if(counterblock)
271 {
272 T* value = NULL;
273
274 value = (T*)((LPBYTE) counterblock + perfcntr->CounterOffset);
275 return *value;
276 }
277
278 return -1;
279 }

283 {
284 int PROC_ID_COUNTER = 784;
285 BOOL processIDexist = FALSE;
286 PPERF_COUNTER_DEFINITION perfcntr = NULL;

288 PPERF_COUNTER_DEFINITION procIDperfcntr = NULL;

291
292 // Get the first counter.
293
294 perfcntr = FirstCounter(perfobj);
295
296 for(DWORD j=0; j<perfobj->NumCounters; j++)
297 {
298 if(perfcntr->CounterNameTitleIndex == PROC_ID_COUNTER)
299 {
300 procIDperfcntr = perfcntr;
301 if(therequestedperfcntr) break;
302 }
303

305 {
306 therequestedperfcntr = perfcntr;
307 if(procIDperfcntr) break;
308 }
309
310 // Get the next counter.
311
312 perfcntr = NextCounter( perfcntr );
313 }
314
315 if(perfobj->NumInstances == PERF_NO_INSTANCES)
316 {
317 counterblock = (PPERF_COUNTER_BLOCK) ((LPBYTE) perfobj + perfobj->DefinitionLength);
318 }
319 else
320 {
321 perfinst = FirstInstance( perfobj );
322
323 for(int k=0; k<perfobj->NumInstances; k++)

326 if(counterblock)

338 // Get the next instance.

340 perfinst = NextInstance( perfinst );
341 }
342 }
343
344 if(processIDexist && counterblock)

346 T *value = NULL;
