396bool DIOWINDOWSSTREAMWIFIREMOTEENUMDEVICES::SetAllConnectionStatus(bool activate)
397{
398 HANDLE handleclient = NULL;
399 DWORD max_client = 2;
400 DWORD current_version = 0;
401 DWORD result = 0;
402 PWLAN_INTERFACE_INFO_LIST interfacelist = NULL;

406 result = WlanOpenHandle(max_client, NULL, &current_version, &handleclient);

412

414 {
415 bool needactivated = false;
416
417 interfacelinfo = (WLAN_INTERFACE_INFO*)&interfacelist->InterfaceInfo[c];
418
419 switch(interfacelinfo->isState)

421 case wlan_interface_state_not_ready : break; // Not ready

431
432 case wlan_interface_state_associating : break; // Attempting to associate with a network
433
434 case wlan_interface_state_discovering : break; // Auto configuration is discovering settings for the network
435
436 case wlan_interface_state_authenticating : break; // In process of authenticating
437

441 if(needactivated)
442 {

607void WLanCallback(WLAN_NOTIFICATION_DATA *scannotificationdata, PVOID context)
608{
609 // Get the data from my struct. If it's null, nothing to do
610 WLAN_CALLBACK_INFO* callbackinfo = (WLAN_CALLBACK_INFO*)context;
611 if(!callbackinfo)
612 {
613 return;
614 }
615
616 // Check the GUID in the struct against the GUID in the notification data, return if they don't match
617 if(memcmp(&callbackinfo->interfaceGUID, &scannotificationdata->InterfaceGuid, sizeof(GUID)) != 0)
618 {
619 return;
620 }
621
622 // If the notification was for a scan complete or failure then we need to set the event
623 if((scannotificationdata->NotificationCode == wlan_notification_acm_scan_complete) || (scannotificationdata->NotificationCode == wlan_notification_acm_scan_fail))
624 {
625 //Set the notification code as the callbackReason
626 callbackinfo->callbackreason = scannotificationdata->NotificationCode;
627
628 //Set the event
629 SetEvent(callbackinfo->scanevent);
630 }
631
632 return;
633}
