130bool DIONTP::GetTimeResponse(XCHAR* urlntpserver, int timeout, bool hardwareuselittleendian, DIONTPRESPONSE& response)
131{
132 if(!xtimer) return false;
133 if((!diostreamudpcfg) || (!diostreamudp)) return false;
134
135 bool status = false;
136
137 memset(&response, 0, sizeof(DIONTPRESPONSE));
138
139 diostreamudpcfg->SetMode(DIOSTREAMMODE_CLIENT);
140 diostreamudpcfg->SetIsUsedDatagrams(false);
141 diostreamudpcfg->GetRemoteURL()->Set(urlntpserver);
142 diostreamudpcfg->SetRemotePort(DIONTP_DEFAULTPORT);
143
144 if(!diostreamudp->Open()) return false;
145
146 DIONTPBASICINFO basicinfo;
147
148 memset((XBYTE*)&basicinfo, 0, sizeof(DIONTPBASICINFO));
149
150 basicinfo.livnmode = 27; //Encoded representation which represents NTP Client Request & NTP version 3.0
151
152 if(diostreamudp->Write((XBYTE*)&basicinfo,sizeof(DIONTPBASICINFO)))
153 {
154 xtimer->Reset();
155
156 DIONTPFULLPACKET fullpacket;
157 XDWORD br = 0;
158 while(!br)
159 {
160 memset((XBYTE*)&fullpacket,0 ,sizeof(DIONTPFULLPACKET));
161
162 XDWORD size = diostreamudp->GetInXBuffer()->GetSize();
163 if(size >= sizeof(DIONTPBASICINFO))
164 {
165 br = diostreamudp->Read((XBYTE*)&fullpacket, size);
166 if(br)
167 {
168 response.stratum = fullpacket.basic.stratum;
169 response.leapindicator = (fullpacket.basic.livnmode & 0xC0) >> 6;
170 response.originatetime = ConvertTimePacket(fullpacket.basic.originatetimestamp , hardwareuselittleendian);
171 response.receivetime = ConvertTimePacket(fullpacket.basic.receivetimestamp , hardwareuselittleendian);
172 response.transmittime = ConvertTimePacket(fullpacket.basic.transmittimestamp , hardwareuselittleendian);
173
174 long long roundtripdelay = (long long)(response.destinationtime - response.originatetime) - (long long)(response.receivetime - response.transmittime);
175 long long localclockoffset = (long long)(response.receivetime - response.originatetime) + (long long)(response.transmittime - response.destinationtime);
176 response.roundtripdelay = (double)(roundtripdelay);
177 response.localclockoffset = (double)(localclockoffset/2);
178
179 status = true;
180 break;
181 }
182 }
183 else
184 {
185 //XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("ntp [%d] %d"), sizeof(DIONTPBASICINFO), size);
186 }
187
188 if((int)xtimer->GetMeasureSeconds()>= timeout)
189 {
190 status = false;
191 break;
192 }
193 }
194 }
195
196 diostreamudp->Close();
197
198 return status;
199}

343XQWORD DIONTP::ConvertTimePacket(DIOTIMEPACKET& timepacket, bool hardwareuselittleendian)
344{
345 XDWORD dwhigh = (hardwareuselittleendian)?SwapDWORD(timepacket.integer) : timepacket.integer;
346 XDWORD dwlow = (hardwareuselittleendian)?SwapDWORD(timepacket.fractional) : timepacket.fractional;
347 XQWORD timedata;
348
349 timedata = ((XQWORD) dwhigh) << 32;
350 timedata |= dwlow;
351
352 return timedata;
353}
