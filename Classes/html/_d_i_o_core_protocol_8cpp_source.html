656XDWORD DIOCOREPROTOCOL::ReceivedMsg(DIOCOREPROTOCOL_HEADER& header, XBUFFER& content)
657{
658 if(!protocolCFG)
659 {
660 return 0;
661 }
662
663 if(!diostream)
664 {
665 return 0;
666 }
667
668 XBUFFER* readbuffer = diostream->GetInXBuffer();
669 if(!readbuffer)
670 {
671 return 0;
672 }
673
674 readbuffer->ResetPosition();
675
676 if(readbuffer->GetSize() < DIOCOREPROTOCOL_HEADER_SIZE_ID)
677 {
678 return 0;
679 }
680
681 XDWORD start = 0;
682 XDWORD sizeread = 0;
683 XDWORD index = 0;
684 bool status = false;
685
686 for(index=0; index<readbuffer->GetSize()-sizeof(XDWORD); index++)
687 {
688 if(readbuffer->Get((XDWORD&)start, index))
689 {
690 if(start == DIOCOREPROTOCOL_HEADER_MAGIC_ID)
691 {
692 break;
693 }
694 }
695 }
696
697 if(readbuffer->Get((XDWORD&)start, index))
698 {
699 sizeread += sizeof(XDWORD);
700
701 // Compress Data
702 if(start == DIOCOREPROTOCOL_HEADER_MAGIC_ID)
703 {
704 XBUFFER headerbuffer;
705 XWORD sizeheader = 0;
706 XWORD sizeheadercmp = 0;
707 XDWORD crc32header = 0;
708
709 if(readbuffer->Get((XWORD&)sizeheader))
710 {
711 sizeread += sizeof(XWORD);
712
713 if(readbuffer->Get((XWORD&)sizeheadercmp))
714 {
715 sizeread += sizeof(XWORD);
716
717 if(readbuffer->Get((XDWORD&)crc32header))
718 {
719 sizeread += sizeof(XDWORD);
720
721 if(sizeheader)
722 {
723 XBYTE* ptrheader = &readbuffer->Get()[DIOCOREPROTOCOL_HEADER_SIZE_ID];
724 XBYTE* ptrcontent = ptrheader;
725 HASHCRC32 hashCRC32;
726 XDWORD crc32headercalc = 0;
727
728 if(sizeheadercmp)
729 {
730 hashCRC32.Do(ptrheader, sizeheadercmp);
731 }
732 else
733 {
734 hashCRC32.Do(ptrheader, sizeheader);
735 }
736
737 crc32headercalc = hashCRC32.GetResultCRC32();
738
739 if(crc32headercalc == crc32header)
740 {
741 if(sizeheadercmp)
742 {
743 headerbuffer.Resize(sizeheader);
744 status = compressor->Decompress(ptrheader, sizeheadercmp, &headerbuffer);
745 ptrcontent += sizeheadercmp;
746 sizeread += sizeheadercmp;
747
748 if(!status)
749 {
750 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: to decompress header"));
751 }
752 }
753 else
754 {
755 status = headerbuffer.Add(ptrheader, sizeheader);
756 ptrcontent += sizeheader;
757 sizeread += sizeheader;
758
759 if(!status)
760 {
761 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: to get header"));
762 }
763 }
764
765 XFILEJSON* headerxfileJSON = header.GetSerializationXFileJSON();
766 if(headerxfileJSON)
767 {
768 header.GetSerializationXFileJSON()->AddBufferLines(XFILETXTFORMATCHAR_UTF8, headerbuffer);
769 header.GetSerializationXFileJSON()->DecodeAllLines();
770 header.DoDeserialize();
771 }
772
773 if(header.GetContentCompressSize() || header.GetContentSize())
774 {
775 HASHCRC32 hashCRC32;
776 XDWORD crc32contentcalc = 0;
777
778 hashCRC32.Do(ptrcontent, header.GetContentCompressSize()?header.GetContentCompressSize():header.GetContentSize());
779
780 crc32contentcalc = hashCRC32.GetResultCRC32();
781
782 if(header.GetContentCRC32() == crc32contentcalc)
783 {
784 if(header.GetContentCompressSize())
785 {
786 content.Resize(header.GetContentSize());
787 status = compressor->Decompress(ptrcontent, header.GetContentCompressSize(), &content);
788 sizeread += header.GetContentCompressSize();
789
790 if(!status)
791 {
792 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: to decompress content"));
793 }
794 }
795 else
796 {
797 if(header.GetContentSize())
798 {
799 status = content.Add(ptrcontent, header.GetContentSize());
800 sizeread += header.GetContentSize();
801
802 if(!status)
803 {
804 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: to get content"));
805 }
806 }
807 }
808 }
809 }
810 else
811 {
812 if(!header.GetContentCRC32())
813 {
814 status = true;
815 }
816 }
817 }
818
819 // ------------------------------------------------------------------------------------------------------------------------
820 /*
821 XBYTE color = XTRACE_COLOR_GREEN;
822 if(GetProtocolCFG()->GetIsServer())
823 {
824 color = XTRACE_COLOR_BLUE;
825 }
826
827 XTRACE_PRINTCOLOR(color, __L("READ -----------------------------------------------------------------------------------------------"));
828 XTRACE_PRINTDATABLOCKCOLOR(color, readbuffer->Get(), sizeread + index);
829 */
830 // ------------------------------------------------------------------------------------------------------------------------
831
832 status = readbuffer->Extract(NULL, 0, sizeread + index);
833
834 if(!status)
835 {
836 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: eliminate buffer used: %d + %d"), sizeread, index);
837 }
838
839 return sizeread;
840 }
841 else
842 {
843 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: not read size header"));
844 }
845 }
846 else
847 {
848 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: not read header CRC"));
849 }
850 }
851 else
852 {
853 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: not read size header cmp"));
854 }
855 }
856 else
857 {
858 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: not read size header"));
859 }
860 }
861 else
862 {
863 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: not found magic"));
864 }
865 }
866 else
867 {
868 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Header] Error to read msg: read magic"));
869 }
870
871 return sizeread;
872}

1695 }

1698 contentresult.CopyFrom(content);
1699 }
1700
1701 switch(header->GetContentType())
1702 {
1703 case DIOCOREPROTOCOL_HEADER_CONTENTTYPE_NONE : break;
1704
1705 case DIOCOREPROTOCOL_HEADER_CONTENTTYPE_BINARY : XTRACE_PRINTDATABLOCKCOLOR(colormsg, contentresult);

1709
1710 string.ConvertFromUTF8(contentresult);
1711 XTRACE_PRINTCOLOR(colormsg, string.Get());
1712 }
1713 break;
1714
1715 case DIOCOREPROTOCOL_HEADER_CONTENTTYPE_JSON : { XFILEJSON xfileJSON;
1716
1717 xfileJSON.AddBufferLines(XFILETXTFORMATCHAR_UTF8, contentresult);
1718 xfileJSON.DecodeAllLines();

1722 }

1742bool DIOCOREPROTOCOL::GenerateHeaderToSend(DIOCOREPROTOCOL_HEADER* header, XBUFFER& headersenddata, XWORD* headersize)
1743{
1744 XFILEJSON* headerxfileJSON = NULL;
1745 XSTRING headerstring;
1746 XBUFFER headerbuffer;
1747 XBUFFER headercmp;
1748 bool isheadercmp = false;
1749 bool status = false;
1750
1751 if(!header)
1752 {
1753 return status;
1754 }
1755
1756 headersenddata.Empty();
1757
1758 headerxfileJSON = header->GetSerializationXFileJSON();
1759 if(!headerxfileJSON)
1760 {
1761 return false;
1762 }
1763
1764 status = header->DoSerialize();
1765 if(!status)
1766 {
1767 return status;
1768 }
1769
1770 headerxfileJSON->EncodeAllLines(false);
1771 headerxfileJSON->GetAllInOneLine(headerstring);
1772
1773 headerstring.ConvertToUTF8(headerbuffer);
1774
1775 if(protocolCFG)
1776 {
1777 if(protocolCFG->GetCompressHeader())
1778 {
1779 if(!compressor->Compress(headerbuffer.Get(), headerbuffer.GetSize(), &headercmp))
1780 {
1781 return status;
1782 }
1783
1784 if(headercmp.GetSize() < headerbuffer.GetSize())
1785 {
1786 isheadercmp = true;
1787 }
1788 else
1789 {
1790 headercmp.Empty();
1791 }
1792 }
1793 }
1794
1795 headersenddata.Add((XDWORD)DIOCOREPROTOCOL_HEADER_MAGIC_ID);
1796 headersenddata.Add((XWORD)headerbuffer.GetSize());
1797 headersenddata.Add((XWORD)headercmp.GetSize());
1798
1799 HASHCRC32 hashCRC32;
1800
1801 if(isheadercmp)
1802 {
1803 hashCRC32.Do(headercmp);
1804 }
1805 else
1806 {
1807 hashCRC32.Do(headerbuffer);
1808 }
1809
1810 headersenddata.Add((XDWORD)hashCRC32.GetResultCRC32());
1811
1812 if(isheadercmp)
1813 {
1814 if(headersize)
1815 {
1816 (*headersize) = headercmp.GetSize();
1817 }
1818
1819 headersenddata.Add(headercmp);
1820 }
1821 else
1822 {
1823 if(headersize)
1824 {
1825 (*headersize) = headerbuffer.GetSize();
1826 }
1827
1828 headersenddata.Add(headerbuffer);
1829 }
1830
1831 return status;
1832}
