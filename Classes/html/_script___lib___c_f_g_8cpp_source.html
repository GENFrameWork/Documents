182void Call_GetFileCFGValue(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
183{
184 if(!library) return;
185 if(!script) return;
186 if(!params) return;
187 if(!returnvalue) return;
188
189 returnvalue->Set();
190
191 if(params->GetSize() < 2)
192 {
193 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
194 return;
195 }
196
197 XSTRING datakey;
198 SCRIPT_LIB_CFG* libraryCFG = (SCRIPT_LIB_CFG*)library;
199
200 if(libraryCFG->GetXFileCFG())
201 {
202 XSTRING* param1 = (XSTRING*)params->Get(0)->GetData();
203 XSTRING* param2 = (XSTRING*)params->Get(1)->GetData();
204
205 XVARIANT* variable = libraryCFG->GetXFileCFG()->GetValue(param1->Get(), param2->Get());
206 if(variable)
207 {
208 variable->ToString(datakey);
209 delete variable;
210 }
211
212 }
213
214 (*returnvalue) = datakey.Get();
215}
