257bool DIOSTREAMTLS::HandShake_Client_Hello()
258{
259 XBUFFER xbuffer;
260 XWORD size16 = 0;
261 XBYTE size24[3] = { 0, 0, 0 };
262 bool status = false;
263
264 //-------------------------------------------------------------------
265 // TLS Plaintext
266
267 xbuffer.Add((XBYTE)DIOSTREAMTLS_CONTENTTYPE_HANDSHAKE); // Content Type
268 xbuffer.Add((XWORD)SwapWORD(DIOSTREAMTLS_VERSION_TLS_1_0)); // legacy record version
269 xbuffer.Add((XWORD)SwapWORD(size16)); // Length
270
271 //-------------------------------------------------------------------
272 // HandSake protot
273 xbuffer.Add((XBYTE)DIOSTREAMTLS_HANDSHAKETYPE_CLIENT_HELLO); // Handshake type: Client Hello (1)
274 xbuffer.Add(size24, 3);
275 xbuffer.Add((XWORD)SwapWORD(DIOSTREAMTLS_VERSION_TLS_1_2)); // Version TLS 1.2
276 xbuffer.Add(random, DIOSTREAMTLS_RANDOM_SIZE); // Random
277 xbuffer.Add((XBYTE)DIOSTREAMTLS_SESSION_ID_SIZE); // Session ID length
278 xbuffer.Add(sessionID, DIOSTREAMTLS_SESSION_ID_SIZE); // Session ID
279
280 XDWORD size = Write(xbuffer.Get(), xbuffer.GetSize());
281 if(size == xbuffer.GetSize()) status = true;
282
283 return status;
284}
