277 bool Rotate (float degrees)
278 {
279 float radians = DEGREE2RADIAN(degrees);
280 float oneminusradians = DEGREE2RADIAN((180-degrees));
281
282
283 float cosine = (float)cos(radians);
284 float sine = (float)sin(radians);
285
286 //float oneminuscosine = (float)cos(oneminusradians);
287 float oneminussine = (float)sin(oneminusradians);
288
289 int w = (int)(height*fabs(oneminussine) + width*fabs(cosine));
290 int h = (int)(height*fabs(cosine) + width*fabs(oneminussine));
291
292 //int bytesppixel = 4;
293
294 GRPBITMAPPIXELFORMATBUFFER<PIXELFORMATBUFFER, COLORTYPE>* bitmap = new GRPBITMAPPIXELFORMATBUFFER<PIXELFORMATBUFFER, COLORTYPE>(w, h, mode);
295 if(!bitmap) return false;
296
297 int w2 = (w/2);
298 int h2 = (h/2);
299
300 COLORTYPE* color;
301
302 for(int y=-h2;y<h2;y++)
303 {
304 for(int x=-w2;x<w2;x++)
305 {
306 int sx = (int)((x*cosine) + (y*sine)) + (width/2);
307 int sy = (int)((y*cosine) - (x*sine)) + (height/2);
308
309 if( (sx>=0) && (sx<(int)width) && (sy>=0) && (sy<(int)height) )
310 {
311 color = (COLORTYPE*)GetPixel(sx,sy);
312 bitmap->PutBlendPixel(x+w2, y+h2, color, color->a);
313 }
314 }
315 }
