335bool XFILE::EraseBlock(XQWORD size)
336{
337 XBYTE* tbuffer;
338 XQWORD position;
339 XQWORD sizetomove;
340 XDWORD nblocks;
341 XDWORD rest;
342 bool status = true;
343
344 if(!isopen) return false;
345
346 if(!GetPosition(position)) return false;
347
348 sizetomove = GetSize() - position - size;
349
350 tbuffer= new XBYTE [XFILE_MAXBUFFER];
351 if(!tbuffer) return false;
352
353 nblocks = (XDWORD)(sizetomove / XFILE_MAXBUFFER);
354 rest = (XDWORD)(sizetomove % XFILE_MAXBUFFER);
355
356 for(XDWORD c=0; c<nblocks; c++)
357 {
358 SetPosition(position + size + ((XQWORD)c * XFILE_MAXBUFFER));
359 if(Read(tbuffer, XFILE_MAXBUFFER))
360 {
361 SetPosition(position + ((XQWORD)c * XFILE_MAXBUFFER));
362 Write(tbuffer, XFILE_MAXBUFFER);
363 }
364 }
365
366 SetPosition(position + size +((XQWORD)nblocks * XFILE_MAXBUFFER));
367 if(Read(tbuffer, rest))
368 {
369 SetPosition(position + ((XQWORD)nblocks * XFILE_MAXBUFFER));
370 Write(tbuffer, rest);
371 }
372
373 SetSize(GetSize() - size);
374
375 delete [] tbuffer;
376
377 return status;
378}

393bool XFILE::InsertBlock(XBYTE* buffer, XDWORD size)
394{
395 XBYTE* movebuffer;
396 XQWORD origin;
397 XQWORD position;
398
399 bool status = true;
400
401 if(!isopen) return false;
402
403 if(GetPosition(position)!=true) return false;
404
405 origin = position;
406
407 movebuffer= new XBYTE [XFILE_MAXBUFFER];
408 if(!movebuffer) return false;
409
410 ActualizeSize();
411
412 XQWORD sizefile = GetSize();
413 XQWORD sizetomove = sizefile-position;
414 XDWORD nblockstomove = (XDWORD)(sizetomove / XFILE_MAXBUFFER);
415 XDWORD resttomove = (XDWORD)(sizetomove % XFILE_MAXBUFFER);
416 XDWORD c;
417
418 position = GetSize() - XFILE_MAXBUFFER;
419
420 for(c=0; c<nblockstomove+1; c++)
421 {
422 XDWORD rb;
423
424 if(c>=nblockstomove)
425 {
426 rb = resttomove;
427 position = origin;
428 }
429 else
430 {
431 rb = XFILE_MAXBUFFER;
432 }
433
434 SetPosition(position);
435 Read(movebuffer, &rb);
436
437 SetPosition(position+size);
438 Write(movebuffer,rb);
439
440 position -= XFILE_MAXBUFFER;
441 }
442
443 SetPosition(origin);
444 Write(buffer,size);
445
446 if(status==true) ActualizeSize();
447
448 delete [] movebuffer;
449
450 return status;
451}

480bool XFILE::OverwriteContent(XCHAR* xpath)
481{
482 bool status=true;
483
484 if(!Open(xpath,false)) return false;
485
486 GetSize();
487
488 XBYTE* buffer;
489 const int sizeblock = 1024;
490 int block = (XDWORD)(cachesize/sizeblock);
491 int mod = (XDWORD)(cachesize%sizeblock);
492
493 buffer = new XBYTE [sizeblock];
494 if(!buffer) return false;
495
496 memset(buffer,0,sizeblock);
497
498 for(int c=0;c<block;c++)
499 {
500 if(Write(buffer,sizeblock)!=true)
501 {
502 status=false;
503 break;
504 }
505 }
506
507 if(status==true)
508 {
509 if(Write(buffer,mod)!=true) status=false;
510 }
511
512 delete [] buffer;
513
514 Close();
515
516 return status;
