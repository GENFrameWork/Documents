390GRPVECTORFILERESULT GRPVECTORFILEDXFTEXTSECTIONBLOCKS::ParserTextSection(XFILETXT* fileTXT)
391{
392 XVECTOR<GRPVECTORFILEDXFTEXTPART*> parts;
393 GRPVECTORFILEDXFTEXTPART* part = NULL;
394 GRPVECTORFILEDXFXDATACTRL* xdatactrl = NULL;
395 int indexline = iniline;
396 XSTRING* line;
397
398 part = new GRPVECTORFILEDXFTEXTPART();
399
400 do{ line = fileTXT->GetLine(indexline);
401 if(line)
402 {
403 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
404
405 if(line && !line->Compare(__L("0"),true))
406 {
407 line = fileTXT->GetLine(indexline + 1);
408 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
409
410 if(IsKnownBlock(*line))
411 {
412 indexline++;
413
414 if(part)
415 {
416 if(part->iniline != -1)
417 {
418 part->endline = indexline - 2;
419
420 parts.Add(part);
421 part = NULL;
422
423 part = new GRPVECTORFILEDXFTEXTPART ();
424 }
425 else
426 {
427 if(part->iniline == -1)
428 {
429 part->name = line->Get();
430 part->iniline = indexline + 1;
431 }
432 }
433 }
434 }
435 }
436 }
437
438 indexline++;
439
440 } while(indexline < endline);
441
442 if(part && (part->iniline != -1))
443 {
444 part->endline = indexline-1;
445
446 parts.Add(part);
447 part = NULL;
448 }
449
450 if(part && part->iniline == -1)
451 {
452 delete part;
453 part = NULL;
454 }
455
456 for(XDWORD c=0; c<parts.GetSize(); c++)
457 {
458 part = parts.Get(c);
459 if(part)
460 {
461 GRPVECTORFILEDXFTEXTBLOCK* block = new GRPVECTORFILEDXFTEXTBLOCK();
462 if(block)
463 {
464 GRPVECTORFILEDXFTEXTSECTIONBLOCKDEF* blockDef = IsKnownBlock(part->name);
465 if(blockDef)
466 {
467 block->SetIsEndBlock(blockDef->isendblock);
468
469 if(blockDef->isendblock)
470 {
471 GRPVECTORFILEDXFTEXTBLOCK* lastblock = blocks.Get(blocks.GetSize()-1);
472 if(lastblock)
473 {
474 block->GetName()->Set(lastblock->GetName()->Get());
475 }
476 }
477 }
478
479 indexline = part->iniline;
480
481 bool nomorevalues = false;
482
483 while(part->endline > indexline)
484 {
485 line = fileTXT->GetLine(indexline);
486 if(line)
487 {
488 indexline++;
489
490 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
491
492 int type = line->ConvertToInt();
493
494 if(!type)
495 {
496 line = fileTXT->GetLine(indexline);
497 if(line)
498 {
499 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
500
501 GRPVECTORFILEDXFTEXTSECTIONENTITIES* sectionEntities = new GRPVECTORFILEDXFTEXTSECTIONENTITIES();
502 if(sectionEntities)
503 {
504 sectionEntities->iniline = indexline - 1;
505 sectionEntities->endline = part->endline;
506
507 if(sectionEntities->IsKnownEntity((*line)))
508 {
509 sectionEntities->SetGrpVectorFile(GetGrpVectorFile());
510 sectionEntities->ParserTextSection(fileTXT);
511 indexline = part->endline - 1;
512
513 block->SetSectionEntities(sectionEntities);
514
515 nomorevalues = true;
516 }
517 }
518 }
519 }
520
521 if(!nomorevalues)
522 {
523 GRPVECTORFILEDXFVALUE* value = new GRPVECTORFILEDXFVALUE();
524 if(value)
525 {
526 GRPVECTORFILEDXFTEXTSECTIONBLOCKDEFTYPE* defType = IsKnownTypeValue(part->name, type);
527 if(!defType)
528 {
529 #ifdef TEST_ONLY_DEFINE_IN_BLOCKS
530
531 XSTRING message;
532
533 message.Format(__L("type data of block %d Not register [%d]"), type);
534
535 GRPVECTORFILE_XEVENT vfevent(GetGrpVectorFile(), GRPVECTORFILE_XEVENTTYPE_PartUnknown);
536
537 vfevent.SetType(GRPVECTORFILETYPE_DXF);
538 vfevent.GetPath()->Set(fileTXT->GetPrimaryFile()->GetPathnamefile());
539 vfevent.GetMessage()->Set(message);
540
541 PostEvent(&vfevent, GetGrpVectorFile());
542
543 indexline++;
544 delete value;
545
546 continue;
547
548 #else
549
550 GRPVECTORFILEDXFTEXTSECTIONGENERICDEFTYPE* genDefType = GetGenericDefType(type);
551 if(genDefType)
552 {
553 value->SetType(genDefType->type);
554 value->GetName()->Set(genDefType->name);
555 value->GetRemark()->Set(genDefType->remark);
556 }
557 else
558 {
559 XSTRING message;
560
561 message.Format(__L("type data of block %d Not register [%d]"), type);
562
563 GRPVECTORFILE_XEVENT vfevent(GetGrpVectorFile(), GRPVECTORFILE_XEVENTTYPE_PARTUNKNOWN);
564
565 vfevent.SetType(GRPVECTORFILETYPE_DXF);
566 vfevent.GetPath()->Set(fileTXT->GetPrimaryFile()->GetPathNameFile());
567 vfevent.GetMsg()->Set(message);
568
569 PostEvent(&vfevent, GetGrpVectorFile());
570
571 indexline++;
572 continue;
573 }
574
575 #endif
576 }
577 else
578 {
579 value->SetType(defType->type);
580 value->GetName()->Set(defType->name);
581 value->GetRemark()->Set(defType->remark);
582 }
583
584 line = fileTXT->GetLine(indexline);
585 if(line)
586 {
587 GRPVECTORFILEDXF::ParserTextFilePrepareLine(line);
588
589 GetVariableFromLine(value->GetName()->Get(), type, line, (*value->GetData()));
590
591 switch(IsXDataControl(type, (*line)))
592 {
593 case GRPVECTORFILEDXFTEXTSECTION_XDATACTRL_STATUS_NOT : if(xdatactrl)
594 {
595 if(value)
596 {
597 xdatactrl->GetValues()->Add(value);
598 }
599 }
600 break;
601
602 case GRPVECTORFILEDXFTEXTSECTION_XDATACTRL_STATUS_INI : if(!xdatactrl)
603 {
604 xdatactrl = new GRPVECTORFILEDXFXDATACTRL();
605 if(xdatactrl)
606 {
607 XSTRING name;
608 line->Copy(1, name);
609
610 xdatactrl->GetName()->Set(name);
611 }
612 }
613 break;
614
615 case GRPVECTORFILEDXFTEXTSECTION_XDATACTRL_STATUS_END : if(xdatactrl)
616 {
617 block->GetXDataCtrlList()->Add(xdatactrl);
618 xdatactrl = NULL;
619 }
620 break;
621 }
622
623
624 if(value)
625 {
626 if((value->GetType() == 2) || (value->GetType() == 3))
627 {
628 XSTRING* data = (XSTRING*)(value->GetData()->GetData());
629 if(data) block->GetName()->Set(data->Get());
630 }
631
632 block->AddValue(value);
633 }
634 }
635 }
636 }
637 }
638
639 indexline++;
640
641 }
642
643 AddBlock(block);
644 }
645 }
646 }
647
648 parts.DeleteContents();
649 parts.DeleteAll();
650
651 #ifdef XTRACE_ACTIVE
652 //ShowTraceAllBlocks();
653 #endif
654
655 return GRPVECTORFILERESULT_OK;
656}
