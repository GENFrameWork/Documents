162bool DIOWINDOWSGPIOPCPARALLEL::Ini()
163{
164 XPATH xpathDLL;
165
166 for(int c=0; c<2; c++)
167 {
168 if(xpath)
169 {
170 xpathDLL.Add((*xpath));
171 xpathDLL.Slash_Add();
172 }
173
174 switch(c)
175 {
176 case 0: xpathDLL.Add("inpout32.dll"); break;
177 case 1: xpathDLL.Add("inpout64.dll"); break;
178 }
179
180 handleDLL = LoadLibrary(xpathDLL.Get());
181 if(handleDLL) break;
182 }
183
184 if(!handleDLL) return false;
185
186 outport = (DIOWGPIOOUT) GetProcAddress(handleDLL, "Out32");
187 inpport = (DIOWGPIOINP) GetProcAddress(handleDLL, "Inp32");;
188 isdriveropen = (DIOWGPIOISDRIVEROPEN) GetProcAddress(handleDLL, "IsInpOutDriverOpen");
189 isdriver64bit = (DIOWGPIOISDRIVER64BITS) GetProcAddress(handleDLL, "IsXP64Bit");
190
191 if((!outport) || (!inpport) || (!isdriveropen) || (!isdriver64bit))
192 {
193 FreeLibrary(handleDLL);
194 return false;
195 }
196
197 return true;
198}
