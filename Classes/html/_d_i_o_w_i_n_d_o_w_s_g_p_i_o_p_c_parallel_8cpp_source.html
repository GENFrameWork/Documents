155bool DIOWINDOWSGPIOPCPARALLEL::Ini()
156{
157 XPATH xpathDLL;
158
159 for(int c=0; c<2; c++)
160 {
161 if(xpath)
162 {
163 xpathDLL.Add((*xpath));
164 xpathDLL.Slash_Add();
165 }
166
167 switch(c)
168 {
169 case 0: xpathDLL.Add("inpout32.dll"); break;
170 case 1: xpathDLL.Add("inpout64.dll"); break;
171 }
172
173 handleDLL = LoadLibrary(xpathDLL.Get());
174 if(handleDLL) break;
175 }
176
177 if(!handleDLL) return false;
178
179 outport = (DIOWGPIOOUT) GetProcAddress(handleDLL, "Out32");
180 inpport = (DIOWGPIOINP) GetProcAddress(handleDLL, "Inp32");;
181 isdriveropen = (DIOWGPIOISDRIVEROPEN) GetProcAddress(handleDLL, "IsInpOutDriverOpen");
182 isdriver64bit = (DIOWGPIOISDRIVER64BITS) GetProcAddress(handleDLL, "IsXP64Bit");
183
184 if((!outport) || (!inpport) || (!isdriveropen) || (!isdriver64bit))
185 {
186 FreeLibrary(handleDLL);
187 return false;
188 }
189
190 return true;
191}
