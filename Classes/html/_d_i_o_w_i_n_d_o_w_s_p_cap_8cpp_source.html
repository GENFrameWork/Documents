197bool DIOWINDOWSPCAP::CreateListNetInterfaces()
198{
199 pcap_if_t* allnetinterfaces;
200 pcap_if_t* netinterface;
201 char errbuf[PCAP_ERRBUF_SIZE];
202
203 // Retrieve the device list
204 if(pcap_findalldevs(&allnetinterfaces, errbuf) == -1) return false;
205
206 // Create list
207 for(netinterface=allnetinterfaces; netinterface; netinterface=netinterface->next)
208 {
209 DIOPCAPNETINTERFACE* _netinterface = new DIOPCAPNETINTERFACE();
210 if(_netinterface)
211 {
212 _netinterface->GetName()->Set(netinterface->name);
213 _netinterface->GetDescription()->Set(netinterface->description);
214
215 if(netinterface->flags & PCAP_IF_UP)
216 {
217 _netinterface->SetIsUp(true);
218 }
219
220 if(netinterface->flags & PCAP_IF_RUNNING)
221 {
222 _netinterface->SetIsRunning(true);
223 }
224
225 if(netinterface->flags & PCAP_IF_WIRELESS)
226 {
227 _netinterface->SetIsWireless(true);
228 }
229
230 if(netinterface->flags & PCAP_IF_LOOPBACK)
231 {
232 _netinterface->SetIsLoopBack(true);
233 }
234
235 netinterfaces.Add(_netinterface);
236 }
237 }
238
239 pcap_freealldevs(allnetinterfaces);
240
241 return true;
242}
