346bool DIOCLIPROTOCOLBUS::ReceivedCommand(XSTRING& originID, XSTRING& command, XVECTOR<XSTRING*>& params, XSTRING& answer)
347{
348 bool status = false;
349
350 answer.Empty();
351
352 DIOCLIPROTOCOLCOMMAND* protocolcommand = GetCommand(command.Get());
353 if(protocolcommand)
354 {
355 if(protocolcommand->GetNParams() < (int)params.GetSize()) return false;
356
357 if(!command.Compare(DIOCLIPROTOCOLBUS_COMMAND_VERSION, true))
358 {
359 answer.Format(__L("%d.%d.%d"), version, subversion, subversionerror);
360 status = true;
361 }
362
363 if(!command.Compare(DIOCLIPROTOCOLBUS_COMMAND_PING, true))
364 {
365 status = true;
366 }
367
368 if(!command.Compare(DIOCLIPROTOCOLBUS_COMMAND_ENUM, true))
369 {
370 enum_mutex->Lock();
371
372 enum_sendoriginID = originID;
373
374 enum_mutex->UnLock();
375
376 threadsendenumrequest->Ini();
377 }
378
379 if(!command.Compare(DIOCLIPROTOCOLBUS_COMMAND_ENUMREQUEST, true))
380 {
381 bool found = false;
382
383 for(XDWORD c=0; c<enum_remotedevices.GetSize(); c++)
384 {
385 XSTRING* device = enum_remotedevices.Get(c);
386 if(device)
387 {
388 if(!device->Compare(originID, true)) found = true;
389 }
390 }
391
392 if(!found)
393 {
394 XSTRING* origin = new XSTRING();
395 if(origin)
396 {
397 (*origin) = originID;
398 enum_remotedevices.Add(origin);
399 }
400 }
401
402 status = true;
403 }
404 }
405
406 return status;
407}
