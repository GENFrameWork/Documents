374bool APPFLOWCHECKRESOURCESHARDWARE::CheckTotalCPUUsageStatus()
375{
376 if(!cfg)
377 {
378 return false;
379 }
380
381 int cpuusage = GEN_XSYSTEM.GetCPUUsageTotal();
382
383 if(cpuusage != XSYSTEM_CPUUSAGE_ERROR)
384 {
385 if(cpuusage > maxtotalCPUusage)
386 {
387 maxtotalCPUusage = cpuusage;
388 }
389
390 ntotalCPUusage += cpuusage;
391 ntotalCPUsamples++;
392 }
393
394 if(ntotalCPUsamples > 5)
395 {
396 if(GetCPUTotalCPUUsageAverange() > (XDWORD)cfg->CheckResourcesHardware_GetTotalCPUUsageLimitPercent())
397 {
398 APPFLOWCHECKRESOURCESHARDWARE_XEVENT xevent(this, APPFLOWCHECKRESOURCESHARDWARE_XEVENT_TYPE_TOTALCPUUSAGELIMIT);
399 xevent.SetActualTotalCPUUsage(cpuusage);
400 PostEvent(&xevent);
401 }
402 }
403
404 return true;
405}
