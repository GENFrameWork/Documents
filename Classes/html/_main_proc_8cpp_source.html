267bool MAINPROC::CreateParams(XCHAR* commandline)
268{
269 XSTRING _commandline = commandline;
270
271 if(_commandline.IsEmpty()) return false;
272
273 int start = 0;
274 XDWORD c = 0;
275
276 do{ if((_commandline.Get()[c] == __C(' ')) || (_commandline.GetSize() == c))
277 {
278 XSTRING* param = new XSTRING();
279 if(param)
280 {
281 _commandline.Copy(start, c, (*param));
282
283 GetExecParams()->Add(param);
284 start = c+1;
285 }
286 }
287
288 c++;
289
290 } while(c <= _commandline.GetSize());
291
292 return true;
293}
