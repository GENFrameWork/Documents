402bool DIOI2C3DACCELEROMETERLSM303DLHC::Configure()
403{
404 if(!IsInitialized()) return false;
405
406 XBYTE value;
407
408 value = 0x00;
409 value |= 0x01 << 5; // Normal mode
410 value |= 0x07; // X/Y/Z axis enable.
411 if(!WriteRegister(DIOI2CLSM303DLHC_CTRL_REG1_A , value)) return false;
412 value = 0x00;
413 ReadRegister(DIOI2CLSM303DLHC_CTRL_REG1_A, value);
414
415 value = 0x00;
416 value |= 0x01 << 6; // 1: data MSB @ lower address
417 value |= 0x01 << 4; // +/- 4g
418 if(!WriteRegister(DIOI2CLSM303DLHC_CTRL_REG4_A, value)) return false;
419
420 return true;

725bool DIOI2C6AXISTRACKINGLSM303DLHC::Read(DIOI2CLSM303DLHCVECTOR& a, DIOI2CLSM303DLHCVECTOR& c)
726{
727 if(!accelerometer) return false;
728 if(!compass) return false;
729
730 XWORD ax = 0;
731 XWORD ay = 0;
732 XWORD az = 0;
733 XWORD cx = 0;
734 XWORD cy = 0;
735 XWORD cz = 0;
736
737 if(!Read(ax, ay, az, cx, cy, cz)) return false;
738
739 // Perform simple lowpass filtering Intended to stabilize heading despite device vibration such as on a UGV
740 filter_ax += ax - (filter_ax >> DIOI2CLSM303DLHC_FILTERSHIFT);
741 filter_ay += ay - (filter_ay >> DIOI2CLSM303DLHC_FILTERSHIFT);
742 filter_az += az - (filter_az >> DIOI2CLSM303DLHC_FILTERSHIFT);
743
744 a.x = (float)(filter_ax >> DIOI2CLSM303DLHC_FILTERSHIFT);
745 a.y = (float)(filter_ay >> DIOI2CLSM303DLHC_FILTERSHIFT);
746 a.z = (float)(filter_az >> DIOI2CLSM303DLHC_FILTERSHIFT);
747
748 // offset and scale
749 c.x = (cx + offset_x) * scale_x;
750 c.y = (cy + offset_y) * scale_y;
751 c.z = (cz + offset_z) * scale_z;
752
753 return true;
754}

789float DIOI2C6AXISTRACKINGLSM303DLHC::Heading(DIOI2CLSM303DLHCVECTOR& from)
790{
791 DIOI2CLSM303DLHCVECTOR a;
792 DIOI2CLSM303DLHCVECTOR c;
793
794 if(!Read(a, c)) return 0.0f;
795
796 DIOI2CLSM303DLHCVECTOR temp_a;
797
798 temp_a.x = a.x;
799 temp_a.y = a.y;
800 temp_a.z = a.z;
801
802 // normalize
803 DIOI2CLSM303DLHCVECTOR::Normalize(temp_a);
804
805 // compute E and N
806 DIOI2CLSM303DLHCVECTOR e;
807 DIOI2CLSM303DLHCVECTOR n;
808
809 DIOI2CLSM303DLHCVECTOR::Cross(c ,temp_a, e);
810
811 DIOI2CLSM303DLHCVECTOR::Normalize(e);
812
813 DIOI2CLSM303DLHCVECTOR::Cross(temp_a, e , n);
814
815 // compute heading
816 float heading = (float)atan2( DIOI2CLSM303DLHCVECTOR::Dot(e , from), DIOI2CLSM303DLHCVECTOR::Dot(n , from)) * (180/PI);
817 if (heading < 0) heading += 360;
818
819 return float(heading);
820}
