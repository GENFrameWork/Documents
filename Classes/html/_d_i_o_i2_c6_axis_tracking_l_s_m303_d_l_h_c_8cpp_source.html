395bool DIOI2C3DACCELEROMETERLSM303DLHC::Configure()
396{
397 if(!IsInitialized()) return false;
398
399 XBYTE value;
400
401 value = 0x00;
402 value |= 0x01 << 5; // Normal mode
403 value |= 0x07; // X/Y/Z axis enable.
404 if(!WriteRegister(DIOI2CLSM303DLHC_CTRL_REG1_A , value)) return false;
405 value = 0x00;
406 ReadRegister(DIOI2CLSM303DLHC_CTRL_REG1_A, value);
407
408 value = 0x00;
409 value |= 0x01 << 6; // 1: data MSB @ lower address
410 value |= 0x01 << 4; // +/- 4g
411 if(!WriteRegister(DIOI2CLSM303DLHC_CTRL_REG4_A, value)) return false;
412
413 return true;

718bool DIOI2C6AXISTRACKINGLSM303DLHC::Read(DIOI2CLSM303DLHCVECTOR& a, DIOI2CLSM303DLHCVECTOR& c)
719{
720 if(!accelerometer) return false;
721 if(!compass) return false;
722
723 XWORD ax = 0;
724 XWORD ay = 0;
725 XWORD az = 0;
726 XWORD cx = 0;
727 XWORD cy = 0;
728 XWORD cz = 0;
729
730 if(!Read(ax, ay, az, cx, cy, cz)) return false;
731
732 // Perform simple lowpass filtering Intended to stabilize heading despite device vibration such as on a UGV
733 filter_ax += ax - (filter_ax >> DIOI2CLSM303DLHC_FILTERSHIFT);
734 filter_ay += ay - (filter_ay >> DIOI2CLSM303DLHC_FILTERSHIFT);
735 filter_az += az - (filter_az >> DIOI2CLSM303DLHC_FILTERSHIFT);
736
737 a.x = (float)(filter_ax >> DIOI2CLSM303DLHC_FILTERSHIFT);
738 a.y = (float)(filter_ay >> DIOI2CLSM303DLHC_FILTERSHIFT);
739 a.z = (float)(filter_az >> DIOI2CLSM303DLHC_FILTERSHIFT);
740
741 // offset and scale
742 c.x = (cx + offset_x) * scale_x;
743 c.y = (cy + offset_y) * scale_y;
744 c.z = (cz + offset_z) * scale_z;
745
746 return true;
747}

782float DIOI2C6AXISTRACKINGLSM303DLHC::Heading(DIOI2CLSM303DLHCVECTOR& from)
783{
784 DIOI2CLSM303DLHCVECTOR a;
785 DIOI2CLSM303DLHCVECTOR c;
786
787 if(!Read(a, c)) return 0.0f;
788
789 DIOI2CLSM303DLHCVECTOR temp_a;
790
791 temp_a.x = a.x;
792 temp_a.y = a.y;
793 temp_a.z = a.z;
794
795 // normalize
796 DIOI2CLSM303DLHCVECTOR::Normalize(temp_a);
797
798 // compute E and N
799 DIOI2CLSM303DLHCVECTOR e;
800 DIOI2CLSM303DLHCVECTOR n;
801
802 DIOI2CLSM303DLHCVECTOR::Cross(c ,temp_a, e);
803
804 DIOI2CLSM303DLHCVECTOR::Normalize(e);
805
806 DIOI2CLSM303DLHCVECTOR::Cross(temp_a, e , n);
807
808 // compute heading
809 float heading = (float)atan2( DIOI2CLSM303DLHCVECTOR::Dot(e , from), DIOI2CLSM303DLHCVECTOR::Dot(n , from)) * (180/PI);
810 if (heading < 0) heading += 360;
811
812 return float(heading);
813}
