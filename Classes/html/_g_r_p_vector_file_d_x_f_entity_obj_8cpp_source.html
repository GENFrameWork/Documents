120{

123 XCHAR* entityname;

126 } ENTITY2CLASS;

128 ENTITY2CLASS entity2class[] = { { __L("3DFACE") , GRPVECTORFILEDXFENTITYOBJTYPE_3DFACE },
129 { __L("ARC") , GRPVECTORFILEDXFENTITYOBJTYPE_ARC },
130 { __L("CIRCLE") , GRPVECTORFILEDXFENTITYOBJTYPE_CIRCLE },

139 };
140
141 GRPVECTORFILEDXFENTITYOBJ* entityobj = NULL;
142
143 if(!entity)
144 {
145 return entityobj;
146 }
147
148 for(int c=0; c<sizeof(entity2class)/sizeof(ENTITY2CLASS); c++)
149 {
150 if(!entity->GetName()->Compare(entity2class[c].entityname, true))
151 {
152 switch(entity2class[c].objtype)
153 {
154 case GRPVECTORFILEDXFENTITYOBJTYPE_UNKNOWN :
155 default : break;
156 case GRPVECTORFILEDXFENTITYOBJTYPE_3DFACE : entityobj = new GRPVECTORFILEDXFENTITYOBJ3DFACE(entity2class[c].entityname); break;
157 case GRPVECTORFILEDXFENTITYOBJTYPE_ARC : entityobj = new GRPVECTORFILEDXFENTITYOBJARC(entity2class[c].entityname); break;
158 case GRPVECTORFILEDXFENTITYOBJTYPE_CIRCLE : entityobj = new GRPVECTORFILEDXFENTITYOBJCIRCLE(entity2class[c].entityname); break;
159 case GRPVECTORFILEDXFENTITYOBJTYPE_ELLIPSE : entityobj = new GRPVECTORFILEDXFENTITYOBJELLIPSE(entity2class[c].entityname); break;
160 case GRPVECTORFILEDXFENTITYOBJTYPE_LINE : entityobj = new GRPVECTORFILEDXFENTITYOBJLINE(entity2class[c].entityname); break;
161 case GRPVECTORFILEDXFENTITYOBJTYPE_LWPOLYLINE : entityobj = new GRPVECTORFILEDXFENTITYOBJLWPOLYLINE(entity2class[c].entityname); break;
162 case GRPVECTORFILEDXFENTITYOBJTYPE_MTEXT : entityobj = new GRPVECTORFILEDXFENTITYOBJMTEXT(entity2class[c].entityname); break;
163 case GRPVECTORFILEDXFENTITYOBJTYPE_POINT : entityobj = new GRPVECTORFILEDXFENTITYOBJPOINT(entity2class[c].entityname); break;

340bool GRPVECTORFILEDXFENTITYOBJ::ApplyData(GRPVECTORFILEDXFENTITY* entity)
341{
342 GRPVECTORFILEDXFVALUE* value;
343
344 value = GetDataValue(__L("G_LAYER_NAME"), entity);
345 if(value)
346 {
347 layername.Set((XCHAR*)(*value->GetData()));
348 }
349
350 value = GetDataValue(__L("G_COLOR_NUMBER"), entity);
351 if(value)
352 {
353 linecolor = (int)(*value->GetData());
354 }
355
356 value = GetDataValue(__L("G_LINETYPE_NAME"), entity);
357 if(value)
358 {
359 linetypename.Set((XCHAR*)(*value->GetData()));
360 }
361
362 value = GetDataValue(__L("G_TYPE_SPACE"), entity);
363 if(value)
364 {
365 typespace = (int)(*value->GetData());
366 }
367
368 value = GetDataValue(__L("G_ENTITY_VISIBILITY"), entity);
369 if(value)
370 {
371 visibility = ((int)(*value->GetData()))?false:true;
372 }
373
