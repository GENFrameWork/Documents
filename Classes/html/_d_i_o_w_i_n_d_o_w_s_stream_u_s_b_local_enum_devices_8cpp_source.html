147 DWORD size;
148 DWORD memberindex = 0;

150 // We will try to get device information set for all USB devices that have a device interface and are currently present on the system (plugged in).
151 //hdevinfo = SetupDiGetClassDevs(NULL, NULL, NULL, DIGCF_DEVICEINTERFACE | DIGCF_PRESENT | DIGCF_ALLCLASSES);

153 hdevinfo = SetupDiGetClassDevs(&GUID_CLASS_USB_DEVICE, NULL, NULL, (DIGCF_PRESENT | DIGCF_DEVICEINTERFACE));

155 {
156 memset(&devinterfacedata, 0, sizeof(SP_DEVICE_INTERFACE_DATA));
157
158 devinterfacedata.cbSize = sizeof(SP_DEVICE_INTERFACE_DATA);
159 memberindex = 0;
160
161 // Next, we will keep calling this SetupDiEnumDeviceInterfaces(..) until this function causes GetLastError() to return ERROR_NO_MORE_ITEMS. With each call the memberindex value needs to be incremented to retrieve the next device interface information.
162 SetupDiEnumDeviceInterfaces(hdevinfo, NULL, interfaceclassguid, memberindex, &devinterfacedata);
163
164 while(GetLastError() != ERROR_NO_MORE_ITEMS)
165 {
166 // As a last step we will need to get some more details for each of device interface information we are able to retrieve. This device interface detail gives us the information we need to identify the device (VID/PID), and decide if it's useful to us. It will also provide a DEVINFO_DATA structure which we can use to know the serial port name for a virtual com port.
167 devinfodata.cbSize = sizeof(devinfodata);
168
169 // Get the required buffer size. Call SetupDiGetDeviceInterfaceDetail with a NULL devinterfacedetaildata pointer, a devinterfacedetaildataSize of zero, and a valid RequiredSize variable. In response to such a call, this function returns the required buffer size at size.
170 SetupDiGetDeviceInterfaceDetail(hdevinfo, &devinterfacedata, NULL, 0, &size, NULL);
171
172 // Allocate memory for the DeviceInterfaceDetail struct. Don't forget to deallocate it later!
173 devinterfacedetaildata = (PSP_DEVICE_INTERFACE_DETAIL_DATA) HeapAlloc(GetProcessHeap(), HEAP_ZERO_MEMORY, size);
174 if(devinterfacedetaildata)
175 {
176 devinterfacedetaildata->cbSize = sizeof(SP_DEVICE_INTERFACE_DETAIL_DATA);
177
178 if(SetupDiGetDeviceInterfaceDetail(hdevinfo, &devinterfacedata, devinterfacedetaildata, size, &size, &devinfodata))
179 {
180 DIOSTREAMDEVICEUSB* device = new DIOSTREAMDEVICEUSB();
181 if(device)
182 {
183 device->SetIndex(devices.GetSize());
184 device->SetType(DIOSTREAMDEVICE_TYPE_USB);

198
199 trash[0].AdjustSize();

360bool DIOWINDOWSSTREAMUSBLOCALENUMDEVICES::GetUSBDescriptor(HDEVINFO& hdevinfo, int memberindex, DIOSTREAMDEVICEUSB* device)
361{
362 SP_DEVINFO_DATA devicedata;
363 DEVINST parentdevinst = 0;
364 wchar_t deviceID[MAX_PATH];
365 XSTRING devIDstring;
366 GUID classGUID = GUID_CLASS_USBHUB;
367 XSTRING guidString;
368 bool status = false;
369
370 devicedata.cbSize = sizeof(SP_DEVINFO_DATA);
371
372 SetupDiEnumDeviceInfo(hdevinfo, memberindex, &devicedata);
373
374 CM_Get_Parent(&parentdevinst, devicedata.DevInst, 0);
375 CM_Get_Device_ID(parentdevinst, deviceID, MAX_PATH, 0);
376
377 devIDstring = deviceID;
378 devIDstring.Character_Change(__C('\\'), __C('#'));
379 devIDstring.Insert(__L("\\\\?\\"), 0);
380 devIDstring.Add(__L("#"));
381
382 guidString.Format( __L("{%08X-%04X-%04X-%02X%02X-%02X%02X%02X%02X%02X%02X}") , classGUID.Data1 , classGUID.Data2, classGUID.Data3
383 , classGUID.Data4[0]
384 , classGUID.Data4[1]
385 , classGUID.Data4[2]
386 , classGUID.Data4[3]
387 , classGUID.Data4[4]
388 , classGUID.Data4[5]
389 , classGUID.Data4[6]
390 , classGUID.Data4[7]);
391
392 devIDstring.Add(guidString);
393
394 HANDLE handle = CreateFile(devIDstring.Get(), GENERIC_WRITE, FILE_SHARE_WRITE, NULL, OPEN_EXISTING, 0, NULL);
395 if(handle)
396 {
397 DWORD usbportnumber = 0;
398 DWORD requiredsize = 0;
399
400 if(SetupDiGetDeviceRegistryProperty(hdevinfo, &devicedata, SPDRP_ADDRESS, NULL, (PBYTE)&usbportnumber, sizeof(usbportnumber), &requiredsize))
401 {
402 USB_DESCRIPTOR_REQUEST* requestpacket = NULL;
403 USB_DEVICE_DESCRIPTOR* devicedescriptor = NULL;
404 int buffersize = sizeof(USB_DESCRIPTOR_REQUEST) + sizeof(USB_DEVICE_DESCRIPTOR);
405 XBYTE* buffer = new XBYTE[buffersize];
406
407 if(buffer)
408 {
409 // We know from out previous discussion that the first part of the buffer contains the request packet, and latter part contains the data to be filled by the IOCTL - in our case the device descriptor
410 requestpacket = (USB_DESCRIPTOR_REQUEST*)buffer;
411 devicedescriptor = (USB_DEVICE_DESCRIPTOR*)((BYTE*)buffer + sizeof(USB_DESCRIPTOR_REQUEST));
412
413 //fill information in packet
414 requestpacket->SetupPacket.bmRequest = 0x80;
415 requestpacket->SetupPacket.bRequest = USB_REQUEST_GET_DESCRIPTOR;
416 requestpacket->ConnectionIndex = usbportnumber;
417 requestpacket->SetupPacket.wValue = (USB_DEVICE_DESCRIPTOR_TYPE << 8 | 0 ); // Since only 1 device descriptor => index : 0
418 requestpacket->SetupPacket.wLength = sizeof(USB_DEVICE_DESCRIPTOR);
419
420 // Issue ioctl
421 DWORD bytesreturned = 0;
422 status = DeviceIoControl(handle, IOCTL_USB_GET_DESCRIPTOR_FROM_NODE_CONNECTION, buffer, buffersize, buffer, buffersize, &bytesreturned, NULL);
423 if(status)
424 {
425 device->GetDescriptor()->SetBCDUSB(devicedescriptor->bcdUSB);
426 device->GetDescriptor()->SetDeviceClass(devicedescriptor->bDeviceClass);
427 device->GetDescriptor()->SetDeviceSubClass(devicedescriptor->bDeviceSubClass);
428 device->GetDescriptor()->SetDeviceProtocol(devicedescriptor->bDeviceProtocol);
429 device->GetDescriptor()->SetMaxPacketSize(devicedescriptor->bMaxPacketSize0);
430 device->GetDescriptor()->SetIDVendor(devicedescriptor->idVendor);
431 device->GetDescriptor()->SetIDProduct(devicedescriptor->idProduct);
432 device->GetDescriptor()->SetBCDDevice(devicedescriptor->bcdDevice);
433 device->GetDescriptor()->SetManufacturer(devicedescriptor->iManufacturer);
434 device->GetDescriptor()->SetProduct(devicedescriptor->iProduct);
435 device->GetDescriptor()->SetSerialNumber(devicedescriptor->iSerialNumber);
436 device->GetDescriptor()->SetNumConfigurations(devicedescriptor->bNumConfigurations);
437 }

459bool DIOWINDOWSSTREAMUSBLOCALENUMDEVICES::SearchAndDeleteEqualDevices()
460{
461 if(devices.IsEmpty()) return false;
462
463 bool found;
464
465 do{ found = false;
466
467 for(int c=0; c<(int)devices.GetSize(); c++)
468 {
469 DIOSTREAMDEVICEUSB* device1 = (DIOSTREAMDEVICEUSB*)devices.Get(c);
470 if(device1)
471 {
472 for(int d=0; d<(int)devices.GetSize(); d++)
473 {
474 DIOSTREAMDEVICEUSB* device2 = (DIOSTREAMDEVICEUSB*)devices.Get(d);
475 if(device2 && (c!=d))
476 {
477 if((!device1->GetLocation()->Compare(device2->GetLocation()->Get()) && (device1->GetClass() == device2->GetClass())))
478 {
479 devices.Delete(device1);
480 delete device1;
481
482 found = true;
483 break;
484 }
485 }
486 }
487
488 if(found) break;
489 }
490 }
491
492 if(!found) break;
493
494 } while(1);
495
496 return true;
497}
