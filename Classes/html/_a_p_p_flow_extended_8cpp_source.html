568void APPFLOWEXTENDED::ThreadFunction_Update(void* param)
569{
570 APPFLOWEXTENDED* appextended = (APPFLOWEXTENDED*)param;
571 if(!appextended)
572 {
573 return;
574 }
575
576 if(appextended->exitincurse)
577 {
578 return;
579 }
580
581 if(appextended->updatemutex)
582 {
583 appextended->updatemutex->Lock();
584 }
585
586 #ifdef APPFLOW_EXTENDED_APPLICATIONSTATUS_ACTIVE
587 if(appextended->GetApplicationStatus())
588 {
589 appextended->GetApplicationStatus()->Update();
590 }
591 #endif
592
593 #ifdef APPFLOW_EXTENDED_INTERNETSTATUS_ACTIVE
594 if(appextended->GetInternetStatus())
595 {
596 appextended->GetInternetStatus()->Update();
597 }
598 #endif
599
600 for(int c=0; c<100; c++)
601 {
602 if(appextended->exitincurse)
603 {
604 return;
605 }
606
607 GEN_XSLEEP.MilliSeconds(10);
608 }
609
610 if(appextended->updatemutex)
611 {
612 appextended->updatemutex->UnLock();
613 }
614}
