419bool SCRIPT_CACHE::Cache_AllList(XVECTOR<XSTRING*>* listscripts)
420{
421 if(!listscripts)
422 {
423 return false;
424 }
425
426 bool status = false;
427
428 for(XDWORD c=0; c<listscripts->GetSize(); c++)
429 {
430 XSTRING* linescripts = listscripts->Get(c);
431 if(linescripts)
432 {
433 if(!linescripts->IsEmpty())
434 {
435 XVECTOR<XSTRING*> namescripts;
436
437 linescripts->Split(__C(','), namescripts);
438
439 XSTRING* namescript = namescripts.Get(0);
440 if(namescript)
441 {
442 SCRIPT::EliminateExtraChars(namescript);
443
444 SCRIPT* script = SCRIPT::Create(namescript->Get());
445 if(script)
446 {
447 XPATH allpath;
448 XDWORD ID = 0;
449 bool incache = false;
450
451 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, allpath);
452
453 for(XDWORD d=0; d<namescripts.GetSize(); d++)
454 {
455 allpath += __C(',');
456 allpath += namescript->Get();
457 }
458
459 ID = GEN_SCRIPT_CACHE.GenerateID(allpath);
460
461 #ifdef SCRIPT_CACHE_ACTIVE
462
463 XSTRING* _script = GEN_SCRIPT_CACHE.Cache_Get(ID);
464 if(_script)
465 {
466 (*script->GetScript()) += _script->Get();
467
468 incache = true;
469 status = !_script->IsEmpty();
470 }
471
472 #endif
473
474 if(!incache)
475 {
476 for(XDWORD d=0; d<namescripts.GetSize(); d++)
477 {
478 namescript = namescripts.Get(d);
479 if(namescript)
480 {
481 XPATH xpath;
482
483 SCRIPT::EliminateExtraChars(namescript);
484
485 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, xpath);
486 xpath.Slash_Add();
487 xpath += namescript->Get();
488
489 status = script->LoadAdd(xpath);
490 if(!status)
491 {
492 break;
493 }
494 }
495 }
496 }
497
498 #ifdef SCRIPT_CACHE_ACTIVE
499 if(status && !incache)
500 {
501 GEN_SCRIPT_CACHE.Cache_Add(ID, script->GetScript());
502 }
503 #endif
504
505 delete script;
506 script = NULL;
507
508 }
509 }
510
511 namescripts.DeleteContents();
512 namescripts.DeleteAll();
513
514 }
515 }
516 }
517
518 return status;
