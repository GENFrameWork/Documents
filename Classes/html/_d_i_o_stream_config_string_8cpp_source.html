140bool DIOSTREAMCONFIGSTRING::SetFromString(XCHAR* string)
141{
142 if(!string) return false;
143
144 DIOURL _string;
145 DIOURL urlall;
146 XSTRING url;
147 XSTRING urladd;
148 XSTRING login;
149 XSTRING password;
150 int endurlaccess;
151 int port = 0;
152
153 remoteURL.Empty();
154
155 _string = string;
156
157 if(_string.HaveHTTPID()) _string.DeleteHTTPID();
158
159 endurlaccess = _string.Find(__L("/"), true);
160 if(endurlaccess == XSTRING_NOTFOUND) endurlaccess = _string.Find(__L("\\"), true);
161 if(endurlaccess == XSTRING_NOTFOUND) endurlaccess = _string.GetSize();
162
163 _string.Copy(endurlaccess+1, urladd);
164 _string.Copy(0, endurlaccess, urlall);
165 urlall.Slash_Delete();
166
167 int havepassword = urlall.Find(__L("@"), true);
168 int haveport = urlall.Find(__L(":"), true, (havepassword == XSTRING_NOTFOUND)?0:havepassword);
169
170 login.AdjustSize(_MAXSTR*10);
171 password.AdjustSize(_MAXSTR*10);
172 url.AdjustSize(_MAXSTR*10);
173
174 if(haveport== XSTRING_NOTFOUND)
175 {
176 if(havepassword == XSTRING_NOTFOUND)
177 urlall.UnFormat(__L("%s"), url.Get());
178 else urlall.UnFormat(__L("%s:%s@%s"), login.Get(), password.Get(), url.Get());
179 }
180 else
181 {
182 if(havepassword == XSTRING_NOTFOUND)
183 urlall.UnFormat(__L("%s:%d"), url.Get(), &port);
184 else urlall.UnFormat(__L("%s:%s@%s:%d"), login.Get(), password.Get(), url.Get(), &port);
185 }
186
187 login.AdjustSize();
188 password.AdjustSize();
189 url.AdjustSize();
190
191
192 if(!login.IsEmpty() || !password.IsEmpty()) remoteURL.AddFormat(__L("%s:%s@"), login.Get(), password.Get());
193
194 remoteURL.Add(url);
195 if(urladd.Get()[0])
196 {
197 remoteURL.Slash_Add();
198 remoteURL.Add(urladd);
199 }
200
201
202 if(port) this->remoteport = port;
203
204 return true;
205}
