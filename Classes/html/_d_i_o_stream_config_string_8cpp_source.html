133bool DIOSTREAMCONFIGSTRING::SetFromString(XCHAR* string)

148 _string = string;
149
150 if(_string.HaveHTTPID()) _string.DeleteHTTPID();

152 endurlaccess = _string.Find(__L("/"), true);

154 if(endurlaccess == XSTRING_NOTFOUND) endurlaccess = _string.GetSize();

157 _string.Copy(0, endurlaccess, urlall);
158 urlall.Slash_Delete();

161 int haveport = urlall.Find(__L(":"), true, (havepassword == XSTRING_NOTFOUND)?0:havepassword);
162
163 login.AdjustSize(_MAXSTR*10);
164 password.AdjustSize(_MAXSTR*10);
165 url.AdjustSize(_MAXSTR*10);
166
167 if(haveport== XSTRING_NOTFOUND)
168 {
169 if(havepassword == XSTRING_NOTFOUND)
170 urlall.UnFormat(__L("%s"), url.Get());
171 else urlall.UnFormat(__L("%s:%s@%s"), login.Get(), password.Get(), url.Get());
172 }
173 else
174 {
175 if(havepassword == XSTRING_NOTFOUND)
176 urlall.UnFormat(__L("%s:%d"), url.Get(), &port);
177 else urlall.UnFormat(__L("%s:%s@%s:%d"), login.Get(), password.Get(), url.Get(), &port);

190 remoteURL.Slash_Add();
191 remoteURL.Add(urladd);
