241bool XTRANSLATION_GEN::Sentence_Add(XDWORD ID, XDWORD codelanguage, XCHAR* sentence, XDWORD fixed)
242{
243 if(!ID) return false;
244 if(!codelanguage) return false;
245 if(!sentence) return false;
246
247 if(Sentence_Get(ID, codelanguage)) return false;
248
249 XTRANSLATION_GEN_SENTENCE* sentenceGEN = new XTRANSLATION_GEN_SENTENCE();
250 if(!sentenceGEN) return false;
251
252 int sizesentence = XSTRING::GetSize(sentence);
253
254 sentenceGEN->ID = ID;
255 sentenceGEN->codelanguage = codelanguage;
256 sentenceGEN->sentence = new XCHAR[sizesentence+1];
257 sentenceGEN->fixed = fixed;
258 if(!sentenceGEN->sentence)
259 {
260 delete sentenceGEN;
261 return false;
262 }
263
264 memset(sentenceGEN->sentence, 0, (sizesentence+1) * sizeof(XCHAR));
265 memcpy(sentenceGEN->sentence, sentence, sizesentence * sizeof(XCHAR));
266
267 sentences.Add(sentenceGEN);
268
269 return true;
270}

460bool XTRANSLATION_GEN::Sentences_AddToTranslation(XDWORD codelanguage)
461{
462 bool existsentenceGENlanguage = false;
463
464 for(XDWORD c=0; c<sentences.GetSize(); c++)
465 {
466 XTRANSLATION_GEN_SENTENCE* sentenceGEN = sentences.Get(c);
467 if(sentenceGEN)
468 {
469 if(sentenceGEN->codelanguage == codelanguage)
470 {
471 GEN_XTRANSLATION.Translate_Add(sentenceGEN->ID, sentenceGEN->sentence, sentenceGEN->fixed);
472 existsentenceGENlanguage = true;
473 }
474 }
475 }
476
477 return existsentenceGENlanguage;
478}
