248bool XTRANSLATION_GEN::Sentence_Add(XDWORD ID, XDWORD codelanguage, XCHAR* sentence, XDWORD fixed)
249{
250 if(!ID) return false;
251 if(!codelanguage) return false;
252 if(!sentence) return false;
253
254 if(Sentence_Get(ID, codelanguage)) return false;
255
256 XTRANSLATION_GEN_SENTENCE* sentenceGEN = new XTRANSLATION_GEN_SENTENCE();
257 if(!sentenceGEN) return false;
258
259 int sizesentence = XSTRING::GetSize(sentence);
260
261 sentenceGEN->ID = ID;
262 sentenceGEN->codelanguage = codelanguage;
263 sentenceGEN->sentence = new XCHAR[sizesentence+1];
264 sentenceGEN->fixed = fixed;
265 if(!sentenceGEN->sentence)
266 {
267 delete sentenceGEN;
268 return false;
269 }
270
271 memset(sentenceGEN->sentence, 0, (sizesentence+1) * sizeof(XCHAR));
272 memcpy(sentenceGEN->sentence, sentence, sizesentence * sizeof(XCHAR));
273
274 sentences.Add(sentenceGEN);
275
276 return true;
277}

292XTRANSLATION_GEN_SENTENCE* XTRANSLATION_GEN::Sentence_Get(XDWORD ID, XDWORD codelanguage)
293{
294 for(XDWORD c=0; c<sentences.GetSize(); c++)
295 {
296 XTRANSLATION_GEN_SENTENCE* sentence = sentences.Get(c);
297 if(sentence)
298 {
299 if((sentence->ID == ID) && (sentence->codelanguage == codelanguage)) return sentence;
300 }
301 }
302
303 return NULL;
304}

508bool XTRANSLATION_GEN::Sentences_AddToTranslation(XDWORD codelanguage)
509{
510 bool existsentenceGENlanguage = false;
511
512 for(XDWORD c=0; c<sentences.GetSize(); c++)
513 {
514 XTRANSLATION_GEN_SENTENCE* sentenceGEN = sentences.Get(c);
515 if(sentenceGEN)
516 {
517 if(sentenceGEN->codelanguage == codelanguage)
518 {
519 GEN_XTRANSLATION.Translate_Add(sentenceGEN->ID, sentenceGEN->sentence, sentenceGEN->fixed);
520 existsentenceGENlanguage = true;
521 }
522 }
523 }
524
525 return existsentenceGENlanguage;
526}
