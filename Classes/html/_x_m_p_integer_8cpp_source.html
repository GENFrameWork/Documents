1224 Y.RightShift(XMPINTEGER_BITSINLIMB * (n - t));
1225
1226 for( i = n; i > t ; i-- )
1227 {

1229 {
1230 Z.GetLimbs()[i - t - 1] = ~0;
1231 }

1234 XUDBL r;
1235

1237 r |= (XUDBL) X.GetLimbs()[i - 1];
1238 r /= Y.GetLimbs()[t];
1239

1241
1242 Z.GetLimbs()[i - t - 1] = (XLIMB) r;
1243 }
1244
1245 Z.GetLimbs()[i - t - 1]++;
1246
1247 do{ Z.GetLimbs()[i - t - 1]--;
1248
1249 if(!T1.LeftSet(0)) return false;
1250
1251 T1.GetLimbs()[0] = (t < 1) ? 0 : Y.GetLimbs()[t - 1];
1252 T1.GetLimbs()[1] = Y.GetLimbs()[t];
1253
1254 if(!T1.Multiplication(&T1, Z.GetLimbs()[i - t - 1])) return false;
1255
1256 if(!T2.LeftSet(0)) return false;
1257
1258 T2.GetLimbs()[0] = (i < 2) ? 0 : X.GetLimbs()[i - 2];
1259 T2.GetLimbs()[1] = (i < 1) ? 0 : X.GetLimbs()[i - 1];
1260 T2.GetLimbs()[2] = X.GetLimbs()[i];
1261
1262 } while(T1.CompareSignedValues(T2) > 0 );
1263
1264 if(!T1.Multiplication(&Y, Z.GetLimbs()[i - t - 1])) return false;
1265
1266 if(!T1.LeftShift(XMPINTEGER_BITSINLIMB * (i - t - 1))) return false;
1267
1268 if(!X.SubtractionSigned(&X, &T1 )) return false;
1269
1270 if(X.CompareSignedValues(0) < 0 )
1271 {
1272 if(!T1.CopyFrom(&Y)) return false;
1273 if(!T1.LeftShift(XMPINTEGER_BITSINLIMB * (i - t - 1))) return false;
1274
1275 if(!X.AdditionSigned(&X, &T1)) return false;
1276
1277 Z.GetLimbs()[i - t - 1]--;
1278 }
1279 }
1280
1281 if(xmpintegerQ != NULL )
1282 {
1283 if(!xmpintegerQ->CopyFrom(&Z)) return false;
1284 xmpintegerQ->SetSign(xmpintegerA->GetSign() * xmpintegerB->GetSign());
1285 }
1286
1287 if(xmpintegerR != NULL )
1288 {
1289 if(!X.RightShift(k)) return false;
1290 X.SetSign(xmpintegerA->GetSign());
1291
1292 if(!xmpintegerR->CopyFrom(&X)) return false;
1293
1294 if(xmpintegerR->CompareSignedValues(0)==0) xmpintegerR->SetSign(1);
1295 }
1296
1297 X.End();
1298 Y.End();
1299 Z.End();
1300 T1.End();
1301 T2.End();
1302
1303 return true;
1304}

1381bool XMPINTEGER::Module(XLIMB* r, XMPINTEGER* xmpintegerA, int integerb)
1382{

1384 XDWORD i;
1385

1388
1389 // handle trivial cases
1390
1391 if(integerb == 1 )
1392 {
1393 (*r) = 0;

1395 }
1396
1397 if(integerb == 2 )
1398 {
1399 (*r) = xmpintegerA->GetLimbs()[0] & 1;
1400 return true;

1413 z = y / integerb;
1414 y -= z * integerb;
1415 }
1416

1418
1419 if( xmpintegerA->GetSign() < 0 && y != 0 ) y = (integerb - y);
1420
1421 *r = y;
1422
1423 return true;

1541bool XMPINTEGER::SlidingWindowExponentiation(XMPINTEGER* A, XMPINTEGER* E, XMPINTEGER* N, XMPINTEGER* _RR)
1542{
1543 XDWORD wbits, wsize, one = 1;
1544 XDWORD i, j, nblimbs;
1545 XDWORD bufsize, nbits;
1546 XLIMB ei, mm, state;
1547 XMPINTEGER RR;
1548 XMPINTEGER T;
1549 XMPINTEGER W[2 << XMPINTEGER_WINDOWSIZE];
1550 XMPINTEGER Apos;
1551 int neg;
1552
1553 if(N->CompareSignedValues(0) < 0 || ( N->GetLimbs()[0] & 1 ) == 0 ) return false;
1554 if(E->CompareSignedValues(0) < 0) return false;
1555
1556 //Init temps and window size
1557
1558 MontgomeryIni( &mm, N );
1559 RR.Ini();
1560 T.Ini();
1561 Apos.Ini();
1562
1563 memset((XBYTE*)W, 0, sizeof(W));
1564
1565 i = E->GetMSB();
1566
1567 wsize = ( i > 671 ) ? 6 : ( i > 239 ) ? 5 :
1568 ( i > 79 ) ? 4 : ( i > 23 ) ? 3 : 1;
1569
1570 if(wsize > XMPINTEGER_WINDOWSIZE) wsize = XMPINTEGER_WINDOWSIZE;
1571
1572 j = N->GetNLimbs() + 1;
1573
1574 if(!Grow(j)) return false;
1575 if(!W[1].Grow(j)) return false;
1576 if(!T.Grow(j*2)) return false;
1577
1578 // Compensate for negative A (and correct at the end)
1579
1580 neg = ( A->GetSign() == -1 );
1581 if(neg)
1582 {
1583 if(!Apos.CopyFrom(A)) return false;
1584 Apos.SetSign(1);
1585 A = &Apos;
1586 }
1587
1588 // If 1st call, pre-compute R^2 mod N
1589
1590 if( _RR == NULL || _RR->GetLimbs() == NULL )
1591 {
1592 if(!RR.LeftSet(1)) return false;
1593 if(!RR.LeftShift(N->GetNLimbs() * 2 * XMPINTEGER_BITSINLIMB)) return false;
1594 if(!RR.Module(&RR, &RR, N)) return false;
1595
1596 _RR->CopyFrom(&RR);
1597
1598 if(_RR != NULL) _RR->CopyFrom(&RR);
1599
1600 } else RR.CopyFrom(_RR);
1601
1602 // W[1] = A * R^2 * R^-1 mod N = A * R mod N
1603
1604 if(A->CompareSignedValues((*N)) >= 0)
1605 {
1606 if(!Module( &W[1], A, N )) return false;
1607 }
1608 else
1609 {
1610 if(!W[1].CopyFrom(A)) return false;
1611 }
1612
1613 MontgomeryMultiplication(&W[1], &RR, N, mm, &T);
1614
1615 // X = R^2 * R^-1 mod N = R mod N
1616
1617 if(!CopyFrom(&RR)) return false;
1618 MontgomeryReduction(this, N, mm, &T);
1619
1620 if(wsize > 1)
1621 {
1622 // W[1 << (wsize - 1)] = W[1] ^ (wsize - 1)
1623
1624 j = one << (wsize - 1);
1625
1626 if(!W[j].Grow(N->GetNLimbs() + 1)) return false;
1627
1628 if(!W[j].CopyFrom(&W[1])) return false;
1629
1630 for(i = 0; i < wsize - 1; i++)
1631 {
1632 MontgomeryMultiplication( &W[j], &W[j], N, mm, &T);
1633 }
1634
1635 // W[i] = W[i - 1] * W[1]
1636
1637 for(i = j + 1; i < (one << wsize); i++)
1638 {
1639 if(!W[i].Grow(N->GetNLimbs() + 1)) return false;
1640 if(!W[i].CopyFrom(&W[i-1])) return false;
1641
1642 MontgomeryMultiplication( &W[i], &W[1], N, mm, &T);
1643 }
1644 }
1645
1646 nblimbs = E->GetNLimbs();
1647 bufsize = 0;
1648 nbits = 0;
1649 wbits = 0;
1650 state = 0;
1651
1652 while(1)
1653 {
1654 if(bufsize == 0)
1655 {
1656 if(nblimbs == 0) break;
1657 nblimbs--;
1658 bufsize = sizeof(XLIMB ) << 3;
1659 }
1660
1661 bufsize--;
1662
1663 ei = (E->GetLimbs()[nblimbs] >> bufsize) & 1;
1664
1665 // skip leading 0s
1666 if(ei == 0 && state == 0) continue;
1667 if(ei == 0 && state == 1)
1668 {
1669 // out of window, square X
1670 MontgomeryMultiplication(this, this, N, mm, &T );
1671 continue;
1672 }
1673
1674 // add ei to current window
1675 state = 2;
1676
1677 nbits++;
1678 wbits |= (ei << (wsize - nbits));
1679
1680 if(nbits == wsize)
1681 {
1682 // X = X^wsize R^-1 mod N
1683 for( i = 0; i < wsize; i++ )
1684 {
1685 MontgomeryMultiplication(this, this, N, mm, &T);
1686 }
1687
1688 // X = X * W[wbits] R^-1 mod N
1689 MontgomeryMultiplication(this, &W[wbits], N, mm, &T);
1690
1691 state--;
1692 nbits = 0;
1693 wbits = 0;
1694 }
1695 }
1696
1697 // process the remaining bits
1698 for(i=0; i<nbits; i++)
1699 {
1700 MontgomeryMultiplication(this, this, N, mm, &T);
1701 wbits <<= 1;
1702 if((wbits & (one << wsize)) != 0) MontgomeryMultiplication(this, &W[1], N, mm, &T);
1703 }
1704
1705 // X = A^E * R * R^-1 mod N = A^E mod N
1706 MontgomeryReduction(this, N, mm, &T);
1707
1708 if(neg)
1709 {
1710 SetSign(-1);
1711 if(!AdditionSigned(N, this)) return false;
1712 }
1713
1714 for(i = (one << (wsize - 1)); i < (one << wsize); i++)
1715 {
1716 W[i].End();
1717 }
1718
1719 W[1].End();
1720 T.End();
1721 Apos.End();
1722
1723 if(_RR == NULL) RR.End();
1724
1725 return true;
1726}
