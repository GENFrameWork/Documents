627bool XDRIVEIMAGEMANAGER::ReadToIMGFile(XDRIVEIMAGEMANAGER_DRIVE& drive, XPATH& pathfileIMG, XQWORD nsectorsperblock)
628{
629 XFILE* xfileIMG;
630 XBUFFER readbuffer;
631 XQWORD nsectorstoread = nsectorsperblock;
632 XQWORD nsectorsread = 0;
633 bool status = 0;
634
635 if(!drive.GetSize()) return status;
636
637 GEN_XFACTORY_CREATE(xfileIMG, Create_File())
638 if(xfileIMG)
639 {
640 if(xfileIMG->Create(pathfileIMG))
641 {
642 XQWORD indexsector=0;
643
644 while(indexsector<drive.GetNumberSectors())
645 {
646 if((indexsector + nsectorstoread) > drive.GetNumberSectors()) nsectorstoread = (drive.GetNumberSectors() - indexsector);
647
648 XDRIVEIMAGEMANAGER_XEVENT xevent(this, XDRIVEIMAGEMANAGER_XEVENT_TYPE_READTOIMGFILE);
649
650 xevent.SetDrive(&drive);
651 xevent.SetIndexSector(indexsector);
652 xevent.SetNumberSectorsInOperation(nsectorstoread);
653 xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_NONE);
654
655 status = Read(drive, indexsector, nsectorstoread, readbuffer);
656 if(!status) xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_READSECTORS);
657
658 if(status)
659 {
660 status = xfileIMG->Write(readbuffer);
661 if(!status) xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_WRITEIMGFILE);
662 }
663
664 if(status) nsectorsread += nsectorstoread;
665
666 lastoperationpercent = (((float)nsectorsread*100)/(float)drive.GetNumberSectors());
667
668 xevent.SetError(lasterror);
669 xevent.SetOperationPercent(lastoperationpercent);
670
671 PostEvent(&xevent);
672
673 if(!status) break;
674
675 indexsector += nsectorstoread;
676 }
677
678 xfileIMG->Close();
679 }
680
681 GEN_XFACTORY.Delete_File(xfileIMG);
682 }
683
684 return status;
685}

701bool XDRIVEIMAGEMANAGER::WriteFromIMGFile(XDRIVEIMAGEMANAGER_DRIVE& drive, XPATH& pathfileIMG, XQWORD nsectorsperblock)
702{
703 XFILE* xfileIMG;
