259bool DIOESP32STREAMSPI::Open()
260{
261 if(!threadconnection) return false;
262 if(!config) return false;
263
264 for(int c=0; c<DIOESP32STREAMSPI_MAXHANDLES; c++)
265 {
266 if(DIOESP32STREAMSPI::handles[c] == this) return false;
267
268 if(!DIOESP32STREAMSPI::handles[c])
269 {
270 indexhandle = c;
271 DIOESP32STREAMSPI::handles[indexhandle] = this;
272 break;
273 }
274 }
275
276 if(indexhandle == -1) return false;
277
278 indexport = config->GetPort()-1;
279
280 if(!DIOESP32STREAMSPI::ports[indexport])
281 {
282 DIOESP32STREAMSPIPORT* port = new DIOESP32STREAMSPIPORT();
283 if(!port) return false;
284
285 hspi = new SPI_HandleTypeDef();
286 if(!hspi)
287 {
288 delete port;
289 return false;
290 }
291
292 switch(config->GetPort())
293 {
294 case 1 : hspi->Instance = SPI1; break;
295 case 2 : hspi->Instance = SPI2; break;
296 default : return false;
297 }
298
299
300
301 hspi->Init.Mode = SPI_MODE_MASTER;
302 hspi->Init.Direction = SPI_DIRECTION_2LINES;
303 hspi->Init.DataSize = SPI_DATASIZE_8BIT;
304 hspi->Init.CLKPolarity = SPI_POLARITY_LOW;
305 hspi->Init.CLKPhase = SPI_PHASE_1EDGE;
306 hspi->Init.NSS = SPI_NSS_HARD_OUTPUT;
307 hspi->Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_2;
308 hspi->Init.FirstBit = SPI_FIRSTBIT_MSB;
309 hspi->Init.TIMode = SPI_TIMODE_DISABLE;
310 hspi->Init.CRCCalculation = SPI_CRCCALCULATION_DISABLE;
311 hspi->Init.CRCPolynomial = 7;
312 hspi->Init.CRCLength = SPI_CRC_LENGTH_DATASIZE;
313 hspi->Init.NSSPMode = SPI_NSS_PULSE_ENABLE;
314
315 if(HAL_SPI_Init(hspi) != HAL_OK)
316 {
317 return false;
318 }
319
320 port->SetHandleSPI(hspi);
321 DIOESP32STREAMSPI::ports[indexport] = port;
322 }
323 else
324 {
325 hspi = DIOESP32STREAMSPI::ports[indexport]->GetHandleSPI();
326 DIOESP32STREAMSPI::ports[indexport]->SetCounterRef(DIOESP32STREAMSPI::ports[indexport]->GetCounterRef()+1);
327 }
328
329 SetEvent(DIOSTREAMSPI_FSMEVENT_CONNECTED);
330
331 status = DIOSTREAMSTATUS_CONNECTED;
332
333 memset(bufferdata, 0, DIOSTREAM_MAXBUFFER);
334
335 ResetXBuffers();
336 ResetConnectionStatistics();
337
338 return threadconnection->Ini();
339}

499bool DIOESP32STREAMSPI::Close()
500{
501 if(!threadconnection) return false;
502
503 threadconnection->End();
504
505 if(GetStatus()==DIOSTREAMSTATUS_DISCONNECTED) return false;
506
507 if(!DIOESP32STREAMSPI::handles[indexhandle]) return false;
508
509 DIOESP32STREAMSPI::handles[indexhandle] = NULL;
510
511 if(!DIOESP32STREAMSPI::ports[indexport]->GetCounterRef())
512 {
513 HAL_SPI_DeInit(DIOESP32STREAMSPI::ports[indexport]->GetHandleSPI());
514
515 delete DIOESP32STREAMSPI::ports[indexport]->GetHandleSPI();
516 delete DIOESP32STREAMSPI::ports[indexport];
517
518 DIOESP32STREAMSPI::ports[indexport] = NULL;
519 }
520 else
521 {
522 DIOESP32STREAMSPI::ports[indexport]->SetCounterRef(DIOESP32STREAMSPI::ports[indexport]->GetCounterRef()-1);
523 }
524
525 return true;

556{
557 int port = -1;
558
559 if(hspi->Instance == SPI1) port = 0;
560 if(hspi->Instance == SPI2) port = 1;
561
562 for(int c=0; c<DIOESP32STREAMSPI_MAXHANDLES; c++)
563 {
564 if(DIOESP32STREAMSPI::handles[c])
565 {
566 if(DIOESP32STREAMSPI::handles[c]->indexport == port)
567 {
568 if(DIOESP32STREAMSPI::handles[c]->GetStatus() == DIOSTREAMSTATUS_DISCONNECTED) return;
569
570 if(!DIOESP32STREAMSPI::handles[c]->IsBlockRead())
571 {
572 DIOESP32STREAMSPI::handles[c]->inbuffer->Add(DIOESP32STREAMSPI::handles[c]->bufferdata, DIOESP32STREAMSPI::handles[c]->sizebufferdata);
573 }
574
575 DIOESP32STREAMSPI::handles[c]->sizebufferdata = 0;
576 }
577 }

624void DIOESP32STREAMSPI::ThreadConnection(void* param)
625{
626 DIOESP32STREAMSPI* diostream = (DIOESP32STREAMSPI*)param;
627 if(!diostream) return;
628
629 //HAL_NVIC_DisableIRQ(TIM3_IRQn);
630
631 if(diostream->GetEvent()==DIOSTREAMSPI_FSMEVENT_NONE) // No hay nuevos Eventos
632 {
633 switch(diostream->GetCurrentState())
634 {
635 case DIOSTREAMSPI_FSMSTATE_NONE : break;
636
637 case DIOSTREAMSPI_FSMSTATE_CONNECTED : break;
638
639 case DIOSTREAMSPI_FSMSTATE_WAITINGTOREAD : if(!diostream->IsBlockRead())
640 {
641 if(!diostream->inbuffer->IsBlocked())
642 {
643 if(diostream->readcache[diostream->indexport].GetSize())
644 {
645 diostream->inbuffer->Add(diostream->readcache[diostream->indexport]);
646 diostream->readcache[diostream->indexport].Delete();
647 }
648 }
649 }
650
651 if(!diostream->IsBlockWrite())
652 {
653 if(!diostream->outbuffer->IsBlocked())
654 {
655 int esize = diostream->outbuffer->GetSize();
656 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
657
658 if(esize)
659 {
660 diostream->outbuffer->Extract(diostream->writebuffer, 0, esize);
661 diostream->WriteDirect(diostream->writebuffer, esize);
662 }
663 }
664 }
665 break;
666
667 case DIOSTREAMSPI_FSMSTATE_DISCONNECTING : break;
668
669 }
670 }
671 else
672 {
673 if(diostream->GetEvent()<DIOSTREAMSPI_LASTEVENT)
674 {
675 diostream->CheckTransition();
676
677 switch(diostream->GetCurrentState())
678 {
679 case DIOSTREAMSPI_FSMSTATE_NONE : break;
680
681 case DIOSTREAMSPI_FSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
682 xevent.SetDIOStream(diostream);
683 diostream->PostEvent(&xevent);
684
685 diostream->SetEvent(DIOSTREAMSPI_FSMEVENT_WAITINGTOREAD);
686 }
687 break;
688
689 case DIOSTREAMSPI_FSMSTATE_WAITINGTOREAD : break;
690
691 case DIOSTREAMSPI_FSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
692 xevent.SetDIOStream(diostream);
693 diostream->PostEvent(&xevent);
694
695 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
696 }
697 break;
698 }
699 }
700 }

702 //HAL_NVIC_EnableIRQ(TIM3_IRQn);
703}
