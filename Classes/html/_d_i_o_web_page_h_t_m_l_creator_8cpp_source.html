786 XFILETXT filetxt;
787 bool status = true;
788 bool intag = false;
789
790 if(!filetxt.Open(xpath)) return false;
791
792 if(filetxt.ReadAllFile())

796 XBUFFER LFdata;

798 filetxt.GetLF(LFdata);
799
800 for(int c=0;c<filetxt.GetNLines();c++)

805 XSTRING inioutput;

807 int result;
808

814 if(Add(*line)) Add(filetxt.GetLF()->Get());
815 }

874DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS DIOWEBPAGEHTMLCREATOR::CheckXTAG(XSTRING* originalline,bool& intag, XSTRING& inioutputline,XSTRING& endoutputline)
875{
876 int index[3];
877
878 inioutputline.Empty();
879 endoutputline.Empty();
880
881 if(!intag)
882 {
883 index[0] = originalline->Find(__L("<?"), true);
884 if(index[0]!=XSTRING_NOTFOUND)
885 {
886 index[1] = originalline->Find(DIOWEBPAGEHTMLCREATOR_TAGXENGINE, true);
887 if(index[1]!=XSTRING_NOTFOUND)
888 {
889 lastXTAG.Empty();
890
891 originalline->Copy(0,index[0],inioutputline);
892
893 index[2] = originalline->Find(__L("?>"),true);
894 if(index[2]!=XSTRING_NOTFOUND)
895 {
896 XSTRING endline;
897
898 originalline->Copy(index[1]+3,index[2],lastXTAG);
899 originalline->Copy(index[2]+2,endoutputline);
900
901 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDEND;
902
903 }
904 else
905 {
906 originalline->Copy(index[1]+3,lastXTAG);
907 intag = true;
908 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDNOTEND;
909 }
910 }
911 }
912
913 inioutputline = originalline->Get();
914 }
915 else
916 {
917 index[0] = originalline->Find(__L("?>"),true);
918 if(index[0]!=XSTRING_NOTFOUND)
919 {
920 XSTRING TAGstring;
921
922 originalline->Copy(0,index[0],TAGstring);
923 lastXTAG+=TAGstring.Get();
924
925 originalline->Copy(index[0]+2,endoutputline);
926
927 intag = false;
928
929 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDEND;
930 }
931
932 lastXTAG += originalline->Get();
933
934 inioutputline.Empty();
935 endoutputline.Empty();
936
937 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDNOTEND;
938 }
939
940 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_NOTFOUND;
941}
