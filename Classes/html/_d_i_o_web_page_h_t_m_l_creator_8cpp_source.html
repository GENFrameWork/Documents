777bool DIOWEBPAGEHTMLCREATOR::AddFile(XPATH& xpath)
778{
779 XFILETXT filetxt;
780 bool status = true;
781 bool intag = false;
782
783 if(!filetxt.Open(xpath)) return false;
784
785 if(filetxt.ReadAllFile())
786 {
787 Empty();
788
789 XBUFFER LFdata;
790
791 filetxt.GetLF(LFdata);
792
793 for(int c=0;c<filetxt.GetNLines();c++)
794 {
795 XSTRING* line = filetxt.GetLine(c);
796 if(line)
797 {
798 XSTRING inioutput;
799 XSTRING endoutput;
800 int result;
801
802 result = CheckXTAG(line, intag, inioutput, endoutput);
803 switch(result)
804 {
805 case DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_NOTFOUND : { if(!line->IsEmpty())
806 {
807 if(Add(*line)) Add(filetxt.GetLF()->Get());
808 }
809 }
810 break;
811
812 case DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDEND : { if(!inioutput.IsEmpty())
813 {
814 if(Add(inioutput)) Add(filetxt.GetLF()->Get());
815 }
816
817 DIOWEBPAGEHTMLCREATOR_XEVENT xevent(this, DIOWEBPAGEHTMLCREATOR_XEVENT_TYPE_XTAGFOUND);
818
819 xevent.SetPageHTMLCreator(this);
820 xevent.SetXTAG(&lastXTAG);
821
822 PostEvent(&xevent);
823
824 if(!endoutput.IsEmpty())
825 {
826 if(Add(endoutput)) Add(filetxt.GetLF()->Get());
827 }
828 }
829 break;
830
831 case DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDNOTEND: { if(!inioutput.IsEmpty())
832 {
833 if(Add(inioutput)) Add(filetxt.GetLF()->Get());
834 }
835
836 if(!endoutput.IsEmpty())
837 {
838 if(Add(endoutput)) Add(filetxt.GetLF()->Get());
839 }
840 }
841 break;
842 }
843 }
844 }
845 }
846
847 filetxt.Close();
848
849 return status;

867DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS DIOWEBPAGEHTMLCREATOR::CheckXTAG(XSTRING* originalline,bool& intag, XSTRING& inioutputline,XSTRING& endoutputline)
868{
869 int index[3];
870
871 inioutputline.Empty();
872 endoutputline.Empty();
873
874 if(!intag)
875 {
876 index[0] = originalline->Find(__L("<?"), true);
877 if(index[0]!=XSTRING_NOTFOUND)
878 {
879 index[1] = originalline->Find(DIOWEBPAGEHTMLCREATOR_TAGXENGINE, true);
880 if(index[1]!=XSTRING_NOTFOUND)
881 {
882 lastXTAG.Empty();
883
884 originalline->Copy(0,index[0],inioutputline);
885
886 index[2] = originalline->Find(__L("?>"),true);
887 if(index[2]!=XSTRING_NOTFOUND)
888 {
889 XSTRING endline;
890
891 originalline->Copy(index[1]+3,index[2],lastXTAG);
892 originalline->Copy(index[2]+2,endoutputline);
893
894 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDEND;
895
896 }
897 else
898 {
899 originalline->Copy(index[1]+3,lastXTAG);
900 intag = true;
901 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDNOTEND;
902 }
903 }
904 }
905
906 inioutputline = originalline->Get();
907 }
908 else
909 {
910 index[0] = originalline->Find(__L("?>"),true);
911 if(index[0]!=XSTRING_NOTFOUND)
912 {
913 XSTRING TAGstring;
914
915 originalline->Copy(0,index[0],TAGstring);
916 lastXTAG+=TAGstring.Get();
917
918 originalline->Copy(index[0]+2,endoutputline);
919
920 intag = false;
921
922 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDEND;
923 }
924
925 lastXTAG += originalline->Get();
926
927 inioutputline.Empty();
928 endoutputline.Empty();
929
930 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_FOUNDNOTEND;
931 }
932
933 return DIOWEBPAGEHTMLCREATORTAGXENGINESTATUS_NOTFOUND;
934}
