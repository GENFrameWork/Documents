121bool DIOBUSPIRATE::Ini(int timeout)
122{
123 if(!diostream) return false;
124
125 if(!diostream->Open()) return false;
126 if(!diostream->WaitToConnected(timeout)) return false;
127
128 if(!WriteCommand(__L(""), timeout)) return false;
129 if(!WriteCommand(__L("m"), timeout)) return false;
130 if(!WriteCommand(__L("1") , timeout)) return false;
131
132 GEN_XSLEEP.Seconds(1);
133 diostream->GetInXBuffer()->Empty();
134
135 return (GetActualProtocol() == DIOBUSPIRATETYPEPROCOTOL_HIZ);

150DIOBUSPIRATETYPEPROCOTOL DIOBUSPIRATE::GetActualProtocol(int timeout)
151{
152 if(!diostream) return DIOBUSPIRATETYPEPROCOTOL_UNKNOWN;
153 if(diostream->GetStatus() != DIOSTREAMSTATUS_CONNECTED) return DIOBUSPIRATETYPEPROCOTOL_UNKNOWN;
154
155 XSTRING promptstatus;
156
157 WriteCommand(__L(""), timeout);
158 ReadPromptStatus(promptstatus, timeout);
159
160 if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_HIZ, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_HIZ;
161 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_1WIRE, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_1WIRE;
162 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_UART, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_UART;
163 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_I2C, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_I2C;
164 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_SPI, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_SPI;
165 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_2WIRE, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_2WIRE;
166 else if(promptstatus.Find(DIOBUSPIRATE_COMMANDMENU_PROTOCOL_3WIRE, false) != XSTRING_NOTFOUND) return DIOBUSPIRATETYPEPROCOTOL_3WIRE;
167
168 return DIOBUSPIRATETYPEPROCOTOL_UNKNOWN;
169}

187bool DIOBUSPIRATE::SetMode(DIOBUSPIRATETYPEPROCOTOL protocol, bool issoftware, int speed, bool issniffer, int timeout)
188{
189 if(!xtimerout) return false;
190 if(!diostream) return false;
191 if(diostream->GetStatus() != DIOSTREAMSTATUS_CONNECTED) return false;
192
193 XSTRING option;
194 bool status = false;
195
196 switch(protocol)
197 {
198 case DIOBUSPIRATETYPEPROCOTOL_I2C : if(WriteCommand(__L("m"), timeout))
199 {
200 option.Format(__L("%d"), protocol);
201 if(WriteCommand(option, timeout))
202 {
203 option.Format(__L("%d"), issoftware?1:2);
204 if(WriteCommand(option, timeout))
205 {
206 option.Format(__L("%d"), speed);
207 if(WriteCommand(option, timeout))
208 {
209 if(issniffer)
210 {
211 option.Format(__L("(2)"), speed);
212 status = WriteCommand(option, timeout);
213
214 } else status = true;
215 }
216 }
217 }
218 }
219
220 GEN_XSLEEP.Seconds(1);
221 diostream->GetInXBuffer()->Empty();
222 break;
223
224 default : break;
225
226 }
227
228 return status;
229}

244bool DIOBUSPIRATE::ReadI2CSnifferCommand(XBUFFER& in, XBUFFER& out)
245{
246 XBUFFER xbuffer;

249

254 {
255 stringsize++;

313bool DIOBUSPIRATE::ReadPromptStatus(XSTRING& promptstatus, int timeout)
314{
315 if(!diostream) return false;
316 if(diostream->GetStatus() != DIOSTREAMSTATUS_CONNECTED) return false;
317
318 promptstatus.Empty();
319
320 if(!xtimerout) return false;
321
322 xtimerout->Reset();
323
324 while(1)
325 {
326 if(diostream->WaitToFilledReadingBuffer(1, 1))
327 {
328 XBYTE data;
329 if(diostream->Read(&data, 1))
330 {
331 if(data != '>')
332 {
333 promptstatus.Add(data);
334 }
335 else return true;
336
337 xtimerout->Reset();
338 }
339 }
340
341 if((int)xtimerout->GetMeasureSeconds() >= timeout) break;
342 }
343
344 return false;
345}

360bool DIOBUSPIRATE::WriteCommand(XCHAR* command, int timeout)
361{
362 if(!diostream) return false;
363 if(diostream->GetStatus() != DIOSTREAMSTATUS_CONNECTED) return false;
364
365 XSTRING _command;
366 bool status;
367
368 _command = command;
369 _command += __L("\r");
370
371 XBUFFER charstr;
372
373 _command.ConvertToASCII(charstr);
374 status = diostream->Write((XBYTE*)charstr.Get(), _command.GetSize())?true:false;
375 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
376
377 return status;
378}
