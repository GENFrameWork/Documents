1026XDWORD XFILERIFF::AdjustSizeOfLists(XFILERIFF_LIST_NODE* node)
1027{
1028 XFILERIFF_LIST* list = NULL;
1029 XDWORD currentsize = 0;
1030
1031 list = node->GetData();
1032 if(!list) return 0;
1033
1034 if(!list->IsTypeList()) return list->GetSize();
1035
1036 for(int c=0; c<node->GetNChildren(); c++)
1037 {
1038 XFILERIFF_LIST_NODE* subnode = node->GetChild(c);
1039 if(subnode)
1040 {
1041 XFILERIFF_LIST* sublist = subnode->GetData();
1042 if(sublist)
1043 {
1044 if(sublist->IsTypeList())
1045 {
1046 currentsize += (AdjustSizeOfLists(subnode) + (sizeof(XDWORD) * 3));
1047 }
1048 else
1049 {
1050 currentsize += (sublist->GetSize() + (sizeof(XDWORD) * 2));
1051 }
1052 }
1053 }
1054 }
1055
1056 list->SetSize(currentsize + sizeof(XDWORD));
1057
1058 if(!WriteListToFile(list)) return false;
1059
1060 return currentsize;
1061}
