1019XDWORD XFILERIFF::AdjustSizeOfLists(XFILERIFF_LIST_NODE* node)
1020{
1021 XFILERIFF_LIST* list = NULL;
1022 XDWORD currentsize = 0;
1023
1024 list = node->GetData();
1025 if(!list) return 0;
1026
1027 if(!list->IsTypeList()) return list->GetSize();
1028
1029 for(int c=0; c<node->GetNChildren(); c++)
1030 {
1031 XFILERIFF_LIST_NODE* subnode = node->GetChild(c);
1032 if(subnode)
1033 {
1034 XFILERIFF_LIST* sublist = subnode->GetData();
1035 if(sublist)
1036 {
1037 if(sublist->IsTypeList())
1038 {
1039 currentsize += (AdjustSizeOfLists(subnode) + (sizeof(XDWORD) * 3));
1040 }
1041 else
1042 {
1043 currentsize += (sublist->GetSize() + (sizeof(XDWORD) * 2));
1044 }
1045 }
1046 }
1047 }
1048
1049 list->SetSize(currentsize + sizeof(XDWORD));
1050
1051 if(!WriteListToFile(list)) return false;
1052
