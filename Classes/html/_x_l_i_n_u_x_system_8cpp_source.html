467int XLINUXSYSTEM::GetCPUUsageTotal()
468{
469 if(xmutexcheckCPUusage)
470 {
471 xmutexcheckCPUusage->Lock();
472 }
473
474 static unsigned int lasttotaluser, lasttotaluserlow, lasttotalsys, lasttotalidle;
475 float percent;
476 int intpercent;
477 FILE* file;
478 unsigned long long totaluser, totaluserlow, totalsys, totalidle, total;
479
480 file = fopen("/proc/stat", "r");
481 fscanf(file, "cpu %llu %llu %llu %llu", &totaluser, &totaluserlow, &totalsys, &totalidle);
482 fclose(file);
483
484 if(totaluser < lasttotaluser || totaluserlow < lasttotaluserlow || totalsys < lasttotalsys || totalidle < lasttotalidle)
485 {
486 // Overflow detection. Just skip this value.
487 percent = -1.0;
488 }
489 else
490 {
491 total = (totaluser - lasttotaluser) + (totaluserlow - lasttotaluserlow) + (totalsys - lasttotalsys);
492 percent = total * 100;
493 total += (totalidle - lasttotalidle);
494 percent /= total;
495 }
496
497 lasttotaluser = totaluser;
498 lasttotaluserlow = totaluserlow;
499 lasttotalsys = totalsys;
500 lasttotalidle = totalidle;
501
502 intpercent = percent;
503
504 if(xmutexcheckCPUusage)
505 {
506 xmutexcheckCPUusage->UnLock();
507 }
508
509 return intpercent;
510}

524int XLINUXSYSTEM::GetCPUUsageForProcessName(XCHAR* processname)
525{
526 XSTRING command;
527 XSTRING result;
528 XBUFFER charcmd;
529
530 command.Format(__L("ps -C %s -o %%cpu"), processname);
531
532 command.ConvertToASCII(charcmd);
533
534 // Open a pipe to the command
535 FILE* pipe = popen((char*)charcmd.Get(), "r");
536 if(!pipe)
537 {
538 return 0; // Error
539 }
540
541 char buffer[128];
542 while(!feof(pipe))
543 {
544 if(fgets(buffer, 128, pipe) != NULL)
545 {
546 result += buffer;
547 }
548 }
549
550 pclose(pipe);
551
552 // Extract the CPU usage percentage from the result
553 double cpupercentage = 0.0f;
554 result.UnFormat(__L("%%CPU\n %lf\n"), &cpupercentage);
555
556 return static_cast<int>(cpupercentage);
557}

702bool XLINUXSYSTEM::ShutDown(XSYSTEM_CHANGESTATUSTYPE type)
703{
704 int opt_reboot = 0;
705 int fd;
706
707 switch(type)
708 {
709 case XSYSTEM_CHANGESTATUSTYPE_REBOOT : opt_reboot = 1; break;
710 case XSYSTEM_CHANGESTATUSTYPE_POWEROFF : opt_reboot = 0; break;
711 default : break;
712 }
713
714 setpriority(PRIO_PROCESS, 0, PRIO_MIN);
715
716 whom = getlogin();
717
718 setpriority(PRIO_PROCESS, 0, PRIO_MIN);
719 signal(SIGINT, int_handler);
720 signal(SIGHUP, int_handler);
721 signal(SIGQUIT, int_handler);
722 signal(SIGTERM, int_handler);
723
724 chdir("/");
725
726 signal(SIGPIPE, SIG_IGN);
727 signal(SIGINT, SIG_IGN); // now there's no turning back...
728
729 kill(1, SIGTSTP); // tell init not to spawn more getty's
730
731 sync();
732
733 signal(SIGTERM, SIG_IGN);
734
735 if(fork() > 0) sleep(1000); // the parent will die soon...
736
737 setpgrp(); // so the shell wont kill us in the fall
738
739 kill(-1, SIGTERM); // a gentle kill of all other processes except init
740 sleep(2);
741
742 kill(-1, SIGKILL); // now use brute force...
743
744 acct(NULL); // turn off accounting
745
746 sync();
747 sleep(2);
748
749 swap_off(); // remove swap files and partitions using swapoff
750
751 unmount_disks(); // unmount disks...
752
753 sync();
754 sleep(1);
755
756 if(opt_reboot)
757 {
758 reboot(LINUX_REBOOT_CMD_RESTART);
759 }
760 else
761 {
762 reboot(LINUX_REBOOT_CMD_POWER_OFF);
763 }
764
765 exit(0); // to quiet gcc
766
767 return true;
768}

876void swap_off()
877{
879 // unmounted. It you don't have swapoff(1) or use mount to
880 // add swapspace, this may not be necessary, but I guess it
881 // won't hurt
882
883 int pid;
884 int result;
885 int status;
886
887 sync();
888
889 if((pid = fork()) < 0)
890 {
891 // Cannot fork for swapoff. Shrug!
892 return;
893 }
894
895 if(!pid)
896 {
897 execl("/sbin/swapoff" , SWAPOFF_ARGS, NULL);
898 execl("/etc/swapoff" , SWAPOFF_ARGS, NULL);
899 execl("/bin/swapoff" , SWAPOFF_ARGS, NULL);
900
901 execlp("swapoff" , SWAPOFF_ARGS, NULL);
902
903 // Cannot exec swapoff, hoping umount will do the trick.
904 exit(0);
905 }
906
907 while((result = wait(&status)) != -1 && result != pid);
908}

918void unmount_disks()
919{
920 // better to use umount directly because it may be smarter than us
921
922 int pid;
923 int result;
924 int status;
925

927

934
935 if(!pid)
936 {
937 execl(_PATH_UMOUNT, UMOUNT_ARGS, NULL);
938
939 // Cannot exec %s, trying umount. _PATH_UMOUNT
940
941 execlp("umount", UMOUNT_ARGS, NULL);
942
943 // Cannot exec umount, giving up on umount.
944 exit(0);
945 }
946
947 while ((result = wait(&status)) != -1 && result != pid);
948
949 // Unmounting any remaining filesystems...
950
951 unmount_disks_ourselves();
952}

962void unmount_disks_ourselves()
963{
964 // unmount all disks
965
966 FILE* mtab;
967 struct mntent* mnt;
968 char* mntlist[128];
969 int i;
970 int n;
971 char* filesys;
972
973 sync();
974
975 if(!(mtab = setmntent(_PATH_MTAB, "r")))
976 {
977 // shutdown: Cannot open %s _PATH_MTAB;
978 return;
979 }
980
981 n = 0;
982 while (n < 100 && (mnt = getmntent(mtab)))
983 {
984 mntlist[n++] = strdup(mnt->mnt_fsname[0] == '/' ?
985 mnt->mnt_fsname : mnt->mnt_dir);
986 }
987
988 endmntent(mtab);
989
990 // we are careful to do this in reverse order of the mtab file
991
992 for(i = n - 1; i >= 0; i--)
993 {
994 filesys = mntlist[i];
995
996 if(umount(mntlist[i]) < 0)
997 {
998 // shutdown: Couldn't umount %s filesys
999 }
1000 }
1001}
