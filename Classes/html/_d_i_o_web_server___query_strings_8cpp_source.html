466int DIOWEBSERVER_QUERYSTRINGS::GetParamsFromURL(XCHAR* url)
467{
468 DIOURL _url;
469 int start;
470 int separator;
471 int end;
472 XSTRING param;
473
474 DelAllParam();
475
476 _url = url;
477
478 start = _url.Find(__L("?"),true);
479 if(start != XSTRING_NOTFOUND)
480 {
481 XSTRING ID;
482 XSTRING value;
483
484 start++;
485
486 do{
487 value.Empty();
488 end = _url.Find(__L("&"),true,start);
489 if(end == XSTRING_NOTFOUND)
490 _url.Copy(start, param);
491 else _url.Copy(start, end, param);
492
493 separator = param.Find(__L("="),true);
494 if(separator == XSTRING_NOTFOUND)
495 {
496 ID = param;
497 }
498 else
499 {
500 param.Copy(0 , separator,ID);
501 param.Copy(separator+1, value);
502 }
503
504 AddParam(ID, value);
505
506 start = end+1;
507
508 } while(end != XSTRING_NOTFOUND);
509 }
510
511
512 return GetNParams();
513}
