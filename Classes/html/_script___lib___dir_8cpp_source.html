156 bool status = false;

158 dir = GEN_XFACTORY.Create_Dir();
159 if(dir)
160 {
161 if(!params->GetSize())
162 {
163 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
164 return;
165 }
166

171 {
172 status = dir->Exist(string->Get());
173 }
174 }
175
176 GEN_XFACTORY.Delete_Dir(dir);
177 }
178
179 (*returnvalue) = status;
