263bool APPFLOWWEBSERVER::SendRequest(DIOWEBSERVER_CONNECTION* connection, DIOWEBHEADER_RESULT headerresult, XBUFFER* data, int timeout, XSTRING* addhead)
264{
265 DIOWEBSERVER_HEADER webserverheader;
266 XSTRING entitytag;
267 HASHMD5 hashmd5;
268 bool iscached = false;
269 bool status = false;
270
271 if(!connection) return false;
272 if(!connection->GetDIOStream()) return false;
273 if(!connection->GetDIOStream()->IsConnected()) return false;
274
275 if(data)
276 {
277 hashmd5.Do((*data));
278 hashmd5.GetResultString(entitytag);
279 iscached = connection->GetRequest()->GetIfNoneMatch()->Compare(entitytag)?false:true;
280 }
281
282 if(iscached)
283 {
284 webserverheader.Create(connection->GetRequest()->GetResource(), 1, 1, DIOWEBHEADER_RESULT_NOTMODIFIED, connection->GetRequest()->GetSize());
285 status = webserverheader.Write(connection->GetDIOStream(), 5);
286 }
287 else
288 {
289 XPATH resourceext;
290 int size = (data ?(int)data->GetSize():-1);
291
292 resourceext = connection->GetRequest()->GetResource()->Get();
293 resourceext.SetOnlyExt();
294
295 webserverheader.Create(connection->GetRequest()->GetResource(), 1, 1, headerresult, size, &entitytag, connection->GetServer()->IsPlayablePage(resourceext));
296 if(addhead) webserverheader.AddLine(addhead->Get());
297 status = webserverheader.Write(connection->GetDIOStream(), 5);
298 if(status)
299 {
300 if(size > 0) status = connection->Send((*data), timeout);
301 }
302 }
303
304 return status;
305}

397bool APPFLOWWEBSERVER::ResolveRequest(DIOWEBSERVER* server, DIOWEBSERVER_CONNECTION* connection, DIOWEBSERVER_REQUEST* request)
398{
399 if(!server) return false;
400 if(!connection) return false;
401 if(!request) return false;
402
403 XSTRING* resource = request->GetResource();
404 if(!resource) return false;
405
406 if(!cfg) return false;
407
408 DIOWEBSERVER_HEADER webserverheader;
409 DIOURL resourceconv;
410 int nparams = 0;
411 DIOWEBSERVER_QUERYSTRINGS querystring;
412 XSTRING extension;
413 bool ispagepluging = false;
414 bool status = false;
415
416 resourceconv.Set(resource->Get());
417 resourceconv.DecodeUnsafeChars();
418
419 if(isauthenticatedaccess)
420 {
421 if(!cfg->WebServer_GetPassword()->IsEmpty())
422 {
423 if(!request->GetLoginPassword(page_login, page_password))
424 {
425 SendRequest(connection, DIOWEBHEADER_RESULT_UNAUTHORIZED, NULL, 5);
426
427 return true;
428 }
429 else
430 {
431 bool ispasswordvalid;
432
433 ispasswordvalid = !page_password.Compare(cfg->WebServer_GetPassword()->Get(), true);
434 if(!ispasswordvalid)
435 {
436 XSTRING leyend;
437
438 leyend = __L("Invalid user or password!");
439 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
440 }
441
442 XSTRING IPstring;
443 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
444
445 APPFLOW_LOG_ENTRY(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server password [%s]: %s"), IPstring.Get(), cfg->WebServer_GetPassword()->Get(), (ispasswordvalid)? __L("Ok."): __L("INVALID!") );
446
447 if(!ispasswordvalid) return true;
448 }
449 }
450 else
451 {
452 bool invalidpassword = false;
453
454 if(page_login.IsEmpty() || page_password.IsEmpty())
455 {
456 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_DOAUTHENTICATE);
457
458 xevent.SetServer(server);
459 xevent.SetConnection(connection);
460 xevent.SetRequest(request);
461
462 PostEvent(&xevent);
463
464 if(xevent.IsRequestAuthenticate())
465 {
466 page_login = __L("usr");
467 page_password = __L("pwd");
468
469 } else invalidpassword = true;
470 }
471 else
472 {
473 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_CHECKAUTHENTICATE);
474
475 xevent.SetServer(server);
476 xevent.SetConnection(connection);
477 xevent.SetRequest(request);
478
479 PostEvent(&xevent);
480
481 if(!xevent.IsRequestAuthenticate()) invalidpassword = true;
482 }
483
484 if(invalidpassword)
485 {
486 XSTRING leyend;
487
488 leyend = __L("Invalid user or password!");
489 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
490
491 APPFLOW_LOG_ENTRY(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
492
493 return true;
494 }
495 }
496 }
497
498 //-------------------------------------------------------------------------------------------------------------------------------
499 /*
500 if(useragentID)
501 {
502 XSTRING browser;
503 XSTRING SO;
504
505 if(useragentID->Get(request->GetUserAgent()->Get(), browser, SO, 5))
506 {
507 APPFLOW_LOG_ENTRY(XLOGLEVEL_INFO, DIOWEBSERVER_LOGSECTIONID_VERBOSE, false, __L("Web browser : %s, with: %s."), browser.Get(), SO.Get());
508 }
509 }
510 */
511
512 //-------------------------------------------------------------------------------------------------------------------------------
513
514 status = false;
515 nparams = querystring.GetParamsFromURL(resourceconv.Get())?true:false;
516
517 querystring.DeleteParamsFromURL(resourceconv);
518
519 if(!resourceconv.Compare(__L("/"),true)) resourceconv.Set(__L("/index.html"));
520
521
522 //-------------------------------------------------------------------------------------------------------------------------------
523
524
525 XPATH xpathfile;
526
527 if(cfg->WebServer_PathResources()->IsEmpty())
528 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_WEB, xpathfile);
529 else xpathfile = cfg->WebServer_PathResources()->Get();
530
531 xpathfile.Slash_Delete();
532 xpathfile += resourceconv.Get();
533
534 //-------------------------------------------------------------------------------------------------------------------------------
535
536 if(resourceconv.GetExtension(extension))
537 {
538 int nplugins = webserver->GetPluginsManager()->Plugins_GetNPlugins();
539 for(int c=0; c<nplugins; c++)
540 {
541 DIOWEBSERVER_PLUGIN* plugin = webserver->GetPluginsManager()->Plugins_Get(c);
542 if(plugin)
543 {
544 if(plugin->PageExtension_IsContent(extension))
545 {
546 DIOWEBHEADER_RESULT headerresult = DIOWEBHEADER_RESULT_OK;
547 XSTRING result;
548
549 status = plugin->PageExtension(xpathfile, request, &querystring, headerresult, result);
550 if(status)
551 {
552 XSTRING head;
553 XSTRING data;
554
555 plugin->SeparedResult(result, head, data);
556
557 status = SendRequest(connection, headerresult, data, 5, &head);
558 }
559 else
560 {
561 XSTRING leyend;
562
563 leyend.Format(__L("Error to execute the %s."), plugin->GetName()->Get());
564
565 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
566 APPFLOW_LOG_ENTRY(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
567 }
568
569 XSTRING IPstring;
570 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
571
572 if(!status)
573 {
574 APPFLOW_LOG_ENTRY((status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR), DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server \"%s\" %s."), IPstring.Get(), resourceconv.Get(), status?__L("sent"):__L("not send"));
575 }
576
577 ispagepluging = true;
578 }
579 }
580 }
581 }
582
583 if(!ispagepluging)
584 {
585 if(!nparams && !isapirestonly)
586 {
587 status = SendRequest(connection, DIOWEBHEADER_RESULT_OK, xpathfile, 5);
588 if(!status)
589 {
590 // Brower close the conextion in thread
591 if(!connection->GetDIOStream()->IsConnected()) status = true;
592 }
593 }
594
595 if(!status)
596 {
597 if(!nparams && !isapirestonly)
598 {
599 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_REQUEST);
600
601 xevent.SetServer(server);
602 xevent.SetConnection(connection);
603 xevent.SetRequest(request);
604
605 PostEvent(&xevent);
606
607 status = xevent.GetStatus();
608 }
609 }
610
611 //-------------------------------------------------------------------------------------------------------------------------------
612
613 if(!status)
614 {
615 DIOWEBPAGEHTMLCREATOR* webHTMLpage = new DIOWEBPAGEHTMLCREATOR();
616 if(webHTMLpage)
617 {
618 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_REQUEST_ENDPOINT);
619
620 xevent.SetServer(server);
621 xevent.SetConnection(connection);
622 xevent.SetRequest(request);
623 xevent.SetQueryString(&querystring);
624
625 PostEvent(&xevent);
626
627 //DIOWEBHEADER_RESULT result = xevent.GetResult();
628
629 status = xevent.GetStatus();
630 if(status)
631 {
632 if(!xevent.GetOutputString()->IsEmpty())
633 {
634 webHTMLpage->Set(xevent.GetOutputString()->Get());
635 status = SendRequest(connection, DIOWEBHEADER_RESULT_OK, (*webHTMLpage), 5);
636 }
637 else
638 {
639 status = SendRequest(connection, DIOWEBHEADER_RESULT_OK, xevent.GetOutputBuffer(), 5);
640 }
641 }
642 else
643 {
644 XSTRING leyend;
645
646 leyend = __L("Error: command or erroneous parameters.");
647 GenerateMessagePage(leyend, (*webHTMLpage));
648
649 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
650
651 APPFLOW_LOG_ENTRY(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
652 }
653
654 XSTRING IPstring;
655 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
656
657 if(!status)
658 {
659 APPFLOW_LOG_ENTRY((status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR), DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server\"%s\" %s."), IPstring.Get(), resourceconv.Get(), status?__L("sent"):__L("not send"));
660 }
661
662 delete webHTMLpage;
663 }
664 }
665 else
666 {
667 XSTRING IPstring;
668 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
669
670 // APPFLOW_LOG_ENTRY(XLOGLEVEL_INFO, DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server of file \"%s\" sent."), IPstring.Get(), xpathfile.Get());
671 }
672 }
673
674 return true;
675}

838void APPFLOWWEBSERVER::HandleEvent_WebServer(DIOWEBSERVER_XEVENT* event)
839{
840 switch(event->GetEventType())
841 {
842 case DIOWEBSERVER_XEVENT_TYPE_REQUEST : ResolveRequest(event->GetServer(), event->GetConnection(), event->GetRequest());
843 break;
844
845 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_CONNECTED : if(!PostEvent(event))
846 {
847 // Accept connection with any version, protocol o resources.
848 event->SetStatus(true);
849 break;
850 }
851 break;
852
853 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_READDATA : PostEvent(event);
854 break;
855
856 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_DISCONNECTED : PostEvent(event);
857 break;
858 }
859}
