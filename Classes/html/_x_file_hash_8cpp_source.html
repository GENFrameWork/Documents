117bool XFILEHASH::Open(XPATH& pathname, bool readonly, bool checkhash, bool checkversion)
118{
119 if(!hash) return false;
120 if(!file) return false;
121
122 if(!hash->GetDefaultSize()) return false;
123
124 if(file->Exist(pathname)!=true) return false;
125
126 bool status = file->Open(pathname,readonly);
127
128 XSTRING IDini(16);
129 XWORD _ID = 0;
130 XWORD _type = 0;
131 XWORD _version = 0;
132
133 if(!file->Read((XBYTE*)IDini.Get(), XSTRING::GetSize(XFILE_ID)*sizeof(XWORD)+1)) return false;
134
135 IDini.AdjustSize();
136
137 if(!file->Read((XBYTE*)&_ID, sizeof(XWORD))) return false;
138 if(_ID!=ID) return false;
139
140 if(!file->Read((XBYTE*)&_type, sizeof(XWORD))) return false;
141 if(_type!=type) return false;
142
143 if(!file->Read((XBYTE*)&_version,sizeof(XWORD))) return false;
144 if(checkversion)
145 {
146 if(version!=_version) return false;
147 }
148
149 version = _version;
150
151 XDWORD sizeIDstring = 0;
152
153 if(!file->Read((XBYTE*)&sizeIDstring, sizeof(XDWORD))) return false;
154 if(sizeIDstring)
155 {
156 XWORD* IDstr = new XWORD[sizeIDstring+1];
157 if(IDstr)
158 {
159 memset((XBYTE*)IDstr, 0, sizeof(XWORD)*(sizeIDstring+1));
160
161 if(!file->Read((XBYTE*)IDstr, sizeof(XWORD)*sizeIDstring))
162 {
163 delete [] IDstr;
164 return false;
165 }
166
167 IDstring.Set((XWORD*)IDstr);
168
169 delete [] IDstr;
170 }
171 }
172
173 XBUFFER hashbuffer((XDWORD)hash->GetDefaultSize(), true);
174 if(!file->Read((XBYTE*)hashbuffer.Get(), hash->GetDefaultSize())) return false;
175
176 file->GetPosition(dataposition);
177
178 XQWORD size = file->GetSize() - GetDataPosition();
179
180 hash->ResetResult();
181
182 if(!hash->Do(file, size, GetDataPosition())) return false;
183
184 if(checkhash)
185 {
186 if(!hashbuffer.Compare(hash->GetResult()))
187 {
188 return false;
189 }
190 }
191
192 if(readonly) hashisupdate = false;
193
194 file->SetPosition(dataposition);
195
196 return status;
197}

212bool XFILEHASH::Create(XPATH& pathname, bool checkHASH)
213{
214 if(!hash) return false;
215 if(!file) return false;
216
217 if(!hash->GetDefaultSize()) return false;
218
219 bool status=file->Create(pathname);
220
221 hash->ResetResult();
222
223 XBUFFER hashbuffer((XDWORD)hash->GetDefaultSize(), true);
224
225 XDWORD sizeIDstring = IDstring.GetSize();
226
227 XSTRING IDini;

485bool XFILEHASH::UpdateHash()
486{
487 if(!hash) return false;
488 if(!file) return false;
489
490 file->Flush();
491
492 hash->ResetResult();
493
494 XQWORD size = file->GetSize() - GetDataPosition();
495
496 if(!hash->Do(file, size, GetDataPosition())) return false;
497
498 file->SetPosition(dataposition-hash->GetDefaultSize());

500 if(!file->Write((XBYTE*)hash->GetResult()->Get(), hash->GetDefaultSize())) return false;
501
502 hashisupdate = true;
