113bool DIOALERTSUDPSERVER::Ini(XCHAR* URL, int port)
114{
115 bool status = false;
116
117 UDPdiostreamcfg = new DIOSTREAMUDPCONFIG();
118 if(!UDPdiostreamcfg) return false;
119
120 UDPdiostreamcfg->SetMode(DIOSTREAMMODE_SERVER);
121 UDPdiostreamcfg->GetRemoteURL()->Set(URL);
122 UDPdiostreamcfg->SetRemotePort(port);
123 UDPdiostreamcfg->SetIsUsedDatagrams(true);
124
125 UDPdiostream = new DIOSTREAMUDPACKNOWLEDGE(UDPdiostreamcfg);
126 if(UDPdiostream)
127 {
128 status = UDPdiostream->Open();
129 if(status)
130 {
131 GEN_XFACTORY_CREATE(xmutexalert, Create_Mutex())
132 if(xmutexalert)
133 {
134 threadread = CREATEXTHREAD(XTHREADGROUPID_DIOALERTS, __L("DIOALERTSUDPSERVER::Ini"), ThreadReadFunction, (void*)this);
135 if(threadread) status = threadread->Ini();
136 }
137 }
138 }
139
140 return status;
141}

366bool DIOALERTSUDPSERVER::ReceivedEvents()
367{
368 if(!UDPdiostream) return false;
369
370 XBUFFER data;
371 XSTRING address;
372 XWORD port;
373 bool status = false;
374
375 if(UDPdiostream->ReadDatagram(address, port, data))
376 {
377 HASHCRC32 hashcrc32;
378 XDWORD CRC32;
379 DIOALERT* alert = NULL;
380
381 if(data.GetSize())
382 {
383 alert = new DIOALERT();
384 if(alert)
385 {
386 XSTRING string;
387 XDWORD dataxdword;
388 XDWORD sizestring;
389 XDWORD applicationversion;
390 XDWORD applicationsubversion;
391 XDWORD applicationsubversionerr;
392
393 data.Extract(CRC32, (data.GetSize()-sizeof(XDWORD)));
394
395 hashcrc32.ResetResult();
396 hashcrc32.Do(data);
397
398 if(CRC32 == hashcrc32.GetResultCRC32())
399 {
400 // Date time
401 data.Extract(dataxdword); alert->GetDateTime()->SetDay(dataxdword); // Day
402 data.Extract(dataxdword); alert->GetDateTime()->SetMonth(dataxdword); // Month
403 data.Extract(dataxdword); alert->GetDateTime()->SetYear(dataxdword); // Year
404 data.Extract(dataxdword); alert->GetDateTime()->SetHours(dataxdword); // Hours
405 data.Extract(dataxdword); alert->GetDateTime()->SetMinutes(dataxdword); // Minutes
406 data.Extract(dataxdword); alert->GetDateTime()->SetSeconds(dataxdword); // Seconds
407
408 data.Extract(sizestring); data.Extract(string, 0, sizestring); alert->Application_GetID()->Set(string); // Aplication String
409 data.Extract(applicationversion);
410 data.Extract(applicationsubversion);
411 data.Extract(applicationsubversionerr);
412
413 alert->Application_SetVersion(applicationversion, applicationsubversion, applicationsubversionerr);
414
415 data.Extract(dataxdword); alert->SetID(dataxdword); // ID
416 data.Extract(dataxdword); alert->SetType(dataxdword); // Type
417 data.Extract(dataxdword); alert->SetLevel((DIOALERTLEVEL)dataxdword); // Level
418 data.Extract(sizestring); data.Extract(string, 0, sizestring); alert->GetOrigin()->Set(string); // Origin
419 data.Extract(sizestring); data.Extract(string, 0, sizestring); alert->GetTitle()->Set(string); // Title
420 data.Extract(sizestring); data.Extract(string, 0, sizestring); alert->Get_Message()->Set(string); // Message
421
422 status = alerts.Add(alert);
423 }
424
425 if(!status) delete alert;
426 }
427 }
428 }
429
430 return status;
431}
