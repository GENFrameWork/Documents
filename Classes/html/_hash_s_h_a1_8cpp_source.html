179void HASHSHA1::Input(XBYTE* input, XQWORD length)
180{
181 if(!length) return;
182
183 if(computed || corrupted)
184 {
185 corrupted = true;
186 return;
187 }
188
189 while(length-- && !corrupted)
190 {
191 if(msgblockindex < 64) msgblock[msgblockindex++] = (*input & 0xFF);
192
193 lengthlow += 8;
194 lengthlow &= 0xFFFFFFFF; // Force it to 32 bits
195
196 if(lengthlow == 0)
197 {
198 lengthhigh++;
199 lengthhigh &= 0xFFFFFFFF; // Force it to 32 bits
200
201 if(lengthhigh == 0) corrupted = true; // Message is too long
202 }
203
204 if(msgblockindex == 64) ProcessMessageBlock();
205
206 input++;
207 }
208}

218void HASHSHA1::ProcessMessageBlock()
219{
220 XDWORD K[] = { 0x5A827999 ,
221 0x6ED9EBA1 ,
222 0x8F1BBCDC ,
223 0xCA62C1D6
224 };
225 int t; // Loop counter
226 XDWORD temp; // Temporary word value
227 XDWORD W[80]; // Word sequence
228 XDWORD A;
229 XDWORD B;
230 XDWORD C;
231 XDWORD D;
232 XDWORD E; // Word buffers
233
234 for(t=0;t<16;t++)
235 {
236 W[t] = ((unsigned) msgblock[t * 4]) << 24;
237 W[t] |= ((unsigned) msgblock[t * 4 + 1]) << 16;
238 W[t] |= ((unsigned) msgblock[t * 4 + 2]) << 8;
239 W[t] |= ((unsigned) msgblock[t * 4 + 3]);
240 }
241
242 for(t=16;t<80;t++)
243 {
244 W[t] = CircularShift(1,W[t-3] ^ W[t-8] ^ W[t-14] ^ W[t-16]);
245 }
246
247 A = H[0];
248 B = H[1];
249 C = H[2];
250 D = H[3];
251 E = H[4];
252
253 for(t=0;t<20;t++)
254 {
255 temp = CircularShift(5,A) + ((B & C) | ((~B) & D)) + E + W[t] + K[0];
256 temp &= 0xFFFFFFFF;
257 E = D;
258 D = C;
259 C = CircularShift(30,B);
260 B = A;
261 A = temp;
262 }
263
264 for(t=20;t<40;t++)
265 {
266 temp = CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[1];
267 temp &= 0xFFFFFFFF;
268 E = D;
269 D = C;
270 C = CircularShift(30,B);
271 B = A;
272 A = temp;
273 }
274
275 for(t=40;t<60;t++)
276 {
277 temp = CircularShift(5,A) + ((B & C) | (B & D) | (C & D)) + E + W[t] + K[2];
278 temp &= 0xFFFFFFFF;
279 E = D;
280 D = C;
281 C = CircularShift(30,B);
282 B = A;
283 A = temp;
284 }
285
286 for(t=60;t<80;t++)
287 {
288 temp = CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[3];
289 temp &= 0xFFFFFFFF;
290 E = D;
291 D = C;
292 C = CircularShift(30,B);
293 B = A;
294 A = temp;
295 }
296
297 H[0] = (H[0] + A) & 0xFFFFFFFF;
298 H[1] = (H[1] + B) & 0xFFFFFFFF;
299 H[2] = (H[2] + C) & 0xFFFFFFFF;
300 H[3] = (H[3] + D) & 0xFFFFFFFF;
301 H[4] = (H[4] + E) & 0xFFFFFFFF;
302
303 msgblockindex = 0;
304}

314void HASHSHA1::PadMessage()
315{
316 // Check to see if the current message block is too small to hold
317 // the initial padding bits and length. If so, we will pad the
318 // block, process it, and then continue padding into a second block.
319
320 if(msgblockindex > 55)
321 {
322 msgblock[msgblockindex++] = 0x80;
323
324 while(msgblockindex < 64)
325 {
326 msgblock[msgblockindex++] = 0;
327 }
328
329 ProcessMessageBlock();
330
331 while(msgblockindex < 56)
332 {
333 msgblock[msgblockindex++] = 0;
334 }
335 }
336 else
337 {
338 msgblock[msgblockindex++] = 0x80;
339
340 while(msgblockindex < 56)
341 {
342 msgblock[msgblockindex++] = 0;
343 }
344
345 }
346
347
348 // Store the message length as the last 8 octets
349
350 msgblock[56] = (lengthhigh >> 24) & 0xFF;
351 msgblock[57] = (lengthhigh >> 16) & 0xFF;
352 msgblock[58] = (lengthhigh >> 8) & 0xFF;
353 msgblock[59] = (lengthhigh) & 0xFF;
354
355 msgblock[60] = (lengthlow >> 24) & 0xFF;
356 msgblock[61] = (lengthlow >> 16) & 0xFF;
357 msgblock[62] = (lengthlow >> 8) & 0xFF;
358 msgblock[63] = (lengthlow) & 0xFF;
359
360 ProcessMessageBlock();
361}
