172void HASHSHA1::Input(XBYTE* input, XQWORD length)
173{
174 if(!length) return;
175
176 if(computed || corrupted)
177 {
178 corrupted = true;
179 return;
180 }
181
182 while(length-- && !corrupted)
183 {
184 if(msgblockindex < 64) msgblock[msgblockindex++] = (*input & 0xFF);
185
186 lengthlow += 8;
187 lengthlow &= 0xFFFFFFFF; // Force it to 32 bits
188
189 if(lengthlow == 0)
190 {
191 lengthhigh++;
192 lengthhigh &= 0xFFFFFFFF; // Force it to 32 bits
193
194 if(lengthhigh == 0) corrupted = true; // Message is too long
195 }
196
197 if(msgblockindex == 64) ProcessMessageBlock();
198
199 input++;
200 }
201}

211void HASHSHA1::ProcessMessageBlock()
212{
213 XDWORD K[] = { 0x5A827999 ,
214 0x6ED9EBA1 ,
215 0x8F1BBCDC ,
216 0xCA62C1D6
217 };
218 int t; // Loop counter
219 XDWORD temp; // Temporary word value
220 XDWORD W[80]; // Word sequence
221 XDWORD A;
222 XDWORD B;
223 XDWORD C;
224 XDWORD D;
225 XDWORD E; // Word buffers
226
227 for(t=0;t<16;t++)
228 {
229 W[t] = ((unsigned) msgblock[t * 4]) << 24;
230 W[t] |= ((unsigned) msgblock[t * 4 + 1]) << 16;
231 W[t] |= ((unsigned) msgblock[t * 4 + 2]) << 8;
232 W[t] |= ((unsigned) msgblock[t * 4 + 3]);
233 }
234
235 for(t=16;t<80;t++)
236 {
237 W[t] = CircularShift(1,W[t-3] ^ W[t-8] ^ W[t-14] ^ W[t-16]);
238 }
239
240 A = H[0];
241 B = H[1];
242 C = H[2];
243 D = H[3];
244 E = H[4];
245
246 for(t=0;t<20;t++)
247 {
248 temp = CircularShift(5,A) + ((B & C) | ((~B) & D)) + E + W[t] + K[0];
249 temp &= 0xFFFFFFFF;
250 E = D;
251 D = C;
252 C = CircularShift(30,B);
253 B = A;
254 A = temp;
255 }
256
257 for(t=20;t<40;t++)
258 {
259 temp = CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[1];
260 temp &= 0xFFFFFFFF;
261 E = D;
262 D = C;
263 C = CircularShift(30,B);
264 B = A;
265 A = temp;
266 }
267
268 for(t=40;t<60;t++)
269 {
270 temp = CircularShift(5,A) + ((B & C) | (B & D) | (C & D)) + E + W[t] + K[2];
271 temp &= 0xFFFFFFFF;
272 E = D;
273 D = C;
274 C = CircularShift(30,B);
275 B = A;
276 A = temp;
277 }
278
279 for(t=60;t<80;t++)
280 {
281 temp = CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[3];
282 temp &= 0xFFFFFFFF;
283 E = D;
284 D = C;
285 C = CircularShift(30,B);
286 B = A;
287 A = temp;
288 }
289
290 H[0] = (H[0] + A) & 0xFFFFFFFF;
291 H[1] = (H[1] + B) & 0xFFFFFFFF;
292 H[2] = (H[2] + C) & 0xFFFFFFFF;
293 H[3] = (H[3] + D) & 0xFFFFFFFF;
294 H[4] = (H[4] + E) & 0xFFFFFFFF;
295
296 msgblockindex = 0;
297}

307void HASHSHA1::PadMessage()
308{
309 // Check to see if the current message block is too small to hold
310 // the initial padding bits and length. If so, we will pad the
311 // block, process it, and then continue padding into a second block.
312
313 if(msgblockindex > 55)
314 {
315 msgblock[msgblockindex++] = 0x80;
316
317 while(msgblockindex < 64)
318 {
319 msgblock[msgblockindex++] = 0;
320 }
321
322 ProcessMessageBlock();
323
324 while(msgblockindex < 56)
325 {
326 msgblock[msgblockindex++] = 0;
327 }
328 }
329 else
330 {
331 msgblock[msgblockindex++] = 0x80;
332
333 while(msgblockindex < 56)
334 {
335 msgblock[msgblockindex++] = 0;
336 }
337
338 }
339
340
341 // Store the message length as the last 8 octets
342
343 msgblock[56] = (lengthhigh >> 24) & 0xFF;
344 msgblock[57] = (lengthhigh >> 16) & 0xFF;
345 msgblock[58] = (lengthhigh >> 8) & 0xFF;
346 msgblock[59] = (lengthhigh) & 0xFF;
347
348 msgblock[60] = (lengthlow >> 24) & 0xFF;
349 msgblock[61] = (lengthlow >> 16) & 0xFF;
350 msgblock[62] = (lengthlow >> 8) & 0xFF;
351 msgblock[63] = (lengthlow) & 0xFF;
352
353 ProcessMessageBlock();
354}

369XDWORD HASHSHA1::CircularShift(int bits, XDWORD word)
370{
371 return ((word << bits) & 0xFFFFFFFF) | ((word & 0xFFFFFFFF) >> (32-bits));
