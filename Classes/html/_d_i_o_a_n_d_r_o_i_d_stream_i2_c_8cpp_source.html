134bool DIOANDROIDSTREAMI2C::Open()
135{
136 if(!config) return false;
137
138 if(!threadconnection) return false;
139
140 XBUFFER charstr;
141
142 (*config->GetLocalDeviceName()).ConvertToASCII(charstr);
143 handle = open(charstr.GetPtrChar(), O_RDWR);
144 if(handle<0) return false;
145
146 if(ioctl(handle, DIOANDROIDSTREAMI2C_SLAVE, config->GetRemoteAddress()) < 0) return false;
147
148 SetEvent(DIOSTREAMI2C_FSMEVENT_CONNECTED);
149
150 status = DIOSTREAMSTATUS_CONNECTED;
151
152 ResetXBuffers();
153 ResetConnectionStatistics();
154
155 return threadconnection->Ini();
156}

215void DIOANDROIDSTREAMI2C::ThreadConnection(void* data)
216{
217 DIOANDROIDSTREAMI2C* diostream = (DIOANDROIDSTREAMI2C*)data;
218 if(!diostream) return;
219
220 if(diostream->GetEvent()==DIOSTREAMI2C_FSMEVENT_NONE) // No hay nuevos Eventos
221 {
222 switch(diostream->GetCurrentState())
223 {
224 case DIOSTREAMI2C_FSMSTATE_NONE : break;
225
226 case DIOSTREAMI2C_FSMSTATE_CONNECTED : break;
227
228 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : { if(diostream->outbuffer->IsBlocked()) break;
229
230 if(!diostream->IsBlockWrite())
231 {
232 XDWORD size = diostream->outbuffer->GetSize();
233 if(size)
234 {
235 diostream->outbuffer->SetBlocked(true);
236 XDWORD bw = diostream->I2C_Write(diostream->config->GetRemoteAddress(), diostream->outbuffer->Get(), size);
237 diostream->outbuffer->SetBlocked(false);
238
239 if(bw)
240 {
241 diostream->outbuffer->Extract(NULL, 0 , size);
242 }
243 }
244 }
245
246
247 if(!diostream->IsBlockRead())
248 {
249 if(diostream->sizeread)
250 {
251 XBYTE* _data = new XBYTE[diostream->sizeread];
252
253 if(_data)
254 {
255 if(diostream->I2C_Read(diostream->config->GetRemoteAddress(), _data, diostream->sizeread))
256 {
257 diostream->inbuffer->Add(_data, diostream->sizeread);
258 //XTRACE_PRINTCOLOR(3, __L("(Addr %04X, size:%d)"), diostream->config->GetRemoteAddress(), diostream->inbuffer->GetSize());
259 //XTRACE_PRINTDATABLOCK(1, (*diostream->inbuffer));
260
261 diostream->sizeread = 0;
262 }
263
264 delete [] _data;
265 }
266 }
267 }
268 }
269 break;
270
271
272 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
273
274 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : break;
275
276 }
277 }
278 else
279 {
280 if(diostream->GetEvent()<DIOSTREAMI2C_LASTEVENT)
281 {
282 diostream->CheckTransition();
283
284 switch(diostream->GetCurrentState())
285 {
286 case DIOSTREAMI2C_FSMSTATE_NONE : break;
287
288 case DIOSTREAMI2C_FSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
289 xevent.SetDIOStream(diostream);
290
291 diostream->PostEvent(&xevent);
292
293 diostream->SetEvent(DIOSTREAMI2C_FSMEVENT_WAITINGTOREAD);
294 }
295 break;
296
297 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : break;
298
299 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
300
301 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
302 xevent.SetDIOStream(diostream);
303 diostream->PostEvent(&xevent);
304
305 diostream->threadconnection->Run(false);
306 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
307 }
308 break;
309 }
310 }
311 }
312}
