127bool DIOANDROIDSTREAMI2C::Open()
128{
129 if(!config) return false;
130
131 if(!threadconnection) return false;
132
133 XBUFFER charstr;
134
135 (*config->GetLocalDeviceName()).ConvertToASCII(charstr);
136 handle = open(charstr.GetPtrChar(), O_RDWR);
137 if(handle<0) return false;
138
139 if(ioctl(handle, DIOANDROIDSTREAMI2C_SLAVE, config->GetRemoteAddress()) < 0) return false;
140
141 SetEvent(DIOSTREAMI2C_FSMEVENT_CONNECTED);
142
143 status = DIOSTREAMSTATUS_CONNECTED;
144
145 ResetXBuffers();
146 ResetConnectionStatistics();
147
148 return threadconnection->Ini();
149}

208void DIOANDROIDSTREAMI2C::ThreadConnection(void* data)
209{
210 DIOANDROIDSTREAMI2C* diostream = (DIOANDROIDSTREAMI2C*)data;
211 if(!diostream) return;
212
213 if(diostream->GetEvent()==DIOSTREAMI2C_FSMEVENT_NONE) // No hay nuevos Eventos
214 {
215 switch(diostream->GetCurrentState())
216 {
217 case DIOSTREAMI2C_FSMSTATE_NONE : break;
218
219 case DIOSTREAMI2C_FSMSTATE_CONNECTED : break;
220
221 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : { if(diostream->outbuffer->IsBlocked()) break;
222
223 if(!diostream->IsBlockWrite())
224 {
225 XDWORD size = diostream->outbuffer->GetSize();
226 if(size)
227 {
228 diostream->outbuffer->SetBlocked(true);
229 XDWORD bw = diostream->I2C_Write(diostream->config->GetRemoteAddress(), diostream->outbuffer->Get(), size);
230 diostream->outbuffer->SetBlocked(false);
231
232 if(bw)
233 {
234 diostream->outbuffer->Extract(NULL, 0 , size);
235 }
236 }
237 }
238
239
240 if(!diostream->IsBlockRead())
241 {
242 if(diostream->sizeread)
243 {
244 XBYTE* _data = new XBYTE[diostream->sizeread];
245
246 if(_data)
247 {
248 if(diostream->I2C_Read(diostream->config->GetRemoteAddress(), _data, diostream->sizeread))
249 {
250 diostream->inbuffer->Add(_data, diostream->sizeread);
251 //XTRACE_PRINTCOLOR(3, __L("(Addr %04X, size:%d)"), diostream->config->GetRemoteAddress(), diostream->inbuffer->GetSize());
252 //XTRACE_PRINTDATABLOCK(1, (*diostream->inbuffer));
253
254 diostream->sizeread = 0;
255 }
256
257 delete [] _data;
258 }
259 }
260 }
261 }
262 break;
263
264
265 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
266
267 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : break;
268
269 }
270 }
271 else
272 {
273 if(diostream->GetEvent()<DIOSTREAMI2C_LASTEVENT)
274 {
275 diostream->CheckTransition();
276
277 switch(diostream->GetCurrentState())
278 {
279 case DIOSTREAMI2C_FSMSTATE_NONE : break;
280
281 case DIOSTREAMI2C_FSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
282 xevent.SetDIOStream(diostream);
283
284 diostream->PostEvent(&xevent);
285
286 diostream->SetEvent(DIOSTREAMI2C_FSMEVENT_WAITINGTOREAD);
287 }
288 break;
289
290 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : break;
291
292 case DIOSTREAMI2C_FSMSTATE_SENDINGDATA : break;
293
294 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
295 xevent.SetDIOStream(diostream);
296 diostream->PostEvent(&xevent);
297
298 diostream->threadconnection->Run(false);
299 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
300 }
301 break;
302 }
303 }
304 }
305}
