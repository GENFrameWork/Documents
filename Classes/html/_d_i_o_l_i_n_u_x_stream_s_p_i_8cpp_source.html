111bool DIOLINUXSTREAMSPI::Open()
112{
113 if(!config) return false;
114
115 XBUFFER charstr;
116
117 (*config->GetLocalDeviceName()).ConvertToASCII(charstr);
118 handle = open(charstr.GetPtrChar(), O_RDWR);
119 //XTRACE_PRINTCOLOR(((handle<0)?4:1), __L("SPI Open %s : %s"), config->GetLocalDeviceName()->Get(), ((handle<0)?__L("Error"):__L("Ok")));
120 if(handle<0) return false;
121
122 // Set SPI parameters.
123 // Why are we reading it afterwriting it? I've no idea, but for now I'm blindly
124 // copying example code I've seen online...
125
126 XBYTE _mode = config->GetSPIMode();
127 XBYTE _bits = config->GetNBitsWord();
128 XDWORD _speed = config->GetSpeed();
129
130 if (ioctl(handle, SPI_IOC_WR_MODE , &_mode) < 0) return false;
131 if (ioctl(handle, SPI_IOC_WR_BITS_PER_WORD , &_bits) < 0) return false;
132 if (ioctl(handle, SPI_IOC_WR_MAX_SPEED_HZ , &_speed) < 0) return false;
133
134 if (ioctl(handle, SPI_IOC_RD_MODE , &_mode) < 0) return false;
135 if (ioctl(handle, SPI_IOC_RD_BITS_PER_WORD , &_bits) < 0) return false;
136 if (ioctl(handle, SPI_IOC_RD_MAX_SPEED_HZ , &_speed) < 0) return false;
137
138 return DIOSTREAMSPI::Open();
139}

192 transf.rx_buf = (unsigned long)bufferread;
193 transf.len = sizeint;

196 transf.bits_per_word = config->GetNBitsWord();
