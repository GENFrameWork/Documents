372bool APPUPDATE::Ini(APPCFG* cfg, APPUPDATE_CFG* updatecfg)
373{
374 if(!cfg) return false;
375
376 this->app_cfg = cfg;
377
378 this->update_cfg.CopyFrom(updatecfg);
379
380 //--------------------------------------------------------------------------------------
381
382 if(!cfg->ApplicationUpdate_IsActive()) return true;
383
384 //--------------------------------------------------------------------------------------
385
386 xscheduler = new XSCHEDULER();
387 if(!xscheduler) return false;
388
389 CreateTaskUpdate();
390
391 SubscribeEvent(XEVENT_TYPE_SCHEDULER, xscheduler);
392 xscheduler->Ini();
393
394 //--------------------------------------------------------------------------------------
395
396 EventManager(APPUPDATE_EVENTOPERATION_REGISTER);
397
398 dioappupdate = new DIOAPPLICATIONUPDATE(update_cfg.GetAppVersion()->GetVersion() ,
399 update_cfg.GetAppVersion()->GetSubVersion() ,
400 update_cfg.GetAppVersion()->GetSubVersionError() ,
401 update_cfg.GetAppName()->Get(),
402 (*update_cfg.GetPathRootApp()));
403
404 if(!dioappupdate) return false;
405
406 dioappupdate->EliminateOldRestorations(cfg->ApplicationUpdate_GetMaxRestorations());
407
408 return true;
409}

425bool APPUPDATE::EventManager(APPUPDATE_EVENTOPERATION eventoperation, XSUBJECT* xsubject, XOBSERVER* observer)
426{
427 XDWORD events[] = { DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE ,
428 DIOAPPLICATIONUPDATE_XEVENT_TYPE_CHECKVERSIONAVAILABLE ,
429 DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE ,
430 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE ,
431 DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES ,
432 DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILE ,
433 DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILES ,
434 DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES ,
435 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILE ,
436 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEEXECFILE ,
437 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES ,
438 DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END ,
439 DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILE ,
440 DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES ,
441 };
442
443
444 for(XDWORD c=0; c< sizeof(events)/sizeof(XDWORD); c++)
445 {
446 switch(eventoperation)
447 {
448 case APPUPDATE_EVENTOPERATION_REGISTER : RegisterEvent(events[c]); break;
449 case APPUPDATE_EVENTOPERATION_DEREGISTER : DeRegisterEvent(events[c]); break;
450 case APPUPDATE_EVENTOPERATION_SUBSCRIBE : SubscribeEvent(events[c] , xsubject, observer); break;
451 case APPUPDATE_EVENTOPERATION_UNSUBSCRIBE : UnSubscribeEvent(events[c] , xsubject, observer); break;
452 default : return false;
453 }
454 }
455
456 return true;
457}

469bool APPUPDATE::Do()
470{
471 DIOAPPLICATIONUPDATE_VERSIONDATA applicationversiondata;
472 XPATH xpathupdate;
473 XPATH xpathbackup;
474 XSTRING string;
475 XSTRING stringresult;
476 bool status = false;
477 XSTRING dir_mask;
478 DIOAPPLICATIONUPDATE_ERROR error;
479
480 dir_mask = DIOAPPLICATIONUPDATE_RESTOREDIR_ID;
481 dir_mask += DIOAPPLICATIONUPDATE_RESTOREDIR_MASK;
482
483 if(!dioappupdate) return false;
484
485 if(update_cfg.GetDolog()) EventManager(APPUPDATE_EVENTOPERATION_SUBSCRIBE, dioappupdate, this);
486
487 // --------------------------------------------------------------------------------------------
488 // Get last update available
489 if(dioappupdate->GetVersionAvailable((*update_cfg.GetURL()), update_cfg.Get_Port(), applicationversiondata, 20, error))
490 {
491 // Check last update available
492 if(dioappupdate->IsNewVersion(applicationversiondata))
493 {
494 // Have new version available
495
496 // Download new version to update available to subdir root/update[x.x.x] Update Version
497 xpathupdate = update_cfg.GetPathRootApp()->Get();
498 xpathupdate.Slash_Add();
499 xpathupdate.AddFormat(dir_mask.Get() , applicationversiondata.GetVersion()
500 , applicationversiondata.GetSubVersion()
501 , applicationversiondata.GetSubVersionError());
502
503 status = dioappupdate->Download((*update_cfg.GetURL()), update_cfg.Get_Port(), xpathupdate, applicationversiondata, 60, error);
504
505 //-----------------------------------------------------------------------
506 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES);
507
508 xevent.SetVersionData(&applicationversiondata);
509 xevent.SetError(error);
510
511 PostEvent(&xevent);
512 //-----------------------------------------------------------------------
513
514 if(status)
515 {
516 // New version update downloaded
517
518
519 //-----------------------------------------------------------------------
520 // Eliminate resources APPLICATION
521 { DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES);
522 xevent.SetVersionData(&applicationversiondata);
523 xevent.SetError(error);
524
525 PostEvent(&xevent);
526 }
527 //-----------------------------------------------------------------------
528
529
530 // Actual files to be updated available to subdir root/update[x.x.x] Actual version
531 xpathbackup = update_cfg.GetPathRootApp()->Get();;
532 xpathbackup.Slash_Add();
533
534 xpathbackup.AddFormat(dir_mask.Get() , update_cfg.GetAppVersion()->GetVersion()
535 , update_cfg.GetAppVersion()->GetSubVersion()
536 , update_cfg.GetAppVersion()->GetSubVersionError());
537
538 // Change actual files for the new version
539 status = dioappupdate->ChangeFilesFromDownload(applicationversiondata, xpathupdate, xpathbackup, error);
540
541 //-----------------------------------------------------------------------
542 { DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES);
543
544 xevent.SetVersionData(&applicationversiondata);
545 xevent.SetError(error);
546
547 PostEvent(&xevent);
548 }
549 //-----------------------------------------------------------------------
550
551 if(!status)
552 {
553 status = dioappupdate->Restore(xpathbackup, error);
554
555 //-----------------------------------------------------------------------
556 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES);
557
558 xevent.SetVersionData(&applicationversiondata);
559 xevent.SetError(error);
560
561 PostEvent(&xevent);
562 //-----------------------------------------------------------------------
563 }
564
565 if(status)
566 {
567 if(applicationversiondata.SystemMustBeInit())
568 {
569 GEN_XSYSTEM.ShutDown(XSYSTEM_CHANGESTATUSTYPE_REBOOT);
570 }
571 else
572 {
573 // Exit & Restart of Application
574
575 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END);
576
577 xevent.SetVersionData(&applicationversiondata);
578 xevent.SetError(error);
579
580 PostEvent(&xevent);
581
582 dioappupdate->RestartApplication(error);
583 }
584 }
585 }
586 }
587 }
588
589 if(update_cfg.GetDolog()) EventManager(APPUPDATE_EVENTOPERATION_UNSUBSCRIBE, dioappupdate, this);
590
591 return status;
592}

756bool APPUPDATE::LogOfEvent(DIOAPPLICATIONUPDATE_XEVENT* event)
757{
758 DIOAPPLICATIONUPDATE_FILE* updatefile = event->GetActualFileToUpdate();
759 XSTRING string;
760
761 switch(event->GetEventType())
762 {
763 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE : if(event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR)
764 {
765 string.Format( __L("Update Version available %d.%d.%d")
766 , event->GetVersionData()->GetVersion()
767 , event->GetVersionData()->GetSubVersion()
768 , event->GetVersionData()->GetSubVersionError());
769 }
770 else
771 {
772 string.Format( __L("No version available to update"));
773 }
774 break;
775
776 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_CHECKVERSIONAVAILABLE : string.Format( __L("Version %d.%d.%d is ")
777 , event->GetVersionData()->GetVersion()
778 , event->GetVersionData()->GetSubVersion()
779 , event->GetVersionData()->GetSubVersionError());
780
781 switch(event->GetVersionComparation())
782 {
783 case -1 : string.Add(__L("old")); break;
784 case 0 : string.Add(__L("equal")); break;
785 case 1 : string.Add(__L("new")); break;
786 }
787
788 if(event->GetVersionComparation() <=0) string.AddFormat(__L(". It will not be updated"));
789
790 break;
791
792
793 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE : if(updatefile) string.Format( __L("Download file %s (%dk).")
794 , updatefile->GetXPathFile()->Get()
795 , updatefile->GetSize()/1024);
796
797 break;
798
799 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE : if(updatefile) string.Format( __L("Unzip file %s (%dk) -> %s.")
800 , updatefile->GetXPathFile()->Get()
801 , updatefile->GetSize()/1024
802 , event->GetActualUnzipFileName()->Get());
803 break;
804
805 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES : string.Format(__L("Downloads file %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
806 break;
807
808 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILE : if(updatefile) string.Format( __L("Backup original file %s (%dk).")
809 , updatefile->GetXPathFile()->Get()
810 , updatefile->GetSize()/1024);
811 break;
812
813 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILES : string.Format(__L("Backup Original file %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
814 break;
815
816 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES : string.Format(__L("Application release blocked resources"));
817 break;
818
819
820 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILE : if(updatefile) string.Format( __L("Copy update file %s (%dk).")
821 , updatefile->GetXPathFile()->Get()
822 , updatefile->GetSize()/1024);
823 break;
824
825 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEEXECFILE : if(updatefile) string.Format( __L("Notify update EXEC file %s (%dk).")
826 , updatefile->GetXPathFile()->Get()
827 , updatefile->GetSize()/1024);
828 break;
829
830 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES : string.Format(__L("Update files %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
831 break;
832
833
834 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END : string.Format(__L("Terminate application"));
835 break;
836
837 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILE : if(updatefile) string.Format( __L("Restore file %s from backup (%dk).")
838 , updatefile->GetXPathFile()->Get()
839 , updatefile->GetSize()/1024);
840 break;
841
842 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES : string.Format(__L("Restore update files %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
843 break;
844 }
845
846 if(string.GetSize())
847 {
848 XLOG::GetInstance().AddEntry((event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR)?XLOGLEVEL_INFO:XLOGLEVEL_ERROR, DIOAPPLICATIONUPDATE_LOG_SECTIONID_APPUPDATE, false, string.Get());
849 return true;
850 }
851
852 return false;
853}

896bool APPUPDATE::CreateTaskUpdate(bool doajust)
897{
898 if(!app_cfg) return false;
899
900 XSCHEDULERTASK* xtask = new XSCHEDULERTASK(xscheduler);
901 if(!xtask) return false;
902
903 bool toactive = false;
904
905 XDATETIME* GEN_XFACTORY_CREATE(start, CreateDateTime())
906 XDATETIME* GEN_XFACTORY_CREATE(end, CreateDateTime())
907 if(start && end)
908 {
909 if(!app_cfg->ApplicationUpdate_GetCheckTime()->IsEmpty())
910 {
911 start->Read();
912
913 start->GetDateTimeFromString((*app_cfg->ApplicationUpdate_GetCheckTime()), XDATETIME_FORMAT_ADDTIME);
914 start->SetSeconds(0);
915 start->SetMilliSeconds(0);
916
917 start->AddDays(1);
918
919 end->Set(start);
920 end->AddSeconds(5);
921
922 xtask->SetTimeLimits(start, end);
923
924 toactive = true;
925 }
926 else
927 {
928 int minutescadence = app_cfg->ApplicationUpdate_GetCheckCadence();
929 if(minutescadence)
930 {
931 start->Read();
932
933 if(doajust)
934 {
935 int minutes = ((start->GetMinutes() / minutescadence) + ((start->GetMinutes() % minutescadence)?1:0)) * minutescadence;
936 if(minutes == start->GetMinutes()) minutes += minutescadence;
937 if(minutes >= 60)
938 {
939 start->AddHours(1);
940 minutes = 0;
941 }
942
943 start->SetMinutes(minutes);
944
945 } else start->AddMinutes(minutescadence);
946
947 start->SetSeconds(0);
948 start->SetMilliSeconds(0);
949
950 xtask->SetTimeLimits(start);
951
952 toactive = true;
953 }
954 }
955 }
956
957 if(toactive)
958 {
959 xtask->SetID(APPUPDATE_TASKID_CHECKUPDATEAPPLICATION);
960 xtask->SetNCycles(1);
961 xtask->SetIsActive(true);
962
963 xscheduler->Task_Add(xtask);
964 }
965 else delete xtask;
966
967 GEN_XFACTORY.DeleteDateTime(start);
968 GEN_XFACTORY.DeleteDateTime(end);
969
970 return true;
971}
