203bool DIOSTREAMUDPACKNOWLEDGE::Open()
204{
205 if(!diostream) return false;
206 if(!URLremote) return false;
207 if(!config) return false;
208
209 if(config->GetRemoteURL()->GetSize())
210 {
211 URLremote->Set(config->GetRemoteURL()->Get());
212 URLremote->ResolveURL(addressremote);
213 }

216
217 if(!diostream->Open()) return false;

219 if(config->IsServer())

223 }
224
225 return true;

241XDWORD DIOSTREAMUDPACKNOWLEDGE::Write(XBYTE* buffer, XDWORD size)
242{
243 if(!diostream) return 0;
244 if(!size) return 0;
245
246 if(addressremote.IsEmpty()) return 0;
247
248 DIOSTREAMUDP* diostreamudp = (DIOSTREAMUDP*)diostream;
249 XDWORD bw = 0;
250
251 if(diostreamudp->WriteDatagram(addressremote, portremote, buffer, size)) bw = size;
252
253 if(diostreamudp->WaitToWriteDatagramsEmpty(timeout))
254 {
255 //XTRACE_PRINTCOLOR(1, __L("Write UDP %s:%d : %s"), addressremote.Get(), portremote, bw==size??__L("[Ok.]"):__L("[Error!]"));
256
257 if(diostreamudp->WaitToGetFirstDatagram(false, timeout) != DIOSTREAMUDP_NOTFOUND)
258 {
259 XSTRING address;
260 XWORD port = 0;
261 XBUFFER xbuffer;
262 XDWORD hashresult[2] = { 0, 0 };
263
264 diostreamudp->ReadDatagram(address, port, xbuffer);
265
266 xbuffer.Extract(hashresult[0]);
267
268 HASHCRC32 hashcrc32;
269
270 hashcrc32.ResetResult();
271 hashcrc32.Do(buffer, size);
272
273 hashresult[1] = hashcrc32.GetResultCRC32();
274
275 if(hashresult[0] != hashresult[1])
276 {
277 //XTRACE_PRINTCOLOR(1, __L("Write received bad CRC!! %08X <> %08X"), hashresult, hashcrc32.GetResultCRC32());
278 bw = 0;
279 }
280 }
281 else
282 {
283 //XTRACE_PRINTCOLOR(1, __L("Write received NOT read"));
284 bw = 0;
285 }
286 }
287 else
288 {
289 //XTRACE_PRINTCOLOR(1, __L("Do not write"));
290 bw = 0;
291 }
292
293 if(!bw)
294 {
295 diostream->Close();
296 diostream->Open();
297 diostream->WaitToConnected(5);
298 }
299
300 return bw;
301}

324 old_port = portremote;

327 portremote = port;

443void DIOSTREAMUDPACKNOWLEDGE::ThreadRunFunction(void* param)
444{
445 DIOSTREAMUDPACKNOWLEDGE* diostreamudpacknowledge = (DIOSTREAMUDPACKNOWLEDGE*)param;
446 if(!diostreamudpacknowledge) return;
447
448 DIOSTREAMUDP* diostreamudp = diostreamudpacknowledge->diostream;
449 if(!diostreamudp) return;
450
451 XBUFFER data;
452 XSTRING address;
453 XWORD port;
454
455 if(diostreamudp->ReadDatagram(address, port, data))
456 {
457 HASHCRC32 hashcrc32;
458
459 hashcrc32.ResetResult();
460 if(hashcrc32.Do(data.Get(), data.GetSize()))
461 {
462 XBUFFER result;
463 XDWORD hashresult = hashcrc32.GetResultCRC32();
464
465 result.Add(hashresult);
466
467 diostreamudp->WriteDatagram(address, port, result);
468 if(diostreamudp->WaitToWriteDatagramsEmpty(diostreamudpacknowledge->timeout))
469 {
470 diostreamudpacknowledge->AddDatagram(false, address, port, data.Get(), data.GetSize());
471 }
472 }
473 }
474}
