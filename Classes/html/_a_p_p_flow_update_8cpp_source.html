432bool APPFLOWUPDATE::EventManager(APPFLOWUPDATE_EVENTOPERATION eventoperation, XSUBJECT* xsubject, XOBSERVER* observer)
433{
434 XDWORD events[] = { DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE ,
435 DIOAPPLICATIONUPDATE_XEVENT_TYPE_CHECKVERSIONAVAILABLE ,
436 DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE ,
437 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE ,
438 DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES ,
439 DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILE ,
440 DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILES ,
441 DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES ,
442 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILE ,
443 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEEXECFILE ,
444 DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES ,
445 DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END ,
446 DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILE ,
447 DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES ,
448 };
449
450
451 for(XDWORD c=0; c< sizeof(events)/sizeof(XDWORD); c++)
452 {
453 switch(eventoperation)
454 {
455 case APPFLOWUPDATE_EVENTOPERATION_REGISTER : RegisterEvent(events[c]); break;
456 case APPFLOWUPDATE_EVENTOPERATION_DEREGISTER : DeRegisterEvent(events[c]); break;
457 case APPFLOWUPDATE_EVENTOPERATION_SUBSCRIBE : SubscribeEvent(events[c] , xsubject, observer); break;
458 case APPFLOWUPDATE_EVENTOPERATION_UNSUBSCRIBE : UnSubscribeEvent(events[c] , xsubject, observer); break;
459 default : return false;
460 }
461 }
462
463 return true;
464}

476bool APPFLOWUPDATE::Do()
477{
478 DIOAPPLICATIONUPDATE_VERSIONDATA applicationversiondata;
479 XPATH xpathupdate;
480 XPATH xpathbackup;
481 XSTRING string;
482 XSTRING stringresult;
483 bool status = false;
484 XSTRING dir_mask;
485 DIOAPPLICATIONUPDATE_ERROR error;
486
487 dir_mask = DIOAPPLICATIONUPDATE_RESTOREDIR_ID;
488 dir_mask += DIOAPPLICATIONUPDATE_RESTOREDIR_MASK;
489
490 if(!dioappupdate) return false;
491
492 if(update_cfg.GetDolog()) EventManager(APPFLOWUPDATE_EVENTOPERATION_SUBSCRIBE, dioappupdate, this);
493
494 // --------------------------------------------------------------------------------------------
495 // Get last update available
496 if(dioappupdate->GetVersionAvailable((*update_cfg.GetURL()), update_cfg.Get_Port(), applicationversiondata, 20, error))
497 {
498 // Check last update available
499 if(dioappupdate->IsNewVersion(applicationversiondata))
500 {
501 // Have new version available
502
503 // Download new version to update available to subdir root/update[x.x.x] Update Version
504 xpathupdate = update_cfg.GetPathRootApp()->Get();
505 xpathupdate.Slash_Add();
506 xpathupdate.AddFormat(dir_mask.Get() , applicationversiondata.GetVersion()
507 , applicationversiondata.GetSubVersion()
508 , applicationversiondata.GetSubVersionError());
509
510 status = dioappupdate->Download((*update_cfg.GetURL()), update_cfg.Get_Port(), xpathupdate, applicationversiondata, 60, error);
511
512 //-----------------------------------------------------------------------
513 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES);
514
515 xevent.SetVersionData(&applicationversiondata);
516 xevent.SetError(error);
517
518 PostEvent(&xevent);
519 //-----------------------------------------------------------------------
520
521 if(status)
522 {
523 // New version update downloaded
524
525
526 //-----------------------------------------------------------------------
527 // Eliminate resources APPLICATION
528 { DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES);
529 xevent.SetVersionData(&applicationversiondata);
530 xevent.SetError(error);
531
532 PostEvent(&xevent);
533 }
534 //-----------------------------------------------------------------------
535
536
537 // Actual files to be updated available to subdir root/update[x.x.x] Actual version
538 xpathbackup = update_cfg.GetPathRootApp()->Get();;
539 xpathbackup.Slash_Add();
540
541 xpathbackup.AddFormat(dir_mask.Get() , update_cfg.GetAppVersion()->GetVersion()
542 , update_cfg.GetAppVersion()->GetSubVersion()
543 , update_cfg.GetAppVersion()->GetSubVersionError());
544
545 // Change actual files for the new version
546 status = dioappupdate->ChangeFilesFromDownload(applicationversiondata, xpathupdate, xpathbackup, error);
547
548 //-----------------------------------------------------------------------
549 { DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES);
550
551 xevent.SetVersionData(&applicationversiondata);
552 xevent.SetError(error);
553
554 PostEvent(&xevent);
555 }
556 //-----------------------------------------------------------------------
557
558 if(!status)
559 {
560 status = dioappupdate->Restore(xpathbackup, error);
561
562 //-----------------------------------------------------------------------
563 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES);
564
565 xevent.SetVersionData(&applicationversiondata);
566 xevent.SetError(error);
567
568 PostEvent(&xevent);
569 //-----------------------------------------------------------------------
570 }
571
572 if(status)
573 {
574 if(applicationversiondata.SystemMustBeInit())
575 {
576 GEN_XSYSTEM.ShutDown(XSYSTEM_CHANGESTATUSTYPE_REBOOT);
577 }
578 else
579 {
580 // Exit & Restart of Application
581
582 DIOAPPLICATIONUPDATE_XEVENT xevent(this, DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END);
583
584 xevent.SetVersionData(&applicationversiondata);
585 xevent.SetError(error);
586
587 PostEvent(&xevent);
588
589 dioappupdate->RestartApplication(error);
590 }
591 }
592 }
593 }
594 }
595
596 if(update_cfg.GetDolog()) EventManager(APPFLOWUPDATE_EVENTOPERATION_UNSUBSCRIBE, dioappupdate, this);
597
598 return status;
599}

763bool APPFLOWUPDATE::LogOfEvent(DIOAPPLICATIONUPDATE_XEVENT* event)
764{
765 DIOAPPLICATIONUPDATE_FILE* updatefile = event->GetActualFileToUpdate();
766 XSTRING string;
767
768 switch(event->GetEventType())
769 {
770 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_GETVERSIONAVAILABLE : if(event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR)
771 {
772 string.Format( __L("Update Version available %d.%d.%d")
773 , event->GetVersionData()->GetVersion()
774 , event->GetVersionData()->GetSubVersion()
775 , event->GetVersionData()->GetSubVersionError());
776 }
777 else
778 {
779 string.Format( __L("No version available to update"));
780 }
781 break;
782
783 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_CHECKVERSIONAVAILABLE : string.Format( __L("Version %d.%d.%d is ")
784 , event->GetVersionData()->GetVersion()
785 , event->GetVersionData()->GetSubVersion()
786 , event->GetVersionData()->GetSubVersionError());
787
788 switch(event->GetVersionComparation())
789 {
790 case -1 : string.Add(__L("old")); break;
791 case 0 : string.Add(__L("equal")); break;
792 case 1 : string.Add(__L("new")); break;
793 }
794
795 if(event->GetVersionComparation() <=0) string.AddFormat(__L(". It will not be updated"));
796
797 break;
798
799
800 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILE : if(updatefile) string.Format( __L("Download file %s (%dk).")
801 , updatefile->GetXPathFile()->Get()
802 , updatefile->GetSize()/1024);
803
804 break;
805
806 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UNZIPFILE : if(updatefile) string.Format( __L("Unzip file %s (%dk) -> %s.")
807 , updatefile->GetXPathFile()->Get()
808 , updatefile->GetSize()/1024
809 , event->GetActualUnzipFileName()->Get());
810 break;
811
812 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_DOWNLOADFILES : string.Format(__L("Downloads file %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
813 break;
814
815 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILE : if(updatefile) string.Format( __L("Backup original file %s (%dk).")
816 , updatefile->GetXPathFile()->Get()
817 , updatefile->GetSize()/1024);
818 break;
819
820 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_BACKUPORIGINALFILES : string.Format(__L("Backup Original file %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
821 break;
822
823 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_RELEASERESOURCES : string.Format(__L("Application release blocked resources"));
824 break;
825
826
827 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILE : if(updatefile) string.Format( __L("Copy update file %s (%dk).")
828 , updatefile->GetXPathFile()->Get()
829 , updatefile->GetSize()/1024);
830 break;
831
832 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEEXECFILE : if(updatefile) string.Format( __L("Notify update EXEC file %s (%dk).")
833 , updatefile->GetXPathFile()->Get()
834 , updatefile->GetSize()/1024);
835 break;
836
837 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_UPDATEFILES : string.Format(__L("Update files %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
838 break;
839
840
841 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_APP_END : string.Format(__L("Terminate application"));
842 break;
843
844 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILE : if(updatefile) string.Format( __L("Restore file %s from backup (%dk).")
845 , updatefile->GetXPathFile()->Get()
846 , updatefile->GetSize()/1024);
847 break;
848
849 case DIOAPPLICATIONUPDATE_XEVENT_TYPE_RESTOREUPDATEFILES : string.Format(__L("Restore update files %s."), (event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR?__L("Ok."):__L("ERROR!")));
850 break;
851 }
852
853 if(string.GetSize())
854 {
855 XLOG::GetInstance().AddEntry((event->GetError() == DIOAPPLICATIONUPDATE_NOT_ERROR)?XLOGLEVEL_INFO:XLOGLEVEL_ERROR, DIOAPPLICATIONUPDATE_LOG_SECTIONID_APPFLOWUPDATE, false, string.Get());
856 return true;
857 }
858
859 return false;
860}

903bool APPFLOWUPDATE::CreateTaskUpdate(bool doajust)
904{
905 if(!cfg) return false;
906
907 XSCHEDULERTASK* xtask = new XSCHEDULERTASK(xscheduler);
908 if(!xtask) return false;
909
910 bool toactive = false;
911
912 XDATETIME* GEN_XFACTORY_CREATE(start, CreateDateTime())
913 XDATETIME* GEN_XFACTORY_CREATE(end, CreateDateTime())
914 if(start && end)
915 {
916 if(!cfg->ApplicationUpdate_GetCheckTime()->IsEmpty())
917 {
918 start->Read();
919
920 start->GetDateTimeFromString((*cfg->ApplicationUpdate_GetCheckTime()), XDATETIME_FORMAT_ADDTIME);
921 start->SetSeconds(0);
922 start->SetMilliSeconds(0);
923
924 start->AddDays(1);
925
926 end->Set(start);
927 end->AddSeconds(5);
928
929 xtask->SetTimeLimits(start, end);
930
931 toactive = true;
932 }
933 else
934 {
935 int minutescadence = cfg->ApplicationUpdate_GetCheckCadence();
936 if(minutescadence)
937 {
938 start->Read();
939
940 if(doajust)
941 {
942 int minutes = ((start->GetMinutes() / minutescadence) + ((start->GetMinutes() % minutescadence)?1:0)) * minutescadence;
943 if(minutes == start->GetMinutes()) minutes += minutescadence;
944 if(minutes >= 60)
945 {
946 start->AddHours(1);
947 minutes = 0;
948 }
949
950 start->SetMinutes(minutes);
951
952 } else start->AddMinutes(minutescadence);
953
954 start->SetSeconds(0);
955 start->SetMilliSeconds(0);
956
957 xtask->SetTimeLimits(start);
958
959 toactive = true;
960 }
961 }
962 }
963
964 if(toactive)
965 {
966 xtask->SetID(APPFLOWUPDATE_TASKID_CHECKUPDATEAPPLICATION);
967 xtask->SetNCycles(1);
968 xtask->SetIsActive(true);
969
970 xscheduler->Task_Add(xtask);
971 }
972 else delete xtask;
973
974 GEN_XFACTORY.DeleteDateTime(start);
975 GEN_XFACTORY.DeleteDateTime(end);
976
977 return true;
978}
