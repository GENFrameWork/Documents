534 int capy = GetHeight();

541

573 bmpinfo.bmiHeader.biClrUsed = 0;

876LRESULT CALLBACK GRPWINDOWSSCREEN::BaseWndProc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam)
877{
878 switch(msg)
879 {
880 case WM_SYSCOMMAND : switch(wparam)
881 {
882 case SC_CLOSE : break;
883 case SC_SCREENSAVE :
884 case SC_MONITORPOWER : return 0;
885 }
886 break;
887
888 case WM_ENDSESSION : break;
889
890 case WM_CLOSE : { GRPWINDOWSSCREEN* screen = GRPWINDOWSSCREEN::GetListScreens()->Get(hwnd);
891 if(screen)
892 {
893 if(screen->IsBlockClose())
894 {
895 return 0;
896 }
897 }
898 }
899 break;
900
901 case WM_DESTROY : { int a=0;
902 a++;
903 }
904 break;
905
906 case WM_POWERBROADCAST : //Computer is suspending
907 if(wparam == PBT_APMSUSPEND)
908 {
909 #ifdef GRP_MAINPROCCTRL_ACTIVE
910 if(mainprocwindows.GetAppMain())
911 {
912 if(mainprocwindows.GetAppMain()->GetApplication()) mainprocwindows.GetAppMain()->GetApplication()->SetSystemChangeStatus(XSYSTEM_CHANGESTATUSTYPE_SUSPEND);
913 }
914 #endif
915 }
916
917 break;
918
919 case WM_QUERYENDSESSION : //Computer is shutting down
920 if(lparam == 0)
921 {
922 #ifdef GRP_MAINPROCCTRL_ACTIVE
923 if(mainprocwindows.GetAppMain())
924 {
925 if(mainprocwindows.GetAppMain()->GetApplication()) mainprocwindows.GetAppMain()->GetApplication()->SetSystemChangeStatus(XSYSTEM_CHANGESTATUSTYPE_REBOOT);
926 }
927 #endif
928 }
929
930 //User is logging off
931 if((lparam & ENDSESSION_LOGOFF) == ENDSESSION_LOGOFF)
932 {
933 #ifdef GRP_MAINPROCCTRL_ACTIVE
934 if(mainprocwindows.GetAppMain())
935 {
936 if(mainprocwindows.GetAppMain()->GetApplication()) mainprocwindows.GetAppMain()->GetApplication()->SetSystemChangeStatus(XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGOFF);
937 }
938 #endif
939 }
940
941 break;
942 }
943
944 return DefWindowProc(hwnd, msg, wparam, lparam);
945}
