148bool DIOSTREAMTCPIPSERVER::WaitToConnected(int timeout)
149{
150 if(!xtimerconnection)
151 {
152 return false;
153 }
154
155 DIOSTREAMTCPIPCONFIG* config = (DIOSTREAMTCPIPCONFIG*)GetConfig();
156 bool status = false;
157
158 if(!config)
159 {
160 return false;
161 }
162
163 xtimerconnection->Reset();
164
165 while(1)
166 {
167 if(config->GetHandleMultiServer() > 0)
168 {
169 status = true;
170 break;
171 }
172
173 if(GetStatus() == DIOSTREAMSTATUS_DISCONNECTED)
174 {
175 break;
176 }
177
178 if((int)xtimerconnection->GetMeasureSeconds() >= timeout)
179 {
180 break;
181 }
182
183 Wait(DIOSTREAM_TIMEINWAITFUNCTIONS);
184 }
185
186 return status;
187}
