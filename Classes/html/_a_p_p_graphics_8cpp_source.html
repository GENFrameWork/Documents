120bool APPGRAPHICS::Ini(XVECTOR<XSTRING*>* execparams)
121{
122 RegisterEvent(GRPXEVENT_TYPE_SCREEN_CREATING);
123 RegisterEvent(GRPXEVENT_TYPE_SCREEN_CREATED);
124 RegisterEvent(GRPXEVENT_TYPE_SCREEN_DELETING);
125 RegisterEvent(GRPXEVENT_TYPE_SCREEN_DELETED);
126
127 if(!APPBASE::Ini(execparams)) return false;
128
129 if(GetExitType() != APPBASE_EXITTYPE_UNKNOWN) return false;
130
131 if(initoptions & APPGRAPHICS_INIOPTION_CREATEMAINSCREEN)
132 {
133 if(!CreateMainScreenProcess((initoptions & APPGRAPHICS_INIOPTION_SHOWMAINSCREEN)?true:false)) return false;
134 }
135

213bool APPGRAPHICS::CreateMainScreenProcess(bool show)
214{
215 if(mainscreen) return false;
216
217 mainscreen = GEN_GRPFACTORY.CreateScreen();
218 if(!mainscreen)
219 {
220 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("Could not initialize the main screen ..."));
221 return false;
222 }
223
224 #ifdef GRP_OPENGL_ACTIVE
225 maincontext = GRPFACTORY::GetInstance().CreateContext();
226 if(!maincontext)
227 {
228 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("Could not initialize the Graphics Context ..."));
229 return false;
230 }
231 #endif
232
233 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATING , this);
234 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
235
236 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETING , this);
237 SubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETED , this);
238
239 #ifndef ANDROID
240
241 GRPXEVENT grpeventini(this, GRPXEVENT_TYPE_SCREEN_CREATING);
242 grpeventini.SetScreen(mainscreen);
243 PostEvent(&grpeventini);
244
245 bool status = !grpeventini.GetError();
246
247 if(status)
248 {
249 GRPVIEWPORT* viewport = mainscreen->GetViewport(0);
250 if(!viewport)
251 {
252 if(!mainscreen->CreateViewport(GRPVIEWPORT_ID_MAIN , 0.0f, 0.0f, (float)mainscreen->GetWidth() , (float)mainscreen->GetHeight(), 0, 0, mainscreen->GetWidth(), mainscreen->GetHeight()))
253 {
254 return false;
255 }
256 }
257
258 status = mainscreen->Create(show);
259 }
260
261 if(status)
262 {
263 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("Screen Created."));
264
265 GRPXEVENT grpeventend(this, GRPXEVENT_TYPE_SCREEN_CREATED);
266 grpeventend.SetScreen(mainscreen);
267 grpeventend.SetError(status);
268 PostEvent(&grpeventend);
269 }
270 else
271 {
272 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("Error to creating screen ..."));
273 }
274
275 if(!status) return false;
276
277 #ifdef INP_ACTIVE
278 if(initoptions & APPGRAPHICS_INIOPTION_INPUT)
279 {
280 keyboard = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_KEYBOARD);
281 if(!keyboard)
282 {
283 keyboard = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_KEYBOARD, mainscreen);
284 if(keyboard) GEN_INPMANAGER.AddDevice(keyboard);
285 }
286
287 mouse = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_MOUSE);
288 if(!mouse)
289 {
290 mouse = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_MOUSE, mainscreen);
291 if(mouse) GEN_INPMANAGER.AddDevice(mouse);
292 }
293
294 touchscreen = GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_TOUCHSCREEN);
295 if(!touchscreen)
296 {
297 touchscreen = INPFACTORY::GetInstance().CreateDevice(INPDEVICE_TYPE_TOUCHSCREEN, mainscreen);
298 if(touchscreen) GEN_INPMANAGER.AddDevice(touchscreen);
299 }
300
301 }
302 #endif
303
304 #endif
305
306 return true;
307}

319bool APPGRAPHICS::DeleteMainScreenProcess()
320{
321 if(!mainscreen)
322 {
323 return false;
324 }
325
326 GRPXEVENT grpeventini(this, GRPXEVENT_TYPE_SCREEN_DELETING);
327 grpeventini.SetScreen(mainscreen);
328 PostEvent(&grpeventini);
329
330 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATING, this);
331 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
332
333 if(mainscreen)
334 {
335 mainscreen->Delete();
336 GRPFACTORY::GetInstance().DeleteScreen(mainscreen);
337 mainscreen = NULL;
338 }
339
340 GRPXEVENT grpeventend(this, GRPXEVENT_TYPE_SCREEN_DELETED);
341 grpeventini.SetScreen(mainscreen);
342 PostEvent(&grpeventend);
343
344 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_DELETED, this);
345 UnSubscribeEvent(GRPXEVENT_TYPE_SCREEN_CREATED , this);
346
347 return true;
348}
