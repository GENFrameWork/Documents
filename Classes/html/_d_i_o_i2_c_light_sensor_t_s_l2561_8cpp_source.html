171bool DIOI2CLIGHTSENSORTSL2561::Get(XWORD& fullspectrum, XWORD& infrared, XDWORD& lux, bool autogain)
172{
173 fullspectrum = 0;
174 infrared = 0;
175 lux = 0;
176
177 if(!diostream) return false;
178
179 XWORD fs, ir, hi, lo;
180 bool status = true;
181
182 if(!autogain)
183 {
184 // autogain not requested. Executing a single sensor read
185
186 status = GetData(fullspectrum, infrared);
187 lux = CalculateLux(fullspectrum, infrared);
188
189 return status;
190 }
191
192 // autogain requested
193 switch(integrationtime)
194 {
195 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_13MS;
196 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_13MS;
197 break;
198
199 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_101MS;
200 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_101MS;
201 break;
202
203 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_402MS;
204 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_402MS;
205 break;
206
207 default : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_402MS;
208 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_402MS;
209 break;
210 }
211
212 // try to adjust the gain
213 status = GetData(fs, ir);
214 if(!status) return false;
215
216 if((fs < lo) && (gain == DIOI2CLIGHTSENSORTSL2561GAIN_1X))
217 {
218 // light too low with this gain
219 // raise the gain and redo the reading
220 SetGain(DIOI2CLIGHTSENSORTSL2561GAIN_16X);
221
222 status = GetData(fs, ir);
223 if(!status) return false;
224
225 // now consider the reading valid after being adjusted
226 fullspectrum = fs;
227 infrared = ir;
228
229 lux = CalculateLux(fullspectrum, infrared);
230
231 return true;
232 }
233
234
235 if((fs > hi) && (gain == DIOI2CLIGHTSENSORTSL2561GAIN_16X))
236 {
237 // light too high with this gain
238 // lower the gain and redo the reading
239 SetGain(DIOI2CLIGHTSENSORTSL2561GAIN_1X);
240
241 status = GetData(fs, ir);
242 if(!status) return false;
243
244 // now consider the reading valid after being adjusted
245 fullspectrum = fs;
246 infrared = ir;
247 lux = CalculateLux(fullspectrum, infrared);
248
249 return true;
250 }
251
252
253 // the reading was valid without gain adjustment (or chip limits encountered!)
254 fullspectrum = fs;
255 infrared = ir;
256
257 lux = CalculateLux(fullspectrum, infrared);
258
259 return true;
260}

355bool DIOI2CLIGHTSENSORTSL2561::ReadData16(XBYTE reg, XWORD& value, int timeout)
356{
357 XBUFFER xbuffer;
358
359 value = 0;
360
361 xbuffer.Add((XBYTE)reg);
362 if(!diostream->Write(xbuffer)) return false;
363 if(!diostream->WaitToFlushOutXBuffer(timeout)) return false;
364
365 xbuffer.Resize(2);
366 if(!diostream->WaitToFilledReadingBuffer(2, timeout)) return false;
367 if(!diostream->Read(xbuffer)) return false;
368 if(xbuffer.GetSize() < 2) return false;
369
370 value = xbuffer.GetByte(1);
371 value <<= 8;
372 value |= xbuffer.GetByte(0);
373
374 return true;
375}

390bool DIOI2CLIGHTSENSORTSL2561::GetData(XWORD& fullspectrum, XWORD& infrared)
391{
392 fullspectrum = 0;
393 infrared = 0;
394
395 if(!Activate()) return false;
396
397 bool status;
398
399 // wait for the internal ADC to complete conversion
400 switch(integrationtime)
401 {
402 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : GEN_XSLEEP.MilliSeconds(20); break;
403 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : GEN_XSLEEP.MilliSeconds(150); break;
404 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : GEN_XSLEEP.MilliSeconds(450); break;
405 }
406
407 status = ReadData16((DIOI2CLIGHTSENSORTSL2561_COMMAND_BIT | DIOI2CLIGHTSENSORTSL2561_WORD_BIT | DIOI2CLIGHTSENSORTSL2561REGISTER_CHAN0_LOW), fullspectrum, timeout);
408 if(status) status = ReadData16((DIOI2CLIGHTSENSORTSL2561_COMMAND_BIT | DIOI2CLIGHTSENSORTSL2561_WORD_BIT | DIOI2CLIGHTSENSORTSL2561REGISTER_CHAN1_LOW), infrared , timeout);
409
410 if(status) status = Deactivate();
411
412 return status;

458XDWORD DIOI2CLIGHTSENSORTSL2561::CalculateLux(XWORD broadband, XWORD inflared)
459{
460 XDWORD chscale;
461 XDWORD channel1;
462 XDWORD channel0;
463 XDWORD clipthreshold;
464 XDWORD ratio1 = 0;
465 int ratio;
466 XDWORD b, m;
467 int temp;
468 XDWORD lux;
469
470 // Make sure the sensor isn't saturated!
471 switch(integrationtime)
472 {
473 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_13MS; break;
474 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_101MS; break;
475 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_402MS; break;
476 default : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_402MS; break;
477 }
478
479 // return 0 lux if the sensor is saturated
480 if((broadband > clipthreshold) || (inflared > clipthreshold)) return 0;
481
482
483 // get the correct scale depending on the intergration time
484 switch(integrationtime)
485 {
486 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : chscale = DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE_TINT0; break;
487 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : chscale = DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE_TINT1; break;
488 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : chscale = (1 << DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE); break;
489 default : chscale = (1 << DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE); break; // No scaling ... integration time = 402ms
490 }
491
492 // scale for gain (1x or 16x)
493 if (!gain) chscale = chscale << 4;
494
495 // scale the channel values
496 channel0 = (broadband * chscale) >> DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE;
497 channel1 = (inflared * chscale) >> DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE;
498
499 // find the ratio of the channel values (Channel1/Channel0)
500 if (channel0 != 0) ratio1 = (channel1 << (DIOI2CLIGHTSENSORTSL2561_LUX_RATIOSCALE+1)) / channel0;
501
502 // round the ratio value
503 ratio = (ratio1 + 1) >> 1;
504
505 #ifdef TSL2561_PACKAGE_CS
506 if ((ratio >= 0) && (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K1C))
507 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B1C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M1C; }
508 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K2C)
509 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B2C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M2C; }
510 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K3C)
511 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B3C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M3C; }
512 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K4C)
513 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B4C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M4C; }
514 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K5C)
515 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B5C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M5C; }
516 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K6C)
517 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B6C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M6C; }
518 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K7C)
519 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B7C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M7C; }
520 else if (ratio > DIOI2CLIGHTSENSORTSL2561_LUX_K8C)
521 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B8C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M8C; }
522 #else
523 if ((ratio >= 0) && (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K1T))
524 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B1T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M1T; }
525 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K2T)
526 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B2T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M2T; }
527 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K3T)
528 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B3T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M3T; }
529 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K4T)
530 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B4T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M4T; }
531 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K5T)
532 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B5T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M5T; }
533 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K6T)
534 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B6T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M6T; }
535 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K7T)
536 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B7T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M7T; }
537 else if (ratio > DIOI2CLIGHTSENSORTSL2561_LUX_K8T)
538 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B8T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M8T; }
539 #endif
540
541 temp = ((channel0 * b) - (channel1 * m));
542
543 // do not allow negative lux value
544 if (temp < 0) temp = 0;
545
546 // round lsb (2^(LUX_SCALE-1))
547 temp += (1 << (DIOI2CLIGHTSENSORTSL2561_LUX_LUXSCALE-1));
548
549 // strip off fractional portion
550 lux = (temp >> DIOI2CLIGHTSENSORTSL2561_LUX_LUXSCALE);
551
552 // Signal I2C had no errors
553 return lux;
554}
