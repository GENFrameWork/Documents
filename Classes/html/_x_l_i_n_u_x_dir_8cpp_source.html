401bool XLINUXDIR::NextSearch(XDIRELEMENT* searchelement)
402{
403 if(!searchelement) return false;
404
405 DIR* dir = (DIR*)searchelement->GetHandle();
406 struct dirent* entry = (struct dirent*)searchelement->GetFindFileData();
407 bool status = false;
408 XSTRING name;
409 XBUFFER patternsearch;
410
411 while(entry != NULL)
412 {
413 name.ConvertFromUTF8((XBYTE*)entry->d_name, strlen(entry->d_name) /*_MAXPATH*/ );
414
415 if((name.Compare(__L("."))) && (name.Compare(__L(".."))))
416 {
417 searchelement->GetPatternSearch()->ConvertToUTF8(patternsearch);
418 if(!fnmatch((char*)patternsearch.Get(), entry->d_name,0))
419 {
420 searchelement->SetType(XDIRELEMENTTYPE_NONE);
421 searchelement->GetNameFile()->Set(name.Get());
422
423 XPATH xpath;
424
425 xpath = searchelement->GetPathSearch()->Get();
426 xpath.Slash_Add();
427 xpath += searchelement->GetNameFile()->Get();
428
429 TypeOfEntry(xpath.Get(), searchelement);
430
431 status = true;
432 }
433 }
434
435 entry = readdir(dir);
436 searchelement->SetFindFileData((void*)entry);
437
438 if(status) break;
439 }
440
441 if(!status) closedir (dir);
442
443 return status;
444}
