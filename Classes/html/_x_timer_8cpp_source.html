418bool XTIMER::GetMeasureString(XSTRING& measure,bool large)
419{
420 XQWORD allseconds = GetMeasureSeconds();
421
422 XDWORD years = XDATETIME_SECONDSYEARS(allseconds);
423 XDWORD months = XDATETIME_SECONDSMONTHS(allseconds);

429

433
434 for(int c=0;c<6;c++)
435 {
436 string.Empty();
437
438 switch(c)
439 {
440 case 0: if(!years) break;
441 if(years>1)
442 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_YEARS), years);
443 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_YEAR));
444 break;
445
446 case 1: if(!months) break;
447 if(months>1)
448 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_MONTHS), months);
449 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_MONTH));
450 break;
451
452 case 2: if(!days) break;
453 if(days>1)
454 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_DAYS), days);
455 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_DAY));
456 break;
457
458 case 3: if(large)
459 {
460 if(!hours) break;
461 if(hours>1)
462 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_HOURS), hours);
463 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_HOUR));
464 }
465 else
466 {
467 string.Format(__L("%02d"), hours);
468 }
469 break;
470
471 case 4: if(large)
472 {
473 if(!minutes) break;
474 if(minutes>1)
475 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_MINUTES), minutes);
476 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_MINUTE));
477 }
478 else

489 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_SECOND));
490 }
491 else
492 {
493 string.Format(__L("%02d"), seconds);
494 }
495 break;
496
497 }
498
499 if((large)||(c<=3))
500 {
501 if(!string.IsEmpty())
502 {
503 if(!measure.IsEmpty()) measure += ", ";
504 measure += string;
505 }
506 }
507 else
508 {
509 if(!string.IsEmpty())
510 {
511 if(!measure.IsEmpty()) measure += ":";
512 measure += string;
513 }
514 }
515 }
516
517 if(measure.IsEmpty())
518 {
519 measure.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ZERO_SECONDS));
520 return false;
521 }
522
523 return true;
524}
