427bool XTIMER::GetMeasureString(XSTRING& measure,bool large)
428{
429 XQWORD allseconds = GetMeasureSeconds();
430
431 XDWORD years = XDATETIME_SECONDSYEARS(allseconds);
432 XDWORD months = XDATETIME_SECONDSMONTHS(allseconds);
433 XDWORD days = XDATETIME_SECONDSDAYS(allseconds);
434
435 XDWORD hours = XDATETIME_SECONDSHOURS(allseconds);
436 XDWORD minutes = XDATETIME_SECONDSMINUTES(allseconds);
437 XDWORD seconds = XDATETIME_SECONDSSECONDS(allseconds);
438
439 measure.Empty();
440
441 XSTRING string;
442
443 for(int c=0;c<6;c++)
444 {
445 string.Empty();
446
447 switch(c)
448 {
449 case 0: if(!years) break;
450 if(years>1)
451 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_YEARS), years);
452 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_YEAR));
453 break;
454
455 case 1: if(!months) break;
456 if(months>1)
457 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_MONTHS), months);
458 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_MONTH));
459 break;
460
461 case 2: if(!days) break;
462 if(days>1)
463 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_DAYS), days);
464 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_DAY));
465 break;
466
467 case 3: if(large)
468 {
469 if(!hours) break;
470 if(hours>1)
471 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_HOURS), hours);
472 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_HOUR));
473 }
474 else
475 {
476 string.Format(__L("%02d"), hours);
477 }
478 break;
479
480 case 4: if(large)
481 {
482 if(!minutes) break;
483 if(minutes>1)
484 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_MINUTES), minutes);
485 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_MINUTE));
486 }
487 else
488 {
489 string.Format(__L("%02d"), minutes);
490 }
491 break;
492
493 case 5: if(large)
494 {
495 if(!seconds) break;
496 if(seconds>1)
497 string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_XX_SECONDS), seconds);
498 else string.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ONE_SECOND));
499 }
500 else
501 {
502 string.Format(__L("%02d"), seconds);
503 }
504 break;
505
506 }
507
508 if((large)||(c<=3))
509 {
510 if(!string.IsEmpty())
511 {
512 if(!measure.IsEmpty()) measure += ", ";
513 measure += string;
514 }
515 }
516 else
517 {
518 if(!string.IsEmpty())
519 {
520 if(!measure.IsEmpty()) measure += ":";
521 measure += string;
522 }
523 }
524 }
525
526 if(measure.IsEmpty())
527 {
528 measure.Format(XT_L(XTRANSLATION_GEN_ID_XTIMER_ZERO_SECONDS));
529 return false;
530 }
531
532 return true;
533}
