244 Error((char*)mysql_error(conn));

246 }
247
248 this->transactionstarted=false;
249
250 return true;
251 }
252 }

431DB_SQL_DATABASE_ENCODING MYSQL_DATABASE::GetDatabaseEncoding()
432{
433 MYSQL* conn = static_cast<MYSQL_CONNECTION*>(this->connection)->connection;
434 if(!conn) return DB_SQL_DATABASE_ENCODING_UNKNOWN;
435
436 const char* name = mysql_character_set_name(conn);
437
438 XSTRING s;
439 s.Set(name);
440
441 if (s.Compare(__L("ucs2")) == 0) return DB_SQL_DATABASE_ENCODING_UNKNOWN;
442 if (s.Compare(__L("utf8")) == 0) return DB_SQL_DATABASE_ENCODING_UTF8;
443
444 return DB_SQL_DATABASE_ENCODING_UNKNOWN;
445}
