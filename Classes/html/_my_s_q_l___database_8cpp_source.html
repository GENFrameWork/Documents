424DB_SQL_DATABASE_ENCODING MYSQL_DATABASE::GetDatabaseEncoding()
425{
426 MYSQL* conn = static_cast<MYSQL_CONNECTION*>(this->connection)->connection;
427 if(!conn) return DB_SQL_DATABASE_ENCODING_UNKNOWN;
428
429 const char* name = mysql_character_set_name(conn);
430
431 XSTRING s;
432 s.Set(name);
433
434 if (s.Compare(__L("ucs2")) == 0) return DB_SQL_DATABASE_ENCODING_UNKNOWN;
435 if (s.Compare(__L("utf8")) == 0) return DB_SQL_DATABASE_ENCODING_UTF8;
436
437 return DB_SQL_DATABASE_ENCODING_UNKNOWN;
438}
