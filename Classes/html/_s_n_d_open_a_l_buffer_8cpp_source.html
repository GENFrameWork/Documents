180XDWORD SNDOPENALBUFFER::GenerateNote(XDWORD frequency, XDWORD duration, XDWORD samplerate)
181{
182 if(!frequency || !duration)
183 {
184 return false;
185 }
186
187 XDWORD size = (XDWORD)((duration/1000) * samplerate);
188
189 xbuffer.SetLocalHardwareUseLittleEndian(false);
190
191 for(XDWORD c=0; c<size; c++)
192 {
193 XWORD data = (short)(32760 * sin(2 * PI * c * frequency/samplerate));
194 xbuffer.Add((XWORD)data);
195 }
196
197 return size;
198}
