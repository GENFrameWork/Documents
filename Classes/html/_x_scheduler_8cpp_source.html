465bool XSCHEDULERTASK::CheckCondition(XDATETIME* xdatetimeactual, XTIMER* xtimeractual)
466{
467 if(!xdatetimeactual) return false;
468
469 XQWORD timeactualsec = xdatetimeactual->GetSeconsFromDate();
470
471 if(xdatetimelimitstart.IsValidDate())
472 {
473 if(timeactualsec < xdatetimelimitstart.GetSeconsFromDate())
474 {
475 isinvalidtimelimit = false;
476
477 xtimeractual->Reset();
478 return false;
479 }
480 else isinvalidtimelimit = true;
481 }
482
483 if(xdatetimelimitend.IsValidDate())
484 {
485 if(timeactualsec > xdatetimelimitend.GetSeconsFromDate())
486 {
487 isinvalidtimelimit = false;
488 return false;
489
490 } else isinvalidtimelimit = true;
491 }
492
493 if(conditiondayweekmask != XSCHEDULER_DAYWEEK_NONE)
494 {
495 if(!(GetDayOfWeekMask(xdatetimeactual) & conditiondayweekmask))
496 {
497 xtimeractual->Reset();
498 return false;
499 }
500 }
501
502 if(!ncyclesactual) return false;
503
504 bool checkcadence = true;
505 if(isstartimmediatelycycles)
506 {
507 if(!ncyclesmade) checkcadence = false;
508 }
509
510 if(checkcadence)
511 {
512 if(xdatetimecadence.IsValidDate())
513 {
514 if(xtimeractual->GetMeasureSeconds() < xdatetimecadence.GetSeconsFromDate()) return false;
515 }
516 }
517
518 if(ncyclesactual>0) ncyclesactual--;
519
520 xtimeractual->Reset();
521
522 ncyclesmade++;
523
524 return true;
525}

539XBYTE XSCHEDULERTASK::GetDayOfWeekMask(XDATETIME* xtimeactual)
540{
541 if(!xtimeactual) return XSCHEDULER_DAYWEEK_NONE;
542
543 XBYTE mask = XSCHEDULER_DAYWEEK_NONE;
544
545 switch(xtimeactual->GetDayOfWeek())
546 {
547 case XDATETIME_DAYWEEK_SUNDAY : mask = XSCHEDULER_DAYWEEK_ISSUNDAY; break;
548 case XDATETIME_DAYWEEK_MONDAY : mask = XSCHEDULER_DAYWEEK_ISMONDAY; break;
549 case XDATETIME_DAYWEEK_TUESDAY : mask = XSCHEDULER_DAYWEEK_ISTUESDAY; break;
550 case XDATETIME_DAYWEEK_WEDNESDAY : mask = XSCHEDULER_DAYWEEK_ISWEDNESDAY; break;
551 case XDATETIME_DAYWEEK_THURSDAY : mask = XSCHEDULER_DAYWEEK_ISTHURSDAY; break;
552 case XDATETIME_DAYWEEK_FRIDAY : mask = XSCHEDULER_DAYWEEK_ISFRIDAY; break;
553 case XDATETIME_DAYWEEK_SATURDAY : mask = XSCHEDULER_DAYWEEK_ISSATURDAY; break;
554 }
555
556 return mask;
557}

950void XSCHEDULER::ThreadScheduler(void* data)
951{
952 XSCHEDULER* xscheduler = (XSCHEDULER*)data;
953 if(!xscheduler)
954 {
955 return;
956 }
957
958 if(!xscheduler->xmutexscheduler)
959 {
960 return;
961 }
962
963 if(!xscheduler->xdatetimeactual)
964 {
965 return;
966 }
967
968 if(!xscheduler->isactive)
969 {
970 return;
971 }
972
973 GEN_XSLEEP.MilliSeconds(10);
974
975 xscheduler->xmutexscheduler->Lock();
976
977 XSCHEDULERTASK* task;
978
979 task = xscheduler->Task_Get(xscheduler->indextask);
980 if(!task)
981 {
982 xscheduler->xmutexscheduler->UnLock();
983 xscheduler->indextask = 0;
984 return;
985 }
986
987 if(xscheduler->xdatetimeactual)
988 {
989 xscheduler->xdatetimeactual->Read();
990 }
991
992 if(task->IsActive())
993 {
994 if(task->CheckCondition(xscheduler->xdatetimeactual, task->GetXTimer()))
995 {
996 xscheduler->indextask++;
997 xscheduler->xmutexscheduler->UnLock();
998
999 XSCHEDULER_XEVENT xevent(xscheduler, XEVENT_TYPE_SCHEDULER);
1000
1001 xevent.SetScheduler(xscheduler);
1002 xevent.SetTask(task);
1003 xevent.SetDateTime(xscheduler->xdatetimeactual);
1004
1005 xscheduler->PostEvent(&xevent);
1006
1007 return;
1008 }
1009 }
1010
1011 xscheduler->indextask++;
1012 xscheduler->xmutexscheduler->UnLock();
1013}
