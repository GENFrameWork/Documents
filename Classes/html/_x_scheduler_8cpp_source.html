458bool XSCHEDULERTASK::CheckCondition(XDATETIME* xdatetimeactual, XTIMER* xtimeractual)
459{
460 if(!xdatetimeactual) return false;
461
462 XQWORD timeactualsec = xdatetimeactual->GetSeconsFromDate();
463
464 if(xdatetimelimitstart.IsValidDate())
465 {
466 if(timeactualsec < xdatetimelimitstart.GetSeconsFromDate())
467 {
468 isinvalidtimelimit = false;
469
470 xtimeractual->Reset();
471 return false;
472 }
473 else isinvalidtimelimit = true;
474 }
475
476 if(xdatetimelimitend.IsValidDate())
477 {
478 if(timeactualsec > xdatetimelimitend.GetSeconsFromDate())
479 {
480 isinvalidtimelimit = false;
481 return false;
482
483 } else isinvalidtimelimit = true;
484 }
485
486 if(conditiondayweekmask != XSCHEDULER_DAYWEEK_NONE)
487 {
488 if(!(GetDayOfWeekMask(xdatetimeactual) & conditiondayweekmask))
489 {
490 xtimeractual->Reset();
491 return false;
492 }
493 }
494
495 if(!ncyclesactual) return false;
496
497 bool checkcadence = true;
498 if(isstartimmediatelycycles)
499 {
500 if(!ncyclesmade) checkcadence = false;
501 }
502
503 if(checkcadence)
504 {
505 if(xdatetimecadence.IsValidDate())
506 {
507 if(xtimeractual->GetMeasureSeconds() < xdatetimecadence.GetSeconsFromDate()) return false;
508 }
509 }
510
511 if(ncyclesactual>0) ncyclesactual--;
512
513 xtimeractual->Reset();
514
515 ncyclesmade++;
516
517 return true;
518}

532XBYTE XSCHEDULERTASK::GetDayOfWeekMask(XDATETIME* xtimeactual)
533{
534 if(!xtimeactual) return XSCHEDULER_DAYWEEK_NONE;
535
536 XBYTE mask = XSCHEDULER_DAYWEEK_NONE;
537
538 switch(xtimeactual->GetDayOfWeek())
539 {
540 case XDATETIME_DAYWEEK_SUNDAY : mask = XSCHEDULER_DAYWEEK_ISSUNDAY; break;
541 case XDATETIME_DAYWEEK_MONDAY : mask = XSCHEDULER_DAYWEEK_ISMONDAY; break;
542 case XDATETIME_DAYWEEK_TUESDAY : mask = XSCHEDULER_DAYWEEK_ISTUESDAY; break;
543 case XDATETIME_DAYWEEK_WEDNESDAY : mask = XSCHEDULER_DAYWEEK_ISWEDNESDAY; break;
544 case XDATETIME_DAYWEEK_THURSDAY : mask = XSCHEDULER_DAYWEEK_ISTHURSDAY; break;
545 case XDATETIME_DAYWEEK_FRIDAY : mask = XSCHEDULER_DAYWEEK_ISFRIDAY; break;
546 case XDATETIME_DAYWEEK_SATURDAY : mask = XSCHEDULER_DAYWEEK_ISSATURDAY; break;
547 }
548
549 return mask;
550}

943void XSCHEDULER::ThreadScheduler(void* data)
944{
945 XSCHEDULER* xscheduler = (XSCHEDULER*)data;
946 if(!xscheduler)
947 {
948 return;
949 }
950
951 if(!xscheduler->xmutexscheduler)
952 {
953 return;
954 }
955
956 if(!xscheduler->xdatetimeactual)
957 {
958 return;
959 }
960
961 if(!xscheduler->isactive)
962 {
963 return;
964 }
965
966 GEN_XSLEEP.MilliSeconds(10);
967
968 xscheduler->xmutexscheduler->Lock();
969
970 XSCHEDULERTASK* task;
971
972 task = xscheduler->Task_Get(xscheduler->indextask);
973 if(!task)
974 {
975 xscheduler->xmutexscheduler->UnLock();
976 xscheduler->indextask = 0;
977 return;
978 }
979
980 if(xscheduler->xdatetimeactual)
981 {
982 xscheduler->xdatetimeactual->Read();
983 }
984
985 if(task->IsActive())
986 {
987 if(task->CheckCondition(xscheduler->xdatetimeactual, task->GetXTimer()))
988 {
989 xscheduler->indextask++;
990 xscheduler->xmutexscheduler->UnLock();
991
992 XSCHEDULER_XEVENT xevent(xscheduler, XEVENT_TYPE_SCHEDULER);
993
994 xevent.SetScheduler(xscheduler);
995 xevent.SetTask(task);
996 xevent.SetDateTime(xscheduler->xdatetimeactual);
997
998 xscheduler->PostEvent(&xevent);
999
1000 return;
1001 }
1002 }
1003
1004 xscheduler->indextask++;
1005 xscheduler->xmutexscheduler->UnLock();
1006}
