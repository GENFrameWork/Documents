600bool XINTERPOLATOR::Update(XQWORD delta)
601{
602 switch(type)
603 {
604 case XINTERPOLATOR_TYPE_LIMIT : Interpolate(delta);
605
606 if(mode == XINTERPOLATOR_LIMITMODE_INCREMENTAL)
607 {
608 if(target<=current) current=target;
609 }
610 else
611 {
612 if(mode == XINTERPOLATOR_LIMITMODE_DECREMENTAL)
613 if(target>=current) current = target;
614 }
615
616 break;
617
618 case XINTERPOLATOR_TYPE_RELATIVE :
619 case XINTERPOLATOR_TYPE_ABSOLUTE : if(current > target) current = target;
620 if(current < 0) current = 0;
621 break;
622
623 case XINTERPOLATOR_TYPE_NONE : break;
624 }
625
626 return true;
627}

671 int margin = 0;
672
673 if(result == target) return false;

731{
732 float fdelta = (float) (delta/100.0f);
733 int itarget = (int) (target);
734 float result = (float) (current);
735 int margin = 0;

741 if(delta!=0)
742 {
743 if(carry<0.0000001f) carry=0.0f;
