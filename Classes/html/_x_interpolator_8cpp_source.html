607bool XINTERPOLATOR::Update(XQWORD delta)
608{
609 switch(type)
610 {
611 case XINTERPOLATOR_TYPE_LIMIT : Interpolate(delta);
612
613 if(mode == XINTERPOLATOR_LIMITMODE_INCREMENTAL)
614 {
615 if(target<=current) current=target;
616 }
617 else
618 {
619 if(mode == XINTERPOLATOR_LIMITMODE_DECREMENTAL)
620 if(target>=current) current = target;
621 }
622
623 break;
624
625 case XINTERPOLATOR_TYPE_RELATIVE :
626 case XINTERPOLATOR_TYPE_ABSOLUTE : if(current > target) current = target;
627 if(current < 0) current = 0;
628 break;
629
630 case XINTERPOLATOR_TYPE_NONE : break;
631 }
632
633 return true;
634}

673bool XINTERPOLATOR::InterpolateLinear(XQWORD delta)
674{
675 float fdelta = (float)(delta/100.0f);
676 int itarget = (int)(target);
677 float result = (float)current;
678 int margin = 0;
679
680 if(result == target) return false;
681
682 if(result < itarget)
683 {
684 if (delta!=0)
685 {
686 if (carry<0.0000001f) carry=0.0f;
687
688 float inc = (speed*fdelta) + carry;
689 carry = inc - float((int)inc);
690
691 result += (int)inc;
692 }
693
694 if(result >= itarget)
695 {
696 result = (float)itarget;
697 carry = 0.0f;
698 }
699

701 else

703 if(delta!=0)

737bool XINTERPOLATOR::InterpolateRelative(XQWORD delta)
738{
739 float fdelta = (float) (delta/100.0f);
740 int itarget = (int) (target);
741 float result = (float) (current);
742 int margin = 0;
743
744 if(result == target) return false;
745
746 if(result < itarget)
747 {
748 if(delta!=0)
749 {
750 if(carry<0.0000001f) carry=0.0f;
751
752 float dif = (float)(itarget - result);
753 float inc = ((speed/100.0f)*fdelta*dif)+carry;
754
755 carry = inc - float((int)inc);
756 result += (int)inc;
757 }
758
759 if(result > (itarget-margin))
760 {
761 result = (float)itarget;
762 carry = 0.0f;
763 }
764 }
765 else

768 {
769 if (carry<0.0000001f) carry=0.0f;

773
774 carry = inc - float((int)inc);
775 result -= (int)inc;
776 }
777
778 if(result < float(itarget+margin))
779 {
780 result = (float)itarget;
