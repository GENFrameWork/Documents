281int DIOLINUXSTREAMTCPIPSERVER::Accept(int handlesocket, void* addr, void* addrlen, XDWORD usec)
282{
283 fd_set fds;
284 struct timeval time_out;
285 int status;
286
287 if(handlesocket==-1)
288 {
289 return -1;
290 }
291
292 FD_ZERO(&fds);
293 FD_SET((unsigned int)handlesocket,&fds);
294
295 time_out.tv_sec = 0;
296 time_out.tv_usec = usec;
297
298 status = (int)select((int)(handlesocket)+1, &fds, NULL, NULL, &time_out);
299 if(status>0)
300 {
301 status = accept(handlesocket, (struct sockaddr*)addr, (socklen_t*)addrlen);
302 }
303 else
304 {
305 status = -1;
306 }
307
308 return status;
309}

323int DIOLINUXSTREAMTCPIPSERVER::IsReadyConnect(int handlesocket)
324{
325 if(handlesocket==-1)
326 {
327 return -1;
328 }
329
330 struct timeval tv;
331 int rc;
332 fd_set fdr;
333 fd_set fdw;
334 fd_set fds;
335
336 FD_ZERO(&fdr);
337 FD_ZERO(&fdw);
338 FD_ZERO(&fds);
339
340 FD_SET((unsigned int)handlesocket, &fdr);
341 FD_SET((unsigned int)handlesocket, &fdw);
342 FD_SET((unsigned int)handlesocket, &fds);
343
344 tv.tv_sec = 0;
345 tv.tv_usec = 100;
346
347 rc = select((int)(handlesocket)+1, &fdr, &fdw, &fds, &tv);
348 if(rc == -1) return -1;
349
350 int status1 = FD_ISSET(handlesocket, &fdr) ? 1 : 0;
351 int status2 = FD_ISSET(handlesocket, &fdw) ? 1 : 0;
352 int status3 = FD_ISSET(handlesocket, &fds) ? 1 : 0;
353
354 if(status1 || status2 )
355 {
356 return 1;
357 }
358
359 if(status3)
360 {
361 return -1;
362 }
363
364 return 0;
365}

492bool DIOLINUXSTREAMTCPIPSERVER::GetHandleServer(DIOSTREAMTCPIP* diostream)
493{
494 int handleserver = (int)config->GetHandleMultiServer();
495 struct sockaddr_in loc_addr = { 0 };
496 struct sockaddr_in rem_addr = { 0 };
497 XSTRING IPstring;
498
499 // ------------------------------------------------------------------------------------------
500
501 memset(&loc_addr,0,sizeof(struct sockaddr_in));
502 memset(&rem_addr,0,sizeof(struct sockaddr_in));
503
504 if(!diostream)
505 {
506 return false;
507 }
508
509 // ------------------------------------------------------------------------------------------
510
511 if(handleserver < 0)
512 {
513 config->GetXMutexMultiServer()->Lock();
514
515 handleserver = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
516 if(handleserver < 0)
517 {
518 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
519 SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
520
521 config->GetXMutexMultiServer()->UnLock();
522
523 return false;
524 }
525
526 int opt = 1;
527 if(setsockopt(handleserver, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0)
528 {
529 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
530 SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
531
532 config->GetXMutexMultiServer()->UnLock();
533
534 return false;
535 }
536
537 #ifdef SO_REUSEPORT
538 opt = 1;
539 if(setsockopt(handleserver, SOL_SOCKET, SO_REUSEPORT, &opt, sizeof(opt)) < 0)
540 {
541 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
542 SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
543 config->GetXMutexMultiServer()->UnLock();
544
545 return false;
546 }
547 #endif
548
549 config->GetLocalIP()->GetXString(IPstring);
550
551 loc_addr.sin_family = AF_INET;
552 if(!config->GetLocalIP()->IsEmpty())
553 {
554 XBUFFER charstr;
555
556 IPstring.ConvertToASCII(charstr);
557 loc_addr.sin_addr.s_addr = inet_addr(charstr.GetPtrChar());
558
559 } else loc_addr.sin_addr.s_addr = htonl(INADDR_ANY);
560
561 loc_addr.sin_port = htons(config->GetRemotePort());
562
563 if(bind(handleserver, (struct sockaddr *)&loc_addr, sizeof(loc_addr)) < 0)
564 {
565 if(errno == EADDRINUSE)
566 {
567 SetLastDIOError(DIOSTREAMERROR_ADDRINUSE);
568
569 } else SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
570
571 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
572
573 config->GetXMutexMultiServer()->UnLock();
574
575 return false;
576 }
577
578 if(listen(handleserver, DIOSTREAMTCPIP_MAXLISTEN) < 0)
579 {
580 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
581 SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
582 config->GetXMutexMultiServer()->UnLock();
583
584 return false;
585 }
586
587 fcntl(handleserver, F_SETFL, fcntl(handleserver, F_GETFL,0) | O_NONBLOCK);
588
589 config->SetHandleMultiServer(handleserver);
590
591 config->GetXMutexMultiServer()->UnLock();
592 }
593
594 // ------------------------------------------------------------------------------------------
595
596 if(enumservers)
597 {
598 if(enumservers->GetXTimer())
599 {
600 if(enumservers->GetXTimer()->GetMeasureSeconds() >= DIOSTREAMTCPIP_TIMEOUTENUMSERVERSIGNAL)
601 {
602 enumservers->SendSignal();
603 enumservers->GetXTimer()->Reset();
604 }
605 }
606 }
607
608 // ------------------------------------------------------------------------------------------
609
610 socklen_t sizeaddr = sizeof(struct sockaddr_in);
611 int handlesocket = Accept(handleserver, (struct sockaddr*)&rem_addr, &sizeaddr, 100/*50*1000*/);
612 if(handlesocket >= 0)
613 {
614 int opt = 1;
615 if(setsockopt(handlesocket, SOL_SOCKET, SO_KEEPALIVE , &opt, sizeof(opt)) < 0)
616 {
617 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
618 SetLastDIOError(DIOSTREAMERROR_UNKNOWN);
619
620 return false;
621 }
622
623 int flags = fcntl(handlesocket, F_GETFL, 0);
624 fcntl(handlesocket, F_SETFL, flags | O_NONBLOCK);
625
626 clientIP.Set(inet_ntoa(rem_addr.sin_addr));
627 }
628 else
629 {
630 if(errno == EADDRINUSE)
631 {
632 SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
633 SetLastDIOError(DIOSTREAMERROR_ADDRINUSE);
634
635 return false;
636 }
637 }
638
639 DIOLINUXSTREAMTCPIP* diowindowsstreamtcpip = (DIOLINUXSTREAMTCPIP*)diostream;
640 if(diowindowsstreamtcpip)
641 {
642 diowindowsstreamtcpip->SetHandleSocket(handlesocket);
643 }
644
645 // ------------------------------------------------------------------------------------------
646
647 return true;
648}

660void DIOLINUXSTREAMTCPIPSERVER::ThreadConnection(void* data)
661{
662 DIOLINUXSTREAMTCPIPSERVER* diostreamserver = (DIOLINUXSTREAMTCPIPSERVER*)data;
663 if(!diostreamserver)
664 {
665 return;
666 }
667
668 diostreamserver->DeleteAllStreamDisconnected();
669
670 for(XDWORD c=0; c<diostreamserver->GetMultiSocketStreams()->GetSize(); c++)
671 {
672 DIOLINUXSTREAMTCPIP* diostream = (DIOLINUXSTREAMTCPIP*)diostreamserver->GetMultiSocketStreams()->Get(c);
673 if(diostream)
674 {
675 if(diostream->GetEvent()==DIOLINUXTCPIPFSMEVENT_NONE) // No hay nuevos Eventos
676 {
677 switch(diostream->GetCurrentState())
678 {
679 case DIOLINUXTCPIPSERVERFSMSTATE_NONE : break;
680
681 case DIOLINUXTCPIPSERVERFSMSTATE_GETTINGCONNECTION : if(diostream->GetHandleSocket() == -1)
682 {
683 diostreamserver->GetHandleServer(diostream);
684 }
685 else
686 {
687 switch(diostreamserver->IsReadyConnect(diostream->GetHandleSocket()))
688 {
689 case -1: diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
690 break;
691
692 case 1: diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_CONNECTED);
693 diostream->SetStatus(DIOSTREAMSTATUS_CONNECTED);
694 break;
695
696 default: diostream->SetStatus(DIOSTREAMSTATUS_GETTINGCONNECTION);
697 break;
698 }
699 }
700 break;
701
702 case DIOLINUXTCPIPSERVERFSMSTATE_CONNECTED : break;
703
704 case DIOLINUXTCPIPSERVERFSMSTATE_WAITINGTOREAD : { struct timeval waitd;
705 fd_set read_flags;
706 fd_set write_flags;
707
708 if(diostream->GetHandleSocket() == -1)
709 {
710 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
711 break;
712 }
713
714 waitd.tv_sec = 0;
715 waitd.tv_usec = 100;
716
717 FD_ZERO(&read_flags);
718 FD_ZERO(&write_flags);
719
720 if(!diostream->IsBlockRead()) FD_SET(diostream->GetHandleSocket(), &read_flags);
721 if(!diostream->IsBlockWrite())
722 {
723 if(diostream->outbuffer->GetSize()) FD_SET(diostream->GetHandleSocket(), &write_flags);
724 }
725
726 int error = select((int)(diostream->GetHandleSocket())+1, &read_flags, &write_flags, (fd_set*)NULL, &waitd);
727 if(error == -1) //Socket ready for reading
728 {
729 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
730 break;
731 }
732
733 if(FD_ISSET(diostream->GetHandleSocket(), &read_flags)) //Socket ready for reading
734 {
735 XBYTE buffer[DIOSTREAM_MAXBUFFER];
736
737 FD_CLR(diostream->GetHandleSocket(), &read_flags);
738
739 XDWORD size = recv(diostream->GetHandleSocket(), (char*)buffer, DIOSTREAM_MAXBUFFER, 0);
740 if(size==-1)
741 {
742 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
743 break;
744 }
745
746 if(size)
747 {
748 diostream->inbuffer->Add(buffer,size);
749 diostream->AddNBytesRead(size);
750 diostreamserver->nbytesread += size;
751 }
752 else
753 {
754 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
755 break;
756 }
757 }
758
759 if(FD_ISSET(diostream->GetHandleSocket(), &write_flags)) //Socket ready for writing
760 {
761 FD_CLR(diostream->GetHandleSocket(), &write_flags);
762
763 int esize = diostream->outbuffer->GetSize();
764 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
765
766 if(esize)
767 {
768 diostream->outbuffer->SetBlocked(true);
769 XDWORD size = send(diostream->GetHandleSocket(), (char*)diostream->outbuffer->Get(), esize,0);
770 diostream->outbuffer->SetBlocked(false);
771
772 if(size == -1)
773 {
774 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_DISCONNECTING);
775 break;
776 }
777
778 if(size)
779 {
780 diostream->outbuffer->Extract(NULL, 0, size);
781
782 diostream->AddNBytesWrite(size);
783 diostreamserver->nbyteswrite += size;
784 }
785 }
786 }
787 }
788 break;
789
790 case DIOLINUXTCPIPSERVERFSMSTATE_SENDINGDATA : break;
791
792 case DIOLINUXTCPIPSERVERFSMSTATE_DISCONNECTING : break;
793
794 }
795 }
796 else
797 {
798 if(diostream->GetEvent()<DIOLINUXTCPIP_LASTEVENT)
799 {
800 diostream->CheckTransition();
801
802 switch(diostream->GetCurrentState())
803 {
804 case DIOLINUXTCPIPSERVERFSMSTATE_NONE : break;
805
806 case DIOLINUXTCPIPSERVERFSMSTATE_GETTINGCONNECTION : if(diostreamserver->enumservers)
807 {
808 if(diostreamserver->enumservers->GetXTimer())
809 {
810 diostreamserver->enumservers->GetXTimer()->AddSeconds(DIOSTREAMTCPIP_TIMEOUTENUMSERVERSIGNAL);
811 }
812 }
813 break;
814
815 case DIOLINUXTCPIPSERVERFSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream, DIOSTREAM_XEVENT_TYPE_CONNECTED);
816 xevent.SetDIOStream(diostream);
817 diostreamserver->PostEvent(&xevent);
818
819 diostream->SetEvent(DIOLINUXTCPIPFSMEVENT_WAITINGTOREAD);
820
821 diostreamserver->CreateStream();
822 }
823 break;
824
825 case DIOLINUXTCPIPSERVERFSMSTATE_WAITINGTOREAD : break;
826
827 case DIOLINUXTCPIPSERVERFSMSTATE_SENDINGDATA : break;
828
829 case DIOLINUXTCPIPSERVERFSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream, DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
830 xevent.SetDIOStream(diostream);
831 diostreamserver->PostEvent(&xevent);
832
833 diostream->SetStatus(DIOSTREAMSTATUS_DISCONNECTED);
834 }
835 break;
836
837 }
838 }
839 }
840 }
841 }
