340bool GRPVIEWPORT::SetCanvasPosition(float x, float y)
341{
342 bool status = true;
343
344 canvas_x = x;
345 canvas_y = y;
346
347 if(canvas_x < 0)
348 {
349 canvas_x = 0;
350 status = false;
351 }
352
353 if(canvas_y < 0)
354 {
355 canvas_y = 0;
356 status = false;
357 }
358
359 if(canvas)
360 {
361 if((canvas_x + width) > canvas->GetWidth())
362 {
363 canvas_x = canvas->GetWidth() - width;
364 status = false;
365 }
366
367 if((canvas_y + height) > canvas->GetHeight())
368 {
369 canvas_y = canvas->GetHeight() - height;
370 status = false;
371 }
372 }
373 else
374 {
375 status = false;
376 }
377
378 return status;
379}
