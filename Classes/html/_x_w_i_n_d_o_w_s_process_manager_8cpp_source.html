197bool XWINDOWSPROCESSMANAGER::Application_Execute(XCHAR* applicationpath, XCHAR* params, XSTRING* in, XSTRING* out, int* returncode)
198{
199 #define CMDLINE_SIZE (10*1024)
200 #define OUTBUF_SIZE (10*1024*1024)
201 #define TEMPBUF_SIZE (10*1024)
202
203 PROCESS_INFORMATION processinfo;
204 STARTUPINFOA startupinfo;
205 SECURITY_ATTRIBUTES saattr;
206 char* cmdline = NULL;
207 char* outbuf = NULL;
208 DWORD bytes_read;
209 char* tempbuf = NULL;
210 DWORD exitcode;
211 HANDLE stdhandle_out_read = NULL;
212 HANDLE stdhandle_out_write = NULL;
213 HANDLE stdhandle_in_read = NULL;
214 HANDLE stdhandle_in_write = NULL;
215 XSTRING _command;
216 bool status = false;
217
218 cmdline = new char[CMDLINE_SIZE];
219 outbuf = new char[OUTBUF_SIZE];
220 tempbuf = new char[TEMPBUF_SIZE];
221
222 if(cmdline && outbuf && tempbuf)
223 {
224 _command.Format(__L("\"%s\""), applicationpath);
225
226 if(params) _command.AddFormat(__L(" %s"), params);
227
228 XBUFFER charstr;
229
230 _command.ConvertToASCII(charstr);
231 memcpy(cmdline, charstr.GetPtrChar(), _command.GetSize()+1);
232
233 memset(&saattr, 0, sizeof(saattr));
234 saattr.nLength = sizeof(SECURITY_ATTRIBUTES);
235 saattr.bInheritHandle = TRUE;
236 saattr.lpSecurityDescriptor = NULL;
237
238 // Create a pipe for the child process's STDOUT.
239 if(!CreatePipe(&stdhandle_out_read, &stdhandle_out_write, &saattr, 0)) return false;
240 // Ensure the read handle to the pipe for STDOUT is not inherited.
241 if(!SetHandleInformation(stdhandle_out_read, HANDLE_FLAG_INHERIT, 0)) return false;
242
243 // Create a pipe for the child process's STDIN.
244 if(!CreatePipe(&stdhandle_in_read, &stdhandle_in_write, &saattr, 0)) return false;
245 // Ensure the read handle to the pipe for STDIN is not inherited.
246 if(!SetHandleInformation(stdhandle_in_write, HANDLE_FLAG_INHERIT, 0)) return false;
247
248 ZeroMemory(&processinfo, sizeof(PROCESS_INFORMATION));
249
250 ZeroMemory(&startupinfo, sizeof(STARTUPINFO));
251 startupinfo.cb = sizeof(startupinfo);
252 startupinfo.hStdError = stdhandle_out_write;
253 startupinfo.hStdOutput = stdhandle_out_write;
254 startupinfo.hStdInput = stdhandle_in_read; // GetStdHandle(STD_INPUT_HANDLE);
255 startupinfo.dwFlags |= STARTF_USESTDHANDLES;
256
257 if(CreateProcessA(NULL, cmdline, NULL, NULL, TRUE, CREATE_NO_WINDOW, NULL, NULL, &startupinfo, &processinfo))
258 {
259 CloseHandle(stdhandle_out_write);
260
261 if(in)
262 {
263 if(in->GetSize())
264 {
265 DWORD bytes_write;
266
267 XBUFFER charstr;
268
269 (*in).ConvertToASCII(charstr);
270 WriteFile(stdhandle_in_write, charstr.Get(), in->GetSize(), &bytes_write, NULL);
271
272 FlushFileBuffers(stdhandle_in_write);
273 }
274 }
275
276 CloseHandle(stdhandle_in_write);
277
278 if(out)
279 {
280 strcpy(outbuf, "");
281
282 for(;;)
283 {
284 if(!ReadFile(stdhandle_out_read, tempbuf, TEMPBUF_SIZE-1, &bytes_read, NULL))
285 {
286 break;
287 }
288
289 if(bytes_read > 0)
290 {
291 tempbuf[bytes_read] = '\0';
292 strcat(outbuf, tempbuf);
293 }
294 }
295
296 (*out) = outbuf;
297 }
298
299 if(WaitForSingleObject(processinfo.hProcess, (out?INFINITE:500)) == WAIT_OBJECT_0)
300 {
301 if(out)
302 {
303 if(GetExitCodeProcess(processinfo.hProcess, &exitcode))
304 {
305 if(returncode) (*returncode) = exitcode;
306 }
307
308 CloseHandle(processinfo.hProcess);
309 CloseHandle(processinfo.hThread);
310 }
311
312 status = true;
313 }
314 }
315 }
316
317 if(cmdline) delete [] cmdline;
318 if(outbuf) delete [] outbuf;
319 if(tempbuf) delete [] tempbuf;
320
321 return status;
322}

647int XWINDOWSPROCESSMANAGER::GetWindowTitleHeight(HWND hwnd)
648{
649 RECT window_rectangle;
650 RECT client_rectangle;
651 int height;
652 int width;
653
654 GetWindowRect(hwnd, &window_rectangle);
655 GetClientRect(hwnd, &client_rectangle);
656
657 height = (window_rectangle.bottom - window_rectangle.top) - (client_rectangle.bottom - client_rectangle.top);
658 width = (window_rectangle.right - window_rectangle.left) - (client_rectangle.right - client_rectangle.left);
659
660 return height - (width/2);
661}

698bool XWINDOWSPROCESSMANAGER::GetChildProcesses(XVECTOR<XPROCESS*>& applist)
699{
700 HANDLE threadSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
701 if(threadSnapshot == INVALID_HANDLE_VALUE)
702 {
703 return false;
704 }
705
706 THREADENTRY32 threadEntry;
707 threadEntry.dwSize = sizeof(THREADENTRY32);
708
709 if(!Thread32First(threadSnapshot, &threadEntry))
710 {
711 CloseHandle(threadSnapshot);
712 return false;
713 }
714
715 do { XPROCESS* mainprocess = Application_GetProcessByID(threadEntry.th32OwnerProcessID, applist);
716 if(mainprocess)
717 {
718 if(mainprocess->GetProcessIDs()->IsEmpty())
719 {
720 mainprocess->GetProcessIDs()->Add(threadEntry.th32OwnerProcessID);
721 }
722
723 mainprocess->GetProcessIDs()->Add(threadEntry.th32ThreadID);
724 }
725
726 } while (Thread32Next(threadSnapshot, &threadEntry));
727
728 CloseHandle(threadSnapshot);
729
730 return true;
731}
