182void* XMEMORY_CONTROL::Assign(XDWORD size, char* pathfile, int line)
183{
184 if(!size) return NULL;
185
186 Lock();
187
188 void* ptr = NULL;
189 ptr = malloc(size);
190 if(!ptr)
191 {
192 UnLock();
193 return NULL;
194 }
195
196 if(!RegisterAssign(ptr, size, pathfile, line))
197 {
198 // XTRACE_PRINTCOLOR(4, __L("[XMemory Control] ALERT: Make Malloc -> The memory allocation could not be registered!"));
199 }
200
201 nassigns++;
202 maxnassigns = __MAX(nassigns, maxnassigns);
203
204 used += size;
205 maxused = __MAX(used, maxused);
206
207 UnLock();
208
209 return ptr;
210}

474XDWORD XMEMORY_CONTROL::CRC32(XBYTE* data, XWORD size)
475{
476 static XDWORD table[0x100];
477 XDWORD crc = 0;
478
479 if(!*table)
480 {
481 for(XWORD c=0; c<0x100; ++c)
482 {
483 table[c] = CRC32ForByte(c);
484 }
485 }
486
487 for(XDWORD c=0; c<size; ++c)
488 {
489 crc = table[(XBYTE)crc ^ ((XBYTE*)data)[c]] ^ crc >> 8;
490 }

510bool XMEMORY_CONTROL::RegisterAssign(void* ptr, XDWORD size, char* pathfile, int line)
511{
512 XDWORD index = 0;
513
514 if(!SearchAssignIndex(true, ptr, index))
515 {
516 if(!ResizeAssignList()) return false;
517 if(!SearchAssignIndex(true, NULL, index)) return false;
518 }
519
520 assignlist[index].ptr = ptr;
521 assignlist[index].size = size;
522
523 XDWORD sizenamemodule = (XDWORD)strlen(pathfile)+1;
524 int c = (int)sizenamemodule;
525
526 while(c>=0)
527 {
528 if((pathfile[c] == '\\') || (pathfile[c] == '/')) break;
529 c--;
530 }
531
532 sizenamemodule -= c;
533 if(sizenamemodule >= XMEMORY_CONTROL_MAXNAMEMODULESIZE) sizenamemodule = (XMEMORY_CONTROL_MAXNAMEMODULESIZE-1);
534
535 memcpy(assignlist[index].namemodule, &pathfile[c+1], sizenamemodule);
536
537 assignlist[index].linemodule = line;
538
539 return true;
540}

616bool XMEMORY_CONTROL::SearchAssignIndex(bool free, void* ptr, XDWORD& index)
617{
618 XDWORD start = 0;
619 XDWORD counter = 0;
620 XDWORD limit = 0;
621 bool found = false;
622
623 index = 0;
624
625 #ifdef MICROCONTROLLER
626 start = 0;
627 #else
628 start = CRC32((XBYTE*)&ptr, sizeof(XDWORD));
629 start &= (nregisterelements -1);
630 #endif
631
632 counter = start;
633 if(counter >= nregisterelements-1) counter = 0;
634
635 limit = (nregisterelements-1);
636
637 for(int c=0; c<2; c++)
638 {
639 while(counter < limit)
640 {
641 if(free)
642 {
643 if(!assignlist[counter].ptr)
644 {
645 index = counter;
646 found = true;
647 break;
648 }
649 }
650 else
651 {
652 if(assignlist[counter].ptr == ptr)
653 {
654 index = counter;
655 found = true;
656 break;
657 }
658 }
659
660 counter++;
661 }
662
663 if(found) return true;
664
665 counter = 0;
666 limit = start;
667 }
668
669 return false;
670}
