200bool DIONODE::GetSubGroups(XVECTOR<XSTRING*>& listsubgroups)
201{
202 XPATH pathgroup;
203
204 pathgroup = group.Get();
205
206 pathgroup.Slash_Delete();
207
208 do{ XSTRING* _subgroup = new XSTRING();
209 if(_subgroup)
210 {
211 pathgroup.GetNamefileExt(*_subgroup);
212 pathgroup.GetPath(pathgroup);
213
214 if(!_subgroup->Compare(pathgroup, true))
215 {
216 listsubgroups.Insert(0, _subgroup);
217 break;
218 }
219
220 pathgroup.Slash_Delete();
221
222 if(_subgroup->IsEmpty())
223 {
224 delete _subgroup;
225 break;
226 }
227 else
228 {
229 listsubgroups.Insert(0, _subgroup);
230 }
231 }
232 else
233 {
234 break;
235 }
236
237 } while(1);
238
239
240 return listsubgroups.IsEmpty()?false:true;
241}
