129bool INPLINUXDEVICEMOUSEX11::Update()
130{
131 if(!created || !enabled)
132 {
133 return false;
134 }
135
136 if(!grpscreenx11)
137 {
138 return false;
139 }
140
141 if(!grpscreenx11->HasFocus())
142 {
143 return false;
144 }
145
146 Window root;
147 Window child;
148 int rootx;
149 int rooty;
150 int mousex = -1;
151 int mousey = -1;
152 unsigned int mask;
153
154 XQueryPointer(grpscreenx11->GetDisplay(), (*grpscreenx11->GetWindow()), &root, &child, &rootx, &rooty, &mousex, &mousey, &mask);
155
156 INPCURSOR* cursor = GetCursor(INPCURSOR_ID_MOUSE);
157 if(cursor)
158 {
159 if((mousex>=0)&&(mousey>=0)&&(mousex < grpscreenx11->GetWidth()) && (mousey < grpscreenx11->GetHeight()))
160 {
161 cursor->SetIsChanged(false);
162
163 if((cursor->GetX() != mousex) || (cursor->GetY() != mousey))
164 {
165 mousey = (grpscreenx11->GetHeight()-mousey);
166 cursor->Set(mousex,mousey);
167 cursor->SetIsChanged(true);
168 }
169 }
170 else
171 {
172 mousex = -1;
173 mousey = -1;
174
175 cursor->Set(mousex,mousey);
176 cursor->SetIsChanged(true);
177 }
178
179 INPBUTTON* button = GetButton(INPBUTTON_ID_MOUSE_LEFT);
180 if(button) cursor->AddPointToMotion(button->IsPressed());
181 }
182
183 //-------------------------------------------------------------
184 // Read Buttons
185
186 for(XDWORD c=0; c<buttons.GetSize(); c++)
187 {
188 INPBUTTON* button=buttons.Get(c);
189 if(button)
190 {
191 if(button->GetState() == INPBUTTON_STATE_RELEASED)
192 {
193 button->SetState(INPBUTTON_STATE_UP);
194 }
195 else
196 {
197 if(button->GetState() == INPBUTTON_STATE_PRESSED) button->SetState(INPBUTTON_STATE_HOLD);
198 }
199 }
200 }
201
202 XEvent event;
203
204 if(XPending(grpscreenx11->GetDisplay()))
205 {
206 XNextEvent(grpscreenx11->GetDisplay(), &event);
207
208 switch(event.type)
209 {
210 case ButtonPress : { INPBUTTON* button = GetButtonByCode(event.xbutton.button);
211 if(button)
212 {
213 switch (button->GetState())
214 {
215 case INPBUTTON_STATE_UP: button->SetState(INPBUTTON_STATE_PRESSED); break;
216 case INPBUTTON_STATE_HOLD: button->SetState(INPBUTTON_STATE_HOLD); break;
217 case INPBUTTON_STATE_PRESSED: button->SetState(INPBUTTON_STATE_HOLD); break;
218 case INPBUTTON_STATE_RELEASED: button->SetState(INPBUTTON_STATE_PRESSED); break;
219 }
220
221 button->SetPressed(true);
222 }
223 }
224 break;
225
226 case ButtonRelease : { INPBUTTON* button = GetButtonByCode(event.xbutton.button);
227 if(button)
228 {
229 switch (button->GetState())
230 {
231 case INPBUTTON_STATE_UP: button->SetState(INPBUTTON_STATE_UP); break;
232 case INPBUTTON_STATE_HOLD: button->SetState(INPBUTTON_STATE_RELEASED); break;
233 case INPBUTTON_STATE_PRESSED: button->SetState(INPBUTTON_STATE_RELEASED); break;
234 case INPBUTTON_STATE_RELEASED: button->SetState(INPBUTTON_STATE_RELEASED); break;
235 }
236
237 button->SetPressed(false);
238 }
239 }
240 break;
241 }
242 }
243
244
245 return true;
246}
