136bool INPLINUXDEVICEMOUSEX11::Update()
137{
138 if(!created || !enabled)
139 {
140 return false;
141 }
142
143 if(!grpscreenx11)
144 {
145 return false;
146 }
147
148 if(!grpscreenx11->HasFocus())
149 {
150 return false;
151 }
152
153 Display* display = grpscreenx11->GetDisplay();
154 Window root;
155 Window child;
156 int rootx;
157 int rooty;
158 int mousex = -1;
159 int mousey = -1;
160 unsigned int mask;
161
162 XQueryPointer(display, (*grpscreenx11->GetWindow()), &root, &child, &rootx, &rooty, &mousex, &mousey, &mask);
163
164 INPCURSOR* cursor = GetCursor(INPCURSOR_ID_MOUSE);
165 if(cursor)
166 {
167 if((mousex>=0)&&(mousey>=0)&&(mousex < grpscreenx11->GetWidth()) && (mousey < grpscreenx11->GetHeight()))
168 {
169 cursor->SetIsChanged(false);
170
171 if((cursor->GetX() != mousex) || (cursor->GetY() != mousey))
172 {
173 mousey = (grpscreenx11->GetHeight()-mousey);
174 cursor->Set(mousex,mousey);
175 cursor->SetIsChanged(true);
176 }
177 }
178 else
179 {
180 mousex = -1;
181 mousey = -1;
182
183 cursor->Set(mousex,mousey);
184 cursor->SetIsChanged(true);
185 }
186
187 INPBUTTON* button = GetButton(INPBUTTON_ID_MOUSE_LEFT);
188 if(button) cursor->AddPointToMotion(button->IsPressed());
189 }
190
191 //-------------------------------------------------------------
192 // Read Buttons
193
194 for(XDWORD c=0; c<buttons.GetSize(); c++)
195 {
196 INPBUTTON* button=buttons.Get(c);
197 if(button)
198 {
199 if(button->GetState() == INPBUTTON_STATE_RELEASED)
200 {
201 button->SetState(INPBUTTON_STATE_UP);
202 }
203 else
204 {
205 if(button->GetState() == INPBUTTON_STATE_PRESSED)
206 {
207 button->SetState(INPBUTTON_STATE_HOLD);
208 }
209 }
210 }
211 }
212
213 XEvent event;
214
215 if(XPending(display))
216 {
217 XNextEvent(display, &event);
218
219 switch(event.type)
220 {
221 case ButtonPress : { INPBUTTON* button = GetButtonByCode(event.xbutton.button);
222 if(button)
223 {
224 switch (button->GetState())
225 {
226 case INPBUTTON_STATE_UP: button->SetState(INPBUTTON_STATE_PRESSED); break;
227 case INPBUTTON_STATE_HOLD: button->SetState(INPBUTTON_STATE_HOLD); break;
228 case INPBUTTON_STATE_PRESSED: button->SetState(INPBUTTON_STATE_HOLD); break;
229 case INPBUTTON_STATE_RELEASED: button->SetState(INPBUTTON_STATE_PRESSED); break;
230 }
231
232 button->SetPressed(true);
233 }
234 }
235 break;
236
237 case ButtonRelease : { INPBUTTON* button = GetButtonByCode(event.xbutton.button);
238 if(button)
239 {
240 switch (button->GetState())
241 {
242 case INPBUTTON_STATE_UP: button->SetState(INPBUTTON_STATE_UP); break;
243 case INPBUTTON_STATE_HOLD: button->SetState(INPBUTTON_STATE_RELEASED); break;
244 case INPBUTTON_STATE_PRESSED: button->SetState(INPBUTTON_STATE_RELEASED); break;
245 case INPBUTTON_STATE_RELEASED: button->SetState(INPBUTTON_STATE_RELEASED); break;
246 }
247
248 button->SetPressed(false);
249 }
250 }
251 break;
252 }
253 }
254
255
256 return true;
257}
