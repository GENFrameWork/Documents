605GRPBITMAPFRAME* GRPBITMAPSEQUENCE::Update()
606{
607 if(!timer) return NULL;
608 if(frames.IsEmpty()) return NULL;
609 if(status!=GRPBITMAPSEQUENCESTATUS_PLAY) return NULL;
610
611 //if(timer->GetMeasureMilliSeconds()<timebyframe) return NULL;
612
613 timer->Reset();
614
615 if(!reverse)
616 {
617 if(frames.GetSize()!=1) frameindex++;
618
619 if(frameindex == frames.GetSize())
620 {
621 frameindex = 0;
622 nloopsmade++;
623 if((nloops)&&(nloops==nloopsmade)) status=GRPBITMAPSEQUENCESTATUS_END;
624 }
625 }
626 else
627 {
628 if(frames.GetSize()!=1) frameindex--;
629
630 if(frameindex<0)
631 {
632 frameindex = (frames.GetSize()-1);
633
634 nloopsmade++;
635 if((nloops)&&(nloops==nloopsmade)) status=GRPBITMAPSEQUENCESTATUS_END;
636 }
637 }
638
639 return (GRPBITMAPFRAME*)frames.Get(frameindex);
640}

674GRPBITMAPSEQUENCE* GRPBITMAPSEQUENCE::Copy()
675{
676 GRPBITMAPSEQUENCE* sequence = new GRPBITMAPSEQUENCE();
677 if(!sequence)
678 {
679 return NULL;
680 }
681
682 for(XDWORD c=0; c<frames.GetSize(); c++)
683 {
684 GRPBITMAPFRAME* frame = frames.Get(c);
685 if(frame)
686 {
687 GRPBITMAP* bitmap = frame->GetBitmap()->Clone();
688 if(bitmap)
689 {
690 sequence->AddFrame(bitmap, frame->GetAjustX(), frame->GetAjustY());
691 }
692 }
693 }
