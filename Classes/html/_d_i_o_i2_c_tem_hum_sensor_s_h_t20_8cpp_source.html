371bool DIOI2CTEMHUMSENSORSHT20::CheckCRC(XWORD datasensor, XBYTE checkvalue)
372{
373 XDWORD remainder = (XDWORD)datasensor << 8;
374
375 remainder |= checkvalue;
376
377 XDWORD divisor = (XDWORD)DIOI2CTEMHUMSENSORSHT20_SHIFTED_DIVISOR;
378
379 for(int c=0; c < 16 ; c++)
380 {
381 if(remainder & (XDWORD)1 << (23 - c))
382 {
383 remainder ^= divisor;
384 }
385
386 divisor >>= 1;
387 }
388
389 return remainder?false:true;
390}
