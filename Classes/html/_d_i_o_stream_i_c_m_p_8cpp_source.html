345XDWORD DIOSTREAMICMP::Read(XBYTE* buffer, XDWORD size)
346{
347 XDWORD esize = 0;
348
349 int index = GetFirstDatagram(false);
350 if(index != DIOSTREAMICMP_NOTFOUND)
351 {
352 DIOSTREAMICMPDATAGRAM* datagram = GetDatagram(index);
353 if(datagram)
354 {
355 esize = datagram->GetData()->GetSize();
356 if(esize >= size)
357 {
358 esize = size;
359
360 memcpy(buffer,(XBYTE*)datagram->GetData()->Get(), size);
361
362 nbytesread+= size;
363 if(xtimernotactivity) xtimernotactivity->Reset();
364
365 DeleteDatagram(index);
366
367 inbuffer->Extract(buffer, 0, size);
368 }
369 }
370 }
371
372 return esize;
373}

700int DIOSTREAMICMP::GetFirstDatagram(bool tosend)
701{
702 int index = DIOSTREAMICMP_NOTFOUND;
703
704 if(datagramsmutex) datagramsmutex->Lock();
705
706 for(int c=0;c<(int)datagrams.GetSize();c++)
707 {
708 DIOSTREAMICMPDATAGRAM* datagram = (DIOSTREAMICMPDATAGRAM*)datagrams.Get(c);
709 if(datagram)
710 {
711 if(datagram->IsToSend() == tosend)
712 {
713 index = c;
714 break;
715 }
716 }
717 }
718
719 if(datagramsmutex) datagramsmutex->UnLock();
720
