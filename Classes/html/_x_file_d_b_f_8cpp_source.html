580bool XFILEDBF_HEADER::Load(XFILE* xfile)
581{
582 // ---------------------------
583 // Read header
584 if(!xfile->SetPosition(8)) return false;
585
586 if(!xfile->Read((XBYTE*)&headersize,2)) return false;
587 if(!headersize) return 0;
588
589 if(headerdata)
590 {
591 delete[] headerdata;
592 headerdata = NULL;
593 }
594
595 headerdata = new XBYTE[headersize];
596 if(!headerdata) return false;
597
598 if(!xfile->SetPosition(0)) return false;
599 if(!xfile->Read(headerdata,headersize)) return false;
600
601 int index=XFILEDBF_FIELDSTART;
602
603 while(headerdata[index]!=0x0D)
604 {
605 XFILEDBF_FIELDTYPE type;
606
607 switch(headerdata[index+11])
608 {
609 case 'C' : type=XFILEDBF_FIELDTYPE_CHARACTER; break;
610 case 'D' : type=XFILEDBF_FIELDTYPE_DATE; break;
611 case 'F' : type=XFILEDBF_FIELDTYPE_FLOATING; break;
612 case 'N' : type=XFILEDBF_FIELDTYPE_BINARYHEX; break;
613 case 'L' : type=XFILEDBF_FIELDTYPE_LOGICAL; break;
614 case 'M' : type=XFILEDBF_FIELDTYPE_MEMO; break;
615 case 'V' : type=XFILEDBF_FIELDTYPE_VARIABLE; break;
616 case 'P' : type=XFILEDBF_FIELDTYPE_PICTURE; break;
617 case 'B' : type=XFILEDBF_FIELDTYPE_BINARY; break;
618 case 'G' : type=XFILEDBF_FIELDTYPE_GENERAL; break;
619 case '2' : type=XFILEDBF_FIELDTYPE_SHORTINT; break;
620 case '4' : type=XFILEDBF_FIELDTYPE_LONGINT; break;
621 case '8' : type=XFILEDBF_FIELDTYPE_DOUBLE; break;
622 default : type=XFILEDBF_FIELDTYPE_UNKNOWN; break;
623 }
624
625 if(type!=XFILEDBF_FIELDTYPE_UNKNOWN)
626 {
627 XSTRING name((char*)&headerdata[index]);
628
629 if(!CreateField(type, name.Get(), headerdata[index+16])) return false;
630 }
631
632 index+=XFILEDBF_FIELDSIZE;
633 }
634
635 return true;
636}

650bool XFILEDBF_HEADER::Create(XFILE* fileb)
651{
652 headersize = (XWORD)(XFILEDBF_FIELDSTART + fields.GetSize()*XFILEDBF_FIELDSIZE+ 1);
653
654 headerdata = new XBYTE [headersize];
655 if(!headerdata) return false;
656
657 memset(headerdata,0,headersize);
658
659 headerdata[0] = 0x03;
660
661 XDATETIME* GEN_XFACTORY_CREATE(xdatetime, CreateDateTime())
662 if(xdatetime)
663 {
664 xdatetime->Read();
665
666 headerdata[1] = xdatetime->GetYear()-1900;
667 headerdata[2] = xdatetime->GetMonth();
668 headerdata[3] = xdatetime->GetDay();
669
670 delete xdatetime;
671 }
672
673 XDWORD* nf = (XDWORD*)&headerdata[4];
674 (*nf)=0;
675
676 XWORD* hs = (XWORD*)(&headerdata[8]);
677 (*hs)=headersize;
678
679 XWORD* rs = (XWORD*)(&headerdata[10]);
680 (*rs)=GetSizeRecord();
681
682
683 int index=XFILEDBF_FIELDSTART;
684
685 for(XDWORD c=0;c<fields.GetSize();c++)
686 {
687 XFILEDBF_FIELD* field = (XFILEDBF_FIELD*)fields.Get(c);
688 char type = ' ';
689
690 if(field)
691 {
692 memset(&headerdata[index],0,XFILEDBF_FIELDSIZE);
693
694
695 XBUFFER charstr;
696
697 field->name.ConvertToASCII(charstr);
698
699 XDWORD size = (int)strlen(charstr);
700 if(size>XFILEDBF_FIELDSIZE) size = XFILEDBF_FIELDSIZE;
701
702 memcpy((char*)(&headerdata[index]), charstr.Get(), size);
703
704 switch(field->type)
705 {
706 case XFILEDBF_FIELDTYPE_CHARACTER : type='C'; break;
707 case XFILEDBF_FIELDTYPE_DATE : type='D'; break;
708 case XFILEDBF_FIELDTYPE_FLOATING : type='F'; break;
709 case XFILEDBF_FIELDTYPE_BINARYHEX : type='N'; break;
710 case XFILEDBF_FIELDTYPE_LOGICAL : type='L'; break;
711 case XFILEDBF_FIELDTYPE_MEMO : type='M'; break;
712 case XFILEDBF_FIELDTYPE_VARIABLE : type='V'; break;
713 case XFILEDBF_FIELDTYPE_PICTURE : type='P'; break;
714 case XFILEDBF_FIELDTYPE_BINARY : type='B'; break;
715 case XFILEDBF_FIELDTYPE_GENERAL : type='G'; break;
716 case XFILEDBF_FIELDTYPE_SHORTINT : type='2'; break;
717 case XFILEDBF_FIELDTYPE_LONGINT : type='4'; break;
718 case XFILEDBF_FIELDTYPE_DOUBLE : type='8'; break;
719 }
720
721 headerdata[index+11]=type;
722 headerdata[index+16]=field->size;
723
724 index+=XFILEDBF_FIELDSIZE;
725 }
726 }
727
728 headerdata[headersize-1] = 0x0D;
729
730 if(!fileb->SetPosition(0)) return false;
731 if(!fileb->Write(headerdata,headersize)) return false;
732
733 return true;
734}
