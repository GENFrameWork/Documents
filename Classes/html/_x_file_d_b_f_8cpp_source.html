573bool XFILEDBF_HEADER::Load(XFILE* xfile)
574{
575 // ---------------------------
576 // Read header
577 if(!xfile->SetPosition(8)) return false;
578
579 if(!xfile->Read((XBYTE*)&headersize,2)) return false;
580 if(!headersize) return 0;
581
582 if(headerdata)
583 {
584 delete[] headerdata;
585 headerdata = NULL;
586 }
587
588 headerdata = new XBYTE[headersize];
589 if(!headerdata) return false;
590
591 if(!xfile->SetPosition(0)) return false;
592 if(!xfile->Read(headerdata,headersize)) return false;
593
594 int index=XFILEDBF_FIELDSTART;
595
596 while(headerdata[index]!=0x0D)
597 {
598 XFILEDBF_FIELDTYPE type;
599
600 switch(headerdata[index+11])
601 {
602 case 'C' : type=XFILEDBF_FIELDTYPE_CHARACTER; break;
603 case 'D' : type=XFILEDBF_FIELDTYPE_DATE; break;
604 case 'F' : type=XFILEDBF_FIELDTYPE_FLOATING; break;
605 case 'N' : type=XFILEDBF_FIELDTYPE_BINARYHEX; break;
606 case 'L' : type=XFILEDBF_FIELDTYPE_LOGICAL; break;
607 case 'M' : type=XFILEDBF_FIELDTYPE_MEMO; break;
608 case 'V' : type=XFILEDBF_FIELDTYPE_VARIABLE; break;
609 case 'P' : type=XFILEDBF_FIELDTYPE_PICTURE; break;
610 case 'B' : type=XFILEDBF_FIELDTYPE_BINARY; break;
611 case 'G' : type=XFILEDBF_FIELDTYPE_GENERAL; break;
612 case '2' : type=XFILEDBF_FIELDTYPE_SHORTINT; break;
613 case '4' : type=XFILEDBF_FIELDTYPE_LONGINT; break;
614 case '8' : type=XFILEDBF_FIELDTYPE_DOUBLE; break;
615 default : type=XFILEDBF_FIELDTYPE_UNKNOWN; break;
616 }
617
618 if(type!=XFILEDBF_FIELDTYPE_UNKNOWN)
619 {
620 XSTRING name((char*)&headerdata[index]);
621
622 if(!CreateField(type, name.Get(), headerdata[index+16])) return false;
623 }
624
625 index+=XFILEDBF_FIELDSIZE;
626 }
627
628 return true;
629}

643bool XFILEDBF_HEADER::Create(XFILE* fileb)
644{
645 headersize = (XWORD)(XFILEDBF_FIELDSTART + fields.GetSize()*XFILEDBF_FIELDSIZE+ 1);
646
647 headerdata = new XBYTE [headersize];
648 if(!headerdata) return false;
649
650 memset(headerdata,0,headersize);
651
652 headerdata[0] = 0x03;
653
654 XDATETIME* GEN_XFACTORY_CREATE(xdatetime, CreateDateTime())
655 if(xdatetime)
656 {
657 xdatetime->Read();
658
659 headerdata[1] = xdatetime->GetYear()-1900;
660 headerdata[2] = xdatetime->GetMonth();
661 headerdata[3] = xdatetime->GetDay();
662
663 delete xdatetime;
664 }
665
666 XDWORD* nf = (XDWORD*)&headerdata[4];
667 (*nf)=0;
668
669 XWORD* hs = (XWORD*)(&headerdata[8]);
670 (*hs)=headersize;
671
672 XWORD* rs = (XWORD*)(&headerdata[10]);
673 (*rs)=GetSizeRecord();
674
675
676 int index=XFILEDBF_FIELDSTART;
677
678 for(XDWORD c=0;c<fields.GetSize();c++)
679 {
680 XFILEDBF_FIELD* field = (XFILEDBF_FIELD*)fields.Get(c);
681 char type = ' ';
682
683 if(field)
684 {
685 memset(&headerdata[index],0,XFILEDBF_FIELDSIZE);
686
687
688 XBUFFER charstr;
689
690 field->name.ConvertToASCII(charstr);
691
692 XDWORD size = (int)strlen(charstr);
693 if(size>XFILEDBF_FIELDSIZE) size = XFILEDBF_FIELDSIZE;
694
695 memcpy((char*)(&headerdata[index]), charstr.Get(), size);
696
697 switch(field->type)
698 {
699 case XFILEDBF_FIELDTYPE_CHARACTER : type='C'; break;
700 case XFILEDBF_FIELDTYPE_DATE : type='D'; break;
701 case XFILEDBF_FIELDTYPE_FLOATING : type='F'; break;
702 case XFILEDBF_FIELDTYPE_BINARYHEX : type='N'; break;
703 case XFILEDBF_FIELDTYPE_LOGICAL : type='L'; break;
704 case XFILEDBF_FIELDTYPE_MEMO : type='M'; break;
705 case XFILEDBF_FIELDTYPE_VARIABLE : type='V'; break;
706 case XFILEDBF_FIELDTYPE_PICTURE : type='P'; break;
707 case XFILEDBF_FIELDTYPE_BINARY : type='B'; break;
708 case XFILEDBF_FIELDTYPE_GENERAL : type='G'; break;
709 case XFILEDBF_FIELDTYPE_SHORTINT : type='2'; break;
710 case XFILEDBF_FIELDTYPE_LONGINT : type='4'; break;
711 case XFILEDBF_FIELDTYPE_DOUBLE : type='8'; break;
712 }
713
714 headerdata[index+11]=type;
715 headerdata[index+16]=field->size;
716
717 index+=XFILEDBF_FIELDSIZE;
718 }
719 }
720
721 headerdata[headersize-1] = 0x0D;
722
723 if(!fileb->SetPosition(0)) return false;
724 if(!fileb->Write(headerdata,headersize)) return false;
725
726 return true;
727}
