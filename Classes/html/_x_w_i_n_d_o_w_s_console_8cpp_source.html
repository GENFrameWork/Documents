558void XWINDOWSCONSOLE::SendEnterKey()
559{
560 #ifndef APPFLOW_GRAPHICS_NOTCONSOLE_ACTIVE
561
562 INPUT ip;
563 // Set up a generic keyboard event.
564 ip.type = INPUT_KEYBOARD;
565 ip.ki.wScan = 0; // hardware scan code for key
566 ip.ki.time = 0;
567 ip.ki.dwExtraInfo = 0;
568
569 // Send the "Enter" key
570 ip.ki.wVk = 0x0D; // virtual-key code for the "Enter" key
571 ip.ki.dwFlags = 0; // 0 for key press
572 SendInput(1, &ip, sizeof(INPUT));
573
574 // Release the "Enter" key
575 ip.ki.dwFlags = KEYEVENTF_KEYUP; // KEYEVENTF_KEYUP for key release
576 SendInput(1, &ip, sizeof(INPUT));
577
578 #endif
579}
