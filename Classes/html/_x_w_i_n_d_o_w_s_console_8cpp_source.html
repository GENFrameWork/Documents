142bool XWINDOWSCONSOLE::SetSize(int width, int height)
143{
144 HWND consolehandle = GetConsoleWindow();
145 RECT rect;
146 RECT rectdesktop;
147
148 if(GetWindowRect(GetDesktopWindow(), &rectdesktop) == FALSE) return false;
149
150 if(GetWindowRect(consolehandle, &rect) == FALSE) return false;
151
152 rect.left = (rectdesktop.right-width)/2;
153 rect.top = (rectdesktop.bottom-height)/2;
154
155 if(MoveWindow(consolehandle, rect.left, rect.top, width, height, TRUE) == FALSE) return false;
156
157 return true;

476void XWINDOWSCONSOLE::SendEnterKey()
477{
478 INPUT ip;
479 // Set up a generic keyboard event.
480 ip.type = INPUT_KEYBOARD;
481 ip.ki.wScan = 0; // hardware scan code for key
482 ip.ki.time = 0;
483 ip.ki.dwExtraInfo = 0;
484
485 // Send the "Enter" key
486 ip.ki.wVk = 0x0D; // virtual-key code for the "Enter" key
487 ip.ki.dwFlags = 0; // 0 for key press
488 SendInput(1, &ip, sizeof(INPUT));
489
490 // Release the "Enter" key
491 ip.ki.dwFlags = KEYEVENTF_KEYUP; // KEYEVENTF_KEYUP for key release
492 SendInput(1, &ip, sizeof(INPUT));
493}
