597 for(int c=0; c<(int)services.GetSize(); c++)
598 {
599 DIOSTREAMDEVICEBLUETOOTHSDPSERVICE* service = (DIOSTREAMDEVICEBLUETOOTHSDPSERVICE*)services.Get(c);
600 if(service)
601 {
602 if(servicetype == service->GetID())
603 {
604 for(int d=0; d<(int)service->GetNProtocols(); d++)
605 {
606 DIOSTREAMDEVICEBLUETOOTHSDPSERVICEPROTOCOL* protocol = (DIOSTREAMDEVICEBLUETOOTHSDPSERVICEPROTOCOL*)service->GetProtocol(d);

608 {
609 int channel = protocol->GetChannel();

611 }
