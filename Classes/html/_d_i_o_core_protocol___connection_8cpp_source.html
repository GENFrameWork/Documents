740{
741 if(GetEvent() == DIOCOREPROTOCOL_CONNECTION_XFSMEVENT_NONE) // Not new event
742 {
743 switch(GetCurrentState())
744 {
745 case DIOCOREPROTOCOL_CONNECTION_XFSMSTATE_NONE : break;
746
747 case DIOCOREPROTOCOL_CONNECTION_XFSMSTATE_CONNECTED : break;
748
749 case DIOCOREPROTOCOL_CONNECTION_XFSMSTATE_KEYEXCHANGE : { DIOCOREPROTOCOL_HEADER header;
750 XBUFFER content;
751
752 if(!protocol)
753 {
754 break;
755 }
756
757 if(IsServer())
758 {
759 if(GetMsg(false, DIOCOREPROTOCOL_HEADER_OPERATION_KEYEXCHANGE, DIOCOREPROTOCOL_KEYEXCHANGE_CLIENT_OPERATION_PARAM, header, content, protocol->GetProtocolCFG()->GetTimeOutNoResponse()))
760 {
761 if(header.GetContentType() == DIOCOREPROTOCOL_HEADER_CONTENTTYPE_BINARY)
762 {
763 if(content.GetSize() == CIPHERCURVE25519_MAXKEY)

765 ciphercurve.CreateSharedKey(content.Get());
766 cipherkey.Set(ciphercurve.GetKey(CIPHERCURVE25519_TYPEKEY_SHARED), CIPHERCURVE25519_MAXKEY);
767
768 messages.DeleteAll();
769
770 Status_Set(DIOCOREPROTOCOL_CONNECTION_STATUS_KEYEXCHANGE);
771 SetEvent(DIOCOREPROTOCOL_CONNECTION_XFSMEVENT_AUTHENTICATION);
772 }
773 }
774 }
775 }
776 else
777 {
778 if(GetMsg(true, DIOCOREPROTOCOL_HEADER_OPERATION_KEYEXCHANGE, DIOCOREPROTOCOL_KEYEXCHANGE_SERVER_OPERATION_PARAM, header, content, protocol->GetProtocolCFG()->GetTimeOutNoResponse()))
779 {
780 if(header.GetContentType() == DIOCOREPROTOCOL_HEADER_CONTENTTYPE_BINARY)
781 {
782 if(content.GetSize() == CIPHERCURVE25519_MAXKEY)
783 {
784 ciphercurve.CreateSharedKey(content.Get());
785 cipherkey.Set(ciphercurve.GetKey(CIPHERCURVE25519_TYPEKEY_SHARED), CIPHERCURVE25519_MAXKEY);
786
787 static bool sended = false;
788 if(!sended)
789 {
790 XBUFFER publickey;
791
792 publickey.Add(ciphercurve.GetKey(CIPHERCURVE25519_TYPEKEY_PUBLIC), CIPHERCURVE25519_MAXKEY);
793
794 sended = SendMsg(header.GetIDMessage(), DIOCOREPROTOCOL_HEADER_OPERATION_KEYEXCHANGE, DIOCOREPROTOCOL_KEYEXCHANGE_CLIENT_OPERATION_PARAM, &publickey);
795 if(sended)
796 {
797 Status_Set(DIOCOREPROTOCOL_CONNECTION_STATUS_KEYEXCHANGE);
798 SetEvent(DIOCOREPROTOCOL_CONNECTION_XFSMEVENT_AUTHENTICATION);
799 }
800
801 sended = false;
802 }
803 }
804 }
805 }
806 }
807 }
808 break;
809
810 case DIOCOREPROTOCOL_CONNECTION_XFSMSTATE_AUTHENTICATION : { DIOCOREPROTOCOL_HEADER header;
811 XBUFFER content;
812
813 if(!protocol)
814 {
815 break;
816 }
817
818 if(IsServer())
819 {

821 {

823 {
824 Status_Set(DIOCOREPROTOCOL_CONNECTION_STATUS_AUTHENTICATED);
825 SetEvent(DIOCOREPROTOCOL_CONNECTION_XFSMEVENT_REGISTRATION);
826 }

829 SetEvent(DIOCOREPROTOCOL_CONNECTION_XFSMEVENT_DISCONNECTED);
830 }
831 }
832
833 }

835 {
836 if(GetMsg(true, DIOCOREPROTOCOL_HEADER_OPERATION_AUTHENTICATE, DIOCOREPROTOCOL_AUTHENTICATION_CHALLENGE_OPERATION_PARAM, header, content, protocol->GetProtocolCFG()->GetTimeOutNoResponse()))
837 {
838 static bool sended = false;

973 {

988 }

992 {

994
995 SendMsg(&ID_message, DIOCOREPROTOCOL_HEADER_OPERATION_AUTHENTICATE, DIOCOREPROTOCOL_AUTHENTICATION_CHALLENGE_OPERATION_PARAM, GetAuthenticationChallenge());
996 }

1000 {
1001 XUUID ID_message;
1002 XFILEJSON classcontent;

1004 if(!serializationmethod)
1005 {
1006 break;
1007 }
1008
1009 if(!GetRegisterData())
1010 {
1011 break;
1012 }
1013
1014 GetRegisterData()->InitializeData(IsServer());
1015
1016 GetRegisterData()->SetSerializationMethod(serializationmethod);
1017 GetRegisterData()->Serialize();
1018
1019 if(serializationmethod)
1020 {
1021 delete serializationmethod;
1022 }
1023
1024 SendMsg(&ID_message, DIOCOREPROTOCOL_HEADER_OPERATION_REGISTERDATA, DIOCOREPROTOCOL_REGISTRATIONDATA_SEND_OPERATION_PARAM, &classcontent);
1025 }
1026 break;
1027
1028 case DIOCOREPROTOCOL_CONNECTION_XFSMSTATE_READY : Status_Set(DIOCOREPROTOCOL_CONNECTION_STATUS_READY);
