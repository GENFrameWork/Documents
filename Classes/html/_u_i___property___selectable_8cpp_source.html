195UI_PROPERTY_SELECTABLE_STATE UI_PROPERTY_SELECTABLE::SetSelectableStateFromString(XCHAR* selectablestr)
196{
197 if(!selectablestr) return UI_PROPERTY_SELECTABLE_STATE_DEACTIVE;
198
199 XSTRING _selectablestr;
200
201 selectable_state = UI_PROPERTY_SELECTABLE_STATE_DEACTIVE;
202
203 _selectablestr = selectablestr;
204
205 if(!_selectablestr.Compare(__L("active"), true)) selectable_state = UI_PROPERTY_SELECTABLE_STATE_ACTIVE;
206 else if(!_selectablestr.Compare(__L("deactive"), true)) selectable_state = UI_PROPERTY_SELECTABLE_STATE_DEACTIVE;
207 else if(!_selectablestr.Compare(__L("preselect"), true)) selectable_state = UI_PROPERTY_SELECTABLE_STATE_PRESELECT;
208 else if(!_selectablestr.Compare(__L("selected"), true)) selectable_state = UI_PROPERTY_SELECTABLE_STATE_SELECTED;
209
210 return selectable_state;
211}
