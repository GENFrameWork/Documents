130bool DIOSPITOUCHSCREENSTMPE610::IniDevice(int chipselect, int timeout)
131{
132 if(!isdiostreamSPIexternal)
133 {
134 diostreamSPIcfg = new DIOSTREAMSPICONFIG();
135 if(!diostreamSPIcfg) return false;
136
137 diostreamSPIcfg->SetSPIMode(DIOSTREAMSPI_MODE_0);
138 diostreamSPIcfg->SetNBitsWord(8);
139 diostreamSPIcfg->SetSpeed(1*1000*1000);
140 diostreamSPIcfg->SetDelay(0);
141 diostreamSPIcfg->SetIsOnlyWrite(false);
142 diostreamSPIcfg->GetLocalDeviceName()->Format(__L("/dev/spidev0.%d"), chipselect);
143
144 diostreamSPI = (DIOSTREAMSPI*)GEN_DIOFACTORY.CreateStreamIO(diostreamSPIcfg);
145 if(!diostreamSPI) return false;
146 }
147 else
148 {
149 if(!diostreamSPI) return false;
150
151 diostreamSPIcfg = (DIOSTREAMSPICONFIG *)diostreamSPI->GetConfig();
152 if(!diostreamSPIcfg) return false;
153 }
154
155 this->timeout = timeout;
156
157 if(!diostreamSPI->Open()) return false;
158
159 if(!diostreamSPI->WaitToConnected(timeout)) return false;
160
161 if(!TouchScreen_Ini()) return false;
162
163 return DIODEVICE::Ini();
164}

394bool DIOSPITOUCHSCREENSTMPE610::TouchScreen_Ini()
395{
396 if(!WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_SYS_CTRL1, DIOSPITOUCHSCREENSTMPE610_SYS_CTRL1_RESET)) return false;
397
398 GEN_XSLEEP.MilliSeconds(250);
399
400 for(XBYTE c=0; c<65; c++)
401 {
402 XBYTE data;
403 ReadRegister(c, data);
404 }
405
406 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_SYS_CTRL2 , 0x0); // turn on clocks!
407 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_CTRL , DIOSPITOUCHSCREENSTMPE610_TSC_CTRL_XYZ | DIOSPITOUCHSCREENSTMPE610_TSC_CTRL_EN); // XYZ and enable!
408 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_EN , DIOSPITOUCHSCREENSTMPE610_INT_EN_TOUCHDET);
409 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_ADC_CTRL1 , DIOSPITOUCHSCREENSTMPE610_ADC_CTRL1_10BIT | (0x6 << 4)); // 96 clocks per conversion
410 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_ADC_CTRL2 , DIOSPITOUCHSCREENSTMPE610_ADC_CTRL2_6_5MHZ);
411 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_CFG , DIOSPITOUCHSCREENSTMPE610_TSC_CFG_4SAMPLE | DIOSPITOUCHSCREENSTMPE610_TSC_CFG_DELAY_1MS | DIOSPITOUCHSCREENSTMPE610_TSC_CFG_SETTLE_5MS);
412 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_FRACT_XYZ , 0x6);
413 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_TH , 1);
414 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_STA , DIOSPITOUCHSCREENSTMPE610_FIFO_STA_RESET);
415 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_STA , 0); // unreset
416 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_I_DRIVE , DIOSPITOUCHSCREENSTMPE610_TSC_I_DRIVE_50MA);
417 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_STA , 0xFF); // reset all ints
418 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_CTRL , DIOSPITOUCHSCREENSTMPE610_INT_CTRL_POL_HIGH | DIOSPITOUCHSCREENSTMPE610_INT_CTRL_ENABLE);
419
420 return true;
421}

436bool DIOSPITOUCHSCREENSTMPE610::ReadRegister(XBYTE reg, XBYTE& data)
437{
438 //diostreamSPI->GetInXBuffer()->Delete();
439
440 XBYTE regdata[2] = { (XBYTE)(reg | 0x80), 0 };
441 XBYTE _data[2] = { 0, 0 };
442
443 bool status = diostreamSPI->Write(regdata, 2)?true:false;
444 //if(status) status = diostreamSPI->WaitToFlushOutXBuffer(timeout);
445
446 if(status) status = diostreamSPI->WaitToFilledReadingBuffer(2, timeout);
447 if(status) status = diostreamSPI->Read(_data, 2)?true:false;
448
449 data = _data[1];
450
451 return status;
452}

467bool DIOSPITOUCHSCREENSTMPE610::WriteRegister(XBYTE reg, XBYTE data)
468{
469 if(!databuffer) return false;
470
471 bool status = diostreamSPI->Write(&reg, 1)?true:false;
472 if(status) status = diostreamSPI->Write(&data, 1)?true:false;
473 if(status) status = diostreamSPI->WaitToFlushOutXBuffer(timeout);
474 if(status) status = diostreamSPI->WaitToFilledReadingBuffer(2, timeout);
475
476 diostreamSPI->GetInXBuffer()->Delete();
477
478 return status;
479}
