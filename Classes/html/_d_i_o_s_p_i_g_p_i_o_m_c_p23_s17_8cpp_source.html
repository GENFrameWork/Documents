103bool DIOSPIGPIOMCP23S17::IniDevice()

151bool DIOSPIGPIOMCP23S17::Configure()
152{
153 XBYTE ioconfig = DIOSPIGPIOMCP23S17_BANK_OFF |
154 DIOSPIGPIOMCP23S17_INT_MIRROR_OFF |
155 DIOSPIGPIOMCP23S17_SEQOP_OFF |
156 DIOSPIGPIOMCP23S17_DISSLW_OFF |
157 DIOSPIGPIOMCP23S17_HAEN_ON |
158 DIOSPIGPIOMCP23S17_ODR_OFF |
159 DIOSPIGPIOMCP23S17_INTPOL_LOW;
160
161 XBYTE addr = 0;
162
163 if(!Write_Register(DIOSPIGPIOMCP23S17_IOCON, addr, ioconfig)) return false;
164
165 // I/O direction
166 if(!Write_Register(DIOSPIGPIOMCP23S17_IODIRA , addr, 0x00)) return false;
167 if(!Write_Register(DIOSPIGPIOMCP23S17_IODIRB , addr, 0xFF)) return false;
168
169 // GPIOB pull ups
170 if(!Write_Register(DIOSPIGPIOMCP23S17_GPPUB , addr, 0xFF)) return false;
171
172 return true;
173}

189bool DIOSPIGPIOMCP23S17::Read_Register(XBYTE reg, XBYTE addr, XBYTE& data)
190{
191 XBYTE control_byte = GetControlByte(DIOSPIGPIOMCP23S17_READ_CMD, addr);
192 XBYTE tx_buffer[3] = { control_byte, reg, 0 };

194
195 data = 0;
196

267bool DIOSPIGPIOMCP23S17::Write_Bit(XBYTE reg, XBYTE addr, XBYTE bitnum, XBYTE data)
268{
269 XBYTE reg_data = 0;
270
271 if(!Read_Register(reg, addr, reg_data)) return false;
272
273 if(data)
274 {
275 reg_data |= (1 << bitnum); // set
276 }
277 else
278 {
279 reg_data &= 0xff ^ (1 << bitnum); // clear
280 }
281
282 return Write_Register(reg, addr, reg_data);
