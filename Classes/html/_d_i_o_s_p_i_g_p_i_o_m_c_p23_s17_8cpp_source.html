110bool DIOSPIGPIOMCP23S17::IniDevice()

158bool DIOSPIGPIOMCP23S17::Configure()
159{
160 XBYTE ioconfig = DIOSPIGPIOMCP23S17_BANK_OFF |
161 DIOSPIGPIOMCP23S17_INT_MIRROR_OFF |
162 DIOSPIGPIOMCP23S17_SEQOP_OFF |
163 DIOSPIGPIOMCP23S17_DISSLW_OFF |
164 DIOSPIGPIOMCP23S17_HAEN_ON |
165 DIOSPIGPIOMCP23S17_ODR_OFF |
166 DIOSPIGPIOMCP23S17_INTPOL_LOW;
167
168 XBYTE addr = 0;
169
170 if(!Write_Register(DIOSPIGPIOMCP23S17_IOCON, addr, ioconfig)) return false;
171
172 // I/O direction
173 if(!Write_Register(DIOSPIGPIOMCP23S17_IODIRA , addr, 0x00)) return false;
174 if(!Write_Register(DIOSPIGPIOMCP23S17_IODIRB , addr, 0xFF)) return false;
175
176 // GPIOB pull ups
177 if(!Write_Register(DIOSPIGPIOMCP23S17_GPPUB , addr, 0xFF)) return false;
178
179 return true;
180}

197{
198 XBYTE control_byte = GetControlByte(DIOSPIGPIOMCP23S17_READ_CMD, addr);

274bool DIOSPIGPIOMCP23S17::Write_Bit(XBYTE reg, XBYTE addr, XBYTE bitnum, XBYTE data)
275{
276 XBYTE reg_data = 0;
277
278 if(!Read_Register(reg, addr, reg_data)) return false;
279
280 if(data)
281 {
282 reg_data |= (1 << bitnum); // set
283 }
284 else
285 {
286 reg_data &= 0xff ^ (1 << bitnum); // clear
287 }
288
