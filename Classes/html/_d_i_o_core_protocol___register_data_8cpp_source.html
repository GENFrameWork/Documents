332bool DIOCOREPROTOCOL_REGISTERDATA::CreateIDMachine(XUUID& ID)
333{
334 XSTRING origin;
335 XBUFFER originbuffer;
336 HASHSHA2* sha2 = NULL;
337 XBUFFER* result = NULL;
338
339 ID.Empty();
340
341 GEN_XSYSTEM.GetOperativeSystemID(origin);
342
343 origin.AddFormat(__L(" %s"), GEN_XSYSTEM.GetBIOSSerialNumber()->Get());
344 origin.AddFormat(__L(" %s"), GEN_XSYSTEM.GetCPUSerialNumber()->Get());
345
346 origin.ConvertToUTF8(originbuffer);
347
348 sha2 = new HASHSHA2(HASHSHA2TYPE_256);
349 if(!sha2)
350 {
351 return false;
352 }
353
354 sha2->Do(originbuffer);
355 result = sha2->GetResult();
356
357 if(!result)
358 {
359 return false;
360 }
361
362 if(!result->GetSize())
363 {
364 return false;
365 }
366
367 XDWORD data1 = 0;
368 XWORD data2 = 0;
369 XWORD data3 = 0;
370 XBYTE data4 = 0;
371 XBYTE data5 = 0;
372 XBYTE* data6 = &result->Get()[12];
373
374 result->Get((XDWORD&)data1);
375 result->Get((XWORD&)data2);
376 result->Get((XWORD&)data3);
377 result->Get((XBYTE&)data4);
378 result->Get((XBYTE&)data5);
379
380 ID.Set(data1, data2, data3, data4, data5, data6);
381
382 ID.GetToString(origin);
383
384 delete sha2;
385
386 return true;
387}
