775
776 int status = XRRSetScreenConfigAndRate(display, conf, root, c, original_rotation, rates[0], CurrentTime);
777
778 return true;
779 }

801XImage* GRPLINUXSCREENX11::CreateXImageFromBuffer(Display* display, int screen, XBYTE* buffer, int width, int height)
802{
803 XImage* image = NULL;
804 double rratio;
805 double gratio;
806 double bratio;
807 int outindex = 0;
808 int numbufferbytes = (4 * (width * height));
809
810 rratio = vinfo.visual->red_mask / 255.0;
811 gratio = vinfo.visual->green_mask / 255.0;
812 bratio = vinfo.visual->blue_mask / 255.0;
813
814 if(vinfo.depth >= 24)
815 {
816 int numnewbufferbytes = (4 * (width * height));
817 XDWORD* newbuffer = (XDWORD*)malloc(numnewbufferbytes);
818
819 for(int index=0;index<numbufferbytes;)
820 {
821 XDWORD b = (buffer[index++] * bratio);
822 XDWORD g = (buffer[index++] * gratio);
823 XDWORD r = (buffer[index++] * rratio);
824 XDWORD a;
825
826 if(Style_Is(GRPSCREENSTYLE_TRANSPARENT))
827 {
828 a = buffer[index++] << 24;
829 }
830 else
831 {
832 a = 0xFF000000;
833 index++;
834 }
835
836 r &= vinfo.visual->red_mask;
837 g &= vinfo.visual->green_mask;
838 b &= vinfo.visual->blue_mask;
839
840 newbuffer[outindex] = r | g | b | a;
841
842 ++outindex;
843 }
844
845 image = XCreateImage (display, vinfo.visual, vinfo.depth, ZPixmap, 0, (char *)newbuffer,width, height, 32, 0);
846 }
847 else
848 {
849 if(vinfo.depth >= 15)
850 {
851 int numnewbufferbytes = (2 * (width * height));
852 XWORD* newbuffer = (XWORD*)malloc(numnewbufferbytes);
853
854 for(int index=0;index<numbufferbytes;)
855 {
856 XDWORD b = (buffer[index++] * bratio);
857 XDWORD g = (buffer[index++] * gratio);
858 XDWORD r = (buffer[index++] * rratio);
859
860 r &= vinfo.visual->red_mask;
861 g &= vinfo.visual->green_mask;
862 b &= vinfo.visual->blue_mask;
863
864 newbuffer[outindex] = r | g | b;
865
866 ++outindex;
867 }
868
869 image = XCreateImage(display, vinfo.visual, vinfo.depth, ZPixmap, 0, (char*)newbuffer, width, height, 16, 0);
870
871 }
872 else
873 {
874 return NULL;
875 }
876 }
877
878 XInitImage(image);
879
880 if(LSBFirst == GetByteOrder())
881 image->byte_order = LSBFirst;
882 else image->byte_order = MSBFirst;
883
884 image->bitmap_bit_order = MSBFirst;
885
886 return image;
887}

899int GRPLINUXSCREENX11::GetTaskBarHeight()
900{
901 int screen = DefaultScreen(display);

903 int screen_height = DisplayHeight(display, screen);
904

907 if(atomworkarea == None)
908 {
909 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Screen X11] Error Atom WORKAREA (Get Taskbar height)"));
910 return 0;
911 }
912
913 Atom actual_type;
914 int actual_format;
915 unsigned long nitems;
916 unsigned long bytes_after;
917 unsigned char* data = nullptr;
918
919 int status = XGetWindowProperty(display, root, atomworkarea, 0, 4 * sizeof(long), False, XA_CARDINAL, &actual_type, &actual_format, &nitems, &bytes_after, &data);
920 if(status != Success || !data)
921 {
922 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Screen X11] Error XGetWindowProperty (Get Taskbar height)"));
923 return 0;
924 }
925
926 long* workarea = reinterpret_cast<long*>(data);
927 int work_width = workarea[2];
928 int work_height = workarea[3];
929 int taskbar_height = screen_height - work_height;
930
931 XFree(data);
932
933 return taskbar_height;
934}

950 if(netsupported == None)

952 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Screen X11] Error NET_SUPPORTED"));
953 return false;
954 }
955
956 // Get the _NET_SUPPORTED property from the root window
957 Atom type;
958 int format;
959 unsigned long nitems;
960 unsigned long bytesafter;
961 Atom* atoms = nullptr;
962
963 int status = XGetWindowProperty(display, root, netsupported, 0, 1024, False, XA_ATOM, &type, &format, &nitems, &bytesafter, reinterpret_cast<unsigned char **>(&atoms));
964 if(status != Success || type != XA_ATOM || format != 32)
965 {
966 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Screen X11] Failed to retrieve _NET_SUPPORTED property"));
967
968 if(atoms)
969 {
970 XFree(atoms);
971 }
972
973 return 1;
974 }
975
976 // Print the list of supported atoms
977 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Screen X11] _NET_SUPPORTED Atoms [%d]: "), nitems);
978
979 for(unsigned long c=0; c<nitems; ++c)
980 {
981 char* atomname = XGetAtomName(display, atoms[c]);
982 if(atomname)
983 {
984 XSTRING _atomname;
985
986 _atomname = atomname;
987
988 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("%s"), _atomname.Get());
989
990 XFree(atomname);
991 }
992 }
993
994 return true;
995}
