111bool UI_ANIMATION::LoadFromFile(XSTRING& resourcename, GRPPROPERTYMODE mode)
112{
113 XPATH xpathimg;
114 GRPBITMAPFILE bitmapfile;
115 GRPBITMAP* bitmap = NULL;
116 XSTRING mask;
117 int indexmask = 0;
118 int indeximg = 0;
119 GRPPROPERTYMODE modealpha = mode;
120 GRPBITMAPFILE_TYPE typefile = bitmapfile.GetTypeFromExtenxion(resourcename.Get());
121
122 modealpha = ChangeModeWithAlpha(typefile, mode);
123
124 while(1)
125 {
126 XSTRING namefile;
127
128 indexmask = resourcename.Copy(UI_ANIMATION_FRAME_INI, UI_ANIMATION_FRAME_END, true, 0, mask);
129
130 namefile = resourcename;
131
132 if(!mask.IsEmpty())
133 {
134 XSTRING result;
135 int masksize = mask.GetSize()+3;
136
137 namefile.DeleteCharacters(indexmask-masksize, masksize);
138
139 mask.Insert(__L("%"), 0);
140 mask.Add(__L("d"));
141
142 result.Format(mask.Get(), indeximg);
143 indeximg++;
144
145 namefile.Insert(result.Get(), indexmask-masksize);
146 }
147
148 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_GRAPHICS, xpathimg);
149 xpathimg.Slash_Add();
150 xpathimg.Add(namefile);
151
152 if(GEN_USERINTERFACE.IsZippedFile())
153 {
154 XFILEUNZIP* unzipfile = GEN_USERINTERFACE.GetUnzipFile();
155 if(unzipfile)
156 {
157 XPATH pathnamefilecmp(APPFLOW_DEFAULT_DIRECTORY_GRAPHICS);
158 XPATH namefileonly;
159
160 pathnamefilecmp.Slash_Add();
161 pathnamefilecmp += namefile;
162
163 pathnamefilecmp.GetNamefileExt(namefileonly);
164
165 bool status = unzipfile->DecompressFile(pathnamefilecmp, (*GEN_USERINTERFACE.GetUnzipPathFile()), namefileonly.Get());
166 if(status)
167 {
168 XPATH unzippathfile_tmp;
169
170 unzippathfile_tmp = GEN_USERINTERFACE.GetUnzipPathFile()->Get();
171 unzippathfile_tmp += namefileonly;
172
173 bitmap = bitmapfile.Load(unzippathfile_tmp, modealpha);
174
175 GEN_USERINTERFACE.DeleteTemporalUnZipFile(unzippathfile_tmp);
176
177 if(bitmap)
178 {
179 bitmaps.Add(bitmap);
180 if(mask.IsEmpty())
181 {
182 break;
183 }
184 }
185 else
186 {
187 break;
188 }
189 }
190 else
191 {
192 break;
193 }
194 }
195 }
196 else
197 {
198 bitmap = bitmapfile.Load(xpathimg, modealpha);
199 if(bitmap)
200 {
201 bitmaps.Add(bitmap);
202 if(mask.IsEmpty())
203 {
204 break;
205 }
206 }
207 else
208 {
209 break;
210 }
211 }
212 }
213
214 return true;
215}
