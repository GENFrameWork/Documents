779bool GRPSCREEN::CreateViewport(XCHAR* ID, float posx, float posy, float width, float height, XDWORD canvasposx, XDWORD canvasposy, XDWORD canvaswidth, XDWORD canvasheight)
780{
781 GRPVIEWPORT* viewport = new GRPVIEWPORT();
782 if(!viewport) return false;
783
784 viewport->GetID()->Set(ID);
785 viewport->SetPosition(posx, posy);
786 viewport->SetSize(width, height);
787
788 GRPPROPERTIES canvasproperties;
789 float cwidth = (float)canvaswidth;
790 float cheight = (float)canvasheight;
791
792 if(width > cwidth) cwidth = width;
793 if(height > cheight) cheight = height;
794
795 canvasproperties.CopyPropertysFrom(this);
796 canvasproperties.SetPosition(canvasposx, canvasposy);
797 canvasproperties.SetSize((XDWORD)cwidth, (XDWORD)cheight);
798
799 if(viewport->CreateCanvas(canvasproperties))
800 {
801 viewport->SetCanvasPosition(posx, posy);
802 viewports.Add(viewport);
803
804 } else return false;
805
806 return true;
807 }

1010bool GRPSCREEN::Rotate(XBYTE* target, GRPCANVAS* screencanvas)
1011{
1012 XDWORD pixelsize = screencanvas->GetBytesperPixel();
1013 XBYTE* source = screencanvas->Buffer_Get();
1014
1015 if(!source || !target)
1016 {
1017 return false;
1018 }
1019
1020 switch(rotation)
1021 {
1022 case GRPSCREENROTATION_NONE : break;
1023
1024 case GRPSCREENROTATION_90_CLOCKWISE : { int target_offset = 0; //((x * height) + (height - y - 1)) * pixelsize;
1025 int source_offset = 0; //(y * width + x) * pixelsize;
1026
1027 for(XDWORD y=0; y<10 /*height*/; y++)
1028 {
1029 for(XDWORD x=0; x<width; x++)
1030 {
1031 //target_offset = (y * width + x) * pixelsize;
1032 //source_offset = (y * height + (width - x - 1)) * pixelsize;
1033
1034 target_offset += pixelsize;
1035 source_offset += pixelsize;
1036
1037 memcpy(target + target_offset, source + source_offset, pixelsize);
1038 }
1039 }
1040 }
1041 break;
1042
1043 case GRPSCREENROTATION_90_ANTICLOCKWISE :
1044 break;
1045
1046 case GRPSCREENROTATION_180 :
1047 break;
1048 }
1049
1050 return true;
1051}
