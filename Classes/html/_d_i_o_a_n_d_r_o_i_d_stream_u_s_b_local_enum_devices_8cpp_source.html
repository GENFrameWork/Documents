248bool DIOANDROIDSTREAMUSBLOCALENUMDEVICES::SearchAndDeleteEqualDevices()
249{
250 if(devices.IsEmpty()) return false;
251
252 bool found;
253
254 do{ found = false;
255
256 for(int c=0; c<devices.GetSize(); c++)
257 {
258 DIOSTREAMDEVICEUSB* device1 = (DIOSTREAMDEVICEUSB*)devices.Get(c);
259 if(device1)
260 {
261 for(int d=0; d<devices.GetSize(); d++)
262 {
263 DIOSTREAMDEVICEUSB* device2 = (DIOSTREAMDEVICEUSB*)devices.Get(d);
264 if(device2 && (c!=d))
265 {
266 if(!device1->GetName()->Compare(device2->GetName()->Get()))
267 {
268 delete device1;
269 devices.Delete(device1);
270
271 found = true;
272 break;
273 }
274 }
275 }
276
277 if(found) break;
278 }
279 }
280
281 if(!found) break;
282
283 } while(1);
284
285 return true;
286}
