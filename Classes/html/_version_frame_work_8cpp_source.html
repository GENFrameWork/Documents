392bool VERSIONFRAMEWORK::SetAppVersion(XCHAR* app_name, XCHAR* app_execname, XDWORD app_version, XDWORD app_subversion, XDWORD app_versionerror, XCHAR* app_owner, XDWORD app_creationyear)
393{
394 app_titlestr.Empty();
395 app_versionstr.Empty();
396
397 this->app_name = app_name;
398 this->app_execname = app_execname;
399 this->app_version = app_version;
400 this->app_subversion = app_subversion;
401 this->app_versionerror = app_versionerror;
402 this->app_owner = app_owner;
403 this->app_creationyear = app_creationyear;
404
405 app_versionstr.Format(__L("%s %d.%d.%d"), app_name?app_name:__L(""), app_version, app_subversion, app_versionerror);
406
407 XSTRING statusstr;
408
409 GetAppVersionStatus(app_version, app_subversion, app_versionerror, statusstr);
410 app_versionstr += statusstr;
411

413 app_titlestr += __L(" Copyright (c) ");
414
415 XSTRING string2;
416
417 XDATETIME* GEN_XFACTORY_CREATE(xdatetime, CreateDateTime())
418 if(!xdatetime) return false;
419
420 xdatetime->Read();
421 string2.Format(((XDWORD)xdatetime->GetYear()>app_creationyear)?__L("%d-%d "):__L("%d "), app_creationyear, xdatetime->GetYear());
422
423 GEN_XFACTORY.DeleteDateTime(xdatetime);
424
425 app_titlestr += string2;
426 if(app_owner) app_titlestr += app_owner;

428 return true;
