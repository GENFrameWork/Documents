316void DIOSTREAMUDPLOCALENUMSERVERS::ThreadDiscoveryLocalServer(void* data)
317{
318 DIOSTREAMUDPLOCALENUMSERVERS* discoveryenum = (DIOSTREAMUDPLOCALENUMSERVERS*)data;
319 if(!discoveryenum) return;
320
321 if(!discoveryenum->GetDIOStreamUDP()) return;
322 if(!discoveryenum->GetDIOStreamUDPCFG()) return;
323
324 discoveryenum->issearching = true;
325
326 DIOSTREAMUDPCONFIG* diostreamudpcfg = discoveryenum->GetDIOStreamUDPCFG();
327 DIOSTREAMUDP* diostreamudp = discoveryenum->GetDIOStreamUDP();
328
329 diostreamudpcfg->SetMode(DIOSTREAMMODE_SERVER);
330 diostreamudpcfg->SetBroadcastModeActive(true);
331 diostreamudpcfg->SetRemotePort(discoveryenum->GetPort());
332 diostreamudpcfg->SetIsUsedDatagrams(true);
333
334 discoveryenum->xtimer->Reset();
335
336 if(diostreamudp->Open())
337 {
338 while(discoveryenum->issearching)
339 {
340 if(discoveryenum->searchtimeout)
341 {
342 if(discoveryenum->xtimer->GetMeasureSeconds()>=discoveryenum->searchtimeout) break;
343 }
344
345 XBUFFER data;
346 XSTRING address;
347 XWORD port;
348
349 if(diostreamudp->ReadDatagram(address, port, data))
350 {
351 XSTRING string;
352 XSTRING string2(256);
353 XDWORD ID = discoveryenum->GetID();
354 HASHCRC32 crc32;
355 XDWORD crc32result[2];
356
357 data.Extract(crc32result[0]);
358
359 string.Format(DIOSTREAMUDPLOCALENUMSERVERSID,ID);
360
361 string2.Set((XWORD*)data.Get());
362 string2.AdjustSize();
363
364 if(!string.Compare(string2))
365 {
366 XBUFFER xbufferexchange;
367
368 string2.ConvertToExchangeXBuffer(xbufferexchange);
369 crc32.Do(xbufferexchange.Get(), xbufferexchange.GetSize());
370
371 crc32result[1] = crc32.GetResultCRC32();
372
373 if(crc32result[0] == crc32result[1])
374 {
375 if(!discoveryenum->IsDeviceAvailable(address.Get()))
376 {
377 DIOSTREAMDEVICEIP* device = new DIOSTREAMDEVICEIP();
378 if(device)
379 {
380 device->GetIP()->Set(address);
381 discoveryenum->devices.Add(device);
382
383 if(!discoveryenum->searchtimeout) break;
384 }
385 }
386 }
387 }
388 }
389 }
390
391 diostreamudp->WaitToFlushXBuffers(5);
392
393 diostreamudp->Close();
394 }
395
396 discoveryenum->xthread->Run(false);
397
398 discoveryenum->issearching = false;
399}
