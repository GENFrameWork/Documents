434bool SCRIPT::LoadScriptAndRun(XVECTOR<XSTRING*>* listscripts, SCRFUNCADJUSTLIBRARYS adjustlibrarys)
435{
436 if(!listscripts)
437 {
438 return false;
439 }
440
441 bool status = false;
442
443 for(XDWORD c=0; c<listscripts->GetSize(); c++)
444 {
445 XSTRING* linescripts = listscripts->Get(c);
446 if(linescripts)
447 {
448 if(!linescripts->IsEmpty())
449 {
450 XVECTOR<XSTRING*> namescripts;
451
452 linescripts->Split(__C(','), namescripts);
453
454 XSTRING* namescript = namescripts.Get(0);
455 if(namescript)
456 {
457 SCRIPT::EliminateExtraChars(namescript);
458
459 SCRIPT* script = SCRIPT::Create(namescript->Get());
460 if(script)
461 {
462 XPATH allpath;
463 XDWORD ID = 0;
464 bool incache = false;
465
466 if(adjustlibrarys)
467 {
468 adjustlibrarys(script);
469 }
470
471 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, allpath);
472
473 for(XDWORD d=0; d<namescripts.GetSize(); d++)
474 {
475 allpath += __C(',');
476 allpath += namescript->Get();
477 }
478
479 #ifdef SCRIPT_CACHE_ACTIVE
480
481 ID = GEN_SCRIPT_CACHE.GenerateID(allpath);
482 XSTRING* _script = GEN_SCRIPT_CACHE.Cache_Get(ID);
483 if(_script)
484 {
485 (*script->GetScript()) += _script->Get();
486
487 incache = true;
488 status = !_script->IsEmpty();
489 }
490
491 #endif
492
493 if(!incache)
494 {
495 for(XDWORD d=0; d<namescripts.GetSize(); d++)
496 {
497 namescript = namescripts.Get(d);
498 if(namescript)
499 {
500 XPATH xpath;
501
502 SCRIPT::EliminateExtraChars(namescript);
503
504 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, xpath);
505 xpath.Slash_Add();
506 xpath += namescript->Get();
507
508 status = script->LoadAdd(xpath);
509 if(!status)
510 {
511 break;
512 }
513 }
514 }
515 }
516
517 #ifdef SCRIPT_CACHE_ACTIVE
518 if(status && !incache)
519 {
520 GEN_SCRIPT_CACHE.Cache_Add(ID, script->GetScript());
521 }
522 #endif
523
524 if(status)
525 {
526 script->Run();
527 }
528
529 delete script;
530 script = NULL;
531
532 }
533 }
534
535 namescripts.DeleteContents();
536 namescripts.DeleteAll();
537
538 }
539 }
540 }
541
542 return status;
