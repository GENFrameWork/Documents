297bool SCRIPT::Load(XPATH& xpath)
298{
299 #ifdef SCRIPT_CACHE_ACTIVE
300
301 xpath.Slash_Normalize(false);
302
303 XDWORD ID = GEN_SCRIPT_CACHE.GenerateID(xpath);
304
305 XSTRING* _script = GEN_SCRIPT_CACHE.Cache_Get(ID);
306 if(_script)
307 {
308 script.Empty();
309 script += _script->Get();
310
311 GetNameScript()->Format(__L("ID%08X"), ID);
312
313 return true;
314 }
315
316 #endif
317
318 if(!xfiletxt) return false;
319
320 bool status = false;
321
322 this->xpath = xpath;
323
324 xpath.GetNamefileExt(namescript);
325
326 if(xfiletxt->Open(xpath))
327 {
328 if(xfiletxt->ReadAllFile()) status = true;
329
330 script.Empty();

333 {
334 script += xfiletxt->GetLine(c)->Get();

342 #ifdef SCRIPT_CACHE_ACTIVE
343 if(status)
344 {

437bool SCRIPT::LoadScriptAndRun(XVECTOR<XSTRING*>* listscripts, SCRFUNCADJUSTLIBRARYS adjustlibrarys)
438{
439 if(!listscripts)
440 {
441 return false;
442 }
443
444 bool status = false;
445
446 for(XDWORD c=0; c<listscripts->GetSize(); c++)
447 {
448 XSTRING* linescripts = listscripts->Get(c);
449 if(linescripts)
450 {
451 if(!linescripts->IsEmpty())
452 {
453 XVECTOR<XSTRING*> namescripts;
454
455 linescripts->Split(__C(','), namescripts);
456
457 XSTRING* namescript = namescripts.Get(0);
458 if(namescript)
459 {
460 SCRIPT::EliminateExtraChars(namescript);
461
462 SCRIPT* script = SCRIPT::Create(namescript->Get());
463 if(script)
464 {
465 XPATH allpath;
466 XDWORD ID = 0;
467 bool incache = false;
468
469 if(adjustlibrarys)
470 {
471 adjustlibrarys(script);
472 }
473
474 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, allpath);
475
476 for(XDWORD d=0; d<namescripts.GetSize(); d++)
477 {
478 allpath += __C(',');
479 allpath += namescript->Get();
480 }
481
482 #ifdef SCRIPT_CACHE_ACTIVE
483
484 allpath.Slash_Normalize(false);
485 ID = GEN_SCRIPT_CACHE.GenerateID(allpath);
486 XSTRING* _script = GEN_SCRIPT_CACHE.Cache_Get(ID);
487 if(_script)
488 {
489 (*script->GetScript()) += _script->Get();
490
491 incache = true;
492 status = !_script->IsEmpty();
493
494 script->GetNameScript()->Format(__L("ID%08X"), ID);
495 }
496
497 #endif
498
499 if(!incache)
500 {
501 for(XDWORD d=0; d<namescripts.GetSize(); d++)
502 {
503 namescript = namescripts.Get(d);
504 if(namescript)
505 {
506 XPATH xpath;
507
508 SCRIPT::EliminateExtraChars(namescript);
509
510 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, xpath);
511 xpath.Slash_Add();
512 xpath += namescript->Get();
513
514 status = script->LoadAdd(xpath);
515 if(!status)
516 {
517 break;
518 }
519 }
520 }
521 }
522
523 #ifdef SCRIPT_CACHE_ACTIVE
524 if(status && !incache)
525 {
526 GEN_SCRIPT_CACHE.Cache_Add(ID, script->GetScript());
527 }
528 #endif
529
530 if(status)
531 {
532 script->Run();
533 }
534
535 delete script;
536 script = NULL;
537
538 }
539 }
540
541 namescripts.DeleteContents();
542 namescripts.DeleteAll();
543
544 }
545 }
546 }
