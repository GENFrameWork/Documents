294bool SCRIPT::Load(XPATH& xpath, bool add)
295{
296 if(!xfiletxt) return false;
297
298 bool status = false;
299
300 if(add)
301 {
302 AddReturnByType();
303 AddReturnByType();
304 }
305
306 this->xpath = xpath;
307
308 xpath.GetNamefileExt(namescript);
309

316 script.Empty();

320 {
321 script += xfiletxt->GetLine(c)->Get();

323 AddReturnByType();
324 }

326 xfiletxt->Close();
327 }

329 return status;

374bool SCRIPT::LoadScriptAndRun(XVECTOR<XSTRING*>* listscripts, SCRFUNCADJUSTLIBRARYS adjustlibrarys)
375{
376 if(!listscripts)
377 {
378 return false;
379 }
380
381 bool status = false;
382
383 for(XDWORD c=0; c<listscripts->GetSize(); c++)
384 {
385 XSTRING* linescripts = listscripts->Get(c);
386 if(linescripts)
387 {
388 if(!linescripts->IsEmpty())
389 {
390 XVECTOR<XSTRING*> namescripts;
391
392 linescripts->Split(__C(','), namescripts);
393
394 XSTRING* namescript = namescripts.Get(0);
395 if(namescript)
396 {
397 SCRIPT::EliminateExtraChars(namescript);
398
399 SCRIPT* script = SCRIPT::Create(namescript->Get());
400 if(script)
401 {
402 if(adjustlibrarys)
403 {
404 adjustlibrarys(script);
405 }
406
407 for(XDWORD d=0; d<namescripts.GetSize(); d++)
408 {
409 namescript = namescripts.Get(d);
410 if(namescript)
411 {
412 XPATH xpath;
413
414 SCRIPT::EliminateExtraChars(namescript);
415
416 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_SCRIPTS, xpath);
417 xpath.Slash_Add();
418 xpath += namescript->Get();
419
420 status = script->Load(xpath, true);
421 if(!status)
422 {
423 break;
424 }
425 }
426 }
427
428 if(status)
429 {
430 script->Run();
431 }
432
433 delete script;
434 script = NULL;
435
436 }
437 }
438
439 namescripts.DeleteContents();
440 namescripts.DeleteAll();
441
442 }
443 }
444 }
445
446 return status;
447}
