169bool HASH::Do(XPATH& xpath, XQWORD size, XQWORD pos)
170{
171 XFILE* xfile;
172 bool status = false;
173
174 GEN_XFACTORY_CREATE(xfile, Create_File())
175 if(!xfile) return false;
176
177 ResetResult();
178
179 if(xfile->Open(xpath, true))
180 {
181 XQWORD _size;
182
183 if(size == HASHALLFILESIZE) _size = xfile->GetSize(); else _size = size;
184
185 status = Do(xfile, _size, pos);
186
187 xfile->Close();
188 }
189
190 if(xfile) GEN_XFACTORY.Delete_File(xfile);
191
192 return status;

209bool HASH::Do(XFILE* xfile, XQWORD size, XQWORD pos)
210{
211 if(!xfile) return false;
212 if(!xfile->IsOpen()) return false;
213
214 XBUFFER xbuffer;
215 XQWORD block = (((size==HASHALLFILESIZE)?xfile->GetSize():size) / HASHMAXFILESIZEBUFFER);
216 XDWORD mod = (((size==HASHALLFILESIZE)?xfile->GetSize():size) % HASHMAXFILESIZEBUFFER);
217 bool status = true;
218
219 if(!xfile->SetPosition(pos)) return false;
220
221 xbuffer.Resize((XDWORD)HASHMAXFILESIZEBUFFER);
222
223 for(XDWORD c=0;c<block;c++)
224 {
225 if(xfile->Read(xbuffer.Get(), HASHMAXFILESIZEBUFFER))
226 {
227 if(!Do(xbuffer))
228 {
229 status = false;
230 break;
231 }
232 }
233 else
234 {
235 status = false;
236 break;
237 }
238 }
239
240 if(status)
241 {
242 xbuffer.Resize(mod);
243 if(xfile->Read(xbuffer.Get(),mod))
244 {
245 if(!Do(xbuffer))
246 {
247 status = false;
248 }
249 }
250 else
251 {
252 status = false;
253 }
254 }
255
256 return status;
257}
