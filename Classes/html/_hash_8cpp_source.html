176bool HASH::Do(XPATH& xpath, XQWORD size, XQWORD pos)
177{
178 XFILE* xfile;
179 bool status = false;
180
181 GEN_XFACTORY_CREATE(xfile, Create_File())
182 if(!xfile) return false;
183
184 ResetResult();
185
186 if(xfile->Open(xpath, true))
187 {
188 XQWORD _size;
189
190 if(size == HASHALLFILESIZE) _size = xfile->GetSize(); else _size = size;
191

194 xfile->Close();

197 if(xfile) GEN_XFACTORY.Delete_File(xfile);

216bool HASH::Do(XFILE* xfile, XQWORD size, XQWORD pos)
217{
218 if(!xfile) return false;
219 if(!xfile->IsOpen()) return false;
220
221 XBUFFER xbuffer;
222 XQWORD block = (((size==HASHALLFILESIZE)?xfile->GetSize():size) / HASHMAXFILESIZEBUFFER);
223 XDWORD mod = (((size==HASHALLFILESIZE)?xfile->GetSize():size) % HASHMAXFILESIZEBUFFER);
224 bool status = true;
225
226 if(!xfile->SetPosition(pos)) return false;
227
228 xbuffer.Resize((XDWORD)HASHMAXFILESIZEBUFFER);
229
230 for(XDWORD c=0;c<block;c++)
231 {
232 if(xfile->Read(xbuffer.Get(), HASHMAXFILESIZEBUFFER))
233 {
234 if(!Do(xbuffer))
235 {
236 status = false;
237 break;
238 }
239 }
240 else
241 {
242 status = false;
243 break;
244 }
245 }
246
247 if(status)
248 {
249 xbuffer.Resize(mod);
250 if(xfile->Read(xbuffer.Get(),mod))
251 {
252 if(!Do(xbuffer))
253 {
254 status = false;
255 }
256 }
257 else
258 {
259 status = false;
260 }
261 }
262
263 return status;
264}
