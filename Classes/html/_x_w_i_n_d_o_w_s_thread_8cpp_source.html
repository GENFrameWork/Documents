229{
230 statusfunc = XTHREADSTATUS_STOP;
231
232 thhandle = CreateThread( NULL, stacksize, Callback, this, 0, NULL);
233 if(thhandle == NULL)
234 {
235 return false;
236 }
237
238 if(run)
239 {
240 if(!Run(true))
241 {
242 CloseHandle(thhandle);
243 return false;
244 }
245 }
246
247 if((priority != XTHREADPRIORITY_UNKNOWN) && (priority != XTHREADPRIORITY_NORMAL))
248 {
249 switch(priority)
250 {
251 case XTHREADPRIORITY_LOW : SetThreadPriority(thhandle, THREAD_PRIORITY_LOWEST); break;
252 case XTHREADPRIORITY_HIGH : SetThreadPriority(thhandle, THREAD_PRIORITY_HIGHEST); break;
253 case XTHREADPRIORITY_REALTIME : SetThreadPriority(thhandle, THREAD_PRIORITY_TIME_CRITICAL); break;
254 default : break;
255 }
256 }
257
258 return true;
259}
