420bool DIOCLIPROTOCOL::SendCommandArg(XCHAR* command, XSTRING* target, XSTRING* answer, int timeoutanswer, va_list* arg)
421{
422 DIOCLIPROTOCOLCOMMAND* protocolcommand = GetCommand(command);
423 if(!protocolcommand) return false;
424
425 if(!arg) return false;
426
427 XSTRING tosend;
428
429 if(answer)
430 {
431 answer->Empty();
432 }
433 else
434 {
435 tosend.Format(__L("%c"), DIOCLIPROTOCOL_MARK_DONOTANSWER);
436 }
437
438 if(!ID.IsEmpty())
439 {
440 tosend.AddFormat(__L("%s%s"), ID.Get(), DIOCLIPROTOCOL_MARK_ORIGIN);
441 }
442
443 if(target)
444 {
445 if(!target->IsEmpty())
446 {
447 tosend.AddFormat(__L("%s%s"), target->Get(), DIOCLIPROTOCOL_MARK_TARGET);
448 }
449 }
450 else
451 {
452 tosend.AddFormat(__L("%s%s"), DIOCLIPROTOCOL_MARK_BROADCAST, DIOCLIPROTOCOL_MARK_TARGET);
453 }
454
455 tosend.AddFormat(__L("%s"), protocolcommand->GetCommand());
456
457 int nparams = protocolcommand->GetNParams();
458 if(nparams)
459 {
460 tosend.AddFormat(__L(" "));
461
462 for(int c=0;c<nparams;c++)
463 {
464 XCHAR* param = (XCHAR*)va_arg((*arg), XCHAR*);
465 if(param)
466 {
467 tosend += param;
468 if(c != nparams-1) tosend += __L(",");
469 }
470 }
471 }
472
473 if(activeCRC)
474 {
475 HASHCRC32 CRC32;
476
477 XBUFFER charstr;
478
479 tosend.ConvertToASCII(charstr);
480 CRC32.Do((XBYTE*)charstr.Get(), tosend.GetSize());
481
482 XDWORD CRC32result = CRC32.GetResultCRC32();
483 tosend.AddFormat(__L("%s%08X"), DIOCLIPROTOCOL_MARK_CRC32, CRC32result);
484 }
485
486 tosend += __L("\n\r");
487
488
489 XBUFFER charstr;
490
491 tosend.ConvertToASCII(charstr);
492 diostream->Write((XBYTE*)charstr.Get(), tosend.GetSize());
493
494 tosend.DeleteCharacter(__C('\n'));
495 tosend.DeleteCharacter(__C('\r'));
496
497 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Protocol CLI] %s Send Ask: %s -> %s %s"), tosend.Get(), ID.Get(), (target?target->Get():DIOCLIPROTOCOL_MARK_BROADCAST), command);
498
499 bool status = true;
500
501 if(answer)
502 {
503 if(xtimerout) xtimerout->Reset();
504
505 while(1)
506 {
507 if(!xtimerout)
508 {
509 status = false;
510 break;
511 }
512
513 if(exitproccess)
514 {
515 status = false;
516 break;
517 }
518
519 if(xtimerout->GetMeasureSeconds() >= (XDWORD)timeoutanswer)
520 {
521 status = false;
522 break;
523 }
524
525 DIOCLIPROTOCOLANSWER* protocolanswer = GetFirstAnswer(command);
526 if(protocolanswer)
527 {
528 answer->Add(protocolanswer->GetAnswer()->Get());
529 DeleteFirstAnswer(command);
530 break;
531 }
532 }
533 }
534
535 return status;
536}
