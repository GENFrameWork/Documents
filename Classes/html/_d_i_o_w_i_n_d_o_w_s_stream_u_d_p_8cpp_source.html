143bool DIOWINDOWSSTREAMUDP::Open()
144{
145 if(!threadconnection) return false;
146
147 if(!inbuffer) return false;
148 if(!outbuffer) return false;
149 if(!config) return true;
150
151 if(!config->IsServer())
152 {
153 if(config->GetRemoteURL()->IsEmpty())
154 {
155 return false;
156 }
157
158 if(config->GetRemoteURL()->IsAURL())
159 {
160 config->GetRemoteURL()->ResolveURL((*config->GetResolvedRemoteURL()));
161 }
162 else
163 {
164 config->GetResolvedRemoteURL()->Set(config->GetRemoteURL()->Get());
165 }
166 }
167
168 SetEvent(DIOWINDOWSUDPFSMEVENT_GETTINGCONNECTION);
169
170 status = DIOSTREAMSTATUS_GETTINGCONNECTION;
171
172 ResetXBuffers();
173
174 ResetConnectionStatistics();
175
176 threadconnection->SetWaitYield(config->GetThreadWaitYield());
177 threadconnection->SetPriority(config->GetThreadPriority());
178
179 return threadconnection->Ini();
180}

201 {

207 }
208

254int DIOWINDOWSSTREAMUDP::IsReadyConnect(SOCKET socket)
255{
256 struct timeval tv;
257 int rc;
258 fd_set fdr;
259 fd_set fdw;
260 fd_set fds;
261
262 if(socket == INVALID_SOCKET) return -1;
263
264 FD_ZERO(&fdr);
265 FD_ZERO(&fdw);
266 FD_ZERO(&fds);
267
268 FD_SET((unsigned int)socket, &fdr);
269 FD_SET((unsigned int)socket, &fdw);
270 FD_SET((unsigned int)socket, &fds);
271
272 tv.tv_sec = 0;
273 tv.tv_usec = 100;
274
275 rc = select((int)(socket)+1,&fdr, &fdw, &fds, &tv);
276 if(rc == SOCKET_ERROR) return -1;
277
278 int status1 = FD_ISSET(socket, &fdr) ? 1 : 0;
279 int status2 = FD_ISSET(socket, &fdw) ? 1 : 0;
280 int status3 = FD_ISSET(socket, &fds) ? 1 : 0;
281
282 //XTRACE_PRINTCOLOR(1, __L("UDP stream: r: %d, w: %d, s: %d"), status1, status2, status3);
283
284 if(config->IsServer())
285 {
286 if(status1 || status2 ) return 1;
287 if(status3) return -1;
288 }
289 else
290 {
291 if((!status1) && status2) return 1;
292 if(status3) return -1;
293 }
294
295 return 0;
296}
