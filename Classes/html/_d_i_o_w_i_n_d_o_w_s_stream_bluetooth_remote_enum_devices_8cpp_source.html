363void DIOWINDOWSSTREAMBLUETOOTHREMOTEENUMDEVICES::SearchServices()
364{
365 for(int c=0;c<(int)devices.GetSize();c++)
366 {
367 DIOSTREAMDEVICEBLUETOOTH* device = (DIOSTREAMDEVICEBLUETOOTH*)devices.Get(c);
368 if(!device) continue;
369
370 HANDLE hlookup;
371 WSAQUERYSET* qs;
372 DWORD qssize;
373 GUID protocol = L2CAP_PROTOCOL_UUID;
374 DWORD flags = LUP_FLUSHCACHE | LUP_RETURN_ALL;
375 XSTRING MACstring;
376 bool done;
377
378 qssize = sizeof(WSAQUERYSET);
379 qs = (WSAQUERYSET*)new XBYTE[qssize];
380
381 ZeroMemory( qs, qssize );
382
383 qs->dwSize = sizeof(WSAQUERYSET);
384
385 device->GetMAC()->GetXString(MACstring);
386
387 qs->lpServiceClassId = (LPGUID)&protocol;
388 //qs->lpServiceClassId = (LPGUID)&PublicBrowseGroupServiceClass_UUID;
389 qs->dwNameSpace = NS_BTH;
390 qs->dwNumberOfCsAddrs = 0;
391 qs->lpszContext = (LPWSTR)MACstring.Get();
392
393 if(WSALookupServiceBegin( qs, flags, &hlookup) == SOCKET_ERROR)
394 {
395 //int error = WSAGetLastError();
396 //XTRACE_PRINT(__L("Search SDP Services Error %d"), error);
397
398 delete [] qs;
399
400 continue;
401 }
402
403 done = false;
404 while(!done)
405 {
406 if(SOCKET_ERROR == WSALookupServiceNext(hlookup, flags, &qssize, qs))
407 {
408 int error = WSAGetLastError();
409 if( error == WSAEFAULT )
410 {
411 delete [] qs;
412 qs = (WSAQUERYSET*)new XBYTE[qssize];
413 }
414 else
415 {
416 if(error == WSA_E_NO_MORE )
417 {
418 done = true;
419 }
420 else
421 {
422 break;
423 }
424 }
425 }
426 else
427 {
428 DIOSTREAMDEVICEBLUETOOTHSDPSERVICE* service = new DIOSTREAMDEVICEBLUETOOTHSDPSERVICE();
429 if(service)
430 {
431 service->GetName()->Set(qs->lpszServiceInstanceName);
432 service->GetDescription()->Set(qs->lpszComment);
433
434 const BLOB* pblob = (BLOB*)qs->lpBlob;
435
436 if(pblob) BluetoothSdpEnumAttributes(pblob->pBlobData, pblob->cbSize, SDP_ServiceCallback, service);
437
438 device->Service_Add(service);
439 }
440 }
441 }
442
443 delete [] qs;
444
445 WSALookupServiceEnd(hlookup);
446
447 }
