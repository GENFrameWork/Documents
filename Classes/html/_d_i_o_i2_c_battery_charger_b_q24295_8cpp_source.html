157void DIOI2CBATTERYCHARGERBQ24295::Update()
158{
159 if(!charging) return;
160
161 // For an unknown reason the stat pin generates an square signal and not holds the low level when is charging
162 // so I have to read the register to know when the charge is done
163 //I read the register every 5 seconds
164
165 XBYTE read_reg_fault;
166 XBYTE read_reg_status;
167
168 if(ReadRegister(DIOI2CBATTERYCHARGERBQ24295_FAULT_REG_REG09 , read_reg_fault) != 0) return; //TODO-GFR detect FAULT errors
169 if(ReadRegister(DIOI2CBATTERYCHARGERBQ24295_SYS_STATUS_REG08 , read_reg_status) != 0) return;
170
171 if(((read_reg_status >> 4) & 0x03) == 0x03) fullycharge = true;
172}

327bool DIOI2CBATTERYCHARGERBQ24295::Initialize()
328{
329 // Charger GPIO Inputs
330 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_STAT] , DIOGPIO_MODE_INPUT);
331
332 // Charger GPIO Outputs
333 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_QON] , DIOGPIO_MODE_OUTPUT);
334 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_CE] , DIOGPIO_MODE_OUTPUT);
335
336 // Charger GPIO Interrupts
337 DIOGPIO_ENTRY* interrupt = GEN_DIOGPIO.GPIOEntry_GetByID(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_INT]);
338 if(interrupt)
339 {
340 interrupt->SetIntFunctionPointer(DIOI2CBATTERYCHARGERBQ24295::Handler_Interrupt);
341 interrupt->SetIntParamPointer((void*)this);
342 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_INT], DIOGPIO_MODE_INTERRUPT);
343 }
344
345 // Charger disabled
346 GEN_DIOGPIO.SetValue(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_CE] , true);
347
348 // Shipping Mode off
349 GEN_DIOGPIO.SetValue(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_QON] , false);
350
351 charging = false;
352 fault = 0;
353 fullycharge = false;
354
355 // Configure the charger ILMAX to 500mA
356 XBYTE read_reg;
357
358 if(!ReadRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
359
360 // IINLIM bits (010) (500mA)
361 read_reg &= 0xF8;
362 read_reg |= 0x02;
363
364 if(!WriteRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
365
366 // Disable the charger watchdog timer and the safety timer
367 if(!ReadRegister(DIOI2CBATTERYCHARGERBQ24295_CHRG_TERM_TIMER_CTRL_REG05, read_reg)) return false;
368
369 // Disable timer
370 read_reg &= 0xC7;
371 if(!WriteRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
372
373 return true;
374}

409bool DIOI2CBATTERYCHARGERBQ24295::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
410{
411 if(!diostream) return false;
412
413 bool status;
414
415 status = diostream->Write(&offset, 1)?true:false;
416 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
417
418 if(!status) return false;
419
420 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
421 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
422
423 return status;
424}

464
