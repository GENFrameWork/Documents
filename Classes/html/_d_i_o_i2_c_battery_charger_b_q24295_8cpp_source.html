150void DIOI2CBATTERYCHARGERBQ24295::Update()
151{
152 if(!charging) return;
153
154 // For an unknown reason the stat pin generates an square signal and not holds the low level when is charging
155 // so I have to read the register to know when the charge is done
156 //I read the register every 5 seconds
157
158 XBYTE read_reg_fault;
159 XBYTE read_reg_status;
160
161 if(ReadRegister(DIOI2CBATTERYCHARGERBQ24295_FAULT_REG_REG09 , read_reg_fault) != 0) return; //TODO-GFR detect FAULT errors
162 if(ReadRegister(DIOI2CBATTERYCHARGERBQ24295_SYS_STATUS_REG08 , read_reg_status) != 0) return;
163
164 if(((read_reg_status >> 4) & 0x03) == 0x03) fullycharge = true;
165}

320bool DIOI2CBATTERYCHARGERBQ24295::Initialize()
321{
322 // Charger GPIO Inputs
323 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_STAT] , DIOGPIO_MODE_INPUT);
324
325 // Charger GPIO Outputs
326 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_QON] , DIOGPIO_MODE_OUTPUT);
327 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_CE] , DIOGPIO_MODE_OUTPUT);
328
329 // Charger GPIO Interrupts
330 DIOGPIO_ENTRY* interrupt = GEN_DIOGPIO.GPIOEntry_GetByID(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_INT]);
331 if(interrupt)
332 {
333 interrupt->SetIntFunctionPointer(DIOI2CBATTERYCHARGERBQ24295::Handler_Interrupt);
334 interrupt->SetIntParamPointer((void*)this);
335 GEN_DIOGPIO.SetMode(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_INT], DIOGPIO_MODE_INTERRUPT);
336 }
337
338 // Charger disabled
339 GEN_DIOGPIO.SetValue(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_CE] , true);
340
341 // Shipping Mode off
342 GEN_DIOGPIO.SetValue(GPIOentryID[DIOI2CBATTERYCHARGERBQ24295_GPIOENTRYID_QON] , false);
343
344 charging = false;
345 fault = 0;
346 fullycharge = false;
347
348 // Configure the charger ILMAX to 500mA
349 XBYTE read_reg;
350
351 if(!ReadRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
352
353 // IINLIM bits (010) (500mA)
354 read_reg &= 0xF8;
355 read_reg |= 0x02;
356
357 if(!WriteRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
358
359 // Disable the charger watchdog timer and the safety timer
360 if(!ReadRegister(DIOI2CBATTERYCHARGERBQ24295_CHRG_TERM_TIMER_CTRL_REG05, read_reg)) return false;
361
362 // Disable timer
363 read_reg &= 0xC7;
364 if(!WriteRegister(DIOI2CBATTERYCHARGERBQ24295_INPUT_SRC_CTRL_REG00, read_reg)) return false;
365
366 return true;
367}

402bool DIOI2CBATTERYCHARGERBQ24295::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
403{
404 if(!diostream) return false;
405
406 bool status;
407
408 status = diostream->Write(&offset, 1)?true:false;
409 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
410
411 if(!status) return false;
412
413 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
414 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
415
416 return status;
417}
