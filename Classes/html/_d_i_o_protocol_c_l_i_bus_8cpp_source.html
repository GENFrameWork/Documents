339bool DIOPROTOCOLCLIBUS::ReceivedCommand(XSTRING& originID, XSTRING& command, XVECTOR<XSTRING*>& params, XSTRING& answer)
340{
341 bool status = false;
342
343 answer.Empty();
344
345 DIOPROTOCOLCLICOMMAND* protocolcommand = GetCommand(command.Get());
346 if(protocolcommand)
347 {
348 if(protocolcommand->GetNParams() < (int)params.GetSize()) return false;
349
350 if(!command.Compare(DIOPROTOCOLCLIBUS_COMMAND_VERSION, true))
351 {
352 answer.Format(__L("%d.%d.%d"), version, subversion, subversionerror);
353 status = true;
354 }
355
356 if(!command.Compare(DIOPROTOCOLCLIBUS_COMMAND_PING, true))
357 {
358 status = true;
359 }
360
361 if(!command.Compare(DIOPROTOCOLCLIBUS_COMMAND_ENUM, true))
362 {
363 enum_mutex->Lock();
364
365 enum_sendoriginID = originID;
366
367 enum_mutex->UnLock();
368
369 threadsendenumrequest->Ini();
370 }
371
372 if(!command.Compare(DIOPROTOCOLCLIBUS_COMMAND_ENUMREQUEST, true))
373 {
374 bool found = false;
375
376 for(XDWORD c=0; c<enum_remotedevices.GetSize(); c++)
377 {
378 XSTRING* device = enum_remotedevices.Get(c);
379 if(device)
380 {
381 if(!device->Compare(originID, true)) found = true;
382 }
383 }
384
385 if(!found)
386 {
387 XSTRING* origin = new XSTRING();
388 if(origin)
389 {
390 (*origin) = originID;
391 enum_remotedevices.Add(origin);
392 }
393 }
394
395 status = true;
396 }
397 }
398
399 return status;
400}
