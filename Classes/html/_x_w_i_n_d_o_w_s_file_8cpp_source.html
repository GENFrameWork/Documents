480FILE* XWINDOWSFILE::CreateStructHandle()
481{
482 if(fp != NULL) return fp;
483
484 switch(mode)
485 {
486 case XWINDOWSFILEMODE_READONLY : fd = _open_osfhandle((intptr_t)filehandle, _O_RDONLY);

534bool XWINDOWSFILE::ActualizeSize()
535{
536 if(!isopen) return false;
537 if(!filehandle) return false;
538
539 if(!isreadonly)
540 {
541 if(!Flush()) return false;
542 }
543
544 LARGE_INTEGER exsize;
545 if(GetFileSizeEx(filehandle, &exsize) == INVALID_FILE_SIZE) return false;
546
547 cachesize = exsize.HighPart;
548 cachesize <<=32;
549 cachesize |= exsize.LowPart;
550
551 return true;
552}
