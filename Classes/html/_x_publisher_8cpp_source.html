451bool XPUBLISHER::PostEvent(XEVENT* event, XSUBJECT* subject)
452{
453 if(!on) return false;
454
455 inevent = true;
456
457 if(xmutexevents) xmutexevents->Lock();
458
459 XVECTOR<XOBSERVER*> observers;
460
461 for(XDWORD c=0; c<eventsvector.GetSize(); c++)
462 {
463 XPUBLISHERENTRY* publisherentry = eventsvector.Get(c);
464 if(publisherentry)
465 {
466 if((publisherentry->type == event->GetEventType()) && (publisherentry->subject == subject))
467 {
468 observers.Add(publisherentry->observer);
469 }
470 }
471 }
472
473 if(xmutexevents) xmutexevents->UnLock();
474
475 for(XDWORD c=0; c<observers.GetSize(); c++)
476 {
477 XOBSERVER* observer = observers.Get(c);
478 if(observer) observer->HandleEvent(event);
479 }
480
481 observers.DeleteAll();
482
483 inevent = false;
484
485 return true;
486}
