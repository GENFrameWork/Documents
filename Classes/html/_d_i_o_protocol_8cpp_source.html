1883bool DIOPROTOCOL::SendCommand(XDWORD type, XDWORD& ID, XBUFFER& xbuffer)
1884{
1885 if(!IsConnected()) return false;
1886
1887 HASHCRC32 crc32;
1888 XDWORD crc32head = 0;
1889 XDWORD crc32body = 0;
1890
1891 DIOPROTOCOL_COMMAND* cmd = GetCommand(type&~DIOPROTOCOL_CMDTYPE_ANSWER);
1892 if(!cmd) return false;
1893
1894 if(xmutexwrites) xmutexwrites->Lock();
1895
1896 XBYTE head[DIOPROTOCOL_SIZEOFHEAD];
1897 XDWORD* ptr_head = (XDWORD*)head;
1898
1899 memset(head, 0, DIOPROTOCOL_SIZEOFHEAD);
1900
1901 if(xbuffer.GetSize())
1902 {
1903 crc32.ResetResult();
1904 crc32.Do(xbuffer);
1905 crc32body = crc32.GetResultCRC32();
1906 }
1907
1908 if(!ID)
1909 {
1910 ID = IDmsgcounter;
1911 ID |= maskID;
1912
1913 IDmsgcounter++;
1914 if(IDmsgcounter > DIOPROTOCOL_CMDTYPE_SECUENCIALLIMIT) IDmsgcounter = 0;
1915 }
1916
1917 ptr_head[0] = DIOPROTOCOL_MSGIDHEAD;
1918 ptr_head[1] = type;
1919 ptr_head[2] = xbuffer.GetSize();
1920 ptr_head[3] = ID;
1921 ptr_head[4] = crc32body;
1922
1923 crc32.ResetResult();
1924 crc32.Do(head, DIOPROTOCOL_SIZEOFHEAD-sizeof(XDWORD));
1925 crc32head = crc32.GetResultCRC32();
1926
1927 ptr_head[5] = crc32head;
1928
1929 bool status = false;
1930
1931 if(xbuffer.GetSize())
1932 {
1933 XBUFFER xbuffersend;
1934
1935 xbuffersend.Add(head, DIOPROTOCOL_SIZEOFHEAD);
1936 xbuffersend.Add(xbuffer);
1937
1938 status = (diostream->Write(xbuffersend) == xbuffersend.GetSize())?true:false;
1939 if(status) diostream->WaitToFlushOutXBuffer(5);
1940 }
1941 else
1942 {
1943 status = (diostream->Write(head, DIOPROTOCOL_SIZEOFHEAD) == DIOPROTOCOL_SIZEOFHEAD)?true:false;
1944 if(status) status = diostream->WaitToFlushOutXBuffer(5);
1945 }
1946
1947 if(xmutexwrites) xmutexwrites->UnLock();
1948
1949 return status;
1950}
