227{
228 DIOLINUXSTREAMWIFIREMOTEENUMDEVICES* enumeration = (DIOLINUXSTREAMWIFIREMOTEENUMDEVICES*)param;
229 if(!enumeration) return;

233 switch(enumeration->GetCurrentState())

235 case DIOLINUXWIFIENUMFSMSTATE_NONE : break;

251 case DIOLINUXWIFIENUMFSMSTATE_NONE : break;
252
253 case DIOLINUXWIFIENUMFSMSTATE_SEARCH : { wireless_scan_head head;
254 wireless_scan* result = NULL;
255 iwrange range;
