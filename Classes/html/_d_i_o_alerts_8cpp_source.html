428XDWORD DIOALERT::CalculateID(bool withdatetime)
429{
430 if(!xdatetime) return false;
431
432 HASHCRC32* hashcrc32 = new HASHCRC32();
433 if(!hashcrc32) return 0;
434
435 XSTRING IDstring;
436 XSTRING levelstring;
437 XDWORD ID = 0;
438 bool status;
439
440 if(withdatetime) xdatetime->GetDateTimeToString(XDATETIME_FORMAT_STANDARD, IDstring);
441
442 levelstring.Format(__L("%s %d"), applicationID.Get(), level);
443 IDstring += levelstring;
444
445 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
446 if(xrand)
447 {
448 levelstring.Format(__L(" %d "), xrand->Between(1000, 10000000) * xrand->Between(1000, 10000000));
449 GEN_XFACTORY.DeleteRand(xrand);
450 }
451
452 IDstring += __L(" "); IDstring += origin;
453 IDstring += __L(" "); IDstring += title;
454 IDstring += __L(" "); IDstring += message;
455 IDstring += levelstring;
456
457 XBUFFER charstr;
458
459 IDstring.ConvertToASCII(charstr);
460 status = hashcrc32->Do(charstr.Get(), IDstring.GetSize());
461
462 if(status) ID = hashcrc32->GetResultCRC32(); else ID = 0;
463
464 delete hashcrc32;
465
466 return ID;
467}

935 va_list arg;

937 va_start(arg, nrecipients);

1257 }
1258
1259 if(statussend) condition->GetLastTimerSend()->Reset();

1724bool DIOALERTS::Sender_UDPSend(DIOALERT* alert)
