510 void FillList (XAVLNODE<K>* n, const K& key, XLIST<K>* list)
511 {
512 // finds all ocurrences of nodes in the tree taking advantage of rotations being invariant
513 if(n == NULL)
514 {
515 return;
516 }
517
518 // in order visit
519 // visit left
520 FillList(n->left, key, list);
521
522 // visit me
523 //if I'm bigger return and don't continue

525 {

529 {
530 // neither bigger or smaller, must be equal
531 list->Add(n->key);
532 }
533
534 // visit right
