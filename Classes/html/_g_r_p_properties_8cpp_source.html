444 case GRPPROPERTYMODE_XX_UNKNOWN : break;

539int GRPPROPERTIES::IsEqualSizeTo(GRPPROPERTIES* properties)
540{
541 if(!properties) return ISLESS;
542
543 int size[2];
544
545 size[0] = width * height;
546 size[1] = properties->width * properties->height;
547
548 if(size[0] > size[1]) return ISGREATER;
549 if(size[0] < size[1]) return ISLESS;
550
551 return ISEQUAL;
