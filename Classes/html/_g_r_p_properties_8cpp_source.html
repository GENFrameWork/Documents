446XBYTE GRPPROPERTIES::GetBitsperPixel()
447{
448 XBYTE bitsppixel = 0;
449
450 switch(mode)
451 {
452 case GRPPROPERTYMODE_XX_UNKNOWN : break;
453 case GRPPROPERTYMODE_08_INDEX : bitsppixel = 8; break;
454
455 case GRPPROPERTYMODE_16_RGBA_4444 :
456 case GRPPROPERTYMODE_16_RGB_565 :
457 case GRPPROPERTYMODE_16_SRGB_565 :
458 case GRPPROPERTYMODE_16_RGBA_5551 : bitsppixel = 16; break;
459
460 case GRPPROPERTYMODE_24_RGB_888 :
461 case GRPPROPERTYMODE_24_BGR_888 : bitsppixel = 24; break;
462
463 case GRPPROPERTYMODE_32_RGBA_8888 :
464 case GRPPROPERTYMODE_32_BGRA_8888 : bitsppixel = 32; break;
465
466 default : bitsppixel = 0; break;
467 }

549int GRPPROPERTIES::IsEqualSizeTo(GRPPROPERTIES* properties)
550{
551 if(!properties) return ISLESS;
552
553 int size[2];
554
555 size[0] = width * height;
556 size[1] = properties->width * properties->height;
557
558 if(size[0] > size[1]) return ISGREATER;
559 if(size[0] < size[1]) return ISLESS;
560
561 return ISEQUAL;
562}
