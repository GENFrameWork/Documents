137bool INPLINUXDEVICEKEYBOARDDIRECT::Update()
138{
139 if((!created)||(!enabled)) return false;
140
141 int bytesread = 0;
142
143 if(old_fd != -1)
144 {
145 XBYTE buffer[INPLINUXDEVICEKEYBOARDDIRECT_OLDMAXBUFFER];
146
147 for(XDWORD e=0; e<keyboards.GetSize(); e++)
148 {
149 errno = 0;
150 bytesread = read(keyboards.Get(e)->GetFileDescriptor(), buffer, sizeof(buffer));
151 if(bytesread == -1 && errno == ENODEV)
152 {
153 // we lost the handle to the event file, need to reclaim it
154 keyboards.DeleteContents();
155 Old_CreateDevices();
156 }
157
158 if(bytesread)
159 {
160 for(int c=0; c<bytesread; c++)
161 {
162 INPBUTTON* button = GetButtonByCode((buffer[c] & 0x7f));
163 if(button)
164 {
165 button->SetPressed((buffer[c] & 0x80) != 0x80);
166
167 if(button->IsPressed())
168 {
169 if(button->GetState() == INPBUTTON_STATE_PRESSED)
170 {
171 button->SetState(INPBUTTON_STATE_HOLD);
172 }
173 else
174 {
175 button->SetState(INPBUTTON_STATE_PRESSED);
176 }
177 }
178 else
179 {
180 button->SetState(INPBUTTON_STATE_RELEASED);
181 }
182
183 //XTRACE_PRINTCOLOR(1,__L("Keypres : code (%d) on : %s "), (buffer[c] & 0x7f), (button->IsPressed()?__L("Press"):__L("Release")));
184 }
185 }
186 }
187 }
188 }
189 else
190 {
191 int eventsize = sizeof(struct input_event);
192 struct input_event event;
193
194 for(XDWORD e=0; e<keyboards.GetSize(); e++)
195 {
196 errno = 0;
197 do{ bytesread = read(keyboards.Get(e)->GetFileDescriptor(), &event, eventsize);
198 if(bytesread == -1 && errno == ENODEV)
199 {
200 // we lost the handle to the event file, need to reclaim it
201 keyboards.DeleteContents();
202 CreateDevices();
203 }
204
205 if(bytesread == eventsize)
206 {
207 if(EV_KEY == event.type)
208 {
209 INPBUTTON* button = GetButtonByCode(event.code);
210 if(button)
211 {
212 button->SetPressed(event.value!=0);
213 //XTRACE_PRINTCOLOR(1,__L("Keypres : code (%d) on : %s "), event.code, (button->IsPressed()?__L("Press"):__L("Release")));
214
215 } else continue;
216
217 switch(event.value)
218 {
219 case 0: button->SetState(INPBUTTON_STATE_RELEASED);
220 break;
221
222 case 1: if(button->GetState() == INPBUTTON_STATE_PRESSED)
223 {
224 button->SetState(INPBUTTON_STATE_HOLD);
225 }
226 else
227 {
228 button->SetState(INPBUTTON_STATE_PRESSED);
229 }
230 break;
231
232 case 2: button->SetState(INPBUTTON_STATE_HOLD);
233 break;
234 }
235 }
236 }
237
238 } while(bytesread == eventsize);
239 }
240 }
241
242 return true;
243}

383bool INPLINUXDEVICEKEYBOARDDIRECT::CreateDevices()
384{
385 INPLINUXFACTORY* inpdevices = (INPLINUXFACTORY*)(&INPFACTORY::GetInstance());
386 XVECTOR<INPLINUXDEVICEID*> devicesID;
387
388 if(inpdevices) inpdevices->GetDeviceHandlers(INPDEVICE_TYPE_KEYBOARD, devicesID);
389
390 for(XDWORD c=0; c<devicesID.GetSize(); c++)
391 {
392 INPLINUXDEVICEID* deviceID = devicesID.Get(c);
393 if(deviceID)
394 {
395 XSTRING event;
396 int filedescriptor = INPLINUXDEVICEID_INVALID;
397
398 event.Format(__L("/dev/input/event%d"), deviceID->GetEventIndex());
399
400 XBUFFER charstr;
401
402 event.ConvertToASCII(charstr);
403 filedescriptor = open(charstr.GetPtrChar(), O_RDONLY | O_NONBLOCK);
404
405 if(filedescriptor != -1)
406 {
407 char keyboardOEMname[_MAXSTR];
408 XSTRING namehardware;
409
410 // we get the keyboard name
411 memset(keyboardOEMname, 0, _MAXSTR);
412
413 ioctl(filedescriptor, EVIOCGNAME(_MAXSTR), keyboardOEMname);
414 namehardware.Set(keyboardOEMname);
415
416 deviceID->SetFileDescriptor(filedescriptor);
417 //deviceID->GetName()->Set(namehardware);
418
419 keyboards.Add(deviceID);
420 }
421 }
422 }
423
424 devicesID.DeleteAll();
425
426 return keyboards.GetSize()?true:false;
427}

439bool INPLINUXDEVICEKEYBOARDDIRECT::Old_CreateDevices()
440{
441 struct termios newsettings;
442
443 old_fd = Old_GetFD(NULL);
444
445 if(ioctl(old_fd, KDGKBMODE, &old_keyboardmode)) return false;
446
447 if(tcgetattr(old_fd, &old_termiosinitialsettings) == -1) return false;
448 if(tcgetattr(old_fd, &newsettings) == -1) return false;
449
450 newsettings.c_lflag &= ~(ICANON | ECHO | ISIG);
451 newsettings.c_iflag = 0;
452 newsettings.c_cc[VMIN] = INPLINUXDEVICEKEYBOARDDIRECT_OLDMAXBUFFER;
453 newsettings.c_cc[VTIME] = 1; /* 0.1 sec intercharacter timeout */
454
455 if(tcsetattr(old_fd, TCSAFLUSH, &newsettings) == -1) return false;
456
457 if(ioctl(old_fd, KDSKBMODE, K_MEDIUMRAW)) return false;
458
459 if(old_fd != -1)
460 {
461 INPLINUXDEVICEID* deviceID = NULL;
462
463 deviceID = new INPLINUXDEVICEID();
464 if(!deviceID) return false;
465
466 deviceID->SetFileDescriptor(old_fd);
467 keyboards.Add(deviceID);
468
469 //XTRACE_PRINTCOLOR((deviceID?1:4),__L("Keyboard handle %d"), old_fd);
470 }
471
472 return true;
473}

531int INPLINUXDEVICEKEYBOARDDIRECT::Old_OpenConsole(const char *fnam)
532{
533 int fd;
534
535 /*
536 * For ioctl purposes we only need some fd and permissions
537 * do not matter. But setfont:activatemap() does a write.
538 */
539 fd = open(fnam, O_RDWR | O_NONBLOCK);
540 if(fd < 0 && errno == EACCES) fd = open(fnam, O_WRONLY | O_NONBLOCK);
541 if (fd < 0 && errno == EACCES) fd = open(fnam, O_RDONLY | O_NONBLOCK);
542 if (fd < 0) return -1;
543
544 if(!Old_IsConsole(fd))
545 {
546 close(fd);
547 return -1;
548 }
549
550 return fd;
551}

565int INPLINUXDEVICEKEYBOARDDIRECT::Old_GetFD(const char *fnam)
566{
567 int fd;
568
569 if (fnam)
570 {
571 fd = Old_OpenConsole(fnam);
572 if (fd >= 0) return fd;
573
574 return -1;
575 }
576
577 fd = Old_OpenConsole("/dev/tty");
578 if (fd >= 0) return fd;
579
580 fd = Old_OpenConsole("/dev/tty0");
581 if (fd >= 0) return fd;
582
583 fd = Old_OpenConsole("/dev/vc/0");
584 if (fd >= 0) return fd;
585
586 fd = Old_OpenConsole("/dev/console");
587 if(fd >= 0) return fd;
588 {
589 for(fd = 0; fd < 3; fd++)
590 {
591 if (Old_IsConsole(fd)) return fd;
592 }
593 }
594
595 return -1;
596}
