673bool XLOGBASE::AddEntry(XLOGLEVEL level, XCHAR* sectionID, bool inmemory, XBYTE* data, XDWORD size, XDWORD sizeline, bool showoffset, bool showtext)
674{
675 if(!IsActive()) return false;
676 if(!mutex) return false;
677
678 mutex->Lock();
679
680 if(!PassFilters(level, sectionID))
681 {
682 mutex->UnLock();
683 return false;
684 }
685
686 XSTRING strdata;
687 XDWORD _size = 0;
688 XDWORD _sizeline = sizeline;
689 int index = 0;
690 XDWORD lines = 0;
691
692 while(_size<size)
693 {
694 XSTRING string;
695
696 if(_sizeline > (size - _size)) _sizeline = (size - _size);
697
698 if(showoffset)
699 {
700 strdata.Format(__L("%04X"),index);
701 string += strdata;
702 string += __L(" ");
703 }
704
705 for(XDWORD c=0; c<_sizeline; c++)
706 {
707 strdata.Format(__L("%02X "),data[index]);
708 string += strdata;
709 _size++;
710 index++;
711 }
712
713 if(_sizeline != sizeline)
714 {
715 for(XDWORD c=0; c<(sizeline-_sizeline); c++)
716 {
717 strdata.Format(__L(" "));
718 string += strdata;
719 }
720 }
721
722 if(showtext)
723 {
724 index -= _sizeline;
725 string += __L(" ");
726
727 for(XDWORD c=0; c<_sizeline; c++)
728 {
729 XCHAR character = (XCHAR)data[index];
730
731 if((character<0x20) || (character>0x80) || (character==__C('%')) || (character==__C('\\'))) character = __C('.');
732
733 strdata.Format(__L("%c"),character);
734 string += strdata;
735 index++;
736 }
737 }
738
739 mutex->UnLock();
740
741 AddEntry(level, lines?NULL:sectionID, inmemory, string.Get());
742
743 mutex->Lock();
744
745 lines++;
746 }
747
748 mutex->UnLock();
749
750 return true;
751}

846bool XLOGBASE::CalculateInitialStatus()
847{
848 XFILE* xfile = filelog->GetPrimaryFile();
849 if(!xfile) return false;
850
851 nlines = 0;
852 nentrys = 0;
853 size = xfile->GetSize();
854
855 entrys.DeleteContents();
856 entrys.DeleteAll();
857
858 XDWORD sizeBOM = 0;
859 XFILETXTFORMATCHAR formatchar = filelog->GetFormatCharFromFile(&sizeBOM);
860 int sizebytescharacter = filelog->SizeOfCharacter(formatchar);
861
862 xfile->SetPosition(sizeBOM);
863
864 bool endfile;
865 XDWORD br;
866 XBUFFER dataline;
867 XBYTE* readbuffer = new XBYTE[XFILETXT_MAXBUFFER];
868 if(!readbuffer) return false;
869
870 memset(readbuffer, 0, XFILETXT_MAXBUFFER);
871
872 do{ XDWORD bufferpos = 0;
873
874 br = XFILETXT_MAXBUFFER;
875 endfile = !xfile->Read(readbuffer, &br);
876 if(!br) break;
877
878 do{ XFILETXTTYPELF typeLF;
879 XDWORD sizeLF = 0;
880 XDWORD sizeline = 0;
881 bool endline = filelog->GetSizeOfLine(formatchar, &readbuffer[bufferpos], typeLF, sizeLF, sizeline, (br-bufferpos));
882
883 if(endline)
884 {
885 if(readbuffer[bufferpos] != 0x20)
886 {
887 nentrys++;
888
889 lastentry = new XLOGENTRY();
890 if(lastentry)
891 {
892 lastentry->position = bufferpos;
893 entrys.Add(lastentry);
894 }
895 }
896
897 if(lastentry) lastentry->size += (sizeline + sizeLF);
898 nlines++;
899
900 sizeline *= sizebytescharacter;
901 sizeline += (sizeLF*sizebytescharacter);
902 bufferpos += sizeline;
903 }
904 else
905 {
906 int lack = (br-bufferpos);
907 bufferpos += lack;
908
909 if(endfile && (bufferpos == br))
910 {
911 if(readbuffer[bufferpos] != 0x20)
912 {
913 nentrys++;
914
915 lastentry = new XLOGENTRY();
916 if(lastentry)
917 {
918 lastentry->position = bufferpos;
919 entrys.Add(lastentry);
920 }
921 }
922
923 if(lastentry) lastentry->size += lack;
924 nlines++;
925 }
926 }
927
928 } while(bufferpos < br);
929
930 } while(!endfile);
931
932 delete [] readbuffer;
933
934 return true;
935}
