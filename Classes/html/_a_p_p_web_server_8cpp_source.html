254bool APPWEBSERVER::SendRequest(DIOWEBSERVER_CONNECTION* connection, DIOWEBHEADER_RESULT headerresult, XBUFFER* data, int timeout, XSTRING* addhead)
255{
256 DIOWEBSERVER_HEADER webserverheader;
257 XSTRING entitytag;
258 HASHMD5 hashmd5;
259 bool iscached = false;
260 bool status = false;
261
262 if(!connection) return false;
263 if(!connection->GetDIOStream()) return false;
264 if(!connection->GetDIOStream()->IsConnected()) return false;
265
266 if(data)
267 {
268 hashmd5.Do((*data));
269 hashmd5.GetResultString(entitytag);
270 iscached = connection->GetRequest()->GetIfNoneMatch()->Compare(entitytag)?false:true;
271 }
272
273 if(iscached)
274 {
275 webserverheader.Create(connection->GetRequest()->GetResource(), 1, 1, DIOWEBHEADER_RESULT_NOTMODIFIED, connection->GetRequest()->GetSize());
276 status = webserverheader.Write(connection->GetDIOStream(), 5);
277 }
278 else
279 {
280 XPATH resourceext;
281 int size = (data ?(int)data->GetSize():-1);
282
283 resourceext = connection->GetRequest()->GetResource()->Get();
284 resourceext.SetOnlyExt();
285
286 webserverheader.Create(connection->GetRequest()->GetResource(), 1, 1, headerresult, size, &entitytag, connection->GetServer()->IsPlayablePage(resourceext));
287 if(addhead) webserverheader.AddLine(addhead->Get());
288 status = webserverheader.Write(connection->GetDIOStream(), 5);
289 if(status)
290 {
291 if(size > 0) status = connection->Send((*data), timeout);
292 }
293 }
294
295 return status;
296}

388bool APPWEBSERVER::ResolveRequest(DIOWEBSERVER* server, DIOWEBSERVER_CONNECTION* connection, DIOWEBSERVER_REQUEST* request)
389{
390 if(!server) return false;
391 if(!connection) return false;
392 if(!request) return false;
393
394 XSTRING* resource = request->GetResource();
395 if(!resource) return false;
396
397 if(!cfg) return false;
398
399 DIOWEBSERVER_HEADER webserverheader;
400 DIOURL resourceconv;
401 int nparams = 0;
402 DIOWEBSERVER_QUERYSTRINGS querystring;
403 XSTRING extension;
404 bool ispagepluging = false;
405 bool status = false;
406
407 resourceconv.Set(resource->Get());
408 resourceconv.DecodeUnsafeChars();
409
410 if(isauthenticatedaccess)
411 {
412 if(!cfg->WebServer_GetPassword()->IsEmpty())
413 {
414 if(!request->GetLoginPassword(page_login, page_password))
415 {
416 SendRequest(connection, DIOWEBHEADER_RESULT_UNAUTHORIZED, NULL, 5);
417
418 return true;
419 }
420 else
421 {
422 bool ispasswordvalid;
423
424 ispasswordvalid = !page_password.Compare(cfg->WebServer_GetPassword()->Get(), true);
425 if(!ispasswordvalid)
426 {
427 XSTRING leyend;
428
429 leyend = __L("Invalid user or password!");
430 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
431 }
432
433 XSTRING IPstring;
434 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
435
436 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server password [%s]: %s"), IPstring.Get(), cfg->WebServer_GetPassword()->Get(), (ispasswordvalid)? __L("Ok."): __L("INVALID!") );
437
438 if(!ispasswordvalid) return true;
439 }
440 }
441 else
442 {
443 bool invalidpassword = false;
444
445 if(page_login.IsEmpty() || page_password.IsEmpty())
446 {
447 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_DOAUTHENTICATE);
448
449 xevent.SetServer(server);
450 xevent.SetConnection(connection);
451 xevent.SetRequest(request);
452
453 PostEvent(&xevent);
454
455 if(xevent.IsRequestAuthenticate())
456 {
457 page_login = __L("usr");
458 page_password = __L("pwd");
459
460 } else invalidpassword = true;
461 }
462 else
463 {
464 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_CHECKAUTHENTICATE);
465
466 xevent.SetServer(server);
467 xevent.SetConnection(connection);
468 xevent.SetRequest(request);
469
470 PostEvent(&xevent);
471
472 if(!xevent.IsRequestAuthenticate()) invalidpassword = true;
473 }
474
475 if(invalidpassword)
476 {
477 XSTRING leyend;
478
479 leyend = __L("Invalid user or password!");
480 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
481
482 GEN_XLOG.AddEntry( XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
483
484 return true;
485 }
486 }
487 }
488
489 //-------------------------------------------------------------------------------------------------------------------------------
490
491 if(useragentID)
492 {
493 XSTRING browser;
494 XSTRING SO;
495
496 if(useragentID->Get(request->GetUserAgent()->Get(), browser, SO, 5))
497 {
498 GEN_XLOG.AddEntry(XLOGLEVEL_INFO, DIOWEBSERVER_LOGSECTIONID_VERBOSE, false, __L("Web browser : %s, with: %s."), browser.Get(), SO.Get());
499 }
500 }
501
502
503 //-------------------------------------------------------------------------------------------------------------------------------
504
505 status = false;
506 nparams = querystring.GetParamsFromURL(resourceconv.Get())?true:false;
507
508 querystring.DeleteParamsFromURL(resourceconv);
509
510 if(!resourceconv.Compare(__L("/"),true)) resourceconv.Set(__L("/index.html"));
511
512
513 //-------------------------------------------------------------------------------------------------------------------------------
514
515
516 XPATH xpathfile;
517
518 if(cfg->WebServer_PathResources()->IsEmpty())
519 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_WEB, xpathfile);
520 else xpathfile = cfg->WebServer_PathResources()->Get();
521
522 xpathfile.Slash_Delete();
523 xpathfile += resourceconv.Get();
524
525 //-------------------------------------------------------------------------------------------------------------------------------
526
527 if(resourceconv.GetExtension(extension))
528 {
529 int nplugins = webserver->GetPluginsManager()->Plugins_GetNPlugins();
530 for(int c=0; c<nplugins; c++)
531 {
532 DIOWEBSERVER_PLUGIN* plugin = webserver->GetPluginsManager()->Plugins_Get(c);
533 if(plugin)
534 {
535 if(plugin->PageExtension_IsContent(extension))
536 {
537 DIOWEBHEADER_RESULT headerresult = DIOWEBHEADER_RESULT_OK;
538 XSTRING result;
539
540 status = plugin->PageExtension(xpathfile, request, &querystring, headerresult, result);
541 if(status)
542 {
543 XSTRING head;
544 XSTRING data;
545
546 plugin->SeparedResult(result, head, data);
547
548 status = SendRequest(connection, headerresult, data, 5, &head);
549 }
550 else
551 {
552 XSTRING leyend;
553
554 leyend.Format(__L("Error to execute the %s."), plugin->GetName()->Get());
555
556 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
557 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
558 }
559
560 XSTRING IPstring;
561 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
562
563 GEN_XLOG.AddEntry((status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR), DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server \"%s\" %s."), IPstring.Get(), resourceconv.Get(), status?__L("sent"):__L("not send"));
564 ispagepluging = true;
565 }
566 }
567 }
568 }
569
570 if(!ispagepluging)
571 {
572 if(!nparams && !isapirestonly)
573 {
574 status = SendRequest(connection, DIOWEBHEADER_RESULT_OK, xpathfile, 5);
575 if(!status)
576 {
577 // Brower close the conextion in thread
578 if(!connection->GetDIOStream()->IsConnected()) status = true;
579 }
580 }
581
582 if(!status)
583 {
584 if(!nparams && !isapirestonly)
585 {
586 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_REQUEST);
587
588 xevent.SetServer(server);
589 xevent.SetConnection(connection);
590 xevent.SetRequest(request);
591
592 PostEvent(&xevent);
593
594 status = xevent.GetStatus();
595 }
596 }
597
598 //-------------------------------------------------------------------------------------------------------------------------------
599
600 if(!status)
601 {
602 DIOWEBPAGEHTMLCREATOR* webHTMLpage = new DIOWEBPAGEHTMLCREATOR();
603 if(webHTMLpage)
604 {
605 DIOWEBSERVER_XEVENT xevent(this, DIOWEBSERVER_XEVENT_TYPE_REQUEST_ENDPOINT);
606
607 xevent.SetServer(server);
608 xevent.SetConnection(connection);
609 xevent.SetRequest(request);
610 xevent.SetQueryString(&querystring);
611
612 PostEvent(&xevent);
613
614 //DIOWEBHEADER_RESULT result = xevent.GetResult();
615
616 status = xevent.GetStatus();
617 if(status)
618 {
619 webHTMLpage->Set(xevent.GetOutput()->Get());
620 status = SendRequest(connection, DIOWEBHEADER_RESULT_OK, (*webHTMLpage), 5);
621 }
622 else
623 {
624 XSTRING leyend;
625
626 leyend = __L("Error: command or erroneous parameters.");
627 GenerateMessagePage(leyend, (*webHTMLpage));
628
629 GenerateResponse_Error(connection, DIOWEBHEADER_RESULT_NOTFOUND, leyend);
630
631 GEN_XLOG.AddEntry(XLOGLEVEL_ERROR, DIOWEBSERVER_LOGSECTIONID, false, leyend.Get());
632 }
633
634 XSTRING IPstring;
635 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
636
637 GEN_XLOG.AddEntry((status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR), DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server\"%s\" %s."), IPstring.Get(), resourceconv.Get(), status?__L("sent"):__L("not send"));
638
639 delete webHTMLpage;
640 }
641 }
642 else
643 {
644 XSTRING IPstring;
645 connection->GetDIOStream()->GetClientIP()->GetXString(IPstring);
646
647 GEN_XLOG.AddEntry(XLOGLEVEL_INFO, DIOWEBSERVER_LOGSECTIONID, false, __L("Request from [%s] to the web server of file \"%s\" sent."), IPstring.Get(), xpathfile.Get());
648 }
649 }
650
651 return true;
652}

754{

756
757 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_REQUEST , (XSUBJECT *)webserver);
758 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_REQUEST);
759
760 if(isauthenticatedaccess)
761 {
762 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_DOAUTHENTICATE);
763 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_CHECKAUTHENTICATE);
764 }
765
766 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_REQUEST_ENDPOINT , (XSUBJECT *)webserver);
767 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_REQUEST_ENDPOINT);
768
769 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_CONNECTED , (XSUBJECT *)webserver);
770 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_CONNECTED);
771
772 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_READDATA , (XSUBJECT *)webserver);
773 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_READDATA);
774
775 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_PONG , (XSUBJECT *)webserver);
776 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_PONG);
777
778 UnSubscribeEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_DISCONNECTED , (XSUBJECT *)webserver);
779 DeRegisterEvent(DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_DISCONNECTED);
780
781 if(webserver)
782 {
783 delete webserver;
784 webserver = NULL;
785 }
786
787 if(pluginPHP)
788 {
789 delete pluginPHP;
790 pluginPHP = NULL;
791 }
792
793 if(useragentID)
794 {
795 delete useragentID;
796 useragentID = NULL;
797 }
798
799 return true;
800}

813void APPWEBSERVER::HandleEvent_WebServer(DIOWEBSERVER_XEVENT* event)
814{
815 switch(event->GetEventType())
816 {
817 case DIOWEBSERVER_XEVENT_TYPE_REQUEST : ResolveRequest(event->GetServer(), event->GetConnection(), event->GetRequest());
818 break;
819
820 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_CONNECTED : if(!PostEvent(event))
821 {
822 // Accept connection with any version, protocol o resources.
823 event->SetStatus(true);
824 break;
825 }
826 break;
827
828 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_READDATA : PostEvent(event);
829 break;
830
831 case DIOWEBSERVER_XEVENT_TYPE_WEBSOCKET_DISCONNECTED : PostEvent(event);
832 break;
833 }
834}
