299bool CIPHERDES::Cipher(XBYTE* input,XDWORD size)
300{
301 if(!size) return false;
302
303 XBUFFER inputpadding;
304 CIPHERDES_CONTEXT ctx;
305 XBYTE iv[8];
306
307 memset(iv,0,8);
308 if(inivector && inivector->GetSize()>=8) memcpy(iv, inivector->Get() ,8);
309
310 inputpadding.Add(input,size);
311 inputpadding.Padding_Add(paddingadjustsize, paddingtype);
312
313 result->Delete();
314 result->Add(input,size);
315 result->Padding_Add(paddingadjustsize, paddingtype);
316
317 CIPHERKEYSYMMETRICAL* key = (CIPHERKEYSYMMETRICAL*)GetKey();
318 if(!key) return false;
319 if(!key->Get()->Get()) return false;
320
321 DESSetKeyCipher(&ctx, (XBYTE *)key->Get()->Get());
322
323 bool status;
324
325 if(this->GetChainingMode() == CIPHERCHAININGMODE_ECB)
326 status = DESCipher_ECB(&ctx, CIPHERDES_ENCRYPT, result->GetSize(), iv, inputpadding.Get(), result->Get());
327 else status = DESCipher_CBC(&ctx, CIPHERDES_ENCRYPT, result->GetSize(), iv, inputpadding.Get(), result->Get());
328
329 return status;
330}

618{

620
621 if(length % 8) return false;
622
623 if(mode==CIPHERDES_ENCRYPT)
624 {
625 while(length > 0)

627 DESCipher(ctx, output, output);
628

630
631 input += 8;
632 output += 8;
633 length -= 8;
634 }

641
642 DESCipher( ctx, input, output);
643

645
646 input += 8;
647 output += 8;
648 length -= 8;
649 }
650 }
651
652 return true;
