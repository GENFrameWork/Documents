205bool XFILEXDB::OpenFile()
206{
207 if(!Open(this->xpath)) return false;
208
209 GetPrimaryFile()->Read((XBYTE*)&nrecords,sizeof(XDWORD));
210
211 if(!nrecords)
212 {
213 Close();
214 return false;
215 }
216
217 indexmap.SetIsMulti(true);
218 for(int c=0;c<nrecords;c++)
219 {
220 XDWORD ID = 0;
221 XDWORD filepos = 0;
222
223 GetPrimaryFile()->Read((XBYTE*)&ID , sizeof(XDWORD));
224 GetPrimaryFile()->Read((XBYTE*)&filepos , sizeof(XDWORD));
225
226 //XTRACE_PRINTCOLOR(1,__L("ID: %08X"), ID);
227
228 indexmap.Add(ID,filepos);
229 }
230
231 indexmap.SetIsMulti(false);
232
233 isopen=true;
234
235 return true;
236}

265XBUFFER* XFILEXDB::GetRecord(XDWORD ID)
266{
267 if(!isopen)
268 {
269 if(!OpenFile()) return NULL;
270 }
271
272 if(!nrecords) return NULL;
273
274 XDWORD filepos = GetPosition(ID);
275 if(!filepos) return NULL;
276
277 XDWORD size;
278
279 if(!GetPrimaryFile()->SetPosition(filepos)) return NULL;
280
281 if(!GetPrimaryFile()->Read((XBYTE*)&size ,sizeof(XDWORD))) return NULL;
282
283 XBUFFER* xbuffer = new XBUFFER();
284 if(xbuffer)
285 {
286 if(size)
287 {
288 xbuffer->Resize(size);
289 xbuffer->FillBuffer();
290 }
291
292 if(!GetPrimaryFile()->Read((XBYTE*)xbuffer->Get() , size))
293 {
294 delete xbuffer;
295 return NULL;
296 }
297
298 return xbuffer;
299 }
