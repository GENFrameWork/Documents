198bool XFILEXDB::OpenFile()
199{
200 if(!Open(this->xpath)) return false;
201
202 GetPrimaryFile()->Read((XBYTE*)&nrecords,sizeof(XDWORD));
203
204 if(!nrecords)
205 {
206 Close();
207 return false;
208 }
209
210 indexmap.SetIsMulti(true);
211 for(int c=0;c<nrecords;c++)
212 {
213 XDWORD ID = 0;
214 XDWORD filepos = 0;
215
216 GetPrimaryFile()->Read((XBYTE*)&ID , sizeof(XDWORD));
217 GetPrimaryFile()->Read((XBYTE*)&filepos , sizeof(XDWORD));
218
219 //XTRACE_PRINTCOLOR(1,__L("ID: %08X"), ID);
220
221 indexmap.Add(ID,filepos);
222 }
223
224 indexmap.SetIsMulti(false);
225
226 isopen=true;
227
228 return true;
229}

258XBUFFER* XFILEXDB::GetRecord(XDWORD ID)
259{
260 if(!isopen)
261 {
262 if(!OpenFile()) return NULL;
263 }
264
265 if(!nrecords) return NULL;
266
267 XDWORD filepos = GetPosition(ID);
268 if(!filepos) return NULL;
269
270 XDWORD size;
271
272 if(!GetPrimaryFile()->SetPosition(filepos)) return NULL;
273
274 if(!GetPrimaryFile()->Read((XBYTE*)&size ,sizeof(XDWORD))) return NULL;
275
276 XBUFFER* xbuffer = new XBUFFER();

278 {
279 if(size)

282 xbuffer->FillBuffer();
283 }
284
