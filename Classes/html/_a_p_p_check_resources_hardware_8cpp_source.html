333bool APPCHECKRESOURCESHARDWARE::CheckTotalCPUUsageStatus()
334{
335 if(!cfg) return false;
336
337 int cpuusage = GEN_XSYSTEM.GetCPUUsageTotal();
338
339 if(cpuusage > maxtotalCPUusage)
340 {
341 maxtotalCPUusage = cpuusage;
342 }
343
344 ntotalCPUusage += cpuusage;
345 ntotalCPUsamples++;
346
347 if(ntotalCPUsamples > 5)
348 {
349 if(GetCPUTotalCPUUsageAverange() > (XDWORD)cfg->CheckResourcesHardware_GetTotalCPUUsageLimitPercent())
350 {
351 APPCHECKRESOURCESHARDWARE_XEVENT xevent(this, APPCHECKRESOURCESHARDWARE_XEVENT_TYPE_TOTALCPUUSAGELIMIT);
352 xevent.SetActualTotalCPUUsage(cpuusage);
353 PostEvent(&xevent);
354 }
355 }
356
357 return true;
358}
