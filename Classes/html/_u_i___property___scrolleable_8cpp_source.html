394{
395 switch(type)
396 {
397 case UI_PROPERTY_SCROLLEABLE_TYPE_UNKWOWN :
398 case UI_PROPERTY_SCROLLEABLE_TYPE_ALL :
399 default : break;
400
401 case UI_PROPERTY_SCROLLEABLE_TYPE_HORIZONTAL : if(!horizontal.active) break;
402
403 if((horizontal.displacement + step) > 0)
404 {
405 horizontal.displacement = 0;
406 break;
407 }
408
409 if(horizontal.limit != -1)
410 {
411 if((horizontal.displacement + step) < horizontal.limit)
412 {
413 horizontal.displacement = horizontal.limit;
414 break;
415 }
416 }
417
418 horizontal.step = step;
419 horizontal.displacement += step;
420 break;
421
422 case UI_PROPERTY_SCROLLEABLE_TYPE_VERTICAL : if(!vertical.active) break;
423
424 if((vertical.displacement + step) > 0)
425 {
426 vertical.displacement = 0;
427 break;
428 }
429
430 if(vertical.limit != -1)
431 {

433 {

435 break;

439 vertical.step = step;
