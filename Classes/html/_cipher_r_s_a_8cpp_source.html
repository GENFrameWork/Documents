232bool CIPHERRSA::SetKey(CIPHERKEY* key, bool integritycheck)
233{
234 if(!key) return false;
235
236 switch(key->GetType())
237 {
238 case CIPHERKEYTYPE_SYMMETRICAL : return false;
239
240 case CIPHERKEYTYPE_PUBLIC : { CIPHERKEYPUBLICRSA* publickey = (CIPHERKEYPUBLICRSA*)key;
241
242 publickey->Get(context.N, context.E); // Assign key public to context
243
244 if(integritycheck)
245 {
246 if(!CheckContextPublicKey()) return false;
247 }
248 }
249 break;
250
251 case CIPHERKEYTYPE_PRIVATE : { CIPHERKEYPRIVATERSA* privatekey = (CIPHERKEYPRIVATERSA*)key;
252 XMPINTEGER P1;
253 XMPINTEGER Q1;
254
255 privatekey->Get(context.P, context.Q, context.D); // Assign key public to context
256
257 if(!P1.SubtractionSigned(&context.P, 1)) return false;
258 if(!Q1.SubtractionSigned(&context.Q, 1)) return false;
259 if(!context.DP.Module(&context.DP, &context.D, &P1)) return false;
260 if(!context.DQ.Module(&context.DQ, &context.D, &Q1)) return false;
261 if(!context.QP.ModularInverse(&context.Q, &context.P)) return false;
262
263 if(integritycheck)
264 {
265 if(!CheckContextPrivateKey()) return false;
266 }
267
268 }
269 break;
270 default : break;
271
272 }
273
274 return CIPHER::SetKey(key);
275}

293{

298

300 XBUFFER output;

553bool CIPHERRSA::GenerateKeys(XDWORD nbits, int exponent, CIPHERKEYPUBLICRSA& publickey, CIPHERKEYPRIVATERSA& privatekey, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
554{
555 XMPINTEGER P1;
556 XMPINTEGER Q1;
557 XMPINTEGER H;
558 XMPINTEGER G;
559 bool status;
560
561 if(nbits<128 || exponent<3 ) return false;
562
563 P1.Ini();
564 Q1.Ini();
565 H.Ini();
566 G.Ini();
567
568 // find primes P and Q with Q < P so that: GCD( E, (P-1)*(Q-1) ) == 1
569 status = context.E.LeftSet(exponent);
570 if(status)
571 {
572 do{ if(!context.P.GeneratePrime((nbits+1)>>1, 0, funcrandom?funcrandom:(XMPINTEGER_FUNCRANDOM)GenerateRandom, NULL)) return false;
573 if(!context.Q.GeneratePrime((nbits+1)>>1, 0, funcrandom?funcrandom:(XMPINTEGER_FUNCRANDOM)GenerateRandom, NULL)) return false;
574
575 if(context.P.CompareSignedValues(context.Q) < 0) context.P.Swap(&context.Q);
576
577 if(context.P.CompareSignedValues(context.Q) == 0) continue;
578
579 if(!context.N.Multiplication(&context.P, &context.Q)) return false;
580
581 if(context.N.GetMSB()!=nbits) continue;
582
583 if(!P1.SubtractionSigned(&context.P, 1)) return false;
584 if(!Q1.SubtractionSigned(&context.Q, 1)) return false;
585
586 if(!H.Multiplication(&P1, &Q1)) return false;
587
588 if(!G.GreatestCommonDivisor(&G, &context.E, &H)) return false;
589
590 } while(G.CompareSignedValues(1) != 0);
591
592
593 // D = E^-1 mod ((P-1)*(Q-1))
594 // DP = D mod (P - 1)
595 // DQ = D mod (Q - 1)
596 // QP = Q^-1 mod P
597
598 if(!context.D.ModularInverse(&context.E, &H)) return false;
599 if(!context.DP.Module(&context.DP, &context.D, &P1)) return false;
600 if(!context.DP.Module(&context.DQ, &context.D, &Q1)) return false;
601 if(!context.QP.ModularInverse(&context.Q, &context.P)) return false;
602
603 context.len = (context.N.GetMSB() + 7) >> 3;
604 }
605
606 P1.End();
607 Q1.End();
608 H.End();
609 G.End();
610
611 if(!status)
612 {
613 context.End();
614 }
615 else
616 {
617 publickey.Set(context.N, context.E);
618 privatekey.Set(context.P, context.Q, context.D);
619 }
620
621 return status;
622}

654bool CIPHERRSA::CheckContextPrivateKey()
655{
656 XMPINTEGER PQ;
657 XMPINTEGER DE;
658 XMPINTEGER P1;
659 XMPINTEGER Q1;
660 XMPINTEGER H;
661 XMPINTEGER I;
662 XMPINTEGER G;
663 XMPINTEGER G2;
664 XMPINTEGER L1;
665 XMPINTEGER L2;
666 XMPINTEGER DP;
667 XMPINTEGER DQ;
668 XMPINTEGER QP;
669
670 if(!CheckContextPublicKey()) return false;
671
672 if((!context.P.GetLimbs()) || (!context.Q.GetLimbs()) || (!context.D.GetLimbs())) return false;
673
674 PQ.Ini();
675 DE.Ini();
676 P1.Ini();
677 Q1.Ini();
678 H.Ini();
679 I.Ini();
680 G.Ini();
681 G2.Ini();
682 L1.Ini();
683 L2.Ini();
684 DP.Ini();
685 DQ.Ini();
686 QP.Ini();
687
688 if(!PQ.Multiplication(&context.P, &context.Q)) return false;
689 if(!DE.Multiplication(&context.D, &context.E)) return false;
690 if(!P1.SubtractionSigned(&context.P, 1)) return false;
691 if(!Q1.SubtractionSigned(&context.Q, 1)) return false;
692 if(!H.Multiplication(&P1, &Q1)) return false;
693 if(!G.GreatestCommonDivisor(&G, &context.E, &H)) return false;
694 if(!G2.GreatestCommonDivisor(&G2, &P1, &Q1)) return false;
695 if(!L1.Division(&L1, &L2, &H, &G2 )) return false;
696 if(!I.Module(&I, &DE, &L1)) return false;
697 if(!DP.Module( &DP, &context.D, &P1)) return false;
698 if(!DQ.Module(&DQ, &context.D, &Q1)) return false;
699 if(!QP.ModularInverse(&context.Q, &context.P)) return false;
700
701 // Check for a valid PKCS1v2 private key
702
703 if((PQ.CompareSignedValues(context.N ) != 0) || (DP.CompareSignedValues(context.DP) != 0) ||
704 (DQ.CompareSignedValues(context.DQ) != 0) || (QP.CompareSignedValues(context.QP) != 0) ||
705 (L2.CompareSignedValues(0) != 0) || (I.CompareSignedValues(1) != 0) ||
706 (G.CompareSignedValues(1) != 0)) return false;
707
708 PQ.End();
709 DE.End();
710 P1.End();
711 Q1.End();
712 H.End();
713 I.End();
714 G.End();
715 G2.End();
716 L1.End();
717 L2.End();
718 DP.End();
719 DQ.End();
720 QP.End();
721
722 return true;
723}

743bool CIPHERRSA::Cipher_PKCS1_V15(XBYTE* buffer, XDWORD size, XBUFFER& output, CIPHERKEYTYPE keytypetouse, CIPHERRSAPKCS1VERSION pkcs1version, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
744{
745 XBUFFER input;
746 int nb_pad;
747 bool status;
748
749 if(!funcrandom) return false;
750
751 int keysize = GetKeySizeInBytes(keytypetouse);
752 if(!keysize) return false;
753
754 //if(keysize < (size + 11)) return false;
755
756 nb_pad = keysize - 3 - size;
757
758 input.Add((XBYTE)0);
759
760 if(keytypetouse == CIPHERKEYTYPE_PUBLIC)
761 {
762 XBYTE data = 0;
763
764 input.Add((XBYTE)CIPHERRSA_CIPHERDATA);
765
766 while(nb_pad > 0)
767 {
768 if(funcrandom(&data, 1, paramrandom))
769 {
770 input.Add((XBYTE)data);
771 nb_pad--;
772 }
773 }
774 }
775 else
776 {
777 input.Add((XBYTE)CIPHERRSA_SINGDATA);
778
779 while(nb_pad-- > 0)
780 {
781 input.Add((XBYTE)0xFF);
782 }
783 }
784
785
786 input.Add((XBYTE)0);
787 input.Add(buffer,size);
788
789 status = false;
790
791 switch(keytypetouse)
792 {
793 case CIPHERKEYTYPE_PUBLIC : status = DoRSAPublicOperation(input, output); break;
794 case CIPHERKEYTYPE_PRIVATE : status = DoRSAPrivateOperation(input, output, funcrandom, paramrandom); break;
795 default : break;
796 }
797
798 return status;
799}

819bool CIPHERRSA::Uncipher_PKCS1_V15(XBYTE* buffer, XDWORD size, XBUFFER& output, CIPHERKEYTYPE keytypetouse, CIPHERRSAPKCS1VERSION pkcs1version, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
820{
821 XBUFFER input;
822 XBUFFER _output;
823 XBYTE pad_chk;
824 int pad_count = 0;
825 XBYTE pad_done = 0;
826 bool status;
827
828 if(!funcrandom) return false;
829
830 int keysize = GetKeySizeInBytes(keytypetouse);
831 if(!keysize) return false;
832
833 if((keysize < 16) || (keysize > XMPINTEGER_MAXSIZE)) return false;
834
835 input.Add(buffer, size);
836
837 status = false;
838
839 switch(keytypetouse)
840 {
841 case CIPHERKEYTYPE_PUBLIC : status = DoRSAPublicOperation(input , _output); break;
842 case CIPHERKEYTYPE_PRIVATE : status = DoRSAPrivateOperation(input, _output, funcrandom, paramrandom); break;
843 default : break;
844 }
845
846 if(!status) return false;
847
848 int index = 0;
849
850 pad_chk = 0;
851 pad_chk |= _output.GetByte(index++);
852
853 pad_chk |= (_output.GetByte(index++)) ^ ((keytypetouse == CIPHERKEYTYPE_PRIVATE)? CIPHERRSA_CIPHERDATA : CIPHERRSA_SINGDATA);
854
855 for(XDWORD i=0; i<size-3; i++)
856 {
857 pad_done |= (_output.GetByte(index+i) == 0);
858 pad_count += (pad_done == 0);
859 }
860
861 index += pad_count;
862 pad_chk |= _output.GetByte(index++); // Must be zero
863
864
865 if(pad_chk) return false;
866
867 output.Add(&_output.Get()[index], (size-index));
868
869 return true;
870}

920bool CIPHERRSA::DoRSAPrivateOperation(XBUFFER& input, XBUFFER& output, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
921{
922 XMPINTEGER T;
923 XMPINTEGER T1;
924 XMPINTEGER T2;
925
926 #if !defined(CIPHERRSA_RSANOCRT)
927 XMPINTEGER* Vi;
928 XMPINTEGER* Vf;
929
930 Vi = &context.Vi;
931 Vf = &context.Vf;
932 #endif
933
934 T.Ini();
935 T1.Ini();
936 T2.Ini();
937
938 if(!T.SetFromXBuffer(input)) return false;
939
940 if(T.CompareSignedValues(context.N)>=0) return false;
941
942 #if defined(CIPHERRSA_RSANOCRT)
943 if(!T.SlidingWindowExponentiation(&T, &context.D, &context.N, &context.RN)) return false;
944 #else
945 if(funcrandom != NULL)
946 {
947 // Blinding T = T * Vi mod N
948 if(!PrepareBlinding(Vi, Vf, funcrandom, paramrandom)) return false;
949 if(!T.Multiplication(&T, Vi )) return false;
950 if(!T.Module( &T, &T, &context.N)) return false;
951 }
952
953 // faster decryption using the CRT
954 // T1 = input ^ dP mod P
955 // T2 = input ^ dQ mod Q
956
957 if(!T1.SlidingWindowExponentiation(&T, &context.DP, &context.P, &context.RP)) return false;
958 if(!T2.SlidingWindowExponentiation(&T, &context.DQ, &context.Q, &context.RQ)) return false;
959
960 // T = (T1 - T2) * (Q^-1 mod P) mod P
961 if(!T.SubtractionSigned(&T1, &T2)) return false;
962 if(!T1.Multiplication(&T, &context.QP)) return false;
963 if(!T.Module(&T, &T1, &context.P)) return false;
964
965 // T = T2 + T * Q
966 if(!T1.Multiplication(&T, &context.Q )) return false;
967 if(!T.AdditionSigned(&T2, &T1)) return false;
968
969 if(funcrandom!= NULL)
970 {
971 // Unblind T = T * Vf mod N
972 if(!T.Multiplication(&T, Vf)) return false;
973 if(!T.Module(&T, &T, &context.N)) return false;
974 }
975 #endif
976
977 int keysize = GetKeySizeInBytes(CIPHERKEYTYPE_PRIVATE);
978 if(!keysize) return false;
979
980 if(!T.GetToXBuffer(output, keysize)) return false;
981
982 T.End();
983 T1.End();
984 T2.End();
985
986 return true;
987}

990#if !defined(CIPHERRSA_RSANOCRT)
1005bool CIPHERRSA::PrepareBlinding(XMPINTEGER* Vi, XMPINTEGER* Vf, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
1006{
1007 int count = 0;
1008
1009 if(context.Vf.GetLimbs()!= NULL)
1010 {
1011 // We already have blinding values, just update them by squaring
1012 if(!context.Vi.Multiplication(&context.Vi, &context.Vi)) return false;
1013 if(!context.Vi.Module(&context.Vi, &context.Vi, &context.N )) return false;
1014 if(!context.Vf.Multiplication(&context.Vf, &context.Vf)) return false;
1015 if(!context.Vf.Module(&context.Vf, &context.Vf, &context.N )) return false;
1016 }
1017 else
1018 {
1019 int keysize = GetKeySizeInBytes(CIPHERKEYTYPE_PRIVATE);
1020 if(!keysize) return false;
1021
1022 // Unblinding value: Vf = random number, invertible mod N
1023 do{ if(count++ > 10) return false;
1024
1025 if(!context.Vf.FillRandom(keysize-1, funcrandom, paramrandom)) return false;
1026 if(!context.Vi.GreatestCommonDivisor(&context.Vi, &context.Vf, &context.N )) return false;
1027
1028 } while(context.Vi.CompareSignedValues(1)!=0);
1029
1030 // Blinding value: Vi = Vf^(-e) mod N
1031 if(!context.Vi.ModularInverse( &context.Vf, &context.N)) return false;
1032 if(!context.Vi.SlidingWindowExponentiation(&context.Vi, &context.E, &context.N, &context.RN )) return false;
1033 }
1034
1035 if(Vi != &context.Vi)
1036 {
1037 if(!Vi->CopyFrom(&context.Vi)) return false;
1038 if(!Vf->CopyFrom(&context.Vf)) return false;
1039 }
1040
1041 return true;
1042}
