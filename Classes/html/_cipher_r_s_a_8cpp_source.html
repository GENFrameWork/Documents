355bool CIPHERKEYRSAPRIVATE::Check(CIPHERKEYRSAPUBLIC& publickey)
356{
357 XMPINTEGER PQ;
358 XMPINTEGER DE;
359 XMPINTEGER P1;
360 XMPINTEGER Q1;
361 XMPINTEGER H;
362 XMPINTEGER I;
363 XMPINTEGER G;
364 XMPINTEGER G2;
365 XMPINTEGER L1;
366 XMPINTEGER L2;
367 XMPINTEGER publicmodulus;
368 XMPINTEGER publicexponent;
369
370 if(!publickey.Check()) return false;
371
372 if((!prime1factor.GetLimbs()) || (!prime2factor.GetLimbs()) || (!exponent.GetLimbs())) return false;
373
374 publickey.Get(publicmodulus, publicexponent);
375
376 PQ.Ini();
377 DE.Ini();
378 P1.Ini();
379 Q1.Ini();
380 H.Ini();
381 I.Ini();
382 G.Ini();
383 G2.Ini();
384 L1.Ini();
385 L2.Ini();
386
387 if(!PQ.Multiplication(&prime1factor, &prime2factor)) return false;
388 if(!DE.Multiplication(&exponent, &publicexponent)) return false;
389 if(!P1.SubtractionSigned(&prime1factor, 1)) return false;
390 if(!Q1.SubtractionSigned(&prime2factor, 1)) return false;
391 if(!H.Multiplication(&P1, &Q1)) return false;
392 if(!G.GreatestCommonDivisor(&G, &publicexponent, &H)) return false;
393 if(!G2.GreatestCommonDivisor(&G2, &P1, &Q1)) return false;
394 if(!L1.Division(&L1, &L2, &H, &G2 )) return false;
395 if(!I.Module(&I, &DE, &L1)) return false;
396
397 // Check for a valid PKCS1v2 private key
398 if((PQ.CompareSignedValues(publicmodulus) != 0) ||
399 (L2.CompareSignedValues(0) != 0) ||
400 (I.CompareSignedValues(1) != 0) ||
401 (G.CompareSignedValues(1) != 0)) return false;
402
403 PQ.End();
404 DE.End();
405 P1.End();
406 Q1.End();
407 H.End();
408 I.End();
409 G.End();
410 G2.End();
411 L1.End();
412 L2.End();
413
414 return true;
415}

591bool CIPHERRSA::SetKey(CIPHERKEY* key, bool integritycheck)
592{
593 if(!key) return false;
594
595 switch(key->GetType())
596 {
597 case CIPHERKEYTYPE_SYMMETRICAL : return false;
598
599 case CIPHERKEYTYPE_PUBLIC : { CIPHERKEYRSAPUBLIC* publickey = (CIPHERKEYRSAPUBLIC*)key;
600
601 publickey->Get(context.N, context.E); // Assign key public to context
602
603 if(integritycheck)
604 {
605 if(!CheckContextPublicKey()) return false;
606 }
607 }
608 break;
609
610 case CIPHERKEYTYPE_PRIVATE : { CIPHERKEYRSAPRIVATE* privatekey = (CIPHERKEYRSAPRIVATE*)key;
611 XMPINTEGER P1;
612 XMPINTEGER Q1;
613
614 privatekey->Get(context.P, context.Q, context.D); // Assign key public to context
615
616 if(!P1.SubtractionSigned(&context.P, 1)) return false;
617 if(!Q1.SubtractionSigned(&context.Q, 1)) return false;
618 if(!context.DP.Module(&context.DP, &context.D, &P1)) return false;
619 if(!context.DQ.Module(&context.DQ, &context.D, &Q1)) return false;
620 if(!context.QP.ModularInverse(&context.Q, &context.P)) return false;
621
622 if(integritycheck)
623 {
624 if(!CheckContextPrivateKey()) return false;
625 }
626
627 }
628 break;
629 default : break;
630
631 }
632
633 return CIPHER::SetKey(key);
634}

912bool CIPHERRSA::GenerateKeys(XDWORD nbits, int exponent, CIPHERKEYRSAPUBLIC& publickey, CIPHERKEYRSAPRIVATE& privatekey, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
913{
914 XMPINTEGER P1;
915 XMPINTEGER Q1;
916 XMPINTEGER H;
917 XMPINTEGER G;
918 bool status;
919
920 if(nbits<128 || exponent<3 ) return false;
921
922 P1.Ini();
923 Q1.Ini();
924 H.Ini();
925 G.Ini();
926
927 // find primes P and Q with Q < P so that: GCD( E, (P-1)*(Q-1) ) == 1
928 status = context.E.LeftSet(exponent);
929 if(status)
930 {
931 do{ if(!context.P.GeneratePrime((nbits+1)>>1, 0, funcrandom?funcrandom:(XMPINTEGER_FUNCRANDOM)GenerateRandom, NULL)) return false;
932 if(!context.Q.GeneratePrime((nbits+1)>>1, 0, funcrandom?funcrandom:(XMPINTEGER_FUNCRANDOM)GenerateRandom, NULL)) return false;
933
934 if(context.P.CompareSignedValues(context.Q) < 0) context.P.Swap(&context.Q);
935
936 if(context.P.CompareSignedValues(context.Q) == 0) continue;
937
938 if(!context.N.Multiplication(&context.P, &context.Q)) return false;
939
940 if(context.N.GetMSB()!=nbits) continue;
941
942 if(!P1.SubtractionSigned(&context.P, 1)) return false;
943 if(!Q1.SubtractionSigned(&context.Q, 1)) return false;
944
945 if(!H.Multiplication(&P1, &Q1)) return false;
946
947 if(!G.GreatestCommonDivisor(&G, &context.E, &H)) return false;
948
949 } while(G.CompareSignedValues(1) != 0);
950
951
952 // D = E^-1 mod ((P-1)*(Q-1))
953 // DP = D mod (P - 1)
954 // DQ = D mod (Q - 1)
955 // QP = Q^-1 mod P
956
957 if(!context.D.ModularInverse(&context.E, &H)) return false;
958 if(!context.DP.Module(&context.DP, &context.D, &P1)) return false;
959 if(!context.DP.Module(&context.DQ, &context.D, &Q1)) return false;
960 if(!context.QP.ModularInverse(&context.Q, &context.P)) return false;
961
962 context.len = (context.N.GetMSB() + 7) >> 3;
963 }
964
965 P1.End();
966 Q1.End();
967 H.End();
968 G.End();
969
970 if(!status)
971 {
972 context.End();
973 }
974 else
975 {
976 publickey.Set(context.N, context.E);
977 privatekey.Set(context.P, context.Q, context.D);
978 }
979
980 return status;
981}

1072 I.End();

1076 L2.End();

1279bool CIPHERRSA::DoRSAPrivateOperation(XBUFFER& input, XBUFFER& output, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
1280{
1281 XMPINTEGER T;
1282 XMPINTEGER T1;
1283 XMPINTEGER T2;
1284
1285 #if !defined(CIPHERRSA_RSANOCRT)
1286 XMPINTEGER* Vi;
1287 XMPINTEGER* Vf;
1288
1289 Vi = &context.Vi;
1290 Vf = &context.Vf;
1291 #endif
1292
1293 T.Ini();
1294 T1.Ini();
1295 T2.Ini();
1296
1297 if(!T.SetFromXBuffer(input)) return false;
1298
1299 if(T.CompareSignedValues(context.N)>=0) return false;
1300
1301 #if defined(CIPHERRSA_RSANOCRT)
1302 if(!T.SlidingWindowExponentiation(&T, &context.D, &context.N, &context.RN)) return false;
1303 #else
1304 if(funcrandom != NULL)
1305 {
1306 // Blinding T = T * Vi mod N
1307 if(!PrepareBlinding(Vi, Vf, funcrandom, paramrandom)) return false;
1308 if(!T.Multiplication(&T, Vi )) return false;
1309 if(!T.Module( &T, &T, &context.N)) return false;
1310 }
1311
1312 // faster decryption using the CRT
1313 // T1 = input ^ dP mod P
1314 // T2 = input ^ dQ mod Q
1315
1316 if(!T1.SlidingWindowExponentiation(&T, &context.DP, &context.P, &context.RP)) return false;
1317 if(!T2.SlidingWindowExponentiation(&T, &context.DQ, &context.Q, &context.RQ)) return false;
1318
1319 // T = (T1 - T2) * (Q^-1 mod P) mod P
1320 if(!T.SubtractionSigned(&T1, &T2)) return false;
1321 if(!T1.Multiplication(&T, &context.QP)) return false;
1322 if(!T.Module(&T, &T1, &context.P)) return false;
1323
1324 // T = T2 + T * Q
1325 if(!T1.Multiplication(&T, &context.Q )) return false;
1326 if(!T.AdditionSigned(&T2, &T1)) return false;
1327
1328 if(funcrandom!= NULL)
1329 {
1330 // Unblind T = T * Vf mod N
1331 if(!T.Multiplication(&T, Vf)) return false;
1332 if(!T.Module(&T, &T, &context.N)) return false;
1333 }
1334 #endif
1335
1336 int keysize = GetKeySizeInBytes(CIPHERKEYTYPE_PRIVATE);
1337 if(!keysize) return false;
1338
1339 if(!T.GetToXBuffer(output, keysize)) return false;
1340
1341 T.End();
1342 T1.End();
1343 T2.End();
1344
1345 return true;
1346}

1364bool CIPHERRSA::PrepareBlinding(XMPINTEGER* Vi, XMPINTEGER* Vf, XMPINTEGER_FUNCRANDOM funcrandom, void* paramrandom)
1365{
1366 int count = 0;
1367
1368 if(context.Vf.GetLimbs()!= NULL)
1369 {
1370 // We already have blinding values, just update them by squaring
1371 if(!context.Vi.Multiplication(&context.Vi, &context.Vi)) return false;
1372 if(!context.Vi.Module(&context.Vi, &context.Vi, &context.N )) return false;
1373 if(!context.Vf.Multiplication(&context.Vf, &context.Vf)) return false;
1374 if(!context.Vf.Module(&context.Vf, &context.Vf, &context.N )) return false;
1375 }
1376 else
1377 {
1378 int keysize = GetKeySizeInBytes(CIPHERKEYTYPE_PRIVATE);
1379 if(!keysize) return false;
1380
1381 // Unblinding value: Vf = random number, invertible mod N
1382 do{ if(count++ > 10) return false;
1383
1384 if(!context.Vf.FillRandom(keysize-1, funcrandom, paramrandom)) return false;
1385 if(!context.Vi.GreatestCommonDivisor(&context.Vi, &context.Vf, &context.N )) return false;
1386
1387 } while(context.Vi.CompareSignedValues(1)!=0);
1388
1389 // Blinding value: Vi = Vf^(-e) mod N
1390 if(!context.Vi.ModularInverse( &context.Vf, &context.N)) return false;
1391 if(!context.Vi.SlidingWindowExponentiation(&context.Vi, &context.E, &context.N, &context.RN )) return false;
1392 }
1393
1394 if(Vi != &context.Vi)
1395 {
1396 if(!Vi->CopyFrom(&context.Vi)) return false;
1397 if(!Vf->CopyFrom(&context.Vf)) return false;
1398 }
1399
1400 return true;
1401}
