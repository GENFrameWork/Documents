102bool DIOWAKEONLAN::SendActivation(DIOMAC* MAC, DIOIP* broadcastIP, int timeout)
103{
104 #define DIOWAKEONLAN_NPACKETS 16
105
106 if(!MAC) return false;
107 if(!broadcastIP) return false;
108
109 DIOSTREAMUDPCONFIG diostreamudpcfg;
110 DIOSTREAMUDP* diostreamudp = NULL;
111 XSTRING broadcastIPstring;
112 XBYTE MACfull[DIOMAC_MAXSIZE] = { 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF };
113 bool status = false;
114
115 broadcastIP->GetXString(broadcastIPstring);
116
117 diostreamudpcfg.SetMode(DIOSTREAMMODE_CLIENT);
118 diostreamudpcfg.SetIsUsedDatagrams(false);
119 diostreamudpcfg.GetRemoteURL()->Set(broadcastIPstring.Get());
120 diostreamudpcfg.SetBroadcastModeActive(true);
121 diostreamudpcfg.SetRemotePort(9);
122
123 diostreamudp = (DIOSTREAMUDP*)GEN_DIOFACTORY.CreateStreamIO(&diostreamudpcfg);
124 if(!diostreamudp) return status;
125
126 if(diostreamudp->Open())
127 {
128 #define SIZEALL (DIOWAKEONLAN_NPACKETS+1) * DIOMAC_MAXSIZE
129 XBYTE tosend[ SIZEALL];
130
131 memset(tosend, 0, SIZEALL);
132
133 memcpy(&tosend[0], MACfull, DIOMAC_MAXSIZE);
134
135 for(XDWORD c=1; c<DIOWAKEONLAN_NPACKETS+1; c++)
136 {
137 memcpy(&tosend[c*DIOMAC_MAXSIZE], MAC->Get(), DIOMAC_MAXSIZE);
