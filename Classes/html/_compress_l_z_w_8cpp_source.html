207XDWORD COMPRESS_LZW::LZW_GIF_Encode(XBYTE* dibbuffer,XBYTE* outbuffer,XDWORD dibwidth,XDWORD dibheight,XWORD colorbit)
208{
209 hash = new XWORD [LZW_MAX_HASHSIZE] ;
210
211 if((hash == NULL) || (dibbuffer == NULL) || (outbuffer == NULL)) return 0;
212
213 bymincode = (colorbit == 1) ? 2 : colorbit;
214 LZWclear = 1 << bymincode;
215 LZWend = LZWclear + 1;
216 origin = curin = dibbuffer;
217 curout = outbuffer;
218 byinbit = 8;
219 byoutbit = 0;
220
221 m_dibwidth = dibwidth;
222 m_dibheight = dibheight;
223 currpixel = 0;
224 currheight = 0;
225 pitch = 4 * ((dibwidth * bymincode + 31) / 32);
226
227 XWORD Old;
228 XBYTE Pixel;
229
230 this->Encode_InitStringTable();
231 this->Encode_WriteIndex(LZWclear);
232
233 Old = this->Encode_GetNextPixel();
234
235 while(currheight<m_dibheight)
236 {
237 Pixel = this->Encode_GetNextPixel();
238
239 if(this->Encode_IsInTable (Old, Pixel))
240 {
241 Old = hash[(Old << 8) | Pixel];
242 }
243 else
244 {
245 this->Encode_WriteIndex(Old);
246 this->Encode_AddStringToTable(Old, Pixel);
247
248 Old = Pixel ;
249 if(currtableindex == LZW_MAX_TABLE_SIZE)
250 {
251 this->Encode_WriteIndex(Pixel);
252 this->Encode_WriteIndex(LZWclear);
253 this->Encode_InitStringTable();
254
255 Old = this->Encode_GetNextPixel();
256 }
257 }
258 }
259
260 this->Encode_WriteIndex(Old);
261 this->Encode_WriteIndex(LZWend);
262
263 delete [] hash;
264
265 return (XDWORD)(curout - outbuffer + 1);
266}

363XWORD COMPRESS_LZW::Decode_GetNextcode()
364{
365 XDWORD dwRet = 0;
366 unsigned int uiAdd = byinbit + bycurrbits;
367
368 if(uiAdd <= 8)
369 {
370 dwRet |= * curin;
371 }
372 else
373 {
374 if(uiAdd <= 16)
375 {
376 dwRet |= * (XWORD *) curin;
377 }
378 else
379 {
380 dwRet |= * (curin + 2);
381 dwRet <<= 16;
382 dwRet |= * (XWORD *) curin;
383 }
384 }
385
386 curin += uiAdd/8;
387 byinbit = uiAdd%8;
388
389 dwRet <<= 32 - uiAdd ;
390 dwRet >>= 32 - bycurrbits ;

537XBYTE COMPRESS_LZW::Encode_GetNextPixel()
538{
539 XBYTE byRet = 0;
540
541 switch(bymincode)
542 {
543 case 8 : byRet = curin[currpixel];
544 break;
545
546 case 4 : byRet = (currpixel % 2 == 0) ? curin[currpixel / 2] >> 4 : curin[currpixel / 2] & 0x0F;
547 break;
548
