375bool GRPBITMAP::FindSubBitmap(GRPBITMAP* bitmapref, int& x, int& y, XBYTE difflimitpercent, XBYTE pixelmargin)
376{
377 x = 0;
378 y = 0;
379
380 if(!bitmapref)
381 {
382 return false;

384
385 GRPBITMAP* _bitmap = bitmapref->ConvertToMode(GetMode());

388 return false;

393 return false;

397 XDWORD* bufferbitmap = (XDWORD*)_bitmap->GetBuffer();
398 XDWORD sizepixel = sizeof(XDWORD);
399 XDWORD bufferscreensize = (GetBufferSize() / sizepixel);
400 XDWORD bufferbmplinesize = _bitmap->GetWidth();
401 XDWORD ndiff = 0;
402 bool found = false;
403
404 for(XDWORD scrpos = 0; scrpos < bufferscreensize; scrpos++)
405 {
406 ndiff = 0;
407 for(XDWORD bmppos = 0; bmppos < bufferbmplinesize; bmppos++)
408 {
409 if(bufferscreen[scrpos + bmppos] != bufferbitmap[bmppos])
410 {
411 if(!IsSimilarPixel(bufferscreen[scrpos + bmppos], bufferbitmap[bmppos], pixelmargin))
412 {
413 ndiff++;
414 }
415 }
416 }
417
418 found = DifferencesPerCent(ndiff, bufferbmplinesize, difflimitpercent);
419 if(found)
420 {
421 found = false;
422
423 XDWORD srcpixelsline = GetWidth();
424 XDWORD scrpos_tmp = scrpos;
425 XDWORD bmppos_tmp = bufferbmplinesize;
426
427 x = (scrpos % GetWidth());
428 y = GetHeight() - (scrpos / srcpixelsline) - _bitmap->GetHeight();
429
430 scrpos_tmp += srcpixelsline;
431
432 for(XDWORD line = 1; line < _bitmap->GetHeight(); line++)
433 {
434 ndiff = 0;
435 for(XDWORD bmppos = 0; bmppos < bufferbmplinesize; bmppos++)

449

456 }

459 {

461 }
462 }
463 }

543bool GRPBITMAP::IsSimilarPixel(XDWORD origin, XDWORD target, XBYTE margin)
544{
545 XBYTE* originRGBA = (XBYTE*)&origin;
546 XBYTE* targetRGBA = (XBYTE*)&target;
547 bool status = false;
548
549 int ncomponent = 0;
550
551 for(int c=0; c<4; c++)
552 {
553 if(abs(originRGBA[c] - targetRGBA[c]) > margin)
554 {
555 ncomponent++;
556 }
557 }
558
559 if(!ncomponent)
560 {
561 status = true;
562 }
