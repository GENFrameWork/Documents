122bool APPFLOWBASE::Ini(XVECTOR<XSTRING*>* execparams)
123{
124 this->execparams = execparams;
125
126 GEN_XFACTORY_CREATE(xtimerglobal, CreateTimer())
127 if(!xtimerglobal) return false;
128
129 xtimerglobal->Reset();
130
131 firstupdate = false;
132
133 XDWORD code = GEN_XSYSTEM.GetLanguageSO();
134
135 if(code != XLANGUAGE_ISO_639_3_CODE_INVALID) XTRANSLATION_GEN::GetInstance().Sentences_AddToTranslation(code);
136
137 if(!AppProc_Ini()) return false;
138
139 return true;
140}

451void APPFLOWBASE::SetSystemChangeStatus(XSYSTEM_CHANGESTATUSTYPE systemchangestatustype)
452{
453 if(this->systemchangestatustype == systemchangestatustype)
454 {
455 return;
456 }
457
458 this->systemchangestatustype = systemchangestatustype;
459
460
461 XSTRING statusstr;
462
463 switch(systemchangestatustype)
464 {
465 case XSYSTEM_CHANGESTATUSTYPE_NONE :
466 default : statusstr = __L("normal"); break;
467 case XSYSTEM_CHANGESTATUSTYPE_REBOOT : statusstr = __L("reboot"); break;
468 case XSYSTEM_CHANGESTATUSTYPE_CONSOLE_CONNECT : statusstr = __L("console connect"); break;
469 case XSYSTEM_CHANGESTATUSTYPE_CONSOLE_DISCONNECT : statusstr = __L("console disconnect"); break;
470 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGON : statusstr = __L("session logon"); break;
471 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGOFF : statusstr = __L("session logooff"); break;
472 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOCK : statusstr = __L("session lock"); break;
473 case XSYSTEM_CHANGESTATUSTYPE_SESSION_UNLOCK : statusstr = __L("session unlock"); break;
474 case XSYSTEM_CHANGESTATUSTYPE_POWEROFF : statusstr = __L("power off"); break;
475 case XSYSTEM_CHANGESTATUSTYPE_SUSPEND : statusstr = __L("suspend"); break;
476 }
477
478 //XTRACE_PRINTCOLOR(XTRACE_COLOR_GREEN, __L("[APP Base System] Mode change to [%s]"), statusstr.Get());
479
480 APPFLOW_XEVENT xevent(this, APPFLOW_XEVENT_TYPE_CHANGESTATUSTYPE);
481 xevent.SetChangeStatusType(systemchangestatustype);
482 PostEvent(&xevent);
483}
