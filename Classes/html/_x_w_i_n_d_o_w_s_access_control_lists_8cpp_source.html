110bool XWINDOWSACCESSCONTROLLISTS::SetFileOnlyPermissionForAdmin(XCHAR* filename)
111{
112 #define EXPLICIT_ACCESS_MAXNUM_ADMIN 2
113
114 PSID everyoneSID = NULL;
115 PSID adminSID = NULL;
116 PACL ACL = NULL;
117 EXPLICIT_ACCESS ea[EXPLICIT_ACCESS_MAXNUM_ADMIN];
118 SID_IDENTIFIER_AUTHORITY SIDauthworld = SECURITY_WORLD_SID_AUTHORITY;
119 SID_IDENTIFIER_AUTHORITY SIDAuthNT = SECURITY_NT_AUTHORITY;
120
121 // Create a well-known SID for the Everyone group.
122 AllocateAndInitializeSid(&SIDauthworld , 1
123 , SECURITY_WORLD_RID
124 , 0, 0, 0, 0, 0, 0, 0
125 , &everyoneSID);
126
127
128 AllocateAndInitializeSid(&SIDAuthNT , 2
129 , SECURITY_BUILTIN_DOMAIN_RID
130 , DOMAIN_ALIAS_RID_ADMINS
131 , 0, 0, 0, 0, 0, 0
132 , &adminSID);
133
134
135 // Initialize an EXPLICIT_ACCESS structure for an ACE.
136 ZeroMemory(&ea, EXPLICIT_ACCESS_MAXNUM_ADMIN * sizeof(EXPLICIT_ACCESS));
137
138 // Set read access for Everyone.
139 ea[0].grfAccessPermissions = GENERIC_READ;
140 ea[0].grfAccessMode = SET_ACCESS;
141 ea[0].grfInheritance = NO_INHERITANCE;
142 ea[0].Trustee.TrusteeForm = TRUSTEE_IS_SID;
143 ea[0].Trustee.TrusteeType = TRUSTEE_IS_WELL_KNOWN_GROUP;
144 ea[0].Trustee.ptstrName = (LPTSTR) everyoneSID;
145
146 // Set full control for Administrators.
147 ea[1].grfAccessPermissions = GENERIC_ALL;
148 ea[1].grfAccessMode = SET_ACCESS;
149 ea[1].grfInheritance = NO_INHERITANCE;
150 ea[1].Trustee.TrusteeForm = TRUSTEE_IS_SID;
151 ea[1].Trustee.TrusteeType = TRUSTEE_IS_GROUP;
152 ea[1].Trustee.ptstrName = (LPTSTR) adminSID;
153
154 // Create a new ACL that contains the new ACEs.
155 SetEntriesInAcl(EXPLICIT_ACCESS_MAXNUM_ADMIN, ea, NULL, &ACL);
156
157 // Initialize a security descriptor.
158 PSECURITY_DESCRIPTOR SD = (PSECURITY_DESCRIPTOR) LocalAlloc(LPTR, SECURITY_DESCRIPTOR_MIN_LENGTH);
159
160 InitializeSecurityDescriptor(SD, SECURITY_DESCRIPTOR_REVISION);
161
162 // Add the ACL to the security descriptor.
163 SetSecurityDescriptorDacl(SD, TRUE // bDaclPresent flag
164 , ACL
165 , FALSE); // not a default DACL
166
167
168 // Change the security attributes
169 BOOL status = SetFileSecurity(filename, DACL_SECURITY_INFORMATION, SD);
170
171 if(everyoneSID) FreeSid(everyoneSID);
172 if(adminSID) FreeSid(adminSID);
173 if(ACL) LocalFree(ACL);
174 if(SD) LocalFree(SD);
175
176 return status?true:false;
177}
