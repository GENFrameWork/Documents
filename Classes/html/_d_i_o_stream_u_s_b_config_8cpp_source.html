276bool DIOSTREAMUSBCONFIG::SetFromString(XCHAR* string)
277{
278 if(!string) return false;
279
280 XSTRING _string = string;
281 int _classdev = 0;
282
283 if(_string.Find(DIOSTREAMUSBLOCALENUMDEVICES_PORTKEY, true) != XSTRING_NOTFOUND)
284 {
285 if(_string.Find(DIOSTREAMUSBLOCALENUMDEVICES_PORTHUB, true) != XSTRING_NOTFOUND)
286 {
287 XSTRING _location;
288
289 _string.UnFormat(__L("%s,%d"), location.Get(), &classdev);
290
291 location = _location;
292 classdev = (DIOSTREAMDEVICEUSBCLASS)_classdev;
293
294 return true;
295 }
296 }
297
298 XDWORD _vendorID = 0;
299 XDWORD _productID = 0;
300
301 _string.UnFormat(__L("0x%04X,0x%04X,%d"), &_vendorID, &_productID, &_classdev);
302
303 vendorID = _vendorID;
304 productID = _productID;
305 classdev = (DIOSTREAMDEVICEUSBCLASS)_classdev;
306
307 return ((vendorID && productID)?true:false);
308}
