129bool DIOI2CPWMCONTROLERPCA9685::SetPWMFrequency(float frequency)
130{
131 if(!IsInitialized()) return false;
132
133 float prescaleval;
134
135 prescaleval = 25000000; // 25MHz
136 prescaleval /= 4096; // 12 bits
137 prescaleval /= frequency;
138 prescaleval -= 1;
139
140 XBYTE prescale = (XBYTE)floor(prescaleval + 0.5f);
141 XBYTE oldmode;
142 bool status;
143
144 status = ReadCommand(DIOI2CPWMCONTROLERPCA9685_MODE1, oldmode);
145 if(!status) return false;
146
147 XBYTE newmode = (oldmode&0x7F) | 0x10; // sleep
148
149 status = SendCommand(DIOI2CPWMCONTROLERPCA9685_MODE1 , newmode); // go to sleep
150 if(!status) return false;
151
152 status = SendCommand(DIOI2CPWMCONTROLERPCA9685_PRESCALE , prescale); // set the prescaler
153 if(!status) return false;
154
155 status = SendCommand(DIOI2CPWMCONTROLERPCA9685_MODE1 , oldmode);
156 if(!status) return false;
157
158 xtimerout->SetMilliSeconds(50);
159
160 return SendCommand(DIOI2CPWMCONTROLERPCA9685_MODE1, (oldmode | 0x80));
161}

178bool DIOI2CPWMCONTROLERPCA9685::SetPWM(XBYTE channel, XWORD on, XWORD off, bool wait)
179{
180 if(!IsInitialized()) return false;
181
182 if(!SendCommand(DIOI2CPWMCONTROLERPCA9685_LED0_ON_L + (4*channel), (on & 0xFF), wait)) return false;
183 if(!SendCommand(DIOI2CPWMCONTROLERPCA9685_LED0_ON_H + (4*channel), (on >> 0x08), wait)) return false;
184 if(!SendCommand(DIOI2CPWMCONTROLERPCA9685_LED0_OFF_L + (4*channel), (off & 0xFF), wait)) return false;
185 if(!SendCommand(DIOI2CPWMCONTROLERPCA9685_LED0_OFF_H + (4*channel), (off >> 0x08), wait)) return false;
186
187 return true;
188}

248bool DIOI2CPWMCONTROLERPCA9685::ReadCommand(XBYTE cmd, XBYTE& value)
249{
250 if(!diostream) return false;
251
252 bool status;
253
254 status = diostream->Write(&cmd, 1)?true:false;
255
256 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
257 if(!status) return false;
258
259 status = diostream->WaitToFilledReadingBuffer(1, timeout);
