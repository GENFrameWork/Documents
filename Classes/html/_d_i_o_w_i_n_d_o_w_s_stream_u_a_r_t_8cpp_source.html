156{

164 XSTRING comport;

166
167 comport.Empty();
168
169 if(!config->GetLocalDeviceName()->IsEmpty())
170 {

172 }
173 else
174 {
175 comport.Format(__L("\\\\.\\COM%d"), config->GetPort());
176 }
177
178 hcom = CreateFile(comport.Get() ,
179 GENERIC_READ|GENERIC_WRITE ,
180 0 ,
181 0 ,
182 OPEN_EXISTING ,
183 FILE_ATTRIBUTE_NORMAL , // FILE_FLAG_OVERLAPPED
184 0);
185
186 if(hcom==INVALID_HANDLE_VALUE) return false;
187
188 if(!SetTimeouts())

210
211

247bool DIOWINDOWSSTREAMUART::Config(XWORD mask)
248{
249 if(!config) return false;
250 if(GetStatus()==DIOSTREAMSTATUS_DISCONNECTED) return false;
251
252 DCB dcb;
253
254 if(!GetCommState(hcom, &dcb)) return false;
255
256 if((mask&DIOSTREAMUARTMASK_BAUDRATE) == DIOSTREAMUARTMASK_BAUDRATE) dcb.BaudRate = config->GetBaudRate();
257
258 if((mask&DIOSTREAMUARTMASK_DATABITS ) == DIOSTREAMUARTMASK_DATABITS)
259 {
260 switch(config->GetDataBits())
261 {
262 case DIOSTREAMUARTDATABIT_5 : dcb.ByteSize = 5; break;
263 case DIOSTREAMUARTDATABIT_6 : dcb.ByteSize = 6; break;
264 case DIOSTREAMUARTDATABIT_7 : dcb.ByteSize = 7; break;
265 case DIOSTREAMUARTDATABIT_8 : dcb.ByteSize = 8; break;
266 default : break;
267 }
268 }
269
270 if((mask&DIOSTREAMUARTMASK_PARITY) == DIOSTREAMUARTMASK_PARITY)
271 {
272 dcb.fParity = TRUE;
273
274 switch(config->GetParity())
275 {
276 case DIOSTREAMUARTPARITY_NONE : dcb.Parity = NOPARITY;
277 dcb.fParity = FALSE;
278 break;
279 case DIOSTREAMUARTPARITY_ODD : dcb.Parity = ODDPARITY; break;
280 case DIOSTREAMUARTPARITY_EVEN : dcb.Parity = EVENPARITY; break;
281 case DIOSTREAMUARTPARITY_MARK : dcb.Parity = MARKPARITY; break;
282 case DIOSTREAMUARTPARITY_SPACE : dcb.Parity = SPACEPARITY; break;
283 }
284 }
285
286
287 if((mask&DIOSTREAMUARTMASK_STOPBITS ) == DIOSTREAMUARTMASK_STOPBITS)
288 {
289 switch(config->GetStopBits())
290 {
291 case DIOSTREAMUARTSTOPBITS_ONE : dcb.StopBits = ONESTOPBIT; break;
292 case DIOSTREAMUARTSTOPBITS_ONEANDAHALF :
293 case DIOSTREAMUARTSTOPBITS_TWO : dcb.StopBits = TWOSTOPBITS; break;
294 }
295 }
296
297 if((mask&DIOSTREAMUARTMASK_FLOWCONTROL) == DIOSTREAMUARTMASK_FLOWCONTROL)
298 {
299 switch(config->GetFlowControl())
300 {
301 case DIOSTREAMUARTFLOWCONTROL_NONE : dcb.fDtrControl = DTR_CONTROL_DISABLE;
302 dcb.fRtsControl = RTS_CONTROL_DISABLE;
303 break;
304
305 case DIOSTREAMUARTFLOWCONTROL_HARD : dcb.fDtrControl = DTR_CONTROL_ENABLE;
306 dcb.fRtsControl = RTS_CONTROL_ENABLE;
307 break;
308
309 case DIOSTREAMUARTFLOWCONTROL_SOFT : dcb.fDtrControl = DTR_CONTROL_DISABLE;
310 dcb.fRtsControl = RTS_CONTROL_DISABLE;
311 break;
312 }
313 }
314
315 dcb.XonChar = 1;
316 dcb.XoffChar = 0;
317
318 if(!SetCommState(hcom,&dcb)) return false;
319
320 return true;
321}

336XDWORD DIOWINDOWSSTREAMUART::ReadDirect(XBYTE* buffer,XDWORD size)
337{
338 if(GetStatus()==DIOSTREAMSTATUS_DISCONNECTED) return 0;
339 if(size == 0) return 0;
340
341 DWORD br = 0;
342 bool status = true;
343
344 if(!hevent)
345 {
346 hevent = CreateEvent(0,TRUE,FALSE,0);
347 if(!hevent) return 0;
348
349 ovi.hEvent = hevent;
350
351 if(!ReadFile(hcom,buffer,size,&br,&ovi))
352 {
353 if(GetLastError()!=ERROR_IO_PENDING)
354 {
355 status = false;
356 }
357 else return ReadDirect(buffer, size);
358 }
359 }
360 else
361 {
362 switch(WaitForSingleObject(ovi.hEvent,readtimeout))
363 {
364 case WAIT_OBJECT_0: if(!GetOverlappedResult(hcom,&ovi,&br,FALSE))
365 status = false;
366 else status = true;
367 break;
368
369 case WAIT_TIMEOUT : CancelIo(hcom);
370 status=false;
371 break;
372
373 default : status=false;
374 break;
375
376 }
377 }
378
379 memset(&ovi,0,sizeof(ovi));
380
381 if(hevent)
382 {
383 CloseHandle(hevent);
384 hevent = NULL;
385 }
386
387 if(!status) return 0;
388
389 return (XDWORD)br;
390}

405XDWORD DIOWINDOWSSTREAMUART::WriteDirect(XBYTE* buffer, XDWORD size)
406{
407 if(GetStatus()==DIOSTREAMSTATUS_DISCONNECTED) return 0;
408
409 if(size == 0) return 0;
410
411 DWORD bw = 0;
412 OVERLAPPED oviwrite = { 0 };
413 bool status;
414
415 oviwrite.hEvent = CreateEvent(0,TRUE,FALSE,0);
416 if(!oviwrite.hEvent) return bw;
417
418 if(!WriteFile(hcom,buffer,size,&bw,&oviwrite))
419 {
420 if(GetLastError()!=ERROR_IO_PENDING)
421 {
422 status = false;
423 }
424 else
425 {
426 switch(WaitForSingleObject(oviwrite.hEvent,writetimeout))
427 {
428 case WAIT_OBJECT_0 : if(!GetOverlappedResult(hcom,&oviwrite,&bw,TRUE))
429 status = false;
430 else status = true;
431 break;
432
433 case WAIT_TIMEOUT : CancelIo(hcom);
434 status=false;
435 break;
436
437 default : status=false;
438 break;
439
440 }
441 }
442
443 } else status = true;
444
445 CloseHandle(oviwrite.hEvent);
446
447 if(!status) return 0;
448
449 return (XDWORD)bw;
450}

714void DIOWINDOWSSTREAMUART::ThreadConnection(void* data)
715{
716 DIOWINDOWSSTREAMUART* diostream = (DIOWINDOWSSTREAMUART*)data;
717 if(!diostream) return;
718
719 if(diostream->GetEvent()==DIOWINDOWSUARTFSMEVENT_NONE) // No hay nuevos Eventos
720 {
721 switch(diostream->GetCurrentState())
722 {
723 case DIOWINDOWSUARTFSMSTATE_NONE : break;
724
725 case DIOWINDOWSUARTFSMSTATE_CONNECTED : break;
726
727 case DIOWINDOWSUARTFSMSTATE_WAITINGTOREAD : { if(!diostream->IsBlockRead())
728 {
729 XBYTE buffer[DIOSTREAM_MAXBUFFER];
730
731 XDWORD br = diostream->ReadDirect(buffer, DIOSTREAM_MAXBUFFER);
732 if(br)
733 {
734 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(2), (XBYTE*)diostream->buffer, (int)br);
735 diostream->inbuffer->Add(buffer, br);
736 }
737 }
738
739 if(!diostream->IsBlockWrite())
740 {
741 int esize = diostream->outbuffer->GetSize();
742 if(esize)
743 {
744 diostream->outbuffer->SetBlocked(true);
745 XDWORD bw = diostream->WriteDirect(diostream->outbuffer->Get(), esize);
746 diostream->outbuffer->SetBlocked(false);
747
748 if(bw) diostream->outbuffer->Extract(NULL, 0, bw);
749
750 }
751 }
752 }
753 break;
754
755 case DIOWINDOWSUARTFSMSTATE_DISCONNECTING : break;
756
757 }
758 }
759 else
760 {
761 if(diostream->GetEvent()<DIOWINDOWSUART_LASTEVENT)
762 {
763 diostream->CheckTransition();
764
765 switch(diostream->GetCurrentState())
766 {
767 case DIOWINDOWSUARTFSMSTATE_NONE : break;
768
769 case DIOWINDOWSUARTFSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
770 xevent.SetDIOStream(diostream);
771 diostream->PostEvent(&xevent);
772
773 diostream->SetEvent(DIOWINDOWSUARTFSMEVENT_WAITINGTOREAD);
774 }
775 break;
776
777 case DIOWINDOWSUARTFSMSTATE_WAITINGTOREAD : break;
778
779 case DIOWINDOWSUARTFSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
780 xevent.SetDIOStream(diostream);
781
782 diostream->PostEvent(&xevent);
783
784 diostream->threadconnection->Run(false);
785 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
786 }
787 break;
788 }
789 }
790 }
791}
