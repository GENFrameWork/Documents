464XQWORDSIG CIPHERCURVE25519::div_by_2_26(const XQWORDSIG v)
465{
466 const XDWORD highword = (XDWORD)(((XQWORD) v) >> 32); // High word of v; no shift needed
467 const XDWORDSIG sign = ((XDWORDSIG) highword) >> 31; // Set to all 1s if v was negative; else set to 0s.
468 const XDWORDSIG roundoff = ((XDWORD) sign) >> 6; // Set to 0x3ffffff if v was negative; else set to 0.
469
470 return (v + roundoff) >> 26; // Should return v / (1<<26)
471}

524void CIPHERCURVE25519::freduce_coefficients(XQWORDSIG* output)
525{
526 XDWORD i;
527
528 output[10] = 0;
529
530 for(i=0; i<10; i+=2)
531 {
532 XQWORDSIG over = div_by_2_26(output[i]);
533
534 output[i] -= over << 26;
535 output[i+1] += over;
536
537 over = div_by_2_25(output[i+1]);
538 output[i+1] -= over << 25;
539 output[i+2] += over;
540 }
541
542 // Now |output[10]| < 2 ^ 38 and all other coefficients are reduced.
543 output[0] += output[10] << 4;
544 output[0] += output[10] << 1;
545 output[0] += output[10];
546
547 output[10] = 0;
548
549 // Now output[1..9] are reduced, and |output[0]| < 2^26 + 19 * 2^38 So |over| will be no more than 77825
550 {
551 XQWORDSIG over = div_by_2_26(output[0]);
552 output[0] -= over << 26;
553 output[1] += over;
554 }
555
556 // Now output[0,2..9] are reduced, and |output[1]| < 2^25 + 77825 So |over| will be no more than 1.
557 {
558 // output[1] fits in 32 bits, so we can use div_s32_by_2_25 here.
559 XDWORDSIG over32 = div_s32_by_2_25((XDWORDSIG)output[1]);
560 output[1] -= over32 << 25;
561 output[2] += over32;
562 }
563
564 // Finally, output[0,1,3..9] are reduced, and output[2] is "nearly reduced":
565 // we have |output[2]| <= 2^26. This is good enough for all of our math, but it will require an extra freduce_coefficients before fcontract.
566}

720void CIPHERCURVE25519::fcontract(XBYTE* output, XQWORDSIG* input)
721{
722 int i;
723 int j;
724
725 for(j=0; j<2; ++j)
726 {
727 for(i=0; i<9; ++i)
728 {
729 if((i & 1) == 1)
730 {
731 // This calculation is a time-invariant way to make input[i] positive by borrowing from the next-larger XQWORDSIG.
732
733 const XDWORDSIG mask = (XDWORDSIG)(input[i]) >> 31;
734 const XDWORDSIG carry = -(((XDWORDSIG)(input[i]) & mask) >> 25);
735
736 input[i] = (XDWORDSIG)(input[i]) + (carry << 25);
737 input[i+1] = (XDWORDSIG)(input[i+1]) - carry;
738
739 }
740 else
741 {
742 const XDWORDSIG mask = (XDWORDSIG)(input[i]) >> 31;
743 const XDWORDSIG carry = -(((XDWORDSIG)(input[i]) & mask) >> 26);
744
745 input[i] = (XDWORDSIG)(input[i]) + (carry << 26);
746 input[i+1] = (XDWORDSIG)(input[i+1]) - carry;
747 }
748 }
749
750 const XDWORDSIG mask = (XDWORDSIG)(input[9]) >> 31;
751 const XDWORDSIG carry = -(((XDWORDSIG)(input[9]) & mask) >> 25);
752
753 input[9] = (XDWORDSIG)(input[9]) + (carry << 25);
754 input[0] = (XDWORDSIG)(input[0]) - (carry * 19);
755 }
756
757 // The first borrow-propagation pass above ended with every XQWORDSIG except (possibly) input[0] non-negative.
758 //
759 // Since each input XQWORDSIG except input[0] is decreased by at most 1 by a borrow-propagation pass, the second borrow-propagation pass
760 // could only have wrapped around to decrease input[0] again if the first pass left input[0] negative *and* input[1] through input[9]
761 // were all zero. In that case, input[1] is now 2^25 - 1, and this last borrow-propagation step will leave input[1] non-negative.
762
763 const XDWORDSIG mask = (XDWORDSIG)(input[0]) >> 31;
764 const XDWORDSIG carry = -(((XDWORDSIG)(input[0]) & mask) >> 26);
765
766 input[0] = (XDWORDSIG)(input[0]) + (carry << 26);
767 input[1] = (XDWORDSIG)(input[1]) - carry;
768
769 // Both passes through the above loop, plus the last 0-to-1 step, are necessary: if input[9] is -1 and input[0] through input[8] are 0,
770 // negative values will remain in the array until the end.
771
772
773 input[1] <<= 2;
774 input[2] <<= 3;
775 input[3] <<= 5;
776 input[4] <<= 6;
777 input[6] <<= 1;
778 input[7] <<= 3;
779 input[8] <<= 4;
780 input[9] <<= 6;
781
782 output[ 0] = 0;
783 output[ 16] = 0;
784
785 output[ 0+0] |= input[0] & 0xff; output[ 0+1] = (input[0] >> 8) & 0xff; output[ 0+2] = (input[0] >> 16) & 0xff; output[ 0+3] = (input[0] >> 24) & 0xff;;
786 output[ 3+0] |= input[1] & 0xff; output[ 3+1] = (input[1] >> 8) & 0xff; output[ 3+2] = (input[1] >> 16) & 0xff; output[ 3+3] = (input[1] >> 24) & 0xff;;
787 output[ 6+0] |= input[2] & 0xff; output[ 6+1] = (input[2] >> 8) & 0xff; output[ 6+2] = (input[2] >> 16) & 0xff; output[ 6+3] = (input[2] >> 24) & 0xff;;
788 output[ 9+0] |= input[3] & 0xff; output[ 9+1] = (input[3] >> 8) & 0xff; output[ 9+2] = (input[3] >> 16) & 0xff; output[ 9+3] = (input[3] >> 24) & 0xff;;
789 output[12+0] |= input[4] & 0xff; output[12+1] = (input[4] >> 8) & 0xff; output[12+2] = (input[4] >> 16) & 0xff; output[12+3] = (input[4] >> 24) & 0xff;;
790 output[16+0] |= input[5] & 0xff; output[16+1] = (input[5] >> 8) & 0xff; output[16+2] = (input[5] >> 16) & 0xff; output[16+3] = (input[5] >> 24) & 0xff;;
791 output[19+0] |= input[6] & 0xff; output[19+1] = (input[6] >> 8) & 0xff; output[19+2] = (input[6] >> 16) & 0xff; output[19+3] = (input[6] >> 24) & 0xff;;
792 output[22+0] |= input[7] & 0xff; output[22+1] = (input[7] >> 8) & 0xff; output[22+2] = (input[7] >> 16) & 0xff; output[22+3] = (input[7] >> 24) & 0xff;;
793 output[25+0] |= input[8] & 0xff; output[25+1] = (input[8] >> 8) & 0xff; output[25+2] = (input[8] >> 16) & 0xff; output[25+3] = (input[8] >> 24) & 0xff;;
794 output[28+0] |= input[9] & 0xff; output[28+1] = (input[9] >> 8) & 0xff; output[28+2] = (input[9] >> 16) & 0xff; output[28+3] = (input[9] >> 24) & 0xff;;
795}

956 nqx2 = t;
