452XQWORDSIG CIPHERCURVE25519::div_by_2_26(const XQWORDSIG v)
453{
454 const XDWORD highword = (XDWORD)(((XQWORD) v) >> 32); // High word of v; no shift needed
455 const XDWORDSIG sign = ((XDWORDSIG) highword) >> 31; // Set to all 1s if v was negative; else set to 0s.
456 const XDWORDSIG roundoff = ((XDWORD) sign) >> 6; // Set to 0x3ffffff if v was negative; else set to 0.
457
458 return (v + roundoff) >> 26; // Should return v / (1<<26)
459}

512void CIPHERCURVE25519::freduce_coefficients(XQWORDSIG* output)
513{
514 XDWORD i;
515
516 output[10] = 0;
517
518 for(i=0; i<10; i+=2)
519 {
520 XQWORDSIG over = div_by_2_26(output[i]);
521
522 output[i] -= over << 26;
523 output[i+1] += over;
524
525 over = div_by_2_25(output[i+1]);
526 output[i+1] -= over << 25;
527 output[i+2] += over;
528 }
529
530 // Now |output[10]| < 2 ^ 38 and all other coefficients are reduced.
531 output[0] += output[10] << 4;
532 output[0] += output[10] << 1;
533 output[0] += output[10];
534
535 output[10] = 0;
536
537 // Now output[1..9] are reduced, and |output[0]| < 2^26 + 19 * 2^38 So |over| will be no more than 77825
538 {
539 XQWORDSIG over = div_by_2_26(output[0]);
540 output[0] -= over << 26;
541 output[1] += over;
542 }
543
544 // Now output[0,2..9] are reduced, and |output[1]| < 2^25 + 77825 So |over| will be no more than 1.
545 {
546 // output[1] fits in 32 bits, so we can use div_s32_by_2_25 here.
547 XDWORDSIG over32 = div_s32_by_2_25((XDWORDSIG)output[1]);
548 output[1] -= over32 << 25;
549 output[2] += over32;
550 }
551
552 // Finally, output[0,1,3..9] are reduced, and output[2] is "nearly reduced":
553 // we have |output[2]| <= 2^26. This is good enough for all of our math, but it will require an extra freduce_coefficients before fcontract.
554}

708void CIPHERCURVE25519::fcontract(XBYTE* output, XQWORDSIG* input)
709{
710 int i;
711 int j;
712
713 for(j=0; j<2; ++j)
714 {
715 for(i=0; i<9; ++i)
716 {
717 if((i & 1) == 1)
718 {
719 // This calculation is a time-invariant way to make input[i] positive by borrowing from the next-larger XQWORDSIG.
720
721 const XDWORDSIG mask = (XDWORDSIG)(input[i]) >> 31;
722 const XDWORDSIG carry = -(((XDWORDSIG)(input[i]) & mask) >> 25);
723
724 input[i] = (XDWORDSIG)(input[i]) + (carry << 25);
725 input[i+1] = (XDWORDSIG)(input[i+1]) - carry;
726
727 }
728 else
729 {
730 const XDWORDSIG mask = (XDWORDSIG)(input[i]) >> 31;
731 const XDWORDSIG carry = -(((XDWORDSIG)(input[i]) & mask) >> 26);
732
733 input[i] = (XDWORDSIG)(input[i]) + (carry << 26);
734 input[i+1] = (XDWORDSIG)(input[i+1]) - carry;
735 }
736 }
737
738 const XDWORDSIG mask = (XDWORDSIG)(input[9]) >> 31;
739 const XDWORDSIG carry = -(((XDWORDSIG)(input[9]) & mask) >> 25);
740
741 input[9] = (XDWORDSIG)(input[9]) + (carry << 25);
742 input[0] = (XDWORDSIG)(input[0]) - (carry * 19);
743 }
744
745 // The first borrow-propagation pass above ended with every XQWORDSIG except (possibly) input[0] non-negative.
746 //
747 // Since each input XQWORDSIG except input[0] is decreased by at most 1 by a borrow-propagation pass, the second borrow-propagation pass
748 // could only have wrapped around to decrease input[0] again if the first pass left input[0] negative *and* input[1] through input[9]
749 // were all zero. In that case, input[1] is now 2^25 - 1, and this last borrow-propagation step will leave input[1] non-negative.
750
751 const XDWORDSIG mask = (XDWORDSIG)(input[0]) >> 31;
752 const XDWORDSIG carry = -(((XDWORDSIG)(input[0]) & mask) >> 26);
753
754 input[0] = (XDWORDSIG)(input[0]) + (carry << 26);
755 input[1] = (XDWORDSIG)(input[1]) - carry;
756
757 // Both passes through the above loop, plus the last 0-to-1 step, are necessary: if input[9] is -1 and input[0] through input[8] are 0,
758 // negative values will remain in the array until the end.
759
760
761 input[1] <<= 2;
762 input[2] <<= 3;
763 input[3] <<= 5;
764 input[4] <<= 6;
765 input[6] <<= 1;
766 input[7] <<= 3;
767 input[8] <<= 4;
768 input[9] <<= 6;
769
770 output[ 0] = 0;
771 output[ 16] = 0;
772
773 output[ 0+0] |= input[0] & 0xff; output[ 0+1] = (input[0] >> 8) & 0xff; output[ 0+2] = (input[0] >> 16) & 0xff; output[ 0+3] = (input[0] >> 24) & 0xff;;
774 output[ 3+0] |= input[1] & 0xff; output[ 3+1] = (input[1] >> 8) & 0xff; output[ 3+2] = (input[1] >> 16) & 0xff; output[ 3+3] = (input[1] >> 24) & 0xff;;
775 output[ 6+0] |= input[2] & 0xff; output[ 6+1] = (input[2] >> 8) & 0xff; output[ 6+2] = (input[2] >> 16) & 0xff; output[ 6+3] = (input[2] >> 24) & 0xff;;
776 output[ 9+0] |= input[3] & 0xff; output[ 9+1] = (input[3] >> 8) & 0xff; output[ 9+2] = (input[3] >> 16) & 0xff; output[ 9+3] = (input[3] >> 24) & 0xff;;
777 output[12+0] |= input[4] & 0xff; output[12+1] = (input[4] >> 8) & 0xff; output[12+2] = (input[4] >> 16) & 0xff; output[12+3] = (input[4] >> 24) & 0xff;;
778 output[16+0] |= input[5] & 0xff; output[16+1] = (input[5] >> 8) & 0xff; output[16+2] = (input[5] >> 16) & 0xff; output[16+3] = (input[5] >> 24) & 0xff;;
779 output[19+0] |= input[6] & 0xff; output[19+1] = (input[6] >> 8) & 0xff; output[19+2] = (input[6] >> 16) & 0xff; output[19+3] = (input[6] >> 24) & 0xff;;
780 output[22+0] |= input[7] & 0xff; output[22+1] = (input[7] >> 8) & 0xff; output[22+2] = (input[7] >> 16) & 0xff; output[22+3] = (input[7] >> 24) & 0xff;;
781 output[25+0] |= input[8] & 0xff; output[25+1] = (input[8] >> 8) & 0xff; output[25+2] = (input[8] >> 16) & 0xff; output[25+3] = (input[8] >> 24) & 0xff;;
782 output[28+0] |= input[9] & 0xff; output[28+1] = (input[9] >> 8) & 0xff; output[28+2] = (input[9] >> 16) & 0xff; output[28+3] = (input[9] >> 24) & 0xff;;
783}
