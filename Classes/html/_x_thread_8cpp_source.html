308bool XTHREAD::WaitToEnd(XDWORD timeout)
309{
310 if(!gotofunction)
311 {
312 statusfunc = XTHREADSTATUS_END;
313 return true;
314 }
315
316 bool statustimeout = false;
317
318 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer())
319 if(!xtimerout) return false;
320
321 while(statusfunc != XTHREADSTATUS_END)
322 {
323 Wait();
324
325 if(xtimerout->GetMeasureSeconds() >= timeout)
326 {
327 statustimeout = true;
328 break;
329 }
330 }
331
332 GEN_XFACTORY.DeleteTimer(xtimerout);
333
334 return (!statustimeout);
335}
