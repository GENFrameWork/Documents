301bool XTHREAD::WaitToEnd(XDWORD timeout)
302{
303 if(!gotofunction)

305 statusfunc = XTHREADSTATUS_END;
306 return true;
307 }

310
311 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer())
312 if(!xtimerout) return false;
313
314 while(statusfunc != XTHREADSTATUS_END)
