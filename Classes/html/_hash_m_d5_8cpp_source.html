191void HASHMD5::Update(XBYTE* input, XQWORD size)
192{
193 // compute number of bytes mod 64
194 XDWORD index = count[0] / 8 % HASHMD5_BLOCKSIZE;
195
196 // Update number of bits
197 if ((count[0] += (XDWORD)(size << 3)) < (XDWORD)(size << 3)) count[1]++;
198
199 count[1] += (XDWORD)(size >> 29);
200
201 // number of bytes we need to fill in buffer
202 XDWORD firstpart = 64 - index;
203
204 XDWORD i;
205
206 // transform as many times as possible.
207 if(size >= firstpart)
208 {
209 // fill buffer first, transform
210 memcpy(&buffer[index], input, firstpart);
211
212 Transform(buffer);
213
214 // transform chunks of HASHMD5_BLOCKSIZE (64 bytes)
215 for(i = firstpart; i + HASHMD5_BLOCKSIZE <= (XDWORD)size; i += HASHMD5_BLOCKSIZE)
216 {
217 Transform(&input[i]);
218 }
219
220 index = 0;
221
222 } else i = 0;
223
224 // buffer remaining input
225 memcpy(&buffer[index], &input[i], (XDWORD)(size-i));
226}

342{

362{

406
407 // Zeroize sensitive information.
408 memset(buffer, 0, sizeof buffer);
409 memset(count, 0, sizeof count);
