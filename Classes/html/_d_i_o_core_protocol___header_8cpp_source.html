866 #endif
867
868 Primitive_Add<XDWORD>(block_index, DIOCOREPROTOCOL_HEADER_VAR_BLOCK_INDEX);

870 Primitive_Add<XDWORD>(block_amount, DIOCOREPROTOCOL_HEADER_VAR_BLOCK_AMOUNT);
871

892bool DIOCOREPROTOCOL_HEADER::Deserialize()
893{
894 XSTRING string;
895
896 string.Empty();
897 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_ID_MESSAGE);
898 ID_message.SetFromString(string);
899
900 #ifdef DIOCOREPROTOCOL_HUMANFORMAT_ACTIVE
901 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_MESSAGE_TYPE);
902 GetMessageTypeFromString(&string, message_type);
903 #else
904 Primitive_Extract<int&>((int&)message_type, DIOCOREPROTOCOL_HEADER_VAR_MESSAGE_TYPE);
905 #endif
906
907 #ifdef DIOCOREPROTOCOL_HUMANFORMAT_ACTIVE
908 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_OPERATION);
909 GetOperationFromString(&string, operation);
910 #else
911 Primitive_Extract<int&>((int&)operation, DIOCOREPROTOCOL_HEADER_VAR_OPERATION);
912 #endif
913
914 Primitive_Extract<XSTRING&>(operation_param, DIOCOREPROTOCOL_HEADER_VAR_OPERATION_PARAM);
915
916 string.Empty();
917 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_DATETIME_SEND);
918 datetime_send->GetDateTimeFromStringISO8601(string, XDATETIME_FORMAT_ISO8601_STANDARD | XDATETIME_FORMAT_TIMEWITHMILLISECONDS);
919
920 #ifdef DIOCOREPROTOCOL_HUMANFORMAT_ACTIVE
921 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_CONTENT_TYPE);
922 GetContentTypeFromString(&string, content_type);
923 #else
924 Primitive_Extract<int&>((int&)content_type, DIOCOREPROTOCOL_HEADER_VAR_CONTENT_TYPE);
925 #endif
926
927 Primitive_Extract<XDWORD&>(block_index, DIOCOREPROTOCOL_HEADER_VAR_BLOCK_INDEX);
928
929 Primitive_Extract<XDWORD&>(block_amount, DIOCOREPROTOCOL_HEADER_VAR_BLOCK_AMOUNT);
930
931 Primitive_Extract<XDWORD&>(content_size, DIOCOREPROTOCOL_HEADER_VAR_CONTENT_SIZE);
932
933 Primitive_Extract<XDWORD&>(content_compresssize, DIOCOREPROTOCOL_HEADER_VAR_CONTENT_COMPRESSSIZE);
934
935 string.Empty();
936 Primitive_Extract<XSTRING&>(string, DIOCOREPROTOCOL_HEADER_VAR_CONTENT_CRC32);
937 string.UnFormat(__L("%08X"), &content_CRC32);
938
939 return true;
940}
