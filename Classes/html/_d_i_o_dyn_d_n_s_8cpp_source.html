209{
210 DIOIP publicIP[2];
211 bool status = false;
212 DIOURL* _url = NULL;
213
214 ischanged = false;
215
216 _url = new DIOURL();
217 if(!_url)
218 {
219 return false;
220 }
221
222 (*_url) = url;
223
224 if(_url->ResolveURL(publicIP[0]))
225 {
226 if(GetPublicIP(publicIP[1], timeout, IPlocal))

228 ischanged = !publicIP[0].Compare(publicIP[1]);

233 }
234
235 status = true;
236 }
237 }
238
239 delete _url;
240

259DIODYNDNS_STATUSCODE DIODYNDNS::Update(DIOURL& url, DIOIP& newpublicip, int timeout, XSTRING* IPlocal)
260{
261 if(!webclient) return DIODYNDNS_STATUSCODE_UNKNOWN;
262
263 DIOURL askurl;
264 XSTRING IPstring;
265 XBUFFER xbuffer;
266 XSTRING result;
267 bool status;
268
269 newpublicip.GetXString(IPstring);

271 askurl.Format(__L("%s:%s@members.dyndns.org/nic/update?hostname=%s&myip=%s&wildcard=NOCHG&mx=NOCHG&backmx=NOCHG"), login.Get(), password.Get(), url.Get(), IPstring.Get());

293DIODYNDNS_STATUSCODE DIODYNDNS::GetError(XSTRING& result)
294{
295 // Other Unknown result from dyndns service.
296 // "good" The update was successful, and the hostname is now updated.
297 // "nochg" The update changed no settings, and is considered abusive. (Additional nochg updates will cause the hostname to become blocked).
298 // "badsys" The system parameter given is not valid. Valid system parameters are dyndns, statdns and custom.
299 // "badagent" The user agent that was sent has been blocked for not following these specifications or no user agent was specified.
300 // "badauth" The username or password specified are incorrect.
301 // "!donator" An option available only to credited users (suchas offline URL) was specified, but the user is not a credited user.
302 // "notfqdn" The hostname specified is not a fully-qualified domain name (not in the form hostname.dyndns.org or domain.com).
303 // "nohost" The hostname specified does not exist (or is not in the service specified in the system parameter).
304 // "!yours" The hostname specified exists, but not under the username specified.
305 // "abuse" The hostname specified is blocked for update abuse.
306 // "numhost" Too many or too few hosts found.
307 // "dnserr" DNS error encountered.
308 // "911" There is a serious problem on our side, such as a database or DNS server failure. The client should stop updating until notified via the status page that the service is back up.
309
310
311 DIODYNDNS_STATUSCODE code = DIODYNDNS_STATUSCODE_UNKNOWN;
312 XCHAR* stringcodes[] = { __L("good") ,
313 __L("nochg") ,
314 __L("badsys") ,
315 __L("badagent") ,
316 __L("badauth") ,
317 __L("!donator") ,
318 __L("notfqdn") ,
319 __L("nohost") ,
320 __L("!yours") ,
321 __L("abuse") ,
322 __L("numhost") ,
323 __L("dnserr") ,
324 __L("911")
325 };
326
327
328 for(int c=0; c<(sizeof(stringcodes)/sizeof(XCHAR*)); c++)
329 {
330 if(result.Find(stringcodes[c],true) != XSTRING_NOTFOUND)
331 {
332 code = (DIODYNDNS_STATUSCODE)(c);
333 break;
334 }
335 }
336
337 return code;
338}
