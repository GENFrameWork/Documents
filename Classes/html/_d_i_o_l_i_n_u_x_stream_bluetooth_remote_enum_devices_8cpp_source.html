297int DIOLINUXSTREAMBLUETOOTHREMOTEENUMDEVICES::IsReadyConnect(int socket)
298{
299 struct timeval tv;
300 int rc;
301 fd_set fdr;
302 fd_set fdw;
303 fd_set fds;
304
305 if(socket==-1) return -1;
306
307 FD_ZERO(&fdr);
308 FD_ZERO(&fdw);
309 FD_ZERO(&fds);
310
311 FD_SET((unsigned int)socket,&fdr);
312 FD_SET((unsigned int)socket,&fdw);
313 FD_SET((unsigned int)socket,&fds);
314
315 tv.tv_sec = 0;
316 tv.tv_usec = 100;
317
318 rc = select(socket+1, &fdr, &fdw, &fds, &tv);
319 if(rc==-1) return -1;
320 if(rc== 2) return -1;
321
322 int status1 = FD_ISSET(socket,&fdr) ? 1 : 0;
323 int status2 = FD_ISSET(socket,&fdw) ? 1 : 0;
324 int status3 = FD_ISSET(socket,&fds) ? 1 : 0;
325
326 if(status3) return -1;
327 if(((status1) || (status2)) && (rc==1)) return 1;
328
329 return 0;
330}

529 // ----------------------------------------------------------------------

541 }

543 sdp_list_free(attrid, 0);
544 sdp_list_free(search, 0);
