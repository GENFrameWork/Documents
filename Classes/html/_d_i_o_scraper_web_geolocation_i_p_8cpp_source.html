497bool DIOSCRAPERWEBGEOLOCATIONIP::Get(DIOIP& IP, DIOGEOLOCATIONIP_RESULT& geolocationIPresult, int timeoutforurl, XSTRING* localIP, bool usecache)
498{
499 bool status = false;
500
501 if(xmutexdo) xmutexdo->Lock();
502
503 this->IP.Set(IP.Get());
504
505 if(usecache)
506 {
507 XSTRING IPstring;
508
509 IP.GetXString(IPstring);
510
511 DIOGEOLOCATIONIP_RESULT* geoIPresult = (DIOGEOLOCATIONIP_RESULT*)cache->Get(IPstring);
512 if(geoIPresult)
513 {
514 geolocationIPresult.CopyFrom(geoIPresult);
515
516 if(xmutexdo) xmutexdo->UnLock();
517 return true;
518 }
519 }
520
521 if(Load(DIOSCRAPERWEBGEOLOCATIONIP_NAMEFILE))
522 {
523 if(Do(DIOSCRAPERWEBGEOLOCATIONIP_NAMESERVICE, timeoutforurl, localIP))
524 {
525 XSTRING country;
526 XSTRING state;
527 XSTRING city;
528 XSTRING ISP;
529 XSTRING organization;
530 double latitude = 0;
531 double longitude = 0;
532 XSTRING string;
533
534 country = GetValue(__L("COUNTRY"));
535 state = GetValue(__L("STATE"));
536 city = GetValue(__L("CITY"));
537 ISP = GetValue(__L("ISP"));
538 organization = GetValue(__L("ORGANIZATION"));
539
540 string = GetValue(__L("LATITUDE"));
541 if(!string.IsEmpty()) string.UnFormat(__L("%f"), &latitude);
542
543 string = GetValue(__L("LONGITUDE"));
544 if(!string.IsEmpty()) string.UnFormat(__L("%f"), &longitude);
545
546 geolocationIPresult.Set(country, state, city, ISP, organization);
547 geolocationIPresult.Set((float)latitude, (float)longitude);
548
549 if(!geolocationIPresult.IsEmpty())
550 {
551 if(usecache)
552 {
553 DIOGEOLOCATIONIP_RESULT* geoIPresult = new DIOGEOLOCATIONIP_RESULT();
554 if(geoIPresult)
555 {
556 XSTRING IPstring;
557
558 IP.GetXString(IPstring);
559
560 geoIPresult->CopyFrom(&geolocationIPresult);
561 cache->Add(IPstring, geoIPresult);
562 }
563 }
564
565 status = true;
566 }
567 }
568 }
569
570 if(xmutexdo) xmutexdo->UnLock();
571
572 return status;
573}
