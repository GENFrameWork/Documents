139{
140 if(!path) return false;

142
143 if(Exist(path)) return true;
144
145 XPATH xpath;
146
147 xpath = path;

149 xpath.Slash_Normalize(true);
150
151 if(recursive)
152 {
153 XPATH xpathsequence;
154 XSTRING pathpart;

156
157 xpathsequence.Empty();

162 xpathsequence.Slash_Add();

178 if(fresult != FR_OK) return false;
179 }
180
181 } else break;
182
183 index++;
184
185 } while(1);
186 }

188 {
189 XBUFFER xbufferexchange;
190 xpath.ConvertToBufferExchange(xbufferexchange);

342{

456bool XSTM32DIR::ReconvertSearchData(XDIRELEMENT* searchelement, FILINFO* fileinfo)
457{
458 if(!searchelement) return false;
459 if(!fileinfo) return false;
460
461 searchelement->GetNameFile()->Set(fileinfo->fname);
462
463 XDATETIME* datetime = searchelement->GetDateTimeFile_Modificated();
464 XWORD date = fileinfo->fdate;
465 XWORD time = fileinfo->ftime;
466
467
468 // bit31:25 Year origin from the 1980 (0..127, e.g. 37 for 2017)
469 // bit24:21 Month (1..12)
470 // bit20:16 Day of the month (1..31)
471
472 // bit15:11 Hour (0..23)
473 // bit10:5 Minute (0..59)
474 // bit4:0 Second / 2 (0..29, e.g. 25 for 50)
475
476 datetime->SetYear(((date >> 9) & 0x7F) + 1980);
477 datetime->SetMonth((date >> 5) & 0x0F);
478 datetime->SetDay((date & 0x1F));
479
480 datetime->SetHours((time >> 11) & 0x1F);
481 datetime->SetMinutes((time >> 5) & 0x3F);
482 datetime->SetSeconds((time & 0x1F) * 2);
483
484 if(fileinfo->fattrib & AM_DIR)
485 searchelement->SetType(XDIRELEMENTTYPE_DIR);
486 else searchelement->SetType(XDIRELEMENTTYPE_FILE);
487
488 return true;
489}
