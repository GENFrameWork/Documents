363bool CIPHER::Cipher(XBYTE* input,XDWORD size)
364{
365 result->Delete();
366 result->Add(input, size);
367
368 CIPHERKEYSYMMETRICAL* key = (CIPHERKEYSYMMETRICAL*)GetKey(0);
369 if(!key) return false;
370 if(!key->Get()->Get()) return false;
371
372 int indexkey = 0;
373
374 for(XDWORD c=0; c<size; c++)
375 {
376 result->Get()[c] ^= key->Get()->GetByte(indexkey);
377
378 indexkey++;
379 if(indexkey >= (int)key->Get()->GetSize()) indexkey = 0;
380 }
381
382 return true;
383}
