645int DIOATCMDS::CalculeBaudRateStreamUART(int timeout)
646{

648 int baudrate = 0;

651 if(!diostreamcfg) return 0;

653 if(diostreamcfg->GetType() != DIOSTREAMTYPE_UART) return 0;

667 {
668 if(threadconnection) threadconnection->Ini();
669
670 echomode = false;
671 if(Basic(timeout)==DIOATCMD_ERROR_NONE) baudrate = baudrates[c];
672
673 echomode = true;
674 if(Basic(timeout)==DIOATCMD_ERROR_NONE) baudrate = baudrates[c];
675
676 if(threadconnection) threadconnection->End();
677
678 if(baudrate) break;
679
680 }
681 }
682
683 diostream->Close();
684

1676{

1678 if(ATcmds.IsEmpty()) return false;

1681 {

1702
1703 case DIOATCMD_TYPEADDITIONALSIZE_BYTES : additionalparam.Resize(sizedata);
1704 diostream->Read((XBYTE*)additionalparam.Get(),sizedata);
1705 break;

1725 }
