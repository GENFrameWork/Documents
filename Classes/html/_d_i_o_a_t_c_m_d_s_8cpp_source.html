652int DIOATCMDS::CalculeBaudRateStreamUART(int timeout)
653{

655 int baudrate = 0;
656
657 DIOSTREAMUARTCONFIG* diostreamcfg = (DIOSTREAMUARTCONFIG*)diostream->GetConfig();
658 if(!diostreamcfg) return 0;
659
660 if(diostreamcfg->GetType() != DIOSTREAMTYPE_UART) return 0;
661
662 for(int c=0;c<sizeof(baudrates)/sizeof(int);c++)

664 diostreamcfg->SetBaudRate(baudrates[c]);
665
666 //XTRACE_PRINT(__L("Open connection: %d bauds"),baudrates[c]);
667
668 diostream->ResetXBuffers();
669
670 diostream->Close();
671 if(!diostream->Open()) return 0;
672
673 if(diostream->WaitToConnected(timeout))
674 {
675 if(threadconnection) threadconnection->Ini();
676
677 echomode = false;
678 if(Basic(timeout)==DIOATCMD_ERROR_NONE) baudrate = baudrates[c];
679
680 echomode = true;
681 if(Basic(timeout)==DIOATCMD_ERROR_NONE) baudrate = baudrates[c];
682
683 if(threadconnection) threadconnection->End();
684
685 if(baudrate) break;
686
687 }
688 }
689
690 diostream->Close();
691
