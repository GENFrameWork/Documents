173bool XWINDOWSREGISTRYKEY::ReadValue(XCHAR* name, XVARIANT& data)
174{
175 if(handlekey == NULL) return false;
176
177 XBUFFER databuffer;
178 XDWORD size = _MAXBUFFER;
179 LSTATUS status;
180
181 databuffer.Resize(size);
182
183 status = RegQueryValueEx(handlekey, name, 0, (DWORD*)&type, (LPBYTE)databuffer.Get(), (DWORD*)&size);
184 if(status != ERROR_SUCCESS) return false;
185
186 SetType(type);
187
188 switch(type)
189 {
190 case REG_NONE : // No value type
191 return false;
192
193 case REG_SZ : // Unicode nul terminated string
194 data = (XCHAR*)databuffer.Get();
195 break;
196
197 case REG_EXPAND_SZ : // Unicode nul terminated string (with environment variable references)
198 break;
199
200 case REG_BINARY : // Free form binary
201 data = databuffer;
202 break;
203
204 case REG_DWORD :
205 //case REG_DWORD_LITTLE_ENDIAN : // 32-bit number (same as REG_DWORD)
206 { XDWORD _data = 0;
207
208 databuffer.Get(_data);
209 SWAPDWORD(_data);
210 data = _data;
211 }
212 break;
213
214 case REG_DWORD_BIG_ENDIAN : { XDWORD _data = 0;
215 databuffer.Get(_data);
216
217 data = _data;
218 }
219 break;
220
221 case REG_LINK : // Symbolic Link (unicode)
222 break;
223
224 case REG_MULTI_SZ : // Multiple Unicode strings
225 data = databuffer;
226 break;
227
228 case REG_RESOURCE_LIST : // Resource list in the resource map
229 break;
230
231 case REG_FULL_RESOURCE_DESCRIPTOR : // Resource list in the hardware description
232 break;
233
234 case REG_RESOURCE_REQUIREMENTS_LIST : break;
235
236 case REG_QWORD :
237 //case REG_QWORD_LITTLE_ENDIAN : // 64-bit number (same as REG_QWORD)
238 { XQWORD _data = 0;
239
240 databuffer.Get(_data);
241 SWAPQWORD(_data);
242 data = _data;
243 }
244 break;
245
246 default : return false;
247 }
248
249 return true;
250}

310{
311 if(handlekey == NULL) return false;

313 LSTATUS status = ERROR_SUCCESS;
314
315 switch(data.GetType())

317 case XVARIANT_TYPE_SHORT :
318 case XVARIANT_TYPE_WORD :
319 case XVARIANT_TYPE_INTEGER :
320 case XVARIANT_TYPE_DWORD : type = REG_DWORD;
321 status = RegSetValueEx(handlekey, name, 0, (DWORD)type, (LPBYTE)data.GetData(), (DWORD)data.GetSize());
322 break;
323
324 case XVARIANT_TYPE_DOUBLEINTEGER :
325 case XVARIANT_TYPE_QWORD : type = REG_QWORD;
326 status = RegSetValueEx(handlekey, name, 0, (DWORD)type, (LPBYTE)data.GetData(), (DWORD)data.GetSize());
327 break;
328 case XVARIANT_TYPE_NULL :
329 case XVARIANT_TYPE_BOOLEAN :
330 case XVARIANT_TYPE_FLOAT :
331 case XVARIANT_TYPE_DOUBLE :
332 case XVARIANT_TYPE_DATE :
333 case XVARIANT_TYPE_TIME :

336 status = RegSetValueEx(handlekey, name, 0, (DWORD)type, (LPBYTE)data.GetData(), (DWORD)data.GetSize());

557bool XWINDOWSREGISTRYMANAGER::CreateKey(HKEY handlekey, XCHAR* subkeystring, XWINDOWSREGISTRYKEY& registrykey)
558{
559 if(registrykey.GetHandle() != NULL) return false;
560
561 HKEY oper_handlekey;
562 DWORD disposition;
563 XPATH subkeystr1;
564 XSTRING subkeystr2;
565 XVECTOR<XSTRING*> partkey;
566 XSTRING subkeystr3;
567 XDWORD error;
568
569 subkeystr1 = subkeystring;
570 subkeystr1.Slash_Normalize(true);
571 subkeystr2 = subkeystr1;
572
573 subkeystr2.Split(__C('\\'), partkey);
574
575 for(XDWORD c=0; c<partkey.GetSize(); c++)
576 {
577 subkeystr3 += partkey.Get(c)->Get();
578
579 error = RegCreateKeyEx(handlekey, subkeystr3.Get(), 0, NULL, 0, KEY_ALL_ACCESS | KEY_WOW64_64KEY, NULL, &oper_handlekey, &disposition);
580 if(error != ERROR_SUCCESS)
581 {
582 RegCloseKey(oper_handlekey);
583 break;
584 }
585 else
586 {
587 if((partkey.GetSize()-1) != c)
588 {
589 RegCloseKey(oper_handlekey);
590 }
591 }
592
593 subkeystr3 +=__C('\\');
594 }
595
596 if(error == ERROR_SUCCESS)
597 {
598 registrykey.SetHandle(oper_handlekey);
599 }
600
601 partkey.DeleteContents();
602 partkey.DeleteAll();
603
604 return (error == ERROR_SUCCESS)?true:false;
605}

759bool XWINDOWSREGISTRYMANAGER::DeleteKey(HKEY handlekey, XCHAR* subkeystring)
760{
761 HKEY operhandlekey = NULL;
762 XSTRING namekey;
763 DWORD size;
764 LONG result;
765 FILETIME ftwrite;
766
767 result = RegDeleteKey(handlekey, subkeystring);
768 if(result == ERROR_SUCCESS) return false;
769
770 result = RegOpenKeyEx(handlekey, subkeystring, 0, KEY_READ, &operhandlekey);
771 if(result != ERROR_SUCCESS)
772 {
773 if(result == ERROR_FILE_NOT_FOUND)
774 return true;
775 else return false;
776 }
777
778 XPATH subkey;
779
780 subkey = subkeystring;
781 subkey.Slash_Add();
782
783 size = _MAXPATH;
784
785 namekey.AdjustSize(size);
786 result = RegEnumKeyEx(operhandlekey, 0, namekey.Get(), &size, NULL, NULL, NULL, &ftwrite);
787 namekey.AdjustSize();
788
789 if(result == ERROR_SUCCESS)
790 {
791 do{ subkey.Add(namekey);
792 if(!DeleteKey(handlekey, namekey)) break;
793
794 size = _MAXPATH;
795
796 namekey.AdjustSize(size);
797 result = RegEnumKeyEx(operhandlekey, 0, namekey.Get(), &size, NULL, NULL, NULL, &ftwrite);
798 namekey.AdjustSize();
799
800 } while(result == ERROR_SUCCESS);
801 }
802
803 RegCloseKey(operhandlekey);
804
805 // Try again to delete the key.
806 result = RegDeleteKey(handlekey, subkeystring);
807
808 if(result == ERROR_SUCCESS) return true;
809
810 return false;
811}
