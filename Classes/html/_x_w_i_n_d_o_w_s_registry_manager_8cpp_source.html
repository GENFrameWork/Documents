166bool XWINDOWSREGISTRYKEY::ReadValue(XCHAR* name, XVARIANT& data)
167{
168 if(handlekey == NULL) return false;
169
170 XBUFFER databuffer;
171 XDWORD size = _MAXBUFFER;
172 LSTATUS status;
173
174 databuffer.Resize(size);
175
176 status = RegQueryValueEx(handlekey, name, 0, (DWORD*)&type, (LPBYTE)databuffer.Get(), (DWORD*)&size);
177 if(status != ERROR_SUCCESS) return false;
178
179 SetType(type);
180
181 switch(type)
182 {
183 case REG_NONE : // No value type
184 return false;
185
186 case REG_SZ : // Unicode nul terminated string
187 data = (XCHAR*)databuffer.Get();
188 break;
189
190 case REG_EXPAND_SZ : // Unicode nul terminated string (with environment variable references)
191 break;
192
193 case REG_BINARY : // Free form binary
194 data = databuffer;
195 break;
196
197 case REG_DWORD :
198 //case REG_DWORD_LITTLE_ENDIAN : // 32-bit number (same as REG_DWORD)
199 { XDWORD _data = 0;
200
201 databuffer.Get(_data);
202 SWAPDWORD(_data);
203 data = _data;
204 }
205 break;
206
207 case REG_DWORD_BIG_ENDIAN : { XDWORD _data = 0;
208 databuffer.Get(_data);
209
210 data = _data;
211 }
212 break;
213
214 case REG_LINK : // Symbolic Link (unicode)
215 break;
216
217 case REG_MULTI_SZ : // Multiple Unicode strings
218 data = databuffer;
219 break;
220
221 case REG_RESOURCE_LIST : // Resource list in the resource map
222 break;
223
224 case REG_FULL_RESOURCE_DESCRIPTOR : // Resource list in the hardware description
225 break;
226
227 case REG_RESOURCE_REQUIREMENTS_LIST : break;
228
229 case REG_QWORD :
230 //case REG_QWORD_LITTLE_ENDIAN : // 64-bit number (same as REG_QWORD)
231 { XQWORD _data = 0;
232
233 databuffer.Get(_data);
234 SWAPQWORD(_data);
235 data = _data;
236 }
237 break;
238
239 default : return false;
240 }
241
242 return true;
243}

550bool XWINDOWSREGISTRYMANAGER::CreateKey(HKEY handlekey, XCHAR* subkeystring, XWINDOWSREGISTRYKEY& registrykey)
551{
552 if(registrykey.GetHandle() != NULL) return false;
553
554 HKEY oper_handlekey;
555 DWORD disposition;
556 XPATH subkeystr1;
557 XSTRING subkeystr2;
558 XVECTOR<XSTRING*> partkey;
559 XSTRING subkeystr3;
560 XDWORD error;
561
562 subkeystr1 = subkeystring;
563 subkeystr1.Slash_Normalize(true);
564 subkeystr2 = subkeystr1;
565
566 subkeystr2.Split(__C('\\'), partkey);
567
568 for(XDWORD c=0; c<partkey.GetSize(); c++)
569 {
570 subkeystr3 += partkey.Get(c)->Get();
571
572 error = RegCreateKeyEx(handlekey, subkeystr3.Get(), 0, NULL, 0, KEY_ALL_ACCESS | KEY_WOW64_64KEY, NULL, &oper_handlekey, &disposition);
573 if(error != ERROR_SUCCESS)
574 {
575 RegCloseKey(oper_handlekey);
576 break;
577 }
578 else
579 {
580 if((partkey.GetSize()-1) != c)
581 {
582 RegCloseKey(oper_handlekey);
583 }
584 }
585
586 subkeystr3 +=__C('\\');
587 }
588
589 if(error == ERROR_SUCCESS)
590 {
591 registrykey.SetHandle(oper_handlekey);
592 }
593
594 partkey.DeleteContents();
595 partkey.DeleteAll();
596
597 return (error == ERROR_SUCCESS)?true:false;
598}

752bool XWINDOWSREGISTRYMANAGER::DeleteKey(HKEY handlekey, XCHAR* subkeystring)
753{
754 HKEY operhandlekey = NULL;
755 XSTRING namekey;
756 DWORD size;
757 LONG result;
758 FILETIME ftwrite;
759
760 result = RegDeleteKey(handlekey, subkeystring);
761 if(result == ERROR_SUCCESS) return false;
762
763 result = RegOpenKeyEx(handlekey, subkeystring, 0, KEY_READ, &operhandlekey);
764 if(result != ERROR_SUCCESS)
765 {
766 if(result == ERROR_FILE_NOT_FOUND)
767 return true;
768 else return false;
769 }
770
771 XPATH subkey;
772
773 subkey = subkeystring;
774 subkey.Slash_Add();
775
776 size = _MAXPATH;
777
778 namekey.AdjustSize(size);
779 result = RegEnumKeyEx(operhandlekey, 0, namekey.Get(), &size, NULL, NULL, NULL, &ftwrite);
780 namekey.AdjustSize();
781
782 if(result == ERROR_SUCCESS)
783 {
784 do{ subkey.Add(namekey);
785 if(!DeleteKey(handlekey, namekey)) break;
786
787 size = _MAXPATH;
788
789 namekey.AdjustSize(size);
790 result = RegEnumKeyEx(operhandlekey, 0, namekey.Get(), &size, NULL, NULL, NULL, &ftwrite);
791 namekey.AdjustSize();
792
793 } while(result == ERROR_SUCCESS);
794 }
795
796 RegCloseKey(operhandlekey);
797
798 // Try again to delete the key.
799 result = RegDeleteKey(handlekey, subkeystring);
800
801 if(result == ERROR_SUCCESS) return true;
802
803 return false;
804}
