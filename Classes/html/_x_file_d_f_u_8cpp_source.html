373bool XFILEDFU::ReadAll()
374{
375 if(!xfilebase) return false;
376
377 bool status = false;
378
379 images.DeleteContents();
380 images.DeleteAll();
381
382 xfilebase->SetPosition(0);
383
384 //------------------------------------------
385 // Read Prefix
386
387 status = xfilebase->Read(prefix.signature, XFILEDFU_PREFIX_SIGNATUREPREXIF_MAX);
388 if(status) status = xfilebase->Read(&prefix.version, 1);
389 if(status) status = xfilebase->Read((XBYTE*)&prefix.imagesize, sizeof(XDWORD));
390 if(status) status = xfilebase->Read(&prefix.ntargets, sizeof(XBYTE));
391
392 if(status)
393 {
394 for(int c=0; c<prefix.ntargets; c++)
395 {
396 XFILEDFU_IMAGE* image = Read_Image();
397 if(image)images.Add(image);
398 }
399 }
400
401 //------------------------------------------
402 // Read Suffix
403
404 status =xfilebase->Read((XBYTE*)&suffix.BCDdevice, sizeof(XWORD));
405 if(status) status = xfilebase->Read((XBYTE*)&suffix.IDproduct, sizeof(XWORD));
406 if(status) status = xfilebase->Read((XBYTE*)&suffix.IDvendor, sizeof(XWORD));
407 if(status) status = xfilebase->Read((XBYTE*)&suffix.BCDDFU, sizeof(XWORD));
408 if(status) status = xfilebase->Read((XBYTE*)&suffix.signature, XFILEDFU_PREFIX_SIGNATURESUFFIX_MAX);
409 if(status) status = xfilebase->Read(&suffix.length, 1);
410 if(status) status = xfilebase->Read((XBYTE*)&suffix.CRC, sizeof(XDWORD));
411
412
413 //--------------------------------------------
414 // Check CRC used in DFU.
415
416 status = CheckCRC();
417
418 return status;
419}

494XFILEDFU_IMAGE* XFILEDFU::Read_Image()
495{
496 XFILEDFU_IMAGE* image = new XFILEDFU_IMAGE();
497 if(!image) return NULL;
498
499 bool status = false;
500
501 status = xfilebase->Read(image->signature, XFILEDFU_PREFIX_SIGNATUREIMAGE_MAX);
502 if(status) xfilebase->Read(&image->alternatesetting, 1);
503 if(status) xfilebase->Read((XBYTE*)&image->targetnameexist, sizeof(XDWORD));
504 if(status) xfilebase->Read((XBYTE*)&image->targetname, XFILEDFU_IMAGEPREFIX_NAME_MAX);
505 if(status) xfilebase->Read((XBYTE*)&image->targetsize, sizeof(XDWORD));
506 if(status) xfilebase->Read((XBYTE*)&image->nelements, sizeof(XDWORD));
507
508 for(XDWORD c=0; c<image->nelements; c++)
509 {
510 XFILEDFU_ELEMENT* element = Read_Element();
511 if(element) image->elements.Add(element);
512 }
513
514 if(!status)
515 {
516 delete image;
517 return NULL;
518 }
519
520 return image;
521}

533XFILEDFU_ELEMENT* XFILEDFU::Read_Element()
534{
535 XFILEDFU_ELEMENT* element = new XFILEDFU_ELEMENT();
536 if(!element) return NULL;
537
538 bool status = false;
539
540 status = xfilebase->Read((XBYTE*)&element->address, sizeof(XDWORD));
541 if(status) xfilebase->Read((XBYTE*)&element->size, sizeof(XDWORD));
542 if(status)
543 {
544 status = false;
545 if(element->size)
546 {
547 status = element->data.Resize(element->size);
548 if(status) status = xfilebase->Read(element->data.Get(), element->data.GetSize());
549 }
550 }
551
552 if(!status)
553 {
554 delete element;
555 return NULL;
556 }
557
558 return element;

571bool XFILEDFU::CheckCRC()
572{
573 XBUFFER buffer;
574 bool status;
575
576 xfilebase->SetPosition(0);
577 buffer.Resize((XDWORD)(xfilebase->GetSize() - sizeof(XDWORD)));
578
579 status = xfilebase->Read(buffer.Get(), buffer.GetSize());
580 if(status)
581 {
582 XDWORD CRC = 0xFFFFFFFF;
583 for (XDWORD i = 0; i < buffer.GetSize(); i++)
584 {
585 CRC = CRCtable[(CRC ^ buffer.Get()[i]) & 0xff] ^ (CRC >> 8);
586 }
587
588 if(CRC != suffix.CRC) status = false;
589 }
590
591 return status;
592}
