366bool XFILEDFU::ReadAll()
367{
368 if(!xfilebase) return false;
369
370 bool status = false;
371
372 images.DeleteContents();
373 images.DeleteAll();
374
375 xfilebase->SetPosition(0);
376
377 //------------------------------------------
378 // Read Prefix
379
380 status = xfilebase->Read(prefix.signature, XFILEDFU_PREFIX_SIGNATUREPREXIF_MAX);
381 if(status) status = xfilebase->Read(&prefix.version, 1);
382 if(status) status = xfilebase->Read((XBYTE*)&prefix.imagesize, sizeof(XDWORD));
383 if(status) status = xfilebase->Read(&prefix.ntargets, sizeof(XBYTE));
384
385 if(status)
386 {
387 for(int c=0; c<prefix.ntargets; c++)
388 {
389 XFILEDFU_IMAGE* image = Read_Image();
390 if(image)images.Add(image);
391 }
392 }
393
394 //------------------------------------------
395 // Read Suffix
396
397 status =xfilebase->Read((XBYTE*)&suffix.BCDdevice, sizeof(XWORD));
398 if(status) status = xfilebase->Read((XBYTE*)&suffix.IDproduct, sizeof(XWORD));
399 if(status) status = xfilebase->Read((XBYTE*)&suffix.IDvendor, sizeof(XWORD));
400 if(status) status = xfilebase->Read((XBYTE*)&suffix.BCDDFU, sizeof(XWORD));
401 if(status) status = xfilebase->Read((XBYTE*)&suffix.signature, XFILEDFU_PREFIX_SIGNATURESUFFIX_MAX);
402 if(status) status = xfilebase->Read(&suffix.length, 1);
403 if(status) status = xfilebase->Read((XBYTE*)&suffix.CRC, sizeof(XDWORD));
404
405
406 //--------------------------------------------
407 // Check CRC used in DFU.
408
409 status = CheckCRC();
410
411 return status;
412}

487XFILEDFU_IMAGE* XFILEDFU::Read_Image()
488{
489 XFILEDFU_IMAGE* image = new XFILEDFU_IMAGE();
490 if(!image) return NULL;
491
492 bool status = false;
493
494 status = xfilebase->Read(image->signature, XFILEDFU_PREFIX_SIGNATUREIMAGE_MAX);
495 if(status) xfilebase->Read(&image->alternatesetting, 1);
496 if(status) xfilebase->Read((XBYTE*)&image->targetnameexist, sizeof(XDWORD));
497 if(status) xfilebase->Read((XBYTE*)&image->targetname, XFILEDFU_IMAGEPREFIX_NAME_MAX);
498 if(status) xfilebase->Read((XBYTE*)&image->targetsize, sizeof(XDWORD));
499 if(status) xfilebase->Read((XBYTE*)&image->nelements, sizeof(XDWORD));
500
501 for(XDWORD c=0; c<image->nelements; c++)
502 {
503 XFILEDFU_ELEMENT* element = Read_Element();
504 if(element) image->elements.Add(element);
505 }
506
507 if(!status)
508 {
509 delete image;
510 return NULL;
511 }
512
513 return image;
514}

535 if(status)

564bool XFILEDFU::CheckCRC()
565{
566 XBUFFER buffer;
567 bool status;
568
569 xfilebase->SetPosition(0);
570 buffer.Resize((XDWORD)(xfilebase->GetSize() - sizeof(XDWORD)));
571
572 status = xfilebase->Read(buffer.Get(), buffer.GetSize());
573 if(status)
574 {
575 XDWORD CRC = 0xFFFFFFFF;
576 for (XDWORD i = 0; i < buffer.GetSize(); i++)
577 {
578 CRC = CRCtable[(CRC ^ buffer.Get()[i]) & 0xff] ^ (CRC >> 8);
579 }
580
581 if(CRC != suffix.CRC) status = false;
582 }
583
584 return status;
585}
