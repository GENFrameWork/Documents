180bool SQLITE_DATABASE::Rollback(XCHAR* savepoint)
181{
182 if(IsTransactionStarted())
183 {
184 if(savepoint == NULL)
185 {
186 int rc = sqlite3_exec(sqlite3database, "ROLLBACK", 0, 0, 0);
187 transactionstarted = false;
188
189 return (rc==SQLITE_OK);
190 }
191 else
192 {
193 DB_SQL_STRING text;
194 text.Format(__L("ROLLBACK TO SAVEPOINT %s"), savepoint);
195
196 XBUFFER rollbacktext;
197
198 text.ConvertToASCII(rollbacktext);
199 int rc = sqlite3_exec(sqlite3database, rollbacktext.GetPtrChar(), 0, 0, 0);
200 transactionstarted = false;
201
202 return (rc==SQLITE_OK);
203 }
204 }
205
206 return false;
207}

307bool SQLITE_DATABASE::GetTables()
308{
309 DB_SQL_QUERY* query = CreateQuery();
310 if(!query) return false;
311
312 bool status = true;
313
314 tables.DeleteContents();
315 tables.DeleteAll();
316
317 query->Set(__L("SELECT name FROM sqlite_master WHERE type='table'"));
318 if(query->Execute())
319 {
320 DB_SQL_RESULT* result=query->GetResult();
321 if(result!=NULL) //hay un result
322 {
323 result->GetNumRows();
324 while(result->HasNext())
325 {
326 result->ProcessRow();
327
328 DB_SQL_ROW* row = result->GetRow();
329 for(XDWORD e=0; e<result->GetNumColumns(); e++)
330 {
331 if(row->Get(e).GetType()==(XVARIANT_TYPE)DB_SQL_VARIANT_TYPE_STRING)
332 {
333 DB_SQL_STRING* tablename = new DB_SQL_STRING();
334 if(tablename)
335 {
336 tablename->Set((XCHAR*)row->Get(0));
337 tables.Add(tablename);
338 }
339 }
340 }
341
342 result->Next();
343 }
344 }
345 }
346 else
347 {
348 Error(__L("Unable to obtain Tables"));
349 status = false;
350 }
351
352 delete(query);
353
354 return status;
355}
