342bool MAINPROCANDROID::OnTouchEvent(AInputEvent* event)
343{
344 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] INI OnTouchEvent"));
345
346 int32_t action = AMotionEvent_getAction(event);
347 size_t pointercount = AMotionEvent_getPointerCount(event);
348
349 if(!mouse) mouse = (INPANDROIDDEVICEMOUSE*)GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_MOUSE);
350
351 if(mouse)
352 {
353 for(size_t c = 0; c<pointercount; ++c)
354 {
355 int32_t pointerID = c;
356 int32_t pointerIndex = AMotionEvent_getPointerId(event,c);
357
358 if(pointerIndex >= mouse->GetCursors()->GetSize())
359 {
360 INPCURSOR* cursor = new INPCURSOR();
361 if(cursor)
362 {
363 cursor->SetID(INPCURSOR_ID(INPCURSOR_ID_TOUCHSCREEN1+pointerIndex));
364 cursor->SetHavePreSelect(false);
365
366 mouse->GetCursors()->Add(cursor);
367 }
368 }
369
370 if(pointerIndex >= mouse->GetButtons()->GetSize())
371 {
372 INPBUTTON* button = new INPBUTTON();
373 if(button)
374 {
375 button->SetID(INPBUTTON_ID_TOUCHSCREEN);
376 mouse->GetButtons()->Add(button);
377 }
378 }
379
380 float x = AMotionEvent_getX(event, pointerID);
381 float y = AMotionEvent_getY(event, pointerID);
382 float pressure = AMotionEvent_getPressure(event, pointerID);
383
384 switch(action & AMOTION_EVENT_ACTION_MASK)
385 {
386 case AMOTION_EVENT_ACTION_UP :
387 case AMOTION_EVENT_ACTION_POINTER_UP : { mouse->AddPosition(pointerIndex, x, y, false);
388 if(mouse->GetButton(pointerIndex))
389 {
390 mouse->GetButton(pointerIndex)->SetPressed(false);
391 mouse->GetButton(pointerIndex)->SetPressure(pressure);
392
393 if(mouse->GetButton(pointerIndex)->GetState() != INPBUTTON_STATE_UP)
394 {
395 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_RELEASED);
396 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d RELEASED"), pointerID,x,y,pressure,action);
397
398 }
399 else
400 {
401 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_UP);
402 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d UP"), pointerID,x,y,pressure,action);
403 }
404 }
405 }
406 break;
407 case AMOTION_EVENT_ACTION_MOVE :
408 case AMOTION_EVENT_ACTION_DOWN :
409 case AMOTION_EVENT_ACTION_POINTER_DOWN : { mouse->AddPosition(pointerIndex,x,y, true);
410 if(mouse->GetButton(pointerIndex))
411 {
412 mouse->GetButton(pointerIndex)->SetPressed(true);
413 mouse->GetButton(pointerIndex)->SetPressure(pressure);
414
415 if(mouse->GetButton(pointerIndex)->GetState() == INPBUTTON_STATE_UP)
416 {
417 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_PRESSED);
418 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d PRESSED"), pointerID,x,y,pressure,action);
419 }
420 else
421 {
422 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_HOLD);
423 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d HOLD"), pointerID,x,y,pressure,action);
424 }
425 }
426 }
427 break;
428
429 case AMOTION_EVENT_ACTION_CANCEL : if (mouse->GetButton(pointerIndex))
430 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_UP);
431 break;
432 }
433 }
434 }
435
436
437 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] END OnTouchEvent"));
438
439 return true;
440}

454bool MAINPROCANDROID::OnKeyboardEvent(AInputEvent* event)

456 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] INI OnKeyboardEvent Android code [%04x]"), AKeyEvent_getKeyCode(event));
457
458 INPDEVICE* keyboard = androidmain.GetKeyboard();
459 if(!keyboard) return false;

462 if(!button) return false;

465 switch(keyEvent)

468 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Keyboard %d ID %d RELEASED"), keyEvent, button->GetID());
469 break;
470
471 case AKEY_EVENT_ACTION_DOWN : button->SetPressed(true);
472 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Keyboard %d ID %d PRESSED"), keyEvent, button->GetID());
473 break;
474 }
475
476 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] END OnKeyboardEvent"));
477
478 return false;

932bool MAINPROCANDROID::GetPackageResourcePath(struct android_app* app, XSTRING& path)
933{
934 JNIEnv* env;
935 JavaVM* vm = app->activity->vm;
936 bool status = false;
937
938 vm->AttachCurrentThread(&env, NULL);
939
940 jclass clazz = env->GetObjectClass(app->activity->clazz);
941 jmethodID methodj = env->GetMethodID(clazz, "getPackageResourcePath", "()Ljava/lang/String;");
942
943 jstring jstr = (jstring)env->CallObjectMethod(app->activity->clazz, methodj);
944 if(jstr != NULL)
945 {
946 const char* str = env->GetStringUTFChars(jstr, NULL);
947 path = str;
948 env->ReleaseStringUTFChars(jstr, str);
949
950 status = true;
951 }
952
953 vm->DetachCurrentThread();
954
955 return status;
956}

990 env->ReleaseStringUTFChars(jpath, str);
