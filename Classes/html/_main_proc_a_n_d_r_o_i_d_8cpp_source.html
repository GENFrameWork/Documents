349bool MAINPROCANDROID::OnTouchEvent(AInputEvent* event)
350{
351 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] INI OnTouchEvent"));
352
353 int32_t action = AMotionEvent_getAction(event);
354 size_t pointercount = AMotionEvent_getPointerCount(event);
355
356 if(!mouse) mouse = (INPANDROIDDEVICEMOUSE*)GEN_INPMANAGER.GetDevice(INPDEVICE_TYPE_MOUSE);
357
358 if(mouse)
359 {
360 for(size_t c = 0; c<pointercount; ++c)
361 {
362 int32_t pointerID = c;
363 int32_t pointerIndex = AMotionEvent_getPointerId(event,c);
364
365 if(pointerIndex >= mouse->GetCursors()->GetSize())
366 {
367 INPCURSOR* cursor = new INPCURSOR();
368 if(cursor)
369 {
370 cursor->SetID(INPCURSOR_ID(INPCURSOR_ID_TOUCHSCREEN1+pointerIndex));
371 cursor->SetHavePreSelect(false);
372
373 mouse->GetCursors()->Add(cursor);
374 }
375 }
376
377 if(pointerIndex >= mouse->GetButtons()->GetSize())
378 {
379 INPBUTTON* button = new INPBUTTON();
380 if(button)
381 {
382 button->SetID(INPBUTTON_ID_TOUCHSCREEN);
383 mouse->GetButtons()->Add(button);
384 }
385 }
386
387 float x = AMotionEvent_getX(event, pointerID);
388 float y = AMotionEvent_getY(event, pointerID);
389 float pressure = AMotionEvent_getPressure(event, pointerID);
390
391 switch(action & AMOTION_EVENT_ACTION_MASK)
392 {
393 case AMOTION_EVENT_ACTION_UP :
394 case AMOTION_EVENT_ACTION_POINTER_UP : { mouse->AddPosition(pointerIndex, x, y, false);
395 if(mouse->GetButton(pointerIndex))
396 {
397 mouse->GetButton(pointerIndex)->SetPressed(false);
398 mouse->GetButton(pointerIndex)->SetPressure(pressure);
399
400 if(mouse->GetButton(pointerIndex)->GetState() != INPBUTTON_STATE_UP)
401 {
402 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_RELEASED);
403 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d RELEASED"), pointerID,x,y,pressure,action);
404
405 }
406 else
407 {
408 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_UP);
409 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d UP"), pointerID,x,y,pressure,action);
410 }
411 }
412 }
413 break;
414 case AMOTION_EVENT_ACTION_MOVE :
415 case AMOTION_EVENT_ACTION_DOWN :
416 case AMOTION_EVENT_ACTION_POINTER_DOWN : { mouse->AddPosition(pointerIndex,x,y, true);
417 if(mouse->GetButton(pointerIndex))
418 {
419 mouse->GetButton(pointerIndex)->SetPressed(true);
420 mouse->GetButton(pointerIndex)->SetPressure(pressure);
421
422 if(mouse->GetButton(pointerIndex)->GetState() == INPBUTTON_STATE_UP)
423 {
424 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_PRESSED);
425 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d PRESSED"), pointerID,x,y,pressure,action);
426 }
427 else
428 {
429 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_HOLD);
430 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Motion %d x %f y %f p %f a %d HOLD"), pointerID,x,y,pressure,action);
431 }
432 }
433 }
434 break;
435
436 case AMOTION_EVENT_ACTION_CANCEL : if (mouse->GetButton(pointerIndex))
437 mouse->GetButton(pointerIndex)->SetState(INPBUTTON_STATE_UP);
438 break;
439 }
440 }
441 }
442
443
444 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] END OnTouchEvent"));
445
446 return true;
447}

461bool MAINPROCANDROID::OnKeyboardEvent(AInputEvent* event)
462{
463 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] INI OnKeyboardEvent Android code [%04x]"), AKeyEvent_getKeyCode(event));
464
465 INPDEVICE* keyboard = androidmain.GetKeyboard();
466 if(!keyboard) return false;
467
468 INPBUTTON* button = keyboard->GetButtonByCode(AKeyEvent_getKeyCode(event));
469 if(!button) return false;
470
471 XDWORD keyEvent = AKeyEvent_getAction(event);
472 switch(keyEvent)
473 {
474 case AKEY_EVENT_ACTION_UP : button->SetPressed(false);
475 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Keyboard %d ID %d RELEASED"), keyEvent, button->GetID());
476 break;
477
478 case AKEY_EVENT_ACTION_DOWN : button->SetPressed(true);
479 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("> ANDROID Keyboard %d ID %d PRESSED"), keyEvent, button->GetID());
480 break;
481 }
482
483 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE , __L("[ANDROID Event] END OnKeyboardEvent"));
484
485 return false;
486}

939bool MAINPROCANDROID::GetPackageResourcePath(struct android_app* app, XSTRING& path)
940{
941 JNIEnv* env;
942 JavaVM* vm = app->activity->vm;
943 bool status = false;
944
945 vm->AttachCurrentThread(&env, NULL);
946
947 jclass clazz = env->GetObjectClass(app->activity->clazz);
948 jmethodID methodj = env->GetMethodID(clazz, "getPackageResourcePath", "()Ljava/lang/String;");
949
950 jstring jstr = (jstring)env->CallObjectMethod(app->activity->clazz, methodj);
951 if(jstr != NULL)
952 {
953 const char* str = env->GetStringUTFChars(jstr, NULL);
954 path = str;
955 env->ReleaseStringUTFChars(jstr, str);
956
957 status = true;
958 }
959
960 vm->DetachCurrentThread();
961
962 return status;
963}

991

993 if(jpath != NULL)
994 {
995 const char* str = env->GetStringUTFChars(jpath, NULL);
996 path = str;
997 env->ReleaseStringUTFChars(jpath, str);
998 status = true;
999 }

1001 vm->DetachCurrentThread();
1002
1003 return status;
