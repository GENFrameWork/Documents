116bool INPWINDOWSCAPTURE::Activate()
117{
118 if(keyhook) return false;
119
120 RegisterEvent(INPCAPTURE_XEVENT_TYPE_PRESSKEY);
121 RegisterEvent(INPCAPTURE_XEVENT_TYPE_UNPRESSKEY);
122
123 // Set up the hook
124
125 HINSTANCE hinst = (HINSTANCE)applicationhandle;
126 if(!hinst) return false;
127
128 keyhook = SetWindowsHookEx(WH_KEYBOARD_LL, LowLevelKeyboardProc, hinst, 0);
129 if(keyhook == NULL)
130 {
131 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Windows Hook Keyboard] Error: Could not hook keyboard. Another program might be interfering."));
132 return false;
133 }
134
135 return true;
136}

148bool INPWINDOWSCAPTURE::Deactivate()
149{
150 if(!keyhook) return false;
151
152 DeRegisterEvent(INPCAPTURE_XEVENT_TYPE_PRESSKEY);
153 DeRegisterEvent(INPCAPTURE_XEVENT_TYPE_UNPRESSKEY);
154
155 //Remove keyboard hook
156 if(UnhookWindowsHookEx(keyhook) == 0)
157 {
158 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Windows Hook Keyboard] Error: Could not unhook keyboard."));
159 return false;
160 }
161
162 //Success
163 keyhook = NULL;
164
165 return true;
166}

182LRESULT CALLBACK INPWINDOWSCAPTURE::LowLevelKeyboardProc(int ncode, WPARAM wparam, LPARAM lparam)
183{
184 bool keydown = false;
185
186 if(ncode == HC_ACTION)
187 {
188 if(wparam == WM_KEYDOWN || wparam == WM_SYSKEYDOWN) keydown = true;
189 if(wparam == WM_KEYUP || wparam == WM_SYSKEYUP) keydown = false;
190 }
191
192 KBDLLHOOKSTRUCT* keydata = (KBDLLHOOKSTRUCT*)lparam;
193
194 //XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("[Windows Hook keyboard] vkCode %04X, scanCode %08X, flags %08X [%c]"), keydata->vkCode, keydata->scanCode, keydata->flags, keydata->vkCode);
195
196 if(capture)
197 {
198 INPCAPTURE_XEVENT xevent(capture, (keydown?INPCAPTURE_XEVENT_TYPE_PRESSKEY:INPCAPTURE_XEVENT_TYPE_UNPRESSKEY));
199
200 if(!keydown)
201 {
202 XCHAR character = (XCHAR)keydata->vkCode;
203
204 capture->SetNKeys(capture->GetNKeys());
205
206 if(capture->GetLimit())
207 {
208 if(capture->GetLimit() >= capture->GetNKeys())
209 {
210 capture->GetString()->Empty();
211 capture->GetBuffer()->Empty();
212 }
213 }
214
215 if(((character >= 0x20) && (character < 0x80)) || (character == 0x13) || (character == 0x10) || (character == 0x09))
216 {
217 capture->GetString()->Add(character);
218 }
219
220 capture->GetBuffer()->Add((XDWORD)keydata->vkCode);
221 }
222
223 xevent.SetVKCode((XDWORD)keydata->vkCode);
224 xevent.SetScanCode((XWORD)keydata->scanCode);
225 xevent.SetFlags((XDWORD)keydata->flags);
226
227 xevent.SetNKeys(capture->GetNKeys());
228 xevent.SetLimit(capture->GetLimit());
229 xevent.GetString()->Set(capture->GetString()->Get());
230 xevent.GetBuffer()->Add(capture->GetBuffer()->Get(), capture->GetBuffer()->GetSize());
231
232 capture->PostEvent(&xevent);
233 if(xevent.IsKeyLocked())
234 {
235 return 1;
236 }
237 else
238 {
239 return 0;
240 }
241 }
242
243 return CallNextHookEx(NULL, ncode, wparam, lparam);
244}
