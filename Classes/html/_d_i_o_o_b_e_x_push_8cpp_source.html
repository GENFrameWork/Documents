430bool DIOOBEXPUSH::Ini(int timeout)
431{
432 if(!diostream) return false;
433 if(!diostream->GetConfig()) return false;
434
435 sendbuffer = new XBUFFER();
436 GEN_XFACTORY_CREATE(xfile, Create_File())
437 GEN_XFACTORY_CREATE(xtimer, CreateTimer());
438
439 issending = false;
440
441 if(xfile && sendbuffer && xtimer)
442 {
443 if(diostream->Open())
444 {
445 if(diostream->WaitToConnected(timeout))
446 {
447 xthreadobex = CREATEXTHREAD(XTHREADGROUPID_DIOSTREAM, __L("DIOOBEXPUSH::Ini"),ThreadSend,(void*)this);
448 if(xthreadobex)
449 {
450 xthreadobex->Ini(false);
451 return true;
452 }
453 }
454
455 diostream->Close();
456 }
457 }
458
459 delete sendbuffer;
460
461 GEN_XFACTORY.Delete_File(xfile);
462 GEN_XFACTORY.DeleteTimer(xtimer);
463
464 sendbuffer = NULL;
465 xfile = NULL;
466 xtimer = NULL;
467
468 return false;
469}

851bool DIOOBEXPUSH::CheckSending()
852{
853 DIOOBEXPUSHFILE* obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(indexfilelist);
854 DIOSTREAMBLUETOOTH* sio = (DIOSTREAMBLUETOOTH*)diostream;
855 XWORD wsize = 0;
856 XDWORD dsize = 0;
857 bool islastblock = false;
858
859 if((!sio)||(!obexpushfile))
860 {
861 if(obexpushfile) obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
862 return false;
863 }
864
865 if(sio->GetStatus()==DIOSTREAMSTATUS_GETTINGCONNECTION)
866 {
867 if(xtimer->GetMeasureSeconds()>=(XDWORD)asktimeout)
868 {
869 asktimeelapsed = xtimer->GetMeasureMilliSeconds();
870 for(int c=0;c<(int)sendfilelist.GetSize();c++)
871 {
872 obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(c);
873 if(obexpushfile) obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
874 }
875
876 return false;
877 }
878
879 return true;
880 }
881
882 if(sio->GetStatus()==DIOSTREAMSTATUS_DISCONNECTED)
883 {
884 haveerror=true;
885 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
886 }
887
888 if(GetEvent()==DIOOBEXPUSH_XFSMEVENT_NONE) // No hay nuevos Eventos
889 {
890 switch(GetCurrentState())
891 {
892 case DIOOBEXPUSH_XFSMSTATE_NONE : break;
893
894 case DIOOBEXPUSH_XFSMSTATE_CONNECT : { XBUFFER answer;
895
896 if(!ReadAnswer(answer)) break;
897
898 //XTRACE_PRINTDATABLOCK(4|XDEBUGTRACELEVEL_WITHCOLOR, answer);
899
900 if(answer.Get()[0]==0xA0)
901 {
902 sendblocksize = (XWORD)answer.Get()[5];
903 sendblocksize <<=8;
904 sendblocksize |= (XWORD)answer.Get()[6];
905
906 SetEvent(DIOOBEXPUSH_XFSMEVENT_INIPUSH);
907 }
908 else
909 {
910 haveerror = true;
911 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
912 }
913 }
914 break;
915
916 case DIOOBEXPUSH_XFSMSTATE_INIPUSH : SetEvent(DIOOBEXPUSH_XFSMEVENT_SENDHEADPUSH);
917 break;
918
919 case DIOOBEXPUSH_XFSMSTATE_SENDHEADPUSH : SetEvent(DIOOBEXPUSH_XFSMEVENT_RECVHEADPUSH);
920 break;
921
922 case DIOOBEXPUSH_XFSMSTATE_SENDBLOCKPUSH : sendbuffer->Delete();
923
924 islastblock = ((filesize - readfilesize) >= sendblocksize)? false:true;
925
926 sendbuffer->Add((!islastblock)?(XBYTE)0x02:(XBYTE)0x82);
927
928 wsize = 0;
929 sendbuffer->Add((XWORD)wsize);
930
931 if(islastblock)
932 readfilesizeblock = (filesize - readfilesize);
933 else readfilesizeblock = sendblocksize -16;
934
935 sendbuffer->Add((!islastblock)?(XBYTE)0x48:(XBYTE)0x49);
936
937 wsize+=3;
938
939 wsize += (int)readfilesizeblock;
940 sendbuffer->Add((XWORD)wsize);
941
942 { XBUFFER datafile((XDWORD)readfilesizeblock, true);
943
944 xfile->Read(datafile.Get(),(XDWORD*)&readfilesizeblock);
945 if(!readfilesizeblock)
946 {
947 haveerror = true;
948 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
949 break;
950 }
951
952 sendbuffer->Add(datafile);
953 }
954
955 if(inlittleendian)
956 wsize = SwapWORD((XWORD)sendbuffer->GetSize());
957 else wsize = (XWORD)sendbuffer->GetSize();
958 sendbuffer->Set((XBYTE*)&wsize,2,1);
959
960 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
961
962 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
963 sio->WaitToFlushOutXBuffer(5);
964
965 if(filesize) obexpushfile->SetSendPercent((XBYTE)((readfilesize*100)/filesize));
966
967 SetEvent(DIOOBEXPUSH_XFSMEVENT_RECVBLOCKPUSH);
968 break;
969
970 case DIOOBEXPUSH_XFSMSTATE_RECVHEADPUSH :
971 case DIOOBEXPUSH_XFSMSTATE_RECVBLOCKPUSH : { XBUFFER answer;
972
973 if(!ReadAnswer(answer)) break;
974
975 //XTRACE_PRINTDATABLOCK(4|XDEBUGTRACELEVEL_WITHCOLOR, answer);
976
977 switch(answer.Get()[0])
978 {
979 case 0x90 : readfilesize+=readfilesizeblock;
980 SetEvent(DIOOBEXPUSH_XFSMEVENT_SENDBLOCKPUSH);
981 break;
982
983 case 0xA0 : haveerror = false;
984 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
985 break;
986
987 default : if(GetEvent()==DIOOBEXPUSH_XFSMEVENT_NONE)
988 {
989 haveerror = true;
990 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
991 }
992 break;
993 }
994
995 }
996 break;
997
998 case DIOOBEXPUSH_XFSMSTATE_ENDPUSH : return false;
999 }
1000 }
1001 else // Nuevo evento
1002 {
1003 if(GetEvent()<DIOOBEXPUSH_LASTEVENT)
1004 {
1005 CheckTransition();
1006
1007 switch(GetCurrentState())
1008 {
1009 case DIOOBEXPUSH_XFSMSTATE_NONE : break;
1010
1011 case DIOOBEXPUSH_XFSMSTATE_CONNECT : sendisstart = false;
1012
1013 asktimeelapsed = xtimer->GetMeasureMilliSeconds();
1014
1015 sendbuffer->Delete();
1016
1017 sendbuffer->Add((XBYTE)0x80);
1018
1019 wsize = 0x0007;
1020 sendbuffer->Add((XWORD)wsize);
1021
1022 sendbuffer->Add((XBYTE)0x10);
1023 sendbuffer->Add((XBYTE)0x00);
1024
1025 wsize = DIOOBEXPUSH_MAXBUFFER;
1026 sendbuffer->Add((XWORD)wsize);
1027
1028 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
1029
1030 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
1031 sio->WaitToFlushOutXBuffer(5);
1032 break;
1033
1034 case DIOOBEXPUSH_XFSMSTATE_INIPUSH : { xtimer->Reset();
1035
1036 obexpushfile->SetIsSend(false);
1037 obexpushfile->SetSendPercent(0);
1038
1039 xpath = (*obexpushfile->GetXPath());
1040
1041 if((!xfile)||(!xfile->Open(xpath,true))||(!sendbuffer))
1042 {
1043 haveerror = true;
1044 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
1045 break;
1046 }
1047
1048 readfilesizeblock = 0;
1049
1050 sendisstart = true;
1051 filesize = xfile->GetSize();
1052 readfilesize = 0;
1053 }
1054 break;
1055
1056 case DIOOBEXPUSH_XFSMSTATE_SENDHEADPUSH : sendbuffer->Delete();
1057
1058 sendbuffer->Add((XBYTE)0x02);
1059
1060 wsize = 0;
1061 sendbuffer->Add((XWORD)wsize);
1062
1063 sendbuffer->Add((XBYTE)0x01);
1064
1065 // Unicode 16 + 1 byte of head;
1066 wsize++;
1067 wsize += (XWORD)((obexpushfile->GetNameSend()->GetSize()+1)*2);
1068 sendbuffer->Add((XWORD)wsize);
1069
1070 XBUFFER xbufferexchange;
1071 (*obexpushfile->GetNameSend())->ConvertToBufferExchange(xbufferexchange, true, true);
1072 sendbuffer->Add(xbufferexchange->Get(), wsize);
1073
1074 sendbuffer->Add((XBYTE)0xC3);
1075 sendbuffer->Add((XDWORD)filesize);
1076
1077 if(inlittleendian)
1078 wsize = SwapWORD((XWORD)sendbuffer->GetSize());
1079 else wsize = (XWORD)sendbuffer->GetSize();
1080 sendbuffer->Set((XBYTE*)&wsize,2,1);
1081
1082 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
1083
1084 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
1085 sio->WaitToFlushOutXBuffer(5);
1086 break;
1087
1088 case DIOOBEXPUSH_XFSMSTATE_RECVHEADPUSH : break;
1089
1090 case DIOOBEXPUSH_XFSMSTATE_SENDBLOCKPUSH : break;
1091
1092 case DIOOBEXPUSH_XFSMSTATE_RECVBLOCKPUSH : break;
1093
1094 case DIOOBEXPUSH_XFSMSTATE_ENDPUSH : if(xfile) xfile->Close();
1095
1096 if(!haveerror)
1097 {
1098 obexpushfile->SetIsSend(true);
1099 obexpushfile->SetSendPercent(100);
1100 obexpushfile->SetSendTimeElapsed(xtimer->GetMeasureMilliSeconds());
1101 obexpushfile->SetError(DIOOBEXPUSHERROR_NONE);
1102
1103 if((indexfilelist+1<(int)sendfilelist.GetSize()))
1104 {
1105 indexfilelist++;
1106 SetEvent(DIOOBEXPUSH_XFSMEVENT_INIPUSH);
1107 }
1108 }
1109 else
1110 {
1111 for(int c=0;c<(int)sendfilelist.GetSize();c++)
1112 {
1113 obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(c);
1114 if(obexpushfile)
1115 {
1116 obexpushfile->SetIsSend(false);
1117
1118 if(sendisstart)
1119 {
1120 obexpushfile->SetError(DIOOBEXPUSHERROR_SENDING);
1121 }
1122 else
1123 {
1124 if(sio->IsRefusedConnection())
1125 {
1126 obexpushfile->SetError(DIOOBEXPUSHERROR_REFUSEDCONNECTION);
1127 }
1128 else
1129 {
1130 obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
1131 }
1132 }
1133 }
1134 }
1135
1136 return false;
1137 }
1138 break;
1139 }
1140 }
1141 }
1142
1143 return true;
1144}
