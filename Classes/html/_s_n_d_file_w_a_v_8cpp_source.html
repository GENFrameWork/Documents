144bool SNDFILEWAV::Open(XCHAR* path)
145{
146 fileRIFF = new XFILERIFF();
147 if(!fileRIFF) return false;
148
149 if(!fileRIFF->Open(path)) return false;
150
151 if(!fileRIFF->ReadAllLists()) return false;
152
153 format_list = fileRIFF->GetChunk(__L("fmt "), __L("WAVE"));
154 if(!format_list) return false;
155
156 fileRIFF->GetFileBase()->SetPosition(format_list->GetPositionFileData());
157
158 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.audio_format , sizeof(XWORD))) return false;
159 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.num_channels , sizeof(XWORD))) return false;
160 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.sample_rate , sizeof(XDWORD))) return false;
161 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.byte_rate , sizeof(XDWORD))) return false;
162 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.block_align , sizeof(XWORD))) return false;
163 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.bits_per_sample , sizeof(XWORD))) return false;
164
165 channels = format.num_channels;
166 samplerate = format.sample_rate;
167
168 data_list = fileRIFF->GetChunk(__L("data"), __L("WAVE"));
169 if(!data_list) return false;
170
171 dataoffsetinfile = data_list->GetPositionFileData();
172 datasizeinfile = data_list->GetSize();
173
174 datasizeread = 0;
175
176 duration = ((XDWORD)(datasizeinfile/channels)/format.byte_rate)*1000;
177 nsamples = (XDWORD)((datasizeinfile*8)/format.bits_per_sample);
178
179 return fileRIFF->GetFileBase()->SetPosition(dataoffsetinfile);
180}

196bool SNDFILEWAV::ReadData(XBYTE* data, XDWORD& sizeread, bool& isfinished)
197{
198 if(!fileRIFF) return false;
199 if(!data_list) return false;
200 if(!dataoffsetinfile) return false;
201
202 XDWORD _sizeread = sizeread;
203
204 if(datasizeread == datasizeinfile)
205 {
206 datasizeread = 0;
207 }
208
209 isfinished = false;
210
211 bool status = fileRIFF->GetFileBase()->Read(data, &_sizeread);
212 if(!_sizeread && !status) return false;
213
214 datasizeread += _sizeread;
215
216 if((sizeread != _sizeread) || (datasizeread >= datasizeinfile))
217 {
218 dataoffsetinfile = data_list->GetPositionFileData();
219 fileRIFF->GetFileBase()->SetPosition(dataoffsetinfile);
220
221 isfinished = true;
222
223 sizeread = _sizeread;
224 }
225
226 return true;
227}
