351void HASHSHA2::Sha2_256_Transf(HASHSHA2_256_CTX* ctx, XBYTE* input, XDWORD blocknb)
352{
353 XDWORD w[64];
354 XDWORD wv[8];
355 XDWORD t1;
356 XDWORD t2;
357 XBYTE* subblock;
358 int i;
359 int j;
360
361 for(i=0; i<(int)blocknb; i++)
362 {
363 subblock = input + (i << 6);
364
365 for(j=0; j<16; j++)
366 {
367 PACK32(&subblock[j << 2], &w[j]);
368 }
369
370 for(j=16; j<64; j++)
371 {
372 SHA256_SCR(j);
373 }
374
375 for(j=0; j<8; j++)
376 {
377 wv[j] = ctx->h[j];
378 }
379
380 for(j=0; j<64; j++)
381 {
382 t1 = wv[7] + SHA256_F2(wv[4]) + CH(wv[4], wv[5], wv[6]) + Sha256_k[j] + w[j];
383 t2 = SHA256_F1(wv[0]) + MAJ(wv[0], wv[1], wv[2]);
384
385 wv[7] = wv[6];
386 wv[6] = wv[5];
387 wv[5] = wv[4];
388 wv[4] = wv[3] + t1;
389 wv[3] = wv[2];
390 wv[2] = wv[1];
391 wv[1] = wv[0];
392 wv[0] = t1 + t2;
393 }
394
395 for(j=0; j<8; j++)
396 {
397 ctx->h[j] += wv[j];
398 }
399 }
400}

458void HASHSHA2::Sha2_256_Update(HASHSHA2_256_CTX* ctx, XBYTE* input, XDWORD size)
459{
460 XDWORD blocknb;
461 XDWORD new_len, rem_len, tmp_len;
462 XBYTE* shifted_input;
463
464 tmp_len = HASHSHA2_256_BLOCK_SIZE - ctx->size;
465 rem_len = size < tmp_len ? size : tmp_len;
466

468
