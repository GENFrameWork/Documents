333void XLINUXDATETIME::GetActualDateTime(XLINUXDATETIME* timed, bool islocal)
334{
335 time_t now;
336 struct tm* datatime;
337
338 now = time(0);
339
340 if(islocal)
341 {
342 datatime = localtime(&now);
343 }
344 else
345 {
346 datatime = gmtime(&now);
347 }
348
349 if(!datatime) return;
350
351 timed->year = datatime->tm_year + 1900;
352 timed->month = datatime->tm_mon+1;
353 timed->day = datatime->tm_mday;
354 timed->hours = datatime->tm_hour;
355 timed->minutes = datatime->tm_min;
356 timed->seconds = datatime->tm_sec;
357
358 struct timeval tv;
359 gettimeofday(&tv, NULL);
360
361 timed->milliseconds = (tv.tv_usec /1000);
362}
