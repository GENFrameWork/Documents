126bool DIOLINUXGPIO::SetMode(DIOGPIO_ENTRY* entry, XWORD mode)
127{
128 if(!entry) return false;
129
130 if(GPIOExport_Add(entry->GetGPIO()))
131 {
132 GPIO_Export(entry->GetGPIO(), true);
133 }
134
135 bool isinput = true;
136
137 if((mode & DIOGPIO_MODE_INPUT) == DIOGPIO_MODE_INPUT) isinput = true;
138 if((mode & DIOGPIO_MODE_OUTPUT) == DIOGPIO_MODE_OUTPUT) isinput = false;
139
140 return GPIO_SetDirection(entry->GetGPIO(), isinput);
