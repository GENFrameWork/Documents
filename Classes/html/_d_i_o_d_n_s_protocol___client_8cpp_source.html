200bool DIODNSPROTOCOLCLIENT::ResolveURL(XCHAR* URL, DIOIP& IPresolved, int querytype, XDWORD timeout)
201{
202 #define ASK_ID 0x55AA
203
204 if(!diostreamudpcfg) return false;
205
206 DIOURL url;
207 XSTRING serverIPDNSstring;
208 XBUFFER askbuffer;
209 XBUFFER answerbuffer;
210 bool status = false;
211
212 url = URL;
213
214 if(!url.IsAURL())
215 {
216 IPresolved.Set(url);
217 return true;
218 }
219
220 serverIP.GetXString(serverIPDNSstring);
221
222 diostreamudpcfg->SetMode(DIOSTREAMMODE_CLIENT);
223 diostreamudpcfg->SetIsUsedDatagrams(false);
224 diostreamudpcfg->GetRemoteURL()->Set(serverIPDNSstring);
225 diostreamudpcfg->SetRemotePort(serverport?serverport:DIODNSPROTOCOL_DEFAULTPORT);
226
227 if(diostreamudp->Open())
228 {
229 DIODNSPROTOCOL_HEADER askheaderDNS;
230 XSTRING askURL;
231 XBUFFER EncodedURL;
232 DIODNSPROTOCOL_QUESTION question;
233
234 askheaderDNS.id = ASK_ID;
235 askheaderDNS.qr = 0; // This is a query
236 askheaderDNS.opcode = 0; // This is a standard query
237 askheaderDNS.aa = 0; // Not Authoritative
238 askheaderDNS.tc = 0; // This message is not truncated
239 askheaderDNS.rd = 1; // Recursion Desired
240 askheaderDNS.ra = 0; // Recursion not available! hey we dont have it (lol)
241 askheaderDNS.z = 0;
242 askheaderDNS.ad = 0;
243 askheaderDNS.cd = 0;
244 askheaderDNS.rcode = 0;
245 askheaderDNS.q_count = SwapWORD(1); // htons(1); //we have only 1 question
246 askheaderDNS.ans_count = 0;
247 askheaderDNS.auth_count = 0;
248 askheaderDNS.add_count = 0;
249
250 askbuffer.Add((XBYTE*)&askheaderDNS, sizeof(DIODNSPROTOCOL_HEADER));
251
252 askURL.Set(URL);
253 EncodeDNSFormat(askURL, EncodedURL);
254 askbuffer.Add(EncodedURL);
255
256 question.qtype = SwapWORD(querytype); // htons(querytype);
257 question.qclass = SwapWORD(1); // htons(1);
258
259 askbuffer.Add((XBYTE*)&question, sizeof(DIODNSPROTOCOL_QUESTION));
260
261 status = diostreamudp->Write(askbuffer);
262 if(status)
263 {
264 status = diostreamudp->WaitToFlushOutXBuffer(3);
265 }
266
267 if(status)
268 {
269 answerbuffer.Empty();
270 answerbuffer.Resize(65535);
271
272 status = diostreamudp->WaitToFilledReadingBuffer(sizeof(DIODNSPROTOCOL_HEADER) + sizeof(RES_RECORD), timeout);
273 if(status)
274 {
275 status = diostreamudp->Read(answerbuffer);
276 }
277 }
278
279 if(status)
280 {
281 status = false;
282
283 DIODNSPROTOCOL_HEADER* answerheaderDNS = (DIODNSPROTOCOL_HEADER*)answerbuffer.Get();
284 if(answerheaderDNS)
285 {
286 if(answerheaderDNS->id == ASK_ID)
287 {
288 int answerscount = SwapWORD(answerheaderDNS->ans_count);
289
290 for(int c=0; c<answerscount; c++)
291 {
292 RES_RECORD* result = new RES_RECORD();
293 if(result)
294 {
295 XBYTE* reader = &answerbuffer.Get()[askbuffer.GetSize()];
296 int type;
297 int stop = 0;
298
299 result->name = GetBufferName(reader, answerbuffer.Get(), &stop);
300 reader += stop;
301
302 result->resource = (DIODNSPROTOCOL_R_DATA*)reader;
303 reader += sizeof(DIODNSPROTOCOL_R_DATA);
304
305 type = SwapWORD(result->resource->type);
306
307 if(type == 1)
308 {
309 IPresolved.Set((XBYTE*)reader);
310 reader = reader + SwapWORD(result->resource->data_len); // ntohs(result->resource->data_len);
311 status = true;
312 }
313 else
314 {
315 result->rdata = GetBufferName(reader, answerbuffer.Get(), &stop);
316 reader += stop;
317
318 XSTRING URLmore;
319
320 URLmore = (char*)result->rdata;
321
322 status = ResolveURL((XCHAR*)URLmore.Get(), IPresolved, querytype, timeout);
323 }
324
325 free(result->name);
326 delete result;
327
328 if(status) break;
329 }
330 }
331 }
332 }
333 }
334
335 diostreamudp->Close();
336 }
337
338 return status;
339}

394bool DIODNSPROTOCOLCLIENT::EncodeDNSFormat(XSTRING& URL, XBUFFER& EncodedURL)
395{
396 int subsize = 0;
397
398 if(URL.IsEmpty())
399 {
400 return false;
401 }
402
403 for(int c=0; c<URL.GetSize(); c++)
404 {
405 for(int d=c; d<URL.GetSize(); d++)
406 {
407 if(URL.Get()[d] != __C('.'))
408 {
409 subsize++;
410 }
411 else
412 {
413 break;
414 }
415 }
416
417 EncodedURL.Add((XBYTE)subsize);
418
419 for(int d=0; d<subsize; d++)
420 {
421 EncodedURL.Add((XBYTE)URL.Get()[c]);
422 c++;
423 }
424
425 subsize = 0;
426 }
427
428 EncodedURL.Add((XBYTE)0x00);
429
430 return true;
431}

446bool DIODNSPROTOCOLCLIENT::DecodeDNSFormat(XBUFFER& EncodedURL, XSTRING& URL)
447{
448 URL.Empty();
449
450 int size = EncodedURL.GetSize();
451 if(!size)
452 {
453 return false;
454 }
455
456 int c=0;
457 while(c < size)
458 {
459 int count = EncodedURL.Get()[c];
460
461 if(!count)
462 {
463 break;
464 }
465
466 c++;
467
468 for(int d=0; d<count; d++)
469 {
470 URL.Add(EncodedURL.Get()[c]);
471 c++;
472 }
473
474 if(EncodedURL.Get()[c])
475 {
476 URL.Add(__C('.'));
477 }
478 }
479
480 return true;
481}

520XBYTE* DIODNSPROTOCOLCLIENT::GetBufferName(XBYTE* reader, XBYTE* buffer, int* count)
521{
522 XBYTE* name;
523 XDWORD p = 0;
524 XDWORD jumped = 0;
525 XDWORD offset;
526 int i;
527 int j;
528
529 *count = 1;
530 name = (XBYTE*)malloc(_MAXSTR);
531
532 name[0] = '\0';
533
534 //read the names in 3www6google3com format
535 while(*reader != 0)
536 {
537 if(*reader >= 192)
538 {
539 offset = (*reader)*256 + *(reader+1) - 49152; //49152 = 11000000 00000000 ;)
540 reader = buffer + offset - 1;
541 jumped = 1; //we have jumped to another location so counting wont go up!
542 }
543 else
544 {
545 name[p++] = *reader;
546 }
547
548 reader = reader + 1;
549
550 if(jumped == 0)
551 {
552 *count = *count + 1; //if we havent jumped to another location then we can count up
553 }
554 }
555
556 name[p] = '\0'; //string complete
557 if(jumped==1)
558 {
559 *count = *count + 1; //number of steps we actually moved forward in the packet
560 }
561
562 //now convert 3www6google3com0 to www.google.com
563 for(i=0; i<(int)strlen((const char*)name); i++)
564 {
565 p = name[i];
566
567 for(j=0; j<(int)p; j++)
568 {
569 name[i]=name[i+1];
570 i = i + 1;
571 }
572
573 name[i]='.';
574 }
575
576 name[i-1]='\0'; //remove the last dot
577
578 return name;
579}
