193bool DIODNSPROTOCOLCLIENT::ResolveURL(XCHAR* URL, DIOIP& IPresolved, int querytype, XDWORD timeout)
194{
195 #define ASK_ID 0x55AA
196
197 if(!diostreamudpcfg) return false;
198
199 DIOURL url;
200 XSTRING serverIPDNSstring;
201 XBUFFER askbuffer;
202 XBUFFER answerbuffer;
203 bool status = false;
204
205 url = URL;
206
207 if(!url.IsAURL())
208 {
209 IPresolved.Set(url);
210 return true;
211 }
212
213 serverIP.GetXString(serverIPDNSstring);
214
215 diostreamudpcfg->SetMode(DIOSTREAMMODE_CLIENT);
216 diostreamudpcfg->SetIsUsedDatagrams(false);
217 diostreamudpcfg->GetRemoteURL()->Set(serverIPDNSstring);
218 diostreamudpcfg->SetRemotePort(serverport?serverport:DIODNSPROTOCOL_DEFAULTPORT);
219
220 if(diostreamudp->Open())
221 {
222 DIODNSPROTOCOL_HEADER askheaderDNS;
223 XSTRING askURL;
224 XBUFFER EncodedURL;
225 DIODNSPROTOCOL_QUESTION question;
226
227 askheaderDNS.id = ASK_ID;
228 askheaderDNS.qr = 0; // This is a query
229 askheaderDNS.opcode = 0; // This is a standard query
230 askheaderDNS.aa = 0; // Not Authoritative
231 askheaderDNS.tc = 0; // This message is not truncated
232 askheaderDNS.rd = 1; // Recursion Desired
233 askheaderDNS.ra = 0; // Recursion not available! hey we dont have it (lol)
234 askheaderDNS.z = 0;
235 askheaderDNS.ad = 0;
236 askheaderDNS.cd = 0;
237 askheaderDNS.rcode = 0;
238 askheaderDNS.q_count = SwapWORD(1); // htons(1); //we have only 1 question
239 askheaderDNS.ans_count = 0;
240 askheaderDNS.auth_count = 0;
241 askheaderDNS.add_count = 0;
242
243 askbuffer.Add((XBYTE*)&askheaderDNS, sizeof(DIODNSPROTOCOL_HEADER));
244
245 askURL.Set(URL);
246 EncodeDNSFormat(askURL, EncodedURL);
247 askbuffer.Add(EncodedURL);
248
249 question.qtype = SwapWORD(querytype); // htons(querytype);
250 question.qclass = SwapWORD(1); // htons(1);
251
252 askbuffer.Add((XBYTE*)&question, sizeof(DIODNSPROTOCOL_QUESTION));
253
254 status = diostreamudp->Write(askbuffer);
255 if(status)
256 {
257 status = diostreamudp->WaitToFlushOutXBuffer(3);
258 }
259
260 if(status)
261 {
262 answerbuffer.Empty();
263 answerbuffer.Resize(65535);
264
265 status = diostreamudp->WaitToFilledReadingBuffer(sizeof(DIODNSPROTOCOL_HEADER) + sizeof(RES_RECORD), timeout);
266 if(status)
267 {
268 status = diostreamudp->Read(answerbuffer);
269 }
270 }
271
272 if(status)
273 {
274 status = false;
275
276 DIODNSPROTOCOL_HEADER* answerheaderDNS = (DIODNSPROTOCOL_HEADER*)answerbuffer.Get();
277 if(answerheaderDNS)
278 {
279 if(answerheaderDNS->id == ASK_ID)
280 {
281 int answerscount = SwapWORD(answerheaderDNS->ans_count);
282
283 for(int c=0; c<answerscount; c++)
284 {
285 RES_RECORD* result = new RES_RECORD();
286 if(result)
287 {
288 XBYTE* reader = &answerbuffer.Get()[askbuffer.GetSize()];
289 int type;
290 int stop = 0;
291
292 result->name = GetBufferName(reader, answerbuffer.Get(), &stop);
293 reader += stop;
294
295 result->resource = (DIODNSPROTOCOL_R_DATA*)reader;
296 reader += sizeof(DIODNSPROTOCOL_R_DATA);
297
298 type = SwapWORD(result->resource->type);
299
300 if(type == 1)
301 {
302 IPresolved.Set((XBYTE*)reader);
303 reader = reader + SwapWORD(result->resource->data_len); // ntohs(result->resource->data_len);
304 status = true;
305 }
306 else
307 {
308 result->rdata = GetBufferName(reader, answerbuffer.Get(), &stop);
309 reader += stop;
310
311 XSTRING URLmore;
312
313 URLmore = (char*)result->rdata;
314
315 status = ResolveURL((XCHAR*)URLmore.Get(), IPresolved, querytype, timeout);
316 }
317
318 free(result->name);
319 delete result;
320
321 if(status) break;
322 }
323 }
324 }
325 }
326 }
327
328 diostreamudp->Close();
329 }
330
331 return status;
332}

387bool DIODNSPROTOCOLCLIENT::EncodeDNSFormat(XSTRING& URL, XBUFFER& EncodedURL)
388{
389 int subsize = 0;
390
391 if(URL.IsEmpty())
392 {
393 return false;
394 }
395
396 for(int c=0; c<URL.GetSize(); c++)
397 {
398 for(int d=c; d<URL.GetSize(); d++)
399 {
400 if(URL.Get()[d] != __C('.'))
401 {
402 subsize++;
403 }
404 else
405 {
406 break;
407 }
408 }
409
410 EncodedURL.Add((XBYTE)subsize);
411
412 for(int d=0; d<subsize; d++)
413 {
414 EncodedURL.Add((XBYTE)URL.Get()[c]);
415 c++;
416 }
417
418 subsize = 0;
419 }
420
421 EncodedURL.Add((XBYTE)0x00);
422
423 return true;
424}

439bool DIODNSPROTOCOLCLIENT::DecodeDNSFormat(XBUFFER& EncodedURL, XSTRING& URL)
440{
441 URL.Empty();
442
443 int size = EncodedURL.GetSize();
444 if(!size)
445 {
446 return false;
447 }
448
449 int c=0;
450 while(c < size)
451 {
452 int count = EncodedURL.Get()[c];
453
454 if(!count)
455 {
456 break;
457 }
458
459 c++;
460
461 for(int d=0; d<count; d++)
462 {
463 URL.Add(EncodedURL.Get()[c]);
464 c++;
465 }
466
467 if(EncodedURL.Get()[c])
468 {
469 URL.Add(__C('.'));
470 }
471 }
472
473 return true;
474}

513XBYTE* DIODNSPROTOCOLCLIENT::GetBufferName(XBYTE* reader, XBYTE* buffer, int* count)
514{
515 XBYTE* name;
516 XDWORD p = 0;
517 XDWORD jumped = 0;
518 XDWORD offset;
519 int i;
520 int j;
521
522 *count = 1;
523 name = (XBYTE*)malloc(_MAXSTR);
524
525 name[0] = '\0';
526
527 //read the names in 3www6google3com format
528 while(*reader != 0)
529 {
530 if(*reader >= 192)
531 {
532 offset = (*reader)*256 + *(reader+1) - 49152; //49152 = 11000000 00000000 ;)
533 reader = buffer + offset - 1;
534 jumped = 1; //we have jumped to another location so counting wont go up!
535 }
536 else
537 {
538 name[p++] = *reader;
539 }
540
541 reader = reader + 1;
542
543 if(jumped == 0)
544 {
545 *count = *count + 1; //if we havent jumped to another location then we can count up
546 }
547 }
548
549 name[p] = '\0'; //string complete
550 if(jumped==1)
551 {
552 *count = *count + 1; //number of steps we actually moved forward in the packet
553 }
554
555 //now convert 3www6google3com0 to www.google.com
556 for(i=0; i<(int)strlen((const char*)name); i++)
557 {
558 p = name[i];
559
560 for(j=0; j<(int)p; j++)
561 {
562 name[i]=name[i+1];
563 i = i + 1;
564 }
565
566 name[i]='.';
567 }
568
569 name[i-1]='\0'; //remove the last dot
570
571 return name;
572}
