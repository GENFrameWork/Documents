136int SCRIPT_LNG_JAVASCRIPT::Run(int* returnval)
137{
138 errorcode = SCRIPT_ERRORCODE_NONE;
139 iscancelexec = false;
140
141 XSTRING _script = script;
142
143 if(HaveMainFunction())
144 {
145 XSTRING mainfunctionname = SCRIPT_LNG_JAVASCRIPT_MAINFUNCTIONNAME;
146 _script.AddFormat(__L("\n%s()\n"), mainfunctionname.Get());

148

152

157

209bool SCRIPT_LNG_JAVASCRIPT::HaveError(int _errorcode)
210{
211 XSTRING currenttoken;
212 XPATH namefile;
213 int errorcode = _errorcode;
214 XSTRING errorstring;
215 XSTRING errorline;
216 int nline = 0;
217
218 currenttoken = duk_safe_to_string(context, -1);
219
220 if(currenttoken.Find(__L("EvalError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_EVAL_ERROR;
221 if(currenttoken.Find(__L("RangeError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_RANGE_ERROR;
222 if(currenttoken.Find(__L("ReferenceError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_REFERENCE_ERROR;
223 if(currenttoken.Find(__L("SyntaxError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_SYNTAX_ERROR;
224 if(currenttoken.Find(__L("TypeError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_TYPE_ERROR;
225 if(currenttoken.Find(__L("URIError") , false) != XSTRING_NOTFOUND) errorcode = SCRIPT_LNG_JAVASCRIPT_ERRORCODE_URI_ERROR;
226
227 namefile.Set(namescript);
228 namefile.SetOnlyNamefileExt();
229
230 SCRIPT_XEVENT xevent(this, SCRIPT_XEVENT_TYPE_ERROR);
231
232 xevent.SetError(errorcode);
233 xevent.GetErrorText()->Set(currenttoken);
234 xevent.GetCurrentToken()->Set(currenttoken);
235 xevent.SetNLine(nline);
236
237 XTRACE_PRINTCOLOR(4, __L("Script [%s] ERROR %d:\"%s\"") , namefile.Get(), errorcode, currenttoken.Get());
238
239 PostEvent(&xevent);
240
241 this->errorcode = errorcode;
242 iscancelexec = true;
243
244
245 return (errorcode == SCRIPT_ERRORCODE_NONE)?false:true;
246
247}

259bool SCRIPT_LNG_JAVASCRIPT::HaveMainFunction()
260{
261 XSTRING mainfunctionname;
262 XSTRING script;
263 int start = 0;
264 int end = 0;
265 XSTRING line;
266
267 script = GetScript()->Get();
268 if(script.IsEmpty()) return false;
269
270 mainfunctionname = SCRIPT_LNG_JAVASCRIPT_MAINFUNCTIONNAME;
271
272 int index = GetScript()->Find(mainfunctionname, false);
273 if(index == XSTRING_NOTFOUND) return false;
274
275 for(start=index; start>0; start--)
276 {
277 if((script.Get()[start] == __C('\r')) || (script.Get()[start] == __C('\n')))
278 {
279 break;
280 }
281 }
282
283 for(end=index; end<(int)script.GetSize() ; end++)
284 {
285 if(script.Get()[end] == __C(')'))
286 {
287 break;
288 }
289 }
290
291 script.Copy(start, end+1, line);
292
293 line.DeleteCharacter(__C('\n'));
294 line.DeleteCharacter(__C('\r'));
295 line.DeleteCharacter(__C(' '));
296
297 XSTRING searchmask;
298
299 searchmask.Format(__L("function%s()"), mainfunctionname.Get());
300
301 if(!line.Compare(searchmask)) return true;
302
303 return false;
304}

318duk_ret_t SCRIPT_LNG_JAVASCRIPT::LibraryCallBack(duk_context* context)
319{
320 XVECTOR<XVARIANT*> params;
321 XVARIANT returnvalue;
322 int nreturnvalues = 0;
323 SCRIPT_LNG_JAVASCRIPT* script = NULL;
324 duk_memory_functions memoryfuncs;
325 void* funcptr = NULL;
326
327 duk_get_memory_functions(context, &memoryfuncs);
328
329 script = (SCRIPT_LNG_JAVASCRIPT*)memoryfuncs.udata;
330 if(!script) return 0;
331
332 duk_push_current_function(context);
333 duk_get_prop_string(context, -1, "\xFF""_ptr");
334 funcptr = (void*)duk_get_pointer(context, -1);
335 if(!funcptr)
336 {
337 return 0;
338 }
339
340 SCRIPT_LIB_FUNCTION* libfunction = script->GetLibraryFunction(funcptr);
341
342 if(!libfunction)
343 {
344 return 0;
345 }
346
347 if(!libfunction->GetFunctionLibrary())
348 {
349 return 0;
350 }
351
352 duk_pop_2(context);
353
354 duk_idx_t nargs;
355 duk_idx_t c;
356
357 nargs = duk_get_top(context);
358
359 for(c = 0; c < nargs; c++)
360 {
361 XVARIANT* variant = new XVARIANT();
362 if(variant)
363 {
364 switch((int)duk_get_type(context, c))
365 {
366 case DUK_TYPE_NONE : // no value, e.g. invalid index
367 break;
368
369 case DUK_TYPE_UNDEFINED : // ECMAScript undefined
370 break;
371
372 case DUK_TYPE_NULL : // ECMAScript null
373 //(*variant) = NULL;
374 break;
375
376 case DUK_TYPE_BOOLEAN : // ECMAScript boolean: 0 or 1
377 (*variant) = (duk_require_boolean(context, c)?true:false);
378 break;
379
380 case DUK_TYPE_NUMBER : // ECMAScript number: double
381 (*variant) = (double)duk_require_number(context, c);
382 break;
383
384 case DUK_TYPE_STRING : // ECMAScript string: CESU-8 / extended UTF-8 encoded
385 (*variant) = (char*)duk_require_string(context, c);
386 break;
387
388 case DUK_TYPE_OBJECT : // ECMAScript object: includes objects, arrays, functions, threads
389 break;
390
391 case DUK_TYPE_BUFFER : // fixed or dynamic, garbage collected byte buffer
392 break;
393
394 case DUK_TYPE_POINTER : // raw void pointer
395 break;
396
397 case DUK_TYPE_LIGHTFUNC : // lightweight function pointer
398 break;
399 }
400
401 params.Add(variant);
402 }
403 }
404
405 libfunction->GetFunctionLibrary()(libfunction->GetLibrary(), script, &params, &returnvalue);
406
407 params.DeleteContents();
408 params.DeleteAll();
409
410
411 switch(returnvalue.GetType())
412 {
413 case XVARIANT_TYPE_NULL : duk_push_null(context);
414 break;
415
416 case XVARIANT_TYPE_BOOLEAN : if((bool)returnvalue)
417 {
418 duk_push_true(context);
419 }
420 else
421 {
422 duk_push_false(context);
423 }
424
425 nreturnvalues++;
426 break;
427
428 case XVARIANT_TYPE_INTEGER : duk_push_int(context, (int)(returnvalue)); nreturnvalues++; break;
429 case XVARIANT_TYPE_CHAR : duk_push_int(context, (int)(returnvalue)); nreturnvalues++; break;
430 case XVARIANT_TYPE_XCHAR : break;
431 case XVARIANT_TYPE_FLOAT : { XDWORD data = returnvalue;
432
433 duk_push_int(context, data);
434 nreturnvalues++;
435 }
436 break;
437
438 case XVARIANT_TYPE_STRING : { XSTRING stringreturnvalue;
439
440 stringreturnvalue = (const XSTRING&)returnvalue;
441
442 XBUFFER charstr;
443
444 stringreturnvalue.ConvertToASCII(charstr);
445 duk_push_string(context, charstr.GetPtrChar());
446
447 nreturnvalues++;
448 }
449 break;
450
451 case XVARIANT_TYPE_DATE : break;
452 case XVARIANT_TYPE_TIME : break;
453 case XVARIANT_TYPE_DATETIME : break;
454 }
455
456
457 // Let Lua know how many return values we've passed
458 return nreturnvalues;
459}
