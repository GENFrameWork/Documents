146 if(!inbuffer) return false;
147 if(!outbuffer) return false;

153 return false;

164 }

168 status = DIOSTREAMSTATUS_GETTINGCONNECTION;
169
170 ResetXBuffers();
171
172 ResetConnectionStatistics();
173

241int DIOWINDOWSSTREAMICMP::IsReadyConnect(SOCKET socket)
242{
243 struct timeval tv;
244 int rc;
245 fd_set fdr;
246 fd_set fdw;
247 fd_set fds;
248
249 if(socket == INVALID_SOCKET) return -1;
250
251 FD_ZERO(&fdr);
252 FD_ZERO(&fdw);
253 FD_ZERO(&fds);
254
255 FD_SET((unsigned int)socket, &fdr);
256 FD_SET((unsigned int)socket, &fdw);
257 FD_SET((unsigned int)socket, &fds);
258
259 tv.tv_sec = 0;
260 tv.tv_usec = 100;
261
262 rc = select((int)(socket)+1,&fdr, &fdw, &fds, &tv);
263 if(rc == SOCKET_ERROR) return -1;
264
265 int status1 = FD_ISSET(socket, &fdr) ? 1 : 0;
266 int status2 = FD_ISSET(socket, &fdw) ? 1 : 0;
267 int status3 = FD_ISSET(socket, &fds) ? 1 : 0;
268
269 if(config->IsServer())
270 {
271 if(status1 || status2 ) return 1;
272 if(status3) return -1;
273 }
274 else
275 {
276 if((!status1) && status2) return 1;
277 if(status3) return -1;
278 }
279
280 return 0;
281}
