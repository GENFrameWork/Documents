181bool APPINTERNETSERVICES::Ini(APPCFG* cfg, XDWORD timeoutgetpublicip)
182{
183 bool status = false;
184
185 this->cfg = cfg;
186
187 xscheduler = new XSCHEDULER();
188 if(!xscheduler) return false;
189
190 XSCHEDULERTASK* xtask;
191
192 //-------------------------------------------------------------------
193
194
195 if(cfg->InternetServices_GetCheckInternetStatusCadence())
196 {
197 xtask = new XSCHEDULERTASK( xscheduler);
198 if(!xtask) return false;
199
200 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, cfg->InternetServices_GetCheckInternetStatusCadence());
201 xtask->SetID(APPINTERNETSERVICES_TASKID_CHECKCONNECTIONINTERNET);
202 xtask->SetIsStartImmediatelyCycles(true);
203 xtask->SetIsActive(true);
204
205 xscheduler->Task_Add(xtask);
206
207 checkinternetconnection = new DIOCHECKINTERNETCONNECTION(cfg->InternetServices_GetCheckInternetStatusCadence()/2);
208 status =(checkinternetconnection)?true:false;
209 }
210
211 //-------------------------------------------------------------------
212
213 if(cfg->InternetServices_GetCheckIPsChangeCadence())
214 {
215 scraperwebpublicIP = new DIOSCRAPERWEBPUBLICIP();
216 status = (scraperwebpublicIP?true:false);
217
218 if(status)
219 {
220 xtask = new XSCHEDULERTASK(xscheduler);
221 if(!xtask) return false;
222
223 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, 5); // Get IP first time
224 xtask->SetID(APPINTERNETSERVICES_TASKID_GETIPS);
225 xtask->SetIsStartImmediatelyCycles(false);
226 xtask->SetIsActive(true);
227
228 xscheduler->Task_Add(xtask);
229
230 #ifdef APP_CFG_DYNDNSMANAGER_ACTIVE
231 if(dyndnsmanager)
232 {
233 for(XDWORD c=0; c<cfg->DNSManager_GetURLs()->GetSize(); c++)
234 {
235 dyndnsmanager->AddDNS((*cfg->DNSManager_GetURL(c)));
236 }
237 }
238 else return true;
239 #endif
240 }
241 else return false;
242 }
243
244 //-------------------------------------------------------------------
245
246 if(cfg->InternetServices_GetUpdateTimeByNTPCadence())
247 {
248 for(XDWORD c=0; c<cfg->InternetServices_GetUpdateTimeNTPServers()->GetSize(); c++)
249 {
250 XSTRING* server = new XSTRING();
251 if(server)
252 {
253 server->Set(cfg->InternetServices_GetUpdateTimeNTPServer(c)->Get());
254 NTPservers.Add(server);
255
256 server = NULL;
257 }
258 }
259
260 xtask = new XSCHEDULERTASK( xscheduler);
261 if(!xtask) return false;
262
263 xtask->SetNCycles(XSCHEDULER_CYCLEFOREVER, cfg->InternetServices_GetUpdateTimeByNTPCadence());
264 xtask->SetID(APPINTERNETSERVICES_TASKID_CHECKNTPDATETIME);
265 xtask->SetIsStartImmediatelyCycles(true);
266 xtask->SetIsActive(true);
267
268 xscheduler->Task_Add(xtask);
269 }
270
271 //-------------------------------------------------------------------
272
273 SubscribeEvent(XEVENT_TYPE_SCHEDULER, xscheduler);
274 if(!xscheduler->Ini()) return false;
275
276 //-------------------------------------------------------------------
277
278 if(timeoutgetpublicip)
279 {
280 if(cfg->InternetServices_GetCheckIPsChangeCadence() > 0)
281 {
282 XTIMER* GEN_XFACTORY_CREATE(timeout, CreateTimer())
283 if(timeout)
284 {
285 do{ if(!publicIP.IsEmpty()) break;
286
287 GEN_XSLEEP.MilliSeconds(50);
288
289 } while(timeout->GetMeasureSeconds() <= timeoutgetpublicip);
290 }
291
292 GEN_XFACTORY.DeleteTimer(timeout);
293 }
294 }
295
296
297 return true;
298}

374bool APPINTERNETSERVICES::ChangeCadenceCheck(APPINTERNETSERVICES_TASKID taskID, int timecadenceseconds, bool startimmediatelycycles)
375{
376 if(!xscheduler)
377 {
378 return false;
379 }
380
381 if(!timecadenceseconds)
382 {
383 return false;
384 }
385
386 XSCHEDULERTASK* task = NULL;
387
388 xscheduler->GetMutexScheduler()->Lock();
389
390 task = xscheduler->Task_GetForID(taskID);
391 if(task)
392 {
393 task->SetNCycles(XSCHEDULER_CYCLEFOREVER, timecadenceseconds);
394 task->SetIsStartImmediatelyCycles(startimmediatelycycles);
395 task->SetIsActive(true);
396
397 task->ResetCondition();
398 }
399
400 xscheduler->GetMutexScheduler()->UnLock();
401
402 return task?true:false;

713bool APPINTERNETSERVICES::UpdateIPs(XSTRING& actualpublicIP)
714{
715 XSTRING actualautomaticlocalIP;
716 DIOSTREAMIPLOCALENUMDEVICES* enumdevices = (DIOSTREAMIPLOCALENUMDEVICES*)GEN_DIOFACTORY.CreateStreamEnumDevices(DIOSTREAMENUMTYPE_IP_LOCAL);
717 bool sendchangeevent = false;
718 bool status = false;
719
720 APPINTERNETSERVICES_XEVENT xevent(this, APPINTERNETSERVICES_XEVENT_TYPE_CHANGEIP);
721
722 APP_LOG_ENTRY(XLOGLEVEL_INFO, APP_CFG_LOG_SECTIONID_CONNEXIONS, false, __L("[Update IPs] Ini Get Local IP... "));
723
724 if(enumdevices)
725 {
726 DIOSTREAMDEVICEIP* device = (DIOSTREAMDEVICEIP*)enumdevices->GetFirstActiveDevice();
727 if(device) device->GetIP()->GetXString(actualautomaticlocalIP);
728
729 alllocalIP.Empty();
730
731 for(XDWORD c=0; c<enumdevices->GetDevices()->GetSize(); c++)
732 {
733 device = (DIOSTREAMDEVICEIP*)enumdevices->GetDevices()->Get(c);
734 if(device)
735 {
736 if((!device->GetIP()->IsEmpty()) &&
737 (!device->GetIP()->IsAPIPA()) &&
738 ((device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_ETHERNET) ||
739 (device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_WIFI) ||
740 (device->GetIPType() == DIOSTREAMIPDEVICE_TYPE_PPP)))
741 {
742 XSTRING locaIPstring;
743
744 if(!alllocalIP.IsEmpty()) alllocalIP.Add(",");
745
746 device->GetIP()->GetXString(locaIPstring);
747 alllocalIP.Add(locaIPstring.Get());
748
749 status = true;
750 }
751 }
752 }
753
754 GEN_DIOFACTORY.DeleteStreamEnumDevices(enumdevices);
755 }
756
757 APP_LOG_ENTRY(status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR, APP_CFG_LOG_SECTIONID_CONNEXIONS, false, __L("[Update IPs] End Get Local IP : [%s] "), actualautomaticlocalIP.Get());
758
759 if(status)
760 {
761 DIOIP ip;
762
763 status = false;
764
765 // Check Local IP changed
766 if(automaticlocalIP.Compare(actualautomaticlocalIP))
767 {
768 xevent.SetIsChangeLocalIP(true);
769 xevent.GetChangeLocalIP()->Set(actualautomaticlocalIP);
770
771 nchangeslocalIP++;
772
773 automaticlocalIP = actualautomaticlocalIP;
774 sendchangeevent = true;
775 }
776
777 APP_LOG_ENTRY(XLOGLEVEL_INFO, APP_CFG_LOG_SECTIONID_CONNEXIONS, false, __L("[Update IPs] Ini Get Public IP... "));
778
779 if(scraperwebpublicIP->Get(ip, 10, NULL, false))
780 {
781 ip.GetXString(actualpublicIP);
782 }
783
784 if(!actualpublicIP.IsEmpty())
785 {
786 // Check Public IP changed
787 if(publicIP.Compare(actualpublicIP))
788 {
789 xevent.SetIsChangePublicIP(true);
790 xevent.GetChangePublicIP()->Set(actualpublicIP);
791
792 nchangespublicIP++;
793
794 publicIP = actualpublicIP;
795 sendchangeevent = true;
796 }
797
798 status = true;
799 }
800
801 if(actualpublicIP.IsEmpty())
802 {
803 APP_LOG_ENTRY(XLOGLEVEL_ERROR, APP_CFG_LOG_SECTIONID_CONNEXIONS, false, __L("[Update IPs] End Get Public IP: Error not IP!"));
804 }
805 else
806 {
807 APP_LOG_ENTRY((status?XLOGLEVEL_INFO:XLOGLEVEL_ERROR), APP_CFG_LOG_SECTIONID_CONNEXIONS, false, __L("[Update IPs] End Get Public IP: [%s] (%s)"), actualpublicIP.Get(), sendchangeevent?__L("has changed"):__L("has not changed"));
808 }
809 }
810
811 if(sendchangeevent)
812 {
813 nchangesIP++;
814
815 xevent.SetNChangesIP(nchangesIP);
816 xevent.SetNChangesLocalIP(nchangeslocalIP);
817 xevent.SetNChangesPublicIP(nchangespublicIP);
818
819 PostEvent(&xevent);
820 }
821
822 return sendchangeevent;
823}

865bool APPINTERNETSERVICES::AdjustTimerByNTP(XVECTOR<XSTRING*>* servers)
866{
867 if(!servers) return false;
868
869 bool status = false;
870
871 DIONTP* ntp = new DIONTP();
872 if(ntp)
873 {
874 DIOURL* url = new DIOURL();
875 if(url)
876 {
877
878 for(XDWORD c=0; c<servers->GetSize(); c++)
879 {
880 (*url) = servers->Get(c)->Get();
881
882 xdatetime_local->Read();
883
884 if(endservices)
885 {
886 break;
887 }
888
889 status = ntp->GetTime((*url), DIONTP_DEFAULTTIMEOUT, GEN_XSYSTEM.HardwareUseLittleEndian(), (*xdatetime_local));
890 if(status)
891 {
892 if(cfg->InternetServices_GetUpdateTimeNTPMeridianDifference() == APP_CFG_INTERNETSERVICES_UPDATETIMENTPMERIDIANDIFFERENCE_AUTO)
893 {
894 xdatetime_local->AddSeconds(xdatetime_local->GetMeridianDifference() * 60);
895 }
896 else
897 {
898 xdatetime_local->AddSeconds(cfg->InternetServices_GetUpdateTimeNTPMeridianDifference() * 60);
899 }
900
901 if(cfg->InternetServices_GetUpdateTimeNTPUseDayLightSaving())
902 {
903 if(xdatetime_local->IsDayLigthSavingTime())
904 {
905 xdatetime_local->AddSeconds(3600);
906 }
907 }
908
909 status = xdatetime_local->Write();
910
911 break;
912 }
913 }
914
915 delete url;
916 }
917
918 delete ntp;
919 }
920
921 if(status)
922 {
923 APPINTERNETSERVICES_XEVENT xevent(this, APPINTERNETSERVICES_XEVENT_TYPE_ADJUSTDATETIME);
924 PostEvent(&xevent);
925 }
926

942{
943 if(checkinternetconnection)

945 checkinternetconnection->GetCheckConnections()->Connections_GetMutex()->Lock();
946 }
947
948 switch(event->GetTask()->GetID())
949 {
950 case APPINTERNETSERVICES_TASKID_CHECKCONNECTIONINTERNET : CheckInternetStatus();
951 break;
952
953 case APPINTERNETSERVICES_TASKID_GETIPS : // XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[APP Internet Services] Check Get IPs..."));

959 if(UpdateIPs(actualpublicIP))

962

968 UpdateDynDNSURLs(actualpublicIP);
969 #endif

971
972 if(!actualpublicIP.IsEmpty())

974 #ifdef APP_CFG_DYNDNSMANAGER_ACTIVE

976 ChangeCadenceCheck(APPINTERNETSERVICES_TASKID_GETIPS, cfg->InternetServices_GetCheckIPsChangeCadence());
977 #endif
978 }
979 }
980 break;
981
982 case APPINTERNETSERVICES_TASKID_CHECKNTPDATETIME : if(CheckInternetConnection())
983 {
984 AdjustTimerByNTP(&NTPservers);
985 }
986 break;
987 }

989

992 checkinternetconnection->GetCheckConnections()->Connections_GetMutex()->UnLock();
