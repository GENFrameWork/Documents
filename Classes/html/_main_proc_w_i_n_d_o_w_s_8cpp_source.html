742bool MAINPROCWINDOWS::IsRunningAsService()
743{
744 // return IsRunningAsService(APPMODE_SERVICE_NAME);
745
746 SERVICE_STATUS_PROCESS ssstatus;
747 DWORD bytesneeded;
748 SC_HANDLE schscmanager;
749
750 schscmanager = OpenSCManager( NULL, // local computer
751 NULL , // servicesActive database
752 SC_MANAGER_ALL_ACCESS ); // full access rights
753 if(!schscmanager) return false;
754
755 // Get a handle to the service.
756 SC_HANDLE schservice = OpenService( schscmanager , // SCM database
757 APPMODE_SERVICE_NAME , // name of service
758 SERVICE_ALL_ACCESS); // full access
759
760 if(!schservice)
761 {
762 CloseServiceHandle(schscmanager);
763 return false;
764 }
765
766 // Check the status in case the service is not stopped.
767
768 if(!QueryServiceStatusEx( schservice, // handle to service
769 SC_STATUS_PROCESS_INFO, // information level
770 (LPBYTE) &ssstatus, // address of structure
771 sizeof(SERVICE_STATUS_PROCESS), // size of structure
772 &bytesneeded ) ) // size needed if buffer is too small
773 {
774 CloseServiceHandle(schservice);
775 CloseServiceHandle(schscmanager);
776 return false;
777 }
778
779 DWORD processid = GetCurrentProcessId();
780 if(processid == ssstatus.dwProcessId)
781 {
782 CloseServiceHandle(schservice);
783 CloseServiceHandle(schscmanager);
784
785 return true;
786 }
787
788 CloseServiceHandle(schservice);
789 CloseServiceHandle(schscmanager);
790
791 if((!ssstatus.dwProcessId) && (ssstatus.dwCurrentState == SERVICE_START_PENDING)) return true;
792
793 return false;
794}
