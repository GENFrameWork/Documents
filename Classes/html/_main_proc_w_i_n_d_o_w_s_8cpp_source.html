680bool MAINPROCWINDOWS::IsRunningAsService()
681{
682 // return IsRunningAsService(APPMODE_SERVICE_NAME);
683
684 SERVICE_STATUS_PROCESS ssstatus;
685 DWORD bytesneeded;
686 SC_HANDLE schscmanager;
687
688 schscmanager = OpenSCManager( NULL, // local computer
689 NULL , // servicesActive database
690 SC_MANAGER_ALL_ACCESS ); // full access rights
691 if(!schscmanager) return false;
692
693 // Get a handle to the service.
694 SC_HANDLE schservice = OpenService( schscmanager , // SCM database
695 APPMODE_SERVICE_NAME , // name of service
696 SERVICE_ALL_ACCESS); // full access
697
698 if(!schservice)
699 {
700 CloseServiceHandle(schscmanager);
701 return false;
702 }
703
704 // Check the status in case the service is not stopped.
705
706 if(!QueryServiceStatusEx( schservice, // handle to service
707 SC_STATUS_PROCESS_INFO, // information level
708 (LPBYTE) &ssstatus, // address of structure
709 sizeof(SERVICE_STATUS_PROCESS), // size of structure
710 &bytesneeded ) ) // size needed if buffer is too small
711 {
712 CloseServiceHandle(schservice);
713 CloseServiceHandle(schscmanager);
714 return false;
715 }
716
717 DWORD processid = GetCurrentProcessId();
718 if(processid == ssstatus.dwProcessId)
719 {
720 CloseServiceHandle(schservice);
721 CloseServiceHandle(schscmanager);
722
723 return true;
724 }
725
726 CloseServiceHandle(schservice);
727 CloseServiceHandle(schscmanager);
728
729 if((!ssstatus.dwProcessId) && (ssstatus.dwCurrentState == SERVICE_START_PENDING)) return true;
730
731 return false;
732
733}

1011DWORD WINAPI Service_WorkerThread(LPVOID lpparam)
1012{
1013 WINDOWSSERVICE* service = (WINDOWSSERVICE*)lpparam;
1014 bool userstop = false;
1015 int status = 0;
1016
1017 if(!service) return false;
1018
1019 #ifdef APP_ACTIVE
1020 if(!mainprocwindows.Ini(&GEN_appmain, APPBASE_APPLICATIONMODE_TYPE_SERVICE))
1021 #else
1022 if(!mainprocwindows.Ini())
1023 #endif
1024 {
1025 status = 1;
1026 }
1027
1028 if(!status)
1029 {
1030 while(1)
1031 {
1032 if(!mainprocwindows.Update())
1033 {
1034 userstop = true;
1035 break;
1036 }
1037 }
1038 }
1039
1040
1041 APPBASE_EXITTYPE exittype = APPBASE_EXITTYPE_UNKNOWN;
1042
1043 #ifdef APP_ACTIVE
1044 exittype = mainprocwindows.GetAppMain()->GetApplication()->GetExitType();
1045 #endif
1046
1047 mainprocwindows.GetXPathExec()->Empty();
1048 mainprocwindows.End();
1049
1050
1051 if(exittype == APPBASE_EXITTYPE_BY_SERVICERELOAD)
1052 {
1053 service->SetMustRestartService(true);
1054 }
1055
1056 if(service->GetHandleStoppedEvent())
1057 {
1058 SetEvent(service->GetHandleStoppedEvent());
1059 }
1060
1061 if(userstop)
1062 {
1063 service->Stop();
1064 }
1065
1066 return ERROR_SUCCESS;
1067}
