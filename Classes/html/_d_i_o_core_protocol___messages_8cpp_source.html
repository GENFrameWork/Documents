298bool DIOCOREPROTOCOL_MESSAGES::Delete(XUUID* IDmessage)
299{
300 if(!IDmessage)
301 {
302 return false;
303 }
304
305 if(GetAll()->IsEmpty())
306 {
307 return false;
308 }
309
310 if(xmutexmessages)
311 {
312 xmutexmessages->Lock();
313 }
314
315
316 bool status = false;
317 XDWORD c = 0;
318
319 while(c < GetAll()->GetSize())
320 {
321 DIOCOREPROTOCOL_MESSAGE* request = NULL;
322 DIOCOREPROTOCOL_MESSAGE* response = NULL;
323 bool statusmessageID[2] = { false, false };
324
325 request = (DIOCOREPROTOCOL_MESSAGE*)GetAll()->GetKey(c);
326 if(request)
327 {
328 if(request->GetHeader())
329 {
330 if(request->GetHeader()->GetIDMessage()->Compare((*IDmessage)))
331 {
332 statusmessageID[0] = true;
333 }
334 }
335 }
336
337 response = (DIOCOREPROTOCOL_MESSAGE*)GetAll()->GetElement(c);
338 if(response)
339 {
340 if(response->GetHeader())
341 {
342 if(response->GetHeader()->GetIDMessage()->Compare((*IDmessage)))
343 {
344 statusmessageID[1] = true;
345 }
346 }
347 }
348
349 if(statusmessageID[0] && statusmessageID[1])
350 {
351 status = GetAll()->Delete(request);
352 if(status)
353 {
354 delete request;
355 delete response;
356
357 status = true;
358 break;
359 }
360 }
361
362 c++;
363 }
364
365 // ----------------------------------------------------------------------------------------------------------------
366
367 if(!status)
368 {
369 XSTRING UUID;
370 IDmessage->GetToString(UUID);
371
372 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[DIO Core Protocol Messages] Delete message %s"), UUID.Get());
373 }
374
375 // ----------------------------------------------------------------------------------------------------------------
376
377 if(xmutexmessages)
378 {
379 xmutexmessages->UnLock();
380 }
381
382 return status;
383}

686{

692 XBYTE color = XTRACE_COLOR_GREEN;
693 if(isserver)
694 {
695 color = XTRACE_COLOR_BLUE;
696 }
697
698 XTRACE_PRINTCOLOR(color, __L("-------------------------------------------------------------------------------"));
699 XTRACE_PRINTCOLOR(color, __L("Actual Messages: [%d]"), allmessages.GetSize());
700
701 for(XDWORD c=0; c<allmessages.GetSize(); c++)
702 {
703 DIOCOREPROTOCOL_MESSAGE* message_request = allmessages.GetKey(c);
704 DIOCOREPROTOCOL_MESSAGE* message_response = allmessages.GetElement(c);
705 XSTRING line;
706
707 line.AddFormat(__L(" [%d] "), c);
708 XTRACE_PRINTCOLOR(color, __L("%s"), line.Get());
709
710 line.Empty();
711 line.AddFormat(__L("%-24s"), __L("request : "));
712 if(message_request)
713 {
714 XSTRING UUID;
715 XSTRING operation;
716
717 switch(message_request->GetAcquisitionType())
718 {
719 case DIOCOREPROTOCOL_MESSAGE_TYPE_ACQUISITION_UNKNOWN : operation = __L("unknown"); break;
720 case DIOCOREPROTOCOL_MESSAGE_TYPE_ACQUISITION_READ : operation = __L("Read"); break;
721 case DIOCOREPROTOCOL_MESSAGE_TYPE_ACQUISITION_WRITE : operation = __L("Write"); break;
722 }
723
724 message_request->GetHeader()->GetIDMessage()->GetToString(UUID);
725 line.AddFormat(__L("%-12s %08X %-42s consumed: %-10s"), operation.Get(), message_request, UUID.Get(), message_request->IsConsumed()?__L("Ok"):__L("none"));
726 }
727 else
728 {
729 line.AddFormat(__L("%-12s %08X %-42s consumed: %-10s"), __L("unknown"), 0, __L("NULL"), __L("none"));
730 }
731
732 XTRACE_PRINTCOLOR(color, __L("%s"), line.Get());
733 if(message_request)
734 {
735 XTRACE_PRINTDATABLOCKCOLOR(color, (*message_request->GetContent()));
736 }
737
738 line.Empty();
739 line.AddFormat(__L("%-24s"), __L("response : "));
740 if(message_response)
741 {
742 XSTRING UUID;
743 XSTRING operation;
744
745 switch(message_response->GetAcquisitionType())

747 case DIOCOREPROTOCOL_MESSAGE_TYPE_ACQUISITION_UNKNOWN : operation = __L("unknown"); break;
