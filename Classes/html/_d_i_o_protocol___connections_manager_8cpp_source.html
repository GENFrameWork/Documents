1636void DIOPROTOCOL_CONNECTIONSMANAGER::ThreadProtocolConnections(void* param)
1637{
1638 DIOPROTOCOL_CONNECTIONSMANAGER* protocolconnections = (DIOPROTOCOL_CONNECTIONSMANAGER*)param;
1639 if(!protocolconnections) return;
1640
1641 if(protocolconnections->isserver)
1642 {
1643 protocolconnections->ManageProtocolConnectionsServer();
1644 }
1645 else
1646 {
1647 protocolconnections->ManageProtocolConnectionsClient();
1648 }
1649
1650 int c=0;
1651 while(c < protocolconnections->ProtocolConnections_GetNAvailable())
1652 {
1653 DIOPROTOCOL_CONNECTION* protocolconnection = (DIOPROTOCOL_CONNECTION*)protocolconnections->protocolconnections.Get(c);
1654 if(protocolconnection)
1655 {
1656 DIOPROTOCOL* protocol = protocolconnection->GetDIOProtocol();
1657 if(protocol)
1658 {
1659 if(protocol->IsConnected())
1660 {
1661 if(!protocol->IsInitialized())
1662 {
1663 if(protocolconnections->isserver)
1664 {
1665 protocolconnection->SetInUse(true, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1666
1667 protocol->GetDIOStream()->GetXTimerNotActivity()->Reset();
1668 protocolconnections->SendEvent(DIOPROTOCOL_CONNECTIONSMANAGER_XEVENT_TYPE_INITPROTOCOL, protocolconnection);
1669 if(protocol->IsInitialized()) protocol->CMD_SendIsInitServerProtocol(true);
1670
1671 protocolconnection->SetInUse(false, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1672 }
1673 else
1674 {
1675 XTIMER* GEN_XFACTORY_CREATE(timerout, CreateTimer())
1676
1677 while(1)
1678 {
1679 if(protocolconnection->GetDIOProtocol()->IsDisconnected()) break;
1680 if(protocol->IsServerInitialized()) break;
1681
1682 if(timerout)
1683 {
1684 if(timerout->GetMeasureSeconds() > 5) break;
1685 }
1686
1687 GEN_XSLEEP.MilliSeconds(10);
1688 }
1689
1690 GEN_XFACTORY.DeleteTimer(timerout);
1691
1692
1693 if(protocol->IsServerInitialized())
1694 {
1695 protocolconnection->SetInUse(true, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1696
1697 protocol->GetDIOStream()->GetXTimerNotActivity()->Reset();
1698 protocolconnections->SendEvent(DIOPROTOCOL_CONNECTIONSMANAGER_XEVENT_TYPE_INITPROTOCOL, protocolconnection);
1699
1700 protocolconnection->SetInUse(false, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1701 }
1702 }
1703
1704 XTRACE_PRINTCOLOR((protocol->IsInitialized()?2:4), __L("Protocol Connection Manager: %s Init Protocol %s") , protocolconnections->isserver?__L("Server"):__L("Client"), protocol->IsInitialized()?__L("Ok!"):__L("ERROR!"));
1705
1706 if(!protocol->IsInitialized())
1707 {
1708 protocolconnections->ProtocolConnections_Disconnect(c);
1709 }
1710 }
1711 else
1712 {
1713 if(protocol->GetDIOStream()->GetXTimerNotActivity()->GetMeasureSeconds() > DIOPROTOCOL_CONNECTIONS_DEFAULTIMECHECKCONNECTIONS)
1714 {
1715 XDWORD milliseconds;
1716
1717 bool status = protocol->CMD_Ping(milliseconds);
1718 if(!status)
1719 {
1720 protocolconnections->ProtocolConnections_Disconnect(c);
1721
1722 XTRACE_PRINTCOLOR(status?1:4, __L("Protocol Connection Manager: Check connection protocol %s"), status?__L("Ok!"):__L("ERROR!"));
1723 }
1724 }
1725 }
1726 }
1727 }
1728 }
1729
1730 c++;
1731 }
1732}
