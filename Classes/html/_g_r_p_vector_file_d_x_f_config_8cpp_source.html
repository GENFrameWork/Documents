130bool GRPVECTORFILEDXFCONFIG::SetHeader(GRPVECTORFILEDXFTEXTSECTIONHEADER* header)
131{
132 this->header = header;
133
134 fileCFG->DeleteAllValues();
135
136 for(XDWORD c=0; c<header->GetVariables()->GetSize(); c++)
137 {
138 XSTRING* namekey = header->GetVariables()->GetKey(c);
139 XVARIANT* variant = header->GetVariables()->GetElement(c);
140
141 if(namekey && variant)
142 {
143 switch(variant->GetType())
144 {
145 case XVARIANT_TYPE_BOOLEAN : fileCFG->AddValue(XFILECFG_VALUETYPE_BOOLEAN , VECTORFILEDXFCONFIG_SECTION_DFX_CFG, namekey->Get(), (bool*)(variant->GetData())); break;
146 case XVARIANT_TYPE_INTEGER : fileCFG->AddValue(XFILECFG_VALUETYPE_INT , VECTORFILEDXFCONFIG_SECTION_DFX_CFG, namekey->Get(), (int*)(variant->GetData())); break;
147 case XVARIANT_TYPE_FLOAT : fileCFG->AddValue(XFILECFG_VALUETYPE_FLOAT , VECTORFILEDXFCONFIG_SECTION_DFX_CFG, namekey->Get(), (float*)(variant->GetData())); break;
148 case XVARIANT_TYPE_STRING : fileCFG->AddValue(XFILECFG_VALUETYPE_STRING , VECTORFILEDXFCONFIG_SECTION_DFX_CFG, namekey->Get(), (XSTRING*)(variant->GetData())); break;
149
150 case XVARIANT_TYPE_NULL :
151 case XVARIANT_TYPE_DOUBLEINTEGER :
152 case XVARIANT_TYPE_DOUBLE :
153 case XVARIANT_TYPE_CHAR :
154 case XVARIANT_TYPE_XCHAR :
155 case XVARIANT_TYPE_DATE :
156 case XVARIANT_TYPE_TIME :
157 case XVARIANT_TYPE_DATETIME :
158 default : return false;
159 break;
160 }
161
162 XCHAR* remark = header->GetVariableRemark((*namekey));
163 if(remark)
164 {
165 fileCFG->AddRemark(VECTORFILEDXFCONFIG_SECTION_DFX_CFG, namekey->Get(), remark, 64, 0);
166 }
167
168 }
169 }
170
171 return true;
172}

212bool GRPVECTORFILEDXFCONFIG::Load_File(XCHAR* pathfile)
213{
214 if(!fileCFG)
215 {
216 return false;
217 }
218
219 if(!header)
220 {
221 return false;
222 }
223
224 XPATH path;
225 bool status = false;
226
227 path = pathfile;
228
229 header->DeleteAllVariables();
230
231 fileCFG->DeleteAllValues();
232
233 if(fileCFG->IniFile(path))
234 {
235 fileCFG->AjustRemarks();
236
237 XFILEINISECTION* sectionINI = fileCFG->GetFileINI()->GetSection(VECTORFILEDXFCONFIG_SECTION_DFX_CFG);
238 if(sectionINI)
239 {
240 XSTRING key;
241 XSTRING value;
242
243 while(sectionINI->GetNextKey(key, value))
244 {
245 XVARIANT variant;
246 XSTRING trash;
247 int type = 0;
248
249 trash.AdjustSize(_MAXSTR);
250
251 key.UnFormat(__L("$%s_%d"), trash.Get(), &type, NULL);
252
253 GRPVECTORFILEDXFTEXTSECTION::GetVariableFromLine(trash.Get(), type, &value, variant);
254
255 if(variant.GetType() != XVARIANT_TYPE_NULL)
256 {
257 header->AddVariable(key.Get(), &variant);
258 }
259 }
260 }
261
262 //status = fileCFG->EndFile();
263
264 fileCFG->GetFileINI()->XFILETXT::Close();
265 fileCFG->DeleteAllRemarks();
266 }
267
268 #ifdef XTRACE_ACTIVE
269 header->ShowTraceAllVariables();
270 #endif
271
272 return status;
273}
