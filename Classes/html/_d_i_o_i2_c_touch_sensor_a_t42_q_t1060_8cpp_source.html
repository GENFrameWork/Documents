270bool DIOI2CTOUCHSENSORAT42QT1060::Initializate()
271{
272 XBYTE hardwareID = 0;
273
274 if(!GetHW_ID(hardwareID)) return false;
275 if(hardwareID == DIOI2CTOUCHSENSORAT42QT1060_HW_ID) return false;
276
277 // Key Mask Enable CHG interrupt for KEYs 0-3
278 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_KEY_MASK_REG, 0x0F)) return false;
279 GEN_XSLEEP.MilliSeconds(10);
280
281 // AKS Mask Enable KEYs 0-3 in the AKS groups
282 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_AKS_MASK_REG, 0x0F)) return false;
283 GEN_XSLEEP.MilliSeconds(10);
284
285 // NTHR Key Threshold value KEY0. Count of Xms pooling
286 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_0, 30)) return false;
287 GEN_XSLEEP.MilliSeconds(10);
288
289 // NTHR Key Threshold value KEY1. Count of Xms pooling
290 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_1, 30)) return false;
291 GEN_XSLEEP.MilliSeconds(10);
292
293 // NTHR Key Threshold value KEY2. Count of Xms pooling
294 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_2, 30)) return false;
295 GEN_XSLEEP.MilliSeconds(10);
296
297 //NTHR Key Threshold value KEY3. Count of Xms pooling
298 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_3, 30)) return false;
299 GEN_XSLEEP.MilliSeconds(10);
300
301 //LP Mode 16ms pooling
302 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_LPM_MODE, 0x01)) return false;
303 GEN_XSLEEP.MilliSeconds(10);
304
305
306 // Charger GPIO Interrupts
307 DIOGPIO_ENTRY* interrupt = GEN_DIOGPIO.GPIOEntry_GetByID(GPIOinterruptentryID);
308 if(interrupt)
309 {
310 interrupt->SetIntFunctionPointer(DIOI2CTOUCHSENSORAT42QT1060::Handler_Interrupt);
311 interrupt->SetIntParamPointer((void*)this);
312 GEN_DIOGPIO.SetMode(GPIOinterruptentryID, DIOGPIO_MODE_INTERRUPT);
313 }
314
315 return true;
316}

350bool DIOI2CTOUCHSENSORAT42QT1060::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
351{
352 if(!diostream) return false;
353
354 bool status;
355
356 status = diostream->Write(&offset, 1)?true:false;
357 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
358
359 if(!status) return false;
360
361 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
362 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
363
364 return status;
365}
