277bool DIOI2CTOUCHSENSORAT42QT1060::Initializate()
278{
279 XBYTE hardwareID = 0;
280
281 if(!GetHW_ID(hardwareID)) return false;
282 if(hardwareID == DIOI2CTOUCHSENSORAT42QT1060_HW_ID) return false;
283
284 // Key Mask Enable CHG interrupt for KEYs 0-3
285 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_KEY_MASK_REG, 0x0F)) return false;
286 GEN_XSLEEP.MilliSeconds(10);
287
288 // AKS Mask Enable KEYs 0-3 in the AKS groups
289 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_AKS_MASK_REG, 0x0F)) return false;
290 GEN_XSLEEP.MilliSeconds(10);
291
292 // NTHR Key Threshold value KEY0. Count of Xms pooling
293 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_0, 30)) return false;
294 GEN_XSLEEP.MilliSeconds(10);
295
296 // NTHR Key Threshold value KEY1. Count of Xms pooling
297 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_1, 30)) return false;
298 GEN_XSLEEP.MilliSeconds(10);
299
300 // NTHR Key Threshold value KEY2. Count of Xms pooling
301 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_2, 30)) return false;
302 GEN_XSLEEP.MilliSeconds(10);
303
304 //NTHR Key Threshold value KEY3. Count of Xms pooling
305 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_NTHR_KEY_3, 30)) return false;
306 GEN_XSLEEP.MilliSeconds(10);
307
308 //LP Mode 16ms pooling
309 if(!WriteRegister(DIOI2CTOUCHSENSORAT42QT1060_LPM_MODE, 0x01)) return false;
310 GEN_XSLEEP.MilliSeconds(10);
311
312
313 // Charger GPIO Interrupts
314 DIOGPIO_ENTRY* interrupt = GEN_DIOGPIO.GPIOEntry_GetByID(GPIOinterruptentryID);
315 if(interrupt)
316 {
317 interrupt->SetIntFunctionPointer(DIOI2CTOUCHSENSORAT42QT1060::Handler_Interrupt);
318 interrupt->SetIntParamPointer((void*)this);
319 GEN_DIOGPIO.SetMode(GPIOinterruptentryID, DIOGPIO_MODE_INTERRUPT);
320 }
321
322 return true;
323}

357bool DIOI2CTOUCHSENSORAT42QT1060::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
358{
359 if(!diostream) return false;
360
361 bool status;
362
363 status = diostream->Write(&offset, 1)?true:false;
364 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
365
366 if(!status) return false;
367
368 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
369 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
370
371 return status;
372}
