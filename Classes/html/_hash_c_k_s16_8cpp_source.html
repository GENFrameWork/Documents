121bool HASHCKS16::Do(XBYTE* input, XQWORD size)
122{
123 if(!size) return false;
124
125 XWORD value = 0;
126
127 resulttmp->Extract(value);
128
129 for(XDWORD c=0; c<size; c++)
130 {
131 value = Update(value, (*(XWORD*)input));
132 input = input + sizeof(XWORD); // need to offset the input by 16 bits, as we are adding 16 by 16
133 }
134
135 value = ~value;
136
137 resulttmp->Add((XWORD)value);
138
139 return true;
140}

270XWORD HASHCKS16::Update(XWORD CKS16, XWORD c)
271{
272 XDWORD tmp = (XDWORD)CKS16 + (XDWORD)c;
273
274 // loop while we have a carry
275 while(tmp > 0xFFFF)
276 {
277 XWORD carry = (tmp >> 16) & (0xFFFF);
278 tmp = tmp + carry;
279 }
280
281 return (XWORD)tmp;
282}
