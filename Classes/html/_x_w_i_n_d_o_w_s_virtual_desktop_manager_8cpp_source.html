573{
574 if(!applicationfilepath) return false;
575 if(!desktopname) return false;
576

578 if(!XSTRING::GetSize(desktopname)) return false;
579

582

584
585 directoryapp.SetOnlyDriveAndPath();
586
587 STARTUPINFO sinfo = { 0 };

589
590 sinfo.cb = sizeof(sinfo);
591 sinfo.lpDesktop = desktopname;

593 //Lanuching a application into dekstop
594 BOOL processreturn = CreateProcess(applicationfilepath , NULL
595 , NULL
596 , NULL
597 , TRUE
598 , NORMAL_PRIORITY_CLASS
599 , NULL
600 , directoryapp.Get()
601 , &sinfo
602 , &pinfo);
603 if(processreturn) status = true;
