416bool DIOI2CAIRQUALITYCCS811::SetEnvironmentalData(float relativehumidity, float temperature)
417{
418 if((temperature < -25) || (temperature > 50)) return false;
419 if((relativehumidity < 0) || (relativehumidity > 100)) return false;
420
421 XDWORD rhum = (XDWORD)(relativehumidity * 1000); //42.348 becomes 42348
422 XDWORD temp = (XDWORD)(temperature * 1000); //23.2 becomes 23200
423
424 XBYTE envdata[4];
425
426 //Split value into 7-bit integer and 9-bit fractional
427
428 //Incorrect way from datasheet.
429 //envData[0] = ((rH % 1000) / 100) > 7 ? (rH / 1000 + 1) << 1 : (rH / 1000) << 1;
430 //envData[1] = 0; //CCS811 only supports increments of 0.5 so bits 7-0 will always be zero
431 //if (((rH % 1000) / 100) > 2 && (((rH % 1000) / 100) < 8))
432 //{
433 // envData[0] |= 1; //Set 9th bit of fractional to indicate 0.5%
434 //}
435
436 //Correct rounding. See issue 8: https://github.com/sparkfun/Qwiic_BME280_CCS811_Combo/issues/8
437 envdata[0] = (rhum + 250) / 500;
438 envdata[1] = 0; //CCS811 only supports increments of 0.5 so bits 7-0 will always be zero
439
440 temp += 25000; //Add the 25C offset
441
442 //Split value into 7-bit integer and 9-bit fractional
443 //envData[2] = ((temp % 1000) / 100) > 7 ? (temp / 1000 + 1) << 1 : (temp / 1000) << 1;
444 //envData[3] = 0;
445 //if (((temp % 1000) / 100) > 2 && (((temp % 1000) / 100) < 8))
446 //{
447 // envData[2] |= 1; //Set 9th bit of fractional to indicate 0.5C
448 //}
449
450 //Correct rounding
451 envdata[2] = (temp + 250) / 500;
452 envdata[3] = 0;
453
454 return WriteRegister(DIOI2CAIRQUALITYCCS811_OFFSET_ENV_DATA, envdata, sizeof(envdata));
455}

500bool DIOI2CAIRQUALITYCCS811::ReadNTC(float& temperature, float& resistance)
501{
502 XBYTE data[4];
503 XWORD vrefcounts = 0;
504 XWORD ntccounts = 0;
505
506 memset(data, 0, sizeof(data));
507
508 bool status = ReadRegister(DIOI2CAIRQUALITYCCS811_OFFSET_NTC, data, sizeof(data));
509
510 vrefcounts = ((XWORD)data[0] << 8) | data[1];
511 ntccounts = ((XWORD)data[2] << 8) | data[3];
512 resistance = ((float)ntccounts * referenceresistance / (float)vrefcounts);
513
514 temperature = (float)log((long)resistance);
515 temperature = (float)(1 / (0.001129148 + (0.000234125 * temperature) + (0.0000000876741 * temperature * temperature * temperature)));
516 temperature = (float)(temperature - 273.15); // Convert Kelvin to Celsius
517
518 return status;
519}
