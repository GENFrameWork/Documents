392void XBUFFERBITS::WriteU1(XDWORD v)
393{
394 bits_left--;
395
396 if(! IsEOF())
397 {
398 // FIXME this is slow, but we must clear bit first
399 // is it better to memset(0) the whole buffer during bs_init() instead?
400 // if we don't do either, we introduce pretty nasty bugs
401 (*(p)) &= ~(0x01 << bits_left);
402 (*(p)) |= ((v & 0x01) << bits_left);

404
405 if(bits_left == 0)

407 p++;

466 };
467
468 XDWORD len;
469
470 if (v == 0)
471 {
472 WriteU1(1);
473 }
474 else
475 {
476 v++;
477
478 if (v >= 0x01000000)
479 {
480 len = 24 + len_table[ v >> 24 ];
481 }
