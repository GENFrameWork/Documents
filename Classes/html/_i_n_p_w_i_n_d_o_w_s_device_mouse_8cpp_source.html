125bool INPWINDOWSDEVICEMOUSE::Update()
126{
127 if(!IsEnabled()) return false;
128
129 if(grpscreen)
130 {
131 if(grpscreen->GetHandle()!=GetForegroundWindow())
132 {
133 if(!grpscreen->Style_Is(GRPSCREENSTYLE_ONTOP))
134 {
135 return false;
136 }
137 }
138 }
139
140 float x=0.0f;
141 float y=0.0f;
142
143 POINT point;
144 if(GetCursorPos(&point))
145 {
146 int screenx;
147 int screeny;
148 int screenw;
149 int screenh;
150
151 if(GetWindowPosition(screenx,screeny,screenw,screenh))
152 {
153 x = (float)(point.x - screenx);
154 y = (float)(screeny - point.y);
155
156 if((x<0)||(y<0)||(x>screenw)||(y>screenh))
157 {
158 x = -1;
159 y = -1;
160 }
161 }
162 }
163
164 for(XDWORD c=0; c<buttons.GetSize(); c++)
165 {
166 INPBUTTON* button = buttons.Get(c);
167 if(button)
168 {
169 INPCURSOR* cursor = cursors.Get(c);
170 if(cursor)
171 {
172 if(x!=cursor->GetX() || y!=cursor->GetY())
173 {
174 cursor->SetIsChanged(true);
175 cursor->SetX(x);
176 cursor->SetY(y);
177 }
178 else cursor->SetIsChanged(false);
179 }
180
181
182 SHORT state = GetAsyncKeyState(button->GetKeyCode());
183 if(state & 0x8000) //key down
184 {
185 if(state & 0x01) //changed
186 {
187 button->SetPressed(true);
188
189 if(button->GetState()!=INPBUTTON_STATE_HOLD) button->SetState(INPBUTTON_STATE_PRESSED);
190
191 if(cursor) cursor->AddPointToMotion(true);
192 }
193 else
194 {
195 button->SetPressed(true);
196
197 button->SetState(INPBUTTON_STATE_HOLD);
198
199 if(cursor) cursor->AddPointToMotion(true);
200 }
201 }
202 else
203 {
204 button->SetPressed(false);
205
206 if (button->GetState() == INPBUTTON_STATE_HOLD || button->GetState() == INPBUTTON_STATE_PRESSED) //changed
207 button->SetState(INPBUTTON_STATE_RELEASED);
208 else button->SetState(INPBUTTON_STATE_UP);
209
210 if(cursor) cursor->AddPointToMotion(false);
211 }
212 }
213 }
214
215 return true;
216}

248bool INPWINDOWSDEVICEMOUSE::GetWindowPosition(int& screenx,int& screeny,int& screenw,int& screenh)
249{
250 if(!grpscreen) return false;
251
252 WINDOWINFO info;
253
254 GetWindowInfo((HWND)grpscreen->GetHandle(),&info);
255
256 screenx = info.rcClient.left;
257 screeny = info.rcClient.bottom;
258
259 screenw = abs(info.rcClient.right - info.rcClient.left);
260 screenh = abs(info.rcClient.bottom - info.rcClient.top);
261
262 return true;
263}
