118bool INPWINDOWSDEVICEMOUSE::Update()
119{
120 if(!IsEnabled()) return false;
121
122 if(grpscreen)
123 {
124 if(grpscreen->GetHandle()!=GetForegroundWindow())
125 {
126 return false;
127 }
128 }
129
130 float x=0.0f;
131 float y=0.0f;
132
133 POINT point;
134 if(GetCursorPos(&point))
135 {
136 int screenx;
137 int screeny;
138 int screenw;
139 int screenh;
140
141 if(GetWindowPosition(screenx,screeny,screenw,screenh))
142 {
143 x = (float)(point.x - screenx);
144 y = (float)(screeny - point.y);
145
146 if((x<0)||(y<0)||(x>screenw)||(y>screenh))
147 {
148 x = -1;
149 y = -1;
150 }
151 }
152 }
153
154 for(XDWORD c=0; c<buttons.GetSize(); c++)
155 {
156 INPBUTTON* button = buttons.Get(c);
157 if(button)
158 {
159 INPCURSOR* cursor = cursors.Get(c);
160 if(cursor)
161 {
162 if(x!=cursor->GetX() || y!=cursor->GetY())
163 {
164 cursor->SetIsChanged(true);
165 cursor->SetX(x);
166 cursor->SetY(y);
167 }
168 else cursor->SetIsChanged(false);
169 }
170
171
172 SHORT state = GetAsyncKeyState(button->GetKeyCode());
173 if(state & 0x8000) //key down
174 {
175 if(state & 0x01) //changed
176 {
177 button->SetPressed(true);
178
179 if(button->GetState()!=INPBUTTON_STATE_HOLD) button->SetState(INPBUTTON_STATE_PRESSED);
180
181 if(cursor) cursor->AddPointToMotion(true);
182 }
183 else
184 {
185 button->SetPressed(true);
186
187 button->SetState(INPBUTTON_STATE_HOLD);
188
189 if(cursor) cursor->AddPointToMotion(true);
190 }
191 }
192 else
193 {
194 button->SetPressed(false);
195
196 if (button->GetState() == INPBUTTON_STATE_HOLD || button->GetState() == INPBUTTON_STATE_PRESSED) //changed
197 button->SetState(INPBUTTON_STATE_RELEASED);
198 else button->SetState(INPBUTTON_STATE_UP);
199
200 if(cursor) cursor->AddPointToMotion(false);
201 }
202 }
203 }
204
205 return true;
206}

238bool INPWINDOWSDEVICEMOUSE::GetWindowPosition(int& screenx,int& screeny,int& screenw,int& screenh)
239{
240 if(!grpscreen) return false;
241
242 WINDOWINFO info;
243
244 GetWindowInfo((HWND)grpscreen->GetHandle(),&info);
245
246 screenx = info.rcClient.left;
247 screeny = info.rcClient.bottom;
248
249 screenw = abs(info.rcClient.right - info.rcClient.left);
250 screenh = abs(info.rcClient.bottom - info.rcClient.top);
251
252 return true;
253}
