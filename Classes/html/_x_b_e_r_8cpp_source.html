606{
607 if(!OIDstring) return false;
608 if(!OIDstring[0]) return false;
609
610 data.Delete();
611
612 XSTRING string;
613 XVECTOR<int> numbers;
614 int start = 0;
615 int end = 0;
616
617 string = OIDstring;
618
619 while(1)
620 {
621 XSTRING numberstring;
622 int indexstring = string.Find(__L("."),true,start);
623
624 if(indexstring != XSTRING_NOTFOUND)
625 {
626 end = indexstring;
627 string.Copy(start,end,numberstring);
628 numbers.Add(numberstring.ConvertToInt());
629 start = end+1;
630 }
631 else
632 {
633 string.Copy(end+1,numberstring);
634 numbers.Add(numberstring.ConvertToInt());
635
636 break;
637 }

640 if(!numbers.GetSize()) return false;

642 //---------------------------------------------------
643 // The firts 2 bytes are special
644 //

646 size_t _first = (size_t)(numbers.Get(0));
647 int first = (int)_first;

651 first += (int)(_first);
652
653 data.Add((XBYTE)first);

655 for(XDWORD c=2;c<numbers.GetSize();c++)

657 XBUFFER _data;

660 XDWORD number = (XDWORD)(_number);

859{
860 static XASN1_OID_PROPERTY* property = NULL;
861 static XBYTE level = 0;
862 XSTRING line;
863
864 if(buffer.IsEmpty()) return false;
865
866 tagtype = (XBYTE)(buffer.Get()[0] & XBER_TAG_MASKTYPE);
867
868 if(!CalculeSize(buffer, size, sizehead)) return false;
869
870 tagclass = XBER_TAG_CLASS(buffer.Get()[0]);
871
872 switch(tagclass)
873 {
874 case XBER_TAGCLASS_UNIVERSAL : break;
875
876 case XBER_TAGCLASS_APPLICATION : break;
877
878 case XBER_TAGCLASS_CONTEXT_SPECIFIC : contextspecificvalue = tagtype;

889 data.Delete();

891
892 line.Format(__L("(%d, %d) %s"), totalposition, size, nametagtype.Get());
893
894 switch(tagtype)
895 {

897 break;

899 case XBER_TAGTYPE_OBJECT_IDENTIFIER : { XSTRING valuestr;
900
901 ConvertToObjetIdentifier(data, value);

910 {
911 property = NULL;
912 }
913 }
914 }
915
916 break;
917
918 case XBER_TAGTYPE_BIT_STRING : unusedbits = data.Get()[0];

922 {

924 if(isconstructed) totalposition++;
925 }
926
927 property = NULL;
928 break;
929
930 case XBER_TAGTYPE_OCTET_STRING : if(property)
931 {

937

979
980 } else return false;
981 }
982
983 level--;
984 }
985 else
986 {
987 switch(tagtype)
988 {
989 case XBER_TAGTYPE_CONTEXT_SPECIFIC : break;
990
991 case XBER_TAGTYPE_BOOLEAN : ConvertToBoolean(data, value);

993

1220{

1223 XBYTE* bits = data.Get();

1225 if(length < 2 && !(length == 1 && bits[0] == 0))

1254 string.AddFormat(__L(".%d"), component);
