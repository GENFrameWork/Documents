514bool XBER::SetINTEGER(XDWORD integer)
515{
516 data.Delete();
517
518 XDWORD size = 0;
519
520 if(integer&0xFF800000)
521 {
522 size = 4;
523 }
524 else
525 {
526 if(integer&0x00FF8000)
527 {
528 size = 3;
529 }
530 else
531 {
532 if(integer&0x0000FF80)
533 {
534 size = 2;
535 }
536 else
537 {
538 size = 1;
539 }
540 }
541 }
542
543 if(!size) return false;

545 for(int c=(size-1);c>=0;c--)
546 {
547 XBYTE byte = (XBYTE)((c?(integer>>(c*8)):(integer))&0xFF);
548
549 data.Add((XBYTE)byte);
550 }
551
552 tagtype = XBER_TAGTYPE_INTEGER;
553 this->size = size;

744 XSTRING string;

747 int end = 0;

749 string = OIDstring;

753 XSTRING numberstring;

756 if(indexstring != XSTRING_NOTFOUND)

758 end = indexstring;
759 string.Copy(start,end,numberstring);

789 XBUFFER _data;
