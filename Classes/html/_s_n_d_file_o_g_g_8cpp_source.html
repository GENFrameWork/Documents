108bool SNDFILEOGG::LoadFile()
109{
110 XFILE* xfile = NULL;
111
112 xbuffer = new XBUFFER(false);
113 if(!xbuffer)
114 {
115 return false;
116 }
117
118 bool status = false;
119
120 GEN_XFACTORY_CREATE(xfile, Create_File())
121 if(!xfile)
122 {
123 xbuffer->Delete();
124 delete xbuffer;
125 xbuffer = NULL;
126 return false;
127 }
128
129 if(!xfile->Open(path))
130 {
131 xbuffer->Delete();
132 delete xbuffer;
133 xbuffer = NULL;
134
135 delete xfile;
136 xfile = NULL;
137
138 return false;
139 }
140
141 //XTRACE_PRINTCOLOR(0,__L("Loading file %s"), xfile->GetPathNameFile());
142
143 xbuffer->Resize((XDWORD)xfile->GetSize());
144 status = xfile->Read(xbuffer->Get(), xbuffer->GetSize());
145 xfile->Close();
146
147 GEN_XFACTORY.Delete_File(xfile);
148
149 // now perform decoding
150 stb_vorbis_info info;
151
152 stream = stb_vorbis_open_memory(xbuffer->Get(), xbuffer->GetSize(), NULL, NULL);
153 if(!stream)
154 {
155 XTRACE_PRINTCOLOR(4,__L("[SND File OGG] File Load Failed: %s"), xfile->GetPathNameFile());
156 return false; // need to check the specific error
157 }
158
159 info = stb_vorbis_get_info(stream);
160
161 channels = info.channels;
162 samplerate = info.sample_rate;
163 nsamples = stb_vorbis_stream_length_in_samples(stream)*channels; // this product is possibly redundant
164
165 duration = (int)(stb_vorbis_stream_length_in_seconds(stream)*1000);
166
167 xbufferdecodeddata = new XBUFFER(false);
168 if(!xbufferdecodeddata)
169 {
170 return false;
171 }
172
173 xbufferdecodeddata->Resize(nsamples*2); // we must multiply by 2 because ogg expects shorts
174
175 // perform the actual decoding
176 int read = stb_vorbis_get_samples_short_interleaved(stream, channels, (short*)(xbufferdecodeddata->Get()), nsamples);
177
178 // try to deallocate the allocacated memory
179 stb_vorbis_close(stream);
180
181 return true;
182}
