103bool GRPBITMAPFILETGAHEADER::Load(XFILE* file)
104{
105 if(!file->Read((XBYTE*)&headerinfo.IDlength , sizeof(XBYTE))) return false;
106 if(!file->Read((XBYTE*)&headerinfo.colourmaptype , sizeof(XBYTE))) return false;
107 if(!file->Read((XBYTE*)&headerinfo.datatypecode , sizeof(XBYTE))) return false;
108 if(!file->Read((XBYTE*)&headerinfo.colourmaporigin , sizeof(XWORD))) return false;
109 if(!file->Read((XBYTE*)&headerinfo.colourmaplength , sizeof(XWORD))) return false;
110 if(!file->Read((XBYTE*)&headerinfo.colourmapdepth , sizeof(XBYTE))) return false;
111 if(!file->Read((XBYTE*)&headerinfo.xorigin , sizeof(XWORD))) return false;
112 if(!file->Read((XBYTE*)&headerinfo.yorigin , sizeof(XWORD))) return false;
113 if(!file->Read((XBYTE*)&headerinfo.width , sizeof(XWORD))) return false;
114 if(!file->Read((XBYTE*)&headerinfo.height , sizeof(XWORD))) return false;
115 if(!file->Read((XBYTE*)&headerinfo.bitsperpixel , sizeof(XBYTE))) return false;
116 if(!file->Read((XBYTE*)&headerinfo.imagedescriptor , sizeof(XBYTE))) return false;
117
118 // Only RGB format valid.
119 if((headerinfo.datatypecode != GRPBITMAPFILETGATYPE_UNCOMPRESS_RGB) &&
120 (headerinfo.datatypecode != GRPBITMAPFILETGATYPE_RLECOMPRESS_RGB)) return false;
121
122 // Only 16,24,32 bits format valid.
123 if((headerinfo.bitsperpixel != 16) &&
124 (headerinfo.bitsperpixel != 24) &&
125 (headerinfo.bitsperpixel != 32)) return false;
126
127 // Only 0,1 colour map types
128 if((headerinfo.colourmaptype != 0) && (headerinfo.colourmaptype != 1)) return false;
129
130 switch(headerinfo.bitsperpixel)
131 {
132 case 16 : mode = GRPPROPERTYMODE_16_RGBA_5551; break;
133 case 24 : mode = GRPPROPERTYMODE_24_RGB_888; break;
134 case 32 : mode = GRPPROPERTYMODE_32_RGBA_8888; break;
135 }
136
137 offsetdata += sizeof(GRPBITMAPFILETGAHEADERINFO);
138 offsetdata += headerinfo.IDlength;
139 offsetdata += headerinfo.colourmaptype * headerinfo.colourmaplength;
140
141 return true;
142}

368GRPBITMAP* GRPBITMAPFILETGA::CreateBitmapFromFile(XPATH& xpath, GRPPROPERTYMODE forcemode)
369{
370 if(!Open(xpath)) return NULL;
371
372 GRPBITMAPFILETGAHEADER header;
373 GRPBITMAP* bitmap = NULL;
374 bool status = true;
375
376 if(!ReadHeaderFile(&header))
377 {
378 Close();
379 return NULL;
380 }
381
382 int width = header.GetWidth();
383 int height = header.GetHeight();
384 int bytesppixel = (header.GetBitsPerPixel()/8);
385 GRP2DCOLOR_RGBA8* colors = NULL;
386 int index = 0;
387
388 bitmap = GRPFACTORY::GetInstance().CreateBitmap(width, height, header.GetMode());
389 if(bitmap)
390 {
391 XBYTE databuffer[8];
392
393 file->SetPosition(header.GetOffsetData());
394
395 XBUFFER* filebuffer = new XBUFFER((height * width * bytesppixel), true);
396
397 file->Read(filebuffer->Get(), height * width * bytesppixel);
398 filebuffer->SetPosition(0);
399
400 colors = new GRP2DCOLOR_RGBA8[(height * width)+ 256 ];
401 if(colors)
402 {
403 while(index < (height * width) )
404 {
405 switch(header.GetType())
406 {
407 case GRPBITMAPFILETGATYPE_UNCOMPRESS_RGB : if(filebuffer->Get(databuffer, bytesppixel))
408 {
409 CreateColor(header.GetMode(), databuffer, colors[index]);
410 index++;
411
412 } else status = false;
413
414 break;
415
416 case GRPBITMAPFILETGATYPE_RLECOMPRESS_RGB : { XBYTE typepacket = 0;
417 int sizepacket = 0;
418
419 if(filebuffer->Get(&typepacket, 1))
420 {
421 sizepacket = (typepacket & 0x7F) + 1;
422
423 if(typepacket & 0x80) // run-length packet
424 {
425 if(filebuffer->Get(databuffer, bytesppixel))
426 {
427 for(int c=0; c<sizepacket; c++)
428 {
429 CreateColor(header.GetMode(), databuffer, colors[index]);
430 index++;
431 }
432 }
433 }
434 else
435 {
436 // raw packet
437 for(int c=0; c<sizepacket; c++)
438 {
439 if(filebuffer->Get(databuffer, bytesppixel))

441 CreateColor(header.GetMode(), databuffer, colors[index]);
442 index++;
443 }
444 }
445 }
446 }
447 }
448
449 break;
450
451 default : break;
452 }
453
454 if(!status) break;
455 }

460 }

463

467 {
468 for(int x=0; x<width; x++)
469 {
470 switch(header.GetMode())

484 }
485 }

492 bitmap = NULL;
493 }

667bool GRPBITMAPFILETGA::CreateFileFromBitmap(XPATH& xpath, GRPBITMAP* bitmap, int quality)
668{
669 if(!bitmap) return false;
670
671 if(!Create(xpath)) return false;
672
673 GRPBITMAPFILETGAHEADER header;
674
675 header.GetHeaderInfo()->IDlength = 0;
676 header.GetHeaderInfo()->colourmaptype = 0;
677 header.GetHeaderInfo()->datatypecode = GRPBITMAPFILETGATYPE_UNCOMPRESS_RGB;
678 header.GetHeaderInfo()->colourmaporigin = 0;
679 header.GetHeaderInfo()->colourmaplength = 0;
680 header.GetHeaderInfo()->colourmapdepth = 0;
681 header.GetHeaderInfo()->xorigin = 0;
682 header.GetHeaderInfo()->yorigin = 0;
683 header.GetHeaderInfo()->width = bitmap->GetWidth();
684 header.GetHeaderInfo()->height = bitmap->GetHeight();
685 header.GetHeaderInfo()->bitsperpixel = 32;
686 header.GetHeaderInfo()->imagedescriptor = 8;
687
688 int width = bitmap->GetWidth();
689 int height = bitmap->GetHeight();
690 int bytesppixel = (header.GetBitsPerPixel()/8);
691 int sizebmp = (header.GetWidth()*header.GetHeight()*bytesppixel);
692 int index = 0;
693 bool status = true;
694
695 if(file->Write((XBYTE*)header.GetHeaderInfo(), sizeof(GRPBITMAPFILETGAHEADERINFO)))
696 {
697 XBYTE* databuffer = new XBYTE[sizebmp];
698 if(databuffer)
699 {
700 for(int y=height-1; y>-1; y--)
701 {
702 for(int x=0; x<width; x++)
703 {
704 GRP2DCOLOR_RGBA8* color = (GRP2DCOLOR_RGBA8*)bitmap->GetPixel(x, y);
705 if(color)
706 {
707 databuffer[index+2] = color->r;
708 databuffer[index+1] = color->g;
709 databuffer[index+0] = color->b;
710 databuffer[index+3] = color->a;
711 }
712
713 index += bytesppixel;
714 }
715 }
716
717 if(!file->Write((XBYTE*)databuffer, sizebmp)) status = false;
718
719 delete [] databuffer;
720
721 } else status = false;
722
723 } else status = false;
724
725 Close();
726
727 return status;
728}

787bool GRPBITMAPFILETGA::CreateColor(GRPPROPERTYMODE mode, XBYTE* databuffer, GRP2DCOLOR_RGBA8& color)
788{
789 if(!databuffer) return false;
790
791 switch(mode)
792 {
793 case GRPPROPERTYMODE_16_RGBA_5551 : color.r = (databuffer[1] & 0x7c) << 1;
794 color.g = ((databuffer[1] & 0x03) << 6) | ((databuffer[0] & 0xe0) >> 2);
795 color.b = (databuffer[0] & 0x1f) << 3;
796 color.a = (databuffer[1] & 0x80)?0:255;
797 break;
798
799 case GRPPROPERTYMODE_24_RGB_888 : color.r = databuffer[2];
800 color.g = databuffer[1];
801 color.b = databuffer[0];
802 color.a = 255;
803 break;
804
805 case GRPPROPERTYMODE_32_RGBA_8888 : color.r = databuffer[2];
806 color.g = databuffer[1];
807 color.b = databuffer[0];
808 color.a = databuffer[3];
809 break;
810 default: break;
811 }
812
813 return true;
814}
