110bool GRPBITMAPFILETGAHEADER::Load(XFILE* file)
111{
112 if(!file->Read((XBYTE*)&headerinfo.IDlength , sizeof(XBYTE))) return false;
113 if(!file->Read((XBYTE*)&headerinfo.colourmaptype , sizeof(XBYTE))) return false;
114 if(!file->Read((XBYTE*)&headerinfo.datatypecode , sizeof(XBYTE))) return false;
115 if(!file->Read((XBYTE*)&headerinfo.colourmaporigin , sizeof(XWORD))) return false;
116 if(!file->Read((XBYTE*)&headerinfo.colourmaplength , sizeof(XWORD))) return false;
117 if(!file->Read((XBYTE*)&headerinfo.colourmapdepth , sizeof(XBYTE))) return false;
118 if(!file->Read((XBYTE*)&headerinfo.xorigin , sizeof(XWORD))) return false;
119 if(!file->Read((XBYTE*)&headerinfo.yorigin , sizeof(XWORD))) return false;
120 if(!file->Read((XBYTE*)&headerinfo.width , sizeof(XWORD))) return false;
121 if(!file->Read((XBYTE*)&headerinfo.height , sizeof(XWORD))) return false;
122 if(!file->Read((XBYTE*)&headerinfo.bitsperpixel , sizeof(XBYTE))) return false;
123 if(!file->Read((XBYTE*)&headerinfo.imagedescriptor , sizeof(XBYTE))) return false;
124
125 // Only RGB format valid.
126 if((headerinfo.datatypecode != GRPBITMAPFILETGATYPE_UNCOMPRESS_RGB) &&
127 (headerinfo.datatypecode != GRPBITMAPFILETGATYPE_RLECOMPRESS_RGB)) return false;
128
129 // Only 16,24,32 bits format valid.
130 if((headerinfo.bitsperpixel != 16) &&
131 (headerinfo.bitsperpixel != 24) &&
132 (headerinfo.bitsperpixel != 32)) return false;
133
134 // Only 0,1 colour map types
135 if((headerinfo.colourmaptype != 0) && (headerinfo.colourmaptype != 1)) return false;
136
137 switch(headerinfo.bitsperpixel)
138 {
139 case 16 : mode = GRPPROPERTYMODE_16_RGBA_5551; break;
140 case 24 : mode = GRPPROPERTYMODE_24_RGB_888; break;
141 case 32 : mode = GRPPROPERTYMODE_32_RGBA_8888; break;
142 }
143
144 offsetdata += sizeof(GRPBITMAPFILETGAHEADERINFO);
145 offsetdata += headerinfo.IDlength;
146 offsetdata += headerinfo.colourmaptype * headerinfo.colourmaplength;
147
148 return true;
149}

406

408 if(colors)

410 while(index < (height * width) )
411 {
412 switch(header.GetType())

415 {
416 CreateColor(header.GetMode(), databuffer, colors[index]);
417 index++;

427 {
428 sizepacket = (typepacket & 0x7F) + 1;
429
430 if(typepacket & 0x80) // run-length packet
431 {
432 if(filebuffer->Get(databuffer, bytesppixel))

674bool GRPBITMAPFILETGA::CreateFileFromBitmap(XPATH& xpath, GRPBITMAP* bitmap, int quality)
675{
676 if(!bitmap) return false;
677
678 if(!Create(xpath)) return false;
679
680 GRPBITMAPFILETGAHEADER header;
681
682 header.GetHeaderInfo()->IDlength = 0;
683 header.GetHeaderInfo()->colourmaptype = 0;
684 header.GetHeaderInfo()->datatypecode = GRPBITMAPFILETGATYPE_UNCOMPRESS_RGB;
685 header.GetHeaderInfo()->colourmaporigin = 0;
686 header.GetHeaderInfo()->colourmaplength = 0;
687 header.GetHeaderInfo()->colourmapdepth = 0;
688 header.GetHeaderInfo()->xorigin = 0;
689 header.GetHeaderInfo()->yorigin = 0;
690 header.GetHeaderInfo()->width = bitmap->GetWidth();
691 header.GetHeaderInfo()->height = bitmap->GetHeight();
692 header.GetHeaderInfo()->bitsperpixel = 32;
693 header.GetHeaderInfo()->imagedescriptor = 8;
694
695 int width = bitmap->GetWidth();
696 int height = bitmap->GetHeight();
697 int bytesppixel = (header.GetBitsPerPixel()/8);
698 int sizebmp = (header.GetWidth()*header.GetHeight()*bytesppixel);
699 int index = 0;
700 bool status = true;
701
702 if(file->Write((XBYTE*)header.GetHeaderInfo(), sizeof(GRPBITMAPFILETGAHEADERINFO)))
703 {
704 XBYTE* databuffer = new XBYTE[sizebmp];
705 if(databuffer)
706 {
707 for(int y=height-1; y>-1; y--)
708 {
709 for(int x=0; x<width; x++)
710 {
711 GRP2DCOLOR_RGBA8* color = (GRP2DCOLOR_RGBA8*)bitmap->GetPixel(x, y);
712 if(color)
713 {
714 databuffer[index+2] = color->r;
715 databuffer[index+1] = color->g;
716 databuffer[index+0] = color->b;
717 databuffer[index+3] = color->a;
718 }
719
720 index += bytesppixel;
721 }
722 }
723
724 if(!file->Write((XBYTE*)databuffer, sizebmp)) status = false;
725
726 delete [] databuffer;
727
728 } else status = false;
729
730 } else status = false;
731
732 Close();
733
734 return status;
735}

794bool GRPBITMAPFILETGA::CreateColor(GRPPROPERTYMODE mode, XBYTE* databuffer, GRP2DCOLOR_RGBA8& color)
795{
796 if(!databuffer) return false;
797
798 switch(mode)
799 {
800 case GRPPROPERTYMODE_16_RGBA_5551 : color.r = (databuffer[1] & 0x7c) << 1;
801 color.g = ((databuffer[1] & 0x03) << 6) | ((databuffer[0] & 0xe0) >> 2);
802 color.b = (databuffer[0] & 0x1f) << 3;
803 color.a = (databuffer[1] & 0x80)?0:255;
804 break;
805
806 case GRPPROPERTYMODE_24_RGB_888 : color.r = databuffer[2];
807 color.g = databuffer[1];
808 color.b = databuffer[0];
809 color.a = 255;
810 break;
811
812 case GRPPROPERTYMODE_32_RGBA_8888 : color.r = databuffer[2];
813 color.g = databuffer[1];
814 color.b = databuffer[0];
815 color.a = databuffer[3];
816 break;
817 default: break;
818 }
819
820 return true;
821}
