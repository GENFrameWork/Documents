384bool INPCURSORMOTION::AddFromLine(INPCURSORMOTION_REDUCEDMODE reducedmode, int reducedfactor, int x1, int y1, int x2, int y2)
385{
386 int x, xinc1, xinc2;
387 int y, yinc1, yinc2;
388 int i, deltax, deltay, npixels;
389 int d, dinc1, dinc2;
390
391 // Calculate deltax and deltay for initialisation
392 deltax = abs(x2 - x1);
393 deltay = abs(y2 - y1);
394
395 // Initialize all vars based on which is the independent variable
396 if(deltax>=deltay)
397 {
398 // x is independent variable }
399 npixels = deltax + 1;
400 d = (2 * deltay) - deltax;
401 dinc1 = deltay >> 1;
402 dinc2 = (deltay - deltax) >> 1;
403 xinc1 = 1;
404 xinc2 = 1;
405 yinc1 = 0;
406 yinc2 = 1;
407 }
408 else
409 {
410 // y is independent variable }
411 npixels = deltay + 1;
412 d = (2 * deltax) - deltay;
413 dinc1 = deltax >> 1;
414 dinc2 = (deltax - deltay) >> 1;
415 xinc1 = 0;
416 xinc2 = 1;
417 yinc1 = 1;
418 yinc2 = 1;
419 }
420
421 // Make sure x and y move in the right directions
422 if(x1>x2)
423 {
424 xinc1 = -xinc1;
425 xinc2 = -xinc2;
426 }
427
428 if(y1>y2)
429 {
430 yinc1 = -yinc1;
431 yinc2 = -yinc2;
432 }
433
434 // Start drawing at
435 x = x1;
436 y = y1;
437
438
439 int steps = CalculateReducedModeSteps(reducedmode,reducedfactor,npixels);
440 int incstep = steps;
441 bool addpoint = false;
442
443 // Draw the pixels }
444 for(i=0;i<npixels;i++)
445 {
446 if(incstep == steps)
447 {
448 addpoint = true;
449 incstep = 0;
450 }
451
452 if(addpoint)
453 AddPoint((float)x, (float)y, 0);
454 else incstep++;
455
456 addpoint = false;
457
458 if(d<0)
459 {
460 d = d + dinc1;
461 x = x + xinc1;
462 y = y + yinc1;
463 }
464 else
465 {
466 d = d + dinc2;
467 x = x + xinc2;
468 y = y + yinc2;
469 }
470 }
471
472 return false;
473}

491bool INPCURSORMOTION::AddFromCircle(INPCURSORMOTION_REDUCEDMODE reducedmode,int reducedfactor,int xc, int yc, int radio)
492{

856INPCURSORMOTION_DIR INPCURSORMOTION::GetDirectionByAngle(bool indetail)
857{
858 if(points.GetSize() < 2) return INPCURSORMOTION_DIR_UNKNOWN;
859
860 float angle = GetAngle();
861
862 if(indetail)
863 {
864 if((angle > 337.5f) || (angle <= 22.5f)) return INPCURSORMOTION_DIR_RIGHT;
865 if((angle > 22.5f) && (angle <= 67.5f)) return INPCURSORMOTION_DIR_RIGHTUP;
866 if((angle > 67.5f) && (angle <= 112.5f)) return INPCURSORMOTION_DIR_UP;
867 if((angle > 112.5f) && (angle <= 157.5f)) return INPCURSORMOTION_DIR_LEFTUP;
868 if((angle > 157.5f) && (angle <= 202.5f)) return INPCURSORMOTION_DIR_LEFT;
869 if((angle > 202.5f) && (angle <= 247.5f)) return INPCURSORMOTION_DIR_LEFTDOWN;
870 if((angle > 247.5f) && (angle <= 292.5f)) return INPCURSORMOTION_DIR_DOWN;
871 if((angle > 292.5f) && (angle <= 337.5f)) return INPCURSORMOTION_DIR_RIGHTDOWN;
872 }
873 else
874 {
875 if((angle > 270) || (angle <= 45)) return INPCURSORMOTION_DIR_RIGHT;
876 if((angle > 45) && (angle <= 135)) return INPCURSORMOTION_DIR_UP;
877 if((angle > 135) && (angle <= 180)) return INPCURSORMOTION_DIR_LEFT;
878 if((angle > 180) && (angle <= 270)) return INPCURSORMOTION_DIR_DOWN;
879 }
880
881 return INPCURSORMOTION_DIR_UNKNOWN;
882}

896INPCURSORMOTION_DIR INPCURSORMOTION::GetDirectionByDifferential(XDWORD* differential)
897{
898 INPCURSORMOTION_DIR dir = INPCURSORMOTION_DIR_UNKNOWN;
899
900 if(differential) (*differential) = 0;
901
902 if(points.GetSize() < 2) return dir;
903
904 int max_up = 0;
905 int max_down = 0;
906 int max_right = 0;
907 int max_left = 0;
908
909 int x_origin = (int)points.Get(0)->GetX();
910 int y_origin = (int)points.Get(0)->GetY();
911
912 for(XDWORD c=1; c<points.GetSize(); c++)
913 {
914 max_right = __MAX(((int)points.Get(c)->GetX() - x_origin), max_right);
915 max_left = __MAX((x_origin - (int)points.Get(c)->GetX()), max_left);
916
917 max_up = __MAX(((int)points.Get(c)->GetY() - y_origin), max_up);
918 max_down = __MAX((y_origin - (int)points.Get(c)->GetY()), max_down);
919 }
920
921 int x = __MAX(max_right, max_left);
922 int y = __MAX(max_up , max_down);
923
924 if(x > y)
925 {
926 if(differential) (*differential) = x;
927
928 dir = INPCURSORMOTION_DIR_LEFT;
929 if(max_right > max_left) dir = INPCURSORMOTION_DIR_RIGHT;
930 }
931 else
932 {
933 if(differential) (*differential) = y;
934
935 dir = INPCURSORMOTION_DIR_DOWN;
936 if(max_up > max_down) dir = INPCURSORMOTION_DIR_UP;
937 }
938
939 return dir;
940}
