169bool DIOSPITFTDISPLAYST7789::Clear(XWORD color)
170{
171 int size = GetSizeBuffer();
172
173 if(!TFT_SetWindow(0, 0, width, height)) return false;
174
175 SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_RAMWR); // write to RAM
176
177 XBYTE chigh = (XBYTE)(color >> 8);
178 XBYTE clow = (XBYTE)(color & 0x00FF);
179
180 canvasbuffer.Delete();
181 canvasbuffer.Resize(size);
182 canvasbuffer.ResetPosition();
183
184 int c = 0;
185
186 do{ canvasbuffer.Set(chigh); c++;
187 canvasbuffer.Set(clow); c++;
188
189 } while(c < size);
190
191 return SendData();
192}

293bool DIOSPITFTDISPLAYST7789::TFT_Init()
294{
295 _colstart = 0;
296 _rowstart = 0;
297 _xstart = 0;
298 _ystart = 0;
299
300 //TFT_Reset();
301
302 if(!SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_SWRESET)) return false; // Software reset
303 GEN_XSLEEP.MilliSeconds(150);
304
305 if(!SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_SLPOUT)) return false; // Out of sleep mode
306 GEN_XSLEEP.MilliSeconds(500);
307
308 if(!SendCommandParams(DIOSPITFTDISPLAYST7789_CMD_ST7789_COLMOD, 1, 0x55)) return false; // Set color mode 0x55 (16-bit color)
309 GEN_XSLEEP.MilliSeconds(10);
310
311 if(!SendCommandParams(DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL , 1, 0x00)) return false; // Memory access ctrl (directions) 0x00 Row addr/col addr, bottom to top refresh
312
313 if(!SendCommandParams(DIOSPITFTDISPLAYST7789_CMD_ST7789_CASET , 4 // Column addr set
314 , 0x00 // XSTART = 0
315 , DIOSPITFTDISPLAYST7789_240x240_XSTART
316 , (DIOSPITFTDISPLAYST7789_TFTWIDTH + DIOSPITFTDISPLAYST7789_240x240_XSTART) >> 8
317 , (DIOSPITFTDISPLAYST7789_TFTWIDTH + DIOSPITFTDISPLAYST7789_240x240_XSTART) & 0xFF)) return false; // XEND = 240
318
319 if(!SendCommandParams(DIOSPITFTDISPLAYST7789_CMD_ST7789_RASET , 4 // Row addr set
320 , 0x00 // YSTART = 0
321 , DIOSPITFTDISPLAYST7789_240x240_YSTART
322 , (DIOSPITFTDISPLAYST7789_TFTHEIGHT + DIOSPITFTDISPLAYST7789_240x240_YSTART) >> 8
323 , (DIOSPITFTDISPLAYST7789_TFTHEIGHT + DIOSPITFTDISPLAYST7789_240x240_YSTART) & 0xFF)) return false; // YEND = 240
324
325
326
327 if(!SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_INVON)) return false; // Inversion ON
328 GEN_XSLEEP.MilliSeconds(10);
329
330 if(!SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_NORON)) return false; // Normal display on
331 GEN_XSLEEP.MilliSeconds(10);
332
333 if(!SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_DISPON)) return false; // Main display turn on, no args, w/delay
334 GEN_XSLEEP.MilliSeconds(20);
335
336 TFT_SetRotation(1);
337
338 return true;
339}

435{
436 SendCommand(DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL);
437
438 rotation = (mode & 3);
439
440 switch(rotation)
441 {
442 case 0: SendData((XBYTE)(DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_MX | DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_MY | DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_RGB));
443 _xstart = _colstart;
444 _ystart = _rowstart;
445 break;
446
447 case 1: SendData((XBYTE)(DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_MY | DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_MV | DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_RGB));
448 _ystart = _colstart;
449 _xstart = _rowstart;
450 break;
451
452 case 2: SendData((XBYTE)(DIOSPITFTDISPLAYST7789_CMD_ST7789_MADCTL_RGB));
