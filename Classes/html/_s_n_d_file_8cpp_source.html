118SNDFILE* SNDFILE::Create(XCHAR* path)
119{
120 XPATH xpath;
121 XSTRING ext;
122 SNDFILE* sndfile = NULL;
123
124 xpath = path;
125 if(xpath.IsEmpty())
126 {
127 return NULL;
128 }
129
130 xpath.GetExt(ext);
131
132 if(!ext.Compare(__L(".ogg"), true))
133 {
134 sndfile = new SNDFILEOGG();
135 }
136
137 if(!ext.Compare(__L(".wav"), true))
138 {
139 sndfile = new SNDFILEWAV();
140 }
141
142 if(sndfile)
143 {
144 XFILE* file = GEN_XFACTORY.Create_File();
145 if(file)
146 {
147 bool exist = file->Exist(xpath);
148 GEN_XFACTORY.Delete_File(file);
149
150 if(!exist)
151 {
152 delete sndfile;
153 return NULL;
154 }
155 }
156
157 sndfile->GetPath()->Set(path);
158
159 if(!sndfile->LoadFile())
160 {
161 delete sndfile;
162 return NULL;
163 }
164
165
166 }
167
168 return sndfile;
169}
