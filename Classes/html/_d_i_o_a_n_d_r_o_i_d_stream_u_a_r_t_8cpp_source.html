176bool DIOANDROIDSTREAMUART::Open()
177{
178 if(!config) return false;
179
180 XSTRING ttyS;
181
182 if(!config->GetLocalDeviceName()->IsEmpty())
183 {
184 ttyS = config->GetLocalDeviceName()->Get();
185 }
186 else
187 {
188 ttyS.Format(__L("/dev/ttyS%d"),config->GetPort());
189 }
190
191 XBUFFER charstr;
192
193 ttyS.ConvertToASCII(charstr);
194 fd= open(charstr.GetPtrChar(), O_RDWR | O_NOCTTY | O_NONBLOCK);
195 if(fd<0) return false;
196
197 SetRTS(false);
198 SetDTR(false);
199
200 CleanBuffers();
201
202 //fcntl(fd, F_SETFL, FNDELAY);
203 //fcntl(fd, F_SETFL, 0);
204
205 if(!Config()) return false;
206
207 SetEvent(DIOANDROIDUARTFSMEVENT_CONNECTED);
208
209 status = DIOSTREAMSTATUS_CONNECTED;
210
211 ResetXBuffers();
212 ResetConnectionStatistics();
213
214 return threadconnection->Ini();
215}

229bool DIOANDROIDSTREAMUART::Config(XWORD mask)
230{
231 if(fd<0) return false;
232
233 struct termios cfg;
234 speed_t speed;
235
236 tcgetattr(fd, &cfg); // save current port settings
237
238 if((mask&DIOSTREAMUARTMASK_BAUDRATE) == DIOSTREAMUARTMASK_BAUDRATE)
239 {
240 switch(this->config->GetBaudRate())
241 {
242 case 50 : speed = B50; break;
243 case 300 : speed = B300; break;
244 case 600 : speed = B600; break;
245 case 1200 : speed = B1200; break;
246 case 4800 : speed = B4800; break;
247 case 9600 : speed = B9600; break;
248 case 19200 : speed = B19200; break;
249 case 38400 : speed = B38400; break;
250 case 57600 : speed = B57600; break;
251 case 115200 : speed = B115200; break;
252 }
253
254 cfsetispeed(&cfg, speed);
255 cfsetospeed(&cfg, speed);
256 }
257
258 if((mask&DIOSTREAMUARTMASK_DATABITS ) == DIOSTREAMUARTMASK_DATABITS)
259 {
260 cfg.c_cflag&= ~CSIZE;
261
262 switch(this->config->GetDataBits())
263 {
264 case DIOSTREAMUARTDATABIT_5 : cfg.c_cflag|= CS5; break;
265 case DIOSTREAMUARTDATABIT_6 : cfg.c_cflag|= CS6; break;
266 case DIOSTREAMUARTDATABIT_7 : cfg.c_cflag|= CS7; break;
267 case DIOSTREAMUARTDATABIT_8 : cfg.c_cflag|= CS8; break;
268 }
269 }
270
271 if((mask&DIOSTREAMUARTMASK_PARITY) == DIOSTREAMUARTMASK_PARITY)
272 {
273 cfg.c_iflag = (IGNBRK| IGNPAR);
274
275 cfg.c_cflag &= ~CMSPAR;
276 cfg.c_cflag &= ~PARENB;
277 cfg.c_cflag &= ~PARODD;
278
279 switch(this->config->GetParity())
280 {
281 case DIOSTREAMUARTPARITY_NONE : cfg.c_cflag &= ~CMSPAR;
282 cfg.c_cflag &= ~PARENB;
283 cfg.c_cflag &= ~PARODD;
284 break;
285
286 case DIOSTREAMUARTPARITY_MARK : cfg.c_cflag |= CMSPAR; // Not available on many systems.
287
288 case DIOSTREAMUARTPARITY_ODD : cfg.c_cflag |= PARENB;
289 cfg.c_cflag |= PARODD;
290 break;
291
292 case DIOSTREAMUARTPARITY_SPACE : cfg.c_cflag |= CMSPAR; // Not available on many systems.
293
294 case DIOSTREAMUARTPARITY_EVEN : cfg.c_cflag |= PARENB;
295 cfg.c_cflag &= ~PARODD;
296 break;
297 }
298 }
299
300 if((mask&DIOSTREAMUARTMASK_STOPBITS ) == DIOSTREAMUARTMASK_STOPBITS)
301 {
302 cfg.c_cflag &= ~CSTOPB;
303
304 switch(this->config->GetStopBits())
305 {
306 case DIOSTREAMUARTSTOPBITS_ONE : cfg.c_cflag &= ~CSTOPB; break;
307 case DIOSTREAMUARTSTOPBITS_ONEANDAHALF :
308 case DIOSTREAMUARTSTOPBITS_TWO : cfg.c_cflag |= CSTOPB; break;
309 }
310 }
311
312 if((mask&DIOSTREAMUARTMASK_FLOWCONTROL) == DIOSTREAMUARTMASK_FLOWCONTROL)
313 {
314 switch(config->GetFlowControl())
315 {
316 case DIOSTREAMUARTFLOWCONTROL_NONE : cfg.c_cflag &= ~CRTSCTS;
317 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
318 break;
319
320 case DIOSTREAMUARTFLOWCONTROL_HARD : cfg.c_cflag |= CRTSCTS;
321 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
322 break;
323
324 case DIOSTREAMUARTFLOWCONTROL_SOFT : cfg.c_cflag &= ~CRTSCTS;
325 cfg.c_iflag |= (IXON | IXOFF | IXANY);
326 break;
327 }
328 }
329
330 if((mask&DIOSTREAMUARTMASK_ALL) == DIOSTREAMUARTMASK_ALL)
331 {
332
333 cfg.c_cflag |= (CLOCAL | CREAD);
334
335 cfg.c_oflag &= ~OPOST;
336 cfg.c_lflag &= ~(ICANON | ECHO | ISIG);
337
338
339 cfg.c_cc[VMIN] = 0; // 0 means use-vtime
340 cfg.c_cc[VTIME] = 1; // time to wait until
341 }
342
343 tcsetattr(fd, TCSANOW, &cfg);
344
345 return true;
346}

643void DIOANDROIDSTREAMUART::ThreadConnection(void* param)
644{
645 DIOANDROIDSTREAMUART* diostream = (DIOANDROIDSTREAMUART*)param;
646 if(!diostream) return;
647
648 if(diostream->GetEvent()==DIOANDROIDUARTFSMEVENT_NONE) // No hay nuevos Eventos
649 {
650 switch(diostream->GetCurrentState())
651 {
652 case DIOANDROIDUARTFSMSTATE_NONE : break;
653
654 case DIOANDROIDUARTFSMSTATE_CONNECTED : break;
655
656 case DIOANDROIDUARTFSMSTATE_WAITINGTOREAD : {
657 if(!diostream->IsBlockRead())
658 {
659 XBYTE buffer[DIOSTREAM_MAXBUFFER];
660
661 XDWORD br = diostream->ReadDirect(buffer, DIOSTREAM_MAXBUFFER);
662 if(br)
663 {
664 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(2), (XBYTE*)diostream->buffer, (int)br);
665 diostream->inbuffer->Add(buffer, br);
666 }
667 }
668
669 if(!diostream->IsBlockWrite())
670 {
671 int esize = diostream->outbuffer->GetSize();
672 if(esize)
673 {
674 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
675
676 diostream->outbuffer->SetBlocked(true);
677 XDWORD bw = diostream->WriteDirect(diostream->outbuffer->Get(), esize);
678 diostream->outbuffer->SetBlocked(false);
679
680 if(bw)
681 {
682 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(4), (XBYTE*)diostream->buffer, (int)bw);
683 diostream->outbuffer->Extract(NULL, 0, bw);
684 }
685 }
686 }
687 }
688 break;
689
690 case DIOANDROIDUARTFSMSTATE_DISCONNECTING : break;
691
692 }
693 }
694 else
695 {
696 if(diostream->GetEvent()<DIOANDROIDUART_LASTEVENT)
697 {
698 diostream->CheckTransition();
699
700 switch(diostream->GetCurrentState())
701 {
702 case DIOANDROIDUARTFSMSTATE_NONE : break;
703
704 case DIOANDROIDUARTFSMSTATE_CONNECTED : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_CONNECTED);
705 xevent.SetDIOStream(diostream);
706 diostream->PostEvent(&xevent);
707
708 diostream->SetEvent(DIOANDROIDUARTFSMEVENT_WAITINGTOREAD);
709 }
710 break;
711
712 case DIOANDROIDUARTFSMSTATE_WAITINGTOREAD : break;
713
714 case DIOANDROIDUARTFSMSTATE_DISCONNECTING : { DIOSTREAM_XEVENT xevent(diostream,DIOSTREAM_XEVENT_TYPE_DISCONNECTED);
715 xevent.SetDIOStream(diostream);
716 diostream->PostEvent(&xevent);
717
718 diostream->threadconnection->Run(false);
719 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
720 }
721 break;
722 }
723 }
724 }
725}
