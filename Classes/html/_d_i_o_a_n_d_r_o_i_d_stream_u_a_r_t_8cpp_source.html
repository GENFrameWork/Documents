169bool DIOANDROIDSTREAMUART::Open()
170{
171 if(!config) return false;
172
173 XSTRING ttyS;
174
175 if(!config->GetLocalDeviceName()->IsEmpty())
176 {
177 ttyS = config->GetLocalDeviceName()->Get();
178 }
179 else
180 {
181 ttyS.Format(__L("/dev/ttyS%d"),config->GetPort());
182 }
183
184 XBUFFER charstr;
185
186 ttyS.ConvertToASCII(charstr);
187 fd= open(charstr.GetPtrChar(), O_RDWR | O_NOCTTY | O_NONBLOCK);
188 if(fd<0) return false;
189
190 SetRTS(false);
191 SetDTR(false);
192
193 CleanBuffers();
194
195 //fcntl(fd, F_SETFL, FNDELAY);
196 //fcntl(fd, F_SETFL, 0);
197
198 if(!Config()) return false;
199
200 SetEvent(DIOANDROIDUARTFSMEVENT_CONNECTED);
201
202 status = DIOSTREAMSTATUS_CONNECTED;
203
204 ResetXBuffers();
205 ResetConnectionStatistics();
206
207 return threadconnection->Ini();

222bool DIOANDROIDSTREAMUART::Config(XWORD mask)
223{
224 if(fd<0) return false;
225
226 struct termios cfg;
227 speed_t speed;
228
229 tcgetattr(fd, &cfg); // save current port settings
230
231 if((mask&DIOSTREAMUARTMASK_BAUDRATE) == DIOSTREAMUARTMASK_BAUDRATE)
232 {
233 switch(this->config->GetBaudRate())
234 {
235 case 50 : speed = B50; break;
236 case 300 : speed = B300; break;
237 case 600 : speed = B600; break;
238 case 1200 : speed = B1200; break;
239 case 4800 : speed = B4800; break;
240 case 9600 : speed = B9600; break;
241 case 19200 : speed = B19200; break;
242 case 38400 : speed = B38400; break;
243 case 57600 : speed = B57600; break;
244 case 115200 : speed = B115200; break;
245 }
246
247 cfsetispeed(&cfg, speed);
248 cfsetospeed(&cfg, speed);
249 }
250
251 if((mask&DIOSTREAMUARTMASK_DATABITS ) == DIOSTREAMUARTMASK_DATABITS)
252 {
253 cfg.c_cflag&= ~CSIZE;
254
255 switch(this->config->GetDataBits())
256 {
257 case DIOSTREAMUARTDATABIT_5 : cfg.c_cflag|= CS5; break;
258 case DIOSTREAMUARTDATABIT_6 : cfg.c_cflag|= CS6; break;
259 case DIOSTREAMUARTDATABIT_7 : cfg.c_cflag|= CS7; break;
260 case DIOSTREAMUARTDATABIT_8 : cfg.c_cflag|= CS8; break;
261 }
262 }
263
264 if((mask&DIOSTREAMUARTMASK_PARITY) == DIOSTREAMUARTMASK_PARITY)
265 {
266 cfg.c_iflag = (IGNBRK| IGNPAR);
267
268 cfg.c_cflag &= ~CMSPAR;
269 cfg.c_cflag &= ~PARENB;
270 cfg.c_cflag &= ~PARODD;
271
272 switch(this->config->GetParity())
273 {
274 case DIOSTREAMUARTPARITY_NONE : cfg.c_cflag &= ~CMSPAR;
275 cfg.c_cflag &= ~PARENB;
276 cfg.c_cflag &= ~PARODD;
277 break;
278
279 case DIOSTREAMUARTPARITY_MARK : cfg.c_cflag |= CMSPAR; // Not available on many systems.
280
281 case DIOSTREAMUARTPARITY_ODD : cfg.c_cflag |= PARENB;
282 cfg.c_cflag |= PARODD;
283 break;
284
285 case DIOSTREAMUARTPARITY_SPACE : cfg.c_cflag |= CMSPAR; // Not available on many systems.
286
287 case DIOSTREAMUARTPARITY_EVEN : cfg.c_cflag |= PARENB;
288 cfg.c_cflag &= ~PARODD;
289 break;
290 }
291 }
292
293 if((mask&DIOSTREAMUARTMASK_STOPBITS ) == DIOSTREAMUARTMASK_STOPBITS)
294 {
295 cfg.c_cflag &= ~CSTOPB;
296
297 switch(this->config->GetStopBits())
298 {
299 case DIOSTREAMUARTSTOPBITS_ONE : cfg.c_cflag &= ~CSTOPB; break;
300 case DIOSTREAMUARTSTOPBITS_ONEANDAHALF :
301 case DIOSTREAMUARTSTOPBITS_TWO : cfg.c_cflag |= CSTOPB; break;
302 }
303 }
304
305 if((mask&DIOSTREAMUARTMASK_FLOWCONTROL) == DIOSTREAMUARTMASK_FLOWCONTROL)
306 {
307 switch(config->GetFlowControl())
308 {
309 case DIOSTREAMUARTFLOWCONTROL_NONE : cfg.c_cflag &= ~CRTSCTS;
310 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
311 break;
312
313 case DIOSTREAMUARTFLOWCONTROL_HARD : cfg.c_cflag |= CRTSCTS;
314 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
315 break;
316
317 case DIOSTREAMUARTFLOWCONTROL_SOFT : cfg.c_cflag &= ~CRTSCTS;
318 cfg.c_iflag |= (IXON | IXOFF | IXANY);
319 break;
320 }
321 }
322
323 if((mask&DIOSTREAMUARTMASK_ALL) == DIOSTREAMUARTMASK_ALL)
324 {
325
326 cfg.c_cflag |= (CLOCAL | CREAD);
327
328 cfg.c_oflag &= ~OPOST;
329 cfg.c_lflag &= ~(ICANON | ECHO | ISIG);
330
331
332 cfg.c_cc[VMIN] = 0; // 0 means use-vtime
333 cfg.c_cc[VTIME] = 1; // time to wait until
334 }
335
336 tcsetattr(fd, TCSANOW, &cfg);
337
338 return true;
339}

636void DIOANDROIDSTREAMUART::ThreadConnection(void* param)
637{
638 DIOANDROIDSTREAMUART* diostream = (DIOANDROIDSTREAMUART*)param;
639 if(!diostream) return;
640
641 if(diostream->GetEvent()==DIOANDROIDUARTFSMEVENT_NONE) // No hay nuevos Eventos
642 {
643 switch(diostream->GetCurrentState())
644 {
645 case DIOANDROIDUARTFSMSTATE_NONE : break;
646
647 case DIOANDROIDUARTFSMSTATE_CONNECTED : break;
648
649 case DIOANDROIDUARTFSMSTATE_WAITINGTOREAD : {
650 if(!diostream->IsBlockRead())
651 {
652 XBYTE buffer[DIOSTREAM_MAXBUFFER];
653
654 XDWORD br = diostream->ReadDirect(buffer, DIOSTREAM_MAXBUFFER);
655 if(br)
656 {
657 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(2), (XBYTE*)diostream->buffer, (int)br);
658 diostream->inbuffer->Add(buffer, br);
659 }
660 }
661
662 if(!diostream->IsBlockWrite())
663 {
664 int esize = diostream->outbuffer->GetSize();
665 if(esize)
666 {
667 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
668
669 diostream->outbuffer->SetBlocked(true);
670 XDWORD bw = diostream->WriteDirect(diostream->outbuffer->Get(), esize);
671 diostream->outbuffer->SetBlocked(false);
672
673 if(bw)
674 {
675 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(4), (XBYTE*)diostream->buffer, (int)bw);
676 diostream->outbuffer->Extract(NULL, 0, bw);
677 }
678 }
679 }
680 }
681 break;
682
683 case DIOANDROIDUARTFSMSTATE_DISCONNECTING : break;
684
685 }
686 }
687 else
688 {
689 if(diostream->GetEvent()<DIOANDROIDUART_LASTEVENT)
690 {
691 diostream->CheckTransition();
692
693 switch(diostream->GetCurrentState())
694 {
695 case DIOANDROIDUARTFSMSTATE_NONE : break;
696
697 case DIOANDROIDUARTFSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
698 xevent.SetDIOStream(diostream);
699 diostream->PostEvent(&xevent);
700
701 diostream->SetEvent(DIOANDROIDUARTFSMEVENT_WAITINGTOREAD);
702 }
703 break;
704
705 case DIOANDROIDUARTFSMSTATE_WAITINGTOREAD : break;
706
707 case DIOANDROIDUARTFSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
708 xevent.SetDIOStream(diostream);
709 diostream->PostEvent(&xevent);
710
711 diostream->threadconnection->Run(false);
712 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
713 }
714 break;
715 }
716 }
717 }
718}
