111bool DIOSPIOLEDDISPLAYSSD1306::IniDevice()
112{
113 if(!DIODEVICESPI::IniDevice()) return false;
114
115 diostreamcfg->SetSPIMode(DIOSTREAMSPI_MODE_3);
116 diostreamcfg->SetNBitsWord(8);
117 diostreamcfg->SetSpeed(16*1000*1000);
118 diostreamcfg->SetDelay(0);
119 diostreamcfg->SetIsOnlyWrite(true);
120
121 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) != DIOGPIO_ID_NOTDEFINED)
122 {
123 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , DIOGPIO_MODE_OUTPUT);
124 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , true);
125 }
126
127 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) != DIOGPIO_ID_NOTDEFINED)
128 {
129 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , DIOGPIO_MODE_OUTPUT);
130 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , true);
131 }
132
133 if(diostream->Open())
134 {
135 if(diostream->WaitToConnected(timeout))
136 {
137 Reset();
138
139 if(!Initializate()) return false;
140
141 return DIODEVICE::Ini();
142 }
143 }
144
145 return false;
146}

160bool DIOSPIOLEDDISPLAYSSD1306::Clear(XBYTE color)
161{
162 XBYTE nblocks;
163
164 nblocks = (width >> 4) * (height >> 3);
165
166 // Set columns
167 SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COLUMN_ADDRESS_SET); // 0x21
168 SendCommand(0x00); // Start
169 SendCommand(width - 1); // End
170
171 //Set rows
172 SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_PAGE_ADDRESS_SET); // 0x22
173 SendCommand(0x00); // Start
174 SendCommand(height - 1); // End
175
176 XBYTE buffer[16];
177
178 memset(buffer, color, 16);
179
180 //Send buffer
181 for(XBYTE c=0; c<nblocks; c++)
182 {
183 SendData(buffer, 16);
184 }
185
186 return true;
187}

291bool DIOSPIOLEDDISPLAYSSD1306::Initializate()
292{
293 Reset();
294
295 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFF)) return false; // 0xAE
296 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_CLOCK_DIV_RATIO_SET)) return false; // 0xD5
297 if(!SendCommand(0xF0)) return false; // Suggest ratio
298 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_MULTIPLEX_RADIO_SET)) return false; // 0xA8
299 if(!SendCommand((64 - 1))) return false; // Display_Height - 1
300
301 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFFSET_SET)) return false; // 0xD3 + No offset
302 if(!SendCommand(00)) return false; // No offset
303 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_CHARGE_PUMP_SET)) return false; // 0x8D
304 if(!SendCommand(0x14)) return false; // Enable charge pump
305 if(!SendCommand((DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_START_LINE_SET | 0x00))) return false; // 0x40 | Start line
306 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_MEMORY_ADDRESSING_SET)) return false; // 0x20
307 if(!SendCommand(0x00)) return false; // Horizontal Addressing - Operate like KS0108
308 if(!SendCommand((DIOSPIOLEDDISPLAYSSD1306_COMMAND_SEGMENT_REMAP_SET | 0x01))) return false; // 0xA0 - Left towards Right
309 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COM_OUTPUT_SCAN_DEC)) return false; // 0xC8 - Up towards Down
310 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COM_PINS_SET)) return false; // 0xDA
311//if(!SendCommand(0x12)) return false; // Sequential COM pin configuration
312
313 if((width == 128) && (height == 64))
314 {
315 if(!SendCommand(0x12)) return false; // Sequential COM pin configuration
316 }
317 if((width == 128) && (height == 32))
318 {
319 if(!SendCommand(0x02)) return false; // Sequential COM pin configuration
320 }
321 if((width == 96) && (height == 16))
322 {
323 if(!SendCommand(0x02)) return false; // Sequential COM pin configuration
324 }
325
326 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_CONSTRAST_SET)) return false; // 0x81
327 if(!SendCommand(0xFF)) return false; // 0xFF
328 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_PRECHARGE_PERIOD_SET)) return false; // 0xD9
329 if(!SendCommand(0xF1)) return false; // 0xF1
330
331 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_VCOMH_DESELECT_LEVEL_SET)) return false; // 0xDB
332 if(!SendCommand(0x20)) return false; // 0x20
333
334 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_ALL_ON_RESUME)) return false; // 0xA4
335 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_NORMAL)) return false; // 0xA6
336 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_SCROLL_DEACTIVATE)) return false; // 0x2E
337
338//if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_ON)) return false; // 0xAF
339
340//Activate(false);
341
342 Clear(0x00);
343
344 Activate(true);
345
346 return true;
347}

359bool DIOSPIOLEDDISPLAYSSD1306::Reset()
360{

362 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) == DIOGPIO_ID_NOTDEFINED) return false;
363
364 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC), false);
365 GEN_XSLEEP.MilliSeconds(10);

368 GEN_XSLEEP.MilliSeconds(10);
369
370 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET), true);
371 GEN_XSLEEP.MilliSeconds(10);
372
373 return true;
