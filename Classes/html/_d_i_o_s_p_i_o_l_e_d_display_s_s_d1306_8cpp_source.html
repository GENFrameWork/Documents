118bool DIOSPIOLEDDISPLAYSSD1306::IniDevice()
119{
120 if(!DIODEVICESPI::IniDevice()) return false;
121
122 diostreamcfg->SetSPIMode(DIOSTREAMSPI_MODE_3);
123 diostreamcfg->SetNBitsWord(8);
124 diostreamcfg->SetSpeed(16*1000*1000);
125 diostreamcfg->SetDelay(0);
126 diostreamcfg->SetIsOnlyWrite(true);
127
128 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) != DIOGPIO_ID_NOTDEFINED)
129 {
130 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , DIOGPIO_MODE_OUTPUT);
131 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) , true);
132 }
133
134 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) != DIOGPIO_ID_NOTDEFINED)
135 {
136 GEN_DIOGPIO.SetMode(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , DIOGPIO_MODE_OUTPUT);
137 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) , true);
138 }
139
140 if(diostream->Open())
141 {
142 if(diostream->WaitToConnected(timeout))
143 {
144 Reset();
145
146 if(!Initializate()) return false;
147
148 return DIODEVICE::Ini();
149 }
150 }
151
152 return false;
153}

167bool DIOSPIOLEDDISPLAYSSD1306::Clear(XBYTE color)
168{
169 XBYTE nblocks;
170
171 nblocks = (width >> 4) * (height >> 3);
172
173 // Set columns
174 SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COLUMN_ADDRESS_SET); // 0x21
175 SendCommand(0x00); // Start
176 SendCommand(width - 1); // End
177
178 //Set rows
179 SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_PAGE_ADDRESS_SET); // 0x22
180 SendCommand(0x00); // Start
181 SendCommand(height - 1); // End
182
183 XBYTE buffer[16];
184
185 memset(buffer, color, 16);
186
187 //Send buffer
188 for(XBYTE c=0; c<nblocks; c++)
189 {
190 SendData(buffer, 16);
191 }
192
193 return true;
194}

298bool DIOSPIOLEDDISPLAYSSD1306::Initializate()
299{
300 Reset();
301
302 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFF)) return false; // 0xAE
303 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_CLOCK_DIV_RATIO_SET)) return false; // 0xD5
304 if(!SendCommand(0xF0)) return false; // Suggest ratio
305 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_MULTIPLEX_RADIO_SET)) return false; // 0xA8
306 if(!SendCommand((64 - 1))) return false; // Display_Height - 1
307
308 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_OFFSET_SET)) return false; // 0xD3 + No offset
309 if(!SendCommand(00)) return false; // No offset
310 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_CHARGE_PUMP_SET)) return false; // 0x8D
311 if(!SendCommand(0x14)) return false; // Enable charge pump
312 if(!SendCommand((DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_START_LINE_SET | 0x00))) return false; // 0x40 | Start line
313 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_MEMORY_ADDRESSING_SET)) return false; // 0x20
314 if(!SendCommand(0x00)) return false; // Horizontal Addressing - Operate like KS0108
315 if(!SendCommand((DIOSPIOLEDDISPLAYSSD1306_COMMAND_SEGMENT_REMAP_SET | 0x01))) return false; // 0xA0 - Left towards Right
316 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COM_OUTPUT_SCAN_DEC)) return false; // 0xC8 - Up towards Down
317 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_COM_PINS_SET)) return false; // 0xDA
318//if(!SendCommand(0x12)) return false; // Sequential COM pin configuration
319
320 if((width == 128) && (height == 64))
321 {
322 if(!SendCommand(0x12)) return false; // Sequential COM pin configuration
323 }
324 if((width == 128) && (height == 32))
325 {
326 if(!SendCommand(0x02)) return false; // Sequential COM pin configuration
327 }
328 if((width == 96) && (height == 16))
329 {
330 if(!SendCommand(0x02)) return false; // Sequential COM pin configuration
331 }
332
333 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_CONSTRAST_SET)) return false; // 0x81
334 if(!SendCommand(0xFF)) return false; // 0xFF
335 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_PRECHARGE_PERIOD_SET)) return false; // 0xD9
336 if(!SendCommand(0xF1)) return false; // 0xF1
337
338 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_VCOMH_DESELECT_LEVEL_SET)) return false; // 0xDB
339 if(!SendCommand(0x20)) return false; // 0x20
340
341 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_ALL_ON_RESUME)) return false; // 0xA4
342 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_NORMAL)) return false; // 0xA6
343 if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_SCROLL_DEACTIVATE)) return false; // 0x2E
344
345//if(!SendCommand(DIOSPIOLEDDISPLAYSSD1306_COMMAND_DISPLAY_ON)) return false; // 0xAF
346
347//Activate(false);
348
349 Clear(0x00);
350
351 Activate(true);
352
353 return true;
354}

366bool DIOSPIOLEDDISPLAYSSD1306::Reset()
367{
368 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET) == DIOGPIO_ID_NOTDEFINED) return false;
369 if(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC) == DIOGPIO_ID_NOTDEFINED) return false;
370
371 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_DC), false);
372 GEN_XSLEEP.MilliSeconds(10);
373
374 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET), false);
375 GEN_XSLEEP.MilliSeconds(10);
376
377 GEN_DIOGPIO.SetValue(GetGPIOEntryID(DIODISPLAYDEVICE_INDEX_GPIOENTRYID_RESET), true);
378 GEN_XSLEEP.MilliSeconds(10);
379
380 return true;
381}
