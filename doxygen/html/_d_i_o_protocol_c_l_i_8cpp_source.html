413bool DIOPROTOCOLCLI::SendCommandArg(XCHAR* command, XSTRING* target, XSTRING* answer, int timeoutanswer, va_list* arg)
414{
415 DIOPROTOCOLCLICOMMAND* protocolcommand = GetCommand(command);
416 if(!protocolcommand) return false;
417
418 if(!arg) return false;
419
420 XSTRING tosend;
421
422 if(answer)
423 {
424 answer->Empty();
425 }
426 else
427 {
428 tosend.Format(__L("%c"), DIOPROTOCOLCLI_MARK_DONOTANSWER);
429 }
430
431 if(!ID.IsEmpty())
432 {
433 tosend.AddFormat(__L("%s%s"), ID.Get(), DIOPROTOCOLCLI_MARK_ORIGIN);
434 }
435
436 if(target)
437 {
438 if(!target->IsEmpty())
439 {
440 tosend.AddFormat(__L("%s%s"), target->Get(), DIOPROTOCOLCLI_MARK_TARGET);
441 }
442 }
443 else
444 {
445 tosend.AddFormat(__L("%s%s"), DIOPROTOCOLCLI_MARK_BROADCAST, DIOPROTOCOLCLI_MARK_TARGET);
446 }
447
448 tosend.AddFormat(__L("%s"), protocolcommand->GetCommand());
449
450 int nparams = protocolcommand->GetNParams();
451 if(nparams)
452 {
453 tosend.AddFormat(__L(" "));
454
455 for(int c=0;c<nparams;c++)
456 {
457 XCHAR* param = (XCHAR*)va_arg((*arg), XCHAR*);
458 if(param)
459 {
460 tosend += param;
461 if(c != nparams-1) tosend += __L(",");
462 }
463 }
464 }
465
466 if(activeCRC)
467 {
468 HASHCRC32 CRC32;
469
470 XBUFFER charstr;
471
472 tosend.ConvertToASCII(charstr);
473 CRC32.Do((XBYTE*)charstr.Get(), tosend.GetSize());
474
475 XDWORD CRC32result = CRC32.GetResultCRC32();
476 tosend.AddFormat(__L("%s%08X"), DIOPROTOCOLCLI_MARK_CRC32, CRC32result);
477 }
478
479 tosend += __L("\n\r");
480
481
482 XBUFFER charstr;
483
484 tosend.ConvertToASCII(charstr);
485 diostream->Write((XBYTE*)charstr.Get(), tosend.GetSize());
486
487 tosend.DeleteCharacter(__C('\n'));
488 tosend.DeleteCharacter(__C('\r'));
489
490 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Protocol CLI] %s Send Ask: %s -> %s %s"), tosend.Get(), ID.Get(), (target?target->Get():DIOPROTOCOLCLI_MARK_BROADCAST), command);
491
492 bool status = true;
493
494 if(answer)
495 {
496 if(xtimerout) xtimerout->Reset();
497
498 while(1)
499 {
500 if(!xtimerout)
501 {
502 status = false;
503 break;
504 }
505
506 if(exitproccess)
507 {
508 status = false;
509 break;
510 }
511
512 if(xtimerout->GetMeasureSeconds() >= (XDWORD)timeoutanswer)
513 {
514 status = false;
515 break;
516 }
517
518 DIOPROTOCOLCLIANSWER* protocolanswer = GetFirstAnswer(command);
519 if(protocolanswer)
520 {
521 answer->Add(protocolanswer->GetAnswer()->Get());
522 DeleteFirstAnswer(command);
523 break;
524 }
525 }
526 }
527
528 return status;
529}
