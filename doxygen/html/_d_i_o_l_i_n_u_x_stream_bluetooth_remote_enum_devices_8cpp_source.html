290int DIOLINUXSTREAMBLUETOOTHREMOTEENUMDEVICES::IsReadyConnect(int socket)
291{
292 struct timeval tv;
293 int rc;
294 fd_set fdr;
295 fd_set fdw;
296 fd_set fds;
297
298 if(socket==-1) return -1;
299
300 FD_ZERO(&fdr);
301 FD_ZERO(&fdw);
302 FD_ZERO(&fds);
303
304 FD_SET((unsigned int)socket,&fdr);
305 FD_SET((unsigned int)socket,&fdw);
306 FD_SET((unsigned int)socket,&fds);
307
308 tv.tv_sec = 0;
309 tv.tv_usec = 100;
310
311 rc = select(socket+1, &fdr, &fdw, &fds, &tv);
312 if(rc==-1) return -1;
313 if(rc== 2) return -1;
314
315 int status1 = FD_ISSET(socket,&fdr) ? 1 : 0;
316 int status2 = FD_ISSET(socket,&fdw) ? 1 : 0;
317 int status3 = FD_ISSET(socket,&fds) ? 1 : 0;
318
319 if(status3) return -1;
320 if(((status1) || (status2)) && (rc==1)) return 1;
321
322 return 0;
323}

496{

499 bdaddr_t origin;
500 bdaddr_t target;

513 XSTRING MACstring;

516 device->GetMAC()->GetXString(MACstring);

518
519 MACstring.ConvertToASCII(charstr);

521
522 // ----------------------------------------------------------------------
523 // Special version Blocking
524 session = sdp_connect(&origin, &target, SDP_RETRY_IF_BUSY);
525 if(!session) return false;
526
527 attrid = sdp_list_append(0, &range);
528 search = sdp_list_append(0, &context->group);

534 }

536 sdp_list_free(attrid, 0);
537 sdp_list_free(search, 0);

579 device->Service_Add(service);

581 }
