234XDWORD DIOSTREAMUDPACKNOWLEDGE::Write(XBYTE* buffer, XDWORD size)
235{
236 if(!diostream) return 0;
237 if(!size) return 0;
238
239 if(addressremote.IsEmpty()) return 0;
240
241 DIOSTREAMUDP* diostreamudp = (DIOSTREAMUDP*)diostream;
242 XDWORD bw = 0;
243
244 if(diostreamudp->WriteDatagram(addressremote, portremote, buffer, size)) bw = size;
245
246 if(diostreamudp->WaitToWriteDatagramsEmpty(timeout))
247 {
248 //XTRACE_PRINTCOLOR(1, __L("Write UDP %s:%d : %s"), addressremote.Get(), portremote, bw==size?__L("Ok"):__L("ERROR!"));
249
250 if(diostreamudp->WaitToGetFirstDatagram(false, timeout) != DIOSTREAMUDP_NOTFOUND)
251 {
252 XSTRING address;
253 XWORD port = 0;
254 XBUFFER xbuffer;
255 XDWORD hashresult[2] = { 0, 0 };
256
257 diostreamudp->ReadDatagram(address, port, xbuffer);
258
259 xbuffer.Extract(hashresult[0]);
260
261 HASHCRC32 hashcrc32;
262
263 hashcrc32.ResetResult();
264 hashcrc32.Do(buffer, size);
265
266 hashresult[1] = hashcrc32.GetResultCRC32();
267
268 if(hashresult[0] != hashresult[1])
269 {
270 //XTRACE_PRINTCOLOR(1, __L("Write received bad CRC!! %08X <> %08X"), hashresult, hashcrc32.GetResultCRC32());
271 bw = 0;
272 }
273 }
274 else
275 {
276 //XTRACE_PRINTCOLOR(1, __L("Write received NOT read"));
277 bw = 0;
278 }
279 }
280 else
281 {
282 //XTRACE_PRINTCOLOR(1, __L("Do not write"));
283 bw = 0;
284 }
285
286 if(!bw)
287 {
288 diostream->Close();
289 diostream->Open();
290 diostream->WaitToConnected(5);
291 }
292
293 return bw;
294}

436void DIOSTREAMUDPACKNOWLEDGE::ThreadRunFunction(void* param)
437{
438 DIOSTREAMUDPACKNOWLEDGE* diostreamudpacknowledge = (DIOSTREAMUDPACKNOWLEDGE*)param;
439 if(!diostreamudpacknowledge) return;
440
441 DIOSTREAMUDP* diostreamudp = diostreamudpacknowledge->diostream;
442 if(!diostreamudp) return;
443
444 XBUFFER data;
445 XSTRING address;
446 XWORD port;
447
448 if(diostreamudp->ReadDatagram(address, port, data))
449 {
450 HASHCRC32 hashcrc32;
451
452 hashcrc32.ResetResult();
453 if(hashcrc32.Do(data.Get(), data.GetSize()))
454 {
455 XBUFFER result;
456 XDWORD hashresult = hashcrc32.GetResultCRC32();
457
458 result.Add(hashresult);
459
460 diostreamudp->WriteDatagram(address, port, result);
461 if(diostreamudp->WaitToWriteDatagramsEmpty(diostreamudpacknowledge->timeout))
462 {
463 diostreamudpacknowledge->AddDatagram(false, address, port, data.Get(), data.GetSize());
464 }
465 }
466 }
467}
