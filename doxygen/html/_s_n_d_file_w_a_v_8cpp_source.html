138{
139 fileRIFF = new XFILERIFF();
140 if(!fileRIFF) return false;
141
142 if(!fileRIFF->Open(path)) return false;
143
144 if(!fileRIFF->ReadAllLists()) return false;
145
146 format_list = fileRIFF->GetChunk(__L("fmt "), __L("WAVE"));

148
149 fileRIFF->GetFileBase()->SetPosition(format_list->GetPositionFileData());

151 if(!fileRIFF->GetFileBase()->Read((XBYTE*)&format.audio_format , sizeof(XWORD))) return false;

189bool SNDFILEWAV::ReadData(XBYTE* data, XDWORD& sizeread, bool& isfinished)
190{
191 if(!fileRIFF) return false;
192 if(!data_list) return false;
193 if(!dataoffsetinfile) return false;
194
195 XDWORD _sizeread = sizeread;
196
197 if(datasizeread == datasizeinfile)
198 {
199 datasizeread = 0;
200 }
201
202 isfinished = false;
203
204 bool status = fileRIFF->GetFileBase()->Read(data, &_sizeread);
205 if(!_sizeread && !status) return false;
206
207 datasizeread += _sizeread;
208
209 if((sizeread != _sizeread) || (datasizeread >= datasizeinfile))
210 {
211 dataoffsetinfile = data_list->GetPositionFileData();
212 fileRIFF->GetFileBase()->SetPosition(dataoffsetinfile);
213
214 isfinished = true;
215
216 sizeread = _sizeread;
217 }
218
219 return true;
220}
