250bool DIOSTREAMTLS::HandShake_Client_Hello()
251{
252 XBUFFER xbuffer;
253 XWORD size16 = 0;
254 XBYTE size24[3] = { 0, 0, 0 };
255 bool status = false;
256
257 //-------------------------------------------------------------------
258 // TLS Plaintext
259
260 xbuffer.Add((XBYTE)DIOSTREAMTLS_CONTENTTYPE_HANDSHAKE); // Content Type
261 xbuffer.Add((XWORD)SwapWORD(DIOSTREAMTLS_VERSION_TLS_1_0)); // legacy record version
262 xbuffer.Add((XWORD)SwapWORD(size16)); // Length
263
264 //-------------------------------------------------------------------
265 // HandSake protot
266 xbuffer.Add((XBYTE)DIOSTREAMTLS_HANDSHAKETYPE_CLIENT_HELLO); // Handshake type: Client Hello (1)
267 xbuffer.Add(size24, 3);
268 xbuffer.Add((XWORD)SwapWORD(DIOSTREAMTLS_VERSION_TLS_1_2)); // Version TLS 1.2
269 xbuffer.Add(random, DIOSTREAMTLS_RANDOM_SIZE); // Random
270 xbuffer.Add((XBYTE)DIOSTREAMTLS_SESSION_ID_SIZE); // Session ID length
271 xbuffer.Add(sessionID, DIOSTREAMTLS_SESSION_ID_SIZE); // Session ID
272
273 XDWORD size = Write(xbuffer.Get(), xbuffer.GetSize());
274 if(size == xbuffer.GetSize()) status = true;
275
276 return status;
277}
