326void XLINUXDATETIME::GetActualDateTime(XLINUXDATETIME* timed, bool islocal)
327{
328 time_t now;
329 struct tm* datatime;
330
331 now = time(0);
332
333 if(islocal)
334 {
335 datatime = localtime(&now);
336 }
337 else
338 {
339 datatime = gmtime(&now);
340 }
341
342 if(!datatime) return;
343
344 timed->year = datatime->tm_year + 1900;
345 timed->month = datatime->tm_mon+1;
346 timed->day = datatime->tm_mday;

354 timed->milliseconds = (tv.tv_usec /1000);
