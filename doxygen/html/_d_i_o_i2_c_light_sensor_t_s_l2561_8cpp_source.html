164bool DIOI2CLIGHTSENSORTSL2561::Get(XWORD& fullspectrum, XWORD& infrared, XDWORD& lux, bool autogain)
165{
166 fullspectrum = 0;
167 infrared = 0;
168 lux = 0;
169
170 if(!diostream) return false;
171
172 XWORD fs, ir, hi, lo;
173 bool status = true;
174
175 if(!autogain)
176 {
177 // autogain not requested. Executing a single sensor read
178
179 status = GetData(fullspectrum, infrared);
180 lux = CalculateLux(fullspectrum, infrared);
181
182 return status;
183 }
184
185 // autogain requested
186 switch(integrationtime)
187 {
188 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_13MS;
189 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_13MS;
190 break;
191
192 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_101MS;
193 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_101MS;
194 break;
195
196 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_402MS;
197 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_402MS;
198 break;
199
200 default : hi = DIOI2CLIGHTSENSORTSL2561_AGC_THI_402MS;
201 lo = DIOI2CLIGHTSENSORTSL2561_AGC_TLO_402MS;
202 break;
203 }
204
205 // try to adjust the gain
206 status = GetData(fs, ir);
207 if(!status) return false;
208
209 if((fs < lo) && (gain == DIOI2CLIGHTSENSORTSL2561GAIN_1X))
210 {
211 // light too low with this gain
212 // raise the gain and redo the reading
213 SetGain(DIOI2CLIGHTSENSORTSL2561GAIN_16X);
214
215 status = GetData(fs, ir);
216 if(!status) return false;
217
218 // now consider the reading valid after being adjusted
219 fullspectrum = fs;
220 infrared = ir;
221
222 lux = CalculateLux(fullspectrum, infrared);
223
224 return true;
225 }
226
227
228 if((fs > hi) && (gain == DIOI2CLIGHTSENSORTSL2561GAIN_16X))
229 {
230 // light too high with this gain
231 // lower the gain and redo the reading
232 SetGain(DIOI2CLIGHTSENSORTSL2561GAIN_1X);
233
234 status = GetData(fs, ir);
235 if(!status) return false;
236
237 // now consider the reading valid after being adjusted
238 fullspectrum = fs;
239 infrared = ir;
240 lux = CalculateLux(fullspectrum, infrared);
241
242 return true;
243 }
244
245
246 // the reading was valid without gain adjustment (or chip limits encountered!)
247 fullspectrum = fs;
248 infrared = ir;
249
250 lux = CalculateLux(fullspectrum, infrared);
251
252 return true;
253}

348bool DIOI2CLIGHTSENSORTSL2561::ReadData16(XBYTE reg, XWORD& value, int timeout)
349{
350 XBUFFER xbuffer;
351
352 value = 0;
353
354 xbuffer.Add((XBYTE)reg);
355 if(!diostream->Write(xbuffer)) return false;
356 if(!diostream->WaitToFlushOutXBuffer(timeout)) return false;
357
358 xbuffer.Resize(2);
359 if(!diostream->WaitToFilledReadingBuffer(2, timeout)) return false;
360 if(!diostream->Read(xbuffer)) return false;
361 if(xbuffer.GetSize() < 2) return false;
362
363 value = xbuffer.GetByte(1);
364 value <<= 8;
365 value |= xbuffer.GetByte(0);
366
367 //XTRACE_PRINTCOLOR(XDEBUG_COLORGREEN, __L("%02X %02X"), xbuffer.GetByte(1), xbuffer.GetByte(0));
368
369 return true;
370}

403 if(status) status = ReadData16((DIOI2CLIGHTSENSORTSL2561_COMMAND_BIT | DIOI2CLIGHTSENSORTSL2561_WORD_BIT | DIOI2CLIGHTSENSORTSL2561REGISTER_CHAN1_LOW), infrared , timeout);

453XDWORD DIOI2CLIGHTSENSORTSL2561::CalculateLux(XWORD broadband, XWORD inflared)
454{
455 XDWORD chscale;
456 XDWORD channel1;
457 XDWORD channel0;
458 XDWORD clipthreshold;
459 XDWORD ratio1 = 0;
460 int ratio;
461 XDWORD b, m;
462 int temp;
463 XDWORD lux;
464
465 // Make sure the sensor isn't saturated!
466 switch(integrationtime)
467 {
468 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_13MS; break;
469 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_101MS; break;
470 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_402MS; break;
471 default : clipthreshold = DIOI2CLIGHTSENSORTSL2561_CLIPPING_402MS; break;
472 }
473
474 // return 0 lux if the sensor is saturated
475 if((broadband > clipthreshold) || (inflared > clipthreshold)) return 0;
476
477
478 // get the correct scale depending on the intergration time
479 switch(integrationtime)
480 {
481 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_13MS : chscale = DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE_TINT0; break;
482 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_101MS : chscale = DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE_TINT1; break;
483 case DIOI2CLIGHTSENSORTSL2561INTEGRATIONTIME_402MS : chscale = (1 << DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE); break;
484 default : chscale = (1 << DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE); break; // No scaling ... integration time = 402ms
485 }
486
487 // scale for gain (1x or 16x)
488 if (!gain) chscale = chscale << 4;
489
490 // scale the channel values
491 channel0 = (broadband * chscale) >> DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE;
492 channel1 = (inflared * chscale) >> DIOI2CLIGHTSENSORTSL2561_LUX_CHSCALE;
493
494 // find the ratio of the channel values (Channel1/Channel0)
495 if (channel0 != 0) ratio1 = (channel1 << (DIOI2CLIGHTSENSORTSL2561_LUX_RATIOSCALE+1)) / channel0;
496
497 // round the ratio value
498 ratio = (ratio1 + 1) >> 1;
499
500 #ifdef TSL2561_PACKAGE_CS
501 if ((ratio >= 0) && (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K1C))
502 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B1C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M1C; }
503 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K2C)
504 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B2C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M2C; }
505 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K3C)
506 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B3C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M3C; }
507 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K4C)
508 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B4C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M4C; }
509 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K5C)
510 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B5C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M5C; }
511 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K6C)
512 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B6C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M6C; }
513 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K7C)
514 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B7C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M7C; }
515 else if (ratio > DIOI2CLIGHTSENSORTSL2561_LUX_K8C)
516 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B8C; m = DIOI2CLIGHTSENSORTSL2561_LUX_M8C; }
517 #else
518 if ((ratio >= 0) && (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K1T))
519 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B1T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M1T; }
520 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K2T)
521 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B2T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M2T; }
522 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K3T)
523 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B3T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M3T; }
524 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K4T)
525 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B4T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M4T; }
526 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K5T)
527 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B5T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M5T; }
528 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K6T)
529 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B6T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M6T; }
530 else if (ratio <= DIOI2CLIGHTSENSORTSL2561_LUX_K7T)
531 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B7T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M7T; }
532 else if (ratio > DIOI2CLIGHTSENSORTSL2561_LUX_K8T)
533 { b = DIOI2CLIGHTSENSORTSL2561_LUX_B8T; m = DIOI2CLIGHTSENSORTSL2561_LUX_M8T; }
534 #endif
535
536 temp = ((channel0 * b) - (channel1 * m));
537
538 // do not allow negative lux value
539 if (temp < 0) temp = 0;
540
541 // round lsb (2^(LUX_SCALE-1))
542 temp += (1 << (DIOI2CLIGHTSENSORTSL2561_LUX_LUXSCALE-1));
543
544 // strip off fractional portion
545 lux = (temp >> DIOI2CLIGHTSENSORTSL2561_LUX_LUXSCALE);
546
547 // Signal I2C had no errors
548 return lux;
549}
