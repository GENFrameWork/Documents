111bool DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES::Search()
112{
113 bool status = false;
114
115 if(threadenumdevices) return false;
116 if(threadenumdevicesevent) return false;
117
118
119 DelAllDevices();
120
121 memset(&hcidevinfo, 0x00, sizeof(hcidevinfo));
122
123 DIOMAC* MAC = GetDeviceSelect();
124 if(MAC)
125 {
126 if(!MAC->IsZero())
127 {
128 XSTRING MACstr;
129
130 MAC->GetXString(MACstr);
131
132 XBUFFER charstr;
133
134 MACstr.ConvertToASCII(charstr);
135 hcideviceID = hci_devid(charstr.GetPtrChar());
136
137 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Blueooth LE enum Remote] Select local bluetooth: %s"), MACstr.Get());
138 }
139 else
140 {
141 hcideviceID = hci_get_route(NULL);
142 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Blueooth LE enum Remote] Default local bluetooth."));
143 }
144 }
145 else
146 {
147 hcideviceID = hci_get_route(NULL);
148 }
149
150 if(hcideviceID < 0) hcideviceID = 0; // use device 0, if device id is invalid
151
152 //XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[LE Remote Search] start device [%d]..."), hcideviceID);
153
154 hcisocket = hci_open_dev(hcideviceID);
155 if(hcisocket == -1)
156 {
157 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Blueooth LE enum Remote] error to hci_copen_dev. "));
158 return false;
159 }
160
161 hcidevinfo.dev_id = hcideviceID;
162
163 // ----------------------------------------------------------------------------------------------
164 // Get old HCI filter
165
166 hci_filter_clear(&old_hcifilter);
167 socklen_t sizefilter = sizeof(old_hcifilter);
168 getsockopt(hcisocket, SOL_HCI, HCI_FILTER, &old_hcifilter, &sizefilter);
169
170
171 // ----------------------------------------------------------------------------------------------
172 // Setup new HCI filter
173
174 hci_filter_clear(&new_hcifilter);
175 hci_filter_set_ptype(HCI_EVENT_PKT , &new_hcifilter);
176 hci_filter_set_event(EVT_LE_META_EVENT , &new_hcifilter);
177
178 setsockopt(hcisocket, SOL_HCI, HCI_FILTER, &new_hcifilter, sizeof(new_hcifilter));
179
180 // ----------------------------------------------------------------------------------------------
181 // Activate search
182
183 hci_le_set_scan_enable(hcisocket, 0x00, 0, 1000);
184 hci_le_set_scan_parameters(hcisocket, 0x00, htobs(0x0080), htobs(0x0080), 0x00, 0x00, 1000);
185 //hci_le_set_scan_parameters(hcisocket, 0x00, htobs(0x0080), htobs(0x0080), 0x00, 0x00, 1000);
186 //hci_le_set_scan_parameters(hcisocket, 0x00, htobs(0x3E8), htobs(0x3E8), 0x00, 0x00, 100);
187 hci_le_set_scan_enable(hcisocket, 0x01, 0, 1000);
188
189
190 GEN_XFACTORY_CREATE(xmutexdevicesevent, Create_Mutex())
191 if(!xmutexdevicesevent) return false;
192
193 threadenumdevices = CREATEXTHREAD(XTHREADGROUPID_DIOSTREAMBLUETOOTH, __L("DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES::DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES"), ThreadEnumDevices, (void*)this);
194 if(!threadenumdevices) return false;

198 status = threadenumdevices->Ini();
199
200 if(GetSearchMode() & DIOSTREAMBLUETOOTHLEREMOTEENUMDEVICES_SEARCHMODE_EVENT)
201 {
202 threadenumdevicesevent = CREATEXTHREAD(XTHREADGROUPID_DIOSTREAMBLUETOOTH, __L("DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES::DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES_EVENT"), ThreadEnumDevicesEvent, (void*)this);
203 if(!threadenumdevicesevent) return false;
204
205 //threadenumdevicesevent->SetPriority(XTHREADPRIORITY_HIGH);
206
207 status = threadenumdevicesevent->Ini();
208 }
209
210 return status;

225bool DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES::StopSearch(bool waitend)
226{
227
228 if(threadenumdevicesevent)
229 {
230 threadenumdevicesevent->End();
231 DELETEXTHREAD(XTHREADGROUPID_DIOSTREAMBLUETOOTH, threadenumdevicesevent);
232
233 threadenumdevicesevent = NULL;
234 }
235
236 if(threadenumdevices)
237 {
238 threadenumdevices->End();
239 DELETEXTHREAD(XTHREADGROUPID_DIOSTREAMBLUETOOTH, threadenumdevices);
240
241 threadenumdevices = NULL;
242 }
243
244 if(xmutexdevicesevent)
245 {
246 GEN_XFACTORY.Delete_Mutex(xmutexdevicesevent);
247 xmutexdevicesevent = NULL;
248 }
249
250
251 // disable LE scan
252 hci_le_set_scan_enable(hcisocket, 0x00, 0, 1000);
253
254 // restore original filter
255 setsockopt(hcisocket, SOL_HCI, HCI_FILTER, &old_hcifilter, sizeof(old_hcifilter));
256
257 close(hcisocket);
258
259 hcisocket = -1;
260
261 // XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[LE Remote Search] stop...."));
262
263 return true;
264}

296{
297 DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES* LEremotedevices = (DIOLINUXSTREAMBLUETOOTHLEREMOTEENUMDEVICES*)param;

300 XBYTE hcieventbuffer[HCI_MAX_EVENT_SIZE];

305
306 hcieventsize = read(LEremotedevices->hcisocket, hcieventbuffer, sizeof(hcieventbuffer));

308 LEmetaevent = (evt_le_meta_event *)(hcieventbuffer + (1 + HCI_EVENT_HDR_SIZE));

311 if(LEmetaevent->subevent == EVT_LE_ADVERTISING_REPORT)

321 LEMACdevicestring.Set(LEMACdevicestr);

359 device_event->GetMAC()->Set(LEMACdevicestring);
