161XWORD DIOATCMDGSM_ALPHABET_8859_1::Traslate(bool from, XWORD data)
162{
163 XDWORD end = (sizeof(table)/sizeof(char[2]));
164 char extended = ((data&0xFF00)>>8);
165 char character = (data&0x00FF);
166 char escape = 0x1B;
167 XDWORD c;
168
169 if(from)
170 {
171 for(c=0;c<127;c++)
172 {
173 if(table[c][1]==character) return (XWORD)table[c][0];
174 }
175
176 for(c=128;c<end;c++)
177 {
178 if(table[c][1]==character) return (XWORD)((escape<<8)|table[c][0]);
179 }
180 }
181 else
182 {
183 int first =0;
184
185 if(extended==escape)
186 first = 128;
187 else end = 127;
188
189 for(c=first;c<end;c++)
190 {
191 if(table[c][0]==character) return (XWORD)table[c][1];
192 }
193 }
194
195 return 0xFF00;
196}

881bool DIOATCMDGSM::CodecPDUFormat(XCHAR* number, XCHAR* message, bool inoctets, int validperiod, XSTRING& length, XSTRING& result)
882{
883 if(!number) return false;
884 if(!message) return false;
885
886 DIOATCMDGSM_ALPHABET_8859_1 alphabet;
887 XSTRING string;
888 XSTRING _number;
889 XSTRING _message;
890 int value;
891 XBYTE* codec7 = new XBYTE[256];
892 XBYTE* codec8 = new XBYTE[256];
893 int size7 = 0;
894 int size8 = 0;
895 int c;
896
897 if((!codec7) || (!codec8)) return false;
898
899 memset(codec7 ,0 , 256);
900 memset(codec8 ,0 , 256);
901
902 result.Empty();
903
904 _number = number;
905 _message = message;
906
907 // No SMS Center.
908 result += __L("00");
909 // SMS-SUBMIT mensaje.
910 result += __L("11");
911 result += __L("00");
912
913 // Size Number (National)
914 string.Format(__L("%02X"),_number.GetSize());
915 result += string;
916
917 if(_number.GetSize()>9)
918 result += __L("91"); // Type Number (International);
919 else result += __L("81"); // Type Number (National);
920
921 // GSM Number
922 string = number;
923 if(string.GetSize()%2) string += __L("F");
924 string.Swab();
925 result += string;
926 result += __L("00");
927
928 // Codec 7/8 Bits
929 if(inoctets)
930 result += __L("02");
931 else result += __L("00");
932
933 // Time Max Msg <28 days >28 weeks
934 if(validperiod>28)
935 value = 168 + validperiod;
936 else value = 197 + validperiod;
937 string.Format(__L("%02X"),value);
938 result += string;
939
940 // Convert message
941 for(c=0;c<(int)_message.GetSize();c++)
942 {
943 XWORD data=alphabet.Traslate(false,(XBYTE)message[c]);
944 XBYTE byte[2];
945
946 byte[1]=(XBYTE)(data>>8);
947 byte[0]=(XBYTE)(data&0x00FF);
948
949 if(byte[1])
950 {
951 codec8[size8]=byte[1];
952 size8++;
953 }
954
955 codec8[size8]=byte[0];
956 size8++;
957 }
958
959
960 // Compress 7 Bits.
961 if(!inoctets)
962 {
963 int shift=0;
964
965 for(c=0;c<size8;c++)
966 {
967 XBYTE mask = (0xFF>>(7-shift));
968 XBYTE data = (codec8[c+1]&mask)<<(XBYTE)(7-shift);
969
970 if(shift!=7)
971 {
972 codec7[size7]=(data|(codec8[c]>>(XBYTE)shift));
973 size7++;
974 }
975
976 shift++;
977 if(shift>7) shift=0;
978 }
979 }
980
981 string.Format(__L("%02X"),_message.GetSize());
982 result += string;
983
984 for(c=0;c<((inoctets)?size8:size7);c++)
985 {
986 string.Format(__L("%02X"),(inoctets)?codec8[c]:codec7[c]);
987 result += string;
988 }
989
990 delete [] codec7;
991 delete [] codec8;
992
993 length.Format(__L("%d"),(result.GetSize()/2)-1);
