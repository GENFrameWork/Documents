276bool XLICENSE::GenerateMachineID(XLICENSEID& licenseID)
277{
278 DIOSTREAMIPLOCALENUMDEVICES* enumdevices = (DIOSTREAMIPLOCALENUMDEVICES*)GEN_DIOFACTORY.CreateStreamEnumDevices(DIOSTREAMENUMTYPE_IP_LOCAL);
279 if(enumdevices)
280 {
281 enumdevices->Search();
282
283 XQWORD basicID = 0;
284 XSTRING basicstring;
285 XBUFFER basicxbuffer;
286
287 for(int c=0;c<(int)enumdevices->GetDevices()->GetSize();c++)
288 {
289 DIOSTREAMDEVICEIP* device = (DIOSTREAMDEVICEIP*)enumdevices->GetDevices()->Get(c);
290 if(device)
291 {
292 DIOMAC* mac = device->GetMAC();
293 if(mac)
294 {
295 XQWORD macID = 0;
296
297 for(int d=0; d<DIOMAC_MAXSIZE; d++)
298 {
299 macID |= mac->Get()[d];
300 macID<<=8;
301 }
302
303 basicID ^= macID;
304 }
305 }
306 }
307
308 GEN_DIOFACTORY.DeleteStreamEnumDevices(enumdevices);
309
310 basicstring.Format(__L("%02X%02X%llX%llX%02X%02X"), XLICENSE_VERSION, XLICENSE_SUBVERSION, basicID, ~basicID, XLICENSE_VERSION, XLICENSE_SUBVERSION);
311
312 HASHSHA1* sha1 = new HASHSHA1();
313 if(sha1)
314 {
315 basicstring.ConvertHexStringToBuffer(basicxbuffer);
316 sha1->Do(basicxbuffer);
317
318 XDWORD index = 0;
319 XDWORD partID = 0;
320 int sizeofpart = sizeof(XDWORD)/sizeof(XBYTE);
321
322 for(int c=0; c<XLICENSE_MAXIDPARTS; c++)
323 {
324 partID = 0;
325 for(int d=0; d<sizeofpart; d++)
326 {
327 partID |= (XBYTE)sha1->GetResult()->GetByte(index++);
328 if(d!=(sizeofpart-1)) partID<<=8;
329 if(index>=sha1->GetResult()->GetSize()) index = 0;
330 }
331
332 licenseID.SetPart(c, partID);
333 }
334
335 delete sha1;
336
337 return true;
338 }
339 }
340
341 return false;
342}

358bool XLICENSE::Generate(XLICENSEID& licenseID, XSTRING* applicationID, XBUFFER* license)
359{
360 CIPHERAES cipherAES;
361 CIPHERKEYSYMMETRICAL key;
362 XBUFFER xbufferkey;
363 XBUFFER xbuffercipher;
364 XBUFFER* _license = license;
365 XSTRING* _applicationID = applicationID;
366
367 if(!_license) _license = this->license;
368 if(!_license) return false;
369
370 if(!_applicationID) _applicationID = &this->applicationID;
371 if(!_applicationID) return false;
372
373 _license->Delete();
374 licenseID.GetXBuffer(xbuffercipher);
375
376 GetBufferKeyFromMachineID((*_applicationID), xbufferkey);
377
378 key.Set(xbufferkey);
379
380 cipherAES.SetKey(&key);
381 cipherAES.SetChainingMode(CIPHERCHAININGMODE_CBC);
382 cipherAES.SetPaddingType(XBUFFER_PADDINGTYPE_NONE);
383
384 if(cipherAES.Cipher(xbuffercipher))
385 {
386 _license->Delete();
387 _license->Add(cipherAES.GetResult());
388
389 return true;
390 }
391
392 return false;
393}

408bool XLICENSE::GetBufferKeyFromMachineID(XSTRING& applicationID, XBUFFER& xbufferkey)
409{
410 HASHSHA1 sha1;
411
412 xbufferkey.Delete();
413
414
415 XBUFFER charstr;
416
417 applicationID.ConvertToASCII(charstr);
418 xbufferkey.Add((XBYTE*)charstr.Get(), applicationID.GetSize());
419
420 sha1.Do(xbufferkey);
421
422 xbufferkey.Delete();
423 xbufferkey.Add(sha1.GetResult());
424
425 xbufferkey.Resize(16);
426
427 return xbufferkey.GetSize()?true:false;

444bool XLICENSE::CipherExpirationDate(bool cipher, XSTRING& applicationID, XSTRING& expirationdate)
445{
446 XBUFFER xbufferkey;
447 CIPHERAES cipherAES;
448 CIPHERKEYSYMMETRICAL key;
449 bool status = false;
450
451 if(!GetBufferKeyFromMachineID(applicationID, xbufferkey)) return false;
452
453 key.Set(xbufferkey);
454
455 cipherAES.SetKey(&key);
456 cipherAES.SetChainingMode(CIPHERCHAININGMODE_CBC);
457 cipherAES.SetPaddingType(XBUFFER_PADDINGTYPE_SPACES);
458
459 if(cipher)
460 {
461 XBUFFER charstr;
462
463 expirationdate.ConvertToASCII(charstr);
464
465 status = cipherAES.Cipher((XBYTE*)charstr.Get(), expirationdate.GetSize());
466 if(status)
467 {
468 cipherAES.GetResult()->ConvertToBase64(expirationdate);
469 return true;
470 }
471 }
472 else
473 {
474 XBUFFER xbuffer;
475
476 xbuffer.ConvertFromBase64(expirationdate);
477
478 status = cipherAES.Uncipher(xbuffer);
479 if(status)
480 {
481 expirationdate.Empty();
482
483 for(XDWORD c=0; c<cipherAES.GetResult()->GetSize(); c++)
484 {
485 expirationdate.Add((XCHAR)cipherAES.GetResult()->GetByte(c));
486 }
487
488 expirationdate.DeleteCharacter(0x20, XSTRINGCONTEXT_TO_END);
489
490 return true;
491 }
492 }
493
494 return status;
495}

661bool XLICENSE::LoadFromBuffer(XBUFFER& xbuffer, XSTRING& applicationID, XBUFFER* license, XSTRING* expirationdate)
662{
663 if(!license) return false;
664
665 XFILEINI fileini;
666 XLICENSEID licenseID;
667 XSTRING licenseIDstring;
668 XSTRING licenseIDstring2;
669 XSTRING licensestring;
670 XSTRING licenseexprirationstring;
671 XBUFFER* _license = license;
672
673 if(!_license) _license = this->license;
674 if(!_license) return false;
675
676 fileini.AddBufferLines(XFILETXTFORMATCHAR_ASCII, xbuffer);
677
678 fileini.ConvertFromLines();
679
680 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_ID , licenseIDstring);
681 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE , licensestring);
682 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_EXPIRATION , licenseexprirationstring);
683
684 GenerateMachineID(licenseID);
685 licenseID.GetXString(licenseIDstring2);
686
687 if(licenseIDstring.Compare(licenseIDstring2)) return false;
688
689 CipherExpirationDate(false, applicationID, licenseexprirationstring);
690
691 if(expirationdate) (*expirationdate) = licenseexprirationstring;
692
693 licensestring.ConvertHexStringToBuffer((*_license));
694
695 return true;

713bool XLICENSE::CheckMasterCreation(XPATH& xpath, XLICENSEID& licenseID, XSTRING& applicationID, int expirationseconds)
714{
715 XFILEINI fileini;
716 XSTRING licensestring;
717
718 if(!fileini.Open(xpath)) return false;
719 fileini.ReadValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE, licensestring);
720 fileini.Close();
721
722 if(!applicationID.Compare(licensestring))
723 {
724 XFILEINI fileini;
725 XSTRING licenseIDstring;
726 XSTRING licensestring;
727 XSTRING licenseexpiration;
728 XBUFFER licencia;
729 XDATETIME* expirationdatetime;
730
731 licenseID.GetXString(licenseIDstring);
732 if(!Generate(licenseID, &applicationID, &licencia)) return false;
733
734 licensestring.ConvertHexStringFromBuffer(licencia);
735
736 GEN_XFACTORY_CREATE(expirationdatetime, CreateDateTime())
737 if(!expirationdatetime) return false;
738
739 expirationdatetime->SetToZero();
740 if(expirationseconds)
741 {
742 expirationdatetime->Read();
743 expirationdatetime->AddSeconds((int)expirationseconds);
744 }
745
746 expirationdatetime->GetDateTimeToString(XDATETIME_FORMAT_STANDARD, licenseexpiration);
747 GEN_XFACTORY.DeleteDateTime(expirationdatetime);
748
749 if(!CipherExpirationDate(true, applicationID, licenseexpiration)) return false;
750
751 if(!fileini.Create(xpath)) return false;
752
753 fileini.CreateSection(XLICENSE_FILESECTION);
754 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_ID , licenseIDstring);
755 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_LICENSE , licensestring);
756 fileini.WriteValue(XLICENSE_FILESECTION, XLICENSE_FILEVALUE_EXPIRATION, licenseexpiration);
757
758 fileini.Close();
759
760 return true;
761 }
762
763 return false;
764}
