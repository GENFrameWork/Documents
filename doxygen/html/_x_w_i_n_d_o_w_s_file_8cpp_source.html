473FILE* XWINDOWSFILE::CreateStructHandle()
474{
475 if(fp != NULL) return fp;
476
477 switch(mode)
478 {
479 case XWINDOWSFILEMODE_READONLY : fd = _open_osfhandle((intptr_t)filehandle, _O_RDONLY);

527bool XWINDOWSFILE::ActualizeSize()
528{
529 if(!isopen) return false;
530 if(!filehandle) return false;
531
532 if(!isreadonly)
533 {
534 if(!Flush()) return false;
535 }
536
537 LARGE_INTEGER exsize;
538 if(GetFileSizeEx(filehandle, &exsize) == INVALID_FILE_SIZE) return false;
539
540 cachesize = exsize.HighPart;
541 cachesize <<=32;
542 cachesize |= exsize.LowPart;
543
544 return true;
