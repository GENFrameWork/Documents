663bool GRPSCREEN::CreateViewport(XCHAR* ID, float posx, float posy, float width, float height, XDWORD canvasposx, XDWORD canvasposy, XDWORD canvaswidth, XDWORD canvasheight)
664{
665 GRPVIEWPORT* viewport = new GRPVIEWPORT();
666 if(!viewport) return false;
667
668 viewport->GetID()->Set(ID);
669 viewport->SetPosition(posx, posy);
670 viewport->SetSize(width, height);
671
672 GRPPROPERTIES canvasproperties;
673
674 canvasproperties.CopyPropertysFrom(this);
675
676 canvasproperties.SetPosition(canvasposx, canvasposy);
677
678 float cwidth = (float)canvaswidth;
679 float cheight = (float)canvasheight;
680
681 if(width > cwidth) cwidth = width;
682 if(height > cheight) cheight = height;
683
684 canvasproperties.SetSize((XDWORD)cwidth, (XDWORD)cheight);
685
686 if(viewport->CreateCanvas(canvasproperties))
687 {
688 viewport->GetCanvas()->GetScreenZone()->Set((int)posx, (int)posy, (int)(posx + width), (int)(posy + height));
689 viewports.Add(viewport);
690
691 // --------------------------------------------------------------------------------------------------
692 // Main Canvas is selected if the name is the name of the main viewport or there is only one viewport.
693
694 if((!viewport->GetID()->Compare(GRPVIEWPORT_ID_MAIN,true)) || viewports.GetSize() == 1)
695 {
696 index_maincanvas = viewports.GetSize()-1;
697 maincanvas = viewport->GetCanvas();
698 }
699
700 } else return false;
701
702 return true;
703 }
