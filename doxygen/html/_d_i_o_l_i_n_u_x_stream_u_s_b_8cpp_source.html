257void DIOLINUXSTREAMUSB::ThreadConnection(void* data)
258{
259 DIOLINUXSTREAMUSB* diostream = (DIOLINUXSTREAMUSB*)data;
260 if(!diostream) return;
261
262 if(diostream->GetEvent()==DIOLINUXUSBFSMEVENT_NONE) // No hay nuevos Eventos
263 {
264 switch(diostream->GetCurrentState())
265 {
266 case DIOLINUXUSBFSMSTATE_NONE : break;
267
268 case DIOLINUXUSBFSMSTATE_CONNECTED : break;
269
270 case DIOLINUXUSBFSMSTATE_WAITINGTOREAD : {
271 if(!diostream->IsBlockRead())
272 {
273 XBYTE buffer[DIOSTREAM_MAXBUFFER];
274
275 XDWORD br = diostream->ReadBuffer(buffer, DIOSTREAM_MAXBUFFER);
276 if(br)
277 {
278 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(2), (XBYTE*)diostream->buffer, (int)br);
279 diostream->inbuffer->Add(buffer, br);
280 }
281 }
282
283 if(!diostream->IsBlockWrite())
284 {
285 int esize = diostream->outbuffer->GetSize();
286 if(esize)
287 {
288 if(esize > DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
289
290 diostream->outbuffer->SetBlocked(true);
291 XDWORD bw = diostream->WriteBuffer(diostream->outbuffer->Get(), esize);
292 diostream->outbuffer->SetBlocked(false);
293
294 if(bw)
295 {
296 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(4), (XBYTE*)diostream->buffer, (int)bw);
297 diostream->outbuffer->Extract(NULL , 0, bw);
298 }
299 else
300 {
301 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(3), (XBYTE*)diostream->buffer, (int)bw);
302 }
303 }
304 }
305 }
306 break;
307
308
309 case DIOLINUXUSBFSMSTATE_SENDINGDATA : break;
310
311 case DIOLINUXUSBFSMSTATE_DISCONNECTING : break;
312
313 }
314 }
315 else
316 {
317 if(diostream->GetEvent()<DIOLINUXUSB_LASTEVENT)
318 {
319 diostream->CheckTransition();
320
321 switch(diostream->GetCurrentState())
322 {
323 case DIOLINUXUSBFSMSTATE_NONE : break;
324
325 case DIOLINUXUSBFSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
326 xevent.SetDIOStream(diostream);
327 diostream->PostEvent(&xevent);
328
329 diostream->SetEvent(DIOLINUXUSBFSMEVENT_WAITINGTOREAD);
330 }
331 break;
332
333 case DIOLINUXUSBFSMSTATE_WAITINGTOREAD : break;
334
335 case DIOLINUXUSBFSMSTATE_SENDINGDATA : break;
336
337 case DIOLINUXUSBFSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
338 xevent.SetDIOStream(diostream);
339 diostream->PostEvent(&xevent);
340
341 diostream->threadconnection->Run(false);
342 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
343 }
344 break;
345 }
346 }
347 }
348}

363XDWORD DIOLINUXSTREAMUSB::ReadBuffer(XBYTE* buffer,XDWORD size)
364{
365 if(GetConnectStatus()==DIOSTREAMSTATUS_DISCONNECTED) return 0;
366
367 fd_set fds;
368 struct timeval tv;
369
370 tv.tv_sec = 0;
371 tv.tv_usec = readtimeout;
372
373 FD_ZERO(&fds);
374 FD_SET(fd, &fds);
375
376 int br = 0;
377
378 if(select(fd+1, &fds, NULL, NULL, &tv) > 0)
379 {
380 br = read(fd, buffer, size);
381
382 if(br< 0)
383 {
384 return 0;
385 }
386 }
387
388 return (XDWORD)br;
389}
