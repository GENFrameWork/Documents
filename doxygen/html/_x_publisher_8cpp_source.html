444bool XPUBLISHER::PostEvent(XEVENT* event, XSUBJECT* subject)
445{
446 if(!on) return false;
447
448 inevent = true;
449
450 if(xmutexevents) xmutexevents->Lock();
451
452 XVECTOR<XOBSERVER*> observers;
453
454 for(XDWORD c=0; c<eventsvector.GetSize(); c++)
455 {
456 XPUBLISHERENTRY* publisherentry = eventsvector.Get(c);
457 if(publisherentry)
458 {
459 if((publisherentry->type == event->GetEventType()) && (publisherentry->subject == subject))
460 {
461 observers.Add(publisherentry->observer);
462 }
463 }
464 }
465
466 if(xmutexevents) xmutexevents->UnLock();
467
468 for(XDWORD c=0; c<observers.GetSize(); c++)
469 {
470 XOBSERVER* observer = observers.Get(c);
471 if(observer) observer->HandleEvent(event);
472 }
473
474 observers.DeleteAll();
475
476
477 inevent = false;
478
479 return true;
480}
