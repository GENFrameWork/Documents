100bool DIOPIFACE::Configure()
101{
102 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IOCON , 8); // Enable hardware addressing
103 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_GPIOA , 0x00); // Turn on port A
104 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IODIRA , 0); // Set port A as an output
105 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_IODIRB , 0xFF); // Set port B as an input
106 Write_Register(DIOSPIGPIOMCP23S17_WRITE_CMD, DIOSPIGPIOMCP23S17_GPPUB , 0xFF); // Turn on port B pullups
107
108 return true;
109}

123bool DIOPIFACE::ReadInputs(XBYTE& value)
124{
125 XBYTE _value[3] = { 0xFF, 0xFF, 0xFF };
126
127 if(!diostream) return false;
128
129 diostream->GetInXBuffer()->Delete();
130
131 if(!Write_Register(DIOSPIGPIOMCP23S17_READ_CMD, DIOPIFACE_INPUTPORT, 0xFF)) return false;
132
133 if(!diostream->WaitToFilledReadingBuffer(3, timeout)) return false;
134
135 bool status = diostream->Read(_value, 3)?true:false;
136
137 value = _value[2] ^ 0xFF;
138
139 return status;
140}
