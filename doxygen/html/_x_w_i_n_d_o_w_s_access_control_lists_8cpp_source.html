103bool XWINDOWSACCESSCONTROLLISTS::SetFileOnlyPermissionForAdmin(XCHAR* filename)
104{
105 #define EXPLICIT_ACCESS_MAXNUM_ADMIN 2
106
107 PSID everyoneSID = NULL;
108 PSID adminSID = NULL;
109 PACL ACL = NULL;
110 EXPLICIT_ACCESS ea[EXPLICIT_ACCESS_MAXNUM_ADMIN];
111 SID_IDENTIFIER_AUTHORITY SIDauthworld = SECURITY_WORLD_SID_AUTHORITY;
112 SID_IDENTIFIER_AUTHORITY SIDAuthNT = SECURITY_NT_AUTHORITY;
113
114 // Create a well-known SID for the Everyone group.
115 AllocateAndInitializeSid(&SIDauthworld , 1
116 , SECURITY_WORLD_RID
117 , 0, 0, 0, 0, 0, 0, 0
118 , &everyoneSID);
119
120
121 AllocateAndInitializeSid(&SIDAuthNT , 2
122 , SECURITY_BUILTIN_DOMAIN_RID
123 , DOMAIN_ALIAS_RID_ADMINS
124 , 0, 0, 0, 0, 0, 0
125 , &adminSID);
126
127
128 // Initialize an EXPLICIT_ACCESS structure for an ACE.
129 ZeroMemory(&ea, EXPLICIT_ACCESS_MAXNUM_ADMIN * sizeof(EXPLICIT_ACCESS));
130
131 // Set read access for Everyone.
132 ea[0].grfAccessPermissions = GENERIC_READ;
133 ea[0].grfAccessMode = SET_ACCESS;
134 ea[0].grfInheritance = NO_INHERITANCE;
135 ea[0].Trustee.TrusteeForm = TRUSTEE_IS_SID;
136 ea[0].Trustee.TrusteeType = TRUSTEE_IS_WELL_KNOWN_GROUP;
137 ea[0].Trustee.ptstrName = (LPTSTR) everyoneSID;
138
139 // Set full control for Administrators.
140 ea[1].grfAccessPermissions = GENERIC_ALL;
141 ea[1].grfAccessMode = SET_ACCESS;
142 ea[1].grfInheritance = NO_INHERITANCE;
143 ea[1].Trustee.TrusteeForm = TRUSTEE_IS_SID;
144 ea[1].Trustee.TrusteeType = TRUSTEE_IS_GROUP;
145 ea[1].Trustee.ptstrName = (LPTSTR) adminSID;
146
147 // Create a new ACL that contains the new ACEs.
148 SetEntriesInAcl(EXPLICIT_ACCESS_MAXNUM_ADMIN, ea, NULL, &ACL);
149
150 // Initialize a security descriptor.
151 PSECURITY_DESCRIPTOR SD = (PSECURITY_DESCRIPTOR) LocalAlloc(LPTR, SECURITY_DESCRIPTOR_MIN_LENGTH);
152
153 InitializeSecurityDescriptor(SD, SECURITY_DESCRIPTOR_REVISION);
154
155 // Add the ACL to the security descriptor.
156 SetSecurityDescriptorDacl(SD, TRUE // bDaclPresent flag
157 , ACL
158 , FALSE); // not a default DACL
159
160
161 // Change the security attributes
162 BOOL status = SetFileSecurity(filename, DACL_SECURITY_INFORMATION, SD);
163
164 if(everyoneSID) FreeSid(everyoneSID);
165 if(adminSID) FreeSid(adminSID);
166 if(ACL) LocalFree(ACL);
167 if(SD) LocalFree(SD);
168
169 return status?true:false;
170}
