260bool MAINPROC::CreateParams(XCHAR* commandline)
261{
262 XSTRING _commandline = commandline;
263
264 if(_commandline.IsEmpty()) return false;
265
266 int start = 0;
267 XDWORD c = 0;
268
269 do{ if((_commandline.Get()[c] == __C(' ')) || (_commandline.GetSize() == c))
270 {
271 XSTRING* param = new XSTRING();
272 if(param)
273 {
274 _commandline.Copy(start, c, (*param));
275
276 GetExecParams()->Add(param);
277 start = c+1;
278 }
279 }
280
281 c++;
282
283 } while(c <= _commandline.GetSize());
284
285 return true;
286}
