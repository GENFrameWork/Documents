136bool INPLINUXDEVICEMOUSEDIRECT::Update()
137{
138 if((!created)||(!enabled)) return false;
139
140 int eventsize = sizeof(struct input_event);
141 int bytesread = 0;
142
143 struct input_event event;
144
145 static int delta = 0;
146 static int hdelta = 0;
147
148 //--------------- clean up old states
149
150 for(XDWORD c=0; c<buttons.GetSize(); c++)
151 {
152 INPBUTTON* button=buttons.Get(c);
153 if(button)
154 {
155 if(button->GetState() == INPBUTTON_STATE_RELEASED)
156 {
157 button->SetState(INPBUTTON_STATE_UP);
158 }
159 else
160 {
161 if(button->GetState() == INPBUTTON_STATE_PRESSED) button->SetState(INPBUTTON_STATE_HOLD);
162 }
163 }
164 }
165
166 //--------------- process
167
168 if(mice.GetSize() == 0)
169 {
170 OpenAllDevicesID();
171 }
172
173 for(XDWORD e=0; e<mice.GetSize(); e++)
174 {
175 while(true)
176 {
177 bytesread=0;
178
179 memset(&event, 0, sizeof(event));
180
181 errno = 0;
182 bytesread = read(mice.Get(e)->GetFileDescriptor(), &event, sizeof(event)/*eventsize*/);
183
184 // if device is diconnected we receive a ENODEV message (happens for example every time a touchscreen is poweredoff)
185
186 if(bytesread == -1 && errno == ENODEV)
187 {
188 // we lost the handle to the event file, need to reclaim it
189 mice.DeleteContents();
190 OpenAllDevicesID();
191 }
192
193 if(bytesread==-1 && errno!=EAGAIN) break;
194
195 if((bytesread < sizeof(struct input_event)) || (bytesread == -1)) break;
196
197 if(bytesread == sizeof(event)/*eventsize*/)
198 {
199 switch(event.type)
200 {
201 case EV_ABS : switch(event.code)
202 {
203 case ABS_X : { mousex = event.value;
204 float x = ((float)mousex / (float)maxx);
205
206 mousex = (int)(x * grpscreen->GetWidth());
207 }
208 break;
209
210 case ABS_Y : { mousey = event.value;
211 float y = ((float)mousey / (float)maxy);
212
213 mousey = (int)(y * grpscreen->GetHeight());
214 mousey = (grpscreen->GetHeight()-mousey);
215
216 }
217 break;
218
219 case ABS_Z : mousez = event.value; break;
220 case REL_HWHEEL : hdelta = event.value; break;
221 case REL_WHEEL : delta = event.value; break;
222 }
223 break;
224
225 case EV_REL : switch(event.code)
226 {
227 case REL_X : mousex += event.value; break;
228 case REL_Y : mousey -= event.value; break;
229 case REL_Z : mousez += event.value; break;
230 case REL_HWHEEL : hdelta += event.value; break;
231 case REL_WHEEL : delta += event.value; break;
232 }
233 break;
234
235
236 case EV_KEY : { INPBUTTON* button = GetButtonByCode(event.code);
237 if(button)
238 {
239 switch(button->GetState())
240 {
241 case INPBUTTON_STATE_UP : (event.value==0) ? button->SetState(INPBUTTON_STATE_UP) : button->SetState(INPBUTTON_STATE_PRESSED); break;
242 case INPBUTTON_STATE_PRESSED : (event.value==0) ? button->SetState(INPBUTTON_STATE_RELEASED) : button->SetState(INPBUTTON_STATE_HOLD); break;
243 case INPBUTTON_STATE_RELEASED :
244 case INPBUTTON_STATE_HOLD : (event.value==0) ? button->SetState(INPBUTTON_STATE_RELEASED) : button->SetState(INPBUTTON_STATE_PRESSED); break;
245 }
246
247 button->SetPressed((event.value==1)?true:false);
248 }
249
250 // emulate mouse button
251 if(event.code == BTN_TOUCH)
252 {
253 button = GetButtonByCode(BTN_LEFT);
254 if(button)
255 {
256 switch(button->GetState())
257 {
258 case INPBUTTON_STATE_UP : (event.value==0) ? button->SetState(INPBUTTON_STATE_UP) : button->SetState(INPBUTTON_STATE_PRESSED); break;
259 case INPBUTTON_STATE_PRESSED : (event.value==0) ? button->SetState(INPBUTTON_STATE_RELEASED) : button->SetState(INPBUTTON_STATE_HOLD); break;
260 case INPBUTTON_STATE_RELEASED :
261 case INPBUTTON_STATE_HOLD : (event.value==0) ? button->SetState(INPBUTTON_STATE_RELEASED) : button->SetState(INPBUTTON_STATE_PRESSED); break;
262 }
263
264 button->SetPressed((event.value==1)?true:false);
265 }
266 }
267 }
268 break;
269
270 #ifdef EV_SYN
271 case EV_SYN : // UpdateMouseData();
272 break;
273 #endif
274
275 default : continue;
276 }
277
278 } else break;
279 }
280 }
281
282 UpdateMouseData();
283
284 return true;
285}

341bool INPLINUXDEVICEMOUSEDIRECT::OpenAllDevicesID()
342{
343 INPLINUXFACTORY* inpdevices = (INPLINUXFACTORY*)(&INPFACTORY::GetInstance());
344 XVECTOR<INPLINUXDEVICEID*> devicesID;
345
346 if(inpdevices) inpdevices->GetDeviceHandlers(INPDEVICE_TYPE_MOUSE, devicesID);
347
348 for(XDWORD c=0; c<devicesID.GetSize(); c++)
349 {
350 INPLINUXDEVICEID* deviceID = devicesID.Get(c);
351 if(deviceID)
352 {
353 XSTRING event;
354 int filedescriptor = INPLINUXDEVICEID_INVALID;
355
356 event.Format(__L("/dev/input/event%d"), deviceID->GetEventIndex());
357
358 XBUFFER charstr;
359
360 event.ConvertToASCII(charstr);
361 filedescriptor = open(charstr.GetPtrChar(), O_RDONLY | O_NONBLOCK);
362
363 if(filedescriptor != -1)
364 {
365 char mouseOEMname[_MAXSTR];
366 XSTRING namehardware;
367
368 // we get the keyboard name
369 memset(mouseOEMname, 0, _MAXSTR);
370
371 ioctl(filedescriptor, EVIOCGNAME(_MAXSTR), mouseOEMname);
372 namehardware.Set(mouseOEMname);
373
374 deviceID->SetFileDescriptor(filedescriptor);
375 //deviceID->GetName()->Set(namehardware);
376
377
378 // get max/min values on each axis
379 int abs[6] = {0};
380 ioctl(filedescriptor, EVIOCGABS(ABS_X), abs);
381 minx = abs[1];
382 maxx = abs[2];
383
384 ioctl(filedescriptor, EVIOCGABS(ABS_Y), abs);
385 miny = abs[1];
386 maxy = abs[2];
387
388 ioctl(filedescriptor, EVIOCGABS(ABS_Z), abs);
389 minz = abs[1];
390 maxz = abs[2];
391
392 mice.Add(deviceID);
393 }
394 }
395 }
396
397 devicesID.DeleteAll();
398
399 return mice.GetSize()?true:false;
400}

438bool INPLINUXDEVICEMOUSEDIRECT::UpdateMouseData()
439{
440 if(!grpscreen) return false;
441
442 for(int e=0;e<GetCursors()->GetSize();e++)
443 {
444 INPCURSOR* cursor = GetCursors()->Get(e);
445 if(!cursor) continue;
446
447 cursor->SetIsChanged(false);
448
449 if((cursor->GetX() != mousex) || (cursor->GetY() != mousey))
450 {
451 cursor->Set(mousex, mousey);
452 cursor->SetIsChanged(true);
453 }
454 }
455
456 INPBUTTON* button = GetButton(INPBUTTON_ID_MOUSE_LEFT);
457 if(button)
458 {
459 if(GetCursors()->Get(0)) GetCursors()->Get(0)->AddPointToMotion(button->IsPressed());
460 }
461
462 return true;
463}
