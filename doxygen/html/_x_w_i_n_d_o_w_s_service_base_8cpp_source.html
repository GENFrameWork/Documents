172void XWINDOWSSERVICEBASE::Stop()
173{
174 DWORD originalstate = status.dwCurrentState;
175
176 try {
177 SetServiceStatus(SERVICE_STOP_PENDING);
178 On_Stop();
179 SetServiceStatus(SERVICE_STOPPED);
180 }
181 catch(DWORD error)
182 {
183 WriteErrorLogEntry(__L("Service Stop"), error);
184 SetServiceStatus(originalstate);
185 }
186 catch(...)
187 {
188 WriteEventLogEntry(__L("Service failed to stop."), EVENTLOG_ERROR_TYPE);
189 SetServiceStatus(originalstate);
190 }

670{
671 SC_HANDLE scmanager = OpenSCManager(0, 0, SC_MANAGER_ALL_ACCESS);
672 if(scmanager == NULL) return false;
673
674 SC_HANDLE service = OpenService(scmanager, servicename, SERVICE_ALL_ACCESS);
675 if(service == NULL) return false;
676
677 DWORD sizeneeded;
678 DWORD buffersize;
679 DWORD error;
680 bool status = false;
681
682 if(service_config) LocalFree(service_config);
683
684 if(!QueryServiceConfig(service, NULL, 0, &sizeneeded))
685 {
686 error = GetLastError();
687 if(error == ERROR_INSUFFICIENT_BUFFER)
688 {
689 buffersize = sizeneeded;
690 service_config = (LPQUERY_SERVICE_CONFIG) LocalAlloc(LMEM_FIXED, buffersize);
691
692 status = true;
693 }
694 }
695
696 if(status)
697 {
698 status = false;
699
700 if(QueryServiceConfig(service, service_config, buffersize, &sizeneeded))
701 {
702 status = true;
703 }
704 }
705
706 CloseServiceHandle(scmanager);
707
708 return status;

956void WINAPI XWINDOWSSERVICEBASE::ServiceCtrlHandler(DWORD control, DWORD eventtype, LPVOID eventdata, LPVOID context)
957{
958 switch(control)
959 {
960 case SERVICE_CONTROL_STOP : service->Stop(); break;
961 case SERVICE_CONTROL_PAUSE : service->Pause(); break;
962 case SERVICE_CONTROL_CONTINUE : service->Continue(); break;
963 case SERVICE_CONTROL_SHUTDOWN : service->Shutdown(); break;
964 case SERVICE_CONTROL_POWEREVENT : service->PowerEvent(eventtype); break;
965 case SERVICE_CONTROL_SESSIONCHANGE : service->SessionChange(eventtype); break;
966 case SERVICE_CONTROL_INTERROGATE : break;
967 default : break;
968 }
