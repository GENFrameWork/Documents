143bool DIOSTREAMI2CCONFIG::SetLocalDeviceNameByPort()
144{
145 int revision = 0;
146
147 switch(GEN_XSYSTEM.GetTypeHardware(&revision))
148 {
149 case XSYSTEM_HARDWARETYPE_PC : localdevicename.Format(__L("/dev/i2c-%d"), port);
150 break;
151
152 case XSYSTEM_HARDWARETYPE_RASPBERRYPI : if(revision == 1)
153 localdevicename = __L("/dev/i2c-0");
154 else localdevicename = __L("/dev/i2c-1");
155 break;
156
157 default : break;
158 }
159
160 return true;
161}
