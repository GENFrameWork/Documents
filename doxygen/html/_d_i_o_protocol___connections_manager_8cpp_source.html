1629void DIOPROTOCOL_CONNECTIONSMANAGER::ThreadProtocolConnections(void* param)
1630{
1631 DIOPROTOCOL_CONNECTIONSMANAGER* protocolconnections = (DIOPROTOCOL_CONNECTIONSMANAGER*)param;
1632 if(!protocolconnections) return;
1633
1634 if(protocolconnections->isserver)
1635 {
1636 protocolconnections->ManageProtocolConnectionsServer();
1637 }
1638 else
1639 {
1640 protocolconnections->ManageProtocolConnectionsClient();
1641 }
1642
1643 int c=0;
1644 while(c < protocolconnections->ProtocolConnections_GetNAvailable())
1645 {
1646 DIOPROTOCOL_CONNECTION* protocolconnection = (DIOPROTOCOL_CONNECTION*)protocolconnections->protocolconnections.Get(c);
1647 if(protocolconnection)
1648 {
1649 DIOPROTOCOL* protocol = protocolconnection->GetDIOProtocol();
1650 if(protocol)
1651 {
1652 if(protocol->IsConnected())
1653 {
1654 if(!protocol->IsInitialized())
1655 {
1656 if(protocolconnections->isserver)
1657 {
1658 protocolconnection->SetInUse(true, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1659
1660 protocol->GetDIOStream()->GetXTimerNotActivity()->Reset();
1661 protocolconnections->SendEvent(DIOPROTOCOL_CONNECTIONSMANAGER_XEVENT_TYPE_INITPROTOCOL, protocolconnection);
1662 if(protocol->IsInitialized()) protocol->CMD_SendIsInitServerProtocol(true);
1663
1664 protocolconnection->SetInUse(false, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1665 }
1666 else
1667 {
1668 XTIMER* GEN_XFACTORY_CREATE(timerout, CreateTimer())
1669
1670 while(1)
1671 {
1672 if(protocolconnection->GetDIOProtocol()->IsDisconnected()) break;
1673 if(protocol->IsServerInitialized()) break;
1674
1675 if(timerout)
1676 {
1677 if(timerout->GetMeasureSeconds() > 5) break;
1678 }
1679
1680 GEN_XSLEEP.MilliSeconds(10);
1681 }
1682
1683 GEN_XFACTORY.DeleteTimer(timerout);
1684
1685
1686 if(protocol->IsServerInitialized())
1687 {
1688 protocolconnection->SetInUse(true, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1689
1690 protocol->GetDIOStream()->GetXTimerNotActivity()->Reset();
1691 protocolconnections->SendEvent(DIOPROTOCOL_CONNECTIONSMANAGER_XEVENT_TYPE_INITPROTOCOL, protocolconnection);
1692
1693 protocolconnection->SetInUse(false, DIOPROTOCOL_CONNECTIONSINUSEID_INIPROTOCOL);
1694 }
1695 }
1696
1697 XTRACE_PRINTCOLOR((protocol->IsInitialized()?2:4), __L("Protocol Connection Manager: %s Init Protocol %s") , protocolconnections->isserver?__L("Server"):__L("Client"), protocol->IsInitialized()?__L("Ok!"):__L("ERROR!"));
1698
1699 if(!protocol->IsInitialized())
1700 {
1701 protocolconnections->ProtocolConnections_Disconnect(c);
1702 }
1703 }
1704 else
1705 {
1706 if(protocol->GetDIOStream()->GetXTimerNotActivity()->GetMeasureSeconds() > DIOPROTOCOL_CONNECTIONS_DEFAULTIMECHECKCONNECTIONS)
1707 {
1708 XDWORD milliseconds;
1709
1710 bool status = protocol->CMD_Ping(milliseconds);
1711 if(!status)
1712 {
1713 protocolconnections->ProtocolConnections_Disconnect(c);
1714
1715 XTRACE_PRINTCOLOR(status?1:4, __L("Protocol Connection Manager: Check connection protocol %s"), status?__L("Ok!"):__L("ERROR!"));
1716 }
1717 }
1718 }
1719 }
1720 }
1721 }
1722
1723 c++;
1724 }
