212 {
213 XSTRING* line = lines.Get(c);
214 if(line)
215 {
216 if(line->Find(DIOWEBHEADER_CONTENT_LENGTH, true) != XSTRING_NOTFOUND)
217 {
218 XSTRING sizestr;

220 int index = line->Find(__L(":"), true);

224
225 contentlength = sizestr.ConvertToQWord();
226
227 if(contentlength !=-1) break;
228 }
229 }
