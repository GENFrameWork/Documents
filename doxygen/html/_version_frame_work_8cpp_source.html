338bool VERSIONFRAMEWORK::SetAppVersion(XCHAR* app_name, XDWORD app_version, XDWORD app_subversion, XDWORD app_versionerror, XCHAR* app_owner, XDWORD app_creationyear)
339{
340 app_titlestr.Empty();
341 app_versionstr.Empty();
342
343 app_versionstr.Format(__L("%s %d.%d.%d"), app_name?app_name:__L(""), app_version, app_subversion, app_versionerror);
344
345 XSTRING statusstr;
346
347 GetAppVersionStatus(app_version, app_subversion, app_versionerror, statusstr);
348 app_versionstr += statusstr;
349
350 app_titlestr = app_versionstr;
351 app_titlestr += __L(" Copyright (c) ");
352
353 XSTRING string2;
354
355 XDATETIME* GEN_XFACTORY_CREATE(xdatetime, CreateDateTime())
356 if(!xdatetime) return false;
357
358 xdatetime->Read();
359 string2.Format(((XDWORD)xdatetime->GetYear()>app_creationyear)?__L("%d-%d "):__L("%d "), app_creationyear, xdatetime->GetYear());
360
361 GEN_XFACTORY.DeleteDateTime(xdatetime);
362
363 app_titlestr += string2;
364 if(app_owner) app_titlestr += app_owner;
365
366 return true;
367}
