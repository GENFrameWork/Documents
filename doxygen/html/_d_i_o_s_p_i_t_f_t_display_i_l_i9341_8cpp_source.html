375bool DIOSPITFTDISPLAYILI9341::TFT_RotateRGB(int rotate, XBYTE bgr)
376{
377 #define MEM_Y (7) /* MY row address order */
378 #define MEM_X (6) /* MX column address order */
379 #define MEM_V (5) /* MV row / column exchange */
380 #define MEM_L (4) /* ML vertical refresh order */
381 #define MEM_H (2) /* MH horizontal refresh order */
382 #define MEM_BGR (3) /* RGB-BGR Order */
383
384 bool status = false;
385
386 switch(rotate)
387 {
388 case 0 : status = WriteRegister(0x36, 1, (1 << MEM_X) | (bgr << MEM_BGR)); break;
389 case 90 : status = WriteRegister(0x36, 1, (1 << MEM_Y) | (1 << MEM_X) |(1 << MEM_V) | (bgr << MEM_BGR)); break;
390 case 180 : status = WriteRegister(0x36, 1, (1 << MEM_Y) | (bgr << MEM_BGR)); break;
391 case 270 : status = WriteRegister(0x36, 1, (1 << MEM_V) | (1 << MEM_L) | (bgr << MEM_BGR)); break;
392 default : break;
393 }
394
395 return status;
396}
