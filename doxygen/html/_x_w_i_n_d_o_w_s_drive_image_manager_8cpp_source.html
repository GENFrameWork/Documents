98bool XWINDOWSDRIVEIMAGEMANAGER::GetDrives(XVECTOR<XDRIVEIMAGEMANAGER_DRIVE*>& drives)
99{
100 drives.DeleteContents();
101 drives.DeleteAll();
102
103 XPATH path;
104 XDWORD drivemask = GetLogicalDrives();
105 XQWORD ID = 0;
106 int c = 0;
107
108 // GetLogicalDrives returns 0 on failure, or a bitmask representing the drives available on the system (bit 0 = A:, bit 1 = B:, etc)
109
110 while(drivemask != 0)
111 {
112 if(drivemask & 1)
113 {
114 // the "A" in path will get incremented by the # of bits
115 // we've shifted
116 path.Format( __L("\\\\.\\%c:\\"), __C('A') + c);
117 if(CheckDriveType(path, ID))
118 {
119 XDRIVEIMAGEMANAGER_DRIVE* drive = new XDRIVEIMAGEMANAGER_DRIVE();
120 if(drive)
121 {
122 XSTRING label;
123
124 drive->GetName()->Format(__L("%c:"), __C('A') + c);
125 drive->GetPath()->Set(path);
126 drive->SetVolume(c);
127 drive->GetLabel()->Set(label);
128 drive->SetID(ID);
129
130 if(GetDriveLabel(drive->GetName()->Get(), label)) drive->GetLabel()->Set(label);
131
132 drives.Add(drive);
133 }
134 }
135 }
136
137 drivemask >>= 1;
138
139 ++c;
140 }
141
142 return true;
143}

157bool XWINDOWSDRIVEIMAGEMANAGER::Open(XDRIVEIMAGEMANAGER_DRIVE& drive)
158{
159 HANDLE handlevolume = GetHandleFromVolume(drive.GetVolume(), GENERIC_READ | GENERIC_WRITE);
160 if(handlevolume == INVALID_HANDLE_VALUE) return false;
161
162 drive.SetHandle((XDWORD)handlevolume);
163
164 XDWORD deviceID = 0;
165 if(!GetDeviceID(handlevolume, deviceID)) return false;
166 drive.SetDeviceID(deviceID);
167
168 if(!LockVolume(handlevolume, true)) return false;
169
170 if(!UnMountVolume(handlevolume)) return false;
171
172 HANDLE rawhandle = GetHandleFromDevice(deviceID, GENERIC_READ | GENERIC_WRITE);
173 if(rawhandle == INVALID_HANDLE_VALUE) return false;
174
175 drive.SetRawHandle((XDWORD)rawhandle);
176
177 XQWORD sectorsize = 0;
178 XQWORD numbersectors = 0;
179
180 if(!GetNumberOfSectors(rawhandle, sectorsize, numbersectors)) return false;
181
182 drive.SetSectorSize(sectorsize);
183 drive.SetNumberSectors(numbersectors);
184
185 return true;
186}

203bool XWINDOWSDRIVEIMAGEMANAGER::Read(XDRIVEIMAGEMANAGER_DRIVE& drive, XQWORD startsector, XQWORD numbersectors, XBUFFER& buffer)
204{
205 XBYTE* _buffer;
206 XDWORD maxbuffer = ((XDWORD)numbersectors * (XDWORD)drive.GetSectorSize());
207 XQWORD maxnsectorsblock = (maxbuffer / drive.GetSectorSize());
208 XQWORD readnsectorsblock = 0;
209 XQWORD sizeread = 0;
210 bool status = false;
211
212 buffer.Delete();
213
214 _buffer = new XBYTE[maxbuffer];
215 if(!_buffer) return false;
216
217 if(maxnsectorsblock > numbersectors) maxnsectorsblock = numbersectors;
218
219 while(readnsectorsblock < numbersectors)
220 {
221 XDRIVEIMAGEMANAGER_XEVENT xevent(this, XDRIVEIMAGEMANAGER_XEVENT_TYPE_READSECTORS);
222
223 xevent.SetDrive(&drive);
224 xevent.SetIndexSector(startsector);
225 xevent.SetNumberSectorsInOperation(maxnsectorsblock);
226
227 sizeread = (maxnsectorsblock * drive.GetSectorSize());
228
229 memset(_buffer, 0, maxbuffer);
230
231 status = ReadSectorDataFromHandle((HANDLE)drive.GetRawHandle(), startsector, drive.GetSectorSize(), _buffer, sizeread);
232 if(status)
233 {
234 xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_NONE);
235
236 buffer.Add(_buffer, (XDWORD)sizeread);
237
238 readnsectorsblock += maxnsectorsblock;
239
240 if((readnsectorsblock + maxnsectorsblock) > numbersectors) maxnsectorsblock = (numbersectors - readnsectorsblock);
241 if((readnsectorsblock + maxnsectorsblock) > drive.GetNumberSectors()) maxnsectorsblock = (drive.GetNumberSectors() - readnsectorsblock);
242 }
243 else
244 {
245 xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_READSECTORS);
246 }
247
248 PostEvent(&xevent);
249
250 if(!status) break;
251
252 }
253
254 delete [] _buffer;
255
256 return status;
257}

598 DWORD nbytesreturned;
599
600
601 path_noslash = path.Get();
602

605 // some calls require no tailing slash, some require a trailing slash...
606

611

614 if(hdevice == INVALID_HANDLE_VALUE) break;
615
616 int sizearray = sizeof(STORAGE_DEVICE_DESCRIPTOR) + 512 - 1;
617 pdevdesc = (PSTORAGE_DEVICE_DESCRIPTOR)new BYTE[sizearray];
618
619 pdevdesc->Size = sizearray;
620
621 // get the device number if the drive is removable or (fixed AND on the usb bus, SD, or MMC (undefined in XP/mingw))

623 if(GetMediaType(hdevice))

651
