326bool XLINUXTRACE::GetHandleNet(XTRACE_TARGET* target)
327{
328 #if(defined(DIO_ACTIVE) && defined(DIO_STREAMUDP_ACTIVE))
329
330 struct sockaddr_in addr;
331 int handle;
332
333 memset(&addr, 0, sizeof(struct sockaddr));
334
335 handle = socket(AF_INET, SOCK_DGRAM, 0);
336 if(handle < 0) return false;
337
338 if(!target->GetIP()[0]) return false;
339
340 addr.sin_family = AF_INET;
341 addr.sin_port = htons(target->GetPort());
342
343 if(target->GetIP()[0] == '*')
344 {
345 addr.sin_addr.s_addr = htonl(INADDR_BROADCAST);
346
347 int yes = 1;
348 setsockopt(handle, SOL_SOCKET, SO_BROADCAST, (char*)&yes, sizeof(yes));
349 }
350 else
351 {
352 addr.sin_addr.s_addr = inet_addr(target->GetIP());
353 }
354
355 connect(handle, (struct sockaddr*)&addr, sizeof(struct sockaddr));
356
357 XQWORD NEThandle = (XQWORD)handle;
358
359 target->SetNETHandle(NEThandle);

361 #endif
362
363 return true;
