97bool GRPWINDOWSCONTEXT::Create(void* handle)
98{
99 // deal with weird windows context creation
100
101
102 #ifdef GRP_OPENGL_ACTIVE
103
104 GRPWINDOWSSCREEN* windowsscreen = (GRPWINDOWSSCREEN*)this->screen;
105 if(!windowsscreen) return false;
106
107 // Choose the corect pixel format
108 windowsscreen->ChooseValidPixelFormat();
109 // make dummy opengl context
110
111 hdlopenglrc = wglCreateContext(windowsscreen->GetHDC());
112 wglMakeCurrent(windowscreen->GetHDC(), hdlopenglrc);
113
114 glewExperimental = true;
115 if(glewInit() != GLEW_OK) return false;
116
117 // Set the pixel format usiong wgl extensions
118 int nPixelFormat;
119 BOOL bValidPixFormat;
120 UINT nMaxFormats = 1;
121 UINT nNumFormats;
122 float pfAttribFList[] = { 0, 0 };
123 int piAttribIList[] = { WGL_DRAW_TO_WINDOW_ARB, GL_TRUE,
124 WGL_SUPPORT_OPENGL_ARB, GL_TRUE,
125 WGL_ACCELERATION_ARB, WGL_FULL_ACCELERATION_ARB,
126 WGL_COLOR_BITS_ARB, 32,
127 WGL_ALPHA_BITS_ARB, 8,
128 WGL_DEPTH_BITS_ARB, 16,
129 WGL_STENCIL_BITS_ARB, 8,
130 WGL_DOUBLE_BUFFER_ARB, GL_TRUE,
131 WGL_PIXEL_TYPE_ARB, WGL_TYPE_RGBA_ARB,
132 WGL_SAMPLE_BUFFERS_ARB, (multisampling != 0) ? 1 : 0, //Number of mutisample buffers (must be 1 at time of writing)
133 WGL_SAMPLES_ARB, (int)multisampling, //Number of samples
134 0, 0 };
135
136 bValidPixFormat = wglChoosePixelFormatARB(windowsscreen->GetHDC(), piAttribIList, pfAttribFList, nMaxFormats, &nPixelFormat, &nNumFormats );
137 if(!bValidPixFormat)
138 {
139 XTRACE_PRINTCOLOR(4,__L("Invalid Pixel Format! Try back..."));
140
141 const int attribList[] ={ WGL_DRAW_TO_WINDOW_ARB , GL_TRUE ,
142 WGL_SUPPORT_OPENGL_ARB , GL_TRUE ,
143 WGL_DOUBLE_BUFFER_ARB , GL_TRUE ,
144 WGL_PIXEL_TYPE_ARB , WGL_TYPE_RGBA_ARB ,
145 WGL_COLOR_BITS_ARB , 32 ,
146 WGL_DEPTH_BITS_ARB , 24 ,
147 WGL_STENCIL_BITS_ARB , 8 ,
148 0, //End
149 };
150
151 bValidPixFormat = wglChoosePixelFormatARB(windowsscreen->GetHDC(), attribList, pfAttribFList, nMaxFormats, &nPixelFormat, &nNumFormats );
152 if(!bValidPixFormat)
153 {
154 XTRACE_PRINTCOLOR(4,__L("\n\n\n\nInvalid Pixel Format!"));
155 return false;
156 }
157 }
158
159 windowsscreen->SetPixelFormat(nPixelFormat);
160
161 // delete old window
162 windowsscreen->Delete();
163
164 // make real window
165 windowsscreen->Create(handle);
166
167 // delete old context
168 Destroy();
169
170 // make real context
171 hdlopenglrc = wglCreateContext(windowsscreen->GetHDC());
172 wglMakeCurrent(windowsscreen->GetHDC(), hdlopenglrc);
173
174 openglctrl.Ini();
175
176 return true;
177
178 #else
179
180 return false;
181
182 #endif
183}
