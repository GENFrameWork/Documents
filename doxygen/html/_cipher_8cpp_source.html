356bool CIPHER::Cipher(XBYTE* input,XDWORD size)
357{
358 result->Delete();
359 result->Add(input, size);
360
361 CIPHERKEYSYMMETRICAL* key = (CIPHERKEYSYMMETRICAL*)GetKey(0);
362 if(!key) return false;
363 if(!key->Get()->Get()) return false;
364
365 int indexkey = 0;
366
367 for(XDWORD c=0; c<size; c++)
368 {
369 result->Get()[c] ^= key->Get()->GetByte(indexkey);
370
371 indexkey++;
372 if(indexkey >= (int)key->Get()->GetSize()) indexkey = 0;
373 }
374
375 return true;
376}
