173bool DIOSTREAMSPI::Open()
174{
175 if(!config) return false;
176 if(!xtimer) return false;
177
178 if(!threadconnection) return false;
179
180 if(!inbuffer) return false;
181 if(!outbuffer) return false;
182
183 if(config->IsDirectAccess())
184 {
185 GPIO_Entry[DIOSTREAMSPI_GPIO_CLK] = GEN_DIOGPIO.GPIOEntry_GetByID(config->GetPinsDirectAccess()[DIOSTREAMSPI_GPIO_CLK]);
186 GPIO_Entry[DIOSTREAMSPI_GPIO_MISO] = GEN_DIOGPIO.GPIOEntry_GetByID(config->GetPinsDirectAccess()[DIOSTREAMSPI_GPIO_MISO]);
187 GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI] = GEN_DIOGPIO.GPIOEntry_GetByID(config->GetPinsDirectAccess()[DIOSTREAMSPI_GPIO_MOSI]);
188 GPIO_Entry[DIOSTREAMSPI_GPIO_CS] = GEN_DIOGPIO.GPIOEntry_GetByID(config->GetPinsDirectAccess()[DIOSTREAMSPI_GPIO_CS]);
189
190
191 if(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK]) GEN_DIOGPIO.SetMode(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK] , DIOGPIO_MODE_INPUT);
192 if(GPIO_Entry[DIOSTREAMSPI_GPIO_MISO]) GEN_DIOGPIO.SetMode(GPIO_Entry[DIOSTREAMSPI_GPIO_MISO] , DIOGPIO_MODE_OUTPUT);
193 if(GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI]) GEN_DIOGPIO.SetMode(GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI] , DIOGPIO_MODE_INPUT);
194 if(GPIO_Entry[DIOSTREAMSPI_GPIO_CS]) GEN_DIOGPIO.SetMode(GPIO_Entry[DIOSTREAMSPI_GPIO_CS] , DIOGPIO_MODE_INPUT);
195 }
196
197 bufferread = new XBYTE[DIOSTREAMSPI_MAXSIZEBUFFER];
198
199 mode = config->GetMode();
200 nbitsword = config->GetNBitsWord();
201 timedelay = config->GetDelay();
202 timedelaydouble *= timedelay;
203 CSline = ((mode & DIOSTREAMSPI_CS_HIGH) == DIOSTREAMSPI_CS_HIGH);
204 lbsfirst = ((mode & DIOSTREAMSPI_LSB_FIRST) == DIOSTREAMSPI_LSB_FIRST);
205
206 if(config->IsDirectAccess())
207 {
208 switch(mode)
209 {
210 case DIOSTREAMSPI_MODE_0:
211 case DIOSTREAMSPI_MODE_1: GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], false);
212 break;
213
214 case DIOSTREAMSPI_MODE_2:
215 case DIOSTREAMSPI_MODE_3: GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], true);
216 break;
217 }
218
219 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CS], CSline);
220 }
221
222 SetEvent(DIOSTREAMSPI_FSMEVENT_CONNECTED);
223
224 status = DIOSTREAMSTATUS_CONNECTED;
225
226 ResetXBuffers();
227
228 ResetConnectionStatistics();
229
230 return threadconnection->Ini();
231}

274bool DIOSTREAMSPI::TransferBuffer(XBYTE* bufferread, XBYTE* bufferwrite, XDWORD size)
275{
276 if(!config) return false;
277 if(!xtimer) return false;
278
279 if(!GPIO_Entry[DIOSTREAMSPI_GPIO_CLK]) return false;
280 if(!GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI]) return false;
281 if(!GPIO_Entry[DIOSTREAMSPI_GPIO_CS]) return false;
282
283 if(config->GetNBitsWord()<=8)
284 {
285 for(XDWORD c=0;c<size;c++)
286 {
287 XWORD readdata;
288
289 TransferOneData(bufferwrite[c], readdata);
290
291 bufferread[c] = (XBYTE)readdata;
292 }
293 }
294 else
295 {
296 for(XDWORD c=0; c<size; c+=2)
297 {
298 XWORD readdata;
299 //XWORD writedata = bufferwrite[c];
300
301 //writedata<<=8;
302 //writedata |= bufferwrite[c];
303
304 TransferOneData(bufferwrite[c], readdata);
305
306 bufferread[c] = (XBYTE)(readdata>>8);
307 bufferread[c+1] = (XBYTE)(readdata&0xFF);
308 }
309
310 }
311
312 return true;
313}

388bool DIOSTREAMSPI::TransferOneData(XWORD writedata,XWORD& readdata)
389{
390 if(!config->IsDirectAccess()) return false;
391
392 XWORD mask = 0x0001;
393
394 if(!lbsfirst) mask<<=(nbitsword-1);
395
396 readdata = 0;
397
398 for(int c=0;c<nbitsword;c++)
399 {
400 switch(mode)
401 {
402 case DIOSTREAMSPI_MODE_0: GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI] , (writedata&mask)?true:false);
403 if(GEN_DIOGPIO.GetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MISO])) readdata |= mask;
404 DIOSTREAMSPI_SLEEP;
405 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], true);
406 DIOSTREAMSPI_SLEEPDOUBLE;
407 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], false);
408 break;
409
410 case DIOSTREAMSPI_MODE_1:
411 case DIOSTREAMSPI_MODE_2:
412 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], true);
413 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI], (writedata&mask)?true:false);
414 if(GEN_DIOGPIO.GetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MISO])) readdata |= mask;
415 DIOSTREAMSPI_SLEEP;
416 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], false);
417 DIOSTREAMSPI_SLEEPDOUBLE;
418 break;
419
420 case DIOSTREAMSPI_MODE_3:
421 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], false);
422 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MOSI], (writedata&mask)?true:false);
423 if(GEN_DIOGPIO.GetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_MISO])) readdata |= mask;
424 DIOSTREAMSPI_SLEEP;
425 GEN_DIOGPIO.SetValue(GPIO_Entry[DIOSTREAMSPI_GPIO_CLK], true);
426 DIOSTREAMSPI_SLEEPDOUBLE;
427 break;
428 }
429
430 if(lbsfirst) mask<<=1; else mask>>=1;
431 }
432
433 return true;
434}

446void DIOSTREAMSPI::ThreadConnection(void* data)
447{
448 DIOSTREAMSPI* diostream = (DIOSTREAMSPI*)data;
449 if(!diostream) return;
450
451 if(diostream->GetEvent()==DIOSTREAMSPI_FSMEVENT_NONE) // No hay nuevos Eventos
452 {
453 switch(diostream->GetCurrentState())
454 {
455 case DIOSTREAMSPI_FSMSTATE_NONE : break;
456
457 case DIOSTREAMSPI_FSMSTATE_CONNECTED : break;
458
459 case DIOSTREAMSPI_FSMSTATE_WAITINGTOREAD : { if(diostream->outbuffer->IsBlocked()) break;
460
461 while(1)
462 {
463 XDWORD size = diostream->outbuffer->GetSize();
464
465 if(!size) break;
466 if(size>(XDWORD)DIOSTREAMSPI_MAXSIZEBUFFER) size = DIOSTREAMSPI_MAXSIZEBUFFER;
467
468 memset(diostream->bufferread, 0 , DIOSTREAMSPI_MAXSIZEBUFFER);
469
470 if(diostream->TransferBuffer(diostream->bufferread, diostream->outbuffer->Get(), size))
471 {
472 if(!diostream->config->IsOnlyWrite())
473 {
474 diostream->inbuffer->Add(diostream->bufferread,size);
475 }
476
477 diostream->outbuffer->Extract(NULL, 0, size);
478 }
479 }
480 }
481 break;
482
483
484 case DIOSTREAMSPI_FSMSTATE_SENDINGDATA : break;
485
486 case DIOSTREAMSPI_FSMSTATE_DISCONNECTING : break;
487
488 }
489 }
490 else
491 {
492 if(diostream->GetEvent()<DIOSTREAMSPI_LASTEVENT)
493 {
494 diostream->CheckTransition();
495
496 switch(diostream->GetCurrentState())
497 {
498 case DIOSTREAMSPI_FSMSTATE_NONE : break;
499
500 case DIOSTREAMSPI_FSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
501 xevent.SetDIOStream(diostream);
502 diostream->PostEvent(&xevent);
503
504 diostream->SetEvent(DIOSTREAMSPI_FSMEVENT_WAITINGTOREAD);
505 }
506 break;
507
508 case DIOSTREAMSPI_FSMSTATE_WAITINGTOREAD : break;
509
510 case DIOSTREAMSPI_FSMSTATE_SENDINGDATA : break;
511
512 case DIOSTREAMSPI_FSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
513 xevent.SetDIOStream(diostream);
514 diostream->PostEvent(&xevent);
515
516 diostream->threadconnection->Run(false);
517 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
518 }
519 break;
520 }
521 }
522 }
523}
