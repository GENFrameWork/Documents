115bool HASHCRC16::Do(XBYTE* input, XQWORD size)
116{
117 if(!size) return false;
118
119 XWORD value = 0;
120
121 resulttmp->Extract(value);
122
123 for(XDWORD c=0; c<size; c++)
124 {
125 value = Update(value, (*input));
126 input++;
127 }
128
129 resulttmp->Add((XWORD)value);
130
131 return true;
132}
