308void CIPHERAES::AESGenTables(void)
309{
310 int pow[256];
311 int log[256];
312 int i;
313 int x;
314 int y;
315 int z;
316
317 // Compute pow and log tables over GF(2^8)
318
319 for(i=0, x=1; i<256; i++)
320 {
321 pow[i] = x;
322 log[x] = i;
323 x = ( x ^ XTIME( x ) ) & 0xFF;
324 }
325
326
327 // Calculate the round constants
328
329 for(i=0, x=1; i<10; i++)
330 {
331 RCON[i] = (XDWORD) x;
332 x = XTIME( x ) & 0xFF;
333 }
334
335
336 // Generate the forward and reverse S-boxes
337
338 FSb[0x00] = 0x63;
339 RSb[0x63] = 0x00;
340
341 for(i=1; i<256; i++)
342 {
343 x = pow[255 - log[i]];
344
345 y = x; y = ( (y << 1) | (y >> 7) ) & 0xFF;
346 x ^= y; y = ( (y << 1) | (y >> 7) ) & 0xFF;
347 x ^= y; y = ( (y << 1) | (y >> 7) ) & 0xFF;
348 x ^= y; y = ( (y << 1) | (y >> 7) ) & 0xFF;
349 x ^= y ^ 0x63;
350
351 FSb[i] = (XBYTE) x;
352 RSb[x] = (XBYTE) i;
353 }
354
355
356 // Generate the forward and reverse tables
357
358 for( i = 0; i < 256; i++ )
359 {
360 x = FSb[i];
361 y = XTIME( x ) & 0xFF;
362 z = ( y ^ x ) & 0xFF;
363
364 FT0[i] = ( (XDWORD) y ) ^
365 ( (XDWORD) x << 8 ) ^
366 ( (XDWORD) x << 16 ) ^
367 ( (XDWORD) z << 24 );
368
369 FT1[i] = ROTL8( FT0[i] );
370 FT2[i] = ROTL8( FT1[i] );
371 FT3[i] = ROTL8( FT2[i] );
372
373 x = RSb[i];
374
375 RT0[i] = ( (XDWORD) MUL( 0x0E, x ) ) ^
376 ( (XDWORD) MUL( 0x09, x ) << 8 ) ^
377 ( (XDWORD) MUL( 0x0D, x ) << 16 ) ^
378 ( (XDWORD) MUL( 0x0B, x ) << 24 );
379
380 RT1[i] = ROTL8( RT0[i] );
381 RT2[i] = ROTL8( RT1[i] );
382 RT3[i] = ROTL8( RT2[i] );
383 }
384}

400bool CIPHERAES::AESSetKeyCipher(CIPHERAES_CONTEXT* ctx, XBYTE* key, XDWORD keysize)
401{
402 XDWORD* RK;
403
404 switch( keysize )
405 {
406 case 128 : ctx->nr = 10; break;
407 case 192 : ctx->nr = 12; break;
408 case 256 : ctx->nr = 14; break;
409 default : return false;
410 }
411
412 ctx->rk = RK = ctx->buf;
413
414 for(XDWORD i = 0; i < (keysize >> 5); i++ )
415 {
416 GET_UINT32_LE( RK[i], key, i << 2 );
417 }
418
419 switch( ctx->nr )
420 {
421 case 10: for(XDWORD i = 0; i < 10; i++, RK += 4 )
422 {
423 RK[4] = RK[0] ^ RCON[i] ^
424 ( (XDWORD) FSb[ ( RK[3] >> 8 ) & 0xFF ] ) ^
425 ( (XDWORD) FSb[ ( RK[3] >> 16 ) & 0xFF ] << 8 ) ^
426 ( (XDWORD) FSb[ ( RK[3] >> 24 ) & 0xFF ] << 16 ) ^
427 ( (XDWORD) FSb[ ( RK[3] ) & 0xFF ] << 24 );
428
429 RK[5] = RK[1] ^ RK[4];
430 RK[6] = RK[2] ^ RK[5];
431 RK[7] = RK[3] ^ RK[6];
432 }
433 break;
434
435 case 12: for(XDWORD i = 0; i < 8; i++, RK += 6 )
436 {
437 RK[6] = RK[0] ^ RCON[i] ^
438 ( (XDWORD) FSb[ ( RK[5] >> 8 ) & 0xFF ] ) ^
439 ( (XDWORD) FSb[ ( RK[5] >> 16 ) & 0xFF ] << 8 ) ^
440 ( (XDWORD) FSb[ ( RK[5] >> 24 ) & 0xFF ] << 16 ) ^
441 ( (XDWORD) FSb[ ( RK[5] ) & 0xFF ] << 24 );
442
443 RK[7] = RK[1] ^ RK[6];
444 RK[8] = RK[2] ^ RK[7];
445 RK[9] = RK[3] ^ RK[8];
446 RK[10] = RK[4] ^ RK[9];
447 RK[11] = RK[5] ^ RK[10];
448 }
449 break;
450
451 case 14: for(XDWORD i = 0; i < 7; i++, RK += 8 )
452 {
453 RK[8] = RK[0] ^ RCON[i] ^
454 ( (XDWORD) FSb[ ( RK[7] >> 8 ) & 0xFF ] ) ^
455 ( (XDWORD) FSb[ ( RK[7] >> 16 ) & 0xFF ] << 8 ) ^
456 ( (XDWORD) FSb[ ( RK[7] >> 24 ) & 0xFF ] << 16 ) ^
457 ( (XDWORD) FSb[ ( RK[7] ) & 0xFF ] << 24 );
458
459 RK[9] = RK[1] ^ RK[8];
460 RK[10] = RK[2] ^ RK[9];
461 RK[11] = RK[3] ^ RK[10];
462
463 RK[12] = RK[4] ^
464 ( (XDWORD) FSb[ ( RK[11] ) & 0xFF ] ) ^
465 ( (XDWORD) FSb[ ( RK[11] >> 8 ) & 0xFF ] << 8 ) ^
466 ( (XDWORD) FSb[ ( RK[11] >> 16 ) & 0xFF ] << 16 ) ^
467 ( (XDWORD) FSb[ ( RK[11] >> 24 ) & 0xFF ] << 24 );
468
469 RK[13] = RK[5] ^ RK[12];
470 RK[14] = RK[6] ^ RK[13];
471 RK[15] = RK[7] ^ RK[14];
472 }
473 break;
474
475 default: break;
476 }
477
478 return true;
479}
