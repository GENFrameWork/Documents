298int XLINUXCONSOLE::GetChar()
299{
300 struct termios oldt, newt;
301 int ch;
302
303 tcgetattr( STDIN_FILENO, &oldt ); /* Obtenemos atributos del terminal */
304 newt = oldt;
305 newt.c_lflag &= ~ICANON;
306 newt.c_lflag &= ~ECHO; /* Eliminamos el Echo */
307
308 tcsetattr( STDIN_FILENO, TCSANOW, &newt ); /* Definimos los nuevos atributos al terminal */
309
310 ch = getchar();
311
312 tcsetattr( STDIN_FILENO, TCSANOW, &oldt ); /* Ponemos los atributos como estaban al principio */
313
314 return ch;
315}
