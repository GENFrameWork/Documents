490bool DIOSCRAPERWEBGEOLOCATIONIP::Get(DIOIP& IP, DIOGEOLOCATIONIP_RESULT& geolocationIPresult, int timeoutforurl, XSTRING* localIP, bool usecache)
491{
492 bool status = false;
493
494 if(xmutexdo) xmutexdo->Lock();
495
496 this->IP.Set(IP.Get());
497
498 if(usecache)
499 {
500 XSTRING IPstring;
501
502 IP.GetXString(IPstring);
503
504 DIOGEOLOCATIONIP_RESULT* geoIPresult = (DIOGEOLOCATIONIP_RESULT*)cache->Get(IPstring);
505 if(geoIPresult)
506 {
507 geolocationIPresult.CopyFrom(geoIPresult);
508
509 if(xmutexdo) xmutexdo->UnLock();
510 return true;
511 }
512 }
513
514 if(Load(DIOSCRAPERWEBGEOLOCATIONIP_NAMEFILE))
515 {
516 if(Do(DIOSCRAPERWEBGEOLOCATIONIP_NAMESERVICE, timeoutforurl, localIP))
517 {
518 XSTRING country;
519 XSTRING state;
520 XSTRING city;
521 XSTRING ISP;
522 XSTRING organization;
523 double latitude = 0;
524 double longitude = 0;
525 XSTRING string;
526
527 country = GetValue(__L("COUNTRY"));
528 state = GetValue(__L("STATE"));
529 city = GetValue(__L("CITY"));
530 ISP = GetValue(__L("ISP"));
531 organization = GetValue(__L("ORGANIZATION"));
532
533 string = GetValue(__L("LATITUDE"));
534 if(!string.IsEmpty()) string.UnFormat(__L("%f"), &latitude);
535
536 string = GetValue(__L("LONGITUDE"));
537 if(!string.IsEmpty()) string.UnFormat(__L("%f"), &longitude);
538
539 geolocationIPresult.Set(country, state, city, ISP, organization);
540 geolocationIPresult.Set((float)latitude, (float)longitude);
541
542 if(!geolocationIPresult.IsEmpty())
543 {
544 if(usecache)
545 {
546 DIOGEOLOCATIONIP_RESULT* geoIPresult = new DIOGEOLOCATIONIP_RESULT();
547 if(geoIPresult)
548 {
549 XSTRING IPstring;
550
551 IP.GetXString(IPstring);
552
553 geoIPresult->CopyFrom(&geolocationIPresult);
554 cache->Add(IPstring, geoIPresult);
555 }
556 }
557
558 status = true;
559 }
560 }
561 }
562
563 if(xmutexdo) xmutexdo->UnLock();
564
565 return status;
566}
