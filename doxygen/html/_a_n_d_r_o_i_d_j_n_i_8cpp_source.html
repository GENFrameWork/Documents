93void ANDROIDJNI::InitializeJNI(android_app* application, jint javaversion)
94{
95 currentjavaVM = application->activity->vm;
96 currentjavaversion = javaversion;
97
98 JNIEnv* env = ANDROIDJNI::GetJNIEnv();
99 if(!env)
100 {
101 return;
102 }
103
104 jobject mainactivity = application->activity->clazz;
105
106 jclass MainClass = env->GetObjectClass(mainactivity);
107 jclass classClass = env->FindClass("java/lang/Class");
108 jclass classLoaderClass = env->FindClass("java/lang/ClassLoader");
109 jmethodID getClassLoaderMethod = env->GetMethodID(classClass, "getClassLoader", "()Ljava/lang/ClassLoader;");
110 jobject classLoader = env->CallObjectMethod(MainClass, getClassLoaderMethod);
111
112 ClassLoader = env->NewGlobalRef(classLoader);
113
114 FindClassMethod = env->GetMethodID(classLoaderClass, "findClass", "(Ljava/lang/String;)Ljava/lang/Class;");
115}

127JNIEnv* ANDROIDJNI::GetJNIEnv()
128{
129 static XDWORD threadslot = 0;
130 if(!threadslot)
131 {
132 pthread_key_create((pthread_key_t*)&threadslot, &JavaEnvDestructor);
133 }
134
135 JNIEnv* env = NULL;
136
137 if(!currentjavaVM)
138 {
139 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[ANDROID] No Current Java VM"));
140 }
141
142 jint getenvresult = currentjavaVM->GetEnv((void**)&env, currentjavaversion);
143 if(getenvresult == JNI_EDETACHED)
144 {
145 jint attachresult = currentjavaVM->AttachCurrentThread(&env, NULL);
146
147 if(attachresult == JNI_ERR)
148 {
149 return NULL;
150 }
151 }
152 else
153 {
154 if(getenvresult != JNI_OK)
155 {
156 return NULL;
157 }
158 }
159
160 return env;
161}

292bool ANDROIDJNI::CheckJavaException()
