217bool UI_COLOR::SetFromString(XCHAR* string)
218{
219 if(!string) return false;
220
221 XSTRING colorstr;
222 int ncommas = 0;
223 int _alpha = 0;
224 bool status = false;
225
226 colorstr = string;
227 if(colorstr.IsEmpty()) return false;
228
229 for(XDWORD c=0; c<colorstr.GetSize(); c++)
230 {
231 XCHAR character = colorstr.Get()[c];
232 if(character == __C(',')) ncommas++;
233 }
234
235 switch(ncommas)
236 {
237 case 0 : // only name
238 { XSTRING* resolved_colorstr = GEN_UI_COLORS.Get(colorstr);
239 if(resolved_colorstr) status = SetFromString(resolved_colorstr->Get());
240 }
241 break;
242
243 case 1 : // name + int alpha
244 { XSTRING _colorstr;
245 XSTRING* resolved_colorstr = NULL;
246
247 _colorstr.AdjustSize(_MAXSTR);
248 colorstr.UnFormat(__L("%s,%d"), _colorstr.Get(), &_alpha);
249 _colorstr.AdjustSize();
250
251 resolved_colorstr = GEN_UI_COLORS.Get(_colorstr);
252 if(resolved_colorstr) status = SetFromString(resolved_colorstr->Get());
253
254 alpha = GetAlphaForPercent(_alpha);
255 }
256 break;
257
258 case 2 : // int colors
259 status = colorstr.UnFormat(__L("%d,%d,%d"), &red, &green, &blue);
260 alpha = 255;
261 break;
262
263 case 3 : // int colors + int alpha
264 status = colorstr.UnFormat(__L("%d,%d,%d,%d"), &red, &green, &blue, &_alpha);
265 alpha = GetAlphaForPercent(_alpha);
266 break;
267
268 default : break;
269 }
270
271 return status;
272}
