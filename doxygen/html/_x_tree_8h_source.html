221 XTREE_NODE_ITERATOR<T>& operator ++ ()
222 {
223 // put the current' nodes children on the stack, and
224 // then move onto the next one by popping it
225 // off the stack and making it the current one
226 if(current)
227 {
228 for( int i = current->GetNChildren()-1; i>=0 ; i--)
229 {
230 stack.Push(current->GetChild(i));
231 }
232 }
233
234 if(!stack.IsEmpty())
235 {
236 current = stack.Top();
237 stack.Pop();
238 }
239 else
240 {
241 (*this) = tree->End();
242 }
243
244 return (*this);
245 }

247 operator bool () { return (current != NULL); }

250 const XTREE_NODE_ITERATOR<T>& operator = (const XTREE_NODE_ITERATOR<T>& iterator)
251 {
252 this->tree = iterator.tree;
253 this->current = iterator.current;
254
255 while(!stack.IsEmpty())
256 {
257 stack.Pop();
258 }
259
260 XTREE_NODE_ITERATOR<T>& it = const_cast <XTREE_NODE_ITERATOR<T>&>(iterator); // remove const-ness
261 int stacksize = it.GetStack()->GetSize();
262 XVECTOR<XTREE_NODE<T>*> vtree(stacksize);
263
264 while(!it.stack.IsEmpty())
265 {
266 vtree.Add(it.stack.Top());
267 it.stack.Pop();
268 }
269
270 // now fill both stacks from the vector (traverse from back)
271 for(int i=vtree.GetSize()-1; i>=0; i--)
272 {
273 it.stack.Push(vtree.Get(i));
274 stack.Push(vtree.Get(i));
275 }
276
277 return (*this);
278 }
