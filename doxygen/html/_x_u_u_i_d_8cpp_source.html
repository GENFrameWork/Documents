368bool XUUID::GenerateRandom()
369{
370 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
371 if(xrand)
372 {
373 xrand->Ini();
374
375 XDATETIME* GEN_XFACTORY_CREATE(datetime, CreateDateTime())
376 if(datetime)
377 {
378 XSTRING datetimestr;
379 HASHCRC32 hashcrc32;
380
381 datetime->Read();
382 datetime->GetDateTimeToString((XDATETIME_FORMAT_YMD | XDATETIME_FORMAT_ADDTIME | XDATETIME_FORMAT_TIMEWITHSECONDS), datetimestr);
383
384 hashcrc32.Do((XBYTE*)datetimestr.Get(), (datetimestr.GetSize() * sizeof(XCHAR)));
385
386 data1 = hashcrc32.GetResultCRC32();
387
388 for(int c=0; c<sizeof(XDWORD); c++)
389 {
390 data1 ^= (XBYTE)xrand->Max(255);
391 if(c!=sizeof(XDWORD)-1) data1 <<= 8;
392 }
393
394 if(xrand->Between(0, 1)) data1 |= 0x80000000;
395
396
397 data2 = (XWORD)xrand->Between(0, 65535);
398 data3 = (XWORD)xrand->Between(0, 65535);
399
400 data4 = (XBYTE)xrand->Between(0, 255);
401 data5 = (XBYTE)xrand->Between(0, 255);
402
403 for(int c=0; c<XUUIDMAXDATA4; c++)
404 {
405 data6[c] = (XBYTE)xrand->Max(255);
406 }
407
408 GEN_XFACTORY.DeleteDateTime(datetime);
409 }
410
411 GEN_XFACTORY.DeleteRand(xrand);
412 }
