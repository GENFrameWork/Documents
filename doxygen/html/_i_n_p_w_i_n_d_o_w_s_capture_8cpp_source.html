109bool INPWINDOWSCAPTURE::Activate()
110{
111 if(keyhook) return false;
112
113 RegisterEvent(INPCAPTURE_XEVENT_TYPE_PRESSKEY);
114 RegisterEvent(INPCAPTURE_XEVENT_TYPE_UNPRESSKEY);
115
116 // Set up the hook
117
118 HINSTANCE hinst = (HINSTANCE)applicationhandle;
119 if(!hinst) return false;
120
121 keyhook = SetWindowsHookEx(WH_KEYBOARD_LL, LowLevelKeyboardProc, hinst, 0);
122 if(keyhook == NULL)
123 {
124 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Windows Hook Keyboard] Error: Could not hook keyboard. Another program might be interfering."));
125 return false;
126 }
127
128 return true;
129}

141bool INPWINDOWSCAPTURE::Deactivate()
142{
143 if(!keyhook) return false;
144
145 DeRegisterEvent(INPCAPTURE_XEVENT_TYPE_PRESSKEY);
146 DeRegisterEvent(INPCAPTURE_XEVENT_TYPE_UNPRESSKEY);
147
148 //Remove keyboard hook
149 if(UnhookWindowsHookEx(keyhook) == 0)
150 {
151 XTRACE_PRINTCOLOR(XTRACE_COLOR_RED, __L("[Windows Hook Keyboard] Error: Could not unhook keyboard."));
152 return false;
153 }
154
155 //Success
156 keyhook = NULL;
157
158 return true;
159}

178LRESULT CALLBACK INPWINDOWSCAPTURE::LowLevelKeyboardProc(int ncode, WPARAM wparam, LPARAM lparam)
179{
180 bool keydown = false;
181
182 if(ncode == HC_ACTION)
183 {
184 if(wparam == WM_KEYDOWN || wparam == WM_SYSKEYDOWN) keydown = true;
185 if(wparam == WM_KEYUP || wparam == WM_SYSKEYUP) keydown = false;
186 }
187
188 KBDLLHOOKSTRUCT* keydata = (KBDLLHOOKSTRUCT*)lparam;
189
190 XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("[Windows Hook keyboard] vkCode %04X, scanCode %08X, flags %08X [%c]"), keydata->vkCode, keydata->scanCode, keydata->flags, keydata->vkCode);
191
192 if(capture)
193 {
194 INPCAPTURE_XEVENT xevent(capture, (keydown?INPCAPTURE_XEVENT_TYPE_PRESSKEY:INPCAPTURE_XEVENT_TYPE_UNPRESSKEY));
195
196 xevent.SetVKCode((XDWORD)keydata->vkCode);
197 xevent.SetScanCode((XWORD)keydata->scanCode);
198 xevent.SetFlags((XDWORD)keydata->flags);
199
200 capture->PostEvent(&xevent);
201 if(xevent.IsKeyLocked())
202 {
203 return 1;
204 }
205 else
206 {
207 return 0;
208 }
209 }
210
211 return CallNextHookEx(NULL, ncode, wparam, lparam);
212}
