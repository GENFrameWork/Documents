394bool XLINUXDIR::NextSearch(XDIRELEMENT* searchelement)
395{
396 if(!searchelement) return false;
397
398 DIR* dir = (DIR*)searchelement->GetHandle();
399 struct dirent* entry = (struct dirent*)searchelement->GetFindFileData();
400 bool status = false;
401 XSTRING name;
402 XBUFFER patternsearch;
403
404 while(entry != NULL)
405 {
406 name.ConvertFromUTF8((XBYTE*)entry->d_name, strlen(entry->d_name) /*_MAXPATH*/ );
407
408 if((name.Compare(__L("."))) && (name.Compare(__L(".."))))
409 {
410 searchelement->GetPatternSearch()->ConvertToUTF8(patternsearch);
411 if(!fnmatch((char*)patternsearch.Get(), entry->d_name,0))
412 {
413 searchelement->SetType(XDIRELEMENTTYPE_NONE);
414 searchelement->GetNameFile()->Set(name.Get());
415
416 XPATH xpath;
417
418 xpath = searchelement->GetPathSearch()->Get();
419 xpath.Slash_Add();
420 xpath += searchelement->GetNameFile()->Get();
421
422 TypeOfEntry(xpath.Get(), searchelement);
423
424 status = true;
425 }
426 }
427
428 entry = readdir(dir);
429 searchelement->SetFindFileData((void*)entry);
430
431 if(status) break;
432 }
433
434 if(!status) closedir (dir);
435
436 return status;
437}
