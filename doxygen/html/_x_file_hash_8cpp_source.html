110bool XFILEHASH::Open(XPATH& pathname, bool readonly, bool checkhash, bool checkversion)
111{
112 if(!hash) return false;
113 if(!file) return false;
114
115 if(!hash->GetDefaultSize()) return false;
116
117 if(file->Exist(pathname)!=true) return false;
118
119 bool status = file->Open(pathname,readonly);
120
121 XSTRING IDini(16);
122 XWORD _ID = 0;
123 XWORD _type = 0;
124 XWORD _version = 0;
125
126 if(!file->Read((XBYTE*)IDini.Get(), XSTRING::GetSize(XFILE_ID)*sizeof(XWORD)+1)) return false;
127
128 IDini.AdjustSize();
129
130 if(!file->Read((XBYTE*)&_ID, sizeof(XWORD))) return false;
131 if(_ID!=ID) return false;
132
133 if(!file->Read((XBYTE*)&_type, sizeof(XWORD))) return false;
134 if(_type!=type) return false;
135
136 if(!file->Read((XBYTE*)&_version,sizeof(XWORD))) return false;
137 if(checkversion)
138 {
139 if(version!=_version) return false;
140 }
141
142 version = _version;
143
144 XDWORD sizeIDstring = 0;
145
146 if(!file->Read((XBYTE*)&sizeIDstring, sizeof(XDWORD))) return false;
147 if(sizeIDstring)
148 {
149 XWORD* IDstr = new XWORD[sizeIDstring+1];
150 if(IDstr)
151 {
152 memset((XBYTE*)IDstr, 0, sizeof(XWORD)*(sizeIDstring+1));
153
154 if(!file->Read((XBYTE*)IDstr, sizeof(XWORD)*sizeIDstring))
155 {
156 delete [] IDstr;
157 return false;
158 }
159
160 IDstring.Set((XWORD*)IDstr);
161
162 delete [] IDstr;
163 }
164 }
165
166 XBUFFER hashbuffer((XDWORD)hash->GetDefaultSize(), true);
167 if(!file->Read((XBYTE*)hashbuffer.Get(), hash->GetDefaultSize())) return false;
168
169 file->GetPosition(dataposition);

172

174

181 return false;
182 }
183 }
184

188
189 return status;
