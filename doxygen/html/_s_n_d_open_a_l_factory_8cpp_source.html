166bool SNDOPENALFACTORY::Update()
167{
168 if(!soundactive)
169 {
170 return false;
171 }
172
173 if(playmutex)
174 {
175 playmutex->Lock();
176 }
177
178 for(XDWORD c=0; c<soundplayitems.GetSize(); c++)
179 {
180 SNDOPENALPLAYITEM* playitem = soundplayitems.Get(c);
181 if(playitem)
182 {
183 SNDITEM* item = playitem->GetItem();
184 SNDOPENALSOURCE* source = playitem->GetSource();
185 if(item && source)
186 {
187 SNDNOTE* soundnote = NULL;
188 SNDFILE* soundfile = NULL;
189
190 switch(item->GetType())
191 {
192 case SNDITEM_TYPE_UNKNOWN : break;
193
194 case SNDITEM_TYPE_NOTE : soundnote = playitem->GetItem()->GetSoundNote();
195 break;
196
197 case SNDITEM_TYPE_FILE : soundfile = playitem->GetItem()->GetSoundFile();
198 break;
199 }
200
201 if(!soundnote && !soundfile)
202 {
203 continue;
204 }
205
206 if(playitem->GetEvent() == SNDOPENALPLAYITEM_XFSMEVENT_NONE) // Not new event
207 {
208 switch(playitem->GetCurrentState())
209 {
210 case SNDOPENALPLAYITEM_XFSMSTATE_NONE : break;
211 case SNDOPENALPLAYITEM_XFSMSTATE_INI : break;
212
213 case SNDOPENALPLAYITEM_XFSMSTATE_PLAY : { XQWORD millisecond = item->GetTimerPlay()->GetMeasureMilliSeconds();
214 bool outoftime = false;
215
216 item->SetPlayingTime((XDWORD)item->GetPlayingTime() + (XDWORD)millisecond);
217 item->SetCurrentPlayingTime((XDWORD)item->GetCurrentPlayingTime() + (XDWORD)millisecond);
218
219 item->GetTimerPlay()->Reset();
220
221 if(item->GetNTimesToPlay() == SNDFACTORY_INLOOP)
222 {
223 if(!source->IsPLaying())
224 {
225 outoftime = true;
226 }
227 }
228 else
229 {
230 if(item->GetCurrentPlayingTime() >= (item->GetDuration()))
231 {
232 outoftime = true;
233 }
234 }
235
236 if(outoftime)
237 {
238 item->SetCurrentPlayingTime(0);
239
240 if(!item->GetCounterPlay())
241 {
242 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_STOP);
243 item->AddOneNTimesPlayed();
244 }
245 else
246 {
247 item->SetCounterPlay(item->GetCounterPlay()-1);
248
249 item->GetTimerPlay()->Reset();
250
251 source->ResetPlay();
252 source->Play();
253 }
254 }
255 }
256 break;
257
258 case SNDOPENALPLAYITEM_XFSMSTATE_PAUSE : break;
259
260 case SNDOPENALPLAYITEM_XFSMSTATE_STOP : break;
261
262 case SNDOPENALPLAYITEM_XFSMSTATE_END : break;
263 }
264 }
265 else // New event
266 {
267 if(playitem->GetEvent() < SNDOPENALPLAYITEM_LASTEVENT)
268 {
269 playitem->CheckTransition();
270
271 switch(playitem->GetCurrentState())
272 {
273 case SNDOPENALPLAYITEM_XFSMSTATE_NONE : break;
274
275 case SNDOPENALPLAYITEM_XFSMSTATE_INI : { switch(item->GetType())
276 {
277 case SNDITEM_TYPE_UNKNOWN : break;
278
279 case SNDITEM_TYPE_NOTE : GenerateBuffer(playitem, soundnote);
280 break;
281
282 case SNDITEM_TYPE_FILE : GenerateBuffer(playitem, soundfile);
283 break;
284 }
285
286 if(item->GetNTimesToPlay())
287 {
288 item->SetCounterPlay(item->GetNTimesToPlay()-1);
289 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_PLAY);
290 }
291
292 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_INI);
293 xevent.SetItem(item);
294 PostEvent(&xevent);
295 }
296 break;
297
298 case SNDOPENALPLAYITEM_XFSMSTATE_PLAY : { item->GetTimerPlay()->Reset();
299 source->Play();
300 item->SetStatus(SNDITEM_STATUS_PLAY);
301
302 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_PLAY);
303 xevent.SetItem(item);
304 PostEvent(&xevent);
305 }
306 break;
307
308 case SNDOPENALPLAYITEM_XFSMSTATE_PAUSE : { source->Pause();
309 item->SetStatus(SNDITEM_STATUS_PAUSE);
310
311 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_PAUSE);
312 xevent.SetItem(item);
313 PostEvent(&xevent);
314 }
315 break;
316
317 case SNDOPENALPLAYITEM_XFSMSTATE_STOP : { source->Stop();
318
319 item->SetStatus(SNDITEM_STATUS_STOP);
320
321 if(item->GetNTimesToPlay() == SNDFACTORY_INLOOP)
322 {
323 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_END);
324 }
325 else
326 {
327 if(!item->GetCounterPlay())
328 {
329 playitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_END);
330 }
331 }
332
333 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_STOP);
334 xevent.SetItem(item);
335 PostEvent(&xevent);
336
337 }
338 break;
339
340 case SNDOPENALPLAYITEM_XFSMSTATE_END : { if(playmutex)
341 {
342 playmutex->Lock();
343 }
344
345 soundplayitems.Delete(playitem);
346 delete playitem;
347
348 if(playmutex)
349 {
350 playmutex->UnLock();
351 }
352
353 item->SetStatus(SNDITEM_STATUS_END);
354
355 SNDFACTORY_XEVENT xevent(this, SNDFACTORY_XEVENT_TYPE_SOUND_END);
356 xevent.SetItem(item);
357 PostEvent(&xevent);
358 }
359 break;
360 }
361 }
362 }
363 }
364 }
365 }
366
367 if(playmutex)
368 {
369 playmutex->UnLock();
370 }
371
372 return true;
373}

442bool SNDOPENALFACTORY::Sound_Play(SNDITEM* item, SNDPLAYCFG* playCFG, int ntimestoplay)
443{
444 if(!soundactive)
445 {
446 return false;
447 }
448
449 if(!SNDFACTORY::Sound_Play(item, playCFG, ntimestoplay))
450 {
451 return false;
452 }
453
454 if(soundplayitems.GetSize() >= maxchannels)
455 {
456 return false;
457 }
458
459 SNDOPENALSOURCE* source = new SNDOPENALSOURCE();
460 if(!source)
461 {
462 return false;
463 }
464
465 SNDOPENALPLAYITEM* soundplayitem = new SNDOPENALPLAYITEM();
466 if(!soundplayitem)
467 {
468 delete source;
469 return false;
470 }
471
472 soundplayitem->SetItem(item);
473 soundplayitem->SetSource(source);
474
475 if(!soundplayitems.Add(soundplayitem))
476 {
477 return false;
478 }
479
480 if(item->GetPlayCFG()->GetVolume() != SNDFACTORY_UNDEFINED)
481 {
482 source->SetVolume(item->GetPlayCFG()->GetVolume());
483 }
484
485 if(item->GetPlayCFG()->GetPitch() != SNDFACTORY_UNDEFINED)
486 {
487 source->SetPitch(item->GetPlayCFG()->GetPitch());
488 }
489
490 if(ntimestoplay == SNDFACTORY_INLOOP)
491 {
492 source->SetInLoop(true);
493 }
494 else
495 {
496 item->SetPlayingTime(0);
497 source->SetInLoop(false);
498 }
499
500 soundplayitem->SetEvent(SNDOPENALPLAYITEM_XFSMEVENT_INI);
501
502 return true;
503}

652bool SNDOPENALFACTORY::Sound_WaitToEnd(SNDITEM* item, int maxtimeout, SNDFACTORY_WAITFUNCTION waitfunction)
653{
654 if(!soundactive)
655 {
656 return false;
657 }
658
659 if(!item)
660 {
661 return false;
662 }
663
664 XTIMER* timeout = NULL;
665 bool status = true;
666
667 timeout = GEN_XFACTORY.CreateTimer();
668 if(!timeout)
669 {
670 return false;
671 }
672
673 timeout->Reset();
674
675 while(item->GetStatus() != SNDITEM_STATUS_END)
676 {
677 if(maxtimeout)
678 {
679 if(timeout->GetMeasureMilliSeconds() >= maxtimeout)
680 {

685

687 {
688 if(!waitfunction(item))
689 {
690 status = false;
691 break;
692 }
693 }
694 else
695 {
696 GEN_XSLEEP.MilliSeconds(100);
697 }
698 }
699
700 GEN_XFACTORY.DeleteTimer(timeout);
701
702 if(waitfunction)
703 {
704 if(!waitfunction(item))
705 {
706 status = false;
707 }
708 }
709
