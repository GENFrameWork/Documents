620bool XDRIVEIMAGEMANAGER::ReadToIMGFile(XDRIVEIMAGEMANAGER_DRIVE& drive, XPATH& pathfileIMG, XQWORD nsectorsperblock)
621{
622 XFILE* xfileIMG;
623 XBUFFER readbuffer;
624 XQWORD nsectorstoread = nsectorsperblock;
625 XQWORD nsectorsread = 0;
626 bool status = 0;
627
628 if(!drive.GetSize()) return status;
629
630 GEN_XFACTORY_CREATE(xfileIMG, Create_File())
631 if(xfileIMG)
632 {
633 if(xfileIMG->Create(pathfileIMG))
634 {
635 XQWORD indexsector=0;
636
637 while(indexsector<drive.GetNumberSectors())
638 {
639 if((indexsector + nsectorstoread) > drive.GetNumberSectors()) nsectorstoread = (drive.GetNumberSectors() - indexsector);
640
641 XDRIVEIMAGEMANAGER_XEVENT xevent(this, XDRIVEIMAGEMANAGER_XEVENT_TYPE_READTOIMGFILE);
642
643 xevent.SetDrive(&drive);
644 xevent.SetIndexSector(indexsector);
645 xevent.SetNumberSectorsInOperation(nsectorstoread);
646 xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_NONE);
647
648 status = Read(drive, indexsector, nsectorstoread, readbuffer);
649 if(!status) xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_READSECTORS);
650
651 if(status)
652 {
653 status = xfileIMG->Write(readbuffer);
654 if(!status) xevent.SetError(XDRIVEIMAGEMANAGER_ERROR_WRITEIMGFILE);
655 }
656
657 if(status) nsectorsread += nsectorstoread;
658
659 lastoperationpercent = (((float)nsectorsread*100)/(float)drive.GetNumberSectors());
660
661 xevent.SetError(lasterror);
662 xevent.SetOperationPercent(lastoperationpercent);
663
664 PostEvent(&xevent);
665
666 if(!status) break;
667
668 indexsector += nsectorstoread;
669 }
670
671 xfileIMG->Close();
672 }
673
674 GEN_XFACTORY.Delete_File(xfileIMG);
675 }
676
677 return status;
678}
