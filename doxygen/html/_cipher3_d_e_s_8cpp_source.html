104bool CIPHER3DES::Cipher(XBYTE* input, XDWORD size)
105{
106 if(!size) return false;
107
108 XBUFFER inputpadding;
109 CIPHER3DES_CONTEXT ctx;
110 XBYTE iv[8];
111
112 memset(iv,0,8);
113 if(inivector && inivector->GetSize()>=8) memcpy(iv, inivector->Get() ,8);
114
115 inputpadding.Add(input, size);
116 inputpadding.Padding_Add(paddingadjustsize, paddingtype);
117
118 result->Delete();
119 result->Add(input, size);
120 result->Padding_Add(paddingadjustsize, paddingtype);
121
122 CIPHERKEYSYMMETRICAL* key = (CIPHERKEYSYMMETRICAL*)GetKey();
123 if(!key) return false;
124 if(!key->Get()->Get()) return false;
125
126 if(key->Get()->GetSize() <= 16)
127 DES3Set2KeyCipher(&ctx, (XBYTE *)key->Get()->Get());
128 else DES3Set3KeyCipher(&ctx, (XBYTE *)key->Get()->Get());
129
130 bool status;
131
132 if(this->GetChainingMode() == CIPHERCHAININGMODE_ECB)
133 status = DES3Cipher_ECB(&ctx, CIPHERDES_ENCRYPT, result->GetSize(), iv, inputpadding.Get(), result->Get());
134 else status = DES3Cipher_CBC(&ctx, CIPHERDES_ENCRYPT, result->GetSize(), iv, inputpadding.Get(), result->Get());
135
136 return status;
137}
