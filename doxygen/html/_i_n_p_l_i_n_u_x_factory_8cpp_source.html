197bool INPLINUXFACTORY::GetDeviceHandlers(INPDEVICE_TYPE devicetype, XVECTOR<INPLINUXDEVICEID*>& devices)
198{
199 bool status = false;
200
201 XFILETXT* xfiletxtdevices = new XFILETXT();
202 if(xfiletxtdevices)
203 {
204 XPATH xpathdevices;
205
206 xpathdevices = INPLINUXDEVICEID_HANDLEFILE;
207
208 if(xfiletxtdevices->Open(xpathdevices))
209 {
210 status = xfiletxtdevices->ReadAllFile();
211
212 for(int c=0; c<xfiletxtdevices->GetNLines(); c++)
213 {
214 XSTRING* line = xfiletxtdevices->GetLine(c);
215 if(line)
216 {
217 int index = line->Find(__L("I:"), true);
218 if(index == 0)
219 {
220 INPLINUXDEVICEID* deviceID = new INPLINUXDEVICEID();
221 if(deviceID)
222 {
223 do{ line = xfiletxtdevices->GetLine(c);
224 if(!line) break;
225 if(line->GetSize() < 2) break;
226
227 char typeline = (char)line->Get()[0];
228 switch(typeline)
229 {
230 case 'I' : break;
231
232 case 'N' : { XSTRING name;
233
234 name.AdjustSize(_MAXSTR);
235 line->UnFormat(__L("N: Name=\"%s\""), name.Get());
236 name.AdjustSize();
237
238 deviceID->GetName()->Set(name);
239 }
240 break;
241
242 case 'P' : break;
243 case 'S' : break;
244 case 'U' : break;
245
246 case 'H' : { XSTRING handlers;
247 int event_index = INPLINUXDEVICEID_INVALID;
248
249 handlers.AdjustSize(_MAXSTR);
250 line->UnFormat(__L("H: Handlers=%s"), handlers.Get());
251 handlers.AdjustSize();
252
253 index = handlers.Find(__L("event"), true);
254 if(index != XSTRING_NOTFOUND)
255 {
256 XSTRING eventstr;
257
258 handlers.Copy(index, eventstr);
259 eventstr.UnFormat(__L("event%d"), &event_index);
260
261 deviceID->SetEventIndex(event_index);
262 }
263 }
264 break;
265
266 case 'B' : { XCHAR* prefix[] = { __L("PROP=") ,
267 __L("EV=") ,
268 __L("KEY=") ,
269 __L("ABS=") ,
270 __L("REL=") ,
271 __L("MSC=")
272 };
273
274 for(int d=0; d< (sizeof(prefix) / sizeof(XCHAR*)); d++)
275 {
276 index = line->Find(prefix[d], true);
277 if(index != XSTRING_NOTFOUND)
278 {
279 switch(d)
280 {
281 case 0 : break;
282
283 case 1 : { XSTRING ev_str;
284
285 ev_str.AdjustSize(_MAXSTR);
286 line->Copy(index+3, ev_str);
287 ev_str.AdjustSize();
288
289 if(!ev_str.Compare(__L("120013"), true)) deviceID->SetType(INPDEVICE_TYPE_KEYBOARD);
290 if(!ev_str.Compare(__L("17"), true)) deviceID->SetType(INPDEVICE_TYPE_MOUSE);
291 if(!ev_str.Compare(__L("b"), true)) deviceID->SetType(INPDEVICE_TYPE_TOUCHSCREEN);
292 }
293 break;
294
295 case 2 : break;
296 case 3 : break;
297 case 4 : break;
298 case 5 : break;
299 }
300 }
301 }
302 }
303 break;
304 }
305
306 c++;
307
308
309 } while(1);
310
311
312 if((deviceID->GetEventIndex() != INPLINUXDEVICEID_INVALID) && (devicetype == deviceID->GetType()))
313 {
314 devices.Add(deviceID);
315
316 #ifdef XTRACE_ACTIVE
317 XSTRING typestr;
318
319 switch(deviceID->GetType())
320 {
321 default :
322 case INPDEVICE_TYPE_NONE : typestr = __L("Unkwown"); break;
323 case INPDEVICE_TYPE_KEYBOARD : typestr = __L("Keyboard"); break;
324 case INPDEVICE_TYPE_MOUSE : typestr = __L("Mouse"); break;
325 case INPDEVICE_TYPE_JOSTICK : typestr = __L("Jostick"); break;
326 case INPDEVICE_TYPE_TOUCHSCREEN : typestr = __L("TouchScreen"); break;
327 case INPDEVICE_TYPE_WIIMOTE : typestr = __L("WII Mote"); break;
328 }
329
330 XTRACE_PRINTCOLOR(XTRACE_COLOR_BLUE, __L("[Input Device] Event [%d] Type: %15s Device [%s]"), deviceID->GetEventIndex(), typestr.Get(), deviceID->GetName()->Get());
331 #endif
332 }
333 else delete deviceID;
334 }
335 }
336 }
337 }
338
339 xfiletxtdevices->Close();
340 }
341
342 delete xfiletxtdevices;
343 }
344
345 return status;
346}
