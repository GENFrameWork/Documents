338XDWORD DIOSTREAMICMP::Read(XBYTE* buffer, XDWORD size)
339{
340 XDWORD esize = 0;
341
342 int index = GetFirstDatagram(false);
343 if(index != DIOSTREAMICMP_NOTFOUND)
344 {
345 DIOSTREAMICMPDATAGRAM* datagram = GetDatagram(index);
346 if(datagram)
347 {
348 esize = datagram->GetData()->GetSize();
349 if(esize >= size)
350 {
351 esize = size;
352
353 memcpy(buffer,(XBYTE*)datagram->GetData()->Get(), size);
354
355 nbytesread+= size;
356 if(xtimernotactivity) xtimernotactivity->Reset();
357
358 DeleteDatagram(index);
359
360 inbuffer->Extract(buffer, 0, size);
361 }
362 }
363 }
364
365 return esize;
366}

693int DIOSTREAMICMP::GetFirstDatagram(bool tosend)
694{
695 int index = DIOSTREAMICMP_NOTFOUND;
696
697 if(datagramsmutex) datagramsmutex->Lock();
698
699 for(int c=0;c<(int)datagrams.GetSize();c++)
