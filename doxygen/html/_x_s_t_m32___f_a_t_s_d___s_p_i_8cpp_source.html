235DSTATUS XSTM32_FATSD_SPI::Disk_Initialize(XBYTE drive)
236{
237 XBYTE n;
238 XBYTE type;
239 XBYTE ocr[4];
240
241 if(drive) return STA_NOINIT;
242 if(status & STA_NODISK) return status;
243
244 // SD Power On
245 SD_PowerOn();
246
247 // SPI Chip Select
248 CS_Select();
249
250 type = 0;
251
252 // Idle
253 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD0, 0) == 1)
254 {
255 timer1 = 100;
256
257 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD8, 0x1AA) == 1)
258 {
259 // SDC Ver2+
260 for(n=0; n<4; n++)
261 {
262 ocr[n] = SPI_ReadByte();
263 }
264
265 if(ocr[2] == 0x01 && ocr[3] == 0xAA)
266 {
267 // 2.7-3.6V
268 do{ if(SD_SendCmd(XSTM32_FATSD_SPI_CMD55, 0) <= 1 && SD_SendCmd(XSTM32_FATSD_SPI_CMD41, 1UL << 30) == 0) break; // ACMD41 with HCS bit
269
270 } while (timer1);
271
272 if(timer1 && SD_SendCmd(XSTM32_FATSD_SPI_CMD58, 0) == 0)
273 {
274 // Check CCS bit
275 for(n=0; n<4; n++)
276 {
277 ocr[n] = SPI_ReadByte();
278 }
279
280 type = (ocr[0] & 0x40) ? 6 : 2;
281 }
282 }
283 }
284 else
285 {
286 // SDC Ver1 or MMC
287 type = (SD_SendCmd(XSTM32_FATSD_SPI_CMD55, 0) <= 1 && SD_SendCmd(XSTM32_FATSD_SPI_CMD41, 0) <= 1) ? 2 : 1; // SDC : MMC
288
289 do{ if(type == 2)
290 {
291 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD55, 0) <= 1 && SD_SendCmd(XSTM32_FATSD_SPI_CMD41, 0) == 0) break; // ACMD41
292 }
293 else
294 {
295 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD1, 0) == 0) break; // CMD1
296 }
297
298 } while (timer1);
299
300 if(!timer1 || SD_SendCmd(XSTM32_FATSD_SPI_CMD16, 512) != 0)
301 {
302 type = 0;
303 }
304 }
305 }
306
307 cardtype = type;
308
309 CS_Deselect();
310
311 SPI_ReadByte(); // Idle (Release DO)
312
313 if(type)
314 {
315 // Clear STA_NOINIT
316 status &= ~STA_NOINIT;
317 }
318 else
319 {
320 // Initialization failed
321 SD_PowerOff();
322 }
323
324 return status;
325}

364DRESULT XSTM32_FATSD_SPI::Disk_Read(XBYTE pdrive, XBYTE* buffer, XDWORD sector, XDWORD count)
365{
366 if(pdrive || !count) return RES_PARERR;
367 if(status & STA_NOINIT) return RES_NOTRDY;
368
369 if (!(cardtype & 4)) sector *= 512; // sector Byte addresultsing
370
371 CS_Select();
372
373 if(count == 1)
374 {
375 if ((SD_SendCmd(XSTM32_FATSD_SPI_CMD17, sector) == 0) && SD_ReadDataBlock(buffer, 512)) count = 0;
376 }
377 else
378 {
379 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD18, sector) == 0)
380 {
381 do{ if(!SD_ReadDataBlock(buffer, 512)) break;
382
383 buffer += 512;
384
385 } while (--count);
386
387 // STOP_TRANSMISSION
388 SD_SendCmd(XSTM32_FATSD_SPI_CMD12, 0);
389 }
390 }
391
392 CS_Deselect();
393
394 SPI_ReadByte(); // Idle Release DO
395
396 return count?RES_ERROR:RES_OK;
397}

415DRESULT XSTM32_FATSD_SPI::Disk_Write(XBYTE pdrive, const XBYTE* buffer, XDWORD sector, XDWORD count)
416{
417 if(pdrive || !count) return RES_PARERR;
418 if(status & STA_NOINIT) return RES_NOTRDY;
419
420 if(status & STA_PROTECT) return RES_WRPRT;
421
422 if(!(cardtype & 4)) sector *= 512; /* 지정 sector를 Byte addresultsing 단위로 변경 */
423
424 CS_Select();
425
426 if(count == 1)
427 {
428 if((SD_SendCmd(XSTM32_FATSD_SPI_CMD24, sector) == 0) && SD_WriteDataBlock(buffer, 0xFE)) count = 0;
429 }
430 else
431 {
432 if(cardtype & 2)
433 {
434 SD_SendCmd(XSTM32_FATSD_SPI_CMD55, 0);
435 SD_SendCmd(XSTM32_FATSD_SPI_CMD23, count); /* ACMD23 */
436 }
437
438 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD25, sector) == 0)
439 {
440 do{ if(!SD_WriteDataBlock(buffer, 0xFC)) break;
441
442 buffer += 512;
443
444 } while (--count);
445
446 if(!SD_WriteDataBlock(0, 0xFD))
447 {
448 count = 1;
449 }
450 }
451 }
452
453 CS_Deselect();
454
455 SPI_ReadByte();
456
457 return (count?RES_ERROR:RES_OK);
458}

476DRESULT XSTM32_FATSD_SPI::Disk_IOTCL(XBYTE drive, XBYTE ctrl, void* buffer)
477{
478 DRESULT result;
479 XBYTE n;
480 XBYTE csd[16];
481 XBYTE* ptr = (XBYTE*)buffer;
482 XWORD csize;
483
484 if(drive) return RES_PARERR;
485
486 result = RES_ERROR;
487
488 if(ctrl == CTRL_POWER)
489 {
490 switch(*ptr)
491 {
492 case 0 : if(SD_CheckPower()) SD_PowerOff(); // Power Off
493 result = RES_OK;
494 break;
495
496 case 1 : SD_PowerOn(); // Power On
497 result = RES_OK;
498 break;
499
500 case 2 : *(ptr + 1) = (XBYTE) SD_CheckPower();
501 result = RES_OK; // Power Check
502 break;
503
504 default : result = RES_PARERR;
505 }
506 }
507 else
508 {
509 if(status & STA_NOINIT) return RES_NOTRDY;
510
511 CS_Select();
512
513 switch(ctrl)
514 {
515 case GET_SECTOR_COUNT : // SD Sector (DWORD)
516 if((SD_SendCmd(XSTM32_FATSD_SPI_CMD9, 0) == 0) && SD_ReadDataBlock(csd, 16))
517 {
518 if((csd[0] >> 6) == 1)
519 {
520 // SDC ver 2.00
521 csize = csd[9] + ((WORD) csd[8] << 8) + 1;
522 *(DWORD*) buffer = (DWORD) csize << 10;
523 }
524 else
525 {
526 // MMC or SDC ver 1.XX
527 n = (csd[5] & 15) + ((csd[10] & 128) >> 7) + ((csd[9] & 3) << 1) + 2;
528 csize = (csd[8] >> 6) + ((WORD) csd[7] << 2) + ((WORD) (csd[6] & 3) << 10) + 1;
529 *(DWORD*) buffer = (DWORD) csize << (n - 9);
530 }
531
532 result = RES_OK;
533 }
534 break;
535
536 case GET_SECTOR_SIZE : // (WORD)
537 *(WORD*) buffer = 512;
538 result = RES_OK;
539 break;
540
541 case CTRL_SYNC : if(SD_ReadyWait() == 0xFF)
542 result = RES_OK;
543 break;
544
545 case MMC_GET_CSD : // CSD (16 bytes)
546 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD9, 0) == 0 && SD_ReadDataBlock(ptr, 16)) result = RES_OK;
547 break;
548
549 case MMC_GET_CID : // CID (16 bytes)
550 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD10, 0) == 0 && SD_ReadDataBlock(ptr, 16)) result = RES_OK;
551 break;
552
553 case MMC_GET_OCR : // OCR (4 bytes)
554 if(SD_SendCmd(XSTM32_FATSD_SPI_CMD58, 0) == 0)
555 {
556 for (n = 0; n < 4; n++)
557 {
558 *ptr++ = SPI_ReadByte();
559 }
560
561 result = RES_OK;
562 }
563
564 default : result = RES_PARERR;
565 }
566
567 CS_Deselect();
568
569 SPI_ReadByte();
570 }
571
572 return result;
573}

725void XSTM32_FATSD_SPI::SD_PowerOn()
726{
727 XBYTE cmd_arg[6];
728 XDWORD count = 0x1FFF;
729
730 CS_Deselect();
731
732 for(int c=0; c<10; c++)
733 {
734 SPI_WriteByte(0xFF);
735 }
736
737 // SPI Chips Select
738 CS_Select();
739
740 // GO_IDLE_STATE
741 cmd_arg[0] = (XSTM32_FATSD_SPI_CMD0 | 0x40);
742 cmd_arg[1] = 0;
743 cmd_arg[2] = 0;
744 cmd_arg[3] = 0;
745 cmd_arg[4] = 0;
746 cmd_arg[5] = 0x95;
747
748
749 for(int c=0; c<6; c++)
750 {
751 SPI_WriteByte(cmd_arg[c]);
752 }
753
754 while((SPI_ReadByte() != 0x01) && count)
755 {
756 count--;
757 }
758
759 CS_Deselect();
760
761 SPI_WriteByte(0XFF);
762

852bool XSTM32_FATSD_SPI::SD_WriteDataBlock(const XBYTE* buffer, XBYTE token)
853{
854 XBYTE resultponse;
855 XBYTE wc;
856 XBYTE i = 0;
857
858 // SD
859 if (SD_ReadyWait() != 0xFF) return FALSE;
860
861 SPI_WriteByte(token);
862
863 if(token != 0xFD)
864 {
865 wc = 0;
866
867 // 512
868 do{ SPI_WriteByte(*buffer++);
869 SPI_WriteByte(*buffer++);
870
871 } while (--wc);
872
873 SPI_ReadByte(); // CRC
874 SPI_ReadByte();
875
876 while(i <= 64)
877 {

883 }

887 }
888
889 if((resultponse & 0x1F) == 0x05) return TRUE;

907XBYTE XSTM32_FATSD_SPI::SD_SendCmd(XBYTE cmd, XDWORD arg)
908{
909 XBYTE crc;
910 XBYTE resultult;
911
912 // SD
913 if(SD_ReadyWait() != 0xFF) return 0xFF;
914
915 SPI_WriteByte(cmd); // Command
916 SPI_WriteByte((XBYTE) (arg >> 24)); // Argument[31..24]
917 SPI_WriteByte((XBYTE) (arg >> 16)); // Argument[23..16]
918 SPI_WriteByte((XBYTE) (arg >> 8)); // Argument[15..8]
919 SPI_WriteByte((XBYTE) arg); // Argument[7..0]
920
921 // CRC
922 crc = 0;
923 if(cmd == XSTM32_FATSD_SPI_CMD0) crc = 0x95; // CRC for CMD0(0)
924 if(cmd == XSTM32_FATSD_SPI_CMD8) crc = 0x87; // CRC for CMD8(0x1AA)
925
926 // CRC
927 SPI_WriteByte(crc);
928
929 // CMD12 Stop Reading
930 if (cmd == XSTM32_FATSD_SPI_CMD12) SPI_ReadByte();
931
932 XBYTE n = 10;
933 do{
934 resultult = SPI_ReadByte();
935
936 } while ((resultult & 0x80) && --n);
937
938 return resultult;
939}
