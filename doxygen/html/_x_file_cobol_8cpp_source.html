153bool XFILECOBOL_PIC::ResolveMask()
154{
155 if(mask.IsEmpty())
156 {
157 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC Missing"));
158 return false;
159 }
160
161 hassign = false;
162 ndecimals = 0;
163 sizedata = 0;
164
165 XDWORD index = 0;
166
167 while(index < mask.GetSize())
168 {
169 switch(mask.Get()[index])
170 {
171 case __C('X') : type = XFILECOBOL_PIC_TYPE_BINARY;
172 maskdecondensed += __C('X');
173 sizedata++;
174 index++;
175 break;
176
177 case __C('A') : type = XFILECOBOL_PIC_TYPE_ALPHA;
178 maskdecondensed += __C('A');
179 sizedata++;
180 index++;
181 break;
182
183 case __C('9') : if(type != XFILECOBOL_PIC_TYPE_DECIMAL) type = XFILECOBOL_PIC_TYPE_NUMERIC;
184 if(type == XFILECOBOL_PIC_TYPE_DECIMAL) ndecimals++;
185 maskdecondensed += __C('9');
186 sizedata++;
187 index++;
188 break;
189
190 case __C(',') : type = XFILECOBOL_PIC_TYPE_DECIMAL;
191 maskdecondensed += __C(',');
192 sizedata++;
193 index++;
194 break;
195
196 case __C('S') : hassign = true;
197 maskdecondensed += __C('S');
198 index++;
199 break;
200
201 case __C('(') : { XSTRING modificator;
202 int sizemodificator;
203
204 int end = mask.Find(__L(")"), true, index+1);
205 if(end == XSTRING_NOTFOUND)
206 {
207 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Missing ')'"), mask.Get());
208 return false;
209 }
210
211 mask.Copy(index, end+1, modificator);
212 modificator.UnFormat(__L("(%d)"), &sizemodificator);
213
214 if(!sizemodificator)
215 {
216 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Modificador is zero '(0)'"), mask.Get());
217 return false;
218 }
219
220 for(int c=0; c<sizemodificator-1; c++)
221 {
222 maskdecondensed += mask.Get()[index-1];
223 sizedata++;
224 if(type == XFILECOBOL_PIC_TYPE_DECIMAL) ndecimals++;
225 }
226
227 index = end;
228 }
229 break;
230
231 case __C(')') : index++;
232 break;
233
234 case __C('V') : type = XFILECOBOL_PIC_TYPE_DECIMAL;
235 maskdecondensed += __C('V');
236 index++;
237 break;
238
239 default : XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Unknown Specifier '%c'"), mask.Get(), mask.Get()[index]);
240 return false;
241 break;
242 }
243 }
244
245 switch(computationaltype)
246 {
247 case 1 : // Float point 4 Bytes
248 break;
249
250 case 2 : // Float Point 8 Bytes
251 break;
252
253 case 3 : // BCD + Sing
254 sizedata = ((sizedata + 1) / 2) + (((sizedata + 1) % 2)?1:0);
255 break;
256
257 case 4 : break;
258
259 case 5 : // Conversion Packet number standard binary data
260 if((sizedata >= 1) && (sizedata <= 2))
261 {
262 sizedata = 1;
263 }
264 else
265 {
266 if((sizedata >=3) && (sizedata <= 4))
267 {
268 sizedata = 2;
269 }
270 else
271 {
272 if((sizedata >= 5) && (sizedata <= 9))
273 {
274 sizedata = 4;
275 }
276 else
277 {
278 if((sizedata >= 10) && (sizedata <= 18))
279 {
280 sizedata = 8;
281 }
282 else
283 {
284 XTRACE_PRINTCOLOR(4, __L("Cobol Reader: Error PIC invalid %s. Size of modificator to long %s > 18 "), mask.Get(), sizedata);
285 return false;
286 }
287 }
288 }
289 }
290 break;
291 }
292
293 return true;
294}

1107bool XFILECOBOL_RECORD::ConvertDataTo(XFILECOBOL_PIC* picture, XBUFFER& bufferdata, XSTRING& datastring)
1108{
1109 datastring.Empty();
1110
1111 if(!picture) return false;
1112 if(!picture->GetMask()) return false;
1113 if(!bufferdata.GetSize()) return false;
1114
1115 XSTRING mask;
1116
1117 mask = picture->GetMaskDecondensed()->Get();
1118 if(!mask.GetSize()) return false;
1119
1120 int indexdata = 0;
1121
1122 switch(picture->GetComputationalType())
1123 {
1124 case 0 : for(XDWORD c=0; c<mask.GetSize(); c++)
1125 {
1126 switch(mask.Get()[c])
1127 {
1128 case __C('S') : { XBYTE lastdata = bufferdata.GetLastByte();
1129 if(lastdata & XFILECOBOL_MASK_SIGN)
1130 {
1131 datastring += __L("-");
1132 lastdata &= ~XFILECOBOL_MASK_SIGN;
1133
1134 bufferdata.Get()[bufferdata.GetSize()-1] = lastdata;
1135 }
1136 }
1137 break;
1138
1139 case __C('9') :
1140 case __C('X') :
1141 case __C('A') : datastring.AddFormat(__L("%c"), bufferdata.Get()[indexdata]);
1142 indexdata++;
1143 break;
1144
1145 case __C('V') :
1146 case __C('.') : datastring += __L(".");
1147 break;
1148 }
1149 }
1150 break;
1151
1152 case 1 : break;
1153
1154 case 3 : { XBYTE lastdata = bufferdata.GetLastByte();
1155 lastdata &= 0x0F;
1156
1157 switch(lastdata)
1158 {
1159 case 0x0C : break;
1160 case 0x0D : datastring = __L("-");
1161 break;
1162 case 0x0F : break;
1163 }
1164
1165 lastdata>>=0x4;
1166
1167 for(XDWORD c=0; c< bufferdata.GetSize()-1; c++)
1168 {

1170 }
1171
1172 datastring.AddFormat(__L("%1X"), lastdata);
1173 }
1174 break;

1177 {
1178 case 1 : { char byte;
1179
1180 bufferdata.Get((XBYTE&)byte);
1181 datastring.Format(__L("%02d"), byte);
1182 }
1183 break;
1184
1185 case 2 : { short word;
1186

1189 }
1190 break;
1191
1192 case 4 : { int integer;
1193
1194 bufferdata.Get((XDWORD&)integer);
1195 datastring.Format(__L("%08d"), integer);
1196 }
1197 break;
1198
1199 case 8 : { long long large;
1200
1201 bufferdata.Get((XQWORD&)large);
1202 datastring.Format(__L("%016d"), large);
1203 }
