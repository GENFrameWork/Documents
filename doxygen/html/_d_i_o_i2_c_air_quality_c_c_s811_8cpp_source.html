409bool DIOI2CAIRQUALITYCCS811::SetEnvironmentalData(float relativehumidity, float temperature)
410{
411 if((temperature < -25) || (temperature > 50)) return false;
412 if((relativehumidity < 0) || (relativehumidity > 100)) return false;
413
414 XDWORD rhum = (XDWORD)(relativehumidity * 1000); //42.348 becomes 42348
415 XDWORD temp = (XDWORD)(temperature * 1000); //23.2 becomes 23200
416
417 XBYTE envdata[4];
418
419 //Split value into 7-bit integer and 9-bit fractional
420
421 //Incorrect way from datasheet.
422 //envData[0] = ((rH % 1000) / 100) > 7 ? (rH / 1000 + 1) << 1 : (rH / 1000) << 1;
423 //envData[1] = 0; //CCS811 only supports increments of 0.5 so bits 7-0 will always be zero
424 //if (((rH % 1000) / 100) > 2 && (((rH % 1000) / 100) < 8))
425 //{
426 // envData[0] |= 1; //Set 9th bit of fractional to indicate 0.5%
427 //}
428
429 //Correct rounding. See issue 8: https://github.com/sparkfun/Qwiic_BME280_CCS811_Combo/issues/8
430 envdata[0] = (rhum + 250) / 500;
431 envdata[1] = 0; //CCS811 only supports increments of 0.5 so bits 7-0 will always be zero
432
433 temp += 25000; //Add the 25C offset
434
435 //Split value into 7-bit integer and 9-bit fractional
436 //envData[2] = ((temp % 1000) / 100) > 7 ? (temp / 1000 + 1) << 1 : (temp / 1000) << 1;
437 //envData[3] = 0;
438 //if (((temp % 1000) / 100) > 2 && (((temp % 1000) / 100) < 8))
439 //{
440 // envData[2] |= 1; //Set 9th bit of fractional to indicate 0.5C
441 //}
442
443 //Correct rounding
444 envdata[2] = (temp + 250) / 500;
445 envdata[3] = 0;
446
447 return WriteRegister(DIOI2CAIRQUALITYCCS811_OFFSET_ENV_DATA, envdata, sizeof(envdata));
448}

493bool DIOI2CAIRQUALITYCCS811::ReadNTC(float& temperature, float& resistance)
494{
495 XBYTE data[4];
496 XWORD vrefcounts = 0;
497 XWORD ntccounts = 0;
498
499 memset(data, 0, sizeof(data));
500
501 bool status = ReadRegister(DIOI2CAIRQUALITYCCS811_OFFSET_NTC, data, sizeof(data));
502
503 vrefcounts = ((XWORD)data[0] << 8) | data[1];
504 ntccounts = ((XWORD)data[2] << 8) | data[3];
505 resistance = ((float)ntccounts * referenceresistance / (float)vrefcounts);
506
507 temperature = (float)log((long)resistance);
508 temperature = (float)(1 / (0.001129148 + (0.000234125 * temperature) + (0.0000000876741 * temperature * temperature * temperature)));
509 temperature = (float)(temperature - 273.15); // Convert Kelvin to Celsius
510
511 return status;
512}
