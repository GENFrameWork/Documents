409GRPBITMAPFILE_TYPE GRPBITMAPFILE::GetTypeFromExtenxion(XCHAR* xpath)
410{
411 XPATH _xpath;
412 XSTRING ext;
413 GRPBITMAPFILE_TYPE type = GRPBITMAPFILE_TYPE_UNKNOWN;
414
415 _xpath = xpath;
416
417 _xpath.GetExt(ext);
418
419 if(!ext.Compare(__L(".jpg"), true)) type = GRPBITMAPFILE_TYPE_JPG;
420 if(!ext.Compare(__L(".jpeg"), true)) type = GRPBITMAPFILE_TYPE_JPG;
421 if(!ext.Compare(__L(".png"), true)) type = GRPBITMAPFILE_TYPE_PNG;
422 if(!ext.Compare(__L(".bmp"), true)) type = GRPBITMAPFILE_TYPE_BMP;
423 if(!ext.Compare(__L(".tga"), true)) type = GRPBITMAPFILE_TYPE_TGA;
424
425 return type;
426}

441GRPBITMAP* GRPBITMAPFILE::Load(XCHAR* xpath, GRPPROPERTYMODE forcemode)
442{
443 if(!xpath) return NULL;
444
445 XPATH _xpath;

447 GRPBITMAPFILEBASE* grpfilebase = NULL;

451

455 default : break;
456

638GRPBITMAPSEQUENCE* GRPBITMAPFILE::LoadSequence(XCHAR* namemask, int inisequence, int endsequence)
639{
640 if(xpath.IsEmpty()) return NULL;
641
642 GRPBITMAPSEQUENCE* bitmapsequence = new GRPBITMAPSEQUENCE();
643 if(!bitmapsequence) return NULL;
644
645 XSTRING namefile;
646
647 //xpath.GetNamefileExt(namefile);
648 //if(!namefile.IsEmpty()) return NULL;
649
650 // The sequence is in individual files in a directory. Sequence numbers must be continuos
651 int nfiles = inisequence;
652
653 while((nfiles < endsequence)|| (endsequence < 0))
654 {
655 XPATH xpathtemp;
656 bool existfile = false;
657
658 namefile.Format(namemask, nfiles);
659 xpathtemp.Create(2, xpath.Get(), namefile.Get());
660
661 XFILE* GEN_XFACTORY_CREATE(xfile, Create_File())
662 if(xfile)
663 {
664 existfile = xfile->Exist(xpathtemp);
665 GEN_XFACTORY.Delete_File(xfile);
666 }
667
668 if(!existfile) break;
669
670 GRPBITMAP* bitmap = Load(xpathtemp);
671 if(!bitmap) break;
672
673 bitmapsequence->AddFrame(bitmap);
674
675 nfiles++;
676 }
677
678 return bitmapsequence;
679}
