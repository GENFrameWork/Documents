968void HASHWHIRLPOOL::Add(XBYTE* source, XDWORD sourceBits, WHIRLPOOL* structpointer)
969{
970 int sourcePos = 0; /* index of leftmost source XBYTE containing data (1 to 8 bits). */
971 int sourceGap = (8 - ((int)sourceBits & 7)) & 7; /* space on source[sourcePos]. */
972 int bufferRem = structpointer->bufferbits & 7; /* occupied bits on buffer[bufferPos]. */
973 int i;
974 XDWORD b;
975 XDWORD carry;
976 XBYTE* buffer = structpointer->buffer;
977 XBYTE* bitLength = structpointer->bitlength;
978 int bufferBits = structpointer->bufferbits;
979 int bufferPos = structpointer->bufferpos;
980
981 // tally the length of the added data:
982
983 XQWORD value = sourceBits;
984
985 for(i=31, carry=0; i>=0 && (carry!= 0 || value!= LL(0)); i--)
986 {
987 carry += bitLength[i] + ((XDWORD)value & 0xff);
988
989 bitLength[i] = (XBYTE)carry;
990
991 carry >>= 8;
992 value >>= 8;
993 }
994
995 // process data in chunks of 8 bits (a more efficient approach would be to take whole-word chunks):
996 while (sourceBits > 8)
997 {
998 // N.B. at least source[sourcePos] and source[sourcePos+1] contain data. */
999 // take a byte from the source:
1000
1001 b = ((source[sourcePos] << sourceGap) & 0xff) | ((source[sourcePos + 1] & 0xff) >> (8 - sourceGap));
1002
1003 // process this byte:
1004
1005 buffer[bufferPos++] |= (XBYTE)(b >> bufferRem);
1006 bufferBits += 8 - bufferRem; /* bufferBits = 8*bufferPos; */
1007
1008 if(bufferBits == HASHWHIRLPOOL_DIGESTBITS)
1009 {
1010 // process data block:
1011 ProcessBuffer(structpointer);
1012
1013 // reset buffer:
1014
1015 bufferBits = bufferPos = 0;
1016 }
1017
1018 buffer[bufferPos] = b << (8 - bufferRem);
1019 bufferBits += bufferRem;
1020
1021 // proceed to remaining data:
1022
1023 sourceBits -= 8;
1024 sourcePos++;
1025 }
1026
1027 // now 0 <= sourceBits <= 8;
1028 // furthermore, all data (if any is left) is in source[sourcePos].
1029
1030 if(sourceBits > 0)
1031 {
1032 b = (source[sourcePos] << sourceGap) & 0xff; /* bits are left-justified on b. */
1033
1034 // process the remaining bits:
1035
1036 buffer[bufferPos] |= b >> bufferRem;
1037
1038 }
1039 else
1040 {
1041 b = 0;
1042 }
1043
1044 if(bufferRem + sourceBits < 8)
1045 {
1046 // all remaining data fits on buffer[bufferPos],
1047 // and there still remains some space.
1048
1049 bufferBits += sourceBits;
1050 }
1051 else
1052 {
1053 // buffer[bufferPos] is full:
1054 bufferPos++;
1055 bufferBits += 8 - bufferRem;
1056 sourceBits -= 8 - bufferRem;
1057
1058 // now 0 <= sourceBits < 8;
1059 // furthermore, all data (if any is left) is in source[sourcePos].
1060 if(bufferBits == HASHWHIRLPOOL_DIGESTBITS)
1061 {
1062 // process data block:
1063 ProcessBuffer(structpointer);
1064
1065 // reset buffer:
1066 bufferBits = bufferPos = 0;
1067 }
1068
1069 buffer[bufferPos] = b << (8 - bufferRem);
1070 bufferBits += (int)sourceBits;
1071 }
1072
1073 structpointer->bufferbits = bufferBits;
1074 structpointer->bufferpos = bufferPos;
1075}
