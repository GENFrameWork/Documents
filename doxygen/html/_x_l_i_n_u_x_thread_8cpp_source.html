311void* XLINUXTHREAD::Callback(void* thread)
312{
313 XLINUXTHREAD* th = NULL;
314 XTHREADFUNCTION func = NULL;
315 void* param;
316
317 th = (XLINUXTHREAD*)thread;
318 if(!th) return NULL;
319
320 func = (XTHREADFUNCTION)th->GetFunction();
321 if(!func) return NULL;
322
323 param = th->GetParam();
324
325 th->gotofunction = true;
326
327 if((th->priority != XTHREADPRIORITY_UNKNOWN) &&
328 (th->priority != XTHREADPRIORITY_NORMAL))
329 {
330 struct sched_param param; // scheduling priority
331
332 int policy = SCHED_FIFO; // scheduling policy
333
334 switch(th->priority)
335 {
336 #ifdef SCHED_IDLE
337 case XTHREADPRIORITY_LOW : policy = SCHED_IDLE;
338 param.sched_priority = sched_get_priority_min(policy);
339 break;
340 #endif
341
342 case XTHREADPRIORITY_HIGH : policy = SCHED_FIFO;
343 param.sched_priority = sched_get_priority_max(policy)/2;
344 break;
345
346 case XTHREADPRIORITY_REALTIME : policy = SCHED_FIFO;
347 param.sched_priority = sched_get_priority_max(policy);
348 break;
349
350 default : break;
351 }
352
353 pthread_setschedparam(th->thhandle, policy, &param);
354 }
355
356 while(1)
357 {
358 if(!th) break;
359
360 if(th->statusfunc == XTHREADSTATUS_EXIT) break;
361 if(th->statusfunc == XTHREADSTATUS_END) break;
362
363 if(th->statusfunc == XTHREADSTATUS_RUN)
364 {
365 if(func)
366 {
367 th->isinfunction = true;
368 func(param);
369 th->isinfunction = false;
370
371 } else break;
372 }
373
374 if(!th) break;
375
376 if(th->waityield) usleep(th->waityield*1000);
377 }
378
379 if(th) th->statusfunc = XTHREADSTATUS_END;
380
381 pthread_exit(0);
382
383 return NULL;
384}
