189bool XSTM32SYSTEM::ShutDown(XSYSTEM_CHANGESTATUSTYPE type)
190{
191 switch(type)
192 {
193 case XSYSTEM_CHANGESTATUSTYPE_REBOOT :
194 #if defined(STM32F407xx) || defined(STM32F411xE) || defined(STM32F446x)
195 { __DSB(); // Ensure all outstanding memory accesses included buffered write are completed before reset
196 SCB->AIRCR = ((0x5FA << SCB_AIRCR_VECTKEY_Pos) | (SCB->AIRCR & SCB_AIRCR_PRIGROUP_Msk) | SCB_AIRCR_SYSRESETREQ_Msk); // Keep priority group unchanged
197 __DSB(); // Ensure completion of memory access
198 while(1); // Wait until reset
199 }
200 #endif
201 break;
202
203 case XSYSTEM_CHANGESTATUSTYPE_SESSION_LOGOFF : break;
204 case XSYSTEM_CHANGESTATUSTYPE_POWEROFF : break;
