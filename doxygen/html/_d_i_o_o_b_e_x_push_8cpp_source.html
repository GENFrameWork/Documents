423bool DIOOBEXPUSH::Ini(int timeout)
424{
425 if(!diostream) return false;
426 if(!diostream->GetConfig()) return false;
427
428 sendbuffer = new XBUFFER();
429 GEN_XFACTORY_CREATE(xfile, Create_File())
430 GEN_XFACTORY_CREATE(xtimer, CreateTimer());
431
432 issending = false;
433
434 if(xfile && sendbuffer && xtimer)
435 {
436 if(diostream->Open())
437 {
438 if(diostream->WaitToConnected(timeout))
439 {
440 xthreadobex = CREATEXTHREAD(XTHREADGROUPID_DIOSTREAM, __L("DIOOBEXPUSH::Ini"),ThreadSend,(void*)this);
441 if(xthreadobex)
442 {
443 xthreadobex->Ini(false);
444 return true;
445 }
446 }
447
448 diostream->Close();
449 }
450 }
451
452 delete sendbuffer;
453
454 GEN_XFACTORY.Delete_File(xfile);
455 GEN_XFACTORY.DeleteTimer(xtimer);
456
457 sendbuffer = NULL;
458 xfile = NULL;
459 xtimer = NULL;
460
461 return false;

844bool DIOOBEXPUSH::CheckSending()
845{
846 DIOOBEXPUSHFILE* obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(indexfilelist);
847 DIOSTREAMBLUETOOTH* sio = (DIOSTREAMBLUETOOTH*)diostream;
848 XWORD wsize = 0;
849 XDWORD dsize = 0;
850 bool islastblock = false;
851
852 if((!sio)||(!obexpushfile))
853 {
854 if(obexpushfile) obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
855 return false;
856 }
857
858 if(sio->GetConnectStatus()==DIOSTREAMSTATUS_GETTINGCONNECTION)
859 {
860 if(xtimer->GetMeasureSeconds()>=(XDWORD)asktimeout)
861 {
862 asktimeelapsed = xtimer->GetMeasureMilliSeconds();
863 for(int c=0;c<(int)sendfilelist.GetSize();c++)
864 {
865 obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(c);
866 if(obexpushfile) obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
867 }
868
869 return false;
870 }
871
872 return true;
873 }
874
875 if(sio->GetConnectStatus()==DIOSTREAMSTATUS_DISCONNECTED)
876 {
877 haveerror=true;
878 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
879 }
880
881 if(GetEvent()==DIOOBEXPUSH_XFSMEVENT_NONE) // No hay nuevos Eventos
882 {
883 switch(GetCurrentState())
884 {
885 case DIOOBEXPUSH_XFSMSTATE_NONE : break;
886
887 case DIOOBEXPUSH_XFSMSTATE_CONNECT : { XBUFFER answer;
888
889 if(!ReadAnswer(answer)) break;
890
891 //XTRACE_PRINTDATABLOCK(4|XDEBUGTRACELEVEL_WITHCOLOR, answer);
892
893 if(answer.Get()[0]==0xA0)
894 {
895 sendblocksize = (XWORD)answer.Get()[5];
896 sendblocksize <<=8;
897 sendblocksize |= (XWORD)answer.Get()[6];
898
899 SetEvent(DIOOBEXPUSH_XFSMEVENT_INIPUSH);
900 }
901 else
902 {
903 haveerror = true;
904 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
905 }
906 }
907 break;
908
909 case DIOOBEXPUSH_XFSMSTATE_INIPUSH : SetEvent(DIOOBEXPUSH_XFSMEVENT_SENDHEADPUSH);
910 break;
911
912 case DIOOBEXPUSH_XFSMSTATE_SENDHEADPUSH : SetEvent(DIOOBEXPUSH_XFSMEVENT_RECVHEADPUSH);
913 break;
914
915 case DIOOBEXPUSH_XFSMSTATE_SENDBLOCKPUSH : sendbuffer->Delete();
916
917 islastblock = ((filesize - readfilesize) >= sendblocksize)? false:true;
918
919 sendbuffer->Add((!islastblock)?(XBYTE)0x02:(XBYTE)0x82);
920
921 wsize = 0;
922 sendbuffer->Add((XWORD)wsize);
923
924 if(islastblock)
925 readfilesizeblock = (filesize - readfilesize);
926 else readfilesizeblock = sendblocksize -16;
927
928 sendbuffer->Add((!islastblock)?(XBYTE)0x48:(XBYTE)0x49);
929
930 wsize+=3;
931
932 wsize += (int)readfilesizeblock;
933 sendbuffer->Add((XWORD)wsize);
934
935 { XBUFFER datafile((XDWORD)readfilesizeblock, true);
936
937 xfile->Read(datafile.Get(),(XDWORD*)&readfilesizeblock);
938 if(!readfilesizeblock)
939 {
940 haveerror = true;
941 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
942 break;
943 }
944
945 sendbuffer->Add(datafile);
946 }
947
948 if(inlittleendian)
949 wsize = SwapWORD((XWORD)sendbuffer->GetSize());
950 else wsize = (XWORD)sendbuffer->GetSize();
951 sendbuffer->Set((XBYTE*)&wsize,2,1);
952
953 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
954
955 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
956 sio->WaitToFlushOutXBuffer(5);
957
958 if(filesize) obexpushfile->SetSendPercent((XBYTE)((readfilesize*100)/filesize));
959
960 SetEvent(DIOOBEXPUSH_XFSMEVENT_RECVBLOCKPUSH);
961 break;
962
963 case DIOOBEXPUSH_XFSMSTATE_RECVHEADPUSH :
964 case DIOOBEXPUSH_XFSMSTATE_RECVBLOCKPUSH : { XBUFFER answer;
965
966 if(!ReadAnswer(answer)) break;
967
968 //XTRACE_PRINTDATABLOCK(4|XDEBUGTRACELEVEL_WITHCOLOR, answer);
969
970 switch(answer.Get()[0])
971 {
972 case 0x90 : readfilesize+=readfilesizeblock;
973 SetEvent(DIOOBEXPUSH_XFSMEVENT_SENDBLOCKPUSH);
974 break;
975
976 case 0xA0 : haveerror = false;
977 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
978 break;
979
980 default : if(GetEvent()==DIOOBEXPUSH_XFSMEVENT_NONE)
981 {
982 haveerror = true;
983 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
984 }
985 break;
986 }
987
988 }
989 break;
990
991 case DIOOBEXPUSH_XFSMSTATE_ENDPUSH : return false;
992 }
993 }
994 else // Nuevo evento
995 {
996 if(GetEvent()<DIOOBEXPUSH_LASTEVENT)
997 {
998 CheckTransition();
999
1000 switch(GetCurrentState())
1001 {
1002 case DIOOBEXPUSH_XFSMSTATE_NONE : break;
1003
1004 case DIOOBEXPUSH_XFSMSTATE_CONNECT : sendisstart = false;
1005
1006 asktimeelapsed = xtimer->GetMeasureMilliSeconds();
1007
1008 sendbuffer->Delete();
1009
1010 sendbuffer->Add((XBYTE)0x80);
1011
1012 wsize = 0x0007;
1013 sendbuffer->Add((XWORD)wsize);
1014
1015 sendbuffer->Add((XBYTE)0x10);
1016 sendbuffer->Add((XBYTE)0x00);
1017
1018 wsize = DIOOBEXPUSH_MAXBUFFER;
1019 sendbuffer->Add((XWORD)wsize);
1020
1021 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
1022
1023 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
1024 sio->WaitToFlushOutXBuffer(5);
1025 break;
1026
1027 case DIOOBEXPUSH_XFSMSTATE_INIPUSH : { xtimer->Reset();
1028
1029 obexpushfile->SetIsSend(false);
1030 obexpushfile->SetSendPercent(0);
1031
1032 xpath = (*obexpushfile->GetXPath());
1033
1034 if((!xfile)||(!xfile->Open(xpath,true))||(!sendbuffer))
1035 {
1036 haveerror = true;
1037 SetEvent(DIOOBEXPUSH_XFSMEVENT_ENDPUSH);
1038 break;
1039 }
1040
1041 readfilesizeblock = 0;
1042
1043 sendisstart = true;
1044 filesize = xfile->GetSize();
1045 readfilesize = 0;
1046 }
1047 break;
1048
1049 case DIOOBEXPUSH_XFSMSTATE_SENDHEADPUSH : sendbuffer->Delete();
1050
1051 sendbuffer->Add((XBYTE)0x02);
1052
1053 wsize = 0;
1054 sendbuffer->Add((XWORD)wsize);
1055
1056 sendbuffer->Add((XBYTE)0x01);
1057
1058 // Unicode 16 + 1 byte of head;
1059 wsize++;
1060 wsize += (XWORD)((obexpushfile->GetNameSend()->GetSize()+1)*2);
1061 sendbuffer->Add((XWORD)wsize);
1062
1063 XBUFFER xbufferexchange;
1064 (*obexpushfile->GetNameSend())->ConvertToBufferExchange(xbufferexchange, true, true);
1065 sendbuffer->Add(xbufferexchange->Get(), wsize);
1066
1067 sendbuffer->Add((XBYTE)0xC3);
1068 sendbuffer->Add((XDWORD)filesize);
1069
1070 if(inlittleendian)
1071 wsize = SwapWORD((XWORD)sendbuffer->GetSize());
1072 else wsize = (XWORD)sendbuffer->GetSize();
1073 sendbuffer->Set((XBYTE*)&wsize,2,1);
1074
1075 //XTRACE_PRINTDATABLOCK(2|XDEBUGTRACELEVEL_WITHCOLOR, (*sendbuffer));
1076
1077 sio->Write(sendbuffer->Get(),sendbuffer->GetSize());
1078 sio->WaitToFlushOutXBuffer(5);
1079 break;
1080
1081 case DIOOBEXPUSH_XFSMSTATE_RECVHEADPUSH : break;
1082
1083 case DIOOBEXPUSH_XFSMSTATE_SENDBLOCKPUSH : break;
1084
1085 case DIOOBEXPUSH_XFSMSTATE_RECVBLOCKPUSH : break;
1086
1087 case DIOOBEXPUSH_XFSMSTATE_ENDPUSH : if(xfile) xfile->Close();
1088
1089 if(!haveerror)
1090 {
1091 obexpushfile->SetIsSend(true);
1092 obexpushfile->SetSendPercent(100);
1093 obexpushfile->SetSendTimeElapsed(xtimer->GetMeasureMilliSeconds());
1094 obexpushfile->SetError(DIOOBEXPUSHERROR_NONE);
1095
1096 if((indexfilelist+1<(int)sendfilelist.GetSize()))
1097 {
1098 indexfilelist++;
1099 SetEvent(DIOOBEXPUSH_XFSMEVENT_INIPUSH);
1100 }
1101 }
1102 else
1103 {
1104 for(int c=0;c<(int)sendfilelist.GetSize();c++)
1105 {
1106 obexpushfile = (DIOOBEXPUSHFILE*)sendfilelist.Get(c);
1107 if(obexpushfile)
1108 {
1109 obexpushfile->SetIsSend(false);
1110
1111 if(sendisstart)
1112 {
1113 obexpushfile->SetError(DIOOBEXPUSHERROR_SENDING);
1114 }
1115 else
1116 {
1117 if(sio->IsRefusedConnection())
1118 {
1119 obexpushfile->SetError(DIOOBEXPUSHERROR_REFUSEDCONNECTION);
1120 }
1121 else
1122 {
1123 obexpushfile->SetError(DIOOBEXPUSHERROR_NOTCONNECTION);
1124 }
1125 }
1126 }
1127 }
1128
1129 return false;
1130 }
1131 break;
1132 }
1133 }
1134 }
1135
1136 return true;
1137}
