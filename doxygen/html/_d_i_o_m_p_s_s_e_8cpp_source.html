149bool DIOMPSSE::Open(int index)
150{
151 if(!dioenumUSBdevices) return false;
152
153 EnumDevices();
154
155 XVECTOR<DIOSTREAMDEVICE*>* devices = dioenumUSBdevices->GetDevices();
156 if(!devices) return false;
157 if(devices->IsEmpty()) return false;
158
159 USBdevice = (DIOSTREAMDEVICEUSB*)devices->Get(index);
160 if(!USBdevice) return false;
161
162 switch(USBdevice->GetDescriptor()->GetBCDDevice())
163 {
164 case 0x200 : if(USBdevice->GetDescriptor()->GetSerialNumber() == 0) chiptype = DIOMPSSE_CHIP_TYPE_BM;
165 break;
166
167 case 0x400 : chiptype = DIOMPSSE_CHIP_TYPE_BM; break;
168 case 0x500 : chiptype = DIOMPSSE_CHIP_TYPE_2232C; break;
169 case 0x600 : chiptype = DIOMPSSE_CHIP_TYPE_R; break;
170 case 0x700 : chiptype = DIOMPSSE_CHIP_TYPE_2232H; break;
171 case 0x800 : chiptype = DIOMPSSE_CHIP_TYPE_4232H; break;
172 case 0x900 : chiptype = DIOMPSSE_CHIP_TYPE_232H; break;
173
174 default : chiptype = DIOMPSSE_CHIP_TYPE_UNKNOWN; break;
175 }
176
177 diostreamUSBcfg->GetResource()->Set((*USBdevice->GetResource()));
178
179 if(diostreambase)
180 {
181 if(diostreambase->IsConnected()) Close();
182 GEN_DIOFACTORY.DeleteStreamIO(diostreambase);
183
184 diostreambase = NULL;
185 }
186
187 diostreambase = GEN_DIOFACTORY.CreateStreamIO(diostreamUSBcfg);
188 if(!diostreambase) return false;
189
190 if(!diostreambase->Open()) return false;
191
192 if(!Reset()) return false;
193
194 isopen = true;
195
196 return isopen;
197}

369bool DIOMPSSE::EnumDevices()
370{
371 if(!dioenumUSBdevices) return false;
372
373 if(!dioenumUSBdevices->Search()) return false;
374
375 XVECTOR<DIOSTREAMDEVICE*>* devices = dioenumUSBdevices->GetDevices();
376 if(!devices) return false;
377 if(devices->IsEmpty()) return false;
378
379 DIOSTREAMDEVICEUSB* device;
380 int index = 0;
381
382 do{ device = (DIOSTREAMDEVICEUSB*)devices->Get(index);
383 if(device)
384 {
385 if((device->GetVendorID() == 0x403) && ((device->GetProductID() == 0x6001) ||
386 (device->GetProductID() == 0x6010) ||
387 (device->GetProductID() == 0x6011) ||
388 (device->GetProductID() == 0x6014)))
389 {
390 index++;
391 }
392 else
393 {
394 devices->Delete(device);
395 delete device;
396 }
397 }
398
399 } while(device);
400
401 return true;
402}

417bool DIOMPSSE::SendCommand(XBYTE command, XBUFFER& buffer)
418{
419 XBUFFER datasend;
420 XBYTE lowsize;
421 XBYTE highsize;
422 bool status;
423
424 if(buffer.GetSize() > 65535) return false;
425
426 lowsize = (buffer.GetSize() & 0x00FF);
427 highsize = ((buffer.GetSize() & 0xFF00) >> 0x08);
428
429 datasend.Add((XBYTE)command);
430 datasend.Add((XBYTE)lowsize);
431 datasend.Add((XBYTE)highsize);
432
433 datasend.Add(buffer);
434
435 status = diostreambase->Write(datasend);
436 if(status) status = diostreambase->WaitToFlushOutXBuffer(5);
437
438 return status;
439}
