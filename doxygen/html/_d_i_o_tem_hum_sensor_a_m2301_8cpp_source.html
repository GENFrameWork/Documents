201bool DIOTEMHUMSENSORAM2301::ReadDirect(float& temperature, float& humidity)
202{
203 if(!xtimer) return false;
204 if(!IsInitialized()) return false;
205
206 int timeelapsedlow;
207 int timeelapsedhigh;
208 XBYTE value[5] = { 0, 0, 0, 0, 0 };
209
210 // Leave it high for a while
211 GEN_DIOGPIO.SetMode(pindata, DIOGPIO_MODE_OUTPUT);
212 GEN_DIOGPIO.SetValue(pindata, DIOTEMHUMSENSORAM2301_HIGH);
213 GEN_XSLEEP.MilliSeconds(1);
214
215 // Set it low to give the start signal
216 GEN_DIOGPIO.SetValue(pindata, DIOTEMHUMSENSORAM2301_LOW);
217 GEN_XSLEEP.MilliSeconds(1);
218 GEN_DIOGPIO.SetValue(pindata, DIOTEMHUMSENSORAM2301_HIGH);
219
220 // Now set the pin high to let the sensor start communicating
221 GEN_DIOGPIO.SetMode(pindata, DIOGPIO_MODE_INPUT);
222
223 GEN_XSLEEP.NanoSeconds(30);
224
225 // Read ACK
226 WaitTo(false, 50);
227
228 WaitTo(true , 100);
229 WaitTo(false, 100);
230
231 // Read values
232 for(int d=0; d<5; d++)
233 {
234 value[d] = 0x00;
235
236 for(int c=0; c<8; c++)
237 {
238 WaitTo(false, 50);
239 WaitTo(true , 100, &timeelapsedlow);
240 WaitTo(false, 100, &timeelapsedhigh);
241
242 value[d]|=(timeelapsedlow>timeelapsedhigh?0:1);
243
244 if(c<7) value[d] <<= 1;
245 }
246 }
247
248
249 // Final State of SDA
250 GEN_DIOGPIO.SetMode(pindata, DIOGPIO_MODE_OUTPUT);
251 GEN_DIOGPIO.SetValue(pindata, DIOTEMHUMSENSORAM2301_HIGH);
252
253 GEN_XSLEEP.MilliSeconds(1);
254
255 // XTRACE_PRINTCOLOR(XTRACE_PRINTCOLOR, __L(" %X %X %X %X "), value[0], value[1], value[2], value[3]);
256
257 // Verify checksum
258 XBYTE x = value[0] + value[1] + value[2] + value[3];
259 if(x != value[4]) return false;
260
261 humidity = (float)(((XWORD)value[0] << 8) | (XWORD)value[1]);
262 humidity /= 10.0f;
263
264 temperature = (float)(((XWORD)value[2] << 8) | (XWORD)value[3]);
265 temperature /= 10.0f;
266
267 if((humidity > 100.0) || (humidity <= 0.0) || (temperature > 80.0) || (temperature < -40.0)) return false;
268
269 return true;
270}

305bool DIOTEMHUMSENSORAM2301::WaitTo(bool tohigh, int timeout, int* timeelapsed)
306{
307 if(!xtimer) return false;
308
309 bool state;
310 bool status = true;
311
312 int _timeelapsed = 0;
313
314 do{ state = GEN_DIOGPIO.GetValue(pindata);
315
316 _timeelapsed++;
317
318 if(tohigh)
319 {
320 if(state) break;
321 }
322 else
323 {
324 if(!state) break;
325 }
326
327 if(_timeelapsed > timeout)
328 {
329 status = false;

331 }
332
333 } while(1);
334
335 if(timeelapsed) (*timeelapsed) = _timeelapsed;

350void DIOTEMHUMSENSORAM2301::ThreadRunFunction(void* param)
351{
352 DIOTEMHUMSENSORAM2301* sensor = (DIOTEMHUMSENSORAM2301*)param;
353 if(!sensor) return;
354
355 if(!sensor->IsInitialized()) return;
356
357 float temperature;
358 float humidity;
359
360 bool status = sensor->ReadDirect(temperature, humidity);
361 if(status)
362 {
363 if(sensor->xmutexread) sensor->xmutexread->Lock();
364
365 if(sensor->nreads>1)
366 {
367 if((abs((int)sensor->temperaturecache - (int)temperature)>10) || (abs((int)sensor->humiditycache - (int)humidity)>10)) status = false;
368 }
369
370 if(status)
371 {
372 sensor->temperaturecache = temperature;
373 sensor->humiditycache = humidity;
374 sensor->nreads++;
375 }
376
377 if(sensor->xmutexread) sensor->xmutexread->UnLock();
378 }
379
380 if(sensor->xtimer) GEN_XSLEEP.MilliSeconds(500);
381
382}
