514SNDITEM* SNDFACTORY::CreateItem(XCHAR* path)
515{
516 if(!soundactive)
517 {
518 return NULL;

520
521 if(!path)
522 {
523 return NULL;
524 }
525
526 SNDITEM* item = new SNDITEM();

528 {
529 return NULL;

533

541 item->GetID()->Set(ID);

544 return NULL;

554 item->SetStatus(SNDITEM_STATUS_NONE);

592SNDITEM* SNDFACTORY::CreateItem(XDWORD frecuency, XDWORD duration)
593{
594 if(!soundactive)
595 {
596 return NULL;
597 }
598
599 if((!frecuency) || (!duration))
600 {
601 return NULL;
602 }
603
604 SNDITEM* item = new SNDITEM();
605 if(!item)
606 {
607 return NULL;
608 }
609
610 item->SetType(SNDITEM_TYPE_NOTE);
611
612 item->GetID()->Format(__L("frec: %d (Hz) - time: %d (msec)"), frecuency, duration);
613
614 SNDNOTE* note = new SNDNOTE();
615 if(!note)
616 {
617 delete item;
618 return NULL;
619 }
620
621 note->SetFrequency(frecuency);
622 note->SetDuration(duration);
623
624 item->SetStatus(SNDITEM_STATUS_NONE);
625 item->SetSoundNote(note);
626
627 sounditems.Add(item);
628
629 return item;
630}

657bool SNDFACTORY::Sound_IsAnyActive()
658{
659 if(!soundactive)
660 {
661 return false;
662 }
663
664 XDWORD counter = 0;
665
666 for(XDWORD c=0; c<sounditems.GetSize(); c++)
667 {
668 SNDITEM* item = sounditems.Get(c);
669 if(item)
670 {
671 if((item->GetStatus() != SNDITEM_STATUS_END) &&
672 (item->GetStatus() != SNDITEM_STATUS_NONE))
673 {
674 counter++;
675 }
676 }
677 }
678
679 return counter?true:false;
680}
