421XDWORD DIOALERT::CalculateID(bool withdatetime)
422{
423 if(!xdatetime) return false;
424
425 HASHCRC32* hashcrc32 = new HASHCRC32();
426 if(!hashcrc32) return 0;
427
428 XSTRING IDstring;
429 XSTRING levelstring;
430 XDWORD ID = 0;
431 bool status;
432
433 if(withdatetime) xdatetime->GetDateTimeToString(XDATETIME_FORMAT_STANDARD, IDstring);
434
435 levelstring.Format(__L("%s %d"), applicationID.Get(), level);
436 IDstring += levelstring;
437
438 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
439 if(xrand)
440 {
441 levelstring.Format(__L(" %d "), xrand->Between(1000, 10000000) * xrand->Between(1000, 10000000));
442 GEN_XFACTORY.DeleteRand(xrand);
443 }
444
445 IDstring += __L(" "); IDstring += origin;
446 IDstring += __L(" "); IDstring += title;
447 IDstring += __L(" "); IDstring += message;
448 IDstring += levelstring;
449
450 XBUFFER charstr;
451
452 IDstring.ConvertToASCII(charstr);
453 status = hashcrc32->Do(charstr.Get(), IDstring.GetSize());
454
455 if(status) ID = hashcrc32->GetResultCRC32(); else ID = 0;
456
457 delete hashcrc32;
458
459 return ID;
460}
