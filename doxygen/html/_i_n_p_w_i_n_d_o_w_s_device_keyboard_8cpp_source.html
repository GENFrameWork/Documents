108bool INPWINDOWSDEVICEKEYBOARD::Update()
109{
110 if((!created)||(!enabled)) return false;
111
112 if(grpscreen)
113 {
114 if(grpscreen->GetHandle()!=GetForegroundWindow()) return false;
115 }
116
117 for(int c=0;c<(int)buttons.GetSize();c++)
118 {
119 INPBUTTON* button = (INPBUTTON*)buttons.Get(c);
120
121 if(button)
122 {
123 if(GetKeyState(button->GetKeyCode()) & 0x80)
124 {
125 button->SetPressed(true);
126 }
127 else
128 {
129 button->SetPressed(false);
130 }
131 }
132 }
133
134 int n=buttons.GetSize();
135 for(int c=0;c<n;c++)
136 {
137 INPBUTTON* button = (INPBUTTON*)buttons.FastGet(c);
138 if(button)
139 {
140 SHORT state = GetAsyncKeyState(button->GetKeyCode());
141 if(state & 0x8000) //key down
142 {
143 if(state & 0x01) //changed
144 {
145 if (button->GetState() != INPBUTTON_STATE_HOLD) button->SetState(INPBUTTON_STATE_PRESSED);
146
147 } else button->SetState(INPBUTTON_STATE_HOLD);
148 }
149 else
150 {
151 if(button->GetState() == INPBUTTON_STATE_HOLD || button->GetState() == INPBUTTON_STATE_PRESSED)
152 {
153 button->SetState(INPBUTTON_STATE_RELEASED);
154
155 } else button->SetState(INPBUTTON_STATE_UP);
156 }
157 }
158 }
159
160 return true;
161}
