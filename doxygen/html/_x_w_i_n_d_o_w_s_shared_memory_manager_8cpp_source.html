106XBYTE* XWINDOWSSHAREDMEMORYMANAGER::Create(XCHAR* ID, XDWORD size, bool ispublic)
107{
108 // Allocate shared memory from the system by creating a shared memory
109 // pool basing it out of the system paging (swapper) file.
110
111 this->ID = ID;
112 this->size = (size + sizeof(XDWORD));
113
114 if(ispublic)
115 {
116 SECURITY_ATTRIBUTES attributes;
117 wchar_t sdd[71] = L"D:"
118 L"(D;OICI;GA;;;BG)" // Deny guests
119 L"(D;OICI;GA;;;AN)" // Deny anonymous
120 L"(A;OICI;GRGWGX;;;AU)" // Allow read, write and execute for Users
121 L"(A;OICI;GA;;;BA)"; // Allow all for Administrators
122
123
124 ZeroMemory(&attributes, sizeof(attributes));
125 attributes.nLength = sizeof(attributes);
126
127 if(ConvertStringSecurityDescriptorToSecurityDescriptor(sdd, SDDL_REVISION_1, &attributes.lpSecurityDescriptor, NULL) != TRUE) return NULL;
128
129 handlefile = CreateFileMapping((HANDLE)NULL,
130 &attributes,
131 PAGE_READWRITE,
132 0,
133 this->size,
134 ID);
135
136 LocalFree(attributes.lpSecurityDescriptor);
137 }
138 else
139 {
140 handlefile = CreateFileMapping((HANDLE)NULL,
141 NULL,
142 PAGE_READWRITE,
143 0,
144 this->size,
145 ID);
146 }
147
148 if(!handlefile) return NULL;
149
150 // Map the file to this process' address space, starting at an address
151 // that should also be available in child processe(s)
152
153 base = (XBYTE*)MapViewOfFileEx(handlefile, FILE_MAP_WRITE, 0, 0, 0, 0);
154 if(base)
155 {
156 memcpy((XBYTE*)base, (XBYTE*)&this->size, sizeof(XDWORD));
157
158 //XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("[WINDOWS Shared Mem] Create Size [%d]"), this->size);
159
160 pointer = base;
161 pointer += sizeof(XDWORD);
162 }
163
164 isserver = true;
165
166 return pointer;
167}

183XBYTE* XWINDOWSSHAREDMEMORYMANAGER::Open(XCHAR* ID, XDWORD& size)
184{
185 // Open the shared memory file by name. The file is based on the
186 // system paging (swapper) file.
187
188 this->ID = ID;
189 this->size = 0;
190
191 size = 0;
192
193 handlefile = OpenFileMapping(FILE_MAP_WRITE, FALSE, ID);
194 if(!handlefile)
195 {
196 //DWORD error = GetLastError();
197 return NULL;
198 }
199
200 // Figure out where to map this file by looking at the address in the
201 // shared memory where the memory was mapped in the parent process.
202
203 base = (XBYTE*)MapViewOfFile(handlefile, FILE_MAP_WRITE, 0, 0, sizeof(XDWORD));
204 if(base)
205 {
206
207 memcpy((XBYTE*)&this->size, base,sizeof(XDWORD));
208
209 //XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("[WINDOWS Shared Mem] Open Size [%d]"), this->size);
210
211 UnmapViewOfFile(base);
212
213 base = (XBYTE*)MapViewOfFile(handlefile, FILE_MAP_WRITE, 0, 0, this->size);
214 if(base)
215 {
216 this->size -= sizeof(XDWORD);
217 size = this->size;
218
219 //XTRACE_PRINTCOLOR(XTRACE_COLOR_PURPLE, __L("[WINDOWS Shared Mem] Open Size [%d]"), this->size);
220
221 pointer = base;
222 pointer += sizeof(XDWORD);
223 }
224 }
225
226 isserver = false;
227
228 return pointer;
229}
