123bool DIOSPITOUCHSCREENSTMPE610::IniDevice(int chipselect, int timeout)

147

149
150 if(!diostreamSPI->Open()) return false;

153
154 if(!TouchScreen_Ini()) return false;

387bool DIOSPITOUCHSCREENSTMPE610::TouchScreen_Ini()
388{
389 if(!WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_SYS_CTRL1, DIOSPITOUCHSCREENSTMPE610_SYS_CTRL1_RESET)) return false;
390
391 GEN_XSLEEP.MilliSeconds(250);
392
393 for(XBYTE c=0; c<65; c++)
394 {
395 XBYTE data;
396 ReadRegister(c, data);
397 }
398
399 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_SYS_CTRL2 , 0x0); // turn on clocks!
400 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_CTRL , DIOSPITOUCHSCREENSTMPE610_TSC_CTRL_XYZ | DIOSPITOUCHSCREENSTMPE610_TSC_CTRL_EN); // XYZ and enable!
401 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_EN , DIOSPITOUCHSCREENSTMPE610_INT_EN_TOUCHDET);
402 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_ADC_CTRL1 , DIOSPITOUCHSCREENSTMPE610_ADC_CTRL1_10BIT | (0x6 << 4)); // 96 clocks per conversion
403 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_ADC_CTRL2 , DIOSPITOUCHSCREENSTMPE610_ADC_CTRL2_6_5MHZ);
404 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_CFG , DIOSPITOUCHSCREENSTMPE610_TSC_CFG_4SAMPLE | DIOSPITOUCHSCREENSTMPE610_TSC_CFG_DELAY_1MS | DIOSPITOUCHSCREENSTMPE610_TSC_CFG_SETTLE_5MS);
405 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_FRACT_XYZ , 0x6);
406 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_TH , 1);
407 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_STA , DIOSPITOUCHSCREENSTMPE610_FIFO_STA_RESET);
408 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_FIFO_STA , 0); // unreset
409 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_TSC_I_DRIVE , DIOSPITOUCHSCREENSTMPE610_TSC_I_DRIVE_50MA);
410 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_STA , 0xFF); // reset all ints
411 WriteRegister(DIOSPITOUCHSCREENSTMPE610_REG_INT_CTRL , DIOSPITOUCHSCREENSTMPE610_INT_CTRL_POL_HIGH | DIOSPITOUCHSCREENSTMPE610_INT_CTRL_ENABLE);
412
413 return true;

429bool DIOSPITOUCHSCREENSTMPE610::ReadRegister(XBYTE reg, XBYTE& data)
430{
431 //diostreamSPI->GetInXBuffer()->Delete();
432
433 XBYTE regdata[2] = { (XBYTE)(reg | 0x80), 0 };
434 XBYTE _data[2] = { 0, 0 };
435
436 bool status = diostreamSPI->Write(regdata, 2)?true:false;
437 //if(status) status = diostreamSPI->WaitToFlushOutXBuffer(timeout);
438
439 if(status) status = diostreamSPI->WaitToFilledReadingBuffer(2, timeout);
440 if(status) status = diostreamSPI->Read(_data, 2)?true:false;
441
442 data = _data[1];
443

460bool DIOSPITOUCHSCREENSTMPE610::WriteRegister(XBYTE reg, XBYTE data)
461{
462 if(!databuffer) return false;
463
464 bool status = diostreamSPI->Write(&reg, 1)?true:false;
465 if(status) status = diostreamSPI->Write(&data, 1)?true:false;
466 if(status) status = diostreamSPI->WaitToFlushOutXBuffer(timeout);
467 if(status) status = diostreamSPI->WaitToFilledReadingBuffer(2, timeout);
468
469 diostreamSPI->GetInXBuffer()->Delete();
470
471 return status;
