292bool DIOSTREAMUARTCONFIG::GetToString(XSTRING* string)
293{
294 if(!string) return false;
295
296 XSTRING xstringtmp;
297
298 string->Empty();
299
300 if(!GetLocalDeviceName()->IsEmpty())
301 {
302 string->Add(GetLocalDeviceName()->Get());
303 string->Add(__L(","));
304 }
305 else
306 {
307 if(!GetPort()) return false;
308 string->Format(__L("%d,"), GetPort());
309 }
310
311 if(!GetBaudRate()) return false;
312 xstringtmp.Format(__L("%d,"), GetBaudRate());
313 string->Add(xstringtmp);
314
315 switch(GetDataBits())
316 {
317 case DIOSTREAMUARTDATABIT_5 : string->Add(__L("5,")); break;
318 case DIOSTREAMUARTDATABIT_6 : string->Add(__L("6,")); break;
319 case DIOSTREAMUARTDATABIT_7 : string->Add(__L("7,")); break;
320 case DIOSTREAMUARTDATABIT_8 : string->Add(__L("8,")); break;
321 default : return false;
322 }
323
324 switch(GetParity())
325 {
326 case DIOSTREAMUARTPARITY_NONE : string->Add(__L("N,")); break;
327 case DIOSTREAMUARTPARITY_ODD : string->Add(__L("O,")); break;
328 case DIOSTREAMUARTPARITY_EVEN : string->Add(__L("E,")); break;
329 case DIOSTREAMUARTPARITY_MARK : string->Add(__L("M,")); break;
330 case DIOSTREAMUARTPARITY_SPACE : string->Add(__L("S,")); break;
331 default : return false;
332 }
333
334 switch(GetFlowControl())
335 {
336 case DIOSTREAMUARTFLOWCONTROL_NONE : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_NONE); break;
337 case DIOSTREAMUARTFLOWCONTROL_HARD : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_SOFT); break;
338 case DIOSTREAMUARTFLOWCONTROL_SOFT : string->Add(DIOSTREAMUART_FLOWCONTROL_STR_HARD); break;
339 default : return false;
340 }
341
342 return true;
343}

374bool DIOSTREAMUARTCONFIG::SetFromString(XCHAR* string)
375{
376 if(!string) return false;
377
378 XSTRING _string = string;
379 XSTRING devicename;
380 int baudrate;
381 int databits;
382 XCHAR parity;
383 XSTRING stopbits;
384 XSTRING flowcontrol;
385
386 devicename.AdjustSize(_MAXSTR);
387 stopbits.AdjustSize(_MAXSTR);
388 flowcontrol.AdjustSize(_MAXSTR);
389
390 _string.UnFormat(__L("%s,%d,%d,%c,%s,%s"), devicename.Get(), &baudrate, &databits, &parity, stopbits.Get(), flowcontrol.Get());
391
392 devicename.AdjustSize();
393 stopbits.AdjustSize();
394 flowcontrol.AdjustSize();
395
396
397 if(devicename.HaveOnlyNumbers())
398 {
399 SetPort(devicename.ConvertToInt());
400 if(!GetPort()) return false;
401 }
402 else GetLocalDeviceName()->Set(devicename);
403
404 if(!baudrate) return false;
405 SetBaudRate(baudrate);
406
407 switch(databits)
408 {
409 case 5 : SetDataBits(DIOSTREAMUARTDATABIT_5); break;
410 case 6 : SetDataBits(DIOSTREAMUARTDATABIT_6); break;
411 case 7 : SetDataBits(DIOSTREAMUARTDATABIT_7); break;
412 case 8 : SetDataBits(DIOSTREAMUARTDATABIT_8); break;
413 default : return false;
414 }
415
416 switch(parity)
417 {
418 case 'N' : SetParity(DIOSTREAMUARTPARITY_NONE); break;
419 case 'O' : SetParity(DIOSTREAMUARTPARITY_ODD); break;
420 case 'E' : SetParity(DIOSTREAMUARTPARITY_EVEN); break;
421 case 'M' : SetParity(DIOSTREAMUARTPARITY_MARK); break;
422 case 'S' : SetParity(DIOSTREAMUARTPARITY_SPACE); break;
423 default : return false;
424 }
425
426 if(!stopbits.Compare(__L("1")))
427 {
428 SetStopBits(DIOSTREAMUARTSTOPBITS_ONE);
429 }
430 else
431 {
432 if(!stopbits.Compare(__L("1.5")))
433 {
434 SetStopBits(DIOSTREAMUARTSTOPBITS_ONEANDAHALF);
435 }
436 else
437 {
438 if(!stopbits.Compare(__L("2")))
439 {
440 SetStopBits(DIOSTREAMUARTSTOPBITS_TWO);
441
442 } else return false;
443 }
444 }
445
446 if(!flowcontrol.Compare(DIOSTREAMUART_FLOWCONTROL_STR_NONE, true))
447 {
448 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_NONE);
449 }
450 else
451 {
452 if(!stopbits.Compare(DIOSTREAMUART_FLOWCONTROL_STR_SOFT) || !stopbits.Compare(__L("SOFT")))
453 {
454 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_SOFT);
455 }
456 else
457 {
458 if(!stopbits.Compare(DIOSTREAMUART_FLOWCONTROL_STR_HARD) || !stopbits.Compare(__L("HARD")))
459 {
460 SetFlowControl(DIOSTREAMUARTFLOWCONTROL_HARD);
461
462 } else return false;
463 }
464 }
465
466 return true;
467}
