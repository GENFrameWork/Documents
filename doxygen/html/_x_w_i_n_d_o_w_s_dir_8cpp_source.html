147bool XWINDOWSDIR::Make(XCHAR* path, bool recursive)
148{
149 if(!path) return false;
150 if(!path[0]) return false;
151
152 if(Exist(path)) return true;
153
154 XPATH xpath;
155
156 xpath = path;
157
158 xpath.Slash_Add();
159 xpath.Slash_Normalize(true);
160
161 if(recursive)
162 {
163 XPATH xpathsequence;
164 XSTRING pathpart;
165 int index = 0;
166
167 xpathsequence.Empty();
168
169 do{

173 xpathsequence += __L("\\");

175 if(pathpart.Character_GetLast()!=__C(':'))
176 {
177 if(!Exist(xpathsequence.Get()))

179 //XTRACE_PRINTCOLOR(2, __L("---> Make: %s"), xpathsequence.Get());

181 int result = CreateDirectory(xpathsequence.Get(), NULL);
182 if(!result) return false;
183 }
184 }
185
186 } else break;
187
188 index++;
189
190 } while(1);
191 }
192 else
193 {

306{
307 if(!xpath) return false;

431{
432 WIN32_FIND_DATA* search = (WIN32_FIND_DATA*)searchelement->GetFindFileData();
433 XDATETIME* datetimefile = NULL;
