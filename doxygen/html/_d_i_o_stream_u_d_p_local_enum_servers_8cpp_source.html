309void DIOSTREAMUDPLOCALENUMSERVERS::ThreadDiscoveryLocalServer(void* data)
310{
311 DIOSTREAMUDPLOCALENUMSERVERS* discoveryenum = (DIOSTREAMUDPLOCALENUMSERVERS*)data;
312 if(!discoveryenum) return;
313
314 if(!discoveryenum->GetDIOStreamUDP()) return;
315 if(!discoveryenum->GetDIOStreamUDPCFG()) return;
316
317 discoveryenum->issearching = true;
318
319 DIOSTREAMUDPCONFIG* diostreamudpcfg = discoveryenum->GetDIOStreamUDPCFG();
320 DIOSTREAMUDP* diostreamudp = discoveryenum->GetDIOStreamUDP();
321
322 diostreamudpcfg->SetMode(DIOSTREAMMODE_SERVER);
323 diostreamudpcfg->SetBroadcastModeActive(true);
324 diostreamudpcfg->SetRemotePort(discoveryenum->GetPort());
325 diostreamudpcfg->SetIsUsedDatagrams(true);
326
327 discoveryenum->xtimer->Reset();
328
329 if(diostreamudp->Open())
330 {
331 while(discoveryenum->issearching)
332 {
333 if(discoveryenum->searchtimeout)
334 {
335 if(discoveryenum->xtimer->GetMeasureSeconds()>=discoveryenum->searchtimeout) break;
336 }
337
338 XBUFFER data;
339 XSTRING address;
340 XWORD port;
341
342 if(diostreamudp->ReadDatagram(address, port, data))
343 {
344 XSTRING string;
345 XSTRING string2(256);
346 XDWORD ID = discoveryenum->GetID();
347 HASHCRC32 crc32;
348 XDWORD crc32result[2];
349
350 data.Extract(crc32result[0]);
351
352 string.Format(DIOSTREAMUDPLOCALENUMSERVERSID,ID);
353
354 string2.Set((XWORD*)data.Get());
355 string2.AdjustSize();
356
357 if(!string.Compare(string2))
358 {
359 XBUFFER xbufferexchange;
360
361 string2.ConvertToExchangeXBuffer(xbufferexchange);
362 crc32.Do(xbufferexchange.Get(), xbufferexchange.GetSize());
363
364 crc32result[1] = crc32.GetResultCRC32();
365
366 if(crc32result[0] == crc32result[1])
367 {
368 if(!discoveryenum->IsDeviceAvailable(address.Get()))
369 {
370 DIOSTREAMDEVICEIP* device = new DIOSTREAMDEVICEIP();
371 if(device)
372 {
373 device->GetIP()->Set(address);
374 discoveryenum->devices.Add(device);
375
376 if(!discoveryenum->searchtimeout) break;
377 }
378 }
379 }
380 }
381 }
382 }
383
384 diostreamudp->WaitToFlushXBuffers(5);
385
386 diostreamudp->Close();
387 }
388
389 discoveryenum->xthread->Run(false);
390
391 discoveryenum->issearching = false;
392}
