111SNDFILE* SNDFILE::Create(XCHAR* path)
112{
113 XPATH xpath;
114 XSTRING ext;
115 SNDFILE* sndfile = NULL;
116
117 xpath = path;
118 if(xpath.IsEmpty())
119 {
120 return NULL;
121 }
122
123 xpath.GetExt(ext);
124
125 if(!ext.Compare(__L(".ogg"), true))
126 {
127 sndfile = new SNDFILEOGG();
128 }
129
130 if(!ext.Compare(__L(".wav"), true))
131 {
132 sndfile = new SNDFILEWAV();
133 }
134
135 if(sndfile)
136 {
137 XFILE* file = GEN_XFACTORY.Create_File();
138 if(file)
139 {
140 bool exist = file->Exist(xpath);
141 GEN_XFACTORY.Delete_File(file);
142
143 if(!exist)
144 {
145 delete sndfile;
146 return NULL;
147 }
148 }
149
150 sndfile->GetPath()->Set(path);
151
152 if(!sndfile->LoadFile())
153 {
154 delete sndfile;
155 return NULL;
156 }
157
158
159 }
160
161 return sndfile;
162}
