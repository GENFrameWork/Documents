269bool DIOSTREAMUSBCONFIG::SetFromString(XCHAR* string)
270{
271 if(!string) return false;
272
273 XSTRING _string = string;
274 int _classdev = 0;
275
276 if(_string.Find(DIOSTREAMUSBLOCALENUMDEVICES_PORTKEY, true) != XSTRING_NOTFOUND)

279 {
280 XSTRING _location;
281
282 _string.UnFormat(__L("%s,%d"), location.Get(), &classdev);
283
284 location = _location;
285 classdev = (DIOSTREAMDEVICEUSBCLASS)_classdev;
286
287 return true;
288 }

292 XDWORD _productID = 0;
293
294 _string.UnFormat(__L("0x%04X,0x%04X,%d"), &_vendorID, &_productID, &_classdev);
295
296 vendorID = _vendorID;
297 productID = _productID;
298 classdev = (DIOSTREAMDEVICEUSBCLASS)_classdev;
