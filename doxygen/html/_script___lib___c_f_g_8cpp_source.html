175void Call_GetFileCFGValue(SCRIPT_LIB* library, SCRIPT* script, XVECTOR<XVARIANT*>* params, XVARIANT* returnvalue)
176{
177 if(!library) return;
178 if(!script) return;
179 if(!params) return;
180 if(!returnvalue) return;
181
182 returnvalue->Set();
183
184 if(params->GetSize() < 2)
185 {
186 script->HaveError(SCRIPT_ERRORCODE_INSUF_PARAMS);
187 return;
188 }
189
190 XSTRING datakey;
191 SCRIPT_LIB_CFG* libraryCFG = (SCRIPT_LIB_CFG*)library;
192
193 if(libraryCFG->GetXFileCFG())
194 {
195 XSTRING* param1 = (XSTRING*)params->Get(0)->GetData();
196 XSTRING* param2 = (XSTRING*)params->Get(1)->GetData();
197
198 XVARIANT* variable = libraryCFG->GetXFileCFG()->GetValue(param1->Get(), param2->Get());
199 if(variable)
200 {
201 variable->ToString(datakey);
202 delete variable;
203 }
204
205 }
206
207 (*returnvalue) = datakey.Get();
208}
