239bool DIOI2CTEMHUMSENSORSHT20::ReadValue(XBYTE cmd, XWORD& value)
240{
241 if(!diostream)
242 {
243 XTRACE_PRINTCOLOR(4, __L("Not datastream comand %02X"), cmd);
244
245 return false;
246 }
247
248 bool status;
249 XBYTE command = cmd;
250
251 status = diostream->Write(&command, 1)?true:false;
252 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
253
254 if(!status)
255 {
256 XTRACE_PRINTCOLOR(4, __L("Not write comand %02X"), cmd);
257
258 return false;
259 }
260
261 GEN_XSLEEP.MilliSeconds(200);
262
263 XBYTE data[3] = { 0, 0, 0 };
264
265 status = diostream->WaitToFilledReadingBuffer(3, timeout);
266 if(status)
267 {
268 status = (diostream->Read(data, 3) == 3)?true:false;
269 if(!status)
270 {
271 XTRACE_PRINTCOLOR(4, __L("Not read comand %02X"), cmd);
272
273 return false;
274 }
275 }
276 else
277 {
278 XTRACE_PRINTCOLOR(4, __L("Not data to read comand %02X"), cmd);
279 return false;
280 }
281
282 XBYTE msb = data[0];
283 XBYTE lsb = data[1];
284 XBYTE checksum = data[2];
285
286 XWORD rawvalue = ((XWORD) msb << 8) | (XWORD) lsb;
287
288 if(!CheckCRC(rawvalue, checksum))
289 {
290 return false;
291 }
292
293
294 value = rawvalue & 0xFFFC;

372bool DIOI2CTEMHUMSENSORSHT20::CheckCRC(XWORD datasensor, XBYTE checkvalue)
373{
374 XDWORD remainder = (XDWORD)datasensor << 8;
375
376 remainder |= checkvalue;
377
378 XDWORD divisor = (XDWORD)DIOI2CTEMHUMSENSORSHT20_SHIFTED_DIVISOR;
379
380 for(int c=0; c < 16 ; c++)
381 {
382 if(remainder & (XDWORD)1 << (23 - c))
383 {
384 remainder ^= divisor;
385 }
386
387 divisor >>= 1;
388 }
389
390 return remainder?false:true;
391}
