663bool XSYSTEM::HostFile(bool add, XCHAR* domain, XCHAR* IP)
664{
665 XPATH xpath;
666 XFILETXT xfileTXT;
667 bool status = false;
668
669 #ifdef WINDOWS
670 xpath = __L("C:\\Windows\\System32\\drivers\\etc\\hosts");
671 #endif
672
673 #if defined(LINUX) || defined(ANDROID)
674 xpath = __L("/etc/hosts");
675 #endif
676
677 if(xfileTXT.Open(xpath, false))
678 {
679 xfileTXT.ReadAllFile();
680
681 if(add)
682 {
683 int nline = IsLineInHostFile(xfileTXT, domain, IP);
684 if(nline == XSTRING_NOTFOUND)
685 {
686 XSTRING line;
687
688 line.Format(__L("%s %s"), IP, domain);
689
690 xfileTXT.AddLine(line);
691 xfileTXT.WriteAllFile();
692 }
693 }
694 else
695 {
696 int nline = IsLineInHostFile(xfileTXT, domain, IP);
697 if(nline != XSTRING_NOTFOUND)
698 {
699 xfileTXT.DeleteLine(nline);
700 status = xfileTXT.WriteAllFile();
701 }
702 }
703
704 xfileTXT.Close();
705 }
706
707 return status;
708}

743int XSYSTEM::IsLineInHostFile(XFILETXT& xfileTXT, XCHAR* domain, XCHAR* IP)
744{
745 bool found = false;
746 int c = 0;
747
748 if(!xfileTXT.GetNLines()) return XSTRING_NOTFOUND;
749
750 do{ XSTRING* line = xfileTXT.GetLine(c);
751 if(line)
752 {
753 int index = line->Find(domain, true);
754 if(index != XSTRING_NOTFOUND)
755 {
756 found = true;
757
758 for(XDWORD d=0; d<line->GetSize(); d++)
759 {
760 XCHAR character = line->Get()[d];
761
762 if((character != __C('\t') && (character != __C(' '))))
763 {
764 if(character == __C('#'))
765 {
766 found = false;
767 }
768
769 break;
770 }
771 }
772
773 if(found) break;
774 }
775 }
776
777 c++;
778
779 } while(c<xfileTXT.GetNLines());
780
781 if(!found) return XSTRING_NOTFOUND;
782
783 return c;
784}
