139bool DIOLINUXPING::Do(XDWORD nretries, XDWORD timebetweenchecks, bool exitfirstgoodreply)
140{
141 #define DIOPING_DEFAULTTIMEOUT 3
142
143 if(!urltarget) return false;
144 if(!xtimer) return false;
145
146 XSTRING targetIP;
147 int handle;
148
149 DeleteAllReplys();
150
151 urltarget->ResolveURL(targetIP);
152
153 handle = socket(AF_INET, SOCK_RAW, IPPROTO_ICMP);
154 if(handle < 0) return false;
155
156 struct sockaddr_in targetaddr = { 0 };
157
158 targetaddr.sin_family = AF_INET;
159
160 XBUFFER charstr;
161
162 targetIP.ConvertToASCII(charstr);
163 targetaddr.sin_addr.s_addr = inet_addr(charstr.GetPtrChar());
164
165 DIOPING_ECHOREQUEST echorequest;
166 int ID = 0xCAFE;
167 int nseq = 2;
168 int nloop = nretries;
169
170 while(nloop)
171 {
172 HASHCRC32 crc32;
173 XDWORD crc32result;
174 XSTRING address;
175 XBUFFER xbuffer;
176
177 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
178 if(!xrand) break;
179
180 memset((XBYTE*)&echorequest, 0, sizeof(DIOPING_ECHOREQUEST));
181
182 // Fill in echo request
183 echorequest.icmphdr.type = DIOPING_ICMP_ECHOREQ;
184 echorequest.icmphdr.code = 0;
185 echorequest.icmphdr.ID = ID;
186 echorequest.icmphdr.seq = nseq++;
187
188 //Fill in some data to send
189 for(int nret=0; nret<DIOPING_REQ_TICKETSIZE; nret++)
190 {
191 echorequest.cdata[nret] = xrand->Max(255);
192 }
193
194 GEN_XFACTORY.DeleteRand(xrand);
195
196 crc32.ResetResult();
197 crc32.Do((XBYTE*)echorequest.cdata, DIOPING_REQ_TICKETSIZE);
198 crc32result = crc32.GetResultCRC32();
199
200 echorequest.applicationID = applicationID;
201 echorequest.time = (XDWORD)xtimer->GetMicroSecondsTickCounter(); // Save tick count when sent
202 echorequest.icmphdr.checksum = CalculeCheckSum((XWORD *)&echorequest, sizeof(DIOPING_ECHOREQUEST)); // Put data in packet and compute checksum
203
204 XDWORD size = sendto(handle,(char*)&echorequest, sizeof(DIOPING_ECHOREQUEST), 0, (sockaddr*)&targetaddr, sizeof(struct sockaddr_in));
205 if(size != sizeof(DIOPING_ECHOREQUEST))
206 {
207 XTRACE_PRINTCOLOR(4, __L("Ping: not write packet! %s"), targetIP.Get());
208 break;
209 }
210
211 xbuffer.Resize(sizeof(DIOPING_ECHOREPLY));
212
213 DIOPING_ECHOREPLY echoreply;
214 struct sockaddr_in originaddr;
215 socklen_t sizeaddr = sizeof(struct sockaddr_in);
216 XSTRING fromIP;
217 bool isreplypacket = false;
218
219 memset(&echoreply, 0, sizeof(DIOPING_ECHOREPLY));
220 memset(&originaddr, 0, sizeaddr);
221
222 xtimer->Reset();
223
224 int flags = fcntl(handle, F_GETFL, 0);
225 fcntl(handle, F_SETFL, flags | O_NONBLOCK);
226
227 do{ size = recvfrom(handle, (char*)xbuffer.Get(), xbuffer.GetSize(), 0, (struct sockaddr *)&originaddr, &sizeaddr);
228 if(size == sizeof(DIOPING_ECHOREPLY))
229 {
230 memcpy((XBYTE*)&echoreply.iphdr, xbuffer.Get(), xbuffer.GetSize());
231
232 if(echoreply.echorequest.icmphdr.type == DIOPING_ICMP_ECHOREPLY)
233 {
234 DIOIP ip;
235 ip.Set(echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b1,
236 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b2,
237 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b3,
238 echoreply.iphdr.iasource.INADDR_UNION.addrbyte.s_b4);
239
240 ip.GetXString(fromIP);
241
242 if(!fromIP.Compare(targetIP)) isreplypacket = true;
243 }
244 }
245
246 if(xtimer->GetMeasureSeconds() >= DIOPING_DEFAULTTIMEOUT) break;
247
248 } while(!isreplypacket);
249
250 if(isreplypacket)
251 {
252 crc32.ResetResult();
253 crc32.Do((XBYTE*)echoreply.echorequest.cdata, DIOPING_REQ_TICKETSIZE);
254
255 if(crc32result == crc32.GetResultCRC32())
256 {
257 XWORD checksum = CalculeCheckSum((XWORD *)&echoreply.echorequest, sizeof(DIOPING_ECHOREQUEST));
258 if(!checksum)
259 {
260 if(echoreply.echorequest.applicationID == applicationID)
261 {
262 XDWORD elapsed = (XDWORD)(xtimer->GetMicroSecondsTickCounter() - echoreply.echorequest.time);
263
264 AddReply(nloop+1, fromIP.Get(), DIOPING_REQ_TICKETSIZE, (elapsed/1000), echoreply.iphdr.TTL);
265
266 DIOPING_XEVENT xevent(this,DIOPING_XEVENT_TYPE_DOPING);
267 xevent.SetPingReply((DIOPINGREPLY*)replys.Get(replys.GetSize()-1));
268 PostEvent(&xevent);
269
270 //XTRACE_PRINTCOLOR(1, __L("Ping: received! %s %s"), fromIP.Get(), targetIP.Get());
271
272 if(!exitfirstgoodreply) GEN_XSLEEP.MilliSeconds(timebetweenchecks);
273 }
274 else
275 {
276 //XTRACE_PRINTCOLOR(4, __L("Ping: application ID not equal! %s"), fromIP.Get());
277 }
278 }
279 else
280 {
281 // XTRACE_PRINTCOLOR(4, __L("Ping: error Checksum! %s "), targetIP.Get());
282 }
283 }
284 else
285 {
286 // XTRACE_PRINTCOLOR(4, __L("Ping: error CRC! %s "), targetIP.Get());
287 }
288 }
289 else
290 {
291 // XTRACE_PRINTCOLOR(4, __L("Ping: invalid packet! %s "), targetIP.Get());
292 }
293
294 if(exitfirstgoodreply) nloop = 0; else nloop--;
295 }
296
297 close(handle);
298
299 bool status = WasConnected();
300
301 if(!status)
302 {
303 // XTRACE_PRINTCOLOR((status?1:4), __L("Ping to [%s]: %s"), targetIP.Get(), (status?__L("Ok."): __L("ERROR!")));
304 }
305
306 return status;
307}
