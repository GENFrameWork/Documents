141bool XWINDOWSCONSOLE::SetSize(int width, int height)
142{
143 HWND consolehandle = GetConsoleWindow();
144 RECT rect;
145 RECT rectdesktop;
146
147 if(GetWindowRect(GetDesktopWindow(), &rectdesktop) == FALSE) return false;
148
149 if(GetWindowRect(consolehandle, &rect) == FALSE) return false;
150
151 rect.left = (rectdesktop.right-width)/2;
152 rect.top = (rectdesktop.bottom-height)/2;
153
154 if(MoveWindow(consolehandle, rect.left, rect.top, width, height, TRUE) == FALSE) return false;
