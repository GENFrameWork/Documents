166bool DIOLINUXSTREAMUART::Open()
167{
168 if(!config) return false;
169
170 XSTRING ttyS;
171
172 if(!config->GetLocalDeviceName()->IsEmpty())
173 {
174 ttyS = config->GetLocalDeviceName()->Get();
175 }
176 else
177 {
178 ttyS.Format(__L("/dev/ttyS%d"),config->GetPort());
179 }
180
181 XBUFFER charstr;
182
183 ttyS.ConvertToASCII(charstr);
184 fd= open(charstr.GetPtrChar(), O_RDWR | O_NOCTTY | O_NONBLOCK);
185 if(fd<0) return false;
186
187 SetRTS(false);
188 SetDTR(false);
189
190 CleanBuffers();
191
192 #ifdef LINUX
193 fcntl(fd, F_SETFL, FNDELAY);
194 #else
195 fcntl(fd, F_SETFL, O_NONBLOCK);
196 #endif
197
198 Config();
199
200 SetEvent(DIOLINUXUARTFSMEVENT_CONNECTED);
201
202 status = DIOSTREAMSTATUS_CONNECTED;
203
204 ResetXBuffers();
205 ResetConnectionStatistics();
206
207 return threadconnection->Ini();
208}

222bool DIOLINUXSTREAMUART::Config(XWORD mask)
223{
224 if(fd<0) return false;
225
226 struct termios cfg;
227 speed_t speed;
228
229 tcgetattr(fd, &cfg); // save current port settings
230
231 if((mask&DIOSTREAMUARTMASK_BAUDRATE) == DIOSTREAMUARTMASK_BAUDRATE)
232 {
233 switch(this->config->GetBaudRate())
234 {
235 case 50 : speed = B50; break;
236 case 300 : speed = B300; break;
237 case 600 : speed = B600; break;
238 case 1200 : speed = B1200; break;
239 case 4800 : speed = B4800; break;
240 case 9600 : speed = B9600; break;
241 case 19200 : speed = B19200; break;
242 case 38400 : speed = B38400; break;
243 case 57600 : speed = B57600; break;
244 case 115200 : speed = B115200; break;
245 }
246
247 cfsetispeed(&cfg, speed);
248 cfsetospeed(&cfg, speed);
249 }
250
251 if((mask&DIOSTREAMUARTMASK_DATABITS ) == DIOSTREAMUARTMASK_DATABITS)
252 {
253 cfg.c_cflag&= ~CSIZE;
254
255 switch(this->config->GetDataBits())
256 {
257 case DIOSTREAMUARTDATABIT_5 : cfg.c_cflag|= CS5; break;
258 case DIOSTREAMUARTDATABIT_6 : cfg.c_cflag|= CS6; break;
259 case DIOSTREAMUARTDATABIT_7 : cfg.c_cflag|= CS7; break;
260 case DIOSTREAMUARTDATABIT_8 : cfg.c_cflag|= CS8; break;
261 }
262 }
263
264 if((mask&DIOSTREAMUARTMASK_PARITY) == DIOSTREAMUARTMASK_PARITY)
265 {
266 cfg.c_iflag = (IGNBRK| IGNPAR);
267
268 #ifdef CMSPAR
269 cfg.c_cflag &= ~CMSPAR;
270 #endif
271 cfg.c_cflag &= ~PARENB;
272 cfg.c_cflag &= ~PARODD;
273
274 switch(this->config->GetParity())
275 {
276 case DIOSTREAMUARTPARITY_NONE :
277 #ifdef CMSPAR
278 cfg.c_cflag &= ~CMSPAR;
279 #endif
280 cfg.c_cflag &= ~PARENB;
281 cfg.c_cflag &= ~PARODD;
282 break;
283 #ifdef CMSPAR
284 case DIOSTREAMUARTPARITY_MARK : cfg.c_cflag |= CMSPAR; // Not available on many systems.
285 #endif
286
287 case DIOSTREAMUARTPARITY_ODD : cfg.c_cflag |= PARENB;
288 cfg.c_cflag |= PARODD;
289 break;
290 #ifdef CMSPAR
291 case DIOSTREAMUARTPARITY_SPACE : cfg.c_cflag |= CMSPAR; // Not available on many systems.
292 #endif
293
294 case DIOSTREAMUARTPARITY_EVEN : cfg.c_cflag |= PARENB;
295 cfg.c_cflag &= ~PARODD;
296 break;
297 }
298 }
299
300 if((mask&DIOSTREAMUARTMASK_STOPBITS ) == DIOSTREAMUARTMASK_STOPBITS)
301 {
302 cfg.c_cflag &= ~CSTOPB;
303
304 switch(this->config->GetStopBits())
305 {
306 case DIOSTREAMUARTSTOPBITS_ONE : cfg.c_cflag &= ~CSTOPB; break;
307 case DIOSTREAMUARTSTOPBITS_ONEANDAHALF :
308 case DIOSTREAMUARTSTOPBITS_TWO : cfg.c_cflag |= CSTOPB; break;
309 }
310 }
311
312 if((mask&DIOSTREAMUARTMASK_FLOWCONTROL) == DIOSTREAMUARTMASK_FLOWCONTROL)
313 {
314 switch(config->GetFlowControl())
315 {
316 case DIOSTREAMUARTFLOWCONTROL_NONE : cfg.c_cflag &= ~CRTSCTS;
317 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
318 break;
319
320 case DIOSTREAMUARTFLOWCONTROL_HARD : cfg.c_cflag |= CRTSCTS;
321 cfg.c_iflag &= ~(IXON | IXOFF | IXANY);
322 break;
323
324 case DIOSTREAMUARTFLOWCONTROL_SOFT : cfg.c_cflag &= ~CRTSCTS;
325 cfg.c_iflag |= (IXON | IXOFF | IXANY);
326 break;
327 }
328 }
329
330 if((mask&DIOSTREAMUARTMASK_ALL) == DIOSTREAMUARTMASK_ALL)
331 {
332
333 cfg.c_cflag |= (CLOCAL | CREAD);
334
335 cfg.c_oflag &= ~OPOST;
336 cfg.c_lflag &= ~(ICANON | ECHO | ISIG);
337
338
339 cfg.c_cc[VMIN] = 0; // 0 means use-vtime
340 cfg.c_cc[VTIME] = 1; // time to wait until
341 }
342
343 tcsetattr(fd, TCSANOW, &cfg);
344
345 return true;
346}

627void DIOLINUXSTREAMUART::ThreadConnection(void* param)
628{
629 DIOLINUXSTREAMUART* diostream = (DIOLINUXSTREAMUART*)param;
630 if(!diostream) return;
631
632 if(diostream->GetEvent()==DIOLINUXUARTFSMEVENT_NONE) // No hay nuevos Eventos
633 {
634 switch(diostream->GetCurrentState())
635 {
636 case DIOLINUXUARTFSMSTATE_NONE : break;
637
638 case DIOLINUXUARTFSMSTATE_CONNECTED : break;
639
640 case DIOLINUXUARTFSMSTATE_WAITINGTOREAD : {
641
642 if(!diostream->IsBlockRead())
643 {
644 XBYTE buffer[DIOSTREAM_MAXBUFFER];
645
646 XDWORD br = diostream->ReadDirect(buffer, DIOSTREAM_MAXBUFFER);
647 if(br)
648 {
649 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(2), (XBYTE*)diostream->buffer, (int)br);
650 diostream->inbuffer->Add(buffer, br);
651 }
652 }
653
654 if(!diostream->IsBlockWrite())
655 {
656 int esize = diostream->outbuffer->GetSize();
657 if(esize)
658 {
659 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
660
661 diostream->outbuffer->SetBlocked(true);
662 XDWORD bw = diostream->WriteDirect(diostream->outbuffer->Get(), esize);
663 diostream->outbuffer->SetBlocked(false);
664
665 if(bw)
666 {
667 //XTRACE_PRINTDATABLOCK(XDEBUGLEVELCOLOR(4), (XBYTE*)diostream->buffer, (int)bw);
668 diostream->outbuffer->Extract(NULL, 0, bw);
669 }
670 }
671 }
672 }
673 break;
674
675 case DIOLINUXUARTFSMSTATE_DISCONNECTING : break;
676
677 }
678 }
679 else
680 {
681 if(diostream->GetEvent()<DIOLINUXUART_LASTEVENT)
682 {
683 diostream->CheckTransition();
684
685 switch(diostream->GetCurrentState())
686 {
687 case DIOLINUXUARTFSMSTATE_NONE : break;
688
689 case DIOLINUXUARTFSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
690 xevent.SetDIOStream(diostream);
691 diostream->PostEvent(&xevent);
692
693 diostream->SetEvent(DIOLINUXUARTFSMEVENT_WAITINGTOREAD);
694 }
695 break;
696
697 case DIOLINUXUARTFSMSTATE_WAITINGTOREAD : break;
698
699 case DIOLINUXUARTFSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
700 xevent.SetDIOStream(diostream);
701 diostream->PostEvent(&xevent);
702
703 diostream->threadconnection->Run(false);
704 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
705 }
706 break;
707 }
708 }
709 }
710}
