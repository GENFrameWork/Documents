200XDWORD COMPRESS_LZW::LZW_GIF_Encode(XBYTE* dibbuffer,XBYTE* outbuffer,XDWORD dibwidth,XDWORD dibheight,XWORD colorbit)
201{
202 hash = new XWORD [LZW_MAX_HASHSIZE] ;
203
204 if((hash == NULL) || (dibbuffer == NULL) || (outbuffer == NULL)) return 0;
205
206 bymincode = (colorbit == 1) ? 2 : colorbit;
207 LZWclear = 1 << bymincode;
208 LZWend = LZWclear + 1;
209 origin = curin = dibbuffer;
210 curout = outbuffer;
211 byinbit = 8;
212 byoutbit = 0;
213
214 m_dibwidth = dibwidth;
215 m_dibheight = dibheight;
216 currpixel = 0;
217 currheight = 0;
218 pitch = 4 * ((dibwidth * bymincode + 31) / 32);
219
220 XWORD Old;
221 XBYTE Pixel;
222
223 this->Encode_InitStringTable();
224 this->Encode_WriteIndex(LZWclear);
225
226 Old = this->Encode_GetNextPixel();
227
228 while(currheight<m_dibheight)
229 {
230 Pixel = this->Encode_GetNextPixel();
231
232 if(this->Encode_IsInTable (Old, Pixel))
233 {
234 Old = hash[(Old << 8) | Pixel];
235 }
236 else
237 {
238 this->Encode_WriteIndex(Old);
239 this->Encode_AddStringToTable(Old, Pixel);
240
241 Old = Pixel ;
242 if(currtableindex == LZW_MAX_TABLE_SIZE)
243 {
244 this->Encode_WriteIndex(Pixel);
245 this->Encode_WriteIndex(LZWclear);
246 this->Encode_InitStringTable();
247
248 Old = this->Encode_GetNextPixel();
249 }
250 }
251 }
252
253 this->Encode_WriteIndex(Old);
254 this->Encode_WriteIndex(LZWend);
255
256 delete [] hash;
257
258 return (XDWORD)(curout - outbuffer + 1);
259}

356XWORD COMPRESS_LZW::Decode_GetNextcode()
357{
358 XDWORD dwRet = 0;
359 unsigned int uiAdd = byinbit + bycurrbits;
360
361 if(uiAdd <= 8)
362 {
363 dwRet |= * curin;
364 }
365 else
366 {
367 if(uiAdd <= 16)
368 {
369 dwRet |= * (XWORD *) curin;
370 }
371 else
372 {
373 dwRet |= * (curin + 2);
374 dwRet <<= 16;
375 dwRet |= * (XWORD *) curin;
376 }
377 }
378
379 curin += uiAdd/8;

381
382 dwRet <<= 32 - uiAdd ;

530XBYTE COMPRESS_LZW::Encode_GetNextPixel()
531{
532 XBYTE byRet = 0;
533
534 switch(bymincode)
535 {
536 case 8 : byRet = curin[currpixel];
