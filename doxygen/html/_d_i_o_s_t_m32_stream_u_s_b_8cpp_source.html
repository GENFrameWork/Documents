317void DIOSTM32STREAMUSB::ThreadConnection(void* param)
318{
319 DIOSTM32STREAMUSB* diostream = (DIOSTM32STREAMUSB*)param;
320 if(!diostream) return;
321
322 if(diostream->GetEvent()==DIOSTM32USBFSMEVENT_NONE) // No hay nuevos Eventos
323 {
324 switch(diostream->GetCurrentState())
325 {
326 case DIOSTM32USBFSMSTATE_NONE : break;
327
328 case DIOSTM32USBFSMSTATE_CONNECTED : break;
329
330 case DIOSTM32USBFSMSTATE_WAITINGTOREAD : { if(!diostream->IsBlockRead())
331 {
332 if(!diostream->inbuffer->IsBlocked())
333 {
334 if(diostream->readcache.GetSize())
335 {
336 diostream->inbuffer->Add(diostream->readcache);
337 diostream->readcache.Delete();
338 }
339 }
340 }
341
342 if(!diostream->IsBlockWrite())
343 {
344 if(!diostream->outbuffer->IsBlocked())
345 {
346 int esize = diostream->outbuffer->GetSize();
347 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
348
349 if(esize)
350 {
351 diostream->outbuffer->Get(diostream->writebuffer, esize, 0);
352 if(diostream->WriteDirect(diostream->writebuffer, esize))
353 {
354 diostream->outbuffer->Extract(NULL, 0, esize);
355 }
356 }
357 }
358 }
359 }
360 break;
361
362 case DIOSTM32USBFSMSTATE_DISCONNECTING : break;
363
364 }
365 }
366 else
367 {
368 if(diostream->GetEvent()<DIOSTM32USB_LASTEVENT)
369 {
370 diostream->CheckTransition();
371
372 switch(diostream->GetCurrentState())
373 {
374 case DIOSTM32USBFSMSTATE_NONE : break;
375
376 case DIOSTM32USBFSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
377 xevent.SetDIOStream(diostream);
378 diostream->PostEvent(&xevent);
379
380 diostream->SetEvent(DIOSTM32USBFSMEVENT_WAITINGTOREAD);
381 }
382 break;
383
384 case DIOSTM32USBFSMSTATE_WAITINGTOREAD : break;
385
386 case DIOSTM32USBFSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
387 xevent.SetDIOStream(diostream);
388 diostream->PostEvent(&xevent);
389
390 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
391 }
392 break;
393 }
394 }
395 }
396}
