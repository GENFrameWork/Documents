104void ANDROIDNATIVEACTIVITY_EVENTLOOP::Run(ANDROIDNATIVEACTIVITY_HANDLER* activityhandler, ANDROIDNATIVEINPUT_HANDLER* inputhandler)
105{
106 int32_t result;
107 int32_t events;
108 android_poll_source* source;
109
110 //app_dummy();
111
112 this->activityhandler = activityhandler;
113 this->inputhandler = inputhandler;
114
115 while (true)
116 {
117 // Event processing loop.
118 while((result = ALooper_pollOnce(enabled?0:-1, NULL, &events, (void**)&source)) >= 0)
119 {
120 // An event has to be processed.
121 if(source != NULL) source->process(application, source);
122
123 // Application is getting destroyed.
124 if(application->destroyRequested) return;
125 }
126
127 if((enabled) && (!quit))
128 {
129 if(activityhandler->OnStep() != STATUS_OK)
130 {
131 quit = true;
132 ANativeActivity_finish(application->activity);
133 }
134 }
135 }
136}

195int32_t ANDROIDNATIVEACTIVITY_EVENTLOOP::ProcessInputEvent(AInputEvent* event, android_app* application)
196{
197 int32_t eventtype = AInputEvent_getType(event);
198
199 this->application = application;
200
201 switch(eventtype)
202 {
203 case AINPUT_EVENT_TYPE_MOTION : switch (AInputEvent_getSource(event))
204 {
205 case AINPUT_SOURCE_TOUCHSCREEN : return inputhandler->OnTouchEvent(event); break;
206 case AINPUT_SOURCE_TRACKBALL : return inputhandler->OnTrackballEvent(event); break;
207 }
208 break;
209
210 case AINPUT_EVENT_TYPE_KEY : return inputhandler->OnKeyboardEvent(event); break;
211 }
212
213 return 0;
214}

227void ANDROIDNATIVEACTIVITY_EVENTLOOP::ProcessApplicationEvent(int32_t command, android_app* application)
228{
229 this->application = application;
230
231 switch(command)
232 {
233 case APP_CMD_CONFIG_CHANGED : activityhandler->OnConfigurationChanged();
234 break;
235
236 case APP_CMD_INIT_WINDOW : activityhandler->OnCreateWindow();
237 break;
238
239 case APP_CMD_DESTROY : activityhandler->OnDestroy();
240 break;
241
242 case APP_CMD_GAINED_FOCUS : Activate();
243 activityhandler->OnGainFocus();
244 break;
245
246 case APP_CMD_LOST_FOCUS : activityhandler->OnLostFocus();
247 Deactivate();
248 break;
249
250 case APP_CMD_LOW_MEMORY : activityhandler->OnLowMemory();
251 break;
252
253 case APP_CMD_PAUSE : activityhandler->OnPause();
254 Deactivate();
255 break;
256
257 case APP_CMD_RESUME : Activate();
258 activityhandler->OnResume();
259 break;
260
261 case APP_CMD_SAVE_STATE : activityhandler->OnSaveState(&application->savedState, &application->savedStateSize);
262 break;
263
264 case APP_CMD_START : activityhandler->OnStart();
265 break;
266
267 case APP_CMD_STOP : activityhandler->OnStop();
268 break;
269
270 case APP_CMD_TERM_WINDOW : activityhandler->OnDestroyWindow();
271 Deactivate();
272 break;
273
274 default : break;
275 }
276}
