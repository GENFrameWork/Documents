332bool DIOSNMP_BASE::Open(XCHAR* IPtarget, int port, bool checkavailableNMS, XCHAR* localIP)
333{
334 if(!IPtarget) return false;
335 if(!IPtarget[0]) return false;
336
337 diostreamudpcfg = new DIOSTREAMUDPCONFIG();
338 if(!diostreamudpcfg) return false;
339
340 diostreamudpcfg->SetMode(DIOSTREAMMODE_CLIENT);
341
342 if(localIP) diostreamudpcfg->GetLocalIP()->Set(localIP);
343
344 diostreamudpcfg->GetRemoteURL()->Set(IPtarget);
345 diostreamudpcfg->SetRemotePort(port);
346
347 bool status = false;
348
349 diostreamudp = (DIOSTREAMUDP*)GEN_DIOFACTORY.CreateStreamIO(diostreamudpcfg);
350 if(diostreamudp)
351 {
352 if(diostreamudp->Open())
353 {
354 if(checkavailableNMS)
355 {
356 //--------------------------------------------
357 // Write something to received a ICMP result.
358 // and check if something is the other side.
359
360 status = diostreamudp->WaitToConnected(3);
361 if(status)
362 {
363 XBYTE data[] = { 0xAA,0x55 };
364 diostreamudp ->Write(data,sizeof(data));
365
366 status = diostreamudp->WaitToDisconnected(3);
367 }
368
369 } else status = true;
370 }
371 }
372
373 if(!status) Close();
374
375 return status;
376}

477bool DIOSNMP_TRAP::Send(XCHAR* community,XCHAR* enterprise,XCHAR* agentip,int generictrap,int specifictrap,XDWORD timestamp,int nvar,...)
478{
479 if(!diostreamudpcfg) return false;
480 if(!diostreamudp) return false;
481
482 XBUFFER xbuffer;
483 DIOSNMP_XBER trap;
484 DIOSNMP_XBER trapdata;
485 DIOSNMP_XBER trapvars;
486 DIOSNMP_XBER trapvar;
487 DIOSNMP_XBER xber;
488
489 // ----------------------------------------------------------------
490 // Trap
491 xber.SetINTEGER(version); trap.Sequence_AddTo(xber); // Version SNMPv1
492 xber.SetOCTETSTRING(community); trap.Sequence_AddTo(xber); // Community
493
494 // ----------------------------------------------------------------
495 // Trap Data
496
497 xber.SetOID(enterprise); trapdata.TrapData_AddTo(xber); // Enterprise
498 xber.SetIPADDRESS(agentip); trapdata.TrapData_AddTo(xber); // IP Address

500 xber.SetINTEGER(specifictrap); trapdata.TrapData_AddTo(xber); // Specific-trap
501 xber.SetTIMETICKS(timestamp); trapdata.TrapData_AddTo(xber); // timestamp
502
503 // ----------------------------------------------------------------
504 // Trap Data Variable
505 // (only available, for the moment, the next types: INTEGER, STRING and OIDs
506 //
507 if(nvar)
508 {
509 if(!(nvar%2))
510 {
511 va_list arg;
512
513 va_start(arg,nvar);
514
515 for(int c=0;c<(nvar/2);c++)
516 {
517 trapvar.Sequence_DeleteAll();
518
519 for(int d=0;d<2;d++)
520 {
521 int type = (int)va_arg(arg, int);
522
523 switch(type)
524 {
525 case XBER_TAGTYPE_INTEGER : { int integer = (int)va_arg(arg, int);
526 xber.SetINTEGER(integer);
527 trapvar.Sequence_AddTo(xber);
528 }
529 break;
530
531 case XBER_TAGTYPE_OCTET_STRING : { XCHAR* OCTETstring = (XCHAR*)va_arg(arg, XCHAR*);
532 if(OCTETstring)
533 {
534 xber.SetOCTETSTRING(OCTETstring);
535 trapvar.Sequence_AddTo(xber);
536 }
537 }
538 break;
539
540 case XBER_TAGTYPE_OID_IRI : { XCHAR* OIDstring = (XCHAR*)va_arg(arg, XCHAR*);
541 if(OIDstring)
542 {
543 xber.SetOID(OIDstring);
544 trapvar.Sequence_AddTo(xber);
545 }
546 }
547 break;
548
549 default : break;
550 }
551 }
552
553 trapvars.Sequence_AddTo(trapvar);
554 }
555
556 va_end(arg);
557
558 trapdata.TrapData_AddTo(trapvars);
559 }
560 }
561
562 trap.Sequence_AddTo(trapdata);
563
564 trap.GetDump(xbuffer);
565
566 //DEBUG_PRINTDATABLOCK(xbuffer.Get(),xbuffer.GetSize(),2);
567
568 XDWORD bw = diostreamudp->Write(xbuffer.Get(),xbuffer.GetSize());
569 if(bw!= xbuffer.GetSize()) return false;
570
571 return true;
572}

636bool DIOSNMP::Get(DIOSNMP_OPERATION operation, XCHAR* community, XCHAR* OIDstr, DIOSNMP_XBER& xberout)
637{
638 #define MAX_READBUFFER 2048
639
640 if(!diostreamudpcfg) return false;
641 if(!diostreamudp) return false;
642
643 XBUFFER xbuffer;
644 XBUFFER xbufferdata;
645 XDWORD number = 0;
646 DIOSNMP_XBER xber;
647 DIOSNMP_XBER value;
648 DIOSNMP_XBER bindings;
649 DIOSNMP_XBER request;
650 DIOSNMP_XBER data_next_request;
651 DIOSNMP_XBER header;
652
653 // ----------------------------------------------------------------
654 // Get-Next
655
656 xber.SetOID(OIDstr); value.Sequence_AddTo(xber); // OID
657 xber.SetNULL(); value.Sequence_AddTo(xber); // NULL
658
659 XRAND* GEN_XFACTORY_CREATE(xrand, CreateRand())
660 if(xrand)
661 {
662 number = xrand->Max(65000);
663 GEN_XFACTORY.DeleteRand(xrand);
664 }
665
666 xber.SetINTEGER(number); bindings.Sequence_AddTo(xber); // Request-ID
667 xber.SetINTEGER(0); bindings.Sequence_AddTo(xber); // Error-status
668 xber.SetINTEGER(0); bindings.Sequence_AddTo(xber); // Error-index
669
670 request.Sequence_AddTo(value);
671 bindings.Sequence_AddTo(request);
672
673 data_next_request.Sequence_AddCommandTo(DIOSNMP_OPERATION_GETNEXTREQUEST, bindings);
674
675 xber.SetINTEGER(version); header.Sequence_AddTo(xber); // Version
676 xber.SetOCTETSTRING(community); header.Sequence_AddTo(xber); // Community
677
678 header.Sequence_AddTo(data_next_request);
679
680 header.GetDump(xbuffer);
681
682 XDWORD bw = diostreamudp->Write(xbuffer.Get(), xbuffer.GetSize());
683 if(bw!= xbuffer.GetSize()) return false;
684
685 xbuffer.Empty();
686 xbuffer.Resize(MAX_READBUFFER);
687
688 XTIMER* GEN_XFACTORY_CREATE(timeout, CreateTimer())
689 if(!timeout) return false;
690
691 do{ XDWORD sizeread = diostreamudp->GetInXBuffer()->GetSize();
692 if(sizeread)
693 {
694 XDWORD br = diostreamudp->Read(xbuffer.Get(), sizeread);
695 if(br)
696 {
697 timeout->Reset();
698 xbufferdata.Add(xbuffer.Get(), br);
699 }
700 }
701
702 if(timeout->GetMeasureSeconds() > 2) break;
703
704 } while(1);
705
706 GEN_XFACTORY.DeleteTimer(timeout);
707
708 if(xbufferdata.GetSize())
709 {
710 xberout.SetTagType(xbufferdata.Get()[0]);
711 xberout.SetSize(xbufferdata.Get()[1]);
712 xberout.GetData()->Add(&xbufferdata.Get()[2], xbufferdata.GetSize());
713 }
714
715 return xbufferdata.GetSize()?true:false;
716}
