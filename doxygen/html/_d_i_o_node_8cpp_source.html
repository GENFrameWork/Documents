193bool DIONODE::GetSubGroups(XVECTOR<XSTRING*>& listsubgroups)
194{
195 XPATH pathgroup;
196
197 pathgroup = group.Get();
198
199 pathgroup.Slash_Delete();
200
201 do{ XSTRING* _subgroup = new XSTRING();
202 if(_subgroup)
203 {
204 pathgroup.GetNamefileExt(*_subgroup);
205 pathgroup.GetPath(pathgroup);
206
207 if(!_subgroup->Compare(pathgroup, true))
208 {
209 listsubgroups.Insert(0, _subgroup);
210 break;
211 }
212
213 pathgroup.Slash_Delete();
214
215 if(_subgroup->IsEmpty())
216 {
217 delete _subgroup;
218 break;
219 }
220 else
221 {
222 listsubgroups.Insert(0, _subgroup);
223 }
224 }
225 else
226 {
227 break;
228 }
229
230 } while(1);
231
232
233 return listsubgroups.IsEmpty()?false:true;
234}
