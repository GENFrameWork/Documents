114bool HASHCKS16::Do(XBYTE* input, XQWORD size)
115{
116 if(!size) return false;
117
118 XWORD value = 0;
119
120 resulttmp->Extract(value);
121
122 for(XDWORD c=0; c<size; c++)
123 {
124 value = Update(value, (*(XWORD*)input));
125 input = input + sizeof(XWORD); // need to offset the input by 16 bits, as we are adding 16 by 16
126 }
127
128 value = ~value;
129
130 resulttmp->Add((XWORD)value);
131
132 return true;
133}

263XWORD HASHCKS16::Update(XWORD CKS16, XWORD c)
264{
265 XDWORD tmp = (XDWORD)CKS16 + (XDWORD)c;
266
267 // loop while we have a carry
268 while(tmp > 0xFFFF)
269 {
270 XWORD carry = (tmp >> 16) & (0xFFFF);
271 tmp = tmp + carry;
272 }
273
274 return (XWORD)tmp;
275}
