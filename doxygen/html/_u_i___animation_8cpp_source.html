104bool UI_ANIMATION::LoadFromFile(XSTRING& resourcename, GRPPROPERTYMODE mode)
105{
106 XPATH xpathimg;
107 GRPBITMAPFILE bitmapfile;
108 GRPBITMAP* bitmap = NULL;
109 XSTRING mask;
110 int indexmask = 0;
111 int indeximg = 0;
112 GRPPROPERTYMODE modealpha = mode;
113 GRPBITMAPFILE_TYPE typefile = bitmapfile.GetTypeFromExtenxion(resourcename.Get());
114
115 modealpha = ChangeModeWithAlpha(typefile, mode);
116
117 while(1)
118 {
119 XSTRING namefile;
120
121 indexmask = resourcename.Copy(UI_ANIMATION_FRAME_INI, UI_ANIMATION_FRAME_END, true, 0, mask);
122
123 namefile = resourcename;
124
125 if(!mask.IsEmpty())
126 {
127 XSTRING result;
128 int masksize = mask.GetSize()+3;
129
130 namefile.DeleteCharacters(indexmask-masksize, masksize);
131
132 mask.Insert(__L("%"), 0);
133 mask.Add(__L("d"));
134
135 result.Format(mask.Get(), indeximg);
136 indeximg++;
137
138 namefile.Insert(result.Get(), indexmask-masksize);
139 }
140
141 GEN_XPATHSMANAGER.GetPathOfSection(XPATHSMANAGERSECTIONTYPE_GRAPHICS, xpathimg);
142 xpathimg.Slash_Add();
143 xpathimg.Add(namefile);
144
145 if(GEN_USERINTERFACE.IsZippedFile())
146 {
147 XFILEUNZIP* unzipfile = GEN_USERINTERFACE.GetUnzipFile();
148 if(unzipfile)
149 {
150 XPATH pathnamefilecmp(APPDEFAULT_DIRECTORY_GRAPHICS);
151 XPATH namefileonly;
152
153 pathnamefilecmp.Slash_Add();
154 pathnamefilecmp += namefile;
155
156 pathnamefilecmp.GetNamefileExt(namefileonly);
157
158 bool status = unzipfile->DecompressFile(pathnamefilecmp, (*GEN_USERINTERFACE.GetUnzipPathFile()), namefileonly.Get());
159 if(status)
160 {
161 XPATH unzippathfile_tmp;
162
163 unzippathfile_tmp = GEN_USERINTERFACE.GetUnzipPathFile()->Get();
164 unzippathfile_tmp += namefileonly;
165
166 bitmap = bitmapfile.Load(unzippathfile_tmp, modealpha);
167
168 GEN_USERINTERFACE.DeleteTemporalUnZipFile(unzippathfile_tmp);
169
170 if(bitmap)
171 {
172 bitmaps.Add(bitmap);
173 if(mask.IsEmpty())
174 {
175 break;
176 }
177 }
178 else
179 {
180 break;
181 }
182 }
183 else
184 {
185 break;
186 }
187 }
188 }
189 else
190 {
191 bitmap = bitmapfile.Load(xpathimg, modealpha);
192 if(bitmap)
193 {
194 bitmaps.Add(bitmap);
195 if(mask.IsEmpty())
196 {
197 break;
198 }
199 }
200 else
201 {
202 break;
203 }
204 }
205 }
206
207 return true;
208}

305GRPPROPERTYMODE UI_ANIMATION::ChangeModeWithAlpha(GRPBITMAPFILE_TYPE typefile, GRPPROPERTYMODE modescreen)
306{
307 GRPPROPERTYMODE mode = GRPPROPERTYMODE_XX_UNKNOWN;
308
309 if((typefile == GRPBITMAPFILE_TYPE_PNG) || (typefile == GRPBITMAPFILE_TYPE_TGA))
310 {
311 switch(modescreen)

313 case GRPPROPERTYMODE_XX_UNKNOWN : break;

315
