223bool DIOLEDNEOPIXELWS2812B::Send()
224{
225 XBYTE* ptrdata = databuffer.Get();
226 int sizedata = databuffer.GetSize();
227
228 DIOGPIO_ENTRY* entry = GEN_DIOGPIO.GPIOEntry_GetByID(GPIOentryID);
229 if(!entry) return false;
230
231 GEN_DIOGPIO.SetValue(entry, false);
232
233 for(int c=0; c<sizedata; c++)
234 {
235 XBYTE databyte = ptrdata[c];
236
237 for(int b=0; b<8; b++)
238 {
239 if(databyte & 0x80)
240 {
241 GEN_DIOGPIO.SetValue(entry, true);
242 GEN_XSLEEP.NanoSeconds(850);
243
244 GEN_DIOGPIO.SetValue(entry, false);
245 GEN_XSLEEP.NanoSeconds(400);
246 }
247 else
248 {
249 GEN_DIOGPIO.SetValue(entry, true);
250 GEN_XSLEEP.NanoSeconds(400);
251
252 GEN_DIOGPIO.SetValue(entry, false);
253 GEN_XSLEEP.NanoSeconds(850);
254 }
255
256 databyte <<= 1;
257 }
258 }
259
260 return true;
261}
