306bool DIOIPADDRESS::Set(XSTRING& addressstring)
307{
308 if(addressstring.IsEmpty()) return false;
309
310 int iip[DIOIP_MAXIPADDRESS] = { 0, 0, 0, 0 };
311 XBYTE bip[DIOIP_MAXIPADDRESS] = { 0, 0, 0, 0 };
312
313 if(addressstring.GetNCharacters(__C('.')) != 3) return false;
314
315 for(XDWORD c=0; c<addressstring.GetSize(); c++)
316 {
317 if((addressstring.Get()[c]<__C('0')) && (addressstring.Get()[c]>__C('9')) && (addressstring.Get()[c]!= __C('.'))) return false;
318 }
319
320 addressstring.UnFormat(__L("%d.%d.%d.%d"), &iip[0], &iip[1], &iip[2], &iip[3]);
321
322 for(int c=0; c<DIOIP_MAXIPADDRESS; c++)
323 {
324 bip[c] = (XBYTE)iip[c];
325 }
326
327 return Set(bip);
328}
