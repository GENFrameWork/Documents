182bool XPATHSMANAGER::GetPathOfSection(XPATHSMANAGERSECTIONTYPE sectiontype, XPATH& xpath, bool addroot)
183{
184 xpath.Empty();
185
186 if(addroot && (sectiontype!=XPATHSMANAGERSECTIONTYPE_ROOT))
187 {
188 for(XDWORD c=0; c<pathsections.GetSize(); c++)
189 {
190 XPATHSMANAGERSECTION* pathsection = pathsections.Get(c);
191 if(pathsection)
192 {
193 if(pathsection->type == XPATHSMANAGERSECTIONTYPE_ROOT)
194 {
195 xpath = pathsection->xpath->Get();
196 xpath.Slash_Add();
197 break;
198 }
199 }
200 }
201 }
202
203 for(XDWORD c=0; c<pathsections.GetSize(); c++)
204 {
205 XPATHSMANAGERSECTION* pathsection = pathsections.Get(c);
206 if(pathsection)
207 {
208 if(pathsection->type == sectiontype)
209 {
210 xpath += pathsection->xpath->Get();
211 break;

213 }
214 }

312{

441bool XPATHSMANAGER::AdjustRootPathDefault(XCHAR* assetsdirname)
442{
443 XDIR* xdir;
444 XPATH xpathroot;
445 XPATH xpathtest;
446 bool founddir = false;
447 int indexfound = 0;
448 bool status = false;
449
450 GEN_XFACTORY_CREATE(xdir, Create_Dir())
451 if(!xdir) return false;
452
453 GetPathOfSection(XPATHSMANAGERSECTIONTYPE_ROOT, xpathroot);
454 xpathroot.Slash_Add();
455
456 do{ indexfound = xpathroot.Find(__L("/"), false, indexfound);
457 if(indexfound == XSTRING_NOTFOUND) indexfound = xpathroot.Find(__L("\\"), false, indexfound);
458
459 if(indexfound != XSTRING_NOTFOUND)
460 {
461 xpathroot.Copy(0, indexfound, xpathtest);
462
463 xpathtest.Slash_Add();
464 xpathtest.Add(assetsdirname);
465
466 founddir = xdir->Exist(xpathtest);
467 if(founddir)
468 {
469 AddPathSection(XPATHSMANAGERSECTIONTYPE_ROOT, xpathtest);
470 status = true;
471 break;
472 }
473
474 indexfound++;
475 }
476
477 } while(indexfound != XSTRING_NOTFOUND);
478
479 if(!founddir)
480 {
481 xpathtest = xpathroot;
482 xpathtest.Slash_Add();
483 xpathtest.Add(assetsdirname);
484
485 if(xdir->Make(xpathtest, false))
486 {
487 AddPathSection(XPATHSMANAGERSECTIONTYPE_ROOT, xpathtest);
488 status = true;
489 }
490 }
491
492 GEN_XFACTORY.Delete_Dir(xdir);
493
494 return status;
495}
