106bool DIOWEBHEADER::Read(DIOSTREAMTCPIP* diostream, int timeout)
107{
108 if(!diostream) return false;
109
110 XSTRING line;
111 bool status;
112
113 DeleteAllLines();
114
115 XTIMER* GEN_XFACTORY_CREATE(xtimerout, CreateTimer())
116 if(!xtimerout) return false;
117
118 do{ status = diostream->ReadStr(line);
119 if(status)
120 {
121 if(line.IsEmpty())
122 {
123 break;
124 }
125 else
126 {
127 //XTRACE_PRINTCOLOR(3, __L("WEB CLIENT HEADER LINE: %s"), line.Get());
128 AddLine(line);
129 }
130 }
131
132 if(timeout)
133 {
134 if(xtimerout->GetMeasureSeconds()>(XDWORD)timeout) break;
135
136 GEN_XSLEEP.MilliSeconds(1);
137 }
138
139 } while((diostream->GetConnectStatus() == DIOSTREAMSTATUS_CONNECTED) || diostream->GetInXBuffer()->GetSize());
140
141 GEN_XFACTORY.DeleteTimer(xtimerout);
142
143
144 if(lines.IsEmpty()) return false;
145
146 return true;

359bool DIOWEBHEADER::GetFormat(bool isserver, int type, void* data)
360{
361 if(lines.IsEmpty()) return false;
362
363 if(!data) return false;
364
365 XSTRING line;
366 int version = 0;
367 int subversion = 0;
368
369 line =((XSTRING*)lines.Get(0))->Get();
370
371 if(isserver)
372 {
373 int code;
374 XSTRING codestring(DIOWEBHEADER_MAXLINE);
375
376 line.UnFormat(__L("HTTP/%d.%d %d %s"),&version,&subversion,&code,codestring.Get());
377 codestring.AdjustSize();
378
379 switch(type)
380 {
381 case 0: (*(int*)data) = version; break;
382 case 1: (*(int*)data) = subversion; break;
383 case 2: (*(int*)data) = code; break;
384 case 3: (*(XSTRING*)data) = codestring; break;
385 }
386 }
387 else
388 {
389 XSTRING method(DIOWEBHEADER_MAXLINE);
390 XSTRING resource(DIOWEBHEADER_MAXLINE);
391
392 if(line.GetSize())
393 {
394 int foundindex = line.Find(__L("HTTP/"), true);
395
396 if(foundindex != XSTRING_NOTFOUND) line.UnFormat(__L("%s %s HTTP/%d.%d"), method.Get(), resource.Get(), &version, &subversion);
397 }
398
399 method.AdjustSize();
400 resource.AdjustSize();
401
402 switch(type)
403 {
404 case 0: if(data) (*(XSTRING*)data) = method; break;
405 case 1: if(data) (*(XSTRING*)data) = resource; break;
406 case 2: if(data) (*(int*)data) = version; break;
407 case 3: if(data) (*(int*)data) = subversion; break;
408 }
409 }
410
411 return true;
412}
