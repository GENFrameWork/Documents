126bool INPLINUXDEVICEKEYBOARDX11::Update()
127{
128 if((!created)||(!enabled)) return false;
129
130 if(!grpscreenx11) return false;
131 if(!grpscreenx11->HasFocus()) return false;
132
133 XEvent event;
134
135 for(int e=0; e<this->buttons.GetSize(); e++)
136 {
137 INPBUTTON* button = buttons.FastGet(e);
138
139 if(button->GetState() == INPBUTTON_STATE_RELEASED) button->SetState(INPBUTTON_STATE_UP);
140 if(button->GetState() == INPBUTTON_STATE_PRESSED) button->SetState(INPBUTTON_STATE_HOLD);
141 }
142
143 while(XCheckWindowEvent(grpscreenx11->GetDisplay(),(*grpscreenx11->GetWindow()), KeyPressMask | KeyReleaseMask ,&event))
144 {
145 XKeyEvent* keyevent = (XKeyEvent *)&event;
146
147 switch(event.type)
148 {
149 case KeyPress : { long keysym = XLookupKeysym (keyevent, 0);
150 INPBUTTON* button = GetButtonByCode(keysym);
151
152 if(!button)
153 {
154 KeySym low = 0;
155 KeySym upper = 0;
156
157 XConvertCase(keysym,&low,&upper);
158
159 button = GetButtonByCode(upper);
160 }
161
162 if(!button) break;
163
164 if(button->GetState() == INPBUTTON_STATE_UP) button->SetState(INPBUTTON_STATE_PRESSED);
165
166 button->SetPressed(true);
167 }
168 break;
169
170 case KeyRelease : { long keysym = XLookupKeysym (keyevent, 0);
171 INPBUTTON* button = GetButtonByCode(keysym);
172
173 if(!button)
174 {
175 KeySym low = 0;
176 KeySym upper = 0;
177
178 XConvertCase(keysym,&low,&upper);
179
180 button = GetButtonByCode(upper);
181 }
182
183 if(!button) break;
184
185 button->SetState(INPBUTTON_STATE_RELEASED);
186
187 button->SetPressed(false);
188 }
189 break;
190 }
191
192 }
193 return true;
194}
