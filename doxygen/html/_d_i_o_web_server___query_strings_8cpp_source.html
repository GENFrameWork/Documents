459int DIOWEBSERVER_QUERYSTRINGS::GetParamsFromURL(XCHAR* url)
460{
461 DIOURL _url;
462 int start;
463 int separator;
464 int end;
465 XSTRING param;
466
467 DelAllParam();
468
469 _url = url;
470
471 start = _url.Find(__L("?"),true);
472 if(start != XSTRING_NOTFOUND)
473 {
474 XSTRING ID;
475 XSTRING value;
476
477 start++;
478
479 do{
480 value.Empty();
481 end = _url.Find(__L("&"),true,start);
482 if(end == XSTRING_NOTFOUND)
483 _url.Copy(start, param);
484 else _url.Copy(start, end, param);
485
486 separator = param.Find(__L("="),true);
487 if(separator == XSTRING_NOTFOUND)
488 {
489 ID = param;
490 }
491 else
492 {
493 param.Copy(0 , separator,ID);
494 param.Copy(separator+1, value);
495 }
496
497 AddParam(ID, value);
498
499 start = end+1;
500
501 } while(end != XSTRING_NOTFOUND);
502 }
503
504
505 return GetNParams();
506}
