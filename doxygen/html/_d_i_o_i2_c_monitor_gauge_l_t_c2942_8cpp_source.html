107float DIOI2CMONITORGAUGELTC2942::GetVoltage (void)
108{
109 if(!diostream) return false;
110
111 float voltage_return = 0;
112 XBYTE buffer[2] = { 0, 0 };
113
114 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_VOL_H, buffer, 2)) return 0;
115
116 XDWORD read_value = buffer[1] | (buffer[0] << 8);
117
118 if(read_value == DIOI2CMONITORGAUGELTC2942_VOLTAGE_READ_ERROR)
119 {
120 return 0;
121 }
122
123 voltage_return = (float)read_value;
124
125 return ((voltage_return* 6000) / 65535)/1000;
126}

140bool DIOI2CMONITORGAUGELTC2942::SetAC(XWORD ac)
141{
142 XBYTE ctrl;
143
144 // Backup control register
145 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL,ctrl)) return false;
146
147 // Shutdown analog section before programming new AC value
148 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl | DIOI2CMONITORGAUGELTC2942_CTL_SHUTDOWN)) return false;
149
150 // Program new AC value
151 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_AC_H, ac >> 8)!=0) return false;
152
153 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_AC_L, ac & 0xFF)!=0) return false;
154
155 // Restore control register
156 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl)!=0) return false;
157
158 return true;
159}

252bool DIOI2CMONITORGAUGELTC2942::Initialize()
253{
254 XBYTE ctrl;
255
256 if(!ReadRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl)) return false;
257
258 // Enable auto measurement of battery voltage and temperature
259 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_ADC_MSK; // clear ADC mode bits
260 ctrl |= DIOI2CMONITORGAUGELTC2942_ADC_AUTO; // configure mode
261
262 // Enable analog section of the chip
263 ctrl &= ~DIOI2CMONITORGAUGELTC2942_CTL_SHUTDOWN;
264
265 // Set prescaler M value
266 //
267 // 0.085mAh • 50momh • M
268 // qLSB = ----------------------- =
269 // RSENSE • 128
270
271 // 0.085mAh • 50momh • 16
272 // qLSB = ----------------------- = 0,0133mAh -> MaxValue: 870,38mAh (not valid)
273 // 40momh • 128
274
275 // 0.085mAh • 50momh • 32
276 // qLSB = ----------------------- = 0,0265625mAh -> MaxValue: 1.740,7mAh valid
277 // 40momh • 128
278
279 // M=32 for 1050mA battery AC max value 39.530
280
281 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_PSCM_MSK; // clear prescaler M bits
282 ctrl |= DIOI2CMONITORGAUGELTC2942_PSCM_32; // configure new value
283
284 // Disable AL/CC pin
285 ctrl &= DIOI2CMONITORGAUGELTC2942_CTL_ALCC_MSK; // clear AL/CC bits
286 ctrl |= DIOI2CMONITORGAUGELTC2942_ALCC_DISABLED; // configure new value
287
288 // Write control register
289 if(!WriteRegister(DIOI2CMONITORGAUGELTC2942_REG_CONTROL, ctrl))
290 {
291 return false;
292 }
293
294 // Checks if there was power off and sets de acumulator in middle value
295 XWORD currentAC = GetAC();
296 if(currentAC >= 0x7FF0) SetAC(DIOI2CMONITORGAUGELTC2942_AC_MAXVALUE/2);
297
298 return true;
299}

333bool DIOI2CMONITORGAUGELTC2942::ReadRegister(XBYTE offset, XBYTE* data, XDWORD sizedata)
334{
335 if(!diostream) return false;
336
337 bool status;
338
339 status = diostream->Write(&offset, 1)?true:false;
340 if(status) status = diostream->WaitToFlushOutXBuffer(timeout);
341
342 if(!status) return false;
343
344 status = diostream->WaitToFilledReadingBuffer(sizedata, timeout);
345 if(status) status = (diostream->Read(data, sizedata) == sizedata)?true:false;
346
347 return status;
348}
