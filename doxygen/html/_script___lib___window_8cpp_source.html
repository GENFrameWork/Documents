948bool FindSubBitmap(GRPBITMAP* bitmapscreen, GRPBITMAP* bitmapref, int& x, int& y, XBYTE difflimitpercent, XBYTE pixelmargin)

971 return false;

976 XDWORD sizepixel = sizeof(XDWORD);

978 XDWORD bufferbmplinesize = _bitmap->GetWidth();

1004
1005 found = false;

1008 XDWORD scrpos_tmp = scrpos;
1009 XDWORD bmppos_tmp = bufferbmplinesize;
1010
1011 x = (scrpos % bitmapscreen->GetWidth());
1012 y = bitmapscreen->GetHeight() - (scrpos / srcpixelsline) - _bitmap->GetHeight();

1053 }

1058 }

1060 }

1066 }
