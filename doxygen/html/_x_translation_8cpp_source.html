215bool XTRANSLATION::Translate_Add(XDWORD ID, XCHAR* sentence, XDWORD fixed)
216{
217 if(!ID) return false;
218 if(!sentence) return false;
219
220 XDWORD sizesentence = XSTRING::GetSize(sentence);
221 if(fixed)
222 {
223 if(sizesentence > fixed) sizesentence = fixed;
224 }
225
226 XCHAR* newsentence = new XCHAR[sizesentence+1];
227 if(newsentence)
228 {
229 memset(newsentence, 0 , (sizesentence+1) * sizeof(XCHAR));
230 memcpy(newsentence, sentence , sizesentence * sizeof(XCHAR));
231
232 sentences[ID] = newsentence;
233 }
234
235 return true;
236}

326 XFILEJSONARRAY* languages = (XFILEJSONARRAY*)xfilejson.GetObject(__L("languages"));
327 if(languages)
328 {
329 for(XDWORD c=0; c<languages->GetValues()->GetSize(); c++)
330 {
331 XFILEJSONVALUE* value = languages->GetValues()->Get(c);
332 if(value)
333 {

451bool XTRANSLATION::Translate_Load()
452{
453 #ifdef XFILE_ACTIVE
454
455 XFILEJSON xfilejson;
456
457 if(!xfilejson.Open(xpath)) return false;
458 if(!xfilejson.ReadAllFile()) return false;
459 if(!xfilejson.DecodeAllLines()) return false;
460
461 int languageindex = GetIndexLanguageByCode(code);
462 if(languageindex == -1) return false;
463
464 XFILEJSONARRAY* translations = (XFILEJSONARRAY*)xfilejson.GetObject(__L("translations"));
465 if(translations)
466 {
467 for(XDWORD c=0; c<translations->GetValues()->GetSize(); c++)
468 {
469 XFILEJSONVALUE* translation = translations->GetValues()->Get(c);
470 if(translation)
471 {
472 XSTRING IDstr = translation->GetName()->Get();
473 if(!IDstr.IsEmpty())
474 {
475 XDWORD ID = IDstr.ConvertToDWord();
476 if(ID < XTRANSLATION_MAXSENTENCES)
477 {
478 XFILEJSONARRAY* translationarray = (XFILEJSONARRAY*)translation->GetValuePointer();
479 if(translationarray)
480 {
481 XDWORD sizesentencefixed = (XDWORD)translationarray->GetValues()->Get(0)->GetValueInteger();
482 XSTRING sentence = (XSTRING)translationarray->GetValues()->Get(languageindex+1)->GetValueString();
483
484 Translate_Add(ID, sentence.Get(), sizesentencefixed);
485 }
486 }
487 }
488 }
489 }
490 }
491
492 xfilejson.Close();
