257bool DIOSTM32STREAMI2C::Open()
258{
259 bool result = false;
260
261 if(!threadconnection) return false;
262 if(!config) return false;
263
264 for(int c=0; c<DIOSTM32STREAMI2C_MAXHANDLES; c++)
265 {
266 if(DIOSTM32STREAMI2C::handles[c] == this) return false;
267
268 if(!DIOSTM32STREAMI2C::handles[c])
269 {
270 indexhandle = c;
271 DIOSTM32STREAMI2C::handles[indexhandle] = this;
272 break;
273 }
274 }
275
276 if(indexhandle == -1) return false;
277
278 if(config->GetLocalAddress() != DIOSTREAMI2CCONFIG_ADDRESSINVALID) localaddress = (config->GetLocalAddress() << 1);
279 if(config->GetRemoteAddress() != DIOSTREAMI2CCONFIG_ADDRESSINVALID) remoteaddress = (config->GetRemoteAddress() << 1);
280
281 indexport = config->GetPort()-1;
282
283 if(!DIOSTM32STREAMI2C::ports[indexport])
284 {
285 DIOSTM32STREAMI2CPORT* port = new DIOSTM32STREAMI2CPORT();
286 if(!port) return false;
287
288 hi2c = new I2C_HandleTypeDef();
289 if(!hi2c)
290 {
291 delete port;
292 return false;
293 }
294
295 switch(config->GetPort())
296 {
297 case 1 : hi2c->Instance = I2C1; break;
298 #if defined(STM32F411xE) || defined(STM32F446x)
299 case 2 : hi2c->Instance = I2C2; break;
300 #endif
301 #if defined(HW_STM32WB55XX)
302 case 3 : hi2c->Instance = I2C3; break;
303 #endif
304
305 default : return false;
306 }
307
308 #if defined(HW_STM32F0XX) || defined(HW_STM32F3XX)
309 hi2c->Init.Timing = 0x2000090E;
310 #endif
311
312 #if defined(HW_STM32WB55XX)
313 hi2c->Init.Timing = 0x10707DBC;
314 //hi2c->Init.Timing = 0x00602173; // (400KHz)
315 #endif
316
317 #if defined(STM32F411xE) || defined(STM32F446x)
318 hi2c->Init.ClockSpeed = 100000;
319 #endif
320
321 hi2c->Init.OwnAddress1 = localaddress;
322
323 switch(config->GetAddressSize())
324 {
325 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_DEFAULT :
326 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_7BITS : hi2c->Init.AddressingMode = I2C_ADDRESSINGMODE_7BIT; break;
327 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_10BITS : hi2c->Init.AddressingMode = I2C_ADDRESSINGMODE_10BIT; break;
328 }
329
330 hi2c->Init.DualAddressMode = I2C_DUALADDRESS_DISABLE;
331 hi2c->Init.OwnAddress2 = 0;
332 #if defined(HW_STM32F0XX) || defined(HW_STM32F3XX) || defined(HW_STM32WB55XX)
333 hi2c->Init.OwnAddress2Masks = I2C_OA2_NOMASK;
334 #endif
335 hi2c->Init.GeneralCallMode = I2C_GENERALCALL_DISABLE;
336 hi2c->Init.NoStretchMode = config->IsStretchActive()?I2C_NOSTRETCH_ENABLE:I2C_NOSTRETCH_DISABLE;
337
338 result =(HAL_I2C_Init(hi2c) == HAL_OK)?true:false;
339 if(result)
340 {
341 #if defined(HW_STM32F0XX) || defined(HW_STM32F3XX) || defined(HW_STM32WB55XX)
342 result = (HAL_I2CEx_ConfigAnalogFilter(hi2c, I2C_ANALOGFILTER_ENABLE) == HAL_OK)?true:false;
343 #endif
344
345 if(result)
346 {
347 result = (HAL_I2CEx_ConfigDigitalFilter(hi2c, 0) == HAL_OK)?true:false;
348 if(result)
349 {
350 if(config->GetMode() == DIOSTREAMMODE_MASTER)
351 {
352 result = (HAL_I2C_IsDeviceReady(hi2c, (uint16_t)remoteaddress, 10, 1000) == HAL_OK)?true:false;
353 }
354 }
355 }
356 }
357
358 port->SetHandleI2C(hi2c);
359 DIOSTM32STREAMI2C::ports[indexport] = port;
360 }
361 else
362 {
363 hi2c = DIOSTM32STREAMI2C::ports[indexport]->GetHandleI2C();
364 DIOSTM32STREAMI2C::ports[indexport]->SetCounterRef(DIOSTM32STREAMI2C::ports[indexport]->GetCounterRef()+1);
365
366 result = true;
367 }
368
369 if(result)
370 {
371 SetEvent(DIOSTREAMI2C_FSMEVENT_CONNECTED);
372
373 status = DIOSTREAMSTATUS_CONNECTED;
374
375 memset(bufferdata, 0, DIOSTREAM_MAXBUFFER);
376
377 ResetXBuffers();
378 ResetConnectionStatistics();
379
380 result =threadconnection->Ini();
381 }
382
383 return result;
384}

402 if(filledto == DIOSTREAM_SOMETHINGTOREAD) sizebufferdata = 1; else sizebufferdata = filledto;

432XDWORD DIOSTM32STREAMI2C::ReadDirect(XBYTE* buffer, XDWORD size)
433{
434 if(!config) return 0;
435 if(GetConnectStatus()==DIOSTREAMSTATUS_DISCONNECTED) return 0;
436
437 XDWORD br = 0;
438
439 switch(config->GetMode())
440 {
441 case DIOSTREAMMODE_MASTER : if(HAL_I2C_Master_Receive(hi2c, (uint16_t)remoteaddress, buffer, size, 1000) == HAL_OK) br = size;
442 break;
443
444 case DIOSTREAMMODE_SEMIMASTER :
445 case DIOSTREAMMODE_SLAVE : if(HAL_I2C_Slave_Receive(hi2c, buffer, size, 1000) == HAL_OK) br = size;
446 break;
447
448 default : break;
449 }
450
451 return br;
452}

696void DIOSTM32STREAMI2C::ThreadConnection(void* param)
697{
698 DIOSTM32STREAMI2C* diostream = (DIOSTM32STREAMI2C*)param;
699 if(!diostream) return;
700
701 if(diostream->GetEvent()==DIOSTREAMI2C_FSMEVENT_NONE)
702 {
703 switch(diostream->GetCurrentState())
704 {
705 case DIOSTREAMI2C_FSMSTATE_NONE : break;
706
707 case DIOSTREAMI2C_FSMSTATE_CONNECTED : break;
708
709 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : if(!diostream->IsBlockRead())
710 {
711 if(!diostream->inbuffer->IsBlocked())
712 {
713 if(diostream->readcache[diostream->indexport].GetSize())
714 {
715 diostream->inbuffer->Add(diostream->readcache[diostream->indexport]);
716 diostream->readcache[diostream->indexport].Delete();
717 }
718 }
719 }
720
721 if(!diostream->IsBlockWrite())
722 {
723 if(!diostream->outbuffer->IsBlocked())
724 {
725 int esize = diostream->outbuffer->GetSize();
726 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
727
728 if(esize)
729 {
730 diostream->outbuffer->Extract(diostream->writebuffer, 0, esize);
731 XDWORD _esize = diostream->WriteDirect(diostream->writebuffer, esize);
732 if(_esize != esize)
733 {
734 diostream->outbuffer->Add(diostream->writebuffer, esize);
735 }
736 }
737 }
738 }
739 break;
740
741 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : break;
742
743 }
744 }
745 else
746 {
747 if(diostream->GetEvent()<DIOSTREAMI2C_LASTEVENT)
748 {
749 diostream->CheckTransition();
750
751 switch(diostream->GetCurrentState())
752 {
753 case DIOSTREAMI2C_FSMSTATE_NONE : break;
754
755 case DIOSTREAMI2C_FSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
756 xevent.SetDIOStream(diostream);
757 diostream->PostEvent(&xevent);
758
759 diostream->SetEvent(DIOSTREAMI2C_FSMEVENT_WAITINGTOREAD);
760 }
761 break;
762
763 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : break;
764
765 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
766 xevent.SetDIOStream(diostream);
767 diostream->PostEvent(&xevent);
768
769 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
770 }
771 break;
772 }
773 }
774 }
775}
