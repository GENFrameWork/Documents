140#define XTRACE_CRC16(ptrbuffer, buffersize, crc) { XBYTE* buffer = ptrbuffer; \
141 XDWORD size = buffersize; \
142 XBYTE x; \
143 crc = 0xFFFF; \
144 while(size--) \
145 { \
146 x = crc >> 8 ^ * buffer++; \
147 x ^= x>>4; \
148 crc = (crc << 8) ^ ((XWORD)(x << 12)) ^ ((XWORD)(x <<5)) ^ ((XWORD)x); \
149 } \
