232bool DIOSCRAPERWEBTRANSLATION::ChangeURL(XCHAR* maskurl, DIOURL& url)
233{
234 XSTRING _maskurl;
235
236 _maskurl = maskurl;
237
238 int found = _maskurl.Find(__L("google"),true);
239 if(found != XSTRING_NOTFOUND)
240 {
241 DIOURL urltext;
242 XBUFFER urlbuffer;
243 XSTRING googlecodetext;
244 XSTRING googlecodetranslate;
245
246 urltext = text.Get();
247 urltext.ConvertToUTF8(urlbuffer);
248 urltext.EncodeUnsafeCharsFromBuffer(urlbuffer);
249
250 GoogleLanguageCode(languagetextcode , googlecodetext);
251 GoogleLanguageCode(languagetranslationcode, googlecodetranslate);
252
253 url.Format(maskurl, urltext.Get(), googlecodetext.Get(), googlecodetext.Get(), googlecodetranslate.Get());
254
255 return true;
256 }
257
258
259 return false;
260}

280bool DIOSCRAPERWEBTRANSLATION::Get(XCHAR* text, XLANGUAGE_CODE languagetextcode, XLANGUAGE_CODE languagetranslationcode, XSTRING& translation, int timeoutforurl, XSTRING* localIP, bool usecache)
281{
282 XSTRING _text;
283 bool status = false;
284
285 if(xmutexdo) xmutexdo->Lock();
286
287 _text.Format(__L("%s %02d %02d"), text, languagetextcode, languagetranslationcode);
288
289 this->text = text;
290 this->languagetextcode = languagetextcode;
291 this->languagetranslationcode = languagetranslationcode;
292
293 DIOTRANSLATION_RESULT* translationresult;
294
295 if(usecache)
296 {
297 translationresult= (DIOTRANSLATION_RESULT*)cache->Get(_text);
298 if(translationresult)
299 {
300 translation = (*translationresult->GetTranslation());
301
302 if(xmutexdo) xmutexdo->UnLock();
303 return true;
304 }
305 }
306
307 if(Load(DIOSCRAPERWEBTRANSLATION_NAMEFILE))
308 {
309 if(Do(DIOSCRAPERWEBTRANSLATION_NAMESERVICE, timeoutforurl, localIP))
310 {
311 translation = GetValue(__L("TRANSLATION"));
312
313 if(usecache)
314 {
315 translationresult = new DIOTRANSLATION_RESULT();
316 if(translationresult)
317 {
318 translationresult->SetLanguageTextCode(languagetextcode);
319 translationresult->SetLanguageTranslationCode(languagetranslationcode);
320 translationresult->GetTranslation()->Set(translation);
321
322 cache->Add(_text, translationresult);
323 }
324 }
325
326 status = true;
327 }
328 }
329
330 if(xmutexdo) xmutexdo->UnLock();
331
332 return status;
333}
