271bool DIOESP32STREAMI2C::Open()
272{
273 bool result = false;
274
275 if(!threadconnection) return false;
276 if(!config) return false;
277
278 for(int c=0; c<DIOESP32STREAMI2C_MAXHANDLES; c++)
279 {
280 if(DIOESP32STREAMI2C::handles[c] == this) return false;
281
282 if(!DIOESP32STREAMI2C::handles[c])
283 {
284 indexhandle = c;
285 DIOESP32STREAMI2C::handles[indexhandle] = this;
286 break;
287 }
288 }
289
290 if(indexhandle == -1) return false;
291
292 if(config->GetLocalAddress() != DIOSTREAMI2CCONFIG_ADDRESSINVALID) localaddress = (config->GetLocalAddress() << 1);
293 if(config->GetRemoteAddress() != DIOSTREAMI2CCONFIG_ADDRESSINVALID) remoteaddress = (config->GetRemoteAddress() << 1);
294
295 indexport = config->GetPort()-1;
296
297 if(!DIOESP32STREAMI2C::ports[indexport])
298 {
299 DIOESP32STREAMI2CPORT* port = new DIOESP32STREAMI2CPORT();
300 if(!port) return false;
301
302 hi2c = new I2C_HandleTypeDef();
303 if(!hi2c)
304 {
305 delete port;
306 return false;
307 }
308
309 switch(config->GetPort())
310 {
311 case 1 : hi2c->Instance = I2C1; break;
312 #if defined(ESP32F411xE) || defined(ESP32F446x)
313 case 2 : hi2c->Instance = I2C2; break;
314 #endif
315 #if defined(HW_ESP32WB55XX)
316 case 3 : hi2c->Instance = I2C3; break;
317 #endif
318
319 default : return false;
320 }
321
322 #if defined(HW_ESP32F0XX) || defined(HW_ESP32F3XX)
323 hi2c->Init.Timing = 0x2000090E;
324 #endif
325
326 #if defined(HW_ESP32WB55XX)
327 hi2c->Init.Timing = 0x10707DBC;
328 //hi2c->Init.Timing = 0x00602173; // (400KHz)
329 #endif
330
331 #if defined(ESP32F411xE) || defined(ESP32F446x)
332 hi2c->Init.ClockSpeed = 100000;
333 #endif
334
335 hi2c->Init.OwnAddress1 = localaddress;
336
337 switch(config->GetAddressSize())
338 {
339 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_DEFAULT :
340 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_7BITS : hi2c->Init.AddressingMode = I2C_ADDRESSINGMODE_7BIT; break;
341 case DIOSTREAMI2CCONFIG_ADDRESSSIZE_10BITS : hi2c->Init.AddressingMode = I2C_ADDRESSINGMODE_10BIT; break;
342 }
343
344 hi2c->Init.DualAddressMode = I2C_DUALADDRESS_DISABLE;
345 hi2c->Init.OwnAddress2 = 0;
346 #if defined(HW_ESP32F0XX) || defined(HW_ESP32F3XX) || defined(HW_ESP32WB55XX)
347 hi2c->Init.OwnAddress2Masks = I2C_OA2_NOMASK;
348 #endif
349 hi2c->Init.GeneralCallMode = I2C_GENERALCALL_DISABLE;
350 hi2c->Init.NoStretchMode = config->IsStretchActive()?I2C_NOSTRETCH_ENABLE:I2C_NOSTRETCH_DISABLE;
351
352 result =(HAL_I2C_Init(hi2c) == HAL_OK)?true:false;
353 if(result)
354 {
355 #if defined(HW_ESP32F0XX) || defined(HW_ESP32F3XX) || defined(HW_ESP32WB55XX)
356 result = (HAL_I2CEx_ConfigAnalogFilter(hi2c, I2C_ANALOGFILTER_ENABLE) == HAL_OK)?true:false;
357 #endif
358
359 if(result)
360 {
361 result = (HAL_I2CEx_ConfigDigitalFilter(hi2c, 0) == HAL_OK)?true:false;
362 if(result)
363 {
364 if(config->GetMode() == DIOSTREAMMODE_MASTER)
365 {
366 result = (HAL_I2C_IsDeviceReady(hi2c, (uint16_t)remoteaddress, 10, 1000) == HAL_OK)?true:false;
367 }
368 }
369 }
370 }
371
372 port->SetHandleI2C(hi2c);
373 DIOESP32STREAMI2C::ports[indexport] = port;
374 }
375 else
376 {
377 hi2c = DIOESP32STREAMI2C::ports[indexport]->GetHandleI2C();
378 DIOESP32STREAMI2C::ports[indexport]->SetCounterRef(DIOESP32STREAMI2C::ports[indexport]->GetCounterRef()+1);
379
380 result = true;
381 }
382
383 if(result)
384 {
385 SetEvent(DIOSTREAMI2C_FSMEVENT_CONNECTED);
386
387 status = DIOSTREAMSTATUS_CONNECTED;
388
389 memset(bufferdata, 0, DIOSTREAM_MAXBUFFER);
390
391 ResetXBuffers();
392 ResetConnectionStatistics();
393
394 result =threadconnection->Ini();
395 }
396
397 return result;
398}

447{

452

456 break;
457
458 case DIOSTREAMMODE_SEMIMASTER :
459 case DIOSTREAMMODE_SLAVE : if(HAL_I2C_Slave_Receive(hi2c, buffer, size, 1000) == HAL_OK) br = size;
460 break;
461
462 default : break;
463 }
464
465 return br;

718void DIOESP32STREAMI2C::ThreadConnection(void* param)
719{
720 DIOESP32STREAMI2C* diostream = (DIOESP32STREAMI2C*)param;
721 if(!diostream) return;
722
723 if(diostream->GetEvent()==DIOSTREAMI2C_FSMEVENT_NONE)
724 {
725 switch(diostream->GetCurrentState())
726 {
727 case DIOSTREAMI2C_FSMSTATE_NONE : break;
728
729 case DIOSTREAMI2C_FSMSTATE_CONNECTED : break;
730
731 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : if(!diostream->IsBlockRead())
732 {
733 if(!diostream->inbuffer->IsBlocked())
734 {
735 if(diostream->readcache[diostream->indexport].GetSize())
736 {
737 diostream->inbuffer->Add(diostream->readcache[diostream->indexport]);
738 diostream->readcache[diostream->indexport].Delete();
739 }
740 }
741 }
742
743 if(!diostream->IsBlockWrite())
744 {
745 if(!diostream->outbuffer->IsBlocked())
746 {
747 int esize = diostream->outbuffer->GetSize();
748 if(esize>DIOSTREAM_MAXBUFFER) esize = DIOSTREAM_MAXBUFFER;
749
750 if(esize)
751 {
752 diostream->outbuffer->Extract(diostream->writebuffer, 0, esize);
753 XDWORD _esize = diostream->WriteDirect(diostream->writebuffer, esize);
754 if(_esize != esize)
755 {
756 diostream->outbuffer->Add(diostream->writebuffer, esize);
757 }
758 }
759 }
760 }
761 break;
762
763 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : break;
764
765 }
766 }
767 else
768 {
769 if(diostream->GetEvent()<DIOSTREAMI2C_LASTEVENT)
770 {
771 diostream->CheckTransition();
772
773 switch(diostream->GetCurrentState())
774 {
775 case DIOSTREAMI2C_FSMSTATE_NONE : break;
776
777 case DIOSTREAMI2C_FSMSTATE_CONNECTED : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_CONNECTED);
778 xevent.SetDIOStream(diostream);
779 diostream->PostEvent(&xevent);
780
781 diostream->SetEvent(DIOSTREAMI2C_FSMEVENT_WAITINGTOREAD);
782 }
783 break;
784
785 case DIOSTREAMI2C_FSMSTATE_WAITINGTOREAD : break;
786
787 case DIOSTREAMI2C_FSMSTATE_DISCONNECTING : { DIOSTREAMXEVENT xevent(diostream,DIOSTREAMXEVENT_TYPE_DISCONNECTED);
788 xevent.SetDIOStream(diostream);
789 diostream->PostEvent(&xevent);
790
791 diostream->status = DIOSTREAMSTATUS_DISCONNECTED;
792 }
793 break;
794 }
795 }
796 }
797}
