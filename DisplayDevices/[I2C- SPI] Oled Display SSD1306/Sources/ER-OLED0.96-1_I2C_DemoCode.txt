void Start(void)
{	SDA=1;
	SCL=1;
    _nop_();
	SDA=0;
    _nop_();
	SCL=0;
}	


void Stop(void)
{
	SCL=0;
	SDA=0;
    _nop_();
	SCL=1;
    _nop_();
	SDA=1;
}	


void Check_Ack(void)//Acknowledge
{
    unsigned char ack=1;
	SDA=1;
    SCL=1;
    _nop_();
    while(ack==1){ack=SDA;}
    _nop_();
    SCL=0;
    return;
}


void SentByte(unsigned char Byte)
{
	uchar i;
	for(i=0;i<8;i++)
     {  SCL=0;
        if((Byte&0x80)==0x80) SDA=1;
       	else SDA=0;
       	SCL=1;
    	_nop_();		
        Byte=Byte<<1;
     }
	SCL=0;
	Check_Ack();
	
}

unsigned char ReceiveByte(void)
{uchar i,rudata=0;
	SCL=0;
	SDA=1;
	for(i=0;i<8;i++)
     {  SCL=1;
    	_nop_();	
		if (SDA==1)rudata|=0x01;
		else rudata|=0x00;
        rudata=rudata<<1;
		SCL=0;
    	_nop_();	
     }	
	Send_ACK();
	return rudata;
}

void Send_ACK(void)
{	SCL=0;
	SDA=0;
    _nop_();	
	SCL=1;
	SCL=0;
}

	{

        Set_Column_Address(0x00);
		
		Start();
		SentByte(Write_Address);
		SentByte(0x40);
		for(j=0;j<0x10;j++)
		{
		    for(k=0;k<0x04;k++)
				SentByte(value);
		    for(k=0;k<0x04;k++)
				SentByte(~value);
		}
		 	Stop();
	}
    return;
}
